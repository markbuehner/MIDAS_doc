  55  subroutine sqlr_readSqlite_avhrr(obsdat, fileName, headerIndexBegin, headerIndexEnd)
  56    !
  57    !:Purpose: To read SQLite avhrr_cloud parameters.
  58    ! 
  59    implicit none
  60
  61    ! Arguments:
  62    type (struct_obs), intent(inout) :: obsdat     ! ObsSpaceData Structure
  63    character(len=*) , intent(in)    :: fileName   ! SQLite filename
  64    integer          , intent(in)    :: headerIndexBegin
  65    integer          , intent(in)    :: headerIndexEnd
  66
  67    ! Locals:
  68    type(fSQL_DATABASE)      :: db   ! type for SQLIte  file handle
  69    type(fSQL_STATEMENT)     :: stmt ! type for precompiled SQLite statements
  70    type(fSQL_STATUS)        :: stat ! type for error status
  71    integer                  :: obsIdo
  72    character(len=128)       :: querySqlite
  73    integer                  :: rowIndex, headerIndex, columnIndex
  74    integer                  :: numberAvhrrRows ,  numberAvhrrColumns
  75    real, allocatable        :: matdata(:,:)
  76    REAL(pre_obsReal) :: CFRAC,MOYRAD,STDRAD
  77
  78    write(*,*) 'sqlr_readSqlite_avhrr: fileName : ', trim(fileName)
  79
  80    if (.not. sqlu_sqlTableExists(trim(fileName), 'avhrr')) then
  81      write(*,*) 'sqlr_readSqlite_avhrr: Table avhrr does not exist :  ... return  '
  82      return
  83    end if
  84    write(*,*) 'sqlr_readSqlite_avhrr: Table avhrr exists: insert contents into obsdat '
  85
  86    call fSQL_open(db, trim(fileName) ,stat)
  87    if (fSQL_error(stat) /= FSQL_OK) then
  88      call utl_abort('sqlr_readSqlite_avhrr: fSQL_open '//fSQL_errmsg(stat))
  89    end if
  90
  91    querySqlite = ' select mean_radiance,stddev_radiance,fractionClearPixels from avhrr where id_obs = ? '
  92    call fSQL_prepare(db, querySqlite , stmt, stat)
  93    write(*,*) 'sqlr_readSqlite_avhrr: obs_getNchanAvhr=',obs_getNchanAvhrr()
  94    do headerIndex = headerIndexBegin, headerIndexEnd
  95      obsIdo = obs_headPrimaryKey(obsdat, headerIndex)
  96      call fSQL_bind_param(stmt, param_index = 1, int_var = obsIdo)
  97      call fSQL_exec_stmt(stmt)
  98      call fSQL_get_many(stmt, nrows = numberAvhrrRows , ncols = numberAvhrrColumns , mode = FSQL_REAL)
  99      allocate(matdata(numberAvhrrRows, numberAvhrrColumns))
 100      matdata(:,:) = MPC_missingValue_R4
 101      call fSQL_fill_matrix(stmt, matdata)
 102
 103      rowIndex = 1
 104      do columnIndex = OBS_CF1, OBS_CF7
 105        if(obs_columnActive_RH(obsdat,columnIndex)) then
 106          CFRAC = matdata(rowIndex,3)
 107          call obs_headSet_r(obsdat,columnIndex,headerIndex, CFRAC)
 108          rowIndex = rowIndex + 6
 109        end if
 110      end do
 111
 112      rowIndex = 1
 113      do columnIndex = OBS_M1C1, OBS_M7C6
 114        if(obs_columnActive_RH(obsdat,columnIndex)) then
 115          MOYRAD = matdata(rowIndex,1) * 100000.d0
 116          call obs_headSet_r(obsdat,columnIndex,headerIndex, MOYRAD)
 117          rowIndex = rowIndex + 1
 118        endif
 119      end do
 120
 121      rowIndex = 1
 122      do columnIndex = OBS_S1C1, OBS_S7C6
 123        if(obs_columnActive_RH(obsdat,columnIndex)) then
 124          STDRAD = matdata(rowIndex,2) * 100000.d0
 125          call obs_headSet_r(obsdat,columnIndex,headerIndex,  STDRAD)
 126          rowIndex = rowIndex + 1
 127        end if
 128      end do
 129
 130      deallocate(matdata)
 131      call fSQL_free_mem(stmt)
 132
 133    end do
 134
 135    call fSQL_finalize(stmt)
 136    call fSQL_close(db, stat) 
 137
 138  end subroutine sqlr_readSqlite_avhrr

 143  subroutine sqlr_readSqlite(obsdat, familyType, fileName)
 144    !
 145    !:Purpose: To read SQLite namelist and files.
 146    ! 
 147    implicit none
 148
 149    ! Arguments:
 150    type (struct_obs), intent(inout) :: obsdat     ! ObsSpaceData Structure
 151    character(len=*) , intent(in)    :: familyType ! Family Type 
 152    character(len=*) , intent(in)    :: fileName   ! SQLite filename
 153
 154    ! Locals:
 155    character(len=12)        :: idStation
 156    integer                  :: codeType, obsDate, obsTime, obsStatus, obsFlag, obsVarno
 157    integer(8)               :: headPrimaryKey
 158    integer(8), allocatable  :: bodyHeadKeys(:), bodyPrimaryKeys(:)
 159    integer(8), allocatable  :: tempBodyKeys(:,:)
 160    logical                  :: finishedWithHeader
 161    real(pre_obsReal)        :: elevReal, xlat, xlon, vertCoord
 162    real                     :: elev    , obsLat, obsLon, elevFact
 163    real                     :: beamAzimuth, beamRangeStart, beamRangeEnd, beamElevation          
 164    real(pre_obsReal)        :: beamAzimuthReal, beamElevationReal
 165    real(pre_obsReal)        :: beamLat, beamLon, beamHeight, beamDistance, beamRange
 166    integer                  :: vertCoordType, vertCoordFact, fnom, fclos, nulnam, ierr, idProf
 167    real                     :: zenithReal, solarZenithReal, CloudCoverReal, solarAzimuthReal
 168    integer                  :: roQcFlag
 169    real(pre_obsReal)        :: geoidUndulation, earthLocRadCurv, obsValue
 170    real(8)                  :: geoidUndulation_R8, earthLocRadCurv_R8, azimuthReal_R8
 171    integer                  :: trackCellNum, iceChartID
 172    real(pre_obsReal)        :: modelWindSpeed
 173    real(8)                  :: modelWindSpeed_R8
 174    integer                  :: iasiImagerCollocationFlag, iasiGeneralQualityFlag
 175    integer                  :: obsSat, landSea, terrainType, instrument, sensor
 176    integer                  :: rowIndex, obsNlv, headerIndex, bodyIndex
 177    integer                  :: numBody, numHeader
 178    real(pre_obsReal), parameter :: zemFact = 0.01
 179    character(len=1024)      :: queryHeader, queryIDs
 180    character(len=256)       :: csqlcrit, selectIDs
 181    character(len=1024)      :: columnsHeader
 182    logical                  :: finished
 183    type(fSQL_DATABASE)      :: db         ! type for SQLIte  file handle
 184    type(fSQL_STATEMENT)     :: stmt,stmt2 ! type for precompiled SQLite statements
 185    type(fSQL_STATUS)        :: stat       !type for error status
 186    character(len=256)       :: sqlDataOrder, extraQueryData
 187    character(len=256), allocatable :: listElemArray(:)
 188    integer, allocatable            :: listElemArrayInteger(:)
 189    integer                  :: numberBodyRows, numberBodyColumns, numberIDsRows, numberIDsColumns
 190    integer                  :: columnIndex
 191
 192    integer, parameter :: lenSqlName    = 60
 193    character(len=lenSqlName), allocatable :: headSqlNames(:), bodySqlNames(:)
 194    character(len=lenSqlName), parameter :: headSqlNamesToRead(32) = (/'ID_STN','LAT','LON','CODTYP', &
 195         'DATE','TIME','STATUS','ELEV','ANTENNA_ALTITUDE','LAND_SEA','ID_SAT','INSTRUMENT','SENSOR', &
 196         'ZENITH','SOLAR_ZENITH','AZIMUTH','SOLAR_AZIMUTH','TERRAIN_TYPE','CLOUD_COVER', &
 197         'FANION_QUAL_IASI_SYS_IND','INDIC_NDX_QUAL_GEOM','RO_QC_FLAG','GEOID_UNDULATION', &
 198         'EARTH_LOCAL_RAD_CURV','CENTER_AZIMUTH','CENTER_ELEVATION','RANGE_START','RANGE_END', &
 199         'ID_PROF','CHARTINDEX','TRACK_CELL_NO','MOD_WIND_SPD'/)
 200    real(8),                   allocatable :: bodyValues(:,:), codtypInFileList(:,:)
 201    logical :: beamRangeFound
 202
 203    ! Namelist variables:
 204    integer                  :: numberElem        ! MUST NOT BE INCLUDED IN NAMELIST!
 205    character(len=256)       :: listElem          ! list of bufr element ids to read
 206    character(len=256)       :: sqlExtraDat       ! can be used e.g. for ' and id_obs in (select id_obs from header where...) '
 207    character(len=256)       :: sqlExtraDat_sat   ! same as sqlExtraDat, but only for SST satellite obs
 208    character(len=256)       :: sqlExtraHeader    ! can be used e.g. for ' id_stn in (...) '
 209    character(len=256)       :: sqlExtraHeader_sat! same as sqlExtraHeader, but only for SST satellite obs
 210    integer                  :: codtyp_sat(10)    ! list of codtyp values of SST satellite obs (default is 88)
 211    character(len=256)       :: sqlNull           ! can be used e.g. for ' and obsvalue is not null '
 212
 213    namelist /NAMSQLtovs/ numberElem,listElem,sqlExtraDat,sqlExtraHeader,sqlNull
 214    namelist /NAMSQLua/   numberElem,listElem,sqlExtraDat,sqlExtraHeader,sqlNull
 215    namelist /NAMSQLai/   numberElem,listElem,sqlExtraDat,sqlExtraHeader,sqlNull
 216    namelist /NAMSQLsw/   numberElem,listElem,sqlExtraDat,sqlExtraHeader,sqlNull
 217    namelist /NAMSQLro/   numberElem,listElem,sqlExtraDat,sqlExtraHeader,sqlNull
 218    namelist /NAMSQLsfc/  numberElem,listElem,sqlExtraDat,sqlExtraHeader,sqlNull
 219    namelist /NAMSQLsc/   numberElem,listElem,sqlExtraDat,sqlExtraHeader,sqlNull
 220    namelist /NAMSQLpr/   numberElem,listElem,sqlExtraDat,sqlExtraHeader,sqlNull
 221    namelist /NAMSQLal/   numberElem,listElem,sqlExtraDat,sqlExtraHeader,sqlNull
 222    namelist /NAMSQLgl/   numberElem,listElem,sqlExtraDat,sqlExtraHeader,sqlNull
 223    namelist /NAMSQLradar/numberElem,listElem,sqlExtraDat,sqlExtraHeader,sqlNull
 224    namelist /NAMSQLsst/  numberElem,listElem,sqlExtraDat,sqlExtraHeader,sqlNull, &
 225                          sqlExtraDat_sat,sqlExtraHeader_sat,codtyp_sat
 226
 227    write(*,*) 'sqlr_readSqlite: fileName   : ', trim(fileName)
 228    write(*,*) 'sqlr_readSqlite: familyType : ', trim(familyType)
 229
 230    ! Get the codtyp from the file to use in determining how to build the main queries
 231    call sqlu_getColumnValuesNum(codtypInFileList, fileName=trim(fileName), &
 232                                 tableName='header', sqlColumnNames=(/'codtyp'/), &
 233                                 extraQuery_opt='group by codtyp')
 234    write(*,*) 'sqlr_readSqlite: codtyp array = ', codtypInFileList(:,:)
 235
 236    ! Default namelist variable values
 237    sqlExtraHeader     = ''
 238    sqlExtraDat        = ''
 239    sqlExtraHeader_sat = ''
 240    sqlExtraDat_sat    = ''
 241    codtyp_sat(:)      = MPC_missingValue_INT
 242    codtyp_sat(1)      = 88
 243    sqlNull            = ''
 244    listElem           = ''
 245    numberElem         = MPC_missingValue_INT
 246    
 247    ! Set the type of vertical coordinate
 248    vertCoordType  = 1
 249    select case(trim(familyType))
 250      case('RA','PR','AL','RO','SF','TM','SC','GL','HY','GP')
 251        vertCoordType = 1
 252      case('UA','AI','SW')
 253        vertCoordType = 2
 254      case('TO')
 255        vertCoordType = 3
 256      case default
 257        call utl_abort('sqlr_readSqlite: unknown family '//trim(familyType))
 258    end select
 259
 260    ! Set multiplying factor for vertical coordinate
 261    select case(trim(familyType))
 262      case('SF','TM','SC')
 263        vertCoordFact = 0
 264      case default
 265        vertCoordFact = 1
 266    end select
 267
 268    ! Set multiplying factor used when adding elevation to vcoord
 269    select case(trim(familyType))
 270      case('PR','SF','TM','GP')
 271        elevFact=1.0
 272      case default
 273        elevFact=0.0
 274    end select
 275
 276    ! Read appropriate namelist based on obs family
 277    nulnam = 0
 278    ierr=fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
 279    select case(trim(familyType))
 280      case('TO')
 281        read(nulnam, nml = NAMSQLtovs, iostat = ierr)
 282        if (ierr /= 0) call utl_abort('sqlr_readSqlite: Error reading namelist: NAMSQLtovs')
 283        if (mmpi_myid == 0) write(*, nml = NAMSQLtovs)
 284      case('UA')
 285        read(nulnam, nml = NAMSQLua, iostat = ierr)
 286        if (ierr /= 0) call utl_abort('sqlr_readSqlite: Error reading namelist: NAMSQLua')
 287        if (mmpi_myid == 0) write(*, nml = NAMSQLua)
 288      case ('AI')
 289        read(nulnam, nml = NAMSQLai, iostat = ierr)
 290        if (ierr /= 0) call utl_abort('sqlr_readSqlite: Error reading namelist: NAMSQLai')
 291        if (mmpi_myid == 0) write(*, nml = NAMSQLai)
 292      case ('SW')
 293        read(nulnam, nml = NAMSQLsw, iostat = ierr)
 294        if (ierr /= 0) call utl_abort('sqlr_readSqlite: Error reading namelist: NAMSQLsw')
 295        if (mmpi_myid == 0) write(*, nml = NAMSQLsw)
 296      case ('PR')
 297        read(nulnam, nml = NAMSQLpr, iostat = ierr)
 298        if (ierr /= 0) call utl_abort('sqlr_readSqlite: Error reading namelist: NAMSQLpr')
 299        if (mmpi_myid == 0) write(*, nml =  NAMSQLpr)
 300      case ('AL')  
 301        read(nulnam, nml = NAMSQLal, iostat = ierr)
 302        if (ierr /= 0) call utl_abort('sqlr_readSqlite: Error reading namelist: NAMSQLal')
 303        if (mmpi_myid == 0) write(*, nml =  NAMSQLal)
 304      case ('RO')     
 305        read(nulnam, nml = NAMSQLro, iostat = ierr)
 306        if (ierr /= 0) call utl_abort('sqlr_readSqlite: Error reading namelist: NAMSQLro')
 307        if (mmpi_myid == 0) write(*, nml = NAMSQLro)
 308      case ('SF','GP','HY')
 309        read(nulnam, nml = NAMSQLsfc, iostat = ierr)
 310        if (ierr /= 0) call utl_abort('sqlr_readSqlite: Error reading namelist: NAMSQLsfc')
 311        if (mmpi_myid == 0) write(*, nml = NAMSQLsfc)        
 312      case ('SC')
 313        read(nulnam, nml = NAMSQLsc, iostat = ierr)
 314        if (ierr /= 0) call utl_abort('sqlr_readSqlite: Error reading namelist: NAMSQLsc')
 315        if (mmpi_myid == 0) write(*, nml = NAMSQLsc)
 316      case ('TM')
 317        read(nulnam, nml = NAMSQLsst, iostat = ierr)
 318        if (ierr /= 0) call utl_abort('sqlr_readSqlite: Error reading namelist: NAMSQLsst')
 319        if (mmpi_myid == 0) write(*, nml = NAMSQLsst)
 320        do rowIndex = 1, size(codtypInFileList,1)
 321          if (any(nint(codtypInFileList(rowIndex,1)) == codtyp_sat(:))) then
 322            write(*,*) 'sqlr_readSqlite: Found satellite SST observation in file:', &
 323                       nint(codtypInFileList(rowIndex,1))
 324            sqlExtraHeader = sqlExtraHeader_sat
 325            sqlExtraDat = sqlExtraDat_sat
 326          end if
 327        end do
 328      case ('GL')
 329        read(nulnam, nml = NAMSQLgl, iostat = ierr)
 330        if (ierr /= 0) call utl_abort('sqlr_readSqlite: Error reading namelist: NAMSQLgl')
 331        if (mmpi_myid == 0) write(*, nml =  NAMSQLgl)
 332      case('RA')
 333        read(nulnam, nml = NAMSQLradar, iostat = ierr)
 334        if (ierr /= 0) call utl_abort('sqlr_readSqlite: Error reading namelist: NAMSQLradar')
 335        if (mmpi_myid == 0) write(*, nml =  NAMSQLradar) 
 336      case default
 337        call utl_abort('sqlr_readSqlite: No namelist read for this family: '//trim(familyType))
 338    end select
 339    ierr=fclos(nulnam)
 340    
 341    if (numberElem /= MPC_missingValue_R4) then
 342      call utl_abort('sqlr_readSqlite: check namelist, numberElem should be removed')
 343    end if
 344    numberElem = count(transfer(listElem, 'a', len(listElem)) == ',') + 1
 345
 346    ! Set the observation variable transforms
 347    if (numberElem > 0) then
 348      call utl_splitString(listElem,',', listElemArray)
 349      call utl_stringArrayToIntegerArray(listElemArray, listElemArrayInteger)
 350      call ovt_setup(listElemArrayInteger)
 351      deallocate(listElemArrayInteger)
 352      deallocate(listElemArray)
 353    end if
 354
 355    ! Compose SQL queries
 356
 357    ! Get list of all header table columns in file, then remove those we don't read
 358    call sqlu_getSqlColumnNames(headSqlNames, fileName=trim(fileName), &
 359                                tableName='header', dataType='all')
 360    do columnIndex = 1, size(headSqlNames)
 361      if (utl_findloc(headSqlNamesToRead,headSqlNames(columnIndex)) == 0) then
 362        headSqlNames(columnIndex) = ''
 363      end if
 364    end do
 365    call utl_removeEmptyStrings(headSqlNames)
 366    do columnIndex = 1, size(headSqlNames)
 367      write(*,*) 'sqlr_readSqlite: headSqlNames   =', columnIndex, &
 368                 trim(headSqlNames(columnIndex))
 369    end do
 370    call utl_combineString(columnsHeader, ',', headSqlNames)
 371    
 372    ! ordering of data that will get read in matdata, bodyPrimaryKeys and bodyHeadKeys
 373    sqlDataOrder = ' order by id_obs, varno, id_data' 
 374
 375    selectIDs  = 'select id_data, id_obs'
 376    csqlcrit = 'varno in ('//trim(listElem)//')'//trim(sqlExtraDat)//trim(SQLNull)
 377
 378    ! Determine names of body columns present in sqlite file
 379    call sqlu_getSqlColumnNames(bodySqlNames, fileName=trim(fileName), &
 380                                tableName='data', dataType='numeric')
 381    do columnIndex = 1, size(bodySqlNames)
 382      write(*,*) 'sqlr_readSqlite: bodySqlNames   =', columnIndex, &
 383                 trim(bodySqlNames(columnIndex))
 384    end do
 385
 386    ! Read all of the columns in the file, then we decide what to do with them later
 387    extraQueryData = 'where '//trim(csqlcrit)//trim(sqlDataOrder)
 388    call sqlu_getColumnValuesNum(bodyValues, fileName=trim(fileName), &
 389                                 tableName='data', sqlColumnNames=bodySqlNames, &
 390                                 extraQuery_opt=extraQueryData)
 391    numberBodyRows    = size(bodyValues,1)
 392    numberBodyColumns = size(bodyValues,2)
 393
 394    ! It is very important that queryIDs and queryData be identical except for the column names being selected
 395    queryIDs  = trim(selectIDs) //' from data where '//trim(csqlcrit)//trim(sqlDataOrder)//';'
 396    write(*,'(4a)') 'sqlr_readSqlite: queryIDs     --> ', trim(queryIDs)
 397
 398    ! Open the sqlite file
 399    call fSQL_open(db, trim(fileName) ,stat)
 400    if (fSQL_error(stat) /= FSQL_OK) then
 401      call utl_abort('sqlr_readSqlite: fSQL_open '//fSQL_errmsg(stat))
 402    end if
 403
 404    ! Read id_data and id_obs columns in the body table ("status" not set when getting integers)
 405    call fSQL_prepare(db, trim(queryIDs) , stmt2, status=stat)
 406    call fSQL_get_many(stmt2, nrows=numberIDsRows, ncols=numberIDsColumns, &
 407                       mode=FSQL_INT8)
 408    allocate(tempBodyKeys(numberIDsRows,numberIDsColumns))
 409    call fSQL_fill_matrix(stmt2, tempBodyKeys)
 410    allocate(bodyPrimaryKeys(numberIDsRows))
 411    allocate(bodyHeadKeys(numberIDsRows))
 412    bodyPrimaryKeys(:) = tempBodyKeys(:,1)
 413    bodyHeadKeys(:)    = tempBodyKeys(:,2)
 414    deallocate(tempBodyKeys)
 415    call fSQL_free_mem(stmt2)
 416    call fSQL_finalize(stmt2)
 417
 418    if (numberIDsRows /= numberBodyRows) then
 419      call utl_abort('sqlr_readSqlite: number of body keys not equal to number of rows in bodyValues')
 420    end if 
 421
 422    headerIndex  = obs_numHeader(obsdat)
 423    bodyIndex = obs_numBody(obsdat)
 424    numHeader = obs_numHeader(obsdat)
 425    numBody   = obs_numBody(obsdat)
 426    write(*,*) 'sqlr_readSqlite: DEBUT numheader  =', numHeader
 427    write(*,*) 'sqlr_readSqlite: DEBUT numbody    =', numBody
 428    write(*,*) 'sqlr_readSqlite:  numberBodyRows numberBodyColumns =', numberBodyRows, numberBodyColumns
 429    write(*,*) 'sqlr_readSqlite: =========================================='
 430
 431    ! Here is a summary of what is going on in the rest of this routine:
 432    ! 
 433    ! for each row of the body table (which has already been read)
 434    !
 435    !   if this is the first id_data associated with a given id_obs
 436    !     -> read associated header and write to obsSpaceData
 437    !
 438    !   -> write body entry in obsSpaceData
 439    !
 440    obsNlv = 0
 441    call fSQL_begin(db)
 442    BODY: do rowIndex = 1, numberIDsRows
 443
 444      ! id_obs, bodyIndex and obsNlv values for this entry
 445      headPrimaryKey = bodyHeadKeys(rowIndex)
 446      bodyIndex = bodyIndex + 1
 447      obsNlv    = obsNlv + 1
 448      call obs_setBodyPrimaryKey(obsdat, bodyIndex, bodyPrimaryKeys(rowIndex))
 449
 450      READHEADER: if (obsNlv == 1) then
 451        headerIndex = headerIndex + 1
 452
 453        ! This is the first body entry associated with a new headPrimaryKey
 454        call obs_headSet_i(obsdat, OBS_RLN, headerIndex, bodyIndex)
 455
 456        ! we read the associated header
 457        if (trim(sqlExtraHeader) == '') then
 458          queryHeader = 'select '//trim(columnsHeader)//' from header where id_obs = ? '
 459        else
 460          queryHeader = 'select '//trim(columnsHeader)//' from header where '//trim(sqlExtraHeader)//' and id_obs = ? '
 461        end if
 462
 463        if (rowIndex == 1) then
 464          write(*,'(4a)') 'sqlr_readSqlite: first queryHeader    --> ', trim(queryHeader)
 465        end if
 466        call fSQL_prepare(db, queryHeader, stmt, stat)
 467        if (fSQL_error(stat)  /= FSQL_OK) then
 468          write(*,*) 'sqlr_readSqlite: problem reading header entry. Query: ', trim(queryHeader)
 469          call sqlu_handleError(stat, 'fSQL_prepare')
 470        end if
 471
 472        call fSQL_bind_param(stmt, param_index = 1, int8_var = headPrimaryKey)
 473
 474        call fSQL_get_row(stmt, finished)
 475        if (finished) then
 476          write(*,*) 'sqlr_readSqlite: problem reading header entry. Query:', trim(queryHeader)
 477          call utl_abort('Problem with fSQL_get_row()')
 478        end if
 479
 480        ! The query result is inserted into variables
 481        terrainType = MPC_missingValue_INT
 482        landSea     = MPC_missingValue_INT
 483        sensor      = MPC_missingValue_INT
 484        cloudCoverReal = MPC_missingValue_R4
 485        elev = 0.
 486        elevReal = 0.
 487        solarAzimuthReal = MPC_missingValue_R4
 488        solarZenithReal  = MPC_missingValue_R4
 489        zenithReal = MPC_missingValue_R4
 490        instrument = MPC_missingValue_INT
 491        azimuthReal_R8 = MPC_missingValue_R8
 492        beamAzimuth    = MPC_missingValue_R4
 493        beamElevation  = MPC_missingValue_R4
 494        beamRangeStart = MPC_missingValue_R4
 495        beamRangeEnd   = MPC_missingValue_R4
 496        obsSat = MPC_missingValue_INT
 497
 498        ! Set some default values
 499        if (trim(familyType) == 'TO') then
 500          call obs_headSet_i(obsdat, OBS_TTYP, headerIndex, terrainType)
 501        end if
 502        call obs_headSet_r(obsdat, OBS_ALT, headerIndex, elevReal)
 503
 504        call obs_setFamily(obsdat, trim(familyType), headerIndex)
 505        call obs_setHeadPrimaryKey(obsdat, headerIndex, headPrimaryKey)
 506        call obs_headSet_i(obsdat, OBS_ONM, headerIndex, headerIndex)
 507        do columnIndex = 1, size(headSqlNames)
 508          select case(trim(headSqlNames(columnIndex)))
 509            case('LAT')
 510              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real_var  = obsLat)
 511              xlat = obsLat * MPC_RADIANS_PER_DEGREE_R8
 512              call obs_headSet_r(obsdat, OBS_LAT, headerIndex, xLat)
 513            case('LON')
 514              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real_var  = obsLon)
 515              if (obsLon < 0.) obsLon = obsLon + 360.
 516              xlon = obsLon * MPC_RADIANS_PER_DEGREE_R8
 517              call obs_headSet_r(obsdat, OBS_LON, headerIndex, xLon)
 518            case('CODTYP')
 519              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = codeType)
 520              call obs_headSet_i(obsdat, OBS_ITY, headerIndex, codeType)
 521            case('DATE')
 522              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = obsDate)
 523              call obs_headSet_i(obsdat, OBS_DAT, headerIndex, obsDate)
 524            case('TIME')
 525              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = obsTime)
 526              obsTime = obsTime/100
 527              call obs_headSet_i(obsdat, OBS_ETM, headerIndex, obsTime)
 528            case('ID_STN')
 529              call fSQL_get_column(stmt, COL_INDEX = columnIndex, char_var  = idStation)
 530              call obs_set_c(obsdat, 'STID' , headerIndex, trim(idStation))
 531            case('STATUS')
 532              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = obsStatus)
 533              call obs_headSet_i(obsdat, OBS_ST1, headerIndex, obsStatus)
 534            case('ELEV','ANTENNA_ALTITUDE')
 535              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real_var  = elev, REAL_MISSING=MPC_missingValue_R4)
 536              elevReal=elev
 537              if (trim(familyType) /= 'TO') then
 538                call obs_headSet_r(obsdat, OBS_ALT ,headerIndex, elevReal)
 539              end if
 540            case('LAND_SEA')
 541              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = landSea, INT_MISSING=MPC_missingValue_INT)
 542              call obs_headSet_i(obsdat, OBS_STYP, headerIndex, landSea)
 543            case('ID_SAT')
 544              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = obsSat, INT_MISSING=MPC_missingValue_INT)
 545              call obs_headSet_i(obsdat, OBS_SAT , headerIndex, obsSat)
 546            case('INSTRUMENT')
 547              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = instrument, INT_MISSING=MPC_missingValue_INT)
 548            case('ZENITH')
 549              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real_var  = zenithReal, REAL_MISSING=MPC_missingValue_R4)
 550              call obs_headSet_r(obsdat, OBS_SZA , headerIndex, real(zenithReal,kind=pre_obsReal))
 551            case('SOLAR_ZENITH')
 552              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real_var  = solarZenithReal, REAL_MISSING=MPC_missingValue_R4)
 553              call obs_headSet_r(obsdat, OBS_SUN , headerIndex, real(solarZenithReal,kind=pre_obsReal))
 554            case('AZIMUTH')
 555              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real8_var = azimuthReal_R8)
 556              call obs_headSet_r(obsdat, OBS_AZA , headerIndex, real(azimuthReal_R8,kind=pre_obsReal))
 557            case('TERRAIN_TYPE')
 558              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = terrainType, INT_MISSING=MPC_missingValue_INT)
 559              call obs_headSet_i(obsdat, OBS_TTYP, headerIndex, terrainType)
 560            case('CLOUD_COVER')
 561              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real_var  = cloudCoverReal, REAL_MISSING=MPC_missingValue_R4)
 562              call obs_headSet_r(obsdat, OBS_CLF , headerIndex, real(cloudCoverReal,kind=pre_obsReal))
 563            case('SOLAR_AZIMUTH')
 564              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real_var  = solarAzimuthReal, REAL_MISSING=MPC_missingValue_R4)
 565              call obs_headSet_r(obsdat, OBS_SAZ , headerIndex, real(solarAzimuthReal,kind=pre_obsReal))
 566            case('SENSOR')
 567              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = sensor, INT_MISSING=MPC_missingValue_INT)
 568            case('FANION_QUAL_IASI_SYS_IND')
 569              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = iasiGeneralQualityFlag, INT_MISSING=MPC_missingValue_INT)
 570              if (obs_columnActive_IH(obsdat,OBS_GQF)) then
 571                call obs_headSet_i(obsdat,OBS_GQF, headerIndex, iasiGeneralQualityFlag)
 572              end if
 573            case('INDIC_NDX_QUAL_GEOM')
 574              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = iasiImagerCollocationFlag, INT_MISSING=MPC_missingValue_INT)
 575              if (obs_columnActive_IH(obsdat,OBS_GQL)) then
 576                call obs_headSet_i(obsdat, OBS_GQL, headerIndex, iasiImagerCollocationFlag)
 577              end if
 578            case('RO_QC_FLAG')
 579              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = roQcFlag, INT_MISSING=MPC_missingValue_INT)
 580              call obs_headSet_i(obsdat, OBS_ROQF, headerIndex, roQcFlag)
 581            case('GEOID_UNDULATION')
 582              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real8_var = geoidUndulation_R8)
 583              geoidUndulation = geoidUndulation_R8
 584              call obs_headSet_r(obsdat, OBS_GEOI, headerIndex, geoidUndulation)
 585            case('EARTH_LOCAL_RAD_CURV')
 586              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real8_var = earthLocRadCurv_R8)
 587              earthLocRadCurv = earthLocRadCurv_R8
 588              call obs_headSet_r(obsdat, OBS_TRAD, headerIndex, earthLocRadCurv)
 589            case('CENTER_AZIMUTH')
 590              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real_var  = beamAzimuth)
 591              beamAzimuthReal   = beamAzimuth   * MPC_RADIANS_PER_DEGREE_R8
 592              call obs_headSet_r(obsdat, OBS_RZAM, headerIndex, beamAzimuthReal)
 593            case('CENTER_ELEVATION')
 594              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real_var  = beamElevation)
 595              beamElevationReal = beamElevation * MPC_RADIANS_PER_DEGREE_R8
 596              call obs_headSet_r(obsdat, OBS_RELE, headerIndex, beamElevationReal)
 597            case('RANGE_START')
 598              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real_var  = beamRangeStart)
 599              call obs_headSet_r(obsdat, OBS_RANS, headerIndex, real(beamRangeStart,kind=pre_obsReal))
 600            case('RANGE_END')
 601              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real_var  = beamRangeEnd)
 602              call obs_headSet_r(obsdat, OBS_RANE, headerIndex, real(beamRangeEnd,kind=pre_obsReal))
 603            case('ID_PROF')
 604              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = idProf)
 605              call obs_headSet_i(obsdat, OBS_PRFL, headerIndex, idProf)
 606            case('CHARTINDEX')
 607              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = iceChartID)
 608              call obs_headSet_i(obsdat, OBS_CHID, headerIndex, iceChartID)
 609            case('TRACK_CELL_NO')
 610              call fSQL_get_column(stmt, COL_INDEX = columnIndex, int_var   = trackCellNum)
 611              if (trackCellNum > 21) trackCellNum = 43 - trackCellNum
 612              call obs_headSet_i(obsdat, OBS_FOV, headerIndex, trackCellNum)
 613            case('MOD_WIND_SPD')
 614              call fSQL_get_column(stmt, COL_INDEX = columnIndex, real8_var = modelWindSpeed_R8)
 615              modelWindSpeed = modelWindSpeed_R8
 616              call obs_headSet_r(obsdat, OBS_MWS, headerIndex, modelWindSpeed)
 617          end select
 618        end do
 619
 620        ! we are done reading this header entry
 621        call fSQL_finalize(stmt)
 622
 623        ! adjust some values that were just read
 624        if (trim(familyType) == 'TO') then
 625          if (instrument == 420) obsSat  = 784
 626          if (codeType == codtyp_get_codtyp('crisfsr') .and. instrument == 620) instrument = 2046
 627          if (sensor == MPC_missingValue_INT) then
 628            sensor = 0
 629            if (instrument == MPC_missingValue_INT) instrument = 0
 630          else
 631            instrument = obsu_cvt_obs_instrum(sensor)
 632          end if
 633          call obs_headSet_i(obsdat, OBS_SAT , headerIndex, obsSat)
 634          call obs_headSet_i(obsdat, OBS_INS , headerIndex, instrument)
 635        end if
 636
 637      end if READHEADER
 638
 639      ! Initialize some obsSpaceData body values
 640      if (obs_columnActive_RB(obsdat,OBS_LATD)) then
 641        call obs_bodySet_r(obsdat, OBS_LATD, bodyIndex, obs_missingValue_R)
 642      end if
 643      if (obs_columnActive_RB(obsdat,OBS_LOND)) then
 644        call obs_bodySet_r(obsdat, OBS_LOND, bodyIndex, obs_missingValue_R)
 645      end if
 646      if (obs_columnActive_RB(obsdat,OBS_SEM)) then
 647        call obs_bodySet_r(obsdat, OBS_SEM, bodyIndex, 0.0)
 648      end if
 649
 650      ! Copy body row into obsspacedata by looping over all columns that were read
 651      beamRangeFound = .false. ! Needed to overwrite obs_ppp, regardless of column order
 652      do columnIndex = 1, size(bodySqlNames)
 653        select case(trim(bodySqlNames(columnIndex)))
 654          case('VCOORD')
 655            vertCoord = bodyValues(rowIndex,columnIndex)
 656            if (trim(familyType) /= 'RA' .and. trim(familyType) /= 'TO') then
 657              vertCoord = vertCoord * vertCoordFact + elevReal * elevFact
 658            end if
 659            call obs_bodySet_r(obsdat, OBS_PPP, bodyIndex, vertCoord)
 660          case('VCOORD_TYPE')
 661            ! This is set earlier based on familyType
 662            call obs_bodySet_i(obsdat, OBS_VCO, bodyIndex, vertCoordType)
 663          case('VARNO')
 664            obsVarno = int(bodyValues(rowIndex,columnIndex))
 665            call obs_bodySet_i(obsdat, OBS_VNM, bodyIndex, obsVarno)
 666          case('OBSVALUE')
 667            obsValue = bodyValues(rowIndex,columnIndex)
 668            if (trim(familyType) == 'TO' .and. obsValue == MPC_missingValue_R8) then
 669              ! Is this really needed???
 670              obsValue = real(MPC_missingValue_R8,pre_obsReal)
 671            end if
 672            call obs_bodySet_r(obsdat, OBS_VAR, bodyIndex, obsValue)
 673          case('FLAG')
 674            obsFlag = int(bodyValues(rowIndex,columnIndex))
 675            if (obsFlag == mpc_missingValue_INT) obsFlag = 0
 676            call obs_bodySet_i(obsdat, OBS_FLG, bodyIndex, obsFlag)
 677          case('SURF_EMISS')
 678            if (obs_columnActive_RB(obsdat,OBS_SEM)) then
 679              call obs_bodySet_r(obsdat, OBS_SEM, bodyIndex, bodyValues(rowIndex,columnIndex) * zemFact)
 680            end if
 681          case('BIAS_CORR')
 682            if (obs_columnActive_RB(obsdat,OBS_BCOR)) then
 683              call obs_bodySet_r(obsdat, OBS_BCOR, bodyIndex, bodyValues(rowIndex,columnIndex))
 684            end if
 685          case('RANGE')
 686            beamRangeFound = .true.
 687            beamRange = bodyValues(rowIndex,columnIndex)
 688            call obs_bodySet_r(obsdat, OBS_LOCI, bodyIndex, beamRange)
 689            ! elevation and azimuths are converted to radians and pre_obsReal precision
 690            beamElevationReal = beamElevation * MPC_RADIANS_PER_DEGREE_R8
 691            beamAzimuthReal   = beamAzimuth   * MPC_RADIANS_PER_DEGREE_R8
 692
 693            call rdv_getlatlonHRfromRange(xlat, xlon, beamElevationReal, beamAzimuthReal, & !in
 694                                          elevReal, beamRange,                            & !in
 695                                          beamLat, beamLon, beamHeight, beamDistance)       !out
 696            call obs_bodySet_r(obsdat, OBS_LATD, bodyIndex, beamLat)
 697            call obs_bodySet_r(obsdat, OBS_LOND, bodyIndex, beamLon)
 698        end select
 699      end do
 700
 701      ! Replace the value of obs_ppp if a radar beam range value was found
 702      if (beamRangeFound) call obs_bodySet_r(obsdat, OBS_PPP, bodyIndex, beamHeight)
 703
 704      ! Activate the 'rejected by selection process' bit if observed value is missing
 705      if (obsValue == MPC_missingValue_R8) then
 706        obsFlag  = obs_bodyElem_i(obsdat, obs_flg, bodyIndex)
 707        call obs_bodySet_i(obsdat, OBS_FLG, bodyIndex, ibset(obsFlag,11))
 708      end if
 709
 710      ! In some cases we need to add extra row(s) to the BODY table
 711      if (.not. filt_bufrCodeAssimilated(obsVarno) .and. &
 712          .not. ovt_bufrCodeSkipped(obsVarno)) then
 713          
 714        ! Add an extra row to the obsSpaceData body table
 715        ! to contain quantity later calculated by ovt_transformObsValue
 716        call obs_setBodyPrimaryKey(obsdat, bodyIndex+1, -1)
 717        call sqlr_addExtraDataRow(obsdat, vertCoord * vertCoordFact + elevReal * elevFact, &
 718                                  ovt_getDestinationBufrCode(obsVarno), &
 719                                  vertCoordType, bodyIndex+1)
 720        bodyIndex = bodyIndex + 1
 721        obsNlv = obsNlv + 1
 722        if (ovt_isWindObs(obsVarno)) then
 723          ! Add an extra row for the other wind component
 724          call obs_setBodyPrimaryKey(obsdat, bodyIndex+1, -1)
 725          call sqlr_addExtraDataRow(obsdat, vertCoord * vertCoordFact + elevReal * elevFact, &
 726                                    ovt_getDestinationBufrCode(obsVarno,extra_opt=.true.), &
 727                                    vertCoordType, bodyIndex+1)
 728          bodyIndex = bodyIndex + 1
 729          obsNlv = obsNlv + 1
 730        end if
 731      end if
 732
 733      ! Logic to determine whether we are finished with this header entry
 734      finishedWithHeader = .false.
 735      if (rowIndex == numberIDsRows) then
 736        ! This is the last body entry, nothing comes after.
 737        finishedWithHeader = .true.
 738      else
 739        if (headPrimaryKey /= bodyHeadKeys(rowIndex+1)) then
 740          ! we just treated the last body entry associated with this id_obs (headPrimaryKey)
 741          ! make header for this id_obs
 742          finishedWithHeader = .true.
 743        end if
 744      end if
 745      if (finishedWithHeader) then
 746        ! Record the number of body rows associated with this header entry
 747        call obs_headSet_i(obsdat, OBS_NLV, headerIndex, obsNlv)
 748        obsNlv = 0 ! and reset counter for next batch of body entries
 749      end if
 750
 751    end do BODY
 752    call fSQL_commit(db)
 753
 754    numHeader = obs_numHeader(obsdat)
 755    numBody   = obs_numBody(obsdat)
 756    write(*,*) 'sqlr_readSqlite: FIN numheader  =', numHeader
 757    write(*,*) 'sqlr_readSqlite: FIN numbody    =', numBody
 758
 759    call fSQL_close(db, stat) 
 760    if (fSQL_error(stat) /= FSQL_OK) then
 761      write(*,*) 'sqlr_readSqlite: problem closing sqlite db', trim(fileName)
 762      call sqlu_handleError(stat, 'fSQL_close')
 763    end if
 764
 765  end subroutine sqlr_readSqlite

 843  ! sqlr_updateSqlite
 844  !--------------------------------------------------------------------------
 845  subroutine sqlr_updateSqlite(db, obsdat, familyType, fileName, fileNumber)
 846    !
 847    !:Purpose: update SQLite files. List of items to update is in the 
 848    !           namSQLUpdate namelist
 849    implicit none
 850    
 851    ! Arguments:
 852    type(fSQL_database), intent(inout) :: db         ! SQL database
 853    type (struct_obs)  , intent(inout) :: obsdat     ! obsSpaceData
 854    character(len=*)   , intent(in)    :: fileName   ! file name  
 855    character(len=*)   , intent(in)    :: familyType ! Observation Family Type 
 856    integer            , intent(in)    :: fileNumber ! FILE NUMBER ASSOCIATED WITH db
 857    
 858    ! Locals:
 859    type(fSQL_statement)        :: stmt ! prepared statement for  SQLite
 860    type(fSQL_status)           :: stat ! type error status
 861    integer                     :: obsRln, obsNlv, obsIdf, obsFlag
 862    integer                     :: obsStatus, last_question, landSea, terrainType
 863    integer(8)                  :: headPrimaryKey, bodyPrimaryKey
 864    integer                     :: itemIndex, headPrimaryKeyIndex, landSeaIndex
 865    integer                     :: headerIndex, bodyIndex
 866    character(len =   3)        :: item
 867    integer                     :: updateList(20), fnom, fclos, nulnam, ierr
 868    character(len =  10)        :: columnName
 869    character(len = 128)        :: query
 870    character(len = 356)        :: itemChar, columnNameChar
 871    logical                     :: back, nonEmptyColumn, nonEmptyColumn_mpiglobal
 872    real(4)                     :: romp, obsValue, scaleFactor, columnValue
 873
 874    integer, parameter :: maxNumberUpdate = 15
 875    ! namelist variables:
 876    integer          :: numberUpdateItems                    ! MUST NOT BE INCLUDED IN NAMELIST!
 877    integer          :: numberUpdateItemsRadar               ! MUST NOT BE INCLUDED IN NAMELIST!
 878    character(len=3) :: itemUpdateList(maxNumberUpdate)      ! List of columns to be updated (e.g.'OMA','OMP')
 879    character(len=3) :: itemUpdateListRadar(maxNumberUpdate) ! List of columns to be updated for Radar data
 880
 881    namelist/namSQLUpdate/ numberUpdateItems,      itemUpdateList,     &
 882                           numberUpdateItemsRadar, itemUpdateListRadar
 883
 884    write(*,*) 'sqlr_updateSqlite: Starting ===================  '
 885
 886    ! set default values of namelist variables
 887    itemUpdateList(:) = ''
 888    itemUpdateListRadar(:) = ''
 889    numberUpdateItems = MPC_missingValue_INT
 890    numberUpdateItemsRadar = MPC_missingValue_INT
 891
 892    ! Read the namelist for directives
 893    nulnam = 0
 894    ierr   = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
 895    read(nulnam,nml = namSQLUpdate, iostat = ierr)
 896    if (ierr /= 0) call utl_abort('sqlr_updateSqlite: Error reading namelist')
 897    if (mmpi_myid == 0) write(*, nml = namSQLUpdate)
 898    ierr = fclos(nulnam)
 899    if (numberUpdateItems /= MPC_missingValue_INT) then
 900      call utl_abort('sqlr_updateSqlite: check namelist section namSQLUpdate, numberUpdateItems should be removed')
 901    end if
 902    if (numberUpdateItemsRadar /= MPC_missingValue_INT) then
 903      call utl_abort('sqlr_updateSqlite: check namelist section namSQLUpdate, numberUpdateItemsRadar should be removed')
 904    end if
 905    numberUpdateItems = 0
 906    do itemIndex = 1, maxNumberUpdate
 907      if (trim(itemUpdateList(itemIndex)) == '') exit
 908      numberUpdateItems = numberUpdateItems + 1
 909    end do
 910    numberUpdateItemsRadar = 0
 911    do itemIndex = 1, maxNumberUpdate
 912      if (trim(itemUpdateListRadar(itemIndex)) == '') exit
 913      numberUpdateItemsRadar = numberUpdateItemsRadar + 1
 914    end do
 915    
 916    ! Append extra sqlite columns to update to itemUpdateList
 917    if (trim(familyType) == 'RA') then
 918      do itemIndex = 1, numberUpdateItemsRadar
 919        numberUpdateItems = numberUpdateItems + 1
 920        itemUpdateList(numberUpdateItems) = itemUpdateListRadar(itemIndex)
 921      end do
 922    end if
 923
