 21  !
 22  use codePrecision_mod
 23  use ramDisk_mod
 24  use midasMpi_mod
 25  use mathPhysConstants_mod
 26  use horizontalCoord_mod
 27  use verticalCoord_mod
 28  use timeCoord_mod
 29  use obsSpaceData_mod
 30  use columnData_mod
 31  use gridStateVector_mod
 32  use gridStateVectorFileIO_mod
 33  use obsFiles_mod
 34  use innovation_mod
 35  use obsErrors_mod
 36  use biasCorrectionConv_mod
 37  use obsSpaceErrorStdDev_mod
 38  use ensembleObservations_mod
 39  use ensembleStateVector_mod
 40  use fileNames_mod
 41  use statetocolumn_mod
 42  implicit none
 43  private
 44
 45  ! public subroutines and functions
 46  public :: omf_oMinusF, omf_oMinusFens

 50    !--------------------------------------------------------------------------
 51    ! omf_oMinusF
 52    !--------------------------------------------------------------------------
 53    subroutine omf_oMinusF(columnTrlOnAnlIncLev, columnTrlOnTrlLev, obsSpaceData, &
 54                           midasMode, addHBHT, addSigmaO)
 55      !
 56      !:Purpose: compute Observation-minus-Forecast (OmF)
 57      !
 58
 59      implicit none
 60      ! Arguments:
 61      type(struct_columnData),target, intent(inout)  :: columnTrlOnAnlIncLev
 62      type(struct_columnData),target, intent(inout)  :: columnTrlOnTrlLev
 63      type(struct_obs),       target, intent(inout)  :: obsSpaceData
 64      type(struct_gsv)             :: stateVectorTrialHighRes
 65      character(len=*), intent(in) :: midasMode
 66      logical, intent(in) :: addHBHT
 67      logical, intent(in) :: addSigmaO
 68
 69      ! locals
 70      type(struct_vco),       pointer :: vco_anl  => null()
 71      type(struct_hco),       pointer :: hco_anl  => null()
 72      type(struct_hco),       pointer :: hco_trl => null()
 73      type(struct_vco),       pointer :: vco_trl => null()
 74      type(struct_hco),       pointer :: hco_core => null()
 75
 76      logical           :: allocHeightSfc
 77
 78      character(len=48) :: obsMpiStrategy
 79      character(len=3)  :: obsColumnMode
 80      character(len=10) :: trialFileName
 81
 82      integer :: dateStampFromObs, get_max_rss
 83
 84      write(*,*) " ---------------------------------------"
 85      write(*,*) " ---  START OF SUBROUTINE oMinusF    ---"
 86      write(*,*) " ---  Computation of the innovation  ---"
 87      write(*,*) " ---------------------------------------"
 88
 89      !
 90      !- 1.  Settings and module initializations
 91      !
 92      write(*,*)
 93      write(*,*) '> omf_oMinusF: setup - START'
 94
 95      obsMpiStrategy = 'LIKESPLITFILES'
 96      obsColumnMode  = 'VAR'
 97      trialFileName  = './trlm_01'
 98
 99      !- 1.3 RAM disk usage
100      call ram_setup
101
102      !- 1.4 Temporal grid and dateStamp from trial file
103      call tim_setup(fileNameForDate_opt=trim(trialFileName))
104
105      !- 1.5 Observation file names and get datestamp, but do not use it
106      call obsf_setup(dateStampFromObs, trim(midasMode))
107
108      !- 1.6 Constants
109      if ( mmpi_myid == 0 ) then
110        call mpc_printConstants(6)
111        call pre_printPrecisions
112      end if
113
114      !- 1.7 Variables of the model states
115      call gsv_setup
116
117      !- 1.8 Set the horizontal domain
118      if ( addHBHT ) then
119        call hco_SetupFromFile(hco_anl, './analysisgrid', 'ANALYSIS') ! IN
120        if ( hco_anl % global ) then
121          hco_core => hco_anl
122        else
123          !- Iniatilized the core (Non-Exteded) analysis grid
124          call hco_SetupFromFile( hco_core, './analysisgrid', 'COREGRID') ! IN
125        end if
126      else
127        call hco_SetupFromFile(hco_anl, trim(trialFileName), ' ') ! IN
128        hco_core => hco_anl
129      end if
130
131      ! 1.9 Setup a column vector following the analysis vertical grid
132      if ( addHBHT ) then
133        call vco_SetupFromFile(vco_anl,        & ! OUT
134                               './analysisgrid') ! IN
135        call col_setVco(columnTrlOnAnlIncLev,vco_anl)
136      end if
137
138      !- 1.10 Setup and read observations
139      call inn_setupObs(obsSpaceData, hco_anl, obsColumnMode, obsMpiStrategy,trim(midasMode))!IN
140
141      ! Apply optional bias corrections when namelist logicals {fam}BiasActive are TRUE
142      ! (Only reverse existing corrections when namelist logicals {fam}RevOnly are TRUE)
143      if (obs_famExist(obsSpaceData,'AI')) call bcc_applyAIBcor(obsSpaceData)    
144      if (obs_famExist(obsSpaceData,'GP')) call bcc_applyGPBcor(obsSpaceData)
145      if (obs_famExist(obsSpaceData,'UA')) call bcc_applyUABcor(obsSpaceData)
146      
147      !- 1.11 Basic setup of columnData module
148      call col_setup
149
150      !- 1.12 Memory allocation for background column data
151      if ( addHBHT ) then
152        call col_allocate(columnTrlOnAnlIncLev, obs_numheader(obsSpaceData),mpiLocal_opt=.true.)
153      end if
154
155      if ( addSigmaO ) then
156        !- 1.13 Initialize the observation error covariances
157        write(*,*)
158        write(*,*) '> omf_oMinusF: Adding sigma_O'
159        call oer_setObsErrors(obsSpaceData, trim(midasMode))
160      end if
161
162      ! Reading trials
163      call inn_getHcoVcoFromTrlmFile( hco_trl, vco_trl )
164      allocHeightSfc = ( vco_trl%Vcode /= 0 )
165
166      call gsv_allocate( stateVectorTrialHighRes, tim_nstepobs, hco_trl, vco_trl,  &
167                         dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true., &
168                         mpi_distribution_opt='Tiles', dataKind_opt=4,  &
169                         allocHeightSfc_opt=allocHeightSfc, hInterpolateDegree_opt='LINEAR', &
170                         beSilent_opt=.false. )
171      call gsv_zero( stateVectorTrialHighRes )
172      call gio_readTrials( stateVectorTrialHighRes )
173      write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
174
175      ! Horizontally interpolate trials to trial columns
176      call inn_setupColumnsOnTrlLev( columnTrlOnTrlLev, obsSpaceData, hco_core, &
177                                       stateVectorTrialHighRes )
178
179      write(*,*)
180      write(*,*) '> omf_oMinusF: setup - END'
181      write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
182
183      !
184      !- 2.  O-F computation
185      !
186
187      !- 2.1 Compute observation innovations
188      write(*,*)
189      write(*,*) '> omf_oMinusF: compute innovation'
190      call inn_computeInnovation(columnTrlOnTrlLev, obsSpaceData, analysisMode_opt=.false.)
191
192      if ( addHBHT ) then
193        write(*,*)
194        write(*,*) '> omf_oMinusF: Adding HBH^T'
195        !- 2.2 Interpolate background columns to analysis levels and setup for linearized H
196        call inn_setupColumnsOnAnlIncLev( columnTrlOnTrlLev, columnTrlOnAnlIncLev )
197        !- 2.3 Compute the background errors in observation space
198        call ose_computeStddev(columnTrlOnAnlIncLev,hco_anl,obsSpaceData)
199      end if
200
201    end subroutine omf_oMinusF

203    !--------------------------------------------------------------------------
204    ! omf_oMinusFens
205    !--------------------------------------------------------------------------
206    subroutine omf_oMinusFens(ensObs, obsSpaceData, nEns, ensPathName, &
207                              midasMode, addHBHT, addSigmaO)
208      !
209      !:Purpose: compute Observation-minus-Forecast (OmF) for ensembles
210      !
211
212      implicit none
213      ! Arguments:
214      type(struct_eob), target, intent(inout)  :: ensObs
215      type(struct_obs), target, intent(inout)  :: obsSpaceData
216      integer, intent(in) :: nEns
217      character(len=*), intent(in) :: ensPathName
218      character(len=*), intent(in) :: midasMode
219      logical, intent(in) :: addHBHT
220      logical, intent(in) :: addSigmaO
221
222      ! locals
223      type(struct_columnData)   :: columTrlOnTrlLev
224      type(struct_columnData)   :: columnTrlOnAnlIncLev
225      type(struct_ens)          :: ensembleTrl4D
226      type(struct_gsv)          :: stateVector4D
227      type(struct_gsv)          :: stateVectorWithZandP4D
228      type(struct_gsv)          :: stateVectorHeightSfc      
229      
230      type(struct_vco), pointer :: vco_anl  => null()
231      type(struct_hco), pointer :: hco_anl  => null()
232      type(struct_hco), pointer :: hco_ens  => null()
233      type(struct_vco), pointer :: vco_ens  => null()
234      type(struct_hco), pointer :: hco_core => null()
235
236      character(len=256) :: ensFileName
237      character(len=48)  :: obsMpiStrategy
238      character(len=3)   :: obsColumnMode
239
240      integer, allocatable :: dateStampList(:)
241      
242      integer :: datestamp, get_max_rss, memberIndex
243
244      write(*,*) " -----------------------------------------"
245      write(*,*) " ---  START OF SUBROUTINE oMinusFens   ---"
246      write(*,*) " ---  Computation of the innovation    ---"
247      write(*,*) " -----------------------------------------"
248
249      !
250      !- 1.  Settings and module initializations
251      !
252      write(*,*)
253      write(*,*) '> omf_oMinusFens: setup - START'
254
255      obsMpiStrategy = 'LIKESPLITFILES'
256      obsColumnMode  = 'VAR'
257 
258      !- 1.3 RAM disk usage
259      call ram_setup
260
261      !- 1.4 Horizontal, vertical and temporal dimensions
262      call fln_ensFileName(ensFileName, ensPathName, memberIndex_opt=1)
263      
264      call tim_setup( fileNameForDate_opt=trim(ensFileName) )
265      allocate(dateStampList(tim_nstepobs))
266      call tim_getstamplist(dateStampList,tim_nstepobs,tim_getDatestamp())
267      
268      call hco_setupFromFile(hco_ens, ensFileName, ' ', 'ENSFILEGRID')
269      call vco_setupFromFile(vco_ens, ensFileName)
270      
271      !- 1.5 Observation file names and get datestamp, but do not use it
272      call obsf_setup(dateStamp, trim(midasMode))
273
274      !- 1.6 Constants
275      if ( mmpi_myid == 0 ) then
276        call mpc_printConstants(6)
277        call pre_printPrecisions
278      end if
279
280      !- 1.7 Variables of the model states
281      call gsv_setup
282
283      !- 1.8 Set the horizontal domain
284      if ( addHBHT ) then
285        call hco_SetupFromFile(hco_anl, './analysisgrid', 'ANALYSIS') ! IN
286        if ( hco_anl % global ) then
287          hco_core => hco_anl
288        else
289          !- Iniatilized the core (Non-Exteded) analysis grid
290          call hco_SetupFromFile( hco_core, './analysisgrid', 'COREGRID') ! IN
291        end if
292      else
293        hco_core => hco_ens
294      end if
295
296      ! 1.9 Setup a column vector following the analysis vertical grid
297      if ( addHBHT ) then
298        call vco_SetupFromFile(vco_anl,        & ! OUT
299                               './analysisgrid') ! IN
300        call col_setVco(columnTrlOnAnlIncLev,vco_anl)
301      end if
302
303      !- 1.10 Setup and read observations
304      call inn_setupObs(obsSpaceData, hco_core, obsColumnMode, obsMpiStrategy, trim(midasMode)) !IN
305
306      ! Apply optional bias corrections when namelist logicals {fam}BiasActive are TRUE
307      ! (Only reverse existing corrections when namelist logicals {fam}RevOnly are TRUE)
308      if (obs_famExist(obsSpaceData,'AI')) call bcc_applyAIBcor(obsSpaceData)    
309      if (obs_famExist(obsSpaceData,'GP')) call bcc_applyGPBcor(obsSpaceData)
310      if (obs_famExist(obsSpaceData,'UA')) call bcc_applyUABcor(obsSpaceData)
311      
312      !- 1.11 Basic setup of columnData module
313      call col_setup
314      call col_setVco(columTrlOnTrlLev, vco_ens)
315      call col_allocate(columTrlOnTrlLev, obs_numheader(obsSpaceData))
316      write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
317
318      !- 1.12 Memory allocation for background column data
319      if ( addHBHT ) then
320        call col_allocate(columnTrlOnAnlIncLev, obs_numheader(obsSpaceData),mpiLocal_opt=.true.)
321      end if
322
323      if ( addSigmaO ) then
324        !- 1.13 Initialize the observation error covariances
325        write(*,*)
326        write(*,*) '> omf_oMinusF: Adding sigma_O'
327        call oer_setObsErrors(obsSpaceData, trim(midasMode))
328      end if
329
330      !- 1.14 Allocate and initialize eob object for storing y-HX values
331      call eob_allocate(ensObs, nEns, obs_numBody(obsSpaceData), obsSpaceData)
332      call eob_zero(ensObs)
333      call eob_setLatLonObs(ensObs)
334
335      !- 1.15 Allocate statevector for storing state with heights and pressures allocated (for s2c_nl)
336      call gsv_allocate(stateVectorWithZandP4D, tim_nstepobs, hco_ens, vco_ens, &
337                        dateStamp_opt=tim_getDateStamp(),  &
338                        mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
339                        dataKind_opt=4, allocHeightSfc_opt=.true.)
340      call gsv_zero(stateVectorWithZandP4D)
341
342      call gsv_allocate(stateVector4D, tim_nstepobs, hco_ens, vco_ens, &
343                        dateStamp_opt=tim_getDateStamp(),  &
344                        mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
345                        dataKind_opt=4, allocHeightSfc_opt=.true., &
346                        allocHeight_opt=.false., allocPressure_opt=.false.)
347      call gsv_zero(stateVector4D)
348      
349      !- 1.16 Read the sfc height from ensemble member 1
350      call gsv_allocate(stateVectorHeightSfc, 1, hco_ens, vco_ens, dateStamp_opt=tim_getDateStamp(),  &
351                        mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
352                        dataKind_opt=4, allocHeightSfc_opt=.true., varNames_opt=(/'P0','TT'/))
353      call gio_readFromFile(stateVectorHeightSfc, ensFileName, ' ', ' ',  &
354                            containsFullField_opt=.true., readHeightSfc_opt=.true.)
355
356      !- 1.17 Reading ensemble
357      call ens_allocate(ensembleTrl4D, nEns, tim_nstepobs, hco_ens, vco_ens, dateStampList)
358      call ens_readEnsemble(ensembleTrl4D, ensPathName, biPeriodic=.false.)
359      
360      write(*,*)
361      write(*,*) '> omf_oMinusFens: setup - END'
362      write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
363
364      !
365      !- 2.  O-F computation
366      !
367      do memberIndex = 1, nEns
368        write(*,*) ''
369        write(*,*) 'oMinusFens: compute O-P for ensemble member ', memberIndex
370        write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
371        
372        !- 2.1 Copy selected member to a stateVector
373        call ens_copyMember(ensembleTrl4D, stateVector4D, memberIndex)
374        call gsv_copy(stateVector4D, stateVectorWithZandP4D, allowVarMismatch_opt=.true., &
375                      beSilent_opt=.true.)
376        call gsv_copyHeightSfc(stateVectorHeightSfc, stateVectorWithZandP4D)
377
378        !- 2.2 Compute and set Yb in ensObs
379        call s2c_nl(stateVectorWithZandP4D, obsSpaceData, columTrlOnTrlLev, hco_ens, &
380                   timeInterpType='LINEAR', dealloc_opt=.false., &
381                   beSilent_opt=.true.)
382        
383        !- 2.3 Compute innovation
384        call inn_computeInnovation(columTrlOnTrlLev, obsSpaceData, analysisMode_opt=.false., beSilent_opt=.true.)
385
386        !- 2.4 Copy to ensObs
387        call eob_setYb(ensObs, memberIndex)
388
389        !- 2.5 Add background errors in observation space
390        if (addHBHT .and. memberIndex == 1) then
391          write(*,*)
392          write(*,*) '> omf_oMinusFens: Adding HBH^T'
393          !  Interpolate background columns to analysis levels and setup for linearized H
394          call inn_setupColumnsOnAnlIncLev(columTrlOnTrlLev, columnTrlOnAnlIncLev)
395          !  Compute the background errors in observation space
396          call ose_computeStddev(columnTrlOnAnlIncLev,hco_anl,obsSpaceData)
397        end if
398
399      end do
400
401      !
402      !- 3.  Ending
403      !
404      call gsv_deallocate(stateVectorWithZandP4D)
405      call gsv_deallocate(stateVector4D)
406      call gsv_deallocate(stateVectorHeightSfc)
407
408    end subroutine omf_oMinusFens
409    
410end module oMinusF_mod
