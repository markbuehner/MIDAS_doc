  85  subroutine fso_setup(hco_anl_in)
  86    !
  87    !:Purpose: Initialise the FSOI module: read the namelist and initialise
  88    !           global variables and structure
  89    !
  90    implicit none
  91
  92    ! Arguments:
  93    type(struct_hco), pointer, intent(in) :: hco_anl_in
  94
  95    ! Locals:
  96    integer :: ierr,nulnam
  97    integer :: fnom,fclos
  98
  99    NAMELIST /NAMFSO/leadTime, nvamaj, nitermax, nsimmax
 100    NAMELIST /NAMFSO/repsg, rdf1fac, forecastPath, fsoMode
 101    NAMELIST /NAMFSO/latMinNorm, latMaxNorm, lonMinNorm, lonMaxNorm
 102    NAMELIST /NAMFSO/includeUVnorm, includeTTnorm, includeP0norm, includeHUnorm, includeTGnorm
 103
 104    ! set default values for namelist variables
 105    leadtime = 12.0d0
 106    nvamaj = 6
 107    nitermax = 100
 108    nsimmax  = 120
 109    repsg    = 1d-5
 110    rdf1fac  = 0.25d0
 111    latMinNorm = -95.0d0
 112    latMaxNorm = 95.0d0
 113    lonMinNorm = -185.0d0
 114    lonMaxNorm = 365.0d0
 115    includeUVnorm=.true.
 116    includeTTnorm=.true.
 117    includeP0norm=.true.
 118    includeHUnorm=.false.
 119    includeTGnorm=.false.
 120    forecastPath = './forecasts'
 121    fsoMode  = 'HFSO'
 122
 123    ! read in the namelist NAMFSO
 124    nulnam = 0
 125    ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
 126    read(nulnam,nml=namfso,iostat=ierr)
 127    if(ierr /= 0) call utl_abort('fso_setup: Error reading namelist')
 128    write(*,nml=namfso)
 129    ierr = fclos(nulnam)
 130
 131    if (trim(fsoMode) /= 'HFSO' .and. trim(fsoMode) /= 'EFSO') then
 132      call utl_abort('fso_setup: Invalid value of fsoMode. Must be HFSO or EFSO')
 133    end if
 134
 135    ! convert Latmin,max and Lonmin,max from degres into RAD
 136    latMinNorm = latMinNorm*MPC_RADIANS_PER_DEGREE_R8
 137    latMaxNorm = latMaxNorm*MPC_RADIANS_PER_DEGREE_R8
 138    lonMinNorm = lonMinNorm*MPC_RADIANS_PER_DEGREE_R8
 139    lonMaxNorm = lonMaxNorm*MPC_RADIANS_PER_DEGREE_R8
 140
 141    call ben_setFsoLeadTime(leadTime)
 142    fso_nsim = 0
 143
 144    hco_anl => hco_anl_in
 145
 146  end subroutine fso_setup

 148  !--------------------------------------------------------------------------
 149  ! fso_ensemble
 150  !--------------------------------------------------------------------------
 151  subroutine fso_ensemble(columnTrlOnAnlIncLev,obsSpaceData)
 152    !
 153    !:Purpose: Perform forecast sensitivity to observation calculation using
 154    !           ensemble approach
 155    !
 156    implicit none
 157
 158    ! Arguments:
 159    type(struct_columnData), target, intent(in)     :: columnTrlOnAnlIncLev
 160    type(struct_obs),        target, intent(inout)  :: obsSpaceData
 161
 162    ! Locals:
 163    type(struct_columnData),target  :: column
 164    type(struct_gsv)                :: statevector_FcstErr, statevector_fso, statevector_HUreference
 165    type(struct_vco), pointer       :: vco_anl
 166    real(8),allocatable             :: ahat(:), zhat(:)
 167    integer                         :: dateStamp_fcst, dateStamp
 168
 169    !for Observation space
 170    integer                         :: headerIndex, bodyIndexBeg, bodyIndexEnd, bodyIndex
 171    real(8)                         :: fso_ori, fso_fin
 172
 173    if (mmpi_myid == 0) write(*,*) 'fso_ensemble: starting'
 174
 175    vco_anl => col_getVco(columnTrlOnAnlIncLev)
 176
 177    nvadim_mpilocal = cvm_nvadim
 178
 179    ! initialize column object for storing "increment"
 180    call col_setVco(column,col_getVco(columnTrlOnAnlIncLev))
 181    call col_allocate(column,col_getNumCol(columnTrlOnAnlIncLev),mpiLocal_opt=.true.)
 182
 183    ! compute dateStamp_fcst
 184    call incdatr(dateStamp_fcst, tim_getDatestamp(), leadTime)
 185    dateStamp = tim_getDatestamp()
 186    write(*,*) 'fso_ensemble: analysis datestamp = ',dateStamp
 187    write(*,*) 'fso_ensemble: forecast datestamp = ',dateStamp_fcst
 188
 189    ! allocate control vector related arrays (these are all mpilocal)
 190    allocate(ahat(nvadim_mpilocal))
 191    allocate(vhat(nvadim_mpilocal))
 192    allocate(zhat(nvadim_mpilocal))
 193
 194    ! initialize control vector related arrays to zero
 195    ahat(:) = 0.0d0
 196    vhat(:) = 0.0d0
 197
 198    ! for statevector_FcstErr
 199    call gsv_allocate(statevector_FcstErr, 1, hco_anl, vco_anl, &
 200                      datestamp_opt=datestamp_fcst, mpi_local_opt=.true., &
 201                      allocHeight_opt=.false., allocPressure_opt=.false.)
 202
 203    ! for statevector_fso
 204    call gsv_allocate(statevector_fso, tim_nstepobsinc, hco_anl, vco_anl, &
 205                      dataKind_opt=pre_incrReal, &
 206                      datestamp_opt=tim_getDatestamp(), mpi_local_opt=.true.)
 207
 208    ! for statevector_HUreference (verifying analysis)
 209    call gsv_allocate(statevector_HUreference, 1, hco_anl, vco_anl, &
 210                      datestamp_opt=datestamp_fcst, mpi_local_opt=.true., &
 211                      allocHeight_opt=.false., allocPressure_opt=.false.)
 212
 213    ! compute forecast error = C * (error_t^fa + error_t^fb)
 214    call calcFcstError(columnTrlOnAnlIncLev,statevector_FcstErr,statevector_HUreference)
 215
 216
 217    ! compute vhat = B_t^T/2 * C * (error_t^fa + error_t^fb)
 218    call bmat_sqrtBT(vhat, nvadim_mpilocal, statevector_FcstErr, useFSOFcst_opt = .true., &
 219    stateVectorRef_opt=statevector_HUreference)
 220
 221    if (mmpi_myid == 0) write(*,*) 'fso: B_t^T/2 * C * (error_t^fa + error_t^fb) max,min:', &
 222        maxval(vhat),minval(vhat)
 223
 224    if( trim(fsoMode) == 'HFSO' ) then
 225      call minimize(nvadim_mpilocal, zhat, column, columnTrlOnAnlIncLev, obsSpaceData)
 226      ahat = zhat + vhat
 227      call bmat_sqrtB(ahat, nvadim_mpilocal, statevector_fso)
 228    elseif( trim(fsoMode) == 'EFSO' ) then
 229      call bmat_sqrtB(vhat, nvadim_mpilocal, statevector_fso)
 230    end if
 231
 232    ! Compute yhat = [R^-1 H B^1/2 ahat], and put in OBS_FSO
 233    call s2c_tl(statevector_fso,column,columnTrlOnAnlIncLev,obsSpaceData)  ! put in column H_horiz B^1/2 ahat
 234    call oop_Htl(column,columnTrlOnAnlIncLev,obsSpaceData,1)          ! Save as OBS_WORK: H_vert H_horiz B^1/2 vhat = H B^1/2 ahat
 235    call rmat_RsqrtInverseAllObs(obsSpaceData,OBS_FSO,OBS_WORK) ! Save as OBS_FSO : R**-1/2 H B^1/2 ahat
 236    call rmat_RsqrtInverseAllObs(obsSpaceData,OBS_FSO,OBS_FSO)  ! Save as OBS_FSO : R**-1 H B^1/2 ahat\
 237
 238    do headerIndex = 1, obs_numHeader(obsSpaceData)
 239
 240      bodyIndexBeg = obs_headElem_i(obsSpaceData,OBS_RLN,headerIndex)
 241      bodyIndexEnd = obs_headElem_i(obsSpaceData,OBS_NLV,headerIndex) + bodyIndexBeg - 1
 242
 243      do bodyIndex = bodyIndexBeg, bodyIndexEnd
 244        if ( obs_bodyElem_i(obsSpaceData,OBS_ASS,bodyIndex) == obs_assimilated ) then
 245          fso_ori = obs_bodyElem_r(obsSpaceData,OBS_FSO,bodyIndex)
 246          fso_fin = fso_ori * obs_bodyElem_r(obsSpaceData,OBS_OMP,bodyIndex)
 247          call obs_bodySet_r(obsSpaceData,OBS_FSO,bodyIndex, fso_fin)
 248        end if
 249      end do
 250
 251    end do
 252
 253    ! print out the information of total FSO for each family
 254    call sumFSO(obsSpaceData)
 255
 256    ! deallocate the control vector related arrays
 257    deallocate(ahat)
 258    deallocate(vhat)
 259    deallocate(zhat)
 260    call col_deallocate(column)
 261
 262    if (mmpi_myid == 0) write(*,*) 'fso_ensemble: Finished'
 263
 264  end subroutine fso_ensemble

 266  !--------------------------------------------------------------------------
 267  ! calcFcstError
 268  !--------------------------------------------------------------------------
 269  subroutine calcFcstError(columnTrlOnAnlIncLev,statevector_out,statevector_verifAnalysis)
 270    !
 271    !:Purpose: Reads the forecast from background and analysis, the verifying
 272    !           analysis based on these inputs, calculates the Forecast error
 273    !
 274    implicit none
 275
 276    ! Arguments:
 277    type(struct_columnData), target, intent(in)     :: columnTrlOnAnlIncLev
 278    type(struct_gsv)       , target, intent(inout)  :: statevector_out, statevector_verifAnalysis
 279
 280    ! Locals:
 281    type(struct_gsv)                :: statevector_fa, statevector_fb, statevector_a
 282    character(len=256)              :: fileName_fa, fileName_fb, fileName_a
 283    logical                         :: faExists
 284    type(struct_gsv)                :: statevector_tempfa, statevector_tempfb
 285    type(struct_vco), pointer       :: vco_anl
 286    integer                         :: dateStamp_fcst, dateStamp
 287
 288    vco_anl => col_getVco(columnTrlOnAnlIncLev)
 289
 290    ! compute dateStamp_fcst
 291    call incdatr(dateStamp_fcst, tim_getDatestamp(), leadTime)
 292    dateStamp = tim_getDatestamp()
 293    write(*,*) 'fso_ensemble: analysis datestamp = ',dateStamp
 294    write(*,*) 'fso_ensemble: forecast datestamp = ',dateStamp_fcst
 295
 296    ! read forecasts from the analysis and background state
 297    fileName_fa = trim(forecastPath) // '/forecast_a'
 298    inquire(file=trim(fileName_fa),exist=faExists)
 299    write(*,*) 'faExists', faExists
 300    call gsv_allocate(statevector_fa, 1, hco_anl, vco_anl, &
 301                      datestamp_opt=datestamp_fcst, mpi_local_opt=.true., &
 302                      hInterpolateDegree_opt='LINEAR', &
 303                      allocHeight_opt=.false., allocPressure_opt=.false.)
 304    call gio_readFromFile(statevector_fa, fileName_fa, ' ', 'P', containsFullField_opt=.true.)
 305
 306    !for statevecotr_tempfa
 307    call gsv_allocate(statevector_tempfa, 1, hco_anl, vco_anl, &
 308                      datestamp_opt=datestamp_fcst, mpi_local_opt=.true., &
 309                      allocHeight_opt=.false., allocPressure_opt=.false.)
 310
 311    !for statevector_fb
 312    fileName_fb = trim(forecastPath) // '/forecast_b'
 313    call gsv_allocate(statevector_fb, 1, hco_anl, vco_anl, &
 314                      datestamp_opt=datestamp_fcst, mpi_local_opt=.true., &
 315                      hInterpolateDegree_opt='LINEAR', &
 316                      allocHeight_opt=.false., allocPressure_opt=.false.)
 317    call gio_readFromFile(statevector_fb, fileName_fb, ' ', 'P', containsFullField_opt=.true.)
 318
 319    !for statevecotr_tempfb
 320    call gsv_allocate(statevector_tempfb, 1, hco_anl, vco_anl, &
 321                      datestamp_opt=datestamp_fcst, mpi_local_opt=.true., &
 322                      allocHeight_opt=.false., allocPressure_opt=.false.)
 323
 324    ! read verifying analysis
 325    fileName_a = trim(forecastPath) // '/analysis'
 326    call gsv_allocate(statevector_a, 1,hco_anl, vco_anl, &
 327                      datestamp_opt=datestamp_fcst, mpi_local_opt=.true., &
 328                      hInterpolateDegree_opt='LINEAR', &
 329                      allocHeight_opt=.false., allocPressure_opt=.false.)
 330    call gio_readFromFile(statevector_a, fileName_a, ' ', 'A', containsFullField_opt=.true.)
 331
 332    ! compute error of both forecasts (overwrite forecasts with error)
 333    call gsv_add(statevector_a, statevector_fa, -1.0d0)
 334    call gsv_add(statevector_a, statevector_fb, -1.0d0)
 335
 336    call gsv_copy(statevector_fa,statevector_tempfa)
 337    call gsv_copy(statevector_fb,statevector_tempfb)
 338    call multEnergyNorm(statevector_tempfa, statevector_a,  &
 339                            latMinNorm, latMaxNorm, lonMinNorm, lonMaxNorm, &
 340                            includeUVnorm, includeTTnorm, includeP0norm,  &
 341                            includeHUnorm, includeTGnorm) ! use analysis as reference state
 342    call multEnergyNorm(statevector_tempfb, statevector_a,  &
 343                            latMinNorm, latMaxNorm, lonMinNorm, lonMaxNorm, &
 344                            includeUVnorm, includeTTnorm, includeP0norm,  &
 345                            includeHUnorm, includeTGnorm) ! use analysis as reference state
 346
 347    ! compute error Norm =  C * (error_t^fa + error_t^fb)
 348    call gsv_add(statevector_fa, statevector_fb, 1.0d0)
 349    call multEnergyNorm(statevector_fb, statevector_a,  &
 350                            latMinNorm, latMaxNorm, lonMinNorm, lonMaxNorm, &
 351                            includeUVnorm, includeTTnorm, includeP0norm,  &
 352                            includeHUnorm, includeTGnorm) ! use analysis as reference state
 353    call gsv_copy(statevector_fb,statevector_out)
 354    call gsv_copy(statevector_a,statevector_verifAnalysis)
 355
 356  end subroutine calcFcstError

 359  ! minimize
 360  !--------------------------------------------------------------------------
 361  subroutine minimize(nvadim,zhat,column,columnTrlOnAnlIncLev,obsSpaceData)
 362    !
 363    !:Purpose: Performs HFSO quasi-Newton minimization
 364    !
 365    implicit none
 366
 367    ! Arguments:
 368    integer,                         intent(in)   :: nvadim
 369    real(8), dimension(nvadim),      intent(out)  :: zhat
 370    type(struct_columnData), target, intent(in)   :: column, columnTrlOnAnlIncLev
 371    type(struct_obs),        target, intent(in)   :: obsSpaceData
 372
 373    ! Locals:
 374    integer                         :: nulout = 6
 375    integer                         :: imode, itermax, isimmax, indic, nmtra
 376    integer                         :: impres, iztrl(10), intunused(1)
 377    real                            :: rspunused(1)
 378    real(8)                         :: zjsp, zxmin, zdf1, zeps, dlgnorm, dlxnorm,zspunused(1)
 379    real(8),allocatable             :: gradJ(:), vatra(:)
 380
 381    call utl_tmg_start(90,'--Minimization')
 382
 383    if (mmpi_myid == 0) write(*,*) 'minimize: starting'
 384
 385    nmtra = (4 + 2*nvamaj)*nvadim
 386    write(*,'(4X,"NVAMAJ = ",I3,/5X,"NMTRA =",I14)') nvamaj,nmtra
 387
 388    columnTrlOnAnlIncLev_ptr => columnTrlOnAnlIncLev
 389    column_ptr  => column
 390    obsSpaceData_ptr => obsSpaceData
 391
 392    allocate(gradJ(nvadim))
 393    allocate(vatra(nmtra))
 394
 395    gradJ(:) = 0.0d0
 396    vatra(:) = 0.0d0
 397    zhat(:)  = 0.0d0
 398
 399    ! Compute zhat by performing variational minimization
 400    ! Set-up for the minimization
 401    if (mmpi_myid == 0) then
 402      impres = 5
 403    else
 404      impres = 0
 405    end if
 406
 407    imode = 0
 408    zeps = repsg
 409    itermax = nitermax
 410    isimmax = nsimmax
 411    zxmin = epsilon(zxmin)
 412    ! initial gradient calculation
 413    indic = 2
 414    call utl_tmg_start(91,'----QuasiNewton')
 415    call simvar(indic,nvadim,zhat,zjsp,gradJ)
 416    call utl_tmg_stop(91)
 417    zdf1 =  rdf1fac * abs(zjsp)
 418
 419    ! print amplitude of initial gradient and cost function value
 420    call prscal(nvadim,gradJ,gradJ,dlgnorm)
 421    dlgnorm = dsqrt(dlgnorm)
 422    call prscal(nvadim,zhat,zhat,dlxnorm)
 423    dlxnorm = dsqrt(dlxnorm)
 424    write(*,*)' |X| = ', dlxnorm
 425    write(*,'(/4X,"J(X) = ",G23.16,4X,"|Grad J(X)| = ",G23.16)') zjsp, dlgnorm
 426    write(*,'(//,10X," Minimization QNA_N1QN3 starts ...",/  &
 427             10x,"DXMIN =",G23.16,2X,"DF1 =",G23.16,2X,"EPSG =",G23.16  &
 428             /,10X,"IMPRES =",I3,2X,"NITER = ",I3,2X,"NSIM = ",I3)') zxmin,zdf1,zeps,impres,itermax,isimmax
 429
 430    ! Do the minimization
 431    call utl_tmg_start(91,'----QuasiNewton')
 432    call qna_n1qn3(simvar, prscal, dcanonb, dcanab, nvadim, zhat,  &
 433                   zjsp, gradJ, zxmin, zdf1, zeps, impres, nulout, imode,   &
 434                   itermax,isimmax, iztrl, vatra, nmtra, intunused, rspunused,  &
 435                   zspunused)
 436    call utl_tmg_stop(91)
 437    call fool_optimizer(obsSpaceData)
 438
 439    write(*,'(//,20X,20("*"),2X    &
 440        ,/,20X,"              Minimization ended with MODE:",I4  &
 441        ,/,20X,"                Total number of iterations:",I4  &
 442        ,/,20X,"               Total number of simulations:",I4)' ) imode,itermax,isimmax
 443
 444    if (mmpi_myid == 0) write(*,*) 'minimize: Finished'
 445
 446    deallocate(vatra)
 447    deallocate(gradJ)
 448
 449    call utl_tmg_stop(90)
 450
 451  end subroutine minimize

 454  ! sumFSO
 455  !--------------------------------------------------------------------------
 456  subroutine sumFSO(obsSpaceData)
 457    !
 458    !:Purpose: Print out the information of total FSO for each family
 459    !
 460    implicit none
 461
 462    ! Arguments:
 463    type(struct_obs), intent(in)  :: obsSpaceData
 464
 465    ! Locals:
 466    real(8)            :: pfso_1
 467    integer            :: bodyIndex,itvs,isens,headerIndex
 468    integer            :: bodyIndexBeg, bodyIndexEnd
 469
 470    integer, parameter :: numFamily = 10
 471    character(len=2), parameter :: familyList(numFamily) = (/'UA','AI','SF','SC','TO','SW','PR','RO','GP','CH'/)
 472    real(8)            :: tfso(numFamily), tfsotov_sensors(tvs_nsensors),totFSO
 473    integer            :: numAss_local(numFamily), numAss_global(numFamily)
 474    integer            :: numAss_sensors_loc(tvs_nsensors), numAss_sensors_glb(tvs_nsensors)
 475    integer            :: ierr, familyIndex
 476
 477    if (mmpi_myid == 0) write(*,*) 'sumFSO: Starting'
 478
 479    ! initialize
 480    do familyIndex = 1, numFamily
 481      tfso(familyIndex) = 0.d0
 482      numAss_local(familyIndex) = 0
 483      numAss_global(familyIndex) = 0
 484    end do
 485
 486    tfsotov_sensors(:) = 0.d0
 487    numAss_sensors_loc(:) = 0
 488    numAss_sensors_glb(:) = 0
 489    totFSO = 0.d0
 490
 491    do familyIndex = 1, numFamily
 492      do bodyIndex = 1, obs_numbody(obsSpaceData)
 493
 494        pfso_1 = obs_bodyElem_r(obsSpaceData,OBS_FSO,bodyIndex)
 495        if ( obs_bodyElem_i(obsSpaceData,OBS_ASS,bodyIndex) == 1 ) then
 496          ! FSO for each family
 497          if (obs_getFamily(obsSpaceData,bodyIndex=bodyIndex) == familyList(familyIndex) ) then
 498            tfso(familyIndex) = tfso(familyIndex) + pfso_1
 499            numAss_local(familyIndex) = numAss_local(familyIndex) + 1
 500          end if
 501        end if
 502
 503      end do
 504    end do
 505
 506    do itvs = 1, tvs_nobtov
 507      headerIndex  = tvs_headerIndex(itvs)
 508      if (headerIndex > 0 ) then
 509        bodyIndexBeg = obs_headElem_i(obsSpaceData,OBS_RLN,headerIndex)
 510        bodyIndexEnd = obs_headElem_i(obsSpaceData,OBS_NLV,headerIndex) + bodyIndexBeg - 1
 511        do bodyIndex = bodyIndexBeg, bodyIndexEnd
 512          pfso_1 = obs_bodyElem_r(obsSpaceData,OBS_FSO,bodyIndex)
 513          if ( obs_bodyElem_i(obsSpaceData,OBS_ASS,bodyIndex) == 1 ) then
 514            isens = tvs_lsensor (itvs)
 515            tfsotov_sensors(isens) =  tfsotov_sensors(isens) + pfso_1
 516            numAss_sensors_loc(isens) = numAss_sensors_loc(isens) + 1
 517          end if
 518        end do
 519      end if
 520    end do
 521
 522    do familyIndex = 1, numFamily
 523      call mmpi_allreduce_sumreal8scalar(tfso(familyIndex),'GRID')
 524      totFSO = totFSO + tfso(familyIndex)
 525      call rpn_comm_allreduce(numAss_local(familyIndex), numAss_global(familyIndex) ,1,'MPI_INTEGER','MPI_SUM','GRID',ierr)
 526    end do
 527
 528    do isens = 1, tvs_nsensors
 529      call mmpi_allreduce_sumreal8scalar(tfsotov_sensors(isens),'GRID')
 530      call rpn_comm_allreduce(numAss_sensors_loc(isens), numAss_sensors_glb(isens) ,1,'MPI_INTEGER','MPI_SUM','GRID',ierr)
 531    end do
 532
 533    if (mmpi_myid == 0) then
 534
 535      write(*,*) ' '
 536      write(*,'(a15,f15.8)') 'Total FSO=', totFSO
 537      write(*,*) ' '
 538
 539      do familyIndex = 1, numFamily
 540        write(*,'(a4,a2,a2,f15.8,a16,i10)') 'FSO-', familyList(familyIndex), '=', tfso(familyIndex),'  Count Number=', numAss_global(familyIndex)
 541      end do
 542      write(*,*) ' '
 543
 544      if (tvs_nsensors > 0) then
 545        write(*,'(1x,a)') 'For TOVS decomposition by sensor:'
 546        write(*,'(1x,a)') '#  plt sat ins    FSO'
 547        do isens = 1, tvs_nsensors
 548          write(*,'(i2,1x,a,1x,a,1x,i2,1x,f15.8,i10)') isens,inst_name(tvs_instruments(isens)), &
 549                platform_name(tvs_platforms(isens)),tvs_satellites(isens),tfsotov_sensors(isens), numAss_sensors_glb(isens)
 550        end do
 551        write(*,*) ' '
 552      end if
 553
 554    end if
 555
 556  end subroutine sumFSO

 558  !--------------------------------------------------------------------------
 559  ! simvar
 560  !--------------------------------------------------------------------------
 561  subroutine simvar(indic,nvadim,zhat,Jtotal,gradJ)
 562    !
 563    !:Purpose: Implement the Variational solver as described in
 564    !           Courtier, 1997, Dual formulation of four-dimentional variational
 565    !           assimilation, Q.J.R., pp2449-2461.
 566    !
 567    !:Arguments:
 568    !   :indic:   Value of indic
 569    !             Note: 1 and 4 are reserved values for call back from m1qn3.
 570    !             For direct calls use other value than 1 and 4.
 571    !             =1 No action taken; =4 Both J(u) and its gradient are computed.
 572    !             =2 Same as 4 (compute J and gradJ) but do not interrupt timer
 573    !             of the minimizer.
 574    !             =3 Compute Jo and gradJo only.
 575    !
 576    !   :nvadim:  Dimension of the control vector in forecast error covariances space
 577    !
 578    !   :zhat:    Control variable in forecast error covariances space
 579    !
 580    !   :Jtotal:  Cost function of the Variational algorithm
 581    !
 582    !   :gradJ:   Gradient of the Variational Cost funtion
 583    !
 584    implicit none
 585
 586    ! Arguments:
 587    integer,                    intent(in)    :: indic
 588    integer,                    intent(in)    :: nvadim
 589    real(8), dimension(nvadim), intent(inout) :: zhat
 590    real(8),                    intent(out)   :: Jtotal
 591    real(8), dimension(nvadim), intent(out)   :: gradJ
 592
 593    ! Locals:
 594    real(8) :: ahat_vhat(nvadim)
 595    real(8) :: Jb, Jobs
 596    type(struct_gsv)          :: statevector
 597    type(struct_vco), pointer :: vco_anl
 598
 599    call utl_tmg_stop(91)
 600    call utl_tmg_stop(90)
 601
 602    if (indic /= 1) then ! No action taken if indic == 1
 603      fso_nsim = fso_nsim + 1
 604
 605      if (mmpi_myid == 0) then
 606        write(*,*) 'simvar: entering for simulation ',fso_nsim
 607        write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 608        call flush(6)
 609      end if
 610
 611      ! note: vhat = B_t^T/2 hat(del x_t)
 612      ahat_vhat(1:nvadim_mpilocal) = zhat(1:nvadim_mpilocal) + vhat(1:nvadim_mpilocal)
 613
 614      ! Computation of background term of cost function:
 615      Jb = dot_product(zhat(1:nvadim_mpilocal),zhat(1:nvadim_mpilocal))/2.d0
 616      call mmpi_allreduce_sumreal8scalar(Jb,'GRID')
 617
 618      vco_anl => col_getVco(columnTrlOnAnlIncLev_ptr)
 619      call gsv_allocate(statevector,tim_nstepobsinc, hco_anl, vco_anl, &
 620                        dataKind_opt=pre_incrReal, mpi_local_opt=.true.)
 621
 622      call bmat_sqrtB(ahat_vhat,nvadim_mpilocal,statevector)
 623
 624      call s2c_tl(statevector,column_ptr,columnTrlOnAnlIncLev_ptr,obsSpaceData_ptr)  ! put in column H_horiz dx
 625      call utl_tmg_start(10,'--Observations')
 626      call utl_tmg_start(18,'----ObsOper_TL')
 627      call oop_Htl(column_ptr,columnTrlOnAnlIncLev_ptr,obsSpaceData_ptr,fso_nsim)  ! Save as OBS_WORK: H_vert H_horiz dx = Hdx
 628
 629      call rmat_RsqrtInverseAllObs(obsSpaceData_ptr,OBS_WORK,OBS_WORK)  ! Save as OBS_WORK : R**-1/2 (Hdx)
 630      call utl_tmg_stop(18)
 631      call utl_tmg_stop(10)
 632
 633      call cfn_calcJo(obsSpaceData_ptr)  ! Store J-obs in OBS_JOBS : 1/2 * R**-1 (Hdx)**2
 634
 635      Jobs = 0.d0
 636      call utl_tmg_start(90,'--Minimization')
 637      call utl_tmg_start(92,'----SumCostFunction')
 638      call cfn_sumJo(obsSpaceData_ptr,Jobs)
 639      Jtotal = Jb + Jobs
 640      if (indic == 3) then
 641        Jtotal = Jobs
 642        if (mmpi_myid == 0) write(*,FMT='(6X,"SIMVAR:  JO = ",G23.16,6X)') Jobs
 643      else
 644        Jtotal = Jb + Jobs
 645        if (mmpi_myid == 0) write(*,FMT='(6X,"SIMVAR:  Jb = ",G23.16,6X,"JO = ",G23.16,6X,"Jt = ",G23.16)') Jb,Jobs,Jtotal
 646      end if
 647      call utl_tmg_stop(92)
 648      call utl_tmg_stop(90)
 649
 650      call utl_tmg_start(10,'--Observations')
 651      call rmat_RsqrtInverseAllObs(obsSpaceData_ptr,OBS_WORK,OBS_WORK)  ! Modify OBS_WORK : R**-1 (Hdx)
 652      call utl_tmg_stop(10)
 653
 654      call col_zero(column_ptr)
 655
 656      call utl_tmg_start(10,'--Observations')
 657      call utl_tmg_start(19,'----ObsOper_AD')
 658      call oop_Had(column_ptr,columnTrlOnAnlIncLev_ptr,obsSpaceData_ptr)   ! Put in column : H_vert**T R**-1 (Hdx)
 659      call utl_tmg_stop(19)
 660      call utl_tmg_stop(10)
 661
 662      call s2c_ad(statevector,column_ptr,columnTrlOnAnlIncLev_ptr,obsSpaceData_ptr)  ! Put in statevector H_horiz**T H_vert**T R**-1 (Hdx)
 663
 664      gradJ(:) = 0.d0
 665      call bmat_sqrtBT(gradJ,nvadim_mpilocal,statevector)
 666      call gsv_deallocate(statevector)
 667
 668      if (indic /= 3) then
 669        gradJ(1:nvadim_mpilocal) = zhat(1:nvadim_mpilocal) + gradJ(1:nvadim_mpilocal)
 670      end if
 671    end if
 672
 673    call utl_tmg_start(90,'--Minimization')
 674    call utl_tmg_start(91,'----QuasiNewton')
 675
 676    if (mmpi_myid == 0) write(*,*) 'simvar: Finished'
 677
 678  end subroutine simvar

 765  !--------------------------------------------------------------------------
 766  ! multEnergyNorm
 767  !--------------------------------------------------------------------------
 768  subroutine multEnergyNorm(statevector_inout, statevector_ref,  &
 769                                latMin, latMax, lonMin, lonMax,      &
 770                                uvNorm,ttNorm,p0Norm,huNorm,tgNorm)
 771    !
 772    !:Purpose: Computes energy norms
 773    !           For some positive definite symmetric matrix defining the energy,
 774    !             total energy = x^T * C * x
 775    !             statevector_inout = C * statevector_inout
 776    !           (Buehner, Du and Bedard, 2018)
 777    !
 778    implicit none
 779
 780    ! Arguments:
 781    type(struct_gsv), intent(inout)  :: statevector_inout
 782    type(struct_gsv), intent(in)     :: statevector_ref
 783    real(8),          intent(in)     :: latMin, latMax, lonMin, lonMax
 784    logical,          intent(in)     :: uvNorm, ttNorm, p0Norm, huNorm, tgNorm
 785
 786    ! Locals:
 787    integer              :: stepIndex, lonIndex, levIndex, latIndex, lonIndex2, latIndex2, nLev_M, nLev_T
 788    real(8)              :: scaleFactor, scaleFactorConst, scaleFactorLat, scaleFactorLon, scaleFactorLev
 789    real(8)              :: pfac, tfac, qfac
 790    real(8)              :: sumScale , sumeu, sumev, sumep, sumet, sumeq
 791    real(8), pointer     :: field_UU(:,:,:,:), field_VV(:,:,:,:), field_T(:,:,:,:), field_LQ(:,:,:,:)
 792    real(8), pointer     :: field_Psfc(:,:,:,:), field_TG(:,:,:,:),Psfc_ptr(:,:,:)
 793    real(8), pointer     :: Press_T(:,:,:)
 794    real(8), pointer     :: Press_M(:,:,:)
 795    real(8), allocatable :: Psfc_ref(:,:)
 796    real(8), parameter   :: T_r = 280.0D0
 797    real(8), parameter   :: Psfc_r = 100000.0D0 ! unit Pa
 798    real(8), parameter   :: sigma = 0.3 ! weight factor for humidity
 799
 800    if (mmpi_myid == 0) write(*,*) 'multEnergyNorm: START'
 801    nullify(Press_T,Press_M)
 802
 803    ! the factors for TT, HU and Ps (for wind is 1)
 804    tfac = mpc_cp_dry_air_r8/T_r                                 ! temperature factor (c_p/T_r)
 805    qfac = sigma*mpc_heat_condens_water_r8**2/(mpc_cp_dry_air_r8*T_r)  ! humidity factor ( (l_p*l_p)/(c_p*T_r) )
 806    pfac = mpc_rgas_dry_air_r8*T_r/(Psfc_r**2)                   ! surface pressure factor (R*T_r/Psfc_r^2)
 807
 808    if (.not. gsv_isAllocated(statevector_inout)) then
 809      call utl_abort('multEnergyNorm: gridStateVector_inout not yet allocated')
 810    end if
 811
 812    nLev_M = gsv_getNumLev(statevector_inout,'MM')
 813    nLev_T = gsv_getNumLev(statevector_inout,'TH')
 814
 815    ! compute 3D log pressure fields
 816    call gsv_getField(statevector_ref,Psfc_ptr,'P0')
 817    allocate(Psfc_ref(statevector_inout%lonPerPEmax,statevector_inout%latPerPEmax))
 818    Psfc_ref(:,:) =  &
 819                  Psfc_ptr(statevector_inout%myLonBeg:statevector_inout%myLonEnd,  &
 820                  statevector_inout%myLatBeg:statevector_inout%myLatEnd, 1)
 821    call czp_fetch3DLevels(statevector_inout%vco, Psfc_ref, &
 822                           fldM_opt=Press_M, fldT_opt=Press_T)
 823    ! dlat * dlon
 824    scaleFactorConst = statevector_inout%hco%dlat*statevector_inout%hco%dlon
 825
 826    ! for wind components if to include in Norm calculation
 827    call gsv_getField(statevector_inout,field_UU,'UU')
 828    call gsv_getField(statevector_inout,field_VV,'VV')
 829    sumeu = 0.0D0
 830    sumev = 0.0D0
 831    sumScale = 0.0D0
 832    if (uvNorm) then
 833      do levIndex = 1, nLev_M
 834        do stepIndex = 1, statevector_inout%numStep
 835          do latIndex = statevector_inout%myLatBeg, statevector_inout%myLatEnd
 836            latIndex2 = latIndex - statevector_inout%myLatBeg + 1
 837            ! IF lat is out of the domain where we want to compute the NRJ norm, we put scaleFactorLat = 0.
 838            if (statevector_inout%hco%lat(latIndex) >= latMin .and. statevector_inout%hco%lat(latIndex) <= latMax) then
 839              scaleFactorLat = cos(statevector_inout%hco%lat(latIndex))
 840            else
 841              scaleFactorLat = 0.0D0
 842            end if
 843            do lonIndex = statevector_inout%myLonBeg, statevector_inout%myLonEnd
 844              lonIndex2 = lonIndex - statevector_inout%myLonBeg + 1
 845              ! Similarly, if lon is out of the domain where we want to compute the NRJ norm, we put scaleFactorLon = 0.
 846              if (statevector_inout%hco%lon(lonIndex) >= lonMin .and. statevector_inout%hco%lon(lonIndex) <= lonMax) then
 847                scaleFactorLon = 1.0D0
 848              else
 849                scaleFactorLon = 0.0D0
 850              end if
 851              ! do all thermo levels for which there is a momentum level above and below
 852              if ( levIndex == nLev_M) then
 853                scaleFactorLev = Press_M(lonIndex2, latIndex2, nLev_M)-Press_T(lonIndex2, latIndex2, nLev_T-1)
 854              else if ( Press_T(lonIndex2, latIndex2, levIndex) < 10000.0D0) then
 855                scaleFactorLev = 0.0D0
 856              else
 857                scaleFactorLev = Press_T(lonIndex2, latIndex2, levIndex+1) -  Press_T(lonIndex2, latIndex2, levIndex)
 858              end if
 859
 860              scaleFactor = scaleFactorConst * scaleFactorLat* scaleFactorLon * scaleFactorLev
 861              sumScale = sumScale + scaleFactor
 862
 863              sumeu = sumeu + &
 864                      0.5 * field_UU(lonIndex,latIndex,levIndex,stepIndex) * field_UU(lonIndex,latIndex,levIndex,stepIndex) * scaleFactor
 865              sumev = sumev + &
 866                      0.5 * field_VV(lonIndex,latIndex,levIndex,stepIndex) * field_VV(lonIndex,latIndex,levIndex,stepIndex) * scaleFactor
 867
 868              field_UU(lonIndex,latIndex,levIndex,stepIndex) = &
 869                   field_UU(lonIndex,latIndex,levIndex,stepIndex) * 0.5 * scaleFactor
 870              field_VV(lonIndex,latIndex,levIndex,stepIndex) = &
 871                   field_VV(lonIndex,latIndex,levIndex,stepIndex) * 0.5 * scaleFactor
 872            end do !lonIndex
 873          end do !latIndex
 874        end do ! stepIndex
 875      end do ! levIndex
 876      call mmpi_allreduce_sumreal8scalar(sumeu,'grid')
 877      call mmpi_allreduce_sumreal8scalar(sumev,'grid')
 878      call mmpi_allreduce_sumreal8scalar(sumScale,'grid')
 879
 880      sumeu = sumeu/sumScale
 881      sumev = sumev/sumScale
 882
 883      field_UU(:,:,:,:) = field_UU(:,:,:,:)/sumScale
 884      field_VV(:,:,:,:) = field_VV(:,:,:,:)/sumScale
 885    else
 886      field_UU(:,:,:,:) = field_UU(:,:,:,:)*0.0D0
 887      field_VV(:,:,:,:) = field_VV(:,:,:,:)*0.0D0
 888    end if ! if uvNorm
 889
 890    if (mmpi_myid == 0)  write(*,*) 'energy for UU=', sumeu
 891    if (mmpi_myid == 0)  write(*,*) 'energy for VV=', sumev
 892
 893    ! for Temperature
 894    call gsv_getField(statevector_inout,field_T,'TT')
 895    sumScale = 0.0D0
 896    sumet = 0.0D0
 897    if (ttNorm) then
 898      do levIndex = 1, nLev_T
 899        do stepIndex = 1, statevector_inout%numStep
 900          do latIndex = statevector_inout%myLatBeg, statevector_inout%myLatEnd
 901            latIndex2 = latIndex - statevector_inout%myLatBeg + 1
 902            ! IF lat is out of the domain where we want to compute the NRJ norm, we put scaleFactorLat = 0.
 903            if (statevector_inout%hco%lat(latIndex) >= latMin .and. statevector_inout%hco%lat(latIndex) <= latMax) then
 904              scaleFactorLat = cos(statevector_inout%hco%lat(latIndex))
 905            else
 906              scaleFactorLat = 0.0D0
 907            end if
 908            do lonIndex = statevector_inout%myLonBeg, statevector_inout%myLonEnd
 909              lonIndex2 = lonIndex - statevector_inout%myLonBeg + 1
 910              ! Similarly, if lon is out of the domain where we want to compute the NRJ norm, we put scaleFactorLon = 0.
 911              if (statevector_inout%hco%lon(lonIndex) >= lonMin .and. statevector_inout%hco%lon(lonIndex) <= lonMax) then
 912                scaleFactorLon = 1.0D0
 913              else
 914                scaleFactorLon = 0.0D0
 915              end if
 916              ! do all thermo levels for which there is a momentum level above and below
 917              if (levIndex == nLev_T) then  !surface
 918                scaleFactorLev =  Press_T(lonIndex2, latIndex2, nLev_T)-Press_T(lonIndex2, latIndex2, nLev_T-1)
 919              else if (levIndex == 1)  then  ! top
 920                scaleFactorLev = 0.0D0
 921              else if ( Press_M(lonIndex2, latIndex2, levIndex-1) < 10000.0D0) then
 922                scaleFactorLev = 0.0D0
 923              else
 924                scaleFactorLev = Press_M(lonIndex2, latIndex2, levIndex ) - Press_M(lonIndex2, latIndex2, levIndex-1)
 925              end if
 926              scaleFactor = scaleFactorConst * scaleFactorLat * scaleFactorLon * scaleFactorLev
 927              sumet = sumet + &
 928                   0.5 * tfac * field_T(lonIndex,latIndex,levIndex,stepIndex) * field_T(lonIndex,latIndex,levIndex,stepIndex) * scaleFactor
 929              sumScale = sumScale + scaleFactor
 930              field_T(lonIndex,latIndex,levIndex,stepIndex) = &
 931                           field_T(lonIndex,latIndex,levIndex,stepIndex) * 0.5 * tfac * scaleFactor
 932            end do
 933          end do
 934        end do ! stepIndex
 935      end do ! levIndex
 936      call mmpi_allreduce_sumreal8scalar(sumet,'grid')
 937      call mmpi_allreduce_sumreal8scalar(sumScale,'grid')
 938      sumet = sumet/sumScale
 939
 940      field_T(:,:,:,:) = field_T(:,:,:,:)/sumScale
 941    else
 942      field_T(:,:,:,:) = field_T(:,:,:,:)*0.0D0
 943    end if ! if ttNorm
 944
 945    if (mmpi_myid == 0)  write(*,*) 'energy for TT=', sumet
 946
 947    ! humidity (set to zero, for now)
 948    call gsv_getField(statevector_inout,field_LQ,'HU')
 949    sumScale = 0.0D0
 950    sumeq = 0.0D0
 951    if (huNorm) then
 952      do levIndex = 1, nLev_T
 953        do stepIndex = 1, statevector_inout%numStep
 954          do latIndex = statevector_inout%myLatBeg, statevector_inout%myLatEnd
 955            latIndex2 = latIndex - statevector_inout%myLatBeg + 1
 956            ! IF lat is out of the domain where we want to compute the NRJ norm, we put scaleFactorLat = 0.
 957            if (statevector_inout%hco%lat(latIndex) >= latMin .and. statevector_inout%hco%lat(latIndex) <= latMax) then
 958              scaleFactorLat = cos(statevector_inout%hco%lat(latIndex))
 959            else
 960              scaleFactorLat = 0.0D0
 961            end if
 962            do lonIndex = statevector_inout%myLonBeg, statevector_inout%myLonEnd
 963              lonIndex2 = lonIndex - statevector_inout%myLonBeg + 1
 964              ! Similarly, if lon is out of the domain where we want to compute the NRJ norm, we put scaleFactorLon = 0.
 965              if (statevector_inout%hco%lon(lonIndex) >= lonMin .and. statevector_inout%hco%lon(lonIndex) <= lonMax) then
 966                scaleFactorLon = 1.0D0
 967              else
 968                scaleFactorLon = 0.0D0
 969              end if
 970              ! do all thermo levels for which there is a momentum level above and below
 971              if ( levIndex == nLev_T) then !surface
 972                scaleFactorLev =  Press_T(lonIndex2, latIndex2, nLev_T) - Press_T(lonIndex2, latIndex2, nLev_T-1)
 973              else if (levIndex == 1)  then  ! top
 974                scaleFactorLev = 0.0D0
 975              else if ( Press_M(lonIndex2, latIndex2, levIndex-1) < 10000.0D0) then
 976                scaleFactorLev = 0.0D0
 977              else
 978                scaleFactorLev = Press_M(lonIndex2, latIndex2, levIndex ) - Press_M(lonIndex2, latIndex2, levIndex-1)
 979              end if
 980
 981              scaleFactor = scaleFactorConst * scaleFactorLat * scaleFactorLon * scaleFactorLev
 982              sumScale = sumScale + scaleFactor
 983

 995      call mmpi_allreduce_sumreal8scalar(sumeq,'grid')

 997      field_LQ(:,:,:,:) = field_LQ(:,:,:,:)/sumScale
 998    else
 999      field_LQ(:,:,:,:) = field_LQ(:,:,:,:)*0.0D0
1000    end if ! if huNorm
1001

1007    sumep = 0.0

1010        do latIndex = statevector_inout%myLatBeg, statevector_inout%myLatEnd
1011            ! IF lat is out of the domain where we want to compute the NRJ norm, we put scaleFactorLat = 0.

1022                scaleFactorLon = 0.0D0

1025              sumScale = sumScale + scaleFactor
1026              sumep = sumep + 0.5 * pfac * &
1027                  field_Psfc(lonIndex,latIndex,1,stepIndex) * field_Psfc(lonIndex,latIndex,1,stepIndex) * scaleFactor
1028              field_Psfc(lonIndex,latIndex,1,stepIndex) = &

1030          end do

1035      call mmpi_allreduce_sumreal8scalar(sumScale,'grid')

1037      field_Psfc(:,:,:,:) =  field_Psfc(:,:,:,:)/sumScale
1038    else
1039      field_Psfc(:,:,:,:) =  field_Psfc(:,:,:,:)*0.0D0
