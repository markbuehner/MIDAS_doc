 17module radvel_mod
 18  ! MODULE radvel_mod (prefix='rdv' category='5. Observation operators')
 19  !
 20  !:Purpose: Containing commonly used functions for the assimilation of Doppler velocity
 21  !
 22  !
 23  use codePrecision_mod
 24  use earthConstants_mod
 25  implicit none
 26  save
 27  private
 28
 29  ! public procedures
 30  public :: rdv_getlatlonHRfromRange, rdv_getRangefromH
 31
 32
 33contains 
 34
 35
 36  subroutine rdv_getlatlonHRfromRange(antennaLat, antennaLon, beamElevation, beamAzimuth, & !in
 37                                      radarAltitude, beamRange,                           & !in
 38                                      latSlant, lonSlant, beamHeight, beamDistance)         !out
 39    !
 40    !:Purpose: Computation of  lat-lon , height  of the trajectory
 41    !           along the radar beam from range of the radar beam
 42    !
 43    ! Angles are expressed in radians
 44    ! Height and distances in meters
 45    !
 46    implicit none
 47
 48    ! Argument 
 49    real(pre_obsReal), intent(in)  :: antennaLat
 50    real(pre_obsReal), intent(in)  :: antennaLon
 51    real(pre_obsReal), intent(in)  :: beamElevation
 52    real(pre_obsReal), intent(in)  :: beamAzimuth
 53    real(pre_obsReal), intent(in)  :: radarAltitude
 54    real(pre_obsReal), intent(in)  :: beamRange
 55    real(pre_obsReal), intent(out) :: LatSlant
 56    real(pre_obsReal), intent(out) :: lonSlant
 57    real(pre_obsReal), intent(out) :: beamHeight
 58    real(pre_obsReal), intent(out) :: beamDistance
 59    ! Local
 60    real(pre_obsReal)              :: Re 
 61    
 62    ! Radius of sphere of equal area from earthconstants_mod.f90
 63    ! ec_wgs_R2 = 6371007.1809
 64    ! effective radius of the earth
 65    Re = ec_wgs_R2 * (4./3.)
 66
 67    !compute height of radar observation
 68    beamHeight = sqrt(beamRange**2.+(Re+radarAltitude)**2.+2.*beamRange*(Re+radarAltitude)*sin(beamElevation))-(Re)
 69
 70    ! distance following surface of the earth from Doviak and Zrnic (2.28c)
 71    beamDistance = atan(beamRange*cos(beamElevation)/(beamRange*sin(beamElevation)+Re+radarAltitude))*Re
 72
 73    ! lat lon of the path along the radar beam  
 74    latSlant = asin( sin(antennaLat)*cos(beamDistance/ec_wgs_R2) + cos(antennaLat)*sin(beamDistance/ec_wgs_R2)*cos(beamAzimuth))
 75    lonSlant = antennaLon + atan2(sin(beamAzimuth)*sin(beamDistance/ec_wgs_R2)*cos(antennaLat), cos(beamDistance/ec_wgs_R2)-sin(antennaLat)*sin(latSlant))
 76
 77  end subroutine rdv_getlatlonHRfromRange
 78
 79  subroutine rdv_getRangefromH(beamHeight, radarAltitude, beamElevation, beamRange)
 80    !
 81    !:Purpose: Computation of range of the radar beam from height of the radar beam
 82    !
 83    implicit none
 84    
 85    ! Argument
 86    real(pre_obsReal) , intent(in)  :: beamHeight
 87    real(pre_obsReal) , intent(in)  :: radarAltitude
 88    real(pre_obsReal) , intent(in)  :: beamElevation
 89    real(pre_obsReal) , intent(out) :: beamRange 
 90    ! Local
 91    real(pre_obsReal)               :: a, b, c, Re
 92
 93    if ( radarAltitude > beamHeight ) then 
 94      !beamHeight is below radar antenna which may cause the equation below to return garbage
 95      !this happens in a few edge cases where its okay to return zero
 96      beamRange = 0.0
 97    else
 98
 99      ! Radius of sphere of equal area from earthconstants_mod.f90
100      ! ec_wgs_R2 = 6371007.1809
101      ! effective radius of the earth
102      Re = ec_wgs_R2*(4./3.)
103
104      a = 1.
105      b = 2.*(Re + radarAltitude)*sin(beamElevation)
106      c = -(Re + beamHeight)**2. + (Re + radarAltitude)**2.
107      ! range of radar beam from height and elevation of the radar beam 
108      beamRange  = (-b + sqrt( b**2. - 4.*a*c )) / (2.*a)
109
110    end if
111  
112     
113  end subroutine rdv_getRangefromH
114
115end module radvel_mod
