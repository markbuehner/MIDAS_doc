 17module regions_mod
 18  ! MODULE regions_mod (prefix='reg' category='7. Low-level data objects')
 19  !
 20  !:Purpose: This is a subset of the EnKF module that deals with 
 21  !           operations involving the generation of a set of regions. 
 22  !           Each region consists of a small number of observations 
 23  !           which is to be assimilated simultaneously in the 
 24  !           sequential algorithm. 
 25  !
 26  use mathPhysConstants_mod
 27  use midasMpi_mod
 28  implicit none
 29
 30  private
 31  public :: struct_reg
 32  public :: reg_getlatitude, reg_getblock, reg_locatestn, reg_init_struct
 33
 34  ! REG_MXOBS : maximum number of observations in a region
 35  ! REG_MXOBS must be at least as large as the number of observations
 36  ! in an individual radiosonde (121 currently)
 37  integer, parameter :: REG_MXOBS = 900
 38
 39  type :: struct_reg
 40   ! horizontal radius in km of the area in which observations are selected
 41   real(8) :: r0_km
 42   ! horizontal radius in km of the area in which an observation can have an impact
 43   real(8) :: r1_km
 44   ! the vertical extend in log(P) of the area in which an observation can have an 
 45   ! impact
 46   real(8) :: rz_logp 
 47   ! as r0km but in radians and adjusted considering that we want an integer
 48   ! number of latitude bands
 49   real(8) :: r0_rad 
 50   real(8) :: r1_rad ! equivalent of r1 in radians
 51   ! the horizontal length scale for the hadamard product (0.5 r1)
 52   real(8) :: scale_hor
 53   ! the vertical length scale for the hadamard product (in log (p))
 54   real(8) :: scale_ver
 55   ! number of latitude bands
 56   integer :: nlatband
 57   ! estimated maximum number of regions that could possibly be formed
 58   integer :: mxreg
 59  end type struct_reg
 60      
 61contains
 62
 63  subroutine reg_getblock(nlatband, r0, latmin, latmax, nlonblock)
 64    !
 65    !:Purpose: We have nlatband latitude bands (bounded by latitudes
 66    !           latmin and latmax). For each band determine how many 
 67    !           longitidunal blocks of at most r0*root(2) radians it 
 68    !           contains. Output is in nlonblock.
 69    !
 70    implicit none
 71
 72    integer :: i, nlatband
 73    real(8) :: bar, dpie, lat, latmin(nlatband), latmax(nlatband), r0
 74    integer :: nlonblock(nlatband)
 75
 76    dpie=MPC_PI_R8
 77      
 78    nlonblock(1)=1
 79    nlonblock(nlatband)=1
 80      
 81    do i = 2, nlatband-1 
 82     if (latmin(i)*latmax(i).gt.0) then
 83      lat = min(abs(latmin(i)),abs(latmax(i)))
 84     else
 85       ! at the equator
 86       lat = 0
 87     end if
 88     bar = (2**0.5)*r0/cos(lat)
 89     nlonblock(i) = int((2.0D0*dpie)/bar)+1
 90   end do 
 91
 92  end subroutine reg_getblock
 93

 95  subroutine reg_getlatitude(r0, nlatband, latmin, latcenter, latmax)
 96    !
 97    !:Purpose: the circle is covered with nlatband latitude bands. 
 98    !           the polar caps (with radius r0 radians) form the first and 
 99    !           last band. Intermediate bands are of width r0*(2**0.5)
100    !           For reach band we have the southern most latitude latmin,
101    !           the central latitude latcenter (at the poles for the extreme 
102    !           bands) and the northern most latitude latcenter
103    !      
104    implicit none
105    integer :: i, nlatband
106    real(8) :: dpie, r0, latmin(nlatband), latcenter(nlatband)
107    real(8) :: latmax(nlatband)
108
109    dpie = MPC_PI_R8
110
111    latmin(1) = -0.5D0*dpie
112    latcenter(1) = -0.5D0*dpie
113    latmax(1) = -0.5D0*dpie+r0
114 
115    if (mmpi_myid == 0) write(*,*) 'get latitude r0 and nlatband: ', r0, nlatband
116    if (mmpi_myid == 0) write(*,*) 'at ', 1, latmin(1), latcenter(1), latmax(1)
117
118    latmin(nlatband) = 0.5D0*dpie-r0
119    latcenter(nlatband) = 0.5D0*dpie
120    latmax(nlatband) = 0.5D0*dpie
121
122    if (mmpi_myid == 0) write(*,*) 'at ', nlatband, latmin(nlatband), latcenter(nlatband), latmax(nlatband)
123    do i = 2, nlatband-1
124      latmin(i) = latmax(i-1)
125      latmax(i) = latmin(i)+r0*(2**0.5)
126      latcenter(i) = latmin(i)+0.5D0*r0*(2**0.5)
127      if (mmpi_myid == 0) write(*,*) 'at ', i, latmin(i), latcenter(i), latmax(i)
128    end do
129 
130  end subroutine reg_getlatitude

133  subroutine getr0(r0km, r0, nlatband)
134    !
135    !:Purpose: The user specifies a radius of r0km (in km) within which 
136    !           stations can be taken together for simultaneous analysis
137    !           in a single batch. For simplicity of the search procedures
138    !           we - instead - use somewhat smaller squares with sides of 
139    !           at most r0km*root(2). We do have circles at the two poles.
140    !           Here we compute the radius r0 in radians such that we arrive 
141    !           exactly at nlatband latitude bands. The equator separates 
142    !           two latitude bands.
143    !         
144    implicit none
145    real(8), intent(in)  :: r0km
146    real(8), intent(out) :: r0
147    integer, intent(out) :: nlatband
148    real(8)              :: dpie, eps, difference, r0new, dminpole, nbandr
149    integer              :: nbandi
150
151    dpie = MPC_PI_R8
152      
153    ! figure out if r0km*root(2) is a divisor of 10000-r0km
154
155    eps = 0.01
156    ! we put a circle with radius r0km around each pole.
157    dminpole = 10000.-r0km
158    ! potential fractional number of latitude bands.
159    nbandr = dminpole/(r0km*(2**0.5))
160    nbandi = nint(nbandr)
161    difference = abs(dble(nbandi)-nbandr)
162    if (difference.gt.eps) then
163      ! increasing number of bands to nbandi
164      nbandi = nbandi+1
165    end if
166    ! decrease r0km to r0new
167    r0new = 10000./(dble(nbandi)*2**0.5 + 1.)
168    ! compute the corresponding r0 in radians       
169    r0 = (r0new/20000.)*dpie
170    ! two hemispheres with a pole
171    nlatband = 2*(nbandi+1)
172      
173  end subroutine getr0

175  subroutine reg_init_struct(lsc, r0_km, r1_km, rz_logp)
176    !
177    !:Purpose: store and derive parameters related to the localization
178    !           in a structure.
179    !
180    implicit none
181    type(struct_reg), intent(out) :: lsc
182    real(8), intent(in) :: r0_km
183    real(8), intent(in) :: r1_km
184    real(8), intent(in) :: rz_logp
185    real(8)  :: dpie
186    integer :: ione
187  
188    lsc%r0_km   = r0_km
189    lsc%r1_km   = r1_km
190    lsc%rz_logp = rz_logp
191
192    call getr0(lsc%r0_km, lsc%r0_rad, lsc%nlatband)
193    lsc%r1_rad = lsc%r1_km/6370.
194    lsc%scale_hor = 0.5*lsc%r1_rad
195    lsc%scale_ver = 0.5*lsc%rz_logp
196
197    dpie = MPC_PI_R8
198    lsc%mxreg = floor(dpie/(lsc%r1_rad+(2**0.5)*lsc%r0_rad)**2)
199
200    ione = 1
201    lsc%mxreg = max(lsc%mxreg, ione)
202
203  end subroutine reg_init_struct

206  subroutine reg_locatestn(r0, lat, lon, nlatband, nlonblock, &
207                           nblockoffset, iblock)
208    !
209    !:Purpose: locate the lat-lon block in which position (lat,lon) falls
210    !
211    !:Arguments:
212    !       r0:        radius of a region (in radians)
213    !       (lat,lon): latitude and longitude in radians
214    !       nlatband:  number of latitude bands
215    !       nlonblock: number of longitudinal blocks at each latitude band
216    !       nblockoffset: offset (the number of blocks south of each
217    !                  latitude band)
218    !       iblock:   block number
219    !
220    implicit none
221    integer :: iblock, ilat, ilon, nlatband
222    integer :: nlonblock(nlatband), nblockoffset(nlatband)
223    real(4) :: lat, lon
224    real(8) :: dpie, r0
225
226    dpie = MPC_PI_R8
227
228    if (lat.le.(-0.5D0*dpie+r0)) then
229      ilat = 1
230    else if (lat.gt.(0.5D0*dpie-r0)) then
231      ilat = nlatband
232    else
233      lat = lat+0.5D0*dpie-r0
234      ilat = 1+ceiling(lat/(r0*(2**0.5)))
235    end if
236    ilat = min(max(ilat,1),nlatband)
237    ilon = ceiling((lon/(2.0D0*dpie))*nlonblock(ilat))
238    ! added for points at the Greenwich meridian
239    ilon = min(max(ilon,1),nlonblock(ilat))
240    iblock = nblockoffset(ilat)+ilon
241
242  end subroutine reg_locatestn
243      
244end module regions_mod
