 67  !--------------------------------------------------------------------------
 68  ! inn_setupObs
 69  !--------------------------------------------------------------------------
 70  subroutine inn_setupobs(obsSpaceData, hco_anl, obsColumnMode, obsMpiStrategy, &
 71       innovationMode_in, obsClean_opt )
 72    !
 73    !:Purpose: To initialize the observation parameters and constants
 74    implicit none
 75
 76    ! Arguments:
 77    type(struct_obs)                        :: obsSpaceData
 78    type(struct_hco), pointer               :: hco_anl
 79    character(len=*)                        :: obsMpiStrategy
 80    character(len=*)                        :: obsColumnMode
 81    character(len=*), intent(in)            :: innovationMode_in
 82    logical,                       optional :: obsClean_opt
 83
 84    ! Locals:
 85    character(len=20) :: nameDimFile
 86    integer :: get_max_rss, ierr, fnom, fclos, unitDimFile, mxstn, mxobs
 87    logical :: obsDimFileExists
 88
 89    write(*,FMT=9000)
 909000 FORMAT(/,1x,' INN_SETUPOBS - Initialisation of observations',/,1x,3('- -----------'))
 91
 92    call utl_tmg_start(10,'--Observations')
 93
 94    !
 95    !- Setup de the mode
 96    !
 97    innovationMode = innovationMode_in
 98
 99    !
100    !- Specify the active observation-array columns
101    !
102    call obs_class_initialize(obsColumnMode)
103    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
104    !
105    !- Allocate memory for observation arrays
106    !
107    nameDimFile = './obs/cmadim'
108    inquire( file=trim(nameDimFile), exist=obsDimFileExists )
109    if ( obsDimFileExists ) then
110      unitDimFile = 0
111      ierr = fnom( unitDimFile, trim(nameDimFile), 'FTN+SEQ+R/O', 0 )
112      read(unitDimFile,*) mxstn
113      read(unitDimFile,*) mxobs
114      ierr=fclos(unitDimFile)  
115      call obs_initialize(obsSpaceData, numHeader_max=mxstn, numBody_max=mxobs, &
116                          mpi_local=obsf_filesSplit())
117    else
118      call obs_initialize(obsSpaceData, mpi_local=obsf_filesSplit())
119    end if
120    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
121
122    !
123    !- Set up the list of elements to be assimilated and flags for rejection
124    !
125    call filt_setup(innovationMode) ! IN
126
127    !
128    !- Initialize TOVS processing
129    !
130    call tvs_setup
131
132    !
133    !- Read the observations from files
134    !
135    call utl_tmg_start(11,'----ReadObsFiles')
136    call obsf_readFiles( obsSpaceData )
137    call utl_tmg_stop(11)
138
139    !
140    !- Initialize GPS processing
141    !
142    if (obs_famExist(obsSpaceData,'RO')) call gps_setupro
143    if (obs_famExist(obsSpaceData,'GP')) call gps_setupgb
144    
145    !
146    !- Filter out data from the obs data base
147    !
148    call filt_suprep(obsSpaceData)
149
150    ! Additional filtering for bias correction if requested 
151    call bcs_setup()
152    call bcs_filterObs(obsSpaceData)
153
154    if ( present(obsClean_opt) ) then
155      if ( obsClean_opt ) then
156        write(*,*) ''
157        write(*,*) 'inn_setupObs: !!WARNING!! Performing a cleanup of obsSpaceData'
158        write(*,*) '              This may make it impossible to update burp files'
159        write(*,*) ''
160        call obs_clean2(obsSpaceData)
161      end if
162    end if
163
164    !
165    !- Set (OBS_IPC and OBS_IPT) or OBS_IP columns according to the chosen strategy
166    !
167    write(*,*)
168    write(*,*) 'inn_setupObs - Using obsMpiStrategy = ', trim(obsMpiStrategy)
169    call setObsMpiStrategy(obsSpaceData,hco_anl,obsMpiStrategy)
170
171    !
172    !- Check if burp files already split
173    !
174    if ( obsf_filesSplit() ) then 
175      ! local observations files, so just do reallocation to reduce memory used
176      call obs_squeeze(obsSpaceData)
177      if ( obs_columnActive_IH(obsSpaceData,OBS_IPC) ) then
178        call obs_MpiRedistribute(obsSpaceData,OBS_IPC)
179      end if
180      write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
181    else
182      ! complete set of obs on each MPI process, only keep subset according to OBS_IP
183      call obs_reduceToMpiLocal(obsSpaceData)
184    end if
185
186    !
187    !- Initialization and memory allocation for TOVS processing
188    !
189    if (obs_famExist(obsSpaceData,'TO') .and. (trim(innovationMode) /= 'thinning')) then
190      call tvs_setupAlloc(obsSpaceData)
191      if (trim(innovationMode) == 'bgck' ) call irbg_setup()
192      ! Initialize non diagonal observation error matrices
193      if ( trim(innovationMode) == 'analysis' .or. trim(innovationMode) == 'FSO') call oer_setInterchanCorr()
194    end if
195
196    call utl_tmg_stop(10)
197
198  end subroutine inn_setupobs

203  subroutine inn_setupColumnsOnTrlLev( columnTrlOnTrlLev, obsSpaceData, hco_core, &
204                                       stateVectorUpdateHighRes, &
205                                       deallocInterpInfoNL_opt )
206    !
207    !:Purpose: To compute vertical (and potentially slanted) columns of trial data interpolated to obs location
208    !
209    implicit none
210
211    ! arguments
212    type(struct_columnData)    :: columnTrlOnTrlLev
213    type(struct_obs)           :: obsSpaceData
214    type(struct_hco), pointer  :: hco_core
215    type(struct_gsv)           :: stateVectorUpdateHighRes
216    logical         , optional :: deallocInterpInfoNL_opt
217
218    ! locals
219    type(struct_vco), pointer :: vco_trl => null()
220    integer                   :: ierr, nulnam, fnom, fclos
221    logical                   :: deallocInterpInfoNL
222    real(8), pointer          :: onecolumn(:)
223
224    character(len=20) :: timeInterpType_nl  ! 'NEAREST' or 'LINEAR'
225    integer           :: numObsBatches      ! number of batches for calling interp setup
226
227    NAMELIST /NAMINN/timeInterpType_nl, numObsBatches
228
229    write(*,*)
230    write(*,*) 'inn_setupColumnsOnTrlLev: START'
231
232    timeInterpType_nl = 'NEAREST'
233    numObsBatches     = 20
234
235    if (utl_isNamelistPresent('naminn','./flnml')) then
236      nulnam = 0
237      ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
238      if (ierr /= 0) call utl_abort('inn_setupColumnsOnTrlLev: Error opening file flnml')
239      read(nulnam,nml=naminn,iostat=ierr)
240      if (ierr /= 0) call utl_abort('inn_setupColumnsOnTrlLev: Error reading namelist')
241      if (mmpi_myid == 0) write(*,nml=naminn)
242      ierr = fclos(nulnam)
243    else
244      write(*,*)
245      write(*,*) 'inn_setupColumnsOnTrlLev: Namelist block NAMINN is missing in the namelist.'
246      write(*,*) '                            The default values will be taken.'
247      if (mmpi_myid == 0) write(*,nml=naminn)
248    end if
249
250    if ( present(deallocInterpInfoNL_opt) ) then
251      deallocInterpInfoNL = deallocInterpInfoNL_opt
252    else
253      deallocInterpInfoNL = .true.
254    end if
255
256    nullify(vco_trl)
257    vco_trl => gsv_getVco(stateVectorUpdateHighRes)
258
259    call col_setVco(columnTrlOnTrlLev,vco_trl)
260    call col_allocate(columnTrlOnTrlLev,obs_numHeader(obsSpaceData),mpiLocal_opt=.true.)
261
262    ! copy latitude from obsSpaceData
263    if ( obs_numHeader(obsSpaceData) > 0 ) then
264      call obs_extractObsRealHeaderColumn(columnTrlOnTrlLev%lat(:), obsSpaceData, OBS_LAT)
265    end if
266
267    call s2c_nl( stateVectorUpdateHighRes, obsSpaceData, columnTrlOnTrlLev, hco_core, &
268                 timeInterpType=timeInterpType_nl, &
269                 moveObsAtPole_opt=.true., numObsBatches_opt=numObsBatches, &
270                 dealloc_opt=deallocInterpInfoNL )
271
272    if ( col_getNumCol(columnTrlOnTrlLev) > 0 .and. col_varExist(columnTrlOnTrlLev,'Z_T ') ) then
273      write(*,*) 'inn_setupBackgroundColumns, statevector->Column 1:'
274      write(*,*) 'Z_T:'
275      onecolumn => col_getColumn(columnTrlOnTrlLev,1,'Z_T ')
276      write(*,*) onecolumn(:)
277      write(*,*) 'Z_M:'
278      onecolumn => col_getColumn(columnTrlOnTrlLev,1,'Z_M ')
279      write(*,*) onecolumn(:)
280
281      nullify(onecolumn)
282    end if
283    if ( col_getNumCol(columnTrlOnTrlLev) > 0 .and. col_varExist(columnTrlOnTrlLev,'P_T ') ) then
284      write(*,*) 'inn_setupBackgroundColumns, statevector->Column 1:'
285      write(*,*) 'P_T:'
286      onecolumn => col_getColumn(columnTrlOnTrlLev,1,'P_T ')
287      write(*,*) onecolumn(:)
288      write(*,*) 'P_M:'
289      onecolumn => col_getColumn(columnTrlOnTrlLev,1,'P_M ')
290      write(*,*) onecolumn(:)
291
292      nullify(onecolumn)
293    end if
294
295    write(*,*) 'inn_setupColumnsOnTrlLev: END'
296
297  end subroutine inn_setupColumnsOnTrlLev

299  !--------------------------------------------------------------------------
300  ! inn_setupColumnsOnAnlIncLev
301  !--------------------------------------------------------------------------
302  subroutine inn_setupColumnsOnAnlIncLev(columnTrlOnTrlLev,columnTrlOnAnlIncLev)
303    !
304    !:Purpose: To create trial data columns on analysis increment levels
305    implicit none
306
307    ! arguments
308    type(struct_columnData) :: columnTrlOnAnlIncLev, columnTrlOnTrlLev
309
310    ! locals
311    integer :: jvar, jlev, columnIndex
312    real(8), pointer :: columnTrlOnAnlIncLev_ptr(:), columnTrlOnTrlLev_ptr(:)
313
314    if (col_getNumCol(columnTrlOnAnlIncLev) == 0) return
315
316    call msg('inn_setupColumnsOnAnlIncLev','START',verb_opt=2)
317
318    !
319    !- Data copying from columnh to columnTrlOnAnlIncLev
320    !
321    
322    ! copy latitude
323    columnTrlOnAnlIncLev%lat(:) = columnTrlOnTrlLev%lat(:)
324
325    ! copy 2D surface variables
326    do jvar = 1, vnl_numvarmax2D
327      if ( .not. col_varExist(columnTrlOnAnlIncLev,vnl_varNameList2D(jvar)) ) cycle
328      do columnIndex = 1, col_getNumCol(columnTrlOnAnlIncLev)
329        columnTrlOnAnlIncLev_ptr  => col_getColumn(columnTrlOnAnlIncLev , columnIndex, vnl_varNameList2D(jvar))
330        columnTrlOnTrlLev_ptr => col_getColumn(columnTrlOnTrlLev, columnIndex, vnl_varNameList2D(jvar))
331        columnTrlOnAnlIncLev_ptr(:) = columnTrlOnTrlLev_ptr(:)
332      end do
333    end do
334
335    !
336    !- Vertical interpolation of 3D variables from trials levels to analysis increment levels
337    !
338    do jvar = 1, vnl_numvarmax3D
339
340      if ( .not. col_varExist(columnTrlOnAnlIncLev,vnl_varNameList3D(jvar)) ) cycle
341      
342      call int_vInterp_col( columnTrlOnTrlLev, columnTrlOnAnlIncLev, &
343                            vnl_varNameList3D(jvar), useColumnPressure_opt=.false.)
344
345      if ( vnl_varNameList3D(jvar) == 'HU  ') then
346        ! Imposing a minimum value for HU
347        do columnIndex = 1, col_getNumCol(columnTrlOnAnlIncLev)
348          columnTrlOnAnlIncLev_ptr => col_getColumn(columnTrlOnAnlIncLev,columnIndex,'HU')
349          do jlev = 1, col_getNumLev(columnTrlOnAnlIncLev,'TH')
350            columnTrlOnAnlIncLev_ptr(jlev) = max(columnTrlOnAnlIncLev_ptr(jlev),col_rhumin)
351          end do
352        end do
353
354      ! Imposing minimum/maximum value for ALL cloud variables at all levels
355      else if (vnl_isCloudVar(vnl_varNameList3D(jvar))) then
356        do columnIndex = 1, col_getNumCol(columnTrlOnAnlIncLev)
357          columnTrlOnAnlIncLev_ptr => col_getColumn(columnTrlOnAnlIncLev,columnIndex,vnl_varNameList3D(jvar))
358          do jlev = 1, col_getNumLev(columnTrlOnAnlIncLev,'TH')
359            columnTrlOnAnlIncLev_ptr(jlev) = max(columnTrlOnAnlIncLev_ptr(jlev), &
360                                                 qlim_getMinValueCloud(vnl_varNameList3D(jvar)))
361            columnTrlOnAnlIncLev_ptr(jlev) = min(columnTrlOnAnlIncLev_ptr(jlev), &
362                                                 qlim_getMaxValueCloud(vnl_varNameList3D(jvar)))
363          end do
364        end do
365
366      else if (trim(vnl_varKindFromVarname(vnl_varNameList3D(jvar))) == 'CH') then
367        ! Imposing boundary values for CH kind variables. This is to prevent
368        ! undesired values usually from vertical extrapolation.
369        do columnIndex = 1, col_getNumCol(columnTrlOnAnlIncLev)
370          columnTrlOnAnlIncLev_ptr => col_getColumn(columnTrlOnAnlIncLev,columnIndex,trim(vnl_varNameList3D(jvar)))
371          if ( col_minValVarKindCH(vnl_varListIndex(vnl_varNameList3D(jvar))) > 1.01*MPC_missingValue_R8 ) then
372            do jlev=1,col_getNumLev(columnTrlOnAnlIncLev,'TH')
373              columnTrlOnAnlIncLev_ptr(jlev) = max(columnTrlOnAnlIncLev_ptr(jlev),col_minValVarKindCH(vnl_varListIndex(vnl_varNameList3D(jvar))))
374            end do
375          end if
376        end do
377      end if
378
379    end do
380
381    ! Print pressure on thermo levels for the first column
382    if (col_varExist(columnTrlOnAnlIncLev,'P_T')) then
383      columnTrlOnAnlIncLev_ptr => col_getColumn(columnTrlOnAnlIncLev,1,'P_T')
384      call msg('inn_setupColumnsOnAnlIncLev', ' after vintprof:'&
385           //new_line('')//'P_T (columnTrlOnAnlIncLev(1)) = '//str(columnTrlOnAnlIncLev_ptr(:)), mpiAll_opt=.false.)
386    end if
387
388    !
389    !- Height adjustments
390    !
391
392    ! Set surface height
393    do columnIndex = 1, col_getNumCol(columnTrlOnAnlIncLev)
394      columnTrlOnAnlIncLev%heightSfc(columnIndex) = columnTrlOnTrlLev%heightSfc(columnIndex)
395    end do
396
397    ! Remove the height offset for the diagnostic levels for backward compatibility only
398    if ( col_varExist(columnTrlOnAnlIncLev,'Z_T') .and. .not.columnTrlOnAnlIncLev%addHeightSfcOffset ) then 
399      do columnIndex = 1, col_getNumCol(columnTrlOnAnlIncLev)
400        columnTrlOnAnlIncLev_ptr => col_getColumn(columnTrlOnAnlIncLev,columnIndex,'Z_T')
401        columnTrlOnAnlIncLev_ptr(col_getNumLev(columnTrlOnAnlIncLev,'TH')) = columnTrlOnAnlIncLev%heightSfc(columnIndex)
402      end do
403    end if
404    if ( col_varExist(columnTrlOnAnlIncLev,'Z_M') .and. .not.columnTrlOnAnlIncLev%addHeightSfcOffset ) then
405      do columnIndex = 1, col_getNumCol(columnTrlOnAnlIncLev)
406        columnTrlOnAnlIncLev_ptr => col_getColumn(columnTrlOnAnlIncLev,columnIndex,'Z_M')
407        columnTrlOnAnlIncLev_ptr(col_getNumLev(columnTrlOnAnlIncLev,'MM')) = columnTrlOnAnlIncLev%heightSfc(columnIndex)
408      end do
409    end if
410
411    ! Print height info of the first original and interpolated columns
412    if ( col_getNumLev(columnTrlOnAnlIncLev,'TH') > 0 .and. col_varExist(columnTrlOnAnlIncLev,'Z_T') ) then
413      columnTrlOnTrlLev_ptr => col_getColumn(columnTrlOnTrlLev,1,'Z_T')
414      columnTrlOnAnlIncLev_ptr => col_getColumn(columnTrlOnAnlIncLev,1,'Z_T')
415      call msg('inn_setupColumnsOnAnlIncLev','vIntProf output:' &
416            //new_line('')//'Z_T (columnTrlOnTrlLev) = '//str(columnTrlOnTrlLev_ptr(:)) &
417            //new_line('')//'Z_T (columnTrlOnAnlIncLev) = '//str(columnTrlOnAnlIncLev_ptr(:)))
418    end if
419    
420    if ( col_getNumLev(columnTrlOnAnlIncLev,'MM') > 0 .and. col_varExist(columnTrlOnAnlIncLev,'Z_M') ) then
421      columnTrlOnTrlLev_ptr => col_getColumn(columnTrlOnTrlLev,1,'Z_M')
422      columnTrlOnAnlIncLev_ptr => col_getColumn(columnTrlOnAnlIncLev,1,'Z_M')
423      call msg('inn_setupColumnsOnAnlIncLev','vIntProf output:'&
424            //new_line('')//'Z_M (columnTrlOnTrlLev) = '//str(columnTrlOnTrlLev_ptr(:)) &
425            //new_line('')//'Z_M (columnTrlOnAnlIncLev) = '//str(columnTrlOnAnlIncLev_ptr(:)))
426    end if
427
428    call msg('inn_setupColumnsOnAnlIncLev','HeightSfc:'//str(columnTrlOnAnlIncLev%heightSfc(1)))
429
430    call msg('inn_setupColumnsOnAnlIncLev','END',verb_opt=2)
431
432  end subroutine inn_setupColumnsOnAnlIncLev

434  !--------------------------------------------------------------------------
435  ! inn_computeInnovation
436  !--------------------------------------------------------------------------
437  subroutine inn_computeInnovation( columnTrlOnTrlLev, obsSpaceData, filterObsAndInitOer_opt, &
438                                    applyVarqcOnNlJo_opt, destObsColumn_opt, &
439                                    beSilent_opt, callFiltTopo_opt, callSetErrGpsgb_opt, &
440                                    analysisMode_opt )
441    !
442    !:Purpose: To initialize observation innovations using the nonlinear H
443    implicit none
444
445    ! Arguments:
446    type(struct_columnData) :: columnTrlOnTrlLev
447    type(struct_obs)        :: obsSpaceData
448    logical, optional       :: filterObsAndInitOer_opt
449    logical, optional       :: applyVarqcOnNlJo_opt
450    integer, optional       :: destObsColumn_opt ! column where result stored, default is OBS_OMP
451    logical, optional       :: beSilent_opt
452    logical, optional       :: callFiltTopo_opt ! whether to make call to FiltTopo
453    logical, optional       :: callSetErrGpsgb_opt ! whether to make call to oer_SETERRGPSGB
454    logical, optional       :: analysisMode_opt ! analysisMode argument for oer_SETERRGPSGB and oop_gpsgb_nl
455    
456    ! Locals:
457    real(8) :: Jo
458    integer :: destObsColumn, get_max_rss
459    logical :: applyVarqcOnNlJo, filterObsAndInitOer, beSilent, callFiltTopo, callSetErrGpsgb, analysisMode 
460    logical, save :: lgpdata = .false.
461
462    call utl_tmg_start(10,'--Observations')
463
464    if ( present(beSilent_opt) ) then
465      beSilent = beSilent_opt
466    else
467      beSilent = .false.
468    end if
469
470    if ( .not. beSilent ) then
471      write(*,*)
472      write(*,*) '--Starting subroutine inn_computeInnovation--'
473      write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
474    end if
475
476    if ( present(filterObsAndInitOer_opt) ) then
477      filterObsAndInitOer = filterObsAndInitOer_opt
478    else
479      filterObsAndInitOer = .true.
480    end if
481
482    if ( present(applyVarqcOnNlJo_opt) ) then
483      applyVarqcOnNlJo = applyVarqcOnNlJo_opt
484    else
485      applyVarqcOnNlJo = .false.
486    end if
487
488    if ( present(destObsColumn_opt) ) then
489      destObsColumn = destObsColumn_opt
490    else
491      destObsColumn = obs_omp
492    end if
493
494    if ( present(callFiltTopo_opt) ) then
495      callFiltTopo = callFiltTopo_opt
496    else
497      callFiltTopo = .true.
498    end if   
499
500    if ( present(callSetErrGpsgb_opt ) ) then
501      callSetErrGpsgb = callSetErrGpsgb_opt
502    else
503      callSetErrGpsgb = .true.   
504    end if    
505
506    if ( present(analysisMode_opt ) ) then
507      analysisMode = analysisMode_opt
508    else
509      analysisMode = .true.
510    end if
511
512    if ( .not.beSilent ) write(*,*) 'oti_timeBinning: Before filtering done in inn_computeInnovation'
513    if ( .not.beSilent ) call oti_timeBinning(obsSpaceData,tim_nstepobs)
514
515    ! Reject observed elements too far below the surface. Pressure values
516    ! for elements slightly below the surface are replaced by the surface
517    ! pressure values of the trial field.
518    !
519    ! GB-GPS (met and ZTD) observations are processed in s/r filt_topoSFC (in obsFilter_mod.ftn90)
520    !
521    if ( filterObsAndInitOer .and. callFiltTopo ) then
522      call filt_topo(columnTrlOnTrlLev,obsSpaceData,beSilent)
523    else
524      if ( mmpi_myid == 0 .and. .not. beSilent ) write(*,*) 'inn_computeInnovation: skip filt_topo'
525    end if
526   
527    ! Remove surface station wind observations
528    if ( trim(innovationMode) == 'analysis' .or. trim(innovationMode) == 'FSO' ) then
529      if ( filterObsAndInitOer ) then
530        call filt_surfaceWind(obsSpaceData, beSilent)
531      else
532        if ( mmpi_myid == 0 .and. .not. beSilent ) write(*,*) 'inn_computeInnovation: skip filt_surfaceWind'
533      end if
534    end if
535
536    ! Find interpolation layer in model profiles
537    if ( col_getNumLev(columnTrlOnTrlLev,'MM') > 1 ) call oop_vobslyrs(columnTrlOnTrlLev, obsSpaceData, beSilent)
538
539    !
540    !- Calculate the innovations [Y - H(Xb)] and place the result in obsSpaceData in destObsColumn column
541    !
542    call utl_tmg_start(17,'----ObsOper_NL')
543    
544    ! Radiosondes
545    call oop_ppp_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, 'UA', destObsColumn)
546
547    ! Aircrafts
548    call oop_ppp_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, 'AI', destObsColumn)
549
550    ! SatWinds
551    if ( filterObsAndInitOer ) then
552      call oer_sw(columnTrlOnTrlLev,obsSpaceData)
553    else
554      if ( mmpi_myid == 0 .and. .not. beSilent ) write(*,*) 'inn_computeInnovation: skip oer_sw'
555    end if
556    
557    call oop_ppp_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, 'SW', destObsColumn)
558
559    ! Surface (SF, UA, SC, GP and RA families)
560    call oop_sfc_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, 'SF', destObsColumn)
561    call oop_sfc_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, 'UA', destObsColumn)
562    call oop_sfc_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, 'SC', destObsColumn)
563    call oop_sfc_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, 'GP', destObsColumn)
564    call oop_sfc_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, 'RA', destObsColumn)
565
566    ! RADAR Doppler velocity
567    if ( filterObsAndInitOer ) then
568     ! Filter Radar for Doppler velocity
569      call filt_radvel(columnTrlOnTrlLev, obsSpaceData, beSilent)
570    else
571      if ( mmpi_myid == 0 .and. .not. beSilent ) write(*,*) 'inn_computeInnovation: skip filt_radvel'
572    end if
573
574    call oop_raDvel_nl(columnTrlOnTrlLev,obsSpaceData, beSilent, 'RA', destObsColumn)  
575
576    ! Sea surface temperature
577    call oop_sst_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, 'TM', destObsColumn)
578
579    ! Sea ice concentration
580    if ( filterObsAndInitOer ) then
581      call filt_iceConcentration(obsSpaceData, beSilent)
582      call filt_backScatAnisIce(obsSpaceData, beSilent)
583      call oer_setErrBackScatAnisIce(obsSpaceData, beSilent, columnTrlOnTrlLev_opt=columnTrlOnTrlLev)
584    else
585      if ( mmpi_myid == 0 .and. .not. beSilent ) write(*,*) 'inn_computeInnovation: skip filt_iceConcentration, filt_backScatAnisIce, and oer_setErrBackScatAnisIce'
586    end if
587
588    call oop_ice_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, 'GL', destObsColumn)
589
590    ! Hydrology
591    call oop_hydro_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, 'HY', destObsColumn)
592
593    ! TOVS / Radiances
594    if (trim(innovationMode) == 'bgck'  ) then
595      call oop_tovs_nl(columnTrlOnTrlLev, obsSpaceData, tim_getDatestamp(),  &
596                       beSilent, bgckMode_opt=.true., destObs_opt=destObsColumn)
597    else
598      call oop_tovs_nl(columnTrlOnTrlLev, obsSpaceData, tim_getDatestamp(),  &
599                       beSilent, bgckMode_opt=.false., destObs_opt=destObsColumn)
600    end if
601
602    ! Profilers
603    call oop_zzz_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, 'PR', destObsColumn)
604
605    ! Aladin winds
606    call oop_zzz_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, 'AL', destObsColumn)
607
608    ! GPS radio occultation
609    if (obs_famExist(obsSpaceData,'RO', localMPI_opt = .true. )) then
610      if ( filterObsAndInitOer ) then
611        call filt_gpsro(columnTrlOnTrlLev, obsSpaceData, beSilent)
612        call oer_SETERRGPSRO(columnTrlOnTrlLev, obsSpaceData, beSilent)
613      else
614        if ( mmpi_myid == 0 .and. .not. beSilent ) write(*,*) 'inn_computeInnovation: skip filt_gpsro, and oer_SETERRGPSRO'
615      end if
616      call oop_gpsro_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, destObsColumn)
617    end if
618
619    ! Chemical constituents
620    call oop_chm_nl(columnTrlOnTrlLev, obsSpaceData, destObsColumn)
621
622    ! GPS ground-based zenith delay
623    if (obs_famExist(obsSpaceData,'GP', localMPI_opt = .true. )) then
624      if ( CallSetErrGpsgb ) then
625        call oer_SETERRGPSGB(columnTrlOnTrlLev, obsSpaceData, beSilent, lgpdata, analysisMode)
626      else
627        if ( mmpi_myid == 0 .and. .not. beSilent ) write(*,*) 'inn_computeInnovation: skip oer_SETERRGPSGB'
628      end if
629      if (lgpdata) call oop_gpsgb_nl(columnTrlOnTrlLev, obsSpaceData, beSilent, &
630                                     destObsColumn, analysisMode_opt=analysisMode)   
631    end if
632
633    call utl_tmg_stop(17)
634
635    ! Save as OBS_WORK : R**-1/2 (d)
636    call rmat_RsqrtInverseAllObs(obsSpaceData,OBS_WORK,destObsColumn)
637
638    ! Store J-obs in OBS_JOBS : 1/2 * R**-1 (d)**2
639    call cfn_calcJo(obsSpaceData)
640
641    ! applying varqc, if asked for
642    if ( applyVarqcOnNlJo ) call vqc_NlTl(obsSpaceData)
643
644    ! Compute Jo components and print
645    call cfn_sumJo(obsSpaceData, Jo, beSilent_opt=beSilent)
646    if ( mmpi_myid == 0 .and. .not. beSilent ) write(*,'(a15,f25.17)') 'Total Jo = ',Jo
647
648    if ( .not.beSilent ) write(*,*) 'oti_timeBinning: After filtering done in inn_computeInnovation'
649    if ( .not.beSilent ) call oti_timeBinning(obsSpaceData,tim_nstepobs)
650
651    if ( .not. beSilent ) then
652      write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
653      write(*,*) '--Done subroutine inn_computeInnovation--'
654    end if
655
656    call utl_tmg_stop(10)
657
658  end subroutine inn_computeInnovation

661  ! setObsMpiStrategy
662  !--------------------------------------------------------------------------
663  subroutine setObsMpiStrategy(obsSpaceData, hco_anl, mpiStrategy)
664    !
665    !:Purpose: To distribute header indices following the chosen strategy,
666    !          current options: "LIKESPLITFILES", "ROUNDROBIN", "LATLONTILES".
667    implicit none
668
669    ! Arguments:
670    type(struct_obs),          intent(inout) :: obsSpaceData
671    type(struct_hco), pointer, intent(in)    :: hco_anl
672    character(len=*),          intent(in)    :: mpiStrategy
673
674    ! Locals:
675    real(8) :: lat_r8, lon_r8
676    real    :: lat_r4, lon_r4
677    real    :: xpos_r4, ypos_r4
678
679    integer :: numHeaderFile, headerIndex, latIndex, lonIndex, ierr
680    integer :: IP, IP_x, IP_y
681    integer :: gdxyfll
682    !
683    !- Determine obs_ipc (column) and obs_ipt (tile) according to distribution strategy
684    !
685    write(*,*)
686    numHeaderFile = obs_numheader(obsSpaceData)
687    write(*,*) 'setObsMpiStrategy: numHeader after reading files = ',numHeaderFile
688    write(*,*) 'setObsMpiStrategy: strategy = ',trim(mpiStrategy)
689
690    ! make sure old PE index is not used (set to -1)
691    do headerIndex = 1, numHeaderFile
692       call obs_headSet_i(obsSpaceData,OBS_IP,headerIndex,-1)
693    end do
694
695    ! set PE index for the file (where reading and updating was/will be done)
696    if ( obs_columnActive_IH(obsSpaceData,OBS_IPF) ) then
697      do headerIndex = 1, numHeaderFile
698        call obs_headSet_i(obsSpaceData,OBS_IPF,headerIndex,mmpi_myid)
699      end do
700    end if
701
702    select case (trim(mpiStrategy))
703    case ('LIKESPLITFILES')
704       !- Keep distribution exactly as it is in the split files:
705       do headerIndex = 1, numHeaderFile
706          if ( obs_columnActive_IH(obsSpaceData,OBS_IPC) ) &
707            call obs_headSet_i(obsSpaceData,OBS_IPC,headerIndex, mmpi_myid)
708          if ( obs_columnActive_IH(obsSpaceData,OBS_IPT) ) &
709            call obs_headSet_i(obsSpaceData,OBS_IPT,headerIndex, mmpi_myid)
710          if ( obs_columnActive_IH(obsSpaceData,OBS_IP) ) &
711            call obs_headSet_i(obsSpaceData,OBS_IP,headerIndex, mmpi_myid)
712       end do
713    case ('ROUNDROBIN')
714       !- Distribute by a round-robin strategy for both obs_ipc and obs_ipt:
715       !  (Only use if files already not split by round robin)
716       do headerIndex = 1, numHeaderFile
717          IP = mod((headerIndex-1),mmpi_nprocs)
718          if ( obs_columnActive_IH(obsSpaceData,OBS_IPC) ) &
719            call obs_headSet_i(obsSpaceData,OBS_IPC,headerIndex, IP)
720          if ( obs_columnActive_IH(obsSpaceData,OBS_IPT) ) &
721            call obs_headSet_i(obsSpaceData,OBS_IPT,headerIndex, IP)
722          if ( obs_columnActive_IH(obsSpaceData,OBS_IP) ) &
723            call obs_headSet_i(obsSpaceData,OBS_IP ,headerIndex, IP)
724       end do
725    case ('LATLONTILES')
726       !- Distribute by latitude/longitude tiles for both obs_ipc and obs_ipt:
727       do headerIndex = 1, numHeaderFile
728          ! compute grid index for each observation header
729          lat_r8 = obs_headElem_r(obsSpaceData,OBS_LAT,headerIndex)
730          lon_r8 = obs_headElem_r(obsSpaceData,OBS_LON,headerIndex)
731          lat_r4 = real(lat_r8) * MPC_DEGREES_PER_RADIAN_R4
732          lon_r4 = real(lon_r8) * MPC_DEGREES_PER_RADIAN_R4
733          ierr = gdxyfll( hco_anl%EZscintID,   & ! IN 
734               xpos_r4, ypos_r4,    & ! OUT
735               lat_r4, lon_r4, 1 )    ! IN
736
737          ! compute correponding mpi task id for each observation
738          latIndex = floor(ypos_r4)
739          lonIndex = floor(xpos_r4)
740          IP_y = mmpi_myidYfromLat(latIndex, hco_anl%nj)
741          IP_x = mmpi_myidXfromLon(lonIndex, hco_anl%ni)
742          IP = IP_x + IP_y*mmpi_npex
743
744          call obs_headSet_i(obsSpaceData,OBS_IPC,headerIndex, IP)
745          call obs_headSet_i(obsSpaceData,OBS_IPT,headerIndex, IP)
746       end do
747    case ('LATLONTILESBALANCED')
748       !- Distribute by latitude/longitude tiles, but with simple & cheap balancing for obs_ipc (1 send or recv):
749       call utl_abort('setObsMpiStrategy: Sorry, LATLONTILESBALANCED no longer available')
750    case default
751       write(*,*)
752       write(*,*) 'ERROR unknown mpiStrategy: ', trim(mpiStrategy)
753       call utl_abort('setObsMpiStrategy')
754    end select
755
756  end subroutine setObsMpiStrategy

759  ! inn_perturbObs
760  !--------------------------------------------------------------------------
761  subroutine inn_perturbObs(obsSpaceData,numAnalyses,indexAnalysis, &
762                            indexBatch,obs_column_index_src,obs_column_index_dest)
763    !
764    !:Purpose: To perturb the innovation vector to simulate effect of
765    !          observation uncertainty
766    !
767    !.. WARNING:: perturbations are not the same when MPI topology changes!!!
768    !
769    implicit none
770
771    ! Arguments:
772    type(struct_obs) :: obsSpaceData
773    integer :: numAnalyses,indexAnalysis,indexBatch
774    integer :: obs_column_index_src,obs_column_index_dest
775
776    ! Locals:
777    integer :: numPerturbations
778    integer :: nrandseed,iseed,indexAnalysis2,indexBody,indexFamily
779    integer, parameter :: numFamily=10
780    real*8  :: zmean,originalOmp
781    real*8  :: scaleFactor(numFamily)
782    character(len=2) :: familyList(numFamily)
783    real*8, save, pointer :: obsPerturbations(:,:) => NULL()
784    logical, save :: firstTime = .true.
785
786    familyList(1)='UA' ; scaleFactor(1)=1.00d0
787    familyList(2)='AI' ; scaleFactor(2)=1.00d0
788    familyList(3)='SF' ; scaleFactor(3)=1.00d0
789    familyList(4)='TO' ; scaleFactor(4)=1.00d0
790    familyList(5)='SW' ; scaleFactor(5)=1.00d0
791    familyList(6)='SC' ; scaleFactor(6)=1.00d0
792    familyList(7)='PR' ; scaleFactor(7)=1.00d0
793    familyList(8)='RO' ; scaleFactor(8)=1.00d0
794    familyList(9)='GP' ; scaleFactor(9)=1.00d0
795    familyList(10)='RA'; scaleFactor(10)=1.00d0    
796
797    numPerturbations = numAnalyses
798
799    if (firstTime) then
800
801       if (.not.associated(obsPerturbations)) then
802          write(*,*) 'perturbObs: allocating space for all perturbations'
803          allocate(obsPerturbations(obs_numBody(obsSpaceData),numPerturbations))
804       end if
805
806       write(*,*) 'perturbObs: computing random numbers'
807
808       ! compute random perturbations
809       do indexAnalysis2 = 1,numPerturbations
810          nrandseed   = indexAnalysis2 + (indexBatch-1)*numAnalyses
811          iseed     = ABS(nrandseed)
812          write(*,*) 'perturbobs: indexAnalysis, iseed=',indexAnalysis2, iseed
813          call rng_setup(iseed) ! JFC: should be called only once, no???
814          do indexBody = 1,obs_numBody(obsSpaceData)
815             obsPerturbations(indexBody,indexAnalysis2)=rng_gaussian()*obs_bodyElem_r(obsSpaceData,OBS_OER,indexBody)
816          end do
817       end do
818
819       ! apply scale factor
820       do indexFamily = 1,numFamily
821          do indexBody = 1,obs_numBody(obsSpaceData)      
822             if (obs_getFamily(obsSpaceData,bodyIndex=indexBody).eq.familyList(indexFamily)) then
823                do indexAnalysis2 = 1,numPerturbations
824                   obsPerturbations(indexBody,indexAnalysis2)=obsPerturbations(indexBody,indexAnalysis2)*scaleFactor(indexFamily)
825                end do
826             end if
827          end do
828       end do
829
830       ! remove ensemble mean
831       do indexBody = 1,obs_numBody(obsSpaceData)      
832          zmean=0.0d0
833          do indexAnalysis2 = 1,numPerturbations
834             zmean=zmean+obsPerturbations(indexBody,indexAnalysis2)
835          end do
836          zmean=zmean/numPerturbations
837          do indexAnalysis2 = 1,numPerturbations
838             obsPerturbations(indexBody,indexAnalysis2)=obsPerturbations(indexBody,indexAnalysis2)-zmean
839          end do
840       end do
841
842       firstTime=.false.
843
844    end if
845
846    ! apply perturbation for current analysis
847    write(*,*) 'perturbObs: applying perturbation for analysis number: ',indexAnalysis
848    do indexBody = 1,obs_numBody(obsSpaceData)      
849       if (obs_bodyElem_i(obsSpaceData,OBS_ASS,indexBody) == obs_assimilated) then
850          originalOmp = obs_bodyElem_r(obsSpaceData,obs_column_index_src,indexBody)
851          call obs_bodySet_r(obsSpaceData,obs_column_index_dest,indexBody,originalOmp+obsPerturbations(indexBody,indexAnalysis))
852       end if
853    end do
854
855  end subroutine inn_perturbObs
