 42  !----------------------------------------------------------------------------------------
 43  ! obgd_computeSSTrial
 44  !----------------------------------------------------------------------------------------
 45  subroutine obgd_computeSSTrial(hco, vco, trialDateStamp, analysisDateStamp, &
 46                                 nmonthsClim, datestampClim, alphaClim, etiket)
 47    !
 48    !: Purpose: to compute SST background   
 49    !           xb(t) = (xa(t-1) - xclim(t-1))*alpha + xclim(t)         
 50    
 51    implicit none
 52
 53    ! Arguments:
 54    type(struct_hco) , intent(in), pointer :: hco               ! horizontal grid
 55    type(struct_vco) , intent(in), pointer :: vco               ! vertical grid
 56    integer          , intent(in)          :: trialDateStamp    ! trial datestamp
 57    integer          , intent(in)          :: analysisDateStamp ! datestamp for last analysis 
 58    integer          , intent(in)          :: nmonthsClim       ! number of climatological fields (= 12)
 59    integer          , intent(in)          :: datestampClim(:)  ! datestamps of input climatology fields
 60    real(4)          , intent(in)          :: alphaClim         ! scalling factor to relax towards climatology
 61    character(len=10), intent(in)          :: etiket            ! etiket from namelist and for trial
 62    
 63    ! Locals:
 64    type(struct_gsv) :: stateVector
 65    real(4), pointer :: stateVector_ptr(:, :, :)
 66    integer          :: lonIndex, latIndex
 67    real(8)          :: climatology_m1(hco % ni, hco % nj), climatology(hco % ni, hco % nj)
 68    
 69    write(*,*) 'obgd_computeSSTrial: starting...'  
 70      
 71    ! get SST analysis
 72    call gsv_allocate(stateVector, 1, hco, vco, dataKind_opt = 4, datestamp_opt = analysisDateStamp, &
 73                      mpi_local_opt = .true., varNames_opt = (/'TM'/))
 74    call gio_readFromFile(stateVector, './analysis', ' ','A', &
 75                          unitConversion_opt=.false., containsFullField_opt=.true.)
 76    call gsv_getField(stateVector, stateVector_ptr)
 77    
 78    call obgd_getClimatology(analysisDateStamp, hco, vco, nmonthsClim, datestampClim, climatology_m1)
 79    call obgd_getClimatology(trialDateStamp   , hco, vco, nmonthsClim, datestampClim, climatology)
 80    
 81    ! compute background state
 82    ! xb(t) = (xa(t-1) - xclim(t-1))*alpha + xclim(t)
 83    do lonIndex = stateVector%myLonBeg, stateVector%myLonEnd 
 84      do latIndex = stateVector%myLatBeg, stateVector%myLatEnd
 85        stateVector_ptr(lonIndex, latIndex, 1) = (stateVector_ptr(lonIndex, latIndex, 1) - &
 86                                                 climatology_m1(lonIndex, latIndex)) * &
 87                                                 alphaClim + climatology(lonIndex, latIndex)
 88      end do
 89    end do      	 
 90  
 91    ! modify dateStamp (from analysis) with trial dateStamp
 92    call gsv_modifyDate( stateVector, trialDateStamp, modifyDateOrigin_opt = .true. )
 93    
 94    ! save trial field
 95    call gio_writeToFile(stateVector, './trial', etiket, typvar_opt = 'P@')
 96
 97    call gsv_deallocate(stateVector)
 98
 99  end subroutine obgd_computeSSTrial

101  !----------------------------------------------------------------------------------------
102  ! obgd_getClimatology
103  !----------------------------------------------------------------------------------------
104  subroutine obgd_getClimatology(dateStamp, hco, vco, nmonthsClim, datestampClim, output)
105    !
106    !: Purpose: 1) to read SST climatological fields from a std file
107    !           2) to interpolate the field in time using the current day (t) in current month (m)    
108    !           SST(t) = SST_clim(m) + (t-1)/(ndays-1) * (SST_clim(m+1) - SST_clim(m)),
109    !           where ndays is a number of days in current month        
110    
111    implicit none
112
113    ! Arguments:
114    integer         , intent(in)          :: dateStamp        ! date stamp for the current day
115    type(struct_hco), intent(in), pointer :: hco              ! horizontal grid
116    type(struct_vco), intent(in), pointer :: vco              ! vertical grid
117    integer         , intent(in)          :: nmonthsClim      ! number of records in the climatology file 
118    integer         , intent(in)          :: datestampClim(:) ! datestamps in the climatology file
119    real(8)         , intent(inout)       :: output(:,:)      ! interpolated SST field from climatology
120  
121    ! locals
122    integer          :: hour, day, month, yyyy, ndays, nextMonth
123    type(struct_gsv) :: stateVector, stateVector_nextMonth
124    real(4), pointer :: clim_ptr(:, :, :), clim_nextMonth_ptr(:, :, :)
125    integer          :: lonIndex, latIndex
126   
127    call tim_dateStampToYYYYMMDDHH(dateStamp, hour, day, month, ndays, yyyy)
128    write(*,'(a,3i5,a,i12,a)') 'obgd_getClimatology: interpolating climatology for day/month/year (datestamp): ', &
129    day, month, yyyy, '(', datestamp, ')'
130
131    if (month == nmonthsClim) then
132      nextMonth = 1
133    else
134      nextMonth = month + 1
135    end if 
136     
137    ! get climatology, current month
138    write(*,*) 'obgd_getClimatology: reading climatology, month: ', month, ', datestamp: ', datestampClim(month) 
139    call gsv_allocate(stateVector, 1, hco, vco, dataKind_opt = 4, &
140                      datestamp_opt = datestampClim(month), mpi_local_opt = .true., &
141                      varNames_opt = (/'TM'/), hInterpolateDegree_opt ='LINEAR')
142    call gio_readFromFile(stateVector, './climatology', ' ',' ', &
143                          unitConversion_opt=.false., containsFullField_opt=.true.)
144    call gsv_getField(stateVector, clim_ptr)
145    
146    ! get climatology, next month
147    write(*,*) 'obgd_getClimatology: reading climatology, month: ', nextMonth, ', datestamp: ', datestampClim(nextMonth) 
148    call gsv_allocate(stateVector_nextMonth, 1, hco, vco, dataKind_opt = 4, &
149                      datestamp_opt = datestampClim(nextMonth), mpi_local_opt = .true., &
150                      varNames_opt = (/'TM'/), hInterpolateDegree_opt ='LINEAR')
151    call gio_readFromFile(stateVector_nextMonth, './climatology', ' ',' ', &
152                          unitConversion_opt=.false., containsFullField_opt=.true.)
153    call gsv_getField(stateVector_nextMonth, clim_nextMonth_ptr)
154
155    do lonIndex = stateVector%myLonBeg, stateVector%myLonEnd 
156      do latIndex = stateVector%myLatBeg, stateVector%myLatEnd
157        output(lonIndex, latIndex) = clim_ptr(lonIndex, latIndex, 1) + real(day - 1) / real(ndays - 1) * &
158                                     (clim_nextMonth_ptr(lonIndex, latIndex, 1) - clim_ptr(lonIndex, latIndex, 1))	 
159      end do
160    end do
161    
162    call gsv_deallocate(stateVector)
163    call gsv_deallocate(stateVector_nextMonth)
164  
165  end subroutine obgd_getClimatology 
166  
167end module oceanBackground_mod
