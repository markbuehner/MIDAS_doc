 183  subroutine ben_setup(hco_anl_in, hco_core_in, vco_anl_in, cvDimPerInstance, &
 184                       mode_opt)
 185    !
 186    !:Purpose: To configure the ensemble B matrix
 187    !
 188    implicit none
 189
 190    ! Arguments:
 191    type(struct_hco), pointer, intent(in) :: hco_anl_in
 192    type(struct_hco), pointer, intent(in) :: hco_core_in
 193    type(struct_vco), pointer, intent(in) :: vco_anl_in
 194
 195    character(len=*), intent(in), optional :: mode_opt
 196
 197    integer, allocatable, intent(out)      :: cvDimPerInstance(:)
 198
 199    ! Locals:
 200    integer        :: fnom, fclos, ierr
 201    integer        :: cvDimStorage(nInstanceMax)
 202    integer        :: nulnam = 0
 203
 204    ! Namelist variables
 205    integer             :: nEns                                   ! number of ensemble members
 206    real(8)             :: scaleFactor(vco_maxNumLevels)             ! level-dependent scaling of variances for all variables 
 207    real(8)             :: scaleFactorHumidity(vco_maxNumLevels)     ! level-dependent scaling of variances for humidity
 208    real(8)             :: advectFactorFSOFcst(vco_maxNumLevels)     ! level-dependent scaling of winds used to advect localization
 209    real(8)             :: advectFactorAssimWindow(vco_maxNumLevels) ! level-dependent scaling of winds used to advect localization
 210    integer             :: ntrunc                                 ! spectral truncation used for horizontal localization function
 211    character(len=256)  :: enspathname                            ! path where ensemble members are located (usually ./ensemble)
 212    real(8)             :: hLocalize(maxNumLocalLength)           ! horiz. localization length scale for each waveband (in km)
 213    real(8)             :: vLocalize(maxNumLocalLength)           ! vert. localization length scale for each waveband (in scale heights)
 214    character(len=256)  :: localizationType                       ! "LevelDependent", "ScaleDependent" or "ScaleDependentWithSpectralLoc"
 215    integer             :: waveBandPeaks(maxNumLocalLength)       ! total wavenumber corresponding to peak of each waveband for SDL
 216    integer             :: waveBandIndexSelected                  ! for multiple NAMBEN blocks, waveband index of this block
 217    logical             :: ensDiagnostic                          ! when `.true.` write diagnostic info related to ens. to files
 218    logical             :: advDiagnostic                          ! when `.true.` write diagnostic info related to advection to files 
 219    character(len=2)    :: ctrlVarHumidity                        ! name of humidity variable used for ensemble perturbations (LQ or HU)
 220    character(len=32)   :: advectTypeAssimWindow                  ! what is advected in assim. window: "amplitude" or "ensPertAnlInc"
 221    character(len=32)   :: advectStartTimeIndexAssimWindow        ! time index where advection originates from "first" or "middle"
 222    logical             :: advectAmplitudeFSOFcst     = .false.   ! activate advection of ens. member amplitudes for FSOI calculation
 223    logical             :: advectAmplitudeAssimWindow = .false.   ! activate advection of ens. member amplitudes for 4D-EnVar
 224    logical             :: advectEnsPertAnlInc        = .false.   ! activate advection of ens. pert. and anl. incr. for 4D-EnVar
 225    logical             :: removeSubEnsMeans                      ! remove mean of each subsensemble defined by "subEnsembleIndex.txt"
 226    logical             :: keepAmplitude                          ! activate storage of ens. amplitudes in instance of struct_ens
 227    character(len=4)    :: includeAnlVar(vnl_numvarmax)           ! list of state variables for this ensemble B matrix; use all if blank
 228    logical             :: ensContainsFullField                   ! indicates full fields and not perturbations are in the ens. files
 229    character(len=24)   :: varianceSmoothing                      ! "none", "horizMean", "footprint" or "footprintLandSeaTopo"
 230    real(8)             :: footprintRadius                        ! parameter for variance smoothing (in meters)
 231    real(8)             :: footprintTopoThreshold                 ! parameter for variance smoothing (in meters)
 232    logical             :: useCmatrixOnly                         ! activate normalization of ens. perturbations by ens. stddev
 233    integer             :: ensDateOfValidity                      ! when set to -1, date in ens. files is ignored (only for 3D ens.)
 234    real(8)             :: huMinValue                             ! minimum humidity value imposed on ensemble members 
 235    character(len=12)   :: hInterpolationDegree                   ! select degree of horizontal interpolation (if needed)
 236    character(len=20)   :: transformVarKindCH                     ! name of transform performed on chemistry-related variables in ens.
 237
 238    ! Namelist
 239    NAMELIST /NAMBEN/nEns, scaleFactor, scaleFactorHumidity, ntrunc, enspathname,             &
 240         hLocalize, vLocalize, LocalizationType, waveBandPeaks, ensDiagnostic, advDiagnostic, &
 241         ctrlVarHumidity, advectFactorFSOFcst, advectFactorAssimWindow, removeSubEnsMeans,    &
 242         keepAmplitude, advectTypeAssimWindow, advectStartTimeIndexAssimWindow, IncludeAnlVar,&
 243         ensContainsFullField, varianceSmoothing, footprintRadius, footprintTopoThreshold,    &
 244         useCmatrixOnly, waveBandIndexSelected, ensDateOfValidity, transformVarKindCH,        &
 245         huMinValue, hInterpolationDegree
 246
 247    if (verbose) write(*,*) 'Entering ben_Setup'
 248
 249    call utl_tmg_start(54,'----B_ENS_Setup')
 250
 251    !- Set the module mode
 252    if ( present(mode_opt) ) then
 253      if ( trim(mode_opt) == 'Analysis' .or. trim(mode_opt) == 'BackgroundCheck') then
 254        ben_mode = trim(mode_opt)
 255        if (mmpi_myid == 0) write(*,*)
 256        if (mmpi_myid == 0) write(*,*) 'ben_setup: Mode activated = ', trim(ben_mode)
 257      else
 258        write(*,*)
 259        write(*,*) 'mode = ', trim(mode_opt)
 260        call utl_abort('ben_setup: unknown mode')
 261      end if
 262    else
 263      ben_mode = 'Analysis'
 264      if (mmpi_myid == 0) write(*,*)
 265      if (mmpi_myid == 0) write(*,*) 'ben_setup: Analysis mode activated (by default)'
 266    end if
 267
 268    !- Open the namelist and loop through it
 269    ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
 270 
 271    instanceLoop: do
 272 
 273      !- Set the default values for the namalist parameters
 274      scaleFactor(:)        =    0.0d0
 275      scaleFactorHumidity(:)=    1.0d0
 276      nEns                  =   10
 277      ntrunc                =   30
 278      enspathname           = 'ensemble'
 279      localizationType      = 'LevelDependent'
 280      waveBandPeaks(:)      =   -1.0d0
 281      waveBandIndexSelected =   -1
 282      ensDiagnostic         = .false.
 283      advDiagnostic         = .false.
 284      hLocalize(:)          =   -1.0d0
 285      hLocalize(1)          = 2800.0d0
 286      vLocalize(:)          =   -1.0d0
 287      vLocalize(1)          =    2.0d0
 288      ctrlVarHumidity       = 'LQ'
 289      advectFactorFSOFcst(:)=   0.0D0
 290      advectTypeAssimWindow = 'amplitude'
 291      advectStartTimeIndexAssimWindow = 'first'
 292      advectFactorAssimWindow(:) = 0.0D0
 293      removeSubEnsMeans     = .false.
 294      keepAmplitude         = .false.
 295      ensContainsFullField  = .true.
 296      includeAnlVar(:)      = ''
 297      varianceSmoothing     = 'none'
 298      footprintRadius        =  250.0d3 ! 250km
 299      footprintTopoThreshold =  200.0d0 ! 200 m
 300      useCmatrixOnly        = .false.
 301      ensDateOfValidity     = MPC_missingValue_INT ! i.e. undefined
 302      transformVarKindCH    = ''
 303      huMinValue            = MPC_missingValue_R8  ! i.e. undefined
 304      hInterpolationDegree  = 'LINEAR' ! or 'CUBIC' or 'NEAREST'
 305      
 306      !- Read the namelist
 307      read(nulnam,nml=namben,iostat=ierr)
 308      if (ierr /= 0) then
 309        if (nInstance >= 1) then
 310          exit instanceLoop
 311        else
 312          call utl_abort('ben_setup: Error reading the first instance namelist')
 313        end if
 314      end if
 315
 316      if (mmpi_myid == 0) write(*,nml=namben)
 317 
 318      ! We have found a valid instance
 319      nInstance = nInstance + 1
 320
 321      !- Adjust some namelist-dependent variables
 322      
 323      ! If zero weight, skip rest of setup
 324      if ( sum(scaleFactor(:)) == 0.0d0 ) then
 325        if (mmpi_myid == 0) write(*,*) 'ben_setup: scaleFactor=0, skipping rest of setup and exit instance loop'
 326        cvDimStorage(nInstance) = 0
 327        bEns(nInstance)%initialized = .false.
 328        exit instanceLoop
 329      end if
 330
 331      if (nInstance > nInstanceMax) then
 332        call utl_abort('ben_setup: the number of instance exceed the maximum currently allowed')
 333      end if
 334
 335      if (trim(ben_mode) == 'BackgroundCheck' .and. nInstance > 1) then
 336        call utl_abort('ben_setup: the background check mode is not compatible with multiple instance')
 337      end if
 338
 339      if ( (huMinValue == MPC_missingValue_R8) .and. &
 340           gsv_varExist(varName='HU') .and. &
 341           (ctrlVarHumidity == 'LQ') ) then
 342        call utl_abort('ben_setup: the value of huMinValue must be specified in namelist NAMBEN')
 343      end if
 344
 345      !- Transfer the info to the structure
 346      bEns(nInstance)%nEns                       = nEns
 347      bEns(nInstance)%scaleFactor(:)             = scaleFactor(:)
 348      bEns(nInstance)%scaleFactorHumidity(:)     = scaleFactorHumidity(:)
 349      bEns(nInstance)%nTrunc                     = nTrunc
 350      bEns(nInstance)%ensPathName                = ensPathName
 351      bEns(nInstance)%hLocalize(:)               = hLocalize(:)
 352      bEns(nInstance)%vLocalize(:)               = vLocalize(:)
 353      bEns(nInstance)%localizationType           = localizationType
 354      bEns(nInstance)%waveBandPeaks(:)           = waveBandPeaks(:)
 355      bEns(nInstance)%waveBandIndexSelected      = waveBandIndexSelected
 356      bEns(nInstance)%ensDiagnostic              = ensDiagnostic
 357      bEns(nInstance)%advDiagnostic              = advDiagnostic
 358      bEns(nInstance)%ctrlVarHumidity            = ctrlVarHumidity
 359      bEns(nInstance)%advectTypeAssimWindow      = advectTypeAssimWindow
 360      bEns(nInstance)%advectStartTimeIndexAssimWindow = advectStartTimeIndexAssimWindow
 361      bEns(nInstance)%advectFactorFSOFcst(:)     = advectFactorFSOFcst(:)
 362      bEns(nInstance)%advectFactorAssimWindow(:) = advectFactorAssimWindow(:)
 363      bEns(nInstance)%advectAmplitudeFSOFcst     = advectAmplitudeFSOFcst
 364      bEns(nInstance)%advectAmplitudeAssimWindow = advectAmplitudeAssimWindow
 365      bEns(nInstance)%advectEnsPertAnlInc        = advectEnsPertAnlInc
 366      bEns(nInstance)%removeSubEnsMeans          = removeSubEnsMeans
 367      bEns(nInstance)%keepAmplitude              = keepAmplitude
 368      bEns(nInstance)%includeAnlVar(:)           = includeAnlVar(:)
 369      bEns(nInstance)%ensContainsFullField       = ensContainsFullField
 370      bEns(nInstance)%varianceSmoothing          = varianceSmoothing
 371      bEns(nInstance)%footprintRadius            = footprintRadius
 372      bEns(nInstance)%footprintTopoThreshold     = footprintTopoThreshold
 373      bEns(nInstance)%useCmatrixOnly             = useCmatrixOnly
 374      bEns(nInstance)%ensDateOfValidity          = ensDateOfValidity
 375      bEns(nInstance)%transformVarKindCH         = transformVarKindCH
 376      bEns(nInstance)%huMinValue                 = huMinValue
 377      bEns(nInstance)%hInterpolationDegree       = hInterpolationDegree
 378      
 379      bEns(nInstance)%hco_ens  => hco_anl_in
 380      bEns(nInstance)%hco_core => hco_core_in
 381      bEns(nInstance)%vco_anl  => vco_anl_in
 382
 383      !- Setup the LAM analysis grid metrics
 384      call agd_SetupFromHCO( hco_anl_in, hco_core_in ) ! IN
 385      
 386      !- Set the instance
 387      call ben_setupOneInstance(nInstance,               & ! IN
 388                                cvDimStorage(nInstance))   ! OUT
 389
 390      bEns(nInstance)%initialized = .true.
 391      
 392    end do instanceLoop
 393
 394    !- Close the namelist
 395    ierr = fclos(nulnam)
 396
 397    !- Set the output control variable dimensions array
 398    allocate(cvDimPerInstance(nInstance))
 399    cvDimPerInstance(:) = cvDimStorage(1:nInstance)
 400
 401    call utl_tmg_stop(54)
 402
 403  end subroutine ben_setup

 408  subroutine ben_setupOneInstance(instanceIndex, cvDim)
 409    !
 410    !:Purpose: To configure a single instance of the ensemble B matrix
 411    !
 412    implicit none
 413
 414    ! Arguments:
 415    integer, intent(in)  :: instanceIndex
 416    integer, intent(out) :: cvDim
 417
 418    ! Locals:
 419    type(struct_gsv) :: statevector_ensMean4D, statevector_oneEnsPert4D
 420
 421    type(struct_gbi) :: gbi_horizontalMean, gbi_landSeaTopo
 422
 423    real(8) :: pSurfRef, delT_hour
 424
 425    real(8), allocatable :: advectFactorFSOFcst_M(:),advectFactorAssimWindow_M(:)
 426
 427    real(8),pointer :: vertLocationEns(:), vertLocationFile(:), vertLocationInc(:)
 428
 429    real(4), pointer :: bin2d(:,:,:)
 430    real(8), pointer :: HeightSfc(:,:)
 431
 432    integer        :: lonPerPE, latPerPE, lonPerPEmax, latPerPEmax
 433    integer        :: myMemberBeg, myMemberEnd, myMemberCount, maxMyMemberCount
 434    integer        :: levIndex, jvar, ierr
 435    integer        :: waveBandIndex, stepIndex
 436    character(len=256) :: ensFileName
 437    integer        :: dateStampFSO, ensDateStampOfValidity, idate, itime, newdate
 438
 439    logical        :: EnsTopMatchesAnlTop, useAnlLevelsOnly
 440
 441    character(len=32)   :: direction, directionEnsPerts, directionAnlInc
 442
 443    if (verbose) write(*,*) 'Entering ben_SetupOneInstance'
 444    
 445    write(*,*) 'ben_setupOneInstance: enspathname = ', trim(bEns(instanceIndex)%ensPathName)
 446
 447    !
 448    !- 1.  B matrix configuration
 449    !
 450
 451    !- 1.1 Number of time step bins
 452    bEns(instanceIndex)%numStep = tim_nstepobsinc
 453    if (bEns(instanceIndex)%numStep /= 1.and.bEns(instanceIndex)%numStep /= 3.and.bEns(instanceIndex)%numStep /= 5.and.bEns(instanceIndex)%numStep /= 7) then
 454      call utl_abort('ben_setupOneInstance: Invalid value for numStep (choose 1 or 3 or 5 or 7)!')
 455    end if
 456
 457    !- 1.2 FSO-related options
 458    bEns(instanceIndex)%numStepAssimWindow = bEns(instanceIndex)%numStep
 459    if (bEns(instanceIndex)%fsoLeadTime > 0.0D0) then
 460      bEns(instanceIndex)%numStep = bEns(instanceIndex)%numStep + 1
 461      call incdatr(dateStampFSO, tim_getDatestamp(), bEns(instanceIndex)%fsoLeadTime)
 462    end if
 463
 464    allocate(bEns(instanceIndex)%dateStampList(bEns(instanceIndex)%numStep))
 465    if (bEns(instanceIndex)%fsoLeadTime > 0.0D0) then
 466      call tim_getstamplist(bEns(instanceIndex)%dateStampList,bEns(instanceIndex)%numStep-1,tim_getDatestamp())
 467      bEns(instanceIndex)%dateStampList(bEns(instanceIndex)%numStep) = dateStampFSO
 468    else
 469      if (bEns(instanceIndex)%ensDateOfValidity == MPC_missingValue_INT) then
 470        call tim_getstamplist(bEns(instanceIndex)%dateStampList,bEns(instanceIndex)%numStep,tim_getDatestamp())
 471      else
 472        if (bEns(instanceIndex)%numStep == 1) then
 473          if (bEns(instanceIndex)%ensDateOfValidity == -1) then
 474            ensDateStampOfValidity = bEns(instanceIndex)%ensDateOfValidity
 475          else
 476            idate = bEns(instanceIndex)%ensDateOfValidity/100
 477            itime = (bEns(instanceIndex)%ensDateOfValidity-idate*100)*1000000
 478            ierr = newdate(ensDateStampOfValidity, idate, itime, 3)
 479          end if
 480          bEns(instanceIndex)%dateStampList(:) = ensDateStampOfValidity
 481        else
 482          call utl_abort('ben_setupOneInstance: A single date of validity cannot be specified for numStep > 1')
 483        end if
 484      end if
 485    end if
 486
 487    !- 1.3 Horizontal grid
 488    bEns(instanceIndex)%ni = bEns(instanceIndex)%hco_ens%ni
 489    bEns(instanceIndex)%nj = bEns(instanceIndex)%hco_ens%nj
 490    if (bEns(instanceIndex)%hco_ens%global) then
 491      if (mmpi_myid == 0) write(*,*)
 492      if (mmpi_myid == 0) write(*,*) 'ben_setupOneInstance: GLOBAL mode activated'
 493    else
 494      if (mmpi_myid == 0) write(*,*)
 495      if (mmpi_myid == 0) write(*,*) 'ben_setupOneInstance: LAM mode activated'
 496    end if
 497
 498    !- 1.4 Vertical levels
 499    if ( mmpi_myid == 0 ) then
 500      call fln_ensfileName(ensFileName, bEns(instanceIndex)%ensPathName, memberIndex_opt=1)
 501      call vco_SetupFromFile(bEns(instanceIndex)%vco_file, ensFileName)
 502    end if
 503    call vco_mpiBcast(bEns(instanceIndex)%vco_file)
 504
 505    !- Do we need to read all the vertical levels from the ensemble?
 506    useAnlLevelsOnly = vco_subsetOrNot(bEns(instanceIndex)%vco_anl, bEns(instanceIndex)%vco_file)
 507    if ( useAnlLevelsOnly ) then
 508      write(*,*)
 509      write(*,*) 'ben_setupOneInstance: only the analysis levels will be read in the ensemble '
 510      bEns(instanceIndex)%vco_ens  => bEns(instanceIndex)%vco_anl ! the ensemble target grid is the analysis grid
 511      call vco_deallocate(bEns(instanceIndex)%vco_file)
 512      bEns(instanceIndex)%vco_file => bEns(instanceIndex)%vco_anl ! only the analysis levels will be read in the ensemble
 513      EnsTopMatchesAnlTop = .true.
 514    else
 515      write(*,*)
 516      write(*,*) 'ben_setupOneInstance: all the vertical levels will be read in the ensemble '
 517      if ( bEns(instanceIndex)%vco_anl%nLev_M > 0 .and. bEns(instanceIndex)%vco_anl%vgridPresent ) then
 518        pSurfRef = 101000.D0
 519        call czp_fetch1DLevels(bEns(instanceIndex)%vco_anl, pSurfRef, &
 520                               profM_opt=vertLocationInc)
 521        call czp_fetch1DLevels(bEns(instanceIndex)%vco_file, pSurfRef, &
 522                               profM_opt=vertLocationFile)
 523      
 524        do levIndex = 1, bEns(instanceIndex)%vco_anl%nLev_M
 525          vertLocationInc(levIndex) = log(vertLocationInc(levIndex))
 526        end do
 527        do levIndex = 1, bEns(instanceIndex)%vco_file%nLev_M
 528          vertLocationFile(levIndex) = log(vertLocationFile(levIndex))
 529        end do
 530
 531        EnsTopMatchesAnlTop = abs( vertLocationFile(1) - vertLocationInc(1) ) < 0.1d0
 532        write(*,*) 'ben_setupOneInstance: EnsTopMatchesAnlTop, presEns, presInc = ', &
 533             EnsTopMatchesAnlTop, vertLocationFile(1), vertLocationInc(1)
 534        deallocate(vertLocationFile)
 535        deallocate(vertLocationInc)
 536      else
 537        ! not sure what this mean when no MM levels
 538        write(*,*) 'ben_setupOneInstance: nLev_M       = ', bEns(instanceIndex)%vco_anl%nLev_M
 539        write(*,*) 'ben_setupOneInstance: vgridPresent = ', bEns(instanceIndex)%vco_anl%vgridPresent
 540        EnsTopMatchesAnlTop = .true.
 541      end if
 542
 543      if ( EnsTopMatchesAnlTop ) then
 544        if ( mmpi_myid == 0 ) write(*,*) 'ben_setupOneInstance: top level of ensemble member and analysis grid match'
 545        bEns(instanceIndex)%vco_ens => bEns(instanceIndex)%vco_anl  ! IMPORTANT: top levels DO match, therefore safe
 546        ! to force members to be on analysis vertical levels
 547      else
 548        if ( mmpi_myid == 0 ) write(*,*) 'ben_setupOneInstance: top level of ensemble member and analysis grid are different, therefore'
 549        if ( mmpi_myid == 0 ) write(*,*) '                      assume member is already be on correct levels - NO CHECKING IS DONE'
 550        bEns(instanceIndex)%vco_ens => bEns(instanceIndex)%vco_file ! IMPORTANT: top levels do not match, therefore must
 551        ! assume file is already on correct vertical levels
 552      end if
 553    end if
 554    
 555    if (bEns(instanceIndex)%vco_anl%Vcode /= bEns(instanceIndex)%vco_ens%Vcode) then
 556      write(*,*) 'ben_setupOneInstance: vco_anl%Vcode = ', bEns(instanceIndex)%vco_anl%Vcode, ', vco_ens%Vcode = ', bEns(instanceIndex)%vco_ens%Vcode
 557      call utl_abort('ben_setupOneInstance: vertical levels of ensemble not compatible with analysis grid')
 558    end if
 559    bEns(instanceIndex)%nLevEns_M  = bEns(instanceIndex)%vco_ens%nLev_M
 560    bEns(instanceIndex)%nLevEns_T  = bEns(instanceIndex)%vco_ens%nLev_T
 561    bEns(instanceIndex)%nLevEns_DP = bEns(instanceIndex)%vco_ens%nLev_Depth
 562    bEns(instanceIndex)%nLevInc_M  = bEns(instanceIndex)%vco_anl%nLev_M
 563    bEns(instanceIndex)%nLevInc_T  = bEns(instanceIndex)%vco_anl%nLev_T
 564    bEns(instanceIndex)%nLevInc_DP = bEns(instanceIndex)%vco_anl%nLev_Depth
 565    bEns(instanceIndex)%topLevIndex_M  = bEns(instanceIndex)%nLevInc_M  - bEns(instanceIndex)%nLevEns_M+1
 566    bEns(instanceIndex)%topLevIndex_T  = bEns(instanceIndex)%nLevInc_T  - bEns(instanceIndex)%nLevEns_T+1
 567    bEns(instanceIndex)%topLevIndex_DP = bEns(instanceIndex)%nLevInc_DP - bEns(instanceIndex)%nLevEns_DP+1
 568
 569    if (bEns(instanceIndex)%vco_anl%Vcode == 5002) then
 570      if ( (bEns(instanceIndex)%nLevEns_T /= (bEns(instanceIndex)%nLevEns_M+1)) .and. (bEns(instanceIndex)%nLevEns_T /= 1 .or. bEns(instanceIndex)%nLevEns_M /= 1) ) then
 571        write(*,*) 'ben_setupOneInstance: nLevEns_T, nLevEns_M = ',bEns(instanceIndex)%nLevEns_T,bEns(instanceIndex)%nLevEns_M
 572        call utl_abort('ben_setupOneInstance: Vcode=5002, nLevEns_T must equal nLevEns_M+1!')
 573      end if
 574    else if (bEns(instanceIndex)%vco_anl%Vcode == 5005) then
 575      if ( bEns(instanceIndex)%nLevEns_T /= bEns(instanceIndex)%nLevEns_M .and. &
 576           bEns(instanceIndex)%nLevEns_T /= 0 .and. &
 577           bEns(instanceIndex)%nLevEns_M /= 0 ) then
 578        write(*,*) 'ben_setup: nLevEns_T, nLevEns_M = ',bEns(instanceIndex)%nLevEns_T,bEns(instanceIndex)%nLevEns_M
 579        call utl_abort('ben_setupOneInstance: Vcode=5005, nLevEns_T must equal nLevEns_M!')
 580      end if
 581    else if (bEns(instanceIndex)%vco_anl%Vcode == 0) then
 582      if ( bEns(instanceIndex)%nLevEns_T /= 0 .and. bEns(instanceIndex)%nLevEns_M /= 0 ) then
 583        write(*,*) 'ben_setup: nLevEns_T, nLevEns_M = ',bEns(instanceIndex)%nLevEns_T, bEns(instanceIndex)%nLevEns_M
 584        call utl_abort('ben_setupOneInstance: surface-only case (Vcode=0), bEns(instanceIndex)%nLevEns_T and nLevEns_M must equal 0!')
 585      end if
 586    else
 587      write(*,*) 'vco_anl%Vcode = ',bEns(instanceIndex)%vco_anl%Vcode
 588      call utl_abort('ben_setupOneInstance: unknown vertical coordinate type!')
 589    end if
 590
 591    if (bEns(instanceIndex)%nLevEns_M.gt.bEns(instanceIndex)%nLevInc_M) then
 592      call utl_abort('ben_setupOneInstance: ensemble has more levels than increment - not allowed!')
 593    end if
 594
 595    if (bEns(instanceIndex)%nLevEns_M.lt.bEns(instanceIndex)%nLevInc_M) then
 596      if (mmpi_myid == 0) write(*,*) 'ben_setupOneInstance: ensemble has less levels than increment'
 597      if (mmpi_myid == 0) write(*,*) '                      some levels near top will have zero increment'
 598    end if
 599
 600    !- 1.5 Bmatrix Weight
 601    if (bEns(instanceIndex)%vco_anl%Vcode == 5002 .or. bEns(instanceIndex)%vco_anl%Vcode == 5005) then
 602      if (bEns(instanceIndex)%nLevEns_M > 0) then
 603        ! Multi-level or momentum-level-only analysis
 604        bEns(instanceIndex)%varNameALFA(:) = varNameALFAatmMM(:)
 605      else
 606        ! Thermo-level-only analysis
 607        bEns(instanceIndex)%varNameALFA(:) = varNameALFAatmTH(:)
 608      end if
 609      allocate(bEns(instanceIndex)%scaleFactor_M(bEns(instanceIndex)%nLevEns_M))
 610      allocate(bEns(instanceIndex)%scaleFactor_T(bEns(instanceIndex)%nLevEns_T))
 611      do levIndex = 1, bEns(instanceIndex)%nLevEns_T
 612        if (bEns(instanceIndex)%scaleFactor(levIndex) > 0.0d0) then 
 613          bEns(instanceIndex)%scaleFactor(levIndex) = sqrt(bEns(instanceIndex)%scaleFactor(levIndex))
 614        else
 615          bEns(instanceIndex)%scaleFactor(levIndex) = 0.0d0
 616        end if
 617      end do
 618      bEns(instanceIndex)%scaleFactor_T(1:bEns(instanceIndex)%nLevEns_T) = bEns(instanceIndex)%scaleFactor(1:bEns(instanceIndex)%nLevEns_T)
 619      if (bEns(instanceIndex)%vco_anl%Vcode == 5002) then
 620        bEns(instanceIndex)%scaleFactor_M(1:bEns(instanceIndex)%nLevEns_M) = bEns(instanceIndex)%scaleFactor(2:(bEns(instanceIndex)%nLevEns_M+1))
 621      else
 622        bEns(instanceIndex)%scaleFactor_M(1:bEns(instanceIndex)%nLevEns_M) = bEns(instanceIndex)%scaleFactor(1:bEns(instanceIndex)%nLevEns_M)
 623      end if
 624
 625      do levIndex = 1, bEns(instanceIndex)%nLevEns_T
 626        if (bEns(instanceIndex)%scaleFactorHumidity(levIndex) > 0.0d0) then 
 627          bEns(instanceIndex)%scaleFactorHumidity(levIndex) = sqrt(bEns(instanceIndex)%scaleFactorHumidity(levIndex))
 628        else
 629          bEns(instanceIndex)%scaleFactorHumidity(levIndex) = 0.0d0
 630        end if
 631      end do
 632      
 633      bEns(instanceIndex)%scaleFactor_SF = bEns(instanceIndex)%scaleFactor_T(bEns(instanceIndex)%nLevEns_T)
 634
 635    else if (bEns(instanceIndex)%nLevEns_DP > 0) then
 636      ! Ocean variables on depth levels
 637      write(*,*) 'nlev_Depth=', bEns(instanceIndex)%nLevEns_DP
 638      bEns(instanceIndex)%varNameALFA(:) = varNameALFAocean(:)
 639      allocate(bEns(instanceIndex)%scaleFactor_DP(bEns(instanceIndex)%nLevEns_DP))
 640      do levIndex = 1, bEns(instanceIndex)%nLevEns_DP
 641        if (bEns(instanceIndex)%scaleFactor(levIndex) > 0.0d0) then 
 642          bEns(instanceIndex)%scaleFactor(levIndex) = sqrt(bEns(instanceIndex)%scaleFactor(levIndex))
 643        else
 644          bEns(instanceIndex)%scaleFactor(levIndex) = 0.0d0
 645        end if
 646      end do
 647      bEns(instanceIndex)%scaleFactor_DP(1:bEns(instanceIndex)%nLevEns_DP) = bEns(instanceIndex)%scaleFactor(1:bEns(instanceIndex)%nLevEns_DP)
 648    else
 649      ! 2D surface variables
 650      bEns(instanceIndex)%varNameALFA(:) = varNameALFAsfc(:)
 651      if (bEns(instanceIndex)%scaleFactor(1) > 0.0d0) then 
 652        bEns(instanceIndex)%scaleFactor_SF = sqrt(bEns(instanceIndex)%scaleFactor(1))
 653      else
 654        call utl_abort('ben_setupOneInstance: with vCode == 0, the scale factor should never be equal to 0')
 655      end if
 656    end if
 657
 658    !- 1.5 Domain Partionning
 659    call mmpi_setup_latbands(bEns(instanceIndex)%nj, latPerPE, latPerPEmax, bEns(instanceIndex)%myLatBeg, bEns(instanceIndex)%myLatEnd)
 660    call mmpi_setup_lonbands(bEns(instanceIndex)%ni, lonPerPE, lonPerPEmax, bEns(instanceIndex)%myLonBeg, bEns(instanceIndex)%myLonEnd)
 661
 662    !- 1.6 Localization
 663    if ( trim(ben_mode) == 'Analysis' ) then
 664
 665      call mmpi_setup_levels(bEns(instanceIndex)%nEns,myMemberBeg,myMemberEnd,myMemberCount)
 666      call rpn_comm_allreduce(myMemberCount, maxMyMemberCount, &
 667           1,"MPI_INTEGER","MPI_MAX","GRID",ierr)
 668      bEns(instanceIndex)%nEnsOverDimension = mmpi_npex * maxMyMemberCount
 669
 670      select case(trim(bEns(instanceIndex)%localizationType))
 671      case('LevelDependent')
 672        if (mmpi_myid == 0) write(*,*)
 673        if (mmpi_myid == 0) write(*,*) 'ben_setupOneInstance: Level-Dependent (Standard) localization will be used'
 674        bEns(instanceIndex)%nWaveBand = 1
 675
 676      case('ScaleDependent','ScaleDependentWithSpectralLoc')
 677        if (mmpi_myid == 0) write(*,*)
 678        if (trim(bEns(instanceIndex)%localizationType) == 'ScaleDependent') then
 679          if (mmpi_myid == 0) write(*,*) 'ben_setupOneInstance: Scale-Dependent localization (SDL) will be used'
 680          bEns(instanceIndex)%nWaveBand             = count(bEns(instanceIndex)%waveBandPeaks >= 0)
 681          bEns(instanceIndex)%nWaveBandForFiltering = bEns(instanceIndex)%nWaveBand
 682        else
 683          if (mmpi_myid == 0) write(*,*) 'ben_setupOneInstance: Scale-Dependent localization with Spectral localization (SDLwSL) will be used'
 684          bEns(instanceIndex)%nWaveBand             = 1
 685          bEns(instanceIndex)%nWaveBandForFiltering = count(bEns(instanceIndex)%waveBandPeaks >= 0)
 686        end if
 687
 688        if ( bEns(instanceIndex)%nWaveBandForFiltering <= 1 ) then
 689          call utl_abort('ben_setupOneInstance: nWaveBandForFiltering <= 1')
 690        end if
 691        ! You must provide nWaveBand wavenumbers in decreasing order
 692        ! e.g. For a 3 wave bands decomposition...
 693        !      wavenumber #1 = where the response function for wave band 1 (hgh res) reaches 1 
 694        !                      and stays at 1 for higher wavenumbers
 695        !      wavenumber #2 = where the response function for wave band 2 reaches 1
 696        !      wavenumber #3 = where the response function for wave band 3 (low res) reaches 1 
 697        !                      and stays at 1 for lower wavenumbers
 698        ! See FilterResponseFunction for further info...
 699
 700        ! Make sure that the wavenumbers are in the correct (decreasing) order
 701        do waveBandIndex = 1, bEns(instanceIndex)%nWaveBandForFiltering-1
 702          if ( bEns(instanceIndex)%waveBandPeaks(waveBandIndex)-bEns(instanceIndex)%waveBandPeaks(waveBandIndex+1) <= 0 ) then
 703            call utl_abort('ben_setupOneInstance: waveBandPeaks are not in decreasing wavenumber order') 
 704          end if
 705        end do
 706
 707        if (trim(bEns(instanceIndex)%localizationType) == 'ScaleDependent') then
 708          ! Make sure that we have valid localization length scales for each wave bands
 709          do  waveBandIndex = 1, bEns(instanceIndex)%nWaveBand
 710            if ( bEns(instanceIndex)%hLocalize(waveBandIndex) <= 0.0d0 ) then
 711              call utl_abort('ben_setupOneInstance: Invalid HORIZONTAL localization length scale')
 712            end if
 713            if ( bEns(instanceIndex)%vLocalize(waveBandIndex) <= 0.0d0 .and. (bEns(instanceIndex)%nLevInc_M > 1 .or. bEns(instanceIndex)%nLevInc_T > 1) ) then
 714              call utl_abort('ben_setupOneInstance: Invalid VERTICAL localization length scale')
 715            end if
 716          end do
 717
 718          ! Make sure the truncation is compatible with the waveBandPeaks
 719          if ( bEns(instanceIndex)%nTrunc < bEns(instanceIndex)%waveBandPeaks(1) ) then
 720            call utl_abort('ben_setupOneInstance: The truncation is not compatible with the your scale-dependent localization')
 721          end if
 722
 723        else ! ScaleDependentWithSpectralLoc
 724          ! Do we have a valid selected waveBand index?
 725          if (bEns(instanceIndex)%waveBandIndexSelected < 1                              .or. &
 726              bEns(instanceIndex)%waveBandIndexSelected > count(bEns(instanceIndex)%waveBandPeaks >= 0)) then
 727            write(*,*) 'ben_setupOneInstance: waveBandIndexSelected = ', bEns(instanceIndex)%waveBandIndexSelected
 728            write(*,*) 'ben_setupOneInstance: number of waveBands   = ', count(bEns(instanceIndex)%waveBandPeaks >= 0)
 729            call utl_abort('ben_setupOneInstance: The selected waveBand index is not valid')
 730          end if
 731
 732          ! Make sure we have only ONE localization length scales for the selected waveBand index
 733          if ( bEns(instanceIndex)%hLocalize(2) > 0.0d0 .or. bEns(instanceIndex)%vLocalize(2) > 0.0d0 ) then
 734            call utl_abort('ben_setupOneInstance: only a single localization length scale must be provided with SDLwSL')
 735          end if
 736        end if
 737
 738      case default
 739        call utl_abort('ben_setupOneInstance: Invalid mode for LocalizationType')
 740      end select
 741
 742      ! Setup the localization
 743      if ( bEns(instanceIndex)%vco_anl%Vcode == 5002 .or. bEns(instanceIndex)%vco_anl%Vcode == 5005 ) then
 744        pSurfRef = 101000.D0
 745        call czp_fetch1DLevels(bEns(instanceIndex)%vco_anl, pSurfRef, &
 746                               profM_opt=vertLocationInc)
 747        allocate(vertLocationEns(bEns(instanceIndex)%nLevEns_M))
 748        do levIndex = 1, bEns(instanceIndex)%nLevEns_M
 749          vertLocationEns(levIndex) = log(vertLocationInc(levIndex+bEns(instanceIndex)%topLevIndex_M-1))
 750        end do
 751        deallocate(vertLocationInc)
 752      else if ( bEns(instanceIndex)%vco_anl%nLev_depth > 0 ) then
 753        allocate(vertLocationEns(bEns(instanceIndex)%vco_anl%nLev_depth))
 754        vertLocationEns(:) = bEns(instanceIndex)%vco_anl%depths(:)
 755      else
 756        pSurfRef = 101000.D0
 757        allocate(vertLocationEns(1))
 758        vertLocationEns(:) = pSurfRef
 759      end if
 760
 761      allocate(bEns(instanceIndex)%locStorage(bEns(instanceIndex)%nWaveBand))
 762      do waveBandIndex = 1, bEns(instanceIndex)%nWaveBand
 763        call loc_setup(bEns(instanceIndex)%locStorage(waveBandIndex), bEns(instanceIndex)%cvDim_mpilocal,          & ! OUT
 764                       bEns(instanceIndex)%hco_ens, bEns(instanceIndex)%vco_ens, bEns(instanceIndex)%nEns,         & ! IN
 765                       vertLocationEns, bEns(instanceIndex)%nTrunc, 'spectral',                               & ! IN
 766                       bEns(instanceIndex)%localizationType, bEns(instanceIndex)%hLocalize(waveBandIndex),         & ! IN
 767                       bEns(instanceIndex)%hLocalize(waveBandIndex+1), bEns(instanceIndex)%vLocalize(waveBandIndex)) ! IN
 768      end do
 769
 770      cvDim = bEns(instanceIndex)%cvDim_mpilocal
 771      deallocate(vertLocationEns)
 772
 773    end if
 774
 775    !- 1.7 Control variables
 776    if      ( bEns(instanceIndex)%ctrlVarHumidity == 'LQ' ) then
 777      write(*,*)
 778      write(*,*) 'ben_setupOneInstance: Humidity control variable = ', bEns(instanceIndex)%ctrlVarHumidity
 779      bEns(instanceIndex)%gsvHUcontainsLQ = .true.
 780    else if ( bEns(instanceIndex)%ctrlVarHumidity == 'HU' ) then
 781      write(*,*)
 782      write(*,*) 'ben_setupOneInstance: Humidity control variable = ', bEns(instanceIndex)%ctrlVarHumidity
 783      bEns(instanceIndex)%gsvHUcontainsLQ = .false.
 784    else
 785      write(*,*)
 786      write(*,*) 'Unknown humidity control variable'
 787      write(*,*) 'Should be LQ or LU, found = ', bEns(instanceIndex)%ctrlVarHumidity
 788      call utl_abort('ben_setupOneInstance')
 789    end if
 790
 791    !
 792    !- 2.  Read/Process the Ensemble
 793    !
 794    
 795    !- 2.1 Identify set of variables for which ensembles are required    
 796    do jvar = 1, vnl_numvarmax
 797      if (trim(bEns(instanceIndex)%includeAnlVar(jvar)) == '') exit
 798      if (.not.gsv_varExist(varName = trim(bEns(instanceIndex)%includeAnlVar(jvar)))) then
 799        write(*,*) 'ben_setupOneInstance: This variable is not a member of ANLVAR: ', trim(bEns(instanceIndex)%includeAnlVar(jvar))
 800        call utl_abort('ben_setupOneInstance: Invalid variable in includeAnlVar')
 801      else
 802        bEns(instanceIndex)%numIncludeAnlVar = bEns(instanceIndex)%numIncludeAnlVar+1
 803      end if
 804    end do
 805    if (bEns(instanceIndex)%numIncludeAnlVar == 0) then
 806      do jvar = 1, vnl_numvarmax
 807        if (.not.gsv_varExist(varName = trim(vnl_varNamelist(jvar)))) cycle
 808        bEns(instanceIndex)%numIncludeAnlVar = bEns(instanceIndex)%numIncludeAnlVar+1
 809        bEns(instanceIndex)%includeAnlVar(bEns(instanceIndex)%numIncludeAnlVar) = vnl_varNamelist(jvar)
 810      end do
 811    end if
 812
 813    if (bEns(instanceIndex)%numIncludeAnlVar == 0) call utl_abort('ben_setupOneInstance: Ensembles not being requested for any variable')
 814
 815    bEns(instanceIndex)%ensShouldNotContainLQvarName=.false.
 816    if (bEns(instanceIndex)%ctrlVarHumidity == 'LQ' .and. .not. bEns(instanceIndex)%ensContainsFullField) then
 817      ! In this particular case, we must force readEnsemble to contains the LQ varName 
 818      ! to be able to read LQ perturbations
 819      do jvar = 1, bEns(instanceIndex)%numIncludeAnlVar
 820        if (bEns(instanceIndex)%includeAnlVar(jvar) == 'LQ')  then
 821          call utl_abort('ben_setup: LQ must not be present in ANLVAR in this case')
 822        end if
 823        if (bEns(instanceIndex)%includeAnlVar(jvar) == 'HU')  then 
 824          bEns(instanceIndex)%includeAnlVar(jvar) = 'LQ'
 825          bEns(instanceIndex)%ensShouldNotContainLQvarName=.true.
 826        end if
 827      end do
 828    end if
 829
 830    !- 2.2 Read the ensemble data
 831    call setupEnsemble(instanceIndex)
 832
 833    if ( trim(ben_mode) /= 'Analysis' ) then
 834      cvDim = 9999 ! Dummy value > 0 to indicate to the background check (s/r ose_compute_HBHT_ensemble)
 835      return
 836    end if
 837
 838    !- 2.3 Convert into a C matrix
 839    if (bEns(instanceIndex)%useCmatrixOnly .or. trim(bEns(instanceIndex)%varianceSmoothing) /= 'none') then
 840      call ens_computeStdDev(bEns(instanceIndex)%ensPerts(1), containsScaledPerts_opt=.true.)
 841      call ens_normalize(bEns(instanceIndex)%ensPerts(1))
 842      bEns(instanceIndex)%ensPertsNormalized = .true.
 843    else
 844      bEns(instanceIndex)%ensPertsNormalized = .false.
 845    end if
 846
 847    !- 2.4 Variance smoothing
 848    if (trim(bEns(instanceIndex)%varianceSmoothing) /= 'none' .and. .not. bEns(instanceIndex)%useCmatrixOnly) then
 849      if (mmpi_myid == 0) write(*,*) 'ben_setupOneInstance: variance smoothing will be performed'
 850
 851      call ens_copyEnsStdDev(bEns(instanceIndex)%ensPerts(1), bEns(instanceIndex)%statevector_ensStdDev)
 852      if (bEns(instanceIndex)%ensDiagnostic) then
 853        call gio_writeToFile(bEns(instanceIndex)%statevector_ensStdDev,'./ens_stddev.fst',       & ! IN
 854                             'STDDEV_RAW', HUcontainsLQ_opt=bEns(instanceIndex)%gsvHUcontainsLQ)  ! IN
 855      end if
 856
 857      call gsv_power(bEns(instanceIndex)%statevector_ensStdDev, 2.d0 ) ! StdDev -> Variance
 858
 859      if (trim(bEns(instanceIndex)%varianceSmoothing) == 'horizMean') then
 860        if (mmpi_myid == 0) write(*,*) 'ben_setup: variance smoothing type = horizMean'
 861        call gbi_setup(gbi_horizontalMean, 'HorizontalMean', &
 862                       bEns(instanceIndex)%statevector_ensStdDev, bEns(instanceIndex)%hco_core)
 863        call gbi_mean(gbi_horizontalMean, bEns(instanceIndex)%statevector_ensStdDev, & ! IN
 864                      bEns(instanceIndex)%statevector_ensStdDev)                       ! OUT
 865        call gbi_deallocate(gbi_horizontalMean)
 866      else if (trim(bEns(instanceIndex)%varianceSmoothing) == 'footprint') then
 867        call gsv_smoothHorizontal(bEns(instanceIndex)%statevector_ensStdDev, & ! INOUT
 868                                  bEns(instanceIndex)%footprintRadius)         ! IN
 869      else if (trim(bEns(instanceIndex)%varianceSmoothing) == 'footprintLandSeaTopo') then
 870        call gbi_setup(gbi_landSeaTopo, 'landSeaTopo', bEns(instanceIndex)%statevector_ensStdDev, &
 871                       bEns(instanceIndex)%hco_core, &
 872                       mpi_distribution_opt='None', writeBinsToFile_opt=bEns(instanceIndex)%ensDiagnostic)
 873        call gsv_getField(gbi_landSeaTopo%statevector_bin2d,bin2d)
 874        HeightSfc => gsv_getHeightSfc(gbi_landSeaTopo%statevector_bin2d)
 875        call gsv_smoothHorizontal(bEns(instanceIndex)%statevector_ensStdDev,                                       & ! INOUT
 876                                  bEns(instanceIndex)%footprintRadius, binInteger_opt=bin2d, binReal_opt=HeightSfc,& ! IN
 877                                  binRealThreshold_opt=bEns(instanceIndex)%footprintTopoThreshold)                   ! IN
 878        call gbi_deallocate(gbi_landSeaTopo)
 879      else
 880        call utl_abort('ben_setupOneInstance: Invalid variance smoothing type = '//trim(bEns(instanceIndex)%varianceSmoothing))
 881      end if
 882
 883      call gsv_power(bEns(instanceIndex)%statevector_ensStdDev, 0.5d0) ! Variance -> StdDev
 884
 885      if (bEns(instanceIndex)%ensDiagnostic) then
 886        call gio_writeToFile(bEns(instanceIndex)%statevector_ensStdDev,'./ens_stddev.fst',& ! IN
 887                             'STDDEV_SMOOT', HUcontainsLQ_opt=bEns(instanceIndex)%gsvHUcontainsLQ)   ! IN
 888      end if
 889    end if
 890
 891    !- 2.5 Pre-compute everything for advection in FSO mode
 892    if (bEns(instanceIndex)%fsoLeadTime > 0.0D0) then
 893      bEns(instanceIndex)%amp3dStepIndexFSOFcst = 1
 894      if ( sum(bEns(instanceIndex)%advectFactorFSOFcst(:)) == 0.0D0 .or. bEns(instanceIndex)%numStep == 1) then
 895        if (mmpi_myid == 0) write(*,*) 'ben_setupOneInstance: advection not activated for FSO'
 896        bEns(instanceIndex)%advectAmplitudeFSOFcst = .false.
 897        bEns(instanceIndex)%numStepAmplitudeFSOFcst = 1
 898      else
 899        if (mmpi_myid == 0) write(*,*) 'ben_setupOneInstance: advection activated in FSO mode'
 900        bEns(instanceIndex)%advectAmplitudeFSOFcst = .true.
 901        bEns(instanceIndex)%numStepAmplitudeFSOFcst = 2
 902        bEns(instanceIndex)%numStepAdvectFSOFcst = nint(bEns(instanceIndex)%fsoLeadTime/6.0D0) + 1
 903        allocate(advectFactorFSOFcst_M(bEns(instanceIndex)%vco_ens%nLev_M))
 904        advectFactorFSOFcst_M(:) = bEns(instanceIndex)%advectFactorFSOFcst(1:bEns(instanceIndex)%vco_ens%nLev_M)
 905        allocate(bEns(instanceIndex)%dateStampListAdvectedFields(bEns(instanceIndex)%numStepAmplitudeFSOFcst))
 906        bEns(instanceIndex)%dateStampListAdvectedFields(1) = tim_getDatestamp()
 907        bEns(instanceIndex)%dateStampListAdvectedFields(2) = bEns(instanceIndex)%dateStampList(bEns(instanceIndex)%numStep)
 908        delT_hour = bEns(instanceIndex)%fsoLeadTime/real(bEns(instanceIndex)%numStepAdvectFSOFcst-1,8) ! time between winds
 909        call utl_tmg_start(55,'------B_ENS_SetupAdvecFSO')
 910        call adv_setup( bEns(instanceIndex)%adv_amplitudeFSOFcst,                                   & ! OUT
 911                        'fromFirstTimeIndex', bEns(instanceIndex)%hco_ens, bEns(instanceIndex)%vco_ens,                 & ! IN
 912                        bEns(instanceIndex)%numStepAmplitudeFSOFcst, bEns(instanceIndex)%dateStampListAdvectedFields,   & ! IN
 913                        bEns(instanceIndex)%numStepAdvectFSOFcst, delT_hour, advectFactorFSOFcst_M, & ! IN
 914                        'MMLevsOnly',                                           & ! IN
 915                        steeringFlowFilename_opt=trim(bEns(instanceIndex)%ensPathName)//'/forecast_for_advection' ) ! IN
 916        call utl_tmg_stop(55)
 917        deallocate(advectFactorFSOFcst_M)
 918      end if
 919    else
 920      bEns(instanceIndex)%numStepAmplitudeFSOFcst = 0
 921    end if
 922
 923    !- 2.6 Pre-compute everything for advection in ANALYSIS mode
 924    if ( sum(bEns(instanceIndex)%advectFactorAssimWindow(:)) == 0.0D0 .or. bEns(instanceIndex)%numStep == 1) then
 925      if (mmpi_myid == 0) write(*,*) 'ben_setupOneInstance: advection not activated in ANALYSIS mode'
 926
 927      bEns(instanceIndex)%advectAmplitudeAssimWindow = .false.
 928      bEns(instanceIndex)%numStepAmplitudeAssimWindow = 1
 929      bEns(instanceIndex)%amp3dStepIndexAssimWindow   = 1
 930
 931    else
 932      if (mmpi_myid == 0) write(*,*) 'ben_setupOneInstance: advection activated in ANALYSIS mode'
 933
 934      delT_hour                 = tim_dstepobsinc
 935      allocate(advectFactorAssimWindow_M(bEns(instanceIndex)%vco_ens%nLev_M))
 936      advectFactorAssimWindow_M(:) = bEns(instanceIndex)%advectFactorAssimWindow(1:bEns(instanceIndex)%vco_ens%nLev_M)
 937      allocate(bEns(instanceIndex)%dateStampListAdvectedFields(bEns(instanceIndex)%numStep))
 938      bEns(instanceIndex)%dateStampListAdvectedFields(:) = bEns(instanceIndex)%dateStampList(:)
 939      call gsv_allocate(statevector_ensMean4D, bEns(instanceIndex)%numStep, bEns(instanceIndex)%hco_ens, bEns(instanceIndex)%vco_ens, &
 940                        datestampList_opt=bEns(instanceIndex)%dateStampListAdvectedFields,     &
 941                        mpi_local_opt=.true., &
 942                        allocHeight_opt=.false., allocPressure_opt=.false.)
 943      call ens_copyEnsMean(bEns(instanceIndex)%ensPerts(1), & ! IN
 944                           statevector_ensMean4D  )   ! OUT
 945
 946      call utl_tmg_start(56,'------B_ENS_SetupAdvecAnl')
 947
 948      select case(trim(bEns(instanceIndex)%advectTypeAssimWindow))
 949      case ('amplitude')
 950        if (mmpi_myid == 0) write(*,*) '         amplitude fields will be advected'
 951        bEns(instanceIndex)%advectAmplitudeAssimWindow  = .true.
 952        bEns(instanceIndex)%numStepAmplitudeAssimWindow = bEns(instanceIndex)%numStep
 953        bEns(instanceIndex)%numStepAdvectAssimWindow    = bEns(instanceIndex)%numStep
 954
 955        select case(trim(bEns(instanceIndex)%advectStartTimeIndexAssimWindow))
 956        case ('first')
 957          direction='fromFirstTimeIndex'
 958          bEns(instanceIndex)%amp3dStepIndexAssimWindow = 1
 959        case ('middle')
 960          direction='fromMiddleTimeIndex'
 961          bEns(instanceIndex)%amp3dStepIndexAssimWindow = (bEns(instanceIndex)%numStepAmplitudeAssimWindow+1)/2
 962        case default
 963          write(*,*)
 964          write(*,*) 'Unsupported starting timeIndex : ', trim(bEns(instanceIndex)%advectStartTimeIndexAssimWindow)
 965          call utl_abort('ben_setupOneInstance')
 966        end select
 967        call adv_setup( bEns(instanceIndex)%adv_amplitudeAssimWindow,                                          & ! OUT
 968                        direction, bEns(instanceIndex)%hco_ens, bEns(instanceIndex)%vco_ens,                                       & ! IN
 969                        bEns(instanceIndex)%numStepAmplitudeAssimWindow, bEns(instanceIndex)%dateStampListAdvectedFields,          & ! IN
 970                        bEns(instanceIndex)%numStepAdvectAssimWindow, delT_hour, advectFactorAssimWindow_M,    & ! IN
 971                        'MMLevsOnly', statevector_steeringFlow_opt = statevector_ensMean4D ) ! IN
 972
 973      case('ensPertAnlInc')
 974        if (mmpi_myid == 0) write(*,*) '         ensPerts and AnalInc will be advected'
 975
 976        if (.not. EnsTopMatchesAnlTop) then
 977          call utl_abort('ben_setupOneInstance: for advectTypeAssimWindow=ensPertAnlInc, ensTop and anlTop must match!')
 978        end if
 979
 980        bEns(instanceIndex)%advectEnsPertAnlInc         = .true.
 981        bEns(instanceIndex)%amp3dStepIndexAssimWindow   = 1
 982        bEns(instanceIndex)%numStepAmplitudeAssimWindow = 1
 983        bEns(instanceIndex)%numStepAdvectAssimWindow    = bEns(instanceIndex)%numStep
 984        
 985        select case(trim(bEns(instanceIndex)%advectStartTimeIndexAssimWindow))
 986        case ('first')
 987          directionEnsPerts='towardFirstTimeIndex'
 988          directionAnlInc  ='towardFirstTimeIndexInverse'
 989        case ('middle')
 990          directionEnsPerts='towardMiddleTimeIndex'
 991          directionAnlInc  ='towardMiddleTimeIndexInverse'
 992        case default
 993          write(*,*)
 994          write(*,*) 'Unsupported starting timeIndex : ', trim(bEns(instanceIndex)%advectStartTimeIndexAssimWindow)
 995          call utl_abort('ben_setupOneInstance')
 996        end select
 997
 998        call adv_setup( bEns(instanceIndex)%adv_ensPerts,                                                              & ! OUT
 999                        directionEnsPerts, bEns(instanceIndex)%hco_ens, bEns(instanceIndex)%vco_ens,                   & ! IN
1000                        bEns(instanceIndex)%numStepAdvectAssimWindow, bEns(instanceIndex)%dateStampListAdvectedFields, & ! IN
1001                        bEns(instanceIndex)%numStepAdvectAssimWindow, delT_hour, advectFactorAssimWindow_M,            & ! IN
1002                        'allLevs', statevector_steeringFlow_opt=statevector_ensMean4D )   ! IN
1003
1004        call adv_setup( bEns(instanceIndex)%adv_analInc,                                                               & ! OUT
1005                        directionAnlInc, bEns(instanceIndex)%hco_ens, bEns(instanceIndex)%vco_ens,                     & ! IN
1006                        bEns(instanceIndex)%numStepAdvectAssimWindow, bEns(instanceIndex)%dateStampListAdvectedFields, & ! IN
1007                        bEns(instanceIndex)%numStepAdvectAssimWindow, delT_hour, advectFactorAssimWindow_M,            & ! IN
1008                        'allLevs', statevector_steeringFlow_opt=statevector_ensMean4D )   ! IN
1009
1010      case default
1011        write(*,*)
1012        write(*,*) 'Unsupported advectTypeAssimWindow : ', trim(bEns(instanceIndex)%advectTypeAssimWindow)
1013        call utl_abort('ben_setupOneInstance')
1014      end select
1015
1016      call utl_tmg_stop(56)
1017
1018      deallocate(advectFactorAssimWindow_M)
1019
1020      !- If wanted, write the ensemble mean
1021      if (bEns(instanceIndex)%advDiagnostic) then
1022        do stepIndex = 1, tim_nstepobsinc
1023          call gio_writeToFile(statevector_ensMean4D,'./ens_mean.fst','ENSMEAN4D',        & ! IN
1024                               stepIndex_opt=stepIndex, HUcontainsLQ_opt=bEns(instanceIndex)%gsvHUcontainsLQ ) ! IN
1025        end do
1026      end if
1027
1028      call gsv_deallocate(statevector_ensMean4D)
1029
1030    end if
1031
1032    !- 2.7 Ensemble perturbations advection
1033    if ( bEns(instanceIndex)%advectEnsPertAnlInc ) then
1034
1035      !- If wanted, write the original ensemble perturbations for member #1
1036      if (bEns(instanceIndex)%advDiagnostic) then
1037        call ens_copyMember(bEns(instanceIndex)%ensPerts(1), statevector_oneEnsPert4D, 1)
1038        do stepIndex = 1, tim_nstepobsinc
1039          call gio_writeToFile(statevector_oneEnsPert4D,'./ens_pert1.fst','ORIGINAL', & ! IN
1040               stepIndex_opt=stepIndex, HUcontainsLQ_opt=bEns(instanceIndex)%gsvHUcontainsLQ )             ! IN
1041        end do
1042      end if
1043
1044      !- Do the advection of all the members
1045      call adv_ensemble_tl( bEns(instanceIndex)%ensPerts(1), &       ! INOUT
1046                            bEns(instanceIndex)%adv_ensPerts, bEns(instanceIndex)%nEns ) ! IN
1047
1048      !- If wanted, write the advected ensemble perturbations for member #1
1049      if (bEns(instanceIndex)%advDiagnostic) then
1050        call ens_copyMember(bEns(instanceIndex)%ensPerts(1), statevector_oneEnsPert4D, 1)
1051        do stepIndex = 1, tim_nstepobsinc
1052          call gio_writeToFile(statevector_oneEnsPert4D,'./ens_pert1_advected.fst','ADVECTED', & ! IN
1053               stepIndex_opt=stepIndex,HUcontainsLQ_opt=bEns(instanceIndex)%gsvHUcontainsLQ )                       ! IN
1054        end do
1055      end if
1056
1057    end if
1058
1059    !- 2.8 Compute and write Std. Dev.
1060    if (bEns(instanceIndex)%ensDiagnostic) call ensembleDiagnostic(instanceIndex,'FullPerturbations')
1061
1062    !- 2.9 Partitioned the ensemble perturbations into wave bands
1063    if (trim(bEns(instanceIndex)%localizationType) == 'ScaleDependent'                .or. &
1064        trim(bEns(instanceIndex)%localizationType) == 'ScaleDependentWithSpectralLoc') then
1065      call ensembleScaleDecomposition(instanceIndex)
1066      if (bEns(instanceIndex)%ensDiagnostic) call ensembleDiagnostic(instanceIndex,'WaveBandPerturbations')
1067    end if
1068
1069    ! Allocate main ensemble Amplitude arrays for improved efficiency - do not know why!!!
1070    if (bEns(instanceIndex)%numStepAmplitudeAssimWindow > 1 .or. bEns(instanceIndex)%numStepAmplitudeFSOFcst > 1 ) then
1071      write(*,*) 'ben_setupOneInstance: WARNING: due to advection being activated, cannot currently used saved ensemble amplitudes!'
1072      bEns(instanceIndex)%useSaveAmp = .false.
1073    else
1074      write(*,*) 'ben_setupOneInstance: using saved memory for ensemble amplitudes for improved efficiency'
1075      bEns(instanceIndex)%useSaveAmp = .true.
1076      call ens_allocate(ensAmplitudeSave(instanceIndex),                 &
1077                        bEns(instanceIndex)%nEnsOverDimension,           &
1078                        bEns(instanceIndex)%numStepAmplitudeAssimWindow, &
1079                        bEns(instanceIndex)%hco_ens,                     &
1080                        bEns(instanceIndex)%vco_ens,                     &
1081                        bEns(instanceIndex)%dateStampList,               &
1082                        hco_core_opt=bEns(instanceIndex)%hco_core,       &
1083                        varNames_opt=bEns(instanceIndex)%varNameALFA, dataKind_opt=8)
1084      call ens_zero(ensAmplitudeSave(instanceIndex))
1085    end if
1086
1087    !- 2.10 Setup en ensGridStateVector to store the amplitude fields (for writing)
1088    if (bEns(instanceIndex)%keepAmplitude) then
1089      write(*,*)
1090      write(*,*) 'ben_setupOneInstance: ensAmplitude fields will be store for potential write to file'
1091      call ens_allocate(bEns(instanceIndex)%ensAmplitudeStorage, bEns(instanceIndex)%nEns, &
1092                        bEns(instanceIndex)%numStepAmplitudeAssimWindow, &
1093                        bEns(instanceIndex)%hco_ens, bEns(instanceIndex)%vco_ens, &
1094                        bEns(instanceIndex)%dateStampList, &
1095                        hco_core_opt=bEns(instanceIndex)%hco_core, &
1096                        varNames_opt=bEns(instanceIndex)%varNameALFA, dataKind_opt=8)
1097    end if
1098
1099  end subroutine ben_setupOneInstance

1186  !--------------------------------------------------------------------------
1187  ! setupEnsemble
1188  !--------------------------------------------------------------------------
1189  subroutine setupEnsemble(instanceIndex)
1190    implicit none
1191
1192    integer, intent(in) :: instanceIndex
1193
1194    real(4), pointer     :: ptr4d_r4(:,:,:,:)
1195    real(8) :: multFactor
1196    integer :: stepIndex,levIndex,lev,waveBandIndex,memberIndex,varIndex
1197    logical :: makeBiPeriodic
1198    character(len=4)  :: varName
1199    character(len=30) :: transform
1200
1201    write(*,*) 'setupEnsemble: Start'
1202    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
1203
1204    !- 1. Memory allocation
1205    allocate(bEns(instanceIndex)%ensPerts(bEns(instanceIndex)%nWaveBand))    
1206    do waveBandIndex = 1, bEns(instanceIndex)%nWaveBand
1207      call ens_allocate(bEns(instanceIndex)%ensPerts(waveBandIndex),  &
1208                        bEns(instanceIndex)%nEns, bEns(instanceIndex)%numStep, &
1209                        bEns(instanceIndex)%hco_ens,  &
1210                        bEns(instanceIndex)%vco_ens, bEns(instanceIndex)%dateStampList, &
1211                        hco_core_opt = bEns(instanceIndex)%hco_core, &
1212                        varNames_opt = bEns(instanceIndex)%includeAnlVar(1:bEns(instanceIndex)%numIncludeAnlVar), &
1213                        hInterpolateDegree_opt = bEns(instanceIndex)%hInterpolationDegree)
1214    end do
1215
1216    !- 2. Read ensemble
1217    makeBiPeriodic = (trim(bEns(instanceIndex)%localizationType) == 'ScaleDependent' .or. &
1218                      trim(bEns(instanceIndex)%localizationType) == 'ScaleDependentWithSpectralLoc' .or. &
1219                      trim(bEns(instanceIndex)%varianceSmoothing) /= 'none')
1220    call ens_readEnsemble(bEns(instanceIndex)%ensPerts(1), bEns(instanceIndex)%ensPathName, makeBiPeriodic,         &
1221                          vco_file_opt = bEns(instanceIndex)%vco_file,                          &
1222                          varNames_opt = bEns(instanceIndex)%includeAnlVar(1:bEns(instanceIndex)%numIncludeAnlVar), & 
1223                          containsFullField_opt=bEns(instanceIndex)%ensContainsFullField)
1224
1225    if ( bEns(instanceIndex)%ctrlVarHumidity == 'LQ' .and. ens_varExist(bEns(instanceIndex)%ensPerts(1),'HU') .and. &
1226         bEns(instanceIndex)%ensContainsFullField ) then
1227      call gvt_transform(bEns(instanceIndex)%ensPerts(1),'HUtoLQ',huMinValue_opt=bEns(instanceIndex)%huMinValue)
1228    else if ( bEns(instanceIndex)%ctrlVarHumidity == 'HU' .and. ens_varExist(bEns(instanceIndex)%ensPerts(1),'HU') .and. &
1229         bEns(instanceIndex)%ensContainsFullField .and. bEns(instanceIndex)%huMinValue /= MPC_missingValue_R8 ) then
1230      call qlim_setMin(bEns(instanceIndex)%ensPerts(1), bEns(instanceIndex)%huMinValue)
1231    else if ( trim(bEns(instanceIndex)%transformVarKindCH) /= '' ) then
1232      do varIndex = 1, bEns(instanceIndex)%numIncludeAnlVar
1233        if ( vnl_varKindFromVarname(bEns(instanceIndex)%includeAnlVar(varIndex)) /= 'CH' ) cycle            
1234
1235        transform = trim(bens(instanceIndex)%transformVarKindCH)//'CH'
1236        call gvt_transform( bEns(instanceIndex)%ensPerts(1), trim(transform), &          
1237                            varName_opt=bEns(instanceIndex)%includeAnlVar(varIndex) ) 
1238      end do
1239    end if
1240
1241    !- 3. From ensemble FORECASTS to ensemble PERTURBATIONS
1242
1243    !- 3.1 remove mean
1244    call ens_computeMean( bEns(instanceIndex)%ensPerts(1), bEns(instanceIndex)%removeSubEnsMeans, numSubEns_opt=bEns(instanceIndex)%numSubEns )
1245    call ens_removeMean( bEns(instanceIndex)%ensPerts(1) )
1246
1247    !- 3.2 normalize and apply scale factors
1248    !$OMP PARALLEL DO PRIVATE (levIndex,varName,lev,ptr4d_r4,stepIndex,memberIndex,multFactor)
1249    do levIndex = 1, ens_getNumK(bEns(instanceIndex)%ensPerts(1))
1250      varName = ens_getVarNameFromK(bEns(instanceIndex)%ensPerts(1),levIndex)
1251      lev = ens_getLevFromK(bEns(instanceIndex)%ensPerts(1),levIndex)
1252
1253      if ( .not. ens_varExist(bEns(instanceIndex)%ensPerts(1), varName) ) cycle 
1254
1255      ptr4d_r4 => ens_getOneLev_r4(bEns(instanceIndex)%ensPerts(1),levIndex)
1256
1257      do stepIndex = 1, bEns(instanceIndex)%numStep
1258        do memberIndex = 1, bEns(instanceIndex)%nEns
1259
1260          if ( vnl_varLevelFromVarname(varName) == 'MM' ) then
1261            multFactor = bEns(instanceIndex)%scaleFactor_M(lev)
1262          else if ( vnl_varLevelFromVarname(varName) == 'TH' ) then
1263            multFactor = bEns(instanceIndex)%scaleFactor_T(lev)
1264          else  if ( vnl_varLevelFromVarname(varName) == 'SF' ) then

1629                     'LatLonMN', maxlevels_opt=bEns(instanceIndex)%nEnsOverDimension, gridDataOrder_opt='kij' )  ! IN
1630
1631      nla_filter    = lst_ben_filter%nla
1632      nphase_filter = lst_ben_filter%nphase
1633    end if
1634
1635    !
1636    !- 1.  Scale decomposition for every wave band except for wave band #1
1637    !
1638    if (trim(bEns(instanceIndex)%localizationType) == 'ScaleDependent') then
1639      waveBandIndexStart     = 2                ! Skip the smallest scales

1643      waveBandIndexLoopDirection = -1

1646      waveBandIndexEnd       = bEns(instanceIndex)%waveBandIndexSelected

1648      waveBandIndexLoopEnd   = waveBandIndexEnd

1677

1679      do levIndex = 1, ens_getNumK(bEns(instanceIndex)%ensPerts(1)) ! Loop on variables and vertical levels
1680
1681        ptr4d_r4 => ens_getOneLev_r4(bEns(instanceIndex)%ensPerts(1),levIndex)
1682          
1683        !- 1.2 GridPoint space -> Spectral Space

1687        end do
1688        !$OMP END PARALLEL DO
1689        !$OMP PARALLEL DO PRIVATE (memberIndex,latIndex,lonIndex)
1690        do latIndex = bEns(instanceIndex)%myLatBeg, bEns(instanceIndex)%myLatEnd

1695          end do

1706          call lst_VarTransform(lst_ben_filter,         & ! IN
1707                                ensPertSP,              & ! OUT
1708                                ensPertGD,              & ! IN 
1709                                kind, bEns(instanceIndex)%nEnsOverDimension ) ! IN
1710        end if

1728          if (bEns(instanceIndex)%hco_ens%global) then
1729            ! Global Mode
1730            call gst_setID(gstFilterID) ! IN
1731            call gst_speree_kij(ensPertSPfiltered, & ! IN
1732                                ensPertGD)           ! OUT
1733          else
1734            ! LAM mode
1735            kind = 'SpectralToGridPoint'
1736            call lst_VarTransform(lst_ben_filter,         & ! IN
1737                                  ensPertSPfiltered,      & ! IN
1738                                  ensPertGD,              & ! OUT
1739                                  kind, bEns(instanceIndex)%nEnsOverDimension ) ! IN
1740          end if
1741
1742          if (trim(bEns(instanceIndex)%localizationType) == 'ScaleDependent') then
1743            ptr4d_r4 => ens_getOneLev_r4(bEns(instanceIndex)%ensPerts(waveBandIndex),levIndex)
1744          else
1745            ptr4d_r4 => ens_getOneLev_r4(bEns(instanceIndex)%ensPerts(1),levIndex)
1746          end if
1747          !$OMP PARALLEL DO PRIVATE (memberIndex,latIndex,lonIndex)
1748          do latIndex = bEns(instanceIndex)%myLatBeg, bEns(instanceIndex)%myLatEnd
1749            do lonIndex = bEns(instanceIndex)%myLonBeg, bEns(instanceIndex)%myLonEnd
1750              do memberIndex = 1, bEns(instanceIndex)%nEns
1751                ptr4d_r4(memberIndex,stepIndex,lonIndex,latIndex) = sngl(ensPertGD(memberIndex,lonIndex,latIndex))
1752              end do
1753            end do
1754          end do
1755          !$OMP END PARALLEL DO
1756
1757        end do ! waveBandIndex
1758      end do ! time bins
1759    end do ! variables&levels
1760
1761    deallocate(ensPertGD)
1762    deallocate(ResponseFunction)
1763    deallocate(ensPertSP)
1764    deallocate(ensPertSPfiltered)
1765
1766    !
1767    !- 2.  For SDL only, Isolate the smallest scales in waveBandIndex = 1 by difference in grid point space
1768    !
