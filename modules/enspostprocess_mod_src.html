  51  !----------------------------------------------------------------------
  52  ! epp_postProcess
  53  !----------------------------------------------------------------------
  54  subroutine epp_postProcess(ensembleTrl, ensembleAnl, &
  55                             stateVectorHeightSfc, stateVectorCtrlTrl, &
  56                             writeTrlEnsemble, outputOnlyEnsMean_opt)
  57    !
  58    !:Purpose:  Perform numerous post-processing steps to the ensemble
  59    !           produced by the LETKF algorithm.
  60    !
  61    implicit none
  62
  63    ! Arguments
  64    type(struct_ens), intent(inout) :: ensembleTrl
  65    type(struct_ens), intent(inout) :: ensembleAnl
  66    type(struct_gsv), intent(in)    :: stateVectorHeightSfc
  67    type(struct_gsv), intent(inout) :: stateVectorCtrlTrl
  68    logical, intent(in)             :: writeTrlEnsemble
  69    logical, optional, intent(in)   :: outputOnlyEnsMean_opt
  70
  71    ! Locals
  72    integer                   :: ierr, nEns, dateStamp, datePrint, timePrint, imode, randomSeedRandomPert
  73    integer                   :: stepIndex, middleStepIndex, nulnam
  74    integer, allocatable      :: dateStampListInc(:)
  75    type(struct_hco), pointer :: hco_ens
  76    type(struct_vco), pointer :: vco_ens
  77    type(struct_gsv)          :: stateVectorMeanAnl, stateVectorMeanAnlRaw, stateVectorMeanTrl
  78    type(struct_gsv)          :: stateVectorMeanInc
  79    type(struct_gsv)          :: stateVectorAnalIncMask
  80    type(struct_gsv)          :: stateVectorStdDevAnl, stateVectorStdDevAnlRaw, stateVectorStdDevAnlPert, stateVectorStdDevTrl
  81    type(struct_gsv)          :: stateVectorMeanIncSubSample
  82    type(struct_gsv)          :: stateVectorMeanAnlSubSample
  83    type(struct_gsv)          :: stateVectorMeanAnlSfcPres
  84    type(struct_gsv)          :: stateVectorMeanAnlSfcPresMpiGlb
  85    type(struct_ens)          :: ensembleTrlSubSample
  86    type(struct_ens)          :: ensembleAnlSubSample
  87    type(struct_ens)          :: ensembleAnlSubSampleUnPert
  88    character(len=12)         :: etiket
  89    type(struct_ens)          :: ensembleAnlInc
  90    type(struct_ens)          :: ensembleAnlIncSubSample
  91    character(len=256)        :: outFileName
  92    character(len=4), pointer :: varNames(:)
  93    character(len=12)         :: hInterpolationDegree = 'LINEAR'
  94    integer, external         :: fnom, fclos, newdate
  95    logical                   :: outputOnlyEnsMean
  96
  97    ! Namelist variables
  98    integer  :: randomSeed           ! seed used for random perturbation additive inflation
  99    logical  :: includeYearInSeed    ! switch for choosing to include year in default random seed
 100    logical  :: writeSubSample       ! write sub-sample members for initializing medium-range fcsts
 101    logical  :: writeSubSampleUnPert ! write unperturbed sub-sample members for medium-range fcsts
 102    real(8)  :: alphaRTPS            ! RTPS coefficient (between 0 and 1; 0 means no relaxation)
 103    real(8)  :: alphaRTPP            ! RTPP coefficient (between 0 and 1; 0 means no relaxation)
 104    real(8)  :: alphaRandomPert      ! Random perturbation additive inflation coeff (0->1)
 105    real(8)  :: alphaRandomPertSubSample ! Random pert. additive inflation coeff for medium-range fcsts
 106    logical  :: huLimitsBeforeRecenter   ! Choose to apply humidity limits before recentering
 107    logical  :: imposeSaturationLimit  ! switch for choosing to impose saturation limit of humidity
 108    logical  :: imposeRttovHuLimits    ! switch for choosing to impose the RTTOV limits on humidity
 109    real(8)  :: weightRecenter(vco_maxNumLevels)  ! weight applied to recentering increment (between 0 and 1; 0 means no recentering)
 110    real(8)  :: weightRecenterLand     ! weight applied to recentering increment for land variables
 111    integer  :: numMembersToRecenter   ! number of members that get recentered on supplied analysis
 112    logical  :: useOptionTableRecenter ! use values in the optiontable file
 113    character(len=8)  :: etiket_anl ! etiket for ensemble output files (member number will be appended)
 114    character(len=8)  :: etiket_inc ! etiket for ensemble output files (member number will be appended)
 115    character(len=8)  :: etiket_trl ! etiket for ensemble output files (member number will be appended)
 116    character(len=12) :: etiket_anlmean ! etiket for mean of analyses and mean of increments files
 117    character(len=12) :: etiket_anlrms  ! etiket for rms of analyses files
 118    character(len=12) :: etiket_anlmean_raw ! etiket for mean of raw analyses
 119    character(len=12) :: etiket_anlrms_raw  ! etiket for rms of raw analyses
 120    character(len=12) :: etiket_anlmeanpert ! etiket for mean of perturbed analyses
 121    character(len=12) :: etiket_anlrmspert  ! etiket for rms of perturbed analyses
 122    character(len=12) :: etiket_trlmean     ! etiket for mean of trials
 123    character(len=12) :: etiket_trlrms      ! etiket for rms of trials
 124    integer  :: numBits ! number of bits when writing ensemble mean and spread
 125    logical  :: useAnalIncMask        ! mask out the increment on the pilot zone
 126    logical  :: writeRawAnalStats     ! write mean and standard deviation of the raw analysis ensemble
 127    logical  :: useMemberAsHuRefState ! use each member as reference state for variable transforms 
 128
 129    NAMELIST /namEnsPostProcModule/randomSeed, includeYearInSeed, writeSubSample, writeSubSampleUnPert,  &
 130                                   alphaRTPS, alphaRTPP, alphaRandomPert, alphaRandomPertSubSample,      &
 131                                   huLimitsBeforeRecenter, imposeSaturationLimit, imposeRttovHuLimits,   &
 132                                   weightRecenter, weightRecenterLand, numMembersToRecenter,  &
 133                                   useOptionTableRecenter,  &
 134                                   etiket_anl, etiket_inc, etiket_trl, etiket_anlmean, etiket_anlrms,    &
 135                                   etiket_anlmeanpert, etiket_anlrmspert,  &
 136                                   etiket_anlmean_raw, etiket_anlrms_raw,  &
 137                                   etiket_trlmean, etiket_trlrms, numBits, useAnalIncMask,  &
 138                                   writeRawAnalStats, useMemberAsHuRefState
 139
 140    if (present(outputOnlyEnsMean_opt)) then
 141      outputOnlyEnsMean = outputOnlyEnsMean_opt
 142    else
 143      outputOnlyEnsMean = .false.
 144    end if
 145
 146    !- Extract the grid definitions and ensemble size
 147    if (ens_isAllocated(ensembleTrl)) then
 148      hco_ens => ens_getHco(ensembleTrl)
 149      vco_ens => ens_getVco(ensembleTrl)
 150      nEns = ens_getNumMembers(ensembleTrl)
 151    else
 152      hco_ens => ens_getHco(ensembleAnl)
 153      vco_ens => ens_getVco(ensembleAnl)
 154      nEns = ens_getNumMembers(ensembleAnl)
 155    end if
 156
 157    !- Setting default namelist variable values
 158    randomSeed            =  -999
 159    includeYearInSeed     = .false.
 160    writeSubSample        = .false.
 161    writeSubSampleUnPert  = .false.
 162    alphaRTPS             =  0.0D0
 163    alphaRTPP             =  0.0D0
 164    alphaRandomPert       =  0.0D0
 165    alphaRandomPertSubSample =  -1.0D0
 166    huLimitsBeforeRecenter = .true.
 167    imposeSaturationLimit = .false.
 168    imposeRttovHuLimits   = .false.
 169    weightRecenter(:)     = -1.0D0 ! means no specified values
 170    weightRecenterLand    = -1.0D0 ! means same recentering for land as other variables
 171    numMembersToRecenter  = -1     ! means all members recentered by default
 172    useOptionTableRecenter = .false.
 173    ! For the next 3 variables, the member number will be appended to this string
 174    ! for files '${trialdate}_006_${member}'
 175    etiket_trl = 'E27_0_0P'
 176    ! for files '${analdate}_000_${member}', 'subspace/${analdate}_000_${member}' and 'subspace_unpert/${analdate}_000_${member}'
 177    ! the member=0000 will contain the mean analysis
 178    etiket_anl = 'E27_0_0P'
 179    ! for files '${analdate}_000_inc_${member}' and 'subspace/${analdate}_000_inc_${member}'
 180    ! the member=0000 will contain the mean increment
 181    etiket_inc = 'E27_0_0P'
 182    !
 183    etiket_anlmean = 'E27_0_0PAVG' ! for file '${analdate}_000_analmean'
 184    etiket_anlrms = 'E27_0_0PRMS'  ! for file '${analdate}_000_analrms'
 185    etiket_anlmeanpert = 'E27_0_0PAVGP'  ! for file '${analdate}_000_analpertmean'
 186    etiket_anlrmspert = 'E27_0_0PRMSP'  ! for file '${analdate}_000_analpertrms'
 187    etiket_anlmean_raw = 'E27_0_0PAVGR'  ! for file '${analdate}_000_analmean_raw'
 188    etiket_anlrms_raw = 'E27_0_0PRMSR'  ! for file '${analdate}_000_analrms_raw'
 189    etiket_trlmean = 'E27_0_0PAVG' ! for file '${trialdate}_006_trialmean'
 190    etiket_trlrms = 'E27_0_0PRMS' ! for file '${trialdate}_006_trialrms'
 191    !
 192    numBits = 16
 193    useAnalIncMask = .false.
 194    writeRawAnalStats = .false.
 195    useMemberAsHuRefState = .false.
 196
 197    !- Read the namelist
 198    nulnam = 0
 199    ierr = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
 200    read(nulnam, nml=namEnsPostProcModule, iostat=ierr)
 201    if ( ierr /= 0) call utl_abort('epp_postProc: Error reading namelist')
 202    if ( mmpi_myid == 0 ) write(*,nml=namEnsPostProcModule)
 203    ierr = fclos(nulnam)
 204
 205    if (alphaRTPS < 0.0D0) alphaRTPS = 0.0D0
 206    if (alphaRTPP < 0.0D0) alphaRTPP = 0.0D0
 207    if (alphaRandomPert < 0.0D0) alphaRandomPert = 0.0D0
 208    if (alphaRandomPertSubSample < 0.0D0) alphaRandomPertSubSample = 0.0D0
 209    if (numMembersToRecenter == -1) numMembersToRecenter = nEns ! default behaviour
 210
 211    if (writeSubSample) then
 212      if (.not.(ens_isAllocated(ensembleTrl).and.ens_isAllocated(ensembleAnl))) then
 213        call utl_abort('epp_postProc: subSample can only be produced if both Anl and Trl ensembles available')
 214      end if
 215    end if
 216
 217    if (writeSubSampleUnPert) then
 218      if (.not.ens_isAllocated(ensembleAnl)) then
 219        call utl_abort('epp_postProc: subSampleUnPert can only be produced if Anl ensemble available')
 220      end if
 221    end if
 222
 223    if (writeRawAnalStats) then
 224      if (.not.ens_isAllocated(ensembleAnl)) then
 225        call utl_abort('epp_postProc: RawAnalStats can only be produced if Anl ensemble available')
 226      end if
 227    end if
 228
 229    if (ens_isAllocated(ensembleTrl)) then
 230      !- Allocate and compute ensemble mean Trl
 231      call gsv_allocate( stateVectorMeanTrl, tim_nstepobsinc, hco_ens, vco_ens, dateStamp_opt=tim_getDateStamp(),  &
 232                         mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
 233                         dataKind_opt=4, allocHeightSfc_opt=.true., &
 234                         hInterpolateDegree_opt = hInterpolationDegree, &
 235                         allocHeight_opt=.false., allocPressure_opt=.false. )
 236      call gsv_zero(stateVectorMeanTrl)
 237      call ens_computeMean(ensembleTrl)
 238      call ens_copyEnsMean(ensembleTrl, stateVectorMeanTrl)
 239      
 240      !- Allocate and compute ensemble spread stddev Trl
 241      call gsv_allocate( stateVectorStdDevTrl, tim_nstepobsinc, hco_ens, vco_ens, dateStamp_opt=tim_getDateStamp(),  &
 242                         mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
 243                         hInterpolateDegree_opt = hInterpolationDegree, &
 244                         dataKind_opt=4, allocHeight_opt=.false., allocPressure_opt=.false. )
 245      call gsv_zero(stateVectorStdDevTrl)
 246      call ens_computeStdDev(ensembleTrl)
 247      call ens_copyEnsStdDev(ensembleTrl, stateVectorStdDevTrl)
 248    end if
 249
 250    if (ens_isAllocated(ensembleAnl)) then
 251      !- Allocate and compute ensemble mean Anl
 252      call gsv_allocate( stateVectorMeanAnl, tim_nstepobsinc, hco_ens, vco_ens, dateStamp_opt=tim_getDateStamp(),  &
 253                         mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
 254                         dataKind_opt=4, allocHeightSfc_opt=.true., &
 255                         hInterpolateDegree_opt = hInterpolationDegree, &
 256                         allocHeight_opt=.false., allocPressure_opt=.false. )
 257      call gsv_zero(stateVectorMeanAnl)
 258      call ens_computeMean(ensembleAnl)
 259      call ens_copyEnsMean(ensembleAnl, stateVectorMeanAnl)
 260
 261      !- Allocate and compute ensemble spread stddev Anl
 262      call gsv_allocate( stateVectorStdDevAnl, tim_nstepobsinc, hco_ens, vco_ens, dateStamp_opt=tim_getDateStamp(),  &
 263                         mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
 264                         hInterpolateDegree_opt = hInterpolationDegree, &
 265                         dataKind_opt=4, allocHeight_opt=.false., allocPressure_opt=.false. )
 266      call gsv_zero(stateVectorStdDevAnl)
 267      call ens_computeStdDev(ensembleAnl)
 268      call ens_copyEnsStdDev(ensembleAnl, stateVectorStdDevAnl)
 269
 270      !- Keep the mean and standard deviation of the raw analysis for outputs, if requested
 271      if (writeRawAnalStats) then
 272        call gsv_allocate( stateVectorMeanAnlRaw, tim_nstepobsinc, hco_ens, vco_ens, dateStamp_opt=tim_getDateStamp(),  &
 273                           mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
 274                           dataKind_opt=4, allocHeightSfc_opt=.true., &
 275                           hInterpolateDegree_opt = hInterpolationDegree, &
 276                           allocHeight_opt=.false., allocPressure_opt=.false. )
 277        call gsv_allocate( stateVectorStdDevAnlRaw, tim_nstepobsinc, hco_ens, vco_ens, dateStamp_opt=tim_getDateStamp(),  &
 278                           mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
 279                           hInterpolateDegree_opt = hInterpolationDegree, &
 280                           dataKind_opt=4, allocHeight_opt=.false., allocPressure_opt=.false. )
 281        call gsv_copy(stateVectorMeanAnl, stateVectorMeanAnlRaw)
 282        call gsv_copy(stateVectorStdDevAnl, stateVectorStdDevAnlRaw)
 283      end if
 284
 285      if (ens_isAllocated(ensembleTrl)) then
 286        !- Apply RTPP, if requested
 287        if (alphaRTPP > 0.0D0) then
 288          call epp_RTPP(ensembleAnl, ensembleTrl, stateVectorMeanAnl, &
 289                        stateVectorMeanTrl, alphaRTPP)
 290          ! recompute the analysis spread stddev
 291          call ens_computeStdDev(ensembleAnl)
 292          call ens_copyEnsStdDev(ensembleAnl, stateVectorStdDevAnl)
 293        end if
 294
 295        !- Apply RTPS, if requested
 296        if (alphaRTPS > 0.0D0) then
 297          call epp_RTPS(ensembleAnl, stateVectorStdDevAnl, stateVectorStdDevTrl, &
 298                        stateVectorMeanAnl, alphaRTPS)
 299          ! recompute the analysis spread stddev
 300          call ens_computeStdDev(ensembleAnl)
 301          call ens_copyEnsStdDev(ensembleAnl, stateVectorStdDevAnl)
 302        end if
 303      end if
 304
 305      !- Impose limits on humidity *before* recentering, if requested
 306      if (huLimitsBeforeRecenter) then
 307        if (imposeSaturationLimit .or. imposeRttovHuLimits) then
 308          if (mmpi_myid == 0) write(*,*) ''
 309          if (mmpi_myid == 0) write(*,*) 'epp_postProcess: limits will be imposed on the humidity of analysis ensemble'
 310          if (mmpi_myid == 0 .and. imposeSaturationLimit ) write(*,*) '              -> Saturation Limit'
 311          if (mmpi_myid == 0 .and. imposeRttovHuLimits   ) write(*,*) '              -> Rttov Limit'
 312          if ( imposeSaturationLimit ) call qlim_saturationLimit(ensembleAnl)
 313          if ( imposeRttovHuLimits   ) call qlim_rttovLimit     (ensembleAnl)
 314          ! And recompute analysis mean
 315          call ens_computeMean(ensembleAnl)
 316          call ens_copyEnsMean(ensembleAnl, stateVectorMeanAnl)
 317        end if
 318      end if
 319
 320      !- Recenter analysis ensemble on supplied analysis
 321      if (any(weightRecenter(:) >= 0.0d0) .or. useOptionTableRecenter) then
 322        write(*,*) 'epp_postProcess: Recenter analyses on supplied analysis'
 323        call epp_hybridRecentering(ensembleAnl, weightRecenter, weightRecenterLand, &
 324                                   useOptionTableRecenter, numMembersToRecenter)
 325        ! And recompute analysis mean
 326        call ens_computeMean(ensembleAnl)
 327        call ens_copyEnsMean(ensembleAnl, stateVectorMeanAnl)
 328        ! And recompute the analysis spread stddev
 329        call ens_computeStdDev(ensembleAnl)
 330        call ens_copyEnsStdDev(ensembleAnl, stateVectorStdDevAnl)
 331      end if
 332
 333      !- Impose limits on humidity *after* recentering, if requested
 334      if (.not.huLimitsBeforeRecenter) then
 335        if (imposeSaturationLimit .or. imposeRttovHuLimits) then
 336          if (mmpi_myid == 0) write(*,*) ''
 337          if (mmpi_myid == 0) write(*,*) 'epp_postProcess: limits will be imposed on the humidity of analysis ensemble'
 338          if (mmpi_myid == 0 .and. imposeSaturationLimit ) write(*,*) '              -> Saturation Limit'
 339          if (mmpi_myid == 0 .and. imposeRttovHuLimits   ) write(*,*) '              -> Rttov Limit'
 340          if ( imposeSaturationLimit ) call qlim_saturationLimit(ensembleAnl)
 341          if ( imposeRttovHuLimits   ) call qlim_rttovLimit     (ensembleAnl)
 342          ! And recompute analysis mean
 343          call ens_computeMean(ensembleAnl)
 344          call ens_copyEnsMean(ensembleAnl, stateVectorMeanAnl)
 345        end if
 346      end if
 347
 348      !- If SubSample requested, copy sub-sample of analysis and trial members
 349      if (writeSubSample) then
 350        ! Copy sub-sampled analysis and trial ensemble members
 351        call epp_selectSubSample(ensembleAnl, ensembleAnlSubSample,  &
 352                                 ensembleTrl, ensembleTrlSubSample)
 353
 354        ! Create subdirectory for outputting sub sample increments
 355        ierr = clib_mkdir_r('subspace')
 356
 357        ! Allocate stateVectors to store and output sub-sampled ensemble mean analysis
 358        call gsv_allocate( stateVectorMeanAnlSubSample, tim_nstepobsinc,  &
 359                           hco_ens, vco_ens, dateStamp_opt=tim_getDateStamp(),  &
 360                           mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
 361                           dataKind_opt=4, allocHeightSfc_opt=.true., &
 362                           hInterpolateDegree_opt = hInterpolationDegree, &
 363                           allocHeight_opt=.false., allocPressure_opt=.false. )
 364        call gsv_zero(stateVectorMeanAnlSubSample)
 365
 366      end if
 367
 368      !- If unperturbed SubSample requested, copy sub-sample of analysis members
 369      if (writeSubSampleUnPert) then
 370        ! Copy sub-sampled analysis ensemble members
 371        call epp_selectSubSample(ensembleAnl, ensembleAnlSubSampleUnPert)
 372
 373        ! Create subdirectory for outputting sub sample members without perturbations
 374        ierr = clib_mkdir_r('subspace_unpert')
 375
 376      end if
 377
 378      !- Apply random additive inflation, if requested
 379      if (alphaRandomPert > 0.0D0) then
 380        ! If namelist value is -999, set random seed using the date (as in standard EnKF)
 381        if (randomSeed == -999) then
 382          imode = -3 ! stamp to printable date and time: YYYYMMDD, HHMMSShh
 383          dateStamp = tim_getDateStamp()
 384          ierr = newdate(dateStamp, datePrint, timePrint, imode)
 385          timePrint = timePrint/1000000
 386          datePrint =  datePrint*100 + timePrint
 387          if (includeYearInSeed) then
 388            ! Remove the century, keeping 2 digits of the year
 389            randomSeedRandomPert = datePrint - 100000000*(datePrint/100000000)
 390          else
 391            ! Remove the year and add 9
 392            randomSeedRandomPert = 9 + datePrint - 1000000*(datePrint/1000000)
 393          end if
 394        else
 395          randomSeedRandomPert = randomSeed
 396        end if
 397        write(*,*) 'epp_postProcess: randomSeed for additive inflation set to ', &
 398             randomSeedRandomPert
 399        if (ens_isAllocated(ensembleTrl)) then
 400          call epp_addRandomPert(ensembleAnl, stateVectorMeanTrl, alphaRandomPert, &
 401               randomSeedRandomPert, useMemberAsHuRefState)
 402        else
 403          call epp_addRandomPert(ensembleAnl, stateVectorMeanAnl, alphaRandomPert, &
 404               randomSeedRandomPert, useMemberAsHuRefState)
 405        end if
 406      end if
 407
 408      !- Recompute the analysis spread stddev after inflation and humidity limits
 409      call gsv_allocate( stateVectorStdDevAnlPert, tim_nstepobsinc, hco_ens, vco_ens, &
 410                         dateStamp_opt=tim_getDateStamp(),  &
 411                         mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
 412                         hInterpolateDegree_opt = hInterpolationDegree, &
 413                         dataKind_opt=4, allocHeight_opt=.false., &
 414                         allocPressure_opt=.false. )
 415      call ens_computeStdDev(ensembleAnl)
 416      call ens_copyEnsStdDev(ensembleAnl, stateVectorStdDevAnlPert)
 417
 418      !- If SubSample requested, do remaining processing and output of sub-sampled members
 419      if (writeSubSample) then
 420
 421        ! Apply random additive inflation to sub-sampled ensemble, if requested
 422        if (alphaRandomPertSubSample > 0.0D0) then
 423          ! If namelist value is -999, set random seed using the date (as in standard EnKF)
 424          if (randomSeed == -999) then
 425            imode = -3 ! stamp to printable date and time: YYYYMMDD, HHMMSShh
 426            dateStamp = tim_getDateStamp()
 427            ierr = newdate(dateStamp, datePrint, timePrint, imode)
 428            timePrint = timePrint/1000000
 429            datePrint =  datePrint*100 + timePrint
 430            if (includeYearInSeed) then
 431              ! Remove the century, keeping 2 digits of the year
 432              randomSeedRandomPert = datePrint - 100000000*(datePrint/100000000)
 433            else
 434              ! Remove the year and add 9
 435              randomSeedRandomPert = 9 + datePrint - 1000000*(datePrint/1000000)
 436            end if
 437          else
 438            randomSeedRandomPert = randomSeed
 439          end if
 440          write(*,*) 'epp_postProcess: randomSeed for additive inflation set to ', &
 441               randomSeedRandomPert
 442          call epp_addRandomPert(ensembleAnlSubSample, stateVectorMeanTrl,  &
 443                                 alphaRandomPertSubSample, randomSeedRandomPert, &
 444                                 useMemberAsHuRefState)
 445        end if
 446
 447        ! Compute analysis mean of sub-sampled ensemble
 448        call ens_computeMean(ensembleAnlSubSample)
 449
 450        ! Shift members to have same mean as full ensemble
 451        call ens_recenter(ensembleAnlSubSample, stateVectorMeanAnl,  &
 452                          recenteringCoeffScalar_opt=1.0D0)
 453
 454        ! Re-compute analysis mean of sub-sampled ensemble
 455        call ens_computeMean(ensembleAnlSubSample)
 456        call ens_copyEnsMean(ensembleAnlSubSample, stateVectorMeanAnlSubSample)
 457        
 458      end if ! writeSubsample
 459
 460      !- If SubSample requested, do remaining processing and output of sub-sampled members
 461      if (writeSubSampleUnPert) then
 462
 463        ! Compute analysis mean of sub-sampled ensemble
 464        call ens_computeMean(ensembleAnlSubSampleUnPert)
 465
 466        ! Shift members to have same mean as full ensemble
 467        call ens_recenter(ensembleAnlSubSampleUnPert, stateVectorMeanAnl,  &
 468                          recenteringCoeffScalar_opt=1.0D0)
 469
 470      end if
 471
 472    end if ! ens_isAllocated(ensembleAnl)
 473
 474    !
 475    !- Transform data before computing the increments and writing to files.
 476    !
 477    if (ens_isAllocated(ensembleAnl)) then
 478      call gvt_transform(ensembleAnl,'AllTransformedToModel',allowOverWrite_opt=.true.)
 479      call gvt_transform(stateVectorMeanAnl,'AllTransformedToModel',allowOverWrite_opt=.true.)
 480      if (writeSubSample) then
 481        call gvt_transform(ensembleAnlSubSample,'AllTransformedToModel',allowOverWrite_opt=.true.)
 482        call gvt_transform(stateVectorMeanAnlSubSample,'AllTransformedToModel',allowOverWrite_opt=.true.)
 483      end if
 484      if (writeSubSampleUnPert) then
 485        call gvt_transform(ensembleAnlSubSampleUnPert,'AllTransformedToModel',allowOverWrite_opt=.true.)
 486      end if
 487    end if
 488
 489    ! When we read ensemble trials we always need to transform them either for incremnets or for writing
 490    if (ens_isAllocated(ensembleTrl)) then
 491      call gvt_transform(ensembleTrl,'AllTransformedToModel',allowOverWrite_opt=.true.)
 492      call gvt_transform(stateVectorCtrlTrl,'AllTransformedToModel',allowOverWrite_opt=.true.)
 493      call gvt_transform(stateVectorMeanTrl,'AllTransformedToModel',allowOverWrite_opt=.true.)
 494      if (writeSubSample) then
 495        call gvt_transform(ensembleTrlSubSample,'AllTransformedToModel',allowOverWrite_opt=.true.)
 496      end if
 497    end if
 498
 499    !
 500    !- Compute increments
 501    !
 502    if (ens_isAllocated(ensembleAnl).and.ens_isAllocated(ensembleTrl)) then
 503
 504      !- Read the analysis mask (in LAM mode only) - N.B. different from land/sea mask!!!
 505      if (.not. hco_ens%global .and. useAnalIncMask) then
 506        call gio_getMaskLAM(stateVectorAnalIncMask, hco_ens, vco_ens, hInterpolationDegree)
 507      end if
 508
 509      ! initialize the vector for the ensemble increment
 510      allocate(dateStampListInc(tim_nstepobsinc))
 511      call tim_getstamplist(dateStampListInc,tim_nstepobsinc,tim_getDatestamp())
 512      nullify(varNames)
 513      call ens_varNamesList(varNames,ensembleTrl)
 514      call ens_allocate(ensembleAnlInc, nEns, tim_nstepobsinc, hco_ens, vco_ens, &
 515                        dateStampListInc, varNames_opt=varNames)
 516      call ens_copy(ensembleTrl,ensembleAnlInc)
 517      deallocate(varNames)
 518
 519      ! Compute the ensemble increments
 520      call ens_add(ensembleAnl, ensembleAnlInc, scaleFactorInOut_opt=-1.0D0)
 521      
 522      !- Mask the ensemble increments for LAM grid and recompute ensemble analysis
 523      if (.not. hco_ens%global .and. useAnalIncMask) then
 524        call ens_applyMaskLAM(ensembleAnlInc, stateVectorAnalIncMask)
 525        call ens_copy(ensembleAnlInc,ensembleAnl)
 526        call ens_add(ensembleTrl,ensembleAnl)
 527      end if
 528
 529      ! Compute mean increment for converted model variables (e.g. VIS and PR)
 530      nullify(varNames)
 531      call gsv_varNamesList(varNames, stateVectorMeanAnl)
 532      call gsv_allocate( stateVectorMeanInc, tim_nstepobsinc, hco_ens, vco_ens, &
 533                         dateStamp_opt=tim_getDateStamp(),  &
 534                         mpi_local_opt=.true., mpi_distribution_opt='Tiles',  &
 535                         hInterpolateDegree_opt = hInterpolationDegree, &
 536                         dataKind_opt=4, allocHeightSfc_opt=.true., varNames_opt=varNames )
 537      call gsv_copy(stateVectorMeanAnl, stateVectorMeanInc)
 538      deallocate(varNames)
 539
 540      call gsv_add(stateVectorCtrlTrl, stateVectorMeanInc, scaleFactor_opt=-1.0D0)
 541
 542      !- Mask the mean increment for LAM grid and recompute the mean analysis
 543      if (.not. hco_ens%global .and. useAnalIncMask) then
 544        call gsv_applyMaskLAM(stateVectorMeanInc, stateVectorAnalIncMask)
 545        call gsv_copy(stateVectorMeanInc,stateVectorMeanAnl)
 546        call gsv_add(stateVectorCtrlTrl,stateVectorMeanAnl)
 547      end if
 548
 549      ! If subsample reqested compute the subsample increments
 550      if (writeSubSample) then
 551        ! Compute ensemble increments for subsample
 552        nullify(varNames)
 553        call ens_varNamesList(varNames,ensembleAnlSubSample)
 554        call ens_allocate(ensembleAnlIncSubSample, ens_getNumMembers(ensembleAnlSubSample), &
 555                          tim_nstepobsinc, hco_ens, vco_ens, dateStampListInc, varNames_opt=varNames)
 556        call ens_copy(ensembleTrlSubSample,ensembleAnlIncSubSample)
 557        deallocate(varNames)
 558        
 559        call ens_add(ensembleAnlSubSample,ensembleAnlIncSubSample,scaleFactorInOut_opt=-1.0D0)
 560
 561        !- Mask the ensemble increment for LAM grid and recompute the mean analysis
 562        if (.not. hco_ens%global .and. useAnalIncMask) then
 563          call ens_applyMaskLAM(ensembleAnlIncSubSample, stateVectorAnalIncMask)
 564          call ens_copy(ensembleAnlIncSubSample,ensembleAnlSubSample)
 565          call ens_add(ensembleTrlSubSample,ensembleAnlSubSample)
 566        end if
 567
 568        ! Compute mean increment with respect to mean of full trial ensemble
 569        nullify(varNames)
 570        call gsv_varNamesList(varNames, stateVectorMeanAnlSubSample)
 571        call gsv_allocate( stateVectorMeanIncSubSample, tim_nstepobsinc,  &
 572                           hco_ens, vco_ens, dateStamp_opt=tim_getDateStamp(),  &
 573                           mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
 574                           dataKind_opt=4, allocHeightSfc_opt=.true., &
 575                           hInterpolateDegree_opt = hInterpolationDegree, &
 576!                           allocHeight_opt=.false., allocPressure_opt=.false., &
 577                           varNames_opt=varNames )
 578        call gsv_copy(stateVectorMeanAnlSubSample, stateVectorMeanIncSubSample)
 579        deallocate(varNames)
 580
 581        call gsv_add(stateVectorCtrlTrl, stateVectorMeanIncSubSample, scaleFactor_opt=-1.0D0)
 582
 583        !- Mask the mean increment for LAM grid and recompute the mean analysis
 584        if (.not. hco_ens%global .and. useAnalIncMask) then
 585          call gsv_applyMaskLAM(stateVectorMeanIncSubSample, stateVectorAnalIncMask)
 586          call gsv_copy(stateVectorMeanIncSubSample,stateVectorMeanAnlSubSample)
 587          call gsv_add(stateVectorCtrlTrl,stateVectorMeanAnlSubSample)
 588        end if
 589      end if
 590
 591    end if !- end of increment computation
 592
 593    !
 594    !- Output everything
 595    !
 596
 597    !- Output ens stddev and mean in trialrms, analrms and analpertrms files
 598
 599    ! determine middle timestep for output of these files
 600    middleStepIndex = (tim_nstepobsinc + 1) / 2
 601
 602    if (ens_isAllocated(ensembleTrl)) then
 603      ! output trialmean, trialrms
 604      call utl_tmg_start(5,'--WriteEnsMeanRms')
 605      call fln_ensTrlFileName(outFileName, '.', tim_getDateStamp())
 606      outFileName = trim(outFileName) // '_trialmean'
 607      call ens_copyMaskToGsv(ensembleTrl, stateVectorMeanTrl)
 608      do stepIndex = 1, tim_nstepobsinc
 609        call gio_writeToFile(stateVectorMeanTrl, outFileName, etiket_trlmean,  &
 610                             typvar_opt='P', writeHeightSfc_opt=.false., numBits_opt=numBits,  &
 611                             stepIndex_opt=stepIndex, containsFullField_opt=.true.)
 612      end do

 614      outFileName = trim(outFileName) // '_trialrms'
 615      call ens_copyMaskToGsv(ensembleTrl, stateVectorStdDevTrl)
 616      call gio_writeToFile(stateVectorStdDevTrl, outFileName, etiket_trlrms,  &
 617                           typvar_opt='P', writeHeightSfc_opt=.false., numBits_opt=numBits, &
 618                           stepIndex_opt=middleStepIndex, containsFullField_opt=.false.)
 619      outFileName = trim(outFileName) // '_ascii'
 620      call epp_printRmsStats(stateVectorStdDevTrl,outFileName,elapsed=0.0D0,ftype='F',nEns=nEns)
 621      call utl_tmg_stop(5)
 622
 623      ! output the trial ensemble if requested (because it was interpolated)
 624      if (writeTrlEnsemble) then
 625        call utl_tmg_start(3,'--WriteEnsemble')
 626        if (.not. outputOnlyEnsMean) then
 627          call ens_writeEnsemble(ensembleTrl, '.', '', etiket_trl, 'P',  &
 628                                 numBits_opt=16, etiketAppendMemberNumber_opt=.true.,  &
 629                                 containsFullField_opt=.true.)
 630        end if
 631        call utl_tmg_stop(3)
 632      end if
 633    end if
 634
 635    ! all outputs related to analysis ensemble
 636    if (ens_isAllocated(ensembleAnl)) then
 637
 638      !- Prepare stateVector with only MeanAnl surface pressure and surface height
 639      if (gsv_isAllocated(stateVectorHeightSfc)) then
 640        call gsv_allocate( stateVectorMeanAnlSfcPres, tim_nstepobsinc, hco_ens, vco_ens,   &
 641                           dateStamp_opt=tim_getDateStamp(),  &
 642                           mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
 643                           hInterpolateDegree_opt = hInterpolationDegree, &
 644                           dataKind_opt=4, allocHeightSfc_opt=.true., varNames_opt=(/'P0'/) )
 645        call gsv_zero(stateVectorMeanAnlSfcPres)
 646        if (mmpi_myid <= (nEns-1)) then
 647          call gsv_allocate( stateVectorMeanAnlSfcPresMpiGlb, tim_nstepobsinc, hco_ens, vco_ens,   &
 648                             dateStamp_opt=tim_getDateStamp(),  &
 649                             mpi_local_opt=.false., &
 650                             hInterpolateDegree_opt = hInterpolationDegree, &
 651                             dataKind_opt=4, allocHeightSfc_opt=.true., varNames_opt=(/'P0'/) )
 652          call gsv_zero(stateVectorMeanAnlSfcPresMpiGlb)
 653        end if
 654        call gsv_copy(stateVectorMeanAnl, stateVectorMeanAnlSfcPres, allowVarMismatch_opt=.true.)
 655        call gsv_copyHeightSfc(stateVectorHeightSfc, stateVectorMeanAnlSfcPres)
 656        call gsv_transposeTilesToMpiGlobal(stateVectorMeanAnlSfcPresMpiGlb, stateVectorMeanAnlSfcPres)
 657      end if
 658      
 659      ! output analmean, analrms
 660      call utl_tmg_start(5,'--WriteEnsMeanRms')
 661      call fln_ensAnlFileName(outFileName, '.', tim_getDateStamp())
 662      outFileName = trim(outFileName) // '_analmean'
 663      call ens_copyMaskToGsv(ensembleAnl, stateVectorMeanAnl)
 664      do stepIndex = 1, tim_nstepobsinc
 665        call gio_writeToFile(stateVectorMeanAnl, outFileName, etiket_anlmean,  &
 666                             typvar_opt='A', writeHeightSfc_opt=.false., numBits_opt=numBits, &
 667                             stepIndex_opt=stepIndex, containsFullField_opt=.true.)
 668      end do
 669      call fln_ensAnlFileName(outFileName, '.', tim_getDateStamp())
 670      outFileName = trim(outFileName) // '_analrms'
 671      call ens_copyMaskToGsv(ensembleAnl, stateVectorStdDevAnl)
 672      call gio_writeToFile(stateVectorStdDevAnl, outFileName, etiket_anlrms,  &
 673                           typvar_opt='A', writeHeightSfc_opt=.false., numBits_opt=numBits, &
 674                           stepIndex_opt=middleStepIndex, containsFullField_opt=.false.)
 675      outFileName = trim(outFileName) // '_ascii'
 676      call epp_printRmsStats(stateVectorStdDevAnl,outFileName,elapsed=0.0D0,ftype='A',nEns=nEns)
 677
 678      ! output analmean_raw and analrms_raw, if requested
 679      if (writeRawAnalStats) then
 680        call fln_ensAnlFileName(outFileName, '.', tim_getDateStamp())
 681        outFileName = trim(outFileName) // '_analmean_raw'
 682        call ens_copyMaskToGsv(ensembleAnl, stateVectorMeanAnlRaw)
 683        do stepIndex = 1, tim_nstepobsinc
 684          call gio_writeToFile(stateVectorMeanAnlRaw, outFileName, etiket_anlmean_raw,  &
 685                               typvar_opt='A', writeHeightSfc_opt=.false., numBits_opt=numBits, &
 686                               stepIndex_opt=stepIndex, containsFullField_opt=.true.)
 687        end do
 688        call fln_ensAnlFileName(outFileName, '.', tim_getDateStamp())
 689        outFileName = trim(outFileName) // '_analrms_raw'
 690        call ens_copyMaskToGsv(ensembleAnl, stateVectorStdDevAnl)
 691        call gio_writeToFile(stateVectorStdDevAnlRaw, outFileName, etiket_anlrms_raw,  &
 692                             typvar_opt='A', writeHeightSfc_opt=.false., numBits_opt=numBits, &
 693                             stepIndex_opt=middleStepIndex, containsFullField_opt=.false.)
 694        outFileName = trim(outFileName) // '_ascii'
 695        call epp_printRmsStats(stateVectorStdDevAnlRaw,outFileName,elapsed=0.0D0,ftype='A',nEns=nEns)
 696      end if  ! writeRawAnalStats
 697
 698      if (alphaRandomPert > 0.0D0) then
 699        ! output analpertmean, analpertrms
 700        call fln_ensAnlFileName( outFileName, '.', tim_getDateStamp() )
 701        outFileName = trim(outFileName) // '_analpertmean'
 702        call ens_copyMaskToGsv(ensembleAnl, stateVectorMeanAnl)
 703        do stepIndex = 1, tim_nstepobsinc
 704          call gio_writeToFile(stateVectorMeanAnl, outFileName, etiket_anlmeanpert,  &
 705                               typvar_opt='A', writeHeightSfc_opt=.false., numBits_opt=numBits, &
 706                               stepIndex_opt=stepIndex, containsFullField_opt=.true.)
 707        end do
 708        call fln_ensAnlFileName( outFileName, '.', tim_getDateStamp() )
 709        outFileName = trim(outFileName) // '_analpertrms'
 710        call ens_copyMaskToGsv(ensembleAnl, stateVectorStdDevAnlPert)
 711        call gio_writeToFile(stateVectorStdDevAnlPert, outFileName, etiket_anlrmspert,  &
 712                             typvar_opt='A', writeHeightSfc_opt=.false., numBits_opt=numBits, &
 713                             stepIndex_opt=middleStepIndex, containsFullField_opt=.false.)
 714        outFileName = trim(outFileName) // '_ascii'
 715        call epp_printRmsStats(stateVectorStdDevAnlPert,outFileName,elapsed=0.0D0,ftype='P',nEns=nEns)
 716      end if
 717      call utl_tmg_stop(5)
 718
 719      !- Output the ensemble mean increment (include MeanAnl Psfc) and ensemble increments
 720      if (ens_isAllocated(ensembleTrl)) then
 721        call utl_tmg_start(5,'--WriteEnsMeanRms')
 722        ! output ensemble mean increment
 723        call fln_ensAnlFileName( outFileName, '.', tim_getDateStamp(), 0, ensFileNameSuffix_opt='inc' )
 724        call ens_copyMaskToGsv(ensembleAnl, stateVectorMeanInc)
 725        ! here we assume 4 digits for the ensemble member!!!!
 726        etiket = trim(etiket_inc) // '0000'
 727        do stepIndex = 1, tim_nstepobsinc
 728          call gio_writeToFile(stateVectorMeanInc, outFileName, etiket,  &
 729                               typvar_opt='R', writeHeightSfc_opt=.false., numBits_opt=numBits, &
 730                               stepIndex_opt=stepIndex, containsFullField_opt=.false.)
 731          if (gsv_isAllocated(stateVectorMeanAnlSfcPres)) then
 732            call gio_writeToFile(stateVectorMeanAnlSfcPres, outFileName, etiket,  &
 733                                 typvar_opt='A', writeHeightSfc_opt=.true., &
 734                                 stepIndex_opt=stepIndex, containsFullField_opt=.true.)
 735          end if
 736        end do
 737        call utl_tmg_stop(5)
 738
 739        !- Output all ensemble member increments
 740        call utl_tmg_start(3,'--WriteEnsemble')
 741        if (.not. outputOnlyEnsMean) then
 742          call ens_writeEnsemble(ensembleAnlInc, '.', '', etiket_inc, 'R',  &
 743                                 numBits_opt=16, etiketAppendMemberNumber_opt=.true.,  &
 744                                 containsFullField_opt=.false., resetTimeParams_opt=.true.)
 745          if (gsv_isAllocated(stateVectorMeanAnlSfcPresMpiGlb)) then
 746            ! Also write the reference (analysis) surface pressure to increment files
 747            call epp_writeToAllMembers(stateVectorMeanAnlSfcPresMpiGlb, nEns,  &
 748                                       etiket='ENS_INC', typvar='A', fileNameSuffix='inc',  &
 749                                       ensPath='.')
 750          end if
 751        end if
 752        call utl_tmg_stop(3)
 753
 754      end if ! allocated(ensembleTrl)
 755
 756      ! output ensemble mean analysis state
 757      call utl_tmg_start(5,'--WriteEnsMeanRms')
 758      call fln_ensAnlFileName( outFileName, '.', tim_getDateStamp(), 0 )
 759      call ens_copyMaskToGsv(ensembleAnl, stateVectorMeanAnl)
 760      ! here we assume 4 digits for the ensemble member!!!!
 761      etiket = trim(etiket_anl) // '0000'
 762      do stepIndex = 1, tim_nstepobsinc
 763        call gio_writeToFile(stateVectorMeanAnl, outFileName, etiket,  &
 764                             typvar_opt='A', writeHeightSfc_opt=.false., numBits_opt=numBits, &
 765                             stepIndex_opt=stepIndex, containsFullField_opt=.true.)
 766      end do
 767      call utl_tmg_stop(5)
 768
 769      !- Output all ensemble member analyses
 770      ! convert transformed to model variables for analysis and trial ensembles
 771      call utl_tmg_start(3,'--WriteEnsemble')
 772      if (.not. outputOnlyEnsMean) then
 773        call ens_writeEnsemble(ensembleAnl, '.', '', etiket_anl, 'A',  &
 774                               numBits_opt=16, etiketAppendMemberNumber_opt=.true.,  &
 775                               containsFullField_opt=.true.)
 776      end if
 777      call utl_tmg_stop(3)
 778
 779      !- Output the sub-sampled ensemble analyses and increments
 780      if (writeSubSample) then
 781        call utl_tmg_start(5,'--WriteEnsMeanRms')
 782        ! Output the ensemble mean increment (include MeanAnl Psfc)
 783        call fln_ensAnlFileName( outFileName, 'subspace', tim_getDateStamp(), 0, ensFileNameSuffix_opt='inc' )
 784        ! here we assume 4 digits for the ensemble member!!!!
 785        etiket = trim(etiket_inc) // '0000'
 786        do stepIndex = 1, tim_nstepobsinc
 787          call gio_writeToFile(stateVectorMeanIncSubSample, outFileName, etiket,  &
 788                               typvar_opt='R', writeHeightSfc_opt=.false., numBits_opt=numBits, &
 789                               stepIndex_opt=stepIndex, containsFullField_opt=.false.)
 790          if (gsv_isAllocated(stateVectorMeanAnlSfcPres)) then
 791            call gio_writeToFile(stateVectorMeanAnlSfcPres, outFileName, etiket,  &
 792                                 typvar_opt='A', writeHeightSfc_opt=.true., &
 793                                 stepIndex_opt=stepIndex, containsFullField_opt=.true.)
 794          end if
 795        end do
 796
 797        ! Output the ensemble mean analysis state
 798        call fln_ensAnlFileName( outFileName, 'subspace', tim_getDateStamp(), 0 )
 799        ! here we assume 4 digits for the ensemble member!!!!
 800        etiket = trim(etiket_anl) // '0000'
 801        do stepIndex = 1, tim_nstepobsinc
 802          call gio_writeToFile(stateVectorMeanAnlSubSample, outFileName, etiket,  &
 803                               typvar_opt='A', writeHeightSfc_opt=.false., numBits_opt=numBits, &
 804                               stepIndex_opt=stepIndex, containsFullField_opt=.true.)
 805        end do
 806        call utl_tmg_stop(5)
 807
 808        ! Output the sub-sampled analysis ensemble members
 809        call utl_tmg_start(3,'--WriteEnsemble')
 810        if (.not. outputOnlyEnsMean) then
 811          call ens_writeEnsemble(ensembleAnlSubSample, 'subspace', '', etiket_anl, 'A',  &
 812                                 numBits_opt=16, etiketAppendMemberNumber_opt=.true.,  &
 813                                 containsFullField_opt=.true.)
 814        end if
 815        call utl_tmg_stop(3)
 816
 817        ! Output the sub-sampled ensemble increments (include MeanAnl Psfc)
 818        call utl_tmg_start(3,'--WriteEnsemble')
 819        if (.not. outputOnlyEnsMean) then
 820          call ens_writeEnsemble(ensembleAnlIncSubSample, 'subspace', '', etiket_inc, 'R',  &
 821                                 numBits_opt=16, etiketAppendMemberNumber_opt=.true.,  &
 822                                 containsFullField_opt=.false., resetTimeParams_opt=.true.)
 823          ! Also write the reference (analysis) surface pressure to increment files
 824          if (gsv_isAllocated(stateVectorMeanAnlSfcPresMpiGlb)) then
 825            call epp_writeToAllMembers(stateVectorMeanAnlSfcPresMpiGlb,  &
 826                                       ens_getNumMembers(ensembleAnlSubSample),  &
 827                                       etiket=etiket_inc, typvar='A', fileNameSuffix='inc',  &
 828                                       ensPath='subspace')
 829          end if
 830        end if
 831        call utl_tmg_stop(3)
 832
 833      end if ! writeSubSample
 834
 835      !- Output the unperturbed sub-sampled ensemble analyses
 836      if (writeSubSampleUnPert) then
 837
 838        ! Output the sub-sampled analysis ensemble members
 839        call utl_tmg_start(3,'--WriteEnsemble')
 840        if (.not. outputOnlyEnsMean) then
 841          call ens_writeEnsemble(ensembleAnlSubSampleUnPert, 'subspace_unpert', '', etiket_anl, 'A',  &
 842                                 numBits_opt=16, etiketAppendMemberNumber_opt=.true.,  &
 843                                 containsFullField_opt=.true.)
 844        end if
 845        call utl_tmg_stop(3)
 846
 847      end if
 848
 849    end if ! ens_isAllocated(ensembleAnl)
 850
 851  end subroutine epp_postProcess

 857                                    fileNameSuffix, ensPath)
 858    !
 859    !:Purpose:   Write the contents of the supplied stateVector to all
 860    !            ensemble member files in an efficient parallel way.
 861    !
 862    implicit none
 863
 864    ! arguments:
 865    type(struct_gsv) :: stateVector
 866    integer          :: nEns
 867    character(len=*) :: etiket
 868    character(len=*) :: typvar
 869    character(len=*) :: fileNameSuffix
 870    character(len=*) :: ensPath
 871
 872    ! locals:
 873    integer            :: memberIndex, stepIndex, writeFilePE(nEns)
 874    character(len=4)   :: memberIndexStr
 875    character(len=256) :: outFileName

 878      writeFilePE(memberIndex) = mod(memberIndex-1, mmpi_nprocs)

 881    do memberIndex = 1, nEns
 882
 883      if (mmpi_myid == writeFilePE(memberIndex)) then
 884
 885        call fln_ensAnlFileName( outFileName, ensPath, tim_getDateStamp(),  &
 886                                 memberIndex, ensFileNameSuffix_opt=fileNameSuffix )
 887        write(memberIndexStr,'(I4.4)') memberIndex
 888
 889        do stepIndex = 1, tim_nstepobsinc
 890          call gio_writeToFile(stateVector, outFileName,  &
 891                               trim(etiket) // memberIndexStr,  &
 892                               typvar_opt=trim(typvar), writeHeightSfc_opt=.true., &
 893                               stepIndex_opt=stepIndex, containsFullField_opt=.true., &
 894                               numBits_opt=16)
 895        end do
 896
 897      end if
 898
 899    end do
 900
 901  end subroutine epp_writeToAllMembers

 907                      stateVectorMeanAnl, alphaRTPS)
 908    !:Purpose: Apply Relaxation To Prior Spread ensemble inflation according
 909    !           to the factor alphaRTPS (usually between 0 and 1).
 910    implicit none
 911
 912    ! Arguments
 913    type(struct_ens) :: ensembleAnl
 914    type(struct_gsv) :: stateVectorStdDevAnl
 915    type(struct_gsv) :: stateVectorStdDevTrl
 916    type(struct_gsv) :: stateVectorMeanAnl
 917    real(8)          :: alphaRTPS
 918
 919    ! Locals
 920    integer :: varLevIndex, latIndex, lonIndex, stepIndex, memberIndex
 921    integer :: nEns, numVarLev, myLonBeg, myLonEnd, myLatBeg, myLatEnd
 922    real(8) :: factorRTPS
 923    real(4), pointer     :: stdDevTrl_ptr_r4(:,:,:,:), stdDevAnl_ptr_r4(:,:,:,:)
 924    real(4), pointer     :: meanAnl_ptr_r4(:,:,:,:), memberAnl_ptr_r4(:,:,:,:)
 925
 926    write(*,*) 'epp_RTPS: Starting'
 927
 928    call gsv_getField(stateVectorStdDevTrl,stdDevTrl_ptr_r4)
 929    call gsv_getField(stateVectorStdDevAnl,stdDevAnl_ptr_r4)
 930    call gsv_getField(stateVectorMeanAnl,meanAnl_ptr_r4)
 931
 932    nEns = ens_getNumMembers(ensembleAnl)
 933    numVarLev = ens_getNumK(ensembleAnl)
 934    call ens_getLatLonBounds(ensembleAnl, myLonBeg, myLonEnd, myLatBeg, myLatEnd)
 935    do varLevIndex = 1, numVarLev
 936      memberAnl_ptr_r4 => ens_getOneLev_r4(ensembleAnl,varLevIndex)
 937      do latIndex = myLatBeg, myLatEnd
 938        do lonIndex = myLonBeg, myLonEnd
 939          do stepIndex = 1, tim_nstepobsinc
 940            ! compute the inflation factor for RTPS
 941            if ( stdDevAnl_ptr_r4(lonIndex,latIndex,varLevIndex,stepIndex) > 0.0 ) then
 942              factorRTPS = 1.0D0 + alphaRTPS *  &
 943                           ( stdDevTrl_ptr_r4(lonIndex,latIndex,varLevIndex,stepIndex) -  &
 944                             stdDevAnl_ptr_r4(lonIndex,latIndex,varLevIndex,stepIndex) ) /  &
 945                           stdDevAnl_ptr_r4(lonIndex,latIndex,varLevIndex,stepIndex)
 946            else
 947              factorRTPS = 0.0D0
 948            end if
 949            ! apply the inflation factor to all Anl members (in place)
 950            if (factorRTPS > 0.0D0) then
 951              do memberIndex = 1, nEns
 952                memberAnl_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex) =  &
 953                     factorRTPS *  &
 954                     ( memberAnl_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex) -  &
 955                       meanAnl_ptr_r4(lonIndex,latIndex,varLevIndex,stepIndex) ) +  &
 956                     meanAnl_ptr_r4(lonIndex,latIndex,varLevIndex,stepIndex)
 957              end do ! memberIndex
 958            end if ! factorRTPS > 0
 959          end do ! stepIndex
 960        end do ! lonIndex
 961      end do ! latIndex
 962    end do ! varLevIndex
 963
 964    write(*,*) 'epp_RTPS: Finished'
 965
 966  end subroutine epp_RTPS

 970  !--------------------------------------------------------------------------
 971  subroutine epp_RTPP(ensembleAnl, ensembleTrl, stateVectorMeanAnl, &
 972                      stateVectorMeanTrl, alphaRTPP)
 973    !:Purpose: Apply Relaxation To Prior Perturbation ensemble inflation according
 974    !           to the factor alphaRTPP (usually between 0 and 1).
 975    implicit none
 976
 977    ! Arguments
 978    type(struct_ens) :: ensembleAnl
 979    type(struct_ens) :: ensembleTrl
 980    type(struct_gsv) :: stateVectorMeanAnl
 981    type(struct_gsv) :: stateVectorMeanTrl
 982    real(8)          :: alphaRTPP
 983
 984    ! Locals
 985    integer :: varLevIndex, latIndex, lonIndex, stepIndex, memberIndex
 986    integer :: nEns, numVarLev, myLonBeg, myLonEnd, myLatBeg, myLatEnd
 987    real(4), pointer     :: meanAnl_ptr_r4(:,:,:,:), meanTrl_ptr_r4(:,:,:,:)
 988    real(4), pointer     :: memberAnl_ptr_r4(:,:,:,:), memberTrl_ptr_r4(:,:,:,:)
 989
 990    write(*,*) 'epp_RTPP: Starting'
 991
 992    call gsv_getField(stateVectorMeanAnl, meanAnl_ptr_r4)
 993    call gsv_getField(stateVectorMeanTrl, meanTrl_ptr_r4)
 994
 995    nEns = ens_getNumMembers(ensembleAnl)
 996    numVarLev = ens_getNumK(ensembleAnl)
 997    call ens_getLatLonBounds(ensembleAnl, myLonBeg, myLonEnd, myLatBeg, myLatEnd)
 998
 999    do varLevIndex = 1, numVarLev
1000      memberAnl_ptr_r4 => ens_getOneLev_r4(ensembleAnl,varLevIndex)
1001      memberTrl_ptr_r4 => ens_getOneLev_r4(ensembleTrl,varLevIndex)
1002      do latIndex = myLatBeg, myLatEnd
1003        do lonIndex = myLonBeg, myLonEnd
1004          do stepIndex = 1, tim_nstepobsinc
1005            ! apply RTPP to all Anl members (in place)
1006            ! member_anl = mean_anl + (1-a)*(member_anl-mean_anl) + a*(member_trl-mean_trl)
1007            do memberIndex = 1, nEns
1008              memberAnl_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex) =  &
1009                   meanAnl_ptr_r4(lonIndex,latIndex,varLevIndex,stepIndex) + &
1010                   (1.0D0 - alphaRTPP) * &
1011                   ( memberAnl_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex) -  &
1012                     meanAnl_ptr_r4(lonIndex,latIndex,varLevIndex,stepIndex) ) +  &
1013                   alphaRTPP * &
1014                   ( memberTrl_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex) -  &
1015                     meanTrl_ptr_r4(lonIndex,latIndex,varLevIndex,stepIndex) )
1016            end do ! memberIndex
1017          end do ! stepIndex
1018        end do ! lonIndex
1019      end do ! latIndex
1020    end do ! varLevIndex
1021
1022    write(*,*) 'epp_RTPP: Finished'
1023
1024  end subroutine epp_RTPP

1029  subroutine epp_addRandomPert(ensembleAnl, stateVectorRefState, alphaRandomPert, &
1030                               randomSeed, useMemberAsHuRefState)
1031    !:Purpose: Apply additive inflation using random perturbations from sampling
1032    !           the B matrix as defined by the regular namelist block NAMBHI, NAMBEN, etc.
1033    !           The scale factor alphaRandomPert (usually between 0 and 1) is used
1034    !           to simply multiply the resulting perturbations before adding to the original
1035    !           members. The perturbations have zero ensemble mean.
1036    implicit none
1037
1038    ! Arguments
1039    type(struct_ens), intent(inout) :: ensembleAnl
1040    type(struct_gsv), intent(in)    :: stateVectorRefState
1041    real(8)         , intent(in)    :: alphaRandomPert
1042    integer         , intent(in)    :: randomSeed
1043    logical         , intent(in)    :: useMemberAsHuRefState
1044
1045    ! Locals
1046    type(struct_gsv)         :: stateVectorPerturbation
1047    type(struct_gsv)         :: stateVectorPerturbationInterp
1048    type(struct_gsv)         :: stateVectorHuRefState
1049    type(struct_gsv)         :: stateVectorHuRefStateInterp
1050    type(struct_gsv)         :: stateVectorP0Ref
1051    type(struct_vco), pointer :: vco_randomPert, vco_ens
1052    type(struct_hco), pointer :: hco_randomPert, hco_ens, hco_core
1053    character(len=12)  :: etiket
1054    real(8), allocatable :: controlVector_mpiglobal(:), controlVector(:)
1055    real(8), allocatable :: perturbationMean(:,:,:)
1056    real(8), pointer     :: PsfcRef(:,:,:,:)
1057    real(8), pointer     :: perturbation_ptr(:,:,:)
1058    real(4), pointer     :: memberAnl_ptr_r4(:,:,:,:)
1059    integer :: cvIndex, memberIndex, varLevIndex, lonIndex, latIndex, stepIndex
1060    integer :: nEns, numVarLev, myLonBeg, myLonEnd, myLatBeg, myLatEnd, varIndex
1061    integer :: middleStepIndex
1062    logical, save :: firstCall = .true.
1063    character(len=4), pointer :: varNamesWithLQ(:)
1064    character(len=4) :: varName
1065
1066    call utl_tmg_start(4,'--AddEnsRandomPert')
1067
1068    ! Determine middle timestep
1069    middleStepIndex = (tim_nstepobsinc + 1) / 2
1070
1071    ! Get ensemble dimensions
1072    nEns = ens_getNumMembers(ensembleAnl)
1073    numVarLev = ens_getNumK(ensembleAnl)
1074    call ens_getLatLonBounds(ensembleAnl, myLonBeg, myLonEnd, myLatBeg, myLatEnd)
1075    vco_ens => ens_getVco(ensembleAnl)
1076    hco_ens => ens_getHco(ensembleAnl)
1077
1078    ! Define the horiz/vertical coordinate for perturbation calculation
1079    nullify(vco_randomPert)
1080    nullify(hco_randomPert)
1081    call hco_setupFromFile(hco_randomPert, './analysisgrid', 'ANALYSIS', 'Analysis' ) ! IN
1082    if ( hco_randomPert % global ) then
1083      etiket = 'BGCK_STDDEV'
1084    else
1085      etiket = 'STDDEV'
1086    end if
1087    call vco_setupFromFile(vco_randomPert, './bgcov', etiket)
1088
1089    ! Special modification to vco_randomPert if a soil variables is present
1090    if ( vco_getNumLev(vco_ens,'OT','I0') > 0 .or. &
1091         vco_getNumLev(vco_ens,'OT','I1') > 0 ) then
1092      write(*,*)
1093      write(*,*) 'epp_addRandomPert: copying number of levels for land surface variables!' 
1094      write(*,*)
1095      vco_randomPert%nlev_Other(:) = vco_ens%nlev_Other(:)
1096    end if
1097
1098    ! Get list of variable names in the ensemble and modify if necessary
1099    nullify(varNamesWithLQ)
1100    call ens_varNamesList(varNamesWithLQ,ensembleAnl)
1101    if (useMemberAsHuRefState .and. ens_varExist(ensembleAnl,'HU')) then
1102      ! Replace HU with LQ so we can apply the transform directly in this routine
1103      do varIndex = 1, size(varNamesWithLQ)
1104        if (varNamesWithLQ(varIndex) == 'HU') varNamesWithLQ(varIndex) = 'LQ'
1105      end do
1106      if (mmpi_myid == 0) write(*,*) 'epp_addRandomPert: varNamesWithLQ = ', varNamesWithLQ(:)
1107    end if
1108
1109    hco_core => hco_randomPert
1110    if (firstCall) then
1111      call utl_tmg_stop(4) ! stop counter, since Bmat has it's own counters
1112      call bmat_setup(hco_randomPert, hco_core, vco_randomPert)
1113      firstCall = .false.
1114      call utl_tmg_start(4,'--AddEnsRandomPert')
1115    end if
1116    call gvt_setup(hco_randomPert, hco_core, vco_randomPert)
1117
1118    call rng_setup(abs(randomSeed))
1119
1120    allocate(controlVector(cvm_nvadim))
1121    allocate(controlVector_mpiglobal(cvm_nvadim_mpiglobal))
1122    allocate(perturbationMean(myLonBeg:myLonEnd,myLatBeg:myLatEnd,numVarLev))
1123    perturbationMean(:,:,:) = 0.0d0
1124
1125    ! NOTE: The following stateVectors will include LQ instead of HU when
1126    !       useMemberAsHuRefState=true. This results in the perturbations
1127    !       being provided by bmat_sqrtB in terms of LQ, allowing the conversion
1128    !       from LQ to HU to be done within this subroutine using the HU field
1129    !       of each ensemble member.
1130    call gsv_allocate(stateVectorPerturbation, 1, hco_randomPert, vco_randomPert, &
1131                      dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true., &
1132                      hInterpolateDegree_opt='LINEAR', &
1133                      varNames_opt=varNamesWithLQ)
1134    call gsv_allocate(stateVectorPerturbationInterp, 1, hco_ens, vco_ens, &
1135                      dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true., &
1136                      hInterpolateDegree_opt='LINEAR', &
1137                      varNames_opt=varNamesWithLQ)
1138    call gsv_getField(stateVectorPerturbationInterp,perturbation_ptr)
1139
1140    ! allocate a minimal reference P0
1141    call gsv_allocate(statevectorP0Ref, 1, hco_ens, vco_randomPert, &
1142                      dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true., &
1143                      hInterpolateDegree_opt='LINEAR', &
1144                      varNames_opt=(/'P0'/))
1145    call gsv_getField(statevectorP0Ref, PsfcRef, 'P0')
1146    PsfcRef(:,:,:,:) = 100000.0D0
1147
1148
1149    ! prepare the reference state HU field for transforming LQ to HU perturbations
1150    if (ens_varExist(ensembleAnl,'HU')) then
1151      call gsv_allocate(stateVectorHuRefState, 1, hco_ens, vco_ens,   &
1152                        dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true., &
1153                        allocHeightSfc_opt=.true., hInterpolateDegree_opt='LINEAR', &
1154                        hExtrapolateDegree_opt='MINIMUM', &
1155                        varNames_opt=(/'HU','P0'/) )
1156      if (.not. useMemberAsHuRefState) then
1157        ! use the single provided state as the reference state and interpolate to perturbation grid
1158        call gsv_copy(stateVectorRefState, stateVectorHuRefState, allowVarMismatch_opt=.true.)
1159        call gsv_allocate(stateVectorHuRefStateInterp, 1, hco_randomPert, vco_randomPert,   &
1160                          dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true., &
1161                          allocHeightSfc_opt=.true., hInterpolateDegree_opt='LINEAR', &
1162                          hExtrapolateDegree_opt='MINIMUM', &
1163                          varNames_opt=(/'HU','P0'/) )
1164        call int_interp_gsv(stateVectorHuRefState, stateVectorHuRefStateInterp)
1165        call gsv_deallocate(stateVectorHuRefState)
1166      end if
1167    end if
1168
1169    do memberIndex = 1, nEns
1170
1171      if( mmpi_myid == 0 ) then
1172        write(*,*) 
1173        write(*,*) 'Computing random perturbation number= ', memberIndex
1174      end if
1175
1176      ! global vector random control vector (independent of mpi topology)
1177      do cvIndex = 1, cvm_nvadim_mpiglobal
1178        controlVector_mpiglobal(cvIndex) = rng_gaussian()
1179      end do
1180      call bmat_reduceToMPILocal( controlVector, controlVector_mpiglobal )
1181
1182      call utl_tmg_stop(4) ! stop counter, since Bmat has it's own counters
1183      if (ens_varExist(ensembleAnl,'HU') .and. .not.useMemberAsHuRefState) then
1184        ! Use supplied reference state for LQ to HU conversion
1185        call bmat_sqrtB(controlVector, cvm_nvadim, &       ! IN
1186                        stateVectorPerturbation,   &       ! OUT
1187                        stateVectorRef_opt=stateVectorHuRefStateInterp) ! IN
1188      else
1189        ! No conversion from LQ to HU done in bmat_sqrtB
1190        call bmat_sqrtB(controlVector, cvm_nvadim, &   ! IN
1191                        stateVectorPerturbation)       ! OUT
1192      end if
1193      call utl_tmg_start(4,'--AddEnsRandomPert')
1194
1195      call int_interp_gsv(stateVectorPerturbation, stateVectorPerturbationInterp, &
1196                          statevectorRef_opt=statevectorP0Ref)
1197
1198      ! If desired, use member itself as reference state for LQ to HU conversion
1199      if (ens_varExist(ensembleAnl,'HU') .and. useMemberAsHuRefState) then
1200        call ens_copyMember(ensembleAnl, stateVectorHuRefState, memberIndex)
1201        call gvt_transform( stateVectorPerturbationInterp,  &          ! INOUT
1202                            'LQtoHU_tlm', &                            ! IN
1203                            stateVectorRef_opt=stateVectorHuRefState ) ! IN
1204      end if
1205
1206      ! scale the perturbation by the specified factor
1207      call gsv_scale(stateVectorPerturbationInterp, alphaRandomPert)
1208
1209      write(*,*) 'epp_addRandomPert: member ', memberIndex, ', perturbation min/maxval = ',  &
1210                 minval(perturbation_ptr), maxval(perturbation_ptr)
1211
1212      do varLevIndex = 1, numVarLev
1213        varName = ens_getVarNameFromK(ensembleAnl,varLevIndex)
1214        memberAnl_ptr_r4 => ens_getOneLev_r4(ensembleAnl,varLevIndex)
1215        do latIndex = myLatBeg, myLatEnd
1216          do lonIndex = myLonBeg, myLonEnd
1217
1218            perturbationMean(lonIndex, latIndex, varLevIndex) =   &
1219                 perturbationMean(lonIndex, latIndex, varLevIndex) +  &
1220                 perturbation_ptr(lonIndex, latIndex, varLevIndex) / real(nEns, 8)
1221
1222            ! Add perturbation to member
1223            do stepIndex = 1, tim_nstepobsinc
1224              memberAnl_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex) =  &
1225                   memberAnl_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex) +  &
1226                   perturbation_ptr(lonIndex, latIndex, varLevIndex)
1227            end do ! stepIndex
1228
1229          end do ! lonIndex
1230        end do ! latIndex
1231      end do ! varLevIndex
1232
1233    end do ! memberIndex
1234
1235    ! remove the ensemble mean of the perturbations
1236    do varLevIndex = 1, numVarLev
1237      memberAnl_ptr_r4 => ens_getOneLev_r4(ensembleAnl,varLevIndex)
1238      do latIndex = myLatBeg, myLatEnd
1239        do lonIndex = myLonBeg, myLonEnd
1240          do stepIndex = 1, tim_nstepobsinc
1241            do memberIndex = 1, nEns
1242              memberAnl_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex) =  &
1243                   memberAnl_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex) -  &
1244                   perturbationMean(lonIndex, latIndex, varLevIndex)
1245            end do ! memberIndex
1246          end do ! stepIndex
1247        end do ! lonIndex
1248      end do ! latIndex
1249    end do ! varLevIndex
1250
1251    deallocate(controlVector)
1252    deallocate(controlVector_mpiglobal)
1253    deallocate(perturbationMean)
1254    call gsv_deallocate(stateVectorPerturbation)
1255    call gsv_deallocate(stateVectorPerturbationInterp)
1256    call gsv_deallocate(stateVectorP0Ref)
1257    if (gsv_isAllocated(stateVectorHuRefStateInterp)) then
1258      call gsv_deallocate(stateVectorHuRefStateInterp)
1259    end if
1260
1261    call utl_tmg_stop(4)
1262
1263  end subroutine epp_addRandomPert

1380    ! check if recentering analysis file exists
1381    inquire(file=recenterAnlFileName, exist=recenterAnlFileExists)
1382    if (.not. recenterAnlFileExists) then
1383      write(*,*) 'epp_hybridRecentering: RecenterAnlFileName = ', recenterAnlFileName
1384      call utl_abort('epp_hybridRecentering: The recentering analysis file does not exist')
1385    end if
1386
1387    hco_ens => ens_getHco(ensembleAnl)
1388    vco_ens => ens_getVco(ensembleAnl)
1389    numMembers = ens_getNumMembers(ensembleAnl)
1390
1391    allocate( weightArray(vco_maxNumLevels,0:numMembers) )
1392    allocate( weightArrayLand(0:numMembers) )
1393
1394    ! read the optiontable file, if requested    
1395    if (useOptionTableRecenter) then
1396      write(*,*) 'epp_hybridRecentering: using optiontable file to specify recentering weights.'
1397      nulFile = 0
1398      status = fnom(nulFile, './optiontable', 'FMT+SEQ+R/O', 0)
1399      read(nulFile,'(a)', IOSTAT=status) textLine
1400      if (status /= 0) then
1401        call utl_abort('epp_hybridRecentering: unable to read optiontable file')

1412        if (mmpi_myid==0) write(*,*) 'weightArray = ', weightArray(1,memberIndex)

1415    else
1416      if (any(weightRecenter(1:vco_getNumLev(vco_ens,'MM')) == -1.d0)) then

1418        write(*,*) 'Number of weightRecenter coefficients needed = ', vco_getNumLev(vco_ens,'MM')

1420        write(*,*) weightRecenter(1:vco_getNumLev(vco_ens,'MM'))
1421        call utl_abort('epp_hybridRecentering: A valid weightRecenter coefficient was not provided for all the vertical levels')
1422      end if
1423      do memberIndex = 0, numMembers
1424        weightArray(:,memberIndex) = weightRecenter(:)
1425      end do
1426    end if
1427

1429      weightArrayLand(memberIndex) = weightRecenterLand
1430    end do
1431    
1432    ! allocate and read in recentering analysis state

1434                       mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
1435                       dataKind_opt=4, allocHeightSfc_opt=.false., &
1436                       hInterpolateDegree_opt = 'LINEAR', &
1437                       allocHeight_opt=.false., allocPressure_opt=.false. )
1438    call gsv_zero(stateVectorRecenterAnl)
1439
1440    do stepIndex = 1, tim_nstepobsinc
1441      call gio_readFromFile( stateVectorRecenterAnl, recenterAnlFileName, ' ', ' ',  &
1442                             stepIndex_opt=stepIndex, containsFullField_opt=.true., &
1443                             readHeightSfc_opt=.false. )
1444    end do
1445    
1446    ! apply recentering
1447    call ens_recenter(ensembleAnl, stateVectorRecenterAnl, &
1448                      recenteringCoeff_opt=weightArray(:,1:numMembers),  &
1449                      numMembersToRecenter_opt=numMembersToRecenter, &
1450                      recenteringCoeffLand_opt=weightArrayLand)

1452    call gsv_deallocate(stateVectorRecenterAnl)

1460  subroutine epp_printRmsStats(stateVectorStdDev,fileName,elapsed,ftype,nEns)
1461    !
1462    !:Purpose: Print statistics of a field to an ASCII output file
1463    !
1464    implicit none
1465
1466    ! arguments
1467    type(struct_gsv)             :: stateVectorStdDev
1468    character(len=*)             :: fileName
1469    real(8), intent(in)          :: elapsed
1470    character(len=1), intent(in) :: ftype
1471    integer, intent(in)          :: nEns
1472
1473    ! locals
1474    real(8), allocatable          :: rmsvalue(:) 
1475    type(struct_vco), pointer     :: vco
1476    type(struct_hco), pointer     :: hco
1477    character(len=4), allocatable :: nomvar_v(:)
1478    character(len=4)              :: varLevel
1479    real(8), allocatable          :: scaleFactor(:)
1480    real(4), allocatable          :: pressureOrDepth(:)
1481    integer :: ierr, latIndex, lonIndex, nulFile
1482    integer :: kIndex, kIndexCount, levIndex, numK, nLev_M, kIndexUU, kIndexVV
1483    real(4), pointer              :: stdDev_ptr_r4(:,:,:)
1484    real(8)                       :: pSfc(1,1)
1485    real(8), pointer              :: pressures_T(:,:,:), pressures_M(:,:,:)
1486    integer, external             :: fnom, fclos
1487    real(8), save, allocatable    :: weight(:,:)
1488    logical, save                 :: firstCall = .true.
1489
1490    vco => gsv_getVco(stateVectorStdDev)
1491    nLev_M = vco_getNumLev(vco,'MM')
1492
1493    numK = gsv_getNumK(stateVectorStdDev)
1494    allocate(nomvar_v(numK))
1495    allocate(pressureOrDepth(numK))
1496    allocate(rmsvalue(numK))
1497    allocate(scaleFactor(numK))
1498
1499    ! compute a 2D weight field used for horizontal averaging
1500    hco => gsv_getHco(stateVectorStdDev)
1501    if (firstCall) then
1502      ! weights are reduced in the overlap region in case of a Yin-Yang grid
1503      allocate(weight(stateVectorStdDev%ni,stateVectorStdDev%nj))
1504      weight(:,:) = 0.0d0
1505      call hco_weight(hco, weight)
1506      firstCall = .false.
1507    end if
1508    ! compute global mean variance accounting for weights
1509    call gsv_getField(stateVectorStdDev,stdDev_ptr_r4)
1510    rmsvalue(:) = 0.0D0
1511    do kIndex = 1, numK
1512      do latIndex = stateVectorStdDev%myLatBeg, stateVectorStdDev%myLatEnd
1513        do lonIndex = stateVectorStdDev%myLonBeg, stateVectorStdDev%myLonEnd
1514          rmsvalue(kIndex) = rmsvalue(kIndex) +  &
1515               (stdDev_ptr_r4(lonIndex,latIndex,kIndex)**2) *  &
1516               weight(lonIndex,latIndex)
1517        end do
1518      end do
1519      call mmpi_allreduce_sumreal8scalar(rmsvalue(kIndex),'GRID')
1520      rmsvalue(kIndex) = rmsvalue(kIndex)**0.5
1521    end do
1522
1523    if (vco%vgridPresent) then
1524      ! compute pressure for a column where Psfc=1000hPa
1525      pSfc(1,1) = 1000.0D2 !1000 hPa
1526      ! pressure levels
1527      call czp_fetch3DLevels(vco, pSfc, fldM_opt=pressures_M, fldT_opt=pressures_T)
1528
1529      ! set the variable name and pressure for each element of column
1530      do kIndex = 1, numK
1531        levIndex = gsv_getLevFromK(stateVectorStdDev, kIndex)
1532        nomvar_v(kIndex) = gsv_getVarNameFromK(stateVectorStdDev,kIndex)
1533        varLevel = vnl_varLevelFromVarname(nomvar_v(kIndex))
1534        if (varLevel == 'MM') then
1535          pressureOrDepth(kIndex) = pressures_M(1,1,levIndex)/Psfc(1,1)
1536        else if (varLevel == 'TH') then
1537          pressureOrDepth(kIndex) = pressures_T(1,1,levIndex)/Psfc(1,1)
1538        else
1539          pressureOrDepth(kIndex) = 1.0
1540        end if
1541      end do
1542      deallocate(pressures_M)
1543      deallocate(pressures_T)
1544
1545    else if (vco%nLev_depth > 0) then
1546
1547      ! set the variable name and depth for each element of column
1548      do kIndex = 1, numK
1549        nomvar_v(kIndex) = gsv_getVarNameFromK(stateVectorStdDev,kIndex)
1550        if (vnl_varLevelFromVarName(nomvar_v(kIndex)) == 'SS') then
1551          pressureOrDepth(kIndex) = 0.0
1552        else
1553          levIndex = gsv_getLevFromK(stateVectorStdDev, kIndex)
1554          pressureOrDepth(kIndex) = vco%depths(levIndex)
1555        end if
1556      end do
1557
1558    else
1559
1560      call utl_abort('epp_printRmsStats: Unknown type of levels')
1561
1562    end if
1563
1564    ! set the scaleFactor and rmsvalue for each element of column
1565    do kIndex = 1, numK
1566      if ( (nomvar_v(kIndex) == 'UU') .or. (nomvar_v(kIndex) == 'VV') ) then
1567        scaleFactor(kIndex) = MPC_KNOTS_PER_M_PER_S_R8
1568      else if (nomvar_v(kIndex) == 'P0') then
1569        scaleFactor(kIndex) = MPC_MBAR_PER_PA_R8
1570      else
1571        scaleFactor(kIndex) = 1.0d0
1572      end if
1573      rmsvalue(kIndex) = scaleFactor(kIndex) * rmsvalue(kIndex)
1574    end do
1575
1576    ! write to file
1577    if (mmpi_myid == 0) then
1578      write(*,*) 'epp_printRmsStats: Opening ascii output file: ', trim(fileName)
1579      nulFile = 0
1580      ierr = fnom (nulFile, fileName, 'SEQ+R/W', 0)
1581      if (ierr /= 0) then
1582        call utl_abort('epp_printRmsStats: Cannot open ascii output file')
1583      end if
1584
1585      kIndexCount = 0
1586      if ( (nomvar_v(1) == 'UU') .and. (nomvar_v(1+nLev_M) == 'VV') ) then        
1587        do levIndex = 1, nLev_M
1588          kIndexCount = kIndexCount + 2
1589          kIndexUU = levIndex
1590          kIndexVV = levIndex + nLev_M
1591          write(nulFile,100) &
1592               elapsed,ftype,nEns,nomvar_v(kIndexUU),pressureOrDepth(kIndexUU),rmsvalue(kIndexUU)
1593          write(nulFile,100) &
1594               elapsed,ftype,nEns,nomvar_v(kIndexVV),pressureOrDepth(kIndexVV),rmsvalue(kIndexVV)
1595        end do
1596      end if
1597      do kIndex = kIndexCount+1, numK
1598        write(nulFile,100) &
1599             elapsed,ftype,nEns,nomvar_v(kIndex),pressureOrDepth(kIndex),rmsvalue(kIndex)   
1600      end do
1601      ierr = fclos(nulFile)
1602    end if
1603
1604100 format(f7.2,1x,A1,1x,I5,1x,A4,1x,f12.7,1x,(2E12.5))
1605
1606    deallocate(nomvar_v)
1607    deallocate(pressureOrDepth)
1608    deallocate(scaleFactor)
1609    deallocate(rmsvalue)
1610
1611  end subroutine epp_printRmsStats
