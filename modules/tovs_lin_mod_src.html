 17module tovs_lin_mod
 18  ! MODULE tovs_lin (prefix='tvslin' category='5. Observation operators')
 19  !

 21  !           tangent-linear and adjoint versions of RTTOV
 22  !
 23  use rttov_interfaces_mod
 24
 25  use rttov_types, only :   &
 26       rttov_profile       ,&
 27       rttov_profile_cloud ,&
 28       rttov_radiance      ,&
 29       rttov_transmission  ,&
 30       rttov_chanprof      ,&
 31       rttov_emissivity
 32  use rttov_const, only : &
 33      gas_unit_specconc  ,&
 34      sensor_id_mw       ,&
 35      surftype_sea       ,&
 36      errorStatus_success

 38  use verticalCoord_mod

 40  use utilities_mod

 43  use columnData_mod

 46  save
 47  private
 48
 49  public :: tvslin_rttov_tl, tvslin_rttov_ad
 50
 51
 52contains

 54  !--------------------------------------------------------------------------
 55  !  tvslin_rttov_tl
 56  !--------------------------------------------------------------------------
 57  subroutine tvslin_rttov_tl(columnAnlInc, columnTrlOnAnlIncLev, obsSpaceData)
 58    !
 59    !:Purpose: Tangent linear of computation of radiance with rttov_tl
 60    !   
 61    implicit none
 62
 63    ! Arguments:
 64    type(struct_obs)        :: obsSpaceData  ! obsSpaceData structure
 65    type(struct_columnData) :: columnAnlInc        ! column structure for pertubation profile
 66    type(struct_columnData) :: columnTrlOnAnlIncLev ! column structure for background profile
 67
 68    ! Locals:
 69    type(struct_vco), pointer :: vco_anl
 70    integer, allocatable :: sensorTovsIndexes(:) 
 71    integer, allocatable :: sensorHeaderIndexes(:) 
 72    integer :: allocStatus(5)
 73    integer :: nobmax
 74    integer :: sensorIndex, tovsIndex
 75    integer :: ilowlvl_M,ilowlvl_T,profileCount,headerIndex,levelIndex,nlv_M,nlv_T
 76    integer :: profileIndex
 77    integer :: Vcode
 78
 79    character(len=4) :: ozoneVarName
 80    logical, allocatable :: surfTypeIsWater(:)
 81    real(8), pointer :: delTT(:), delHU(:), delP(:)
 82    real(8), pointer :: delO3(:)
 83    real(8), pointer :: delCLW(:)
 84    real(8), pointer :: delCIW(:), delRF(:), delSF(:)
 85    integer :: btCount
 86    integer,external :: omp_get_num_threads
 87    integer :: nthreads, max_nthreads
 88    integer :: btIndex, bodyIndex
 89    integer :: instrum
 90    integer :: sensorType   !sensor type(1=infrared; 2=microwave; 3=high resolution, 4=polarimetric)
 91    integer :: errorStatus
 92    integer,allocatable :: sensorBodyIndexes(:)
 93    logical,allocatable :: lchannel_subset(:,:)
 94    real(8), allocatable :: surfem1(:)
 95    integer, allocatable  :: frequencies(:)
 96    type(rttov_emissivity), pointer :: emissivity_local(:)
 97    type(rttov_emissivity), pointer :: emissivity_tl(:)
 98    type(rttov_radiance) :: radiancedata_d   ! radiances full structure buffer used in rttov calls
 99    type(rttov_radiance) :: radiancedata_tl  ! tl radiances full structure buffer used in rttov calls
100    type(rttov_transmission) :: transmission       ! transmission
101    type(rttov_transmission) :: transmission_tl    ! transmission tl
102    type(rttov_profile), pointer :: profilesdata_tl(:) ! tl profiles buffer used in rttov calls
103    type(rttov_profile_cloud), pointer :: cld_profiles_tl(:) !tl profiles buffer used in RttovScatt calls
104    type(rttov_chanprof), pointer :: chanprof(:)
105    logical, pointer :: calcemis(:)
106    logical :: runObsOperatorWithClw_tl
107    logical :: runObsOperatorWithHydrometeors_tl
108    integer :: asw
109    real(8) :: obsOMP
110    type (rttov_profile), pointer :: profiles(:)
111    type(rttov_profile_cloud), pointer :: cld_profiles(:)
112         
113    if (tvs_nobtov == 0) return       ! exit if there are not tovs data
114
115    write(*,*) 'tvslin_rttov_tl: Starting'
116
117    call tvs_getProfile(profiles, 'tlad', cld_profiles)
118
119    if (.not. tvs_useO3Climatology .and. .not. col_varExist(columnTrlOnAnlIncLev,'TO3') .and. .not.  col_varExist(columnTrlOnAnlIncLev,'O3L') ) then
120      call utl_abort('tvslin_rttov_tl: if tvs_useO3Climatology is set to .true. the ozone variable must be included as an analysis variable in NAMSTATE.')
121    else if (.not.tvs_useO3Climatology) then 
122      if (col_varExist(columnTrlOnAnlIncLev,'TO3')) then
123        ozoneVarName = 'TO3'
124      else
125        ozoneVarName = 'O3L'
126      end if 
127    end if
128
129    !  1.  Set index for model's lowest level and model top
130
131    nlv_M = col_getNumLev(columnTrlOnAnlIncLev,'MM')
132    nlv_T = col_getNumLev(columnTrlOnAnlIncLev,'TH')
133
134    if ( col_getPressure(columnTrlOnAnlIncLev,1,1,'TH') < col_getPressure(columnTrlOnAnlIncLev,nlv_T,1,'TH') ) then
135      ilowlvl_M = nlv_M
136      ilowlvl_T = nlv_T
137    else
138      ilowlvl_M = 1
139      ilowlvl_T = 1
140    end if
141
142    vco_anl => col_getVco(columnTrlOnAnlIncLev)
143    Vcode = vco_anl%Vcode
144    
145  
146    !     1.  Get number of threads available and allocate memory for some variables
147    !     .   ---------------------------------------------------------------------- 
148
149    !$omp parallel 
150    max_nthreads = omp_get_num_threads()
151    !$omp end parallel
152
153    allocStatus(:) = 0
154    allocate ( sensorTovsIndexes(tvs_nobtov), stat = allocStatus(1) )
155    call utl_checkAllocationStatus(allocStatus(1:1), ' tvslin_rttov_tl sensorTovsIndexes')
156    
157    ! 2.  Computation of hx for tovs data only
158
159    
160    ! Loop over all sensors specified by user
161
162    sensor_loop:  do sensorIndex = 1, tvs_nsensors
163
164      runObsOperatorWithClw_tl = col_varExist(columnTrlOnAnlIncLev,'LWCR') .and. &
165                                 tvs_isInstrumUsingCLW(tvs_instruments(sensorIndex)) .and. &
166                                 tvs_mwInstrumUsingCLW_tl
167      runObsOperatorWithHydrometeors_tl = col_varExist(columnTrlOnAnlIncLev,'LWCR') .and. &
168                                          col_varExist(columnTrlOnAnlIncLev,'IWCR') .and. &
169                                          tvs_isInstrumUsingHydrometeors(tvs_instruments(sensorIndex)) .and. &
170                                          tvs_mwInstrumUsingHydrometeors_tl
171       
172      sensorType = tvs_coefs(sensorIndex) % coef % id_sensor
173      instrum = tvs_coefs(sensorIndex) % coef % id_inst
174      !  loop over all obs.
175      profileCount = 0
176      do tovsIndex = 1, tvs_nobtov
177        !    Currently processed sensor?
178        if ( tvs_lsensor(tovsIndex) == sensorIndex ) then
179          profileCount = profileCount + 1
180          sensorTovsIndexes(profileCount) = tovsIndex
181        end if
182      end do
183     
184      if (profileCount == 0) cycle sensor_loop
185      nobmax = sensorTovsIndexes(profileCount)
186      !     compute the number of calculated radiances for one call
187      btCount = tvs_countRadiances(sensorTovsIndexes(1:profileCount), obsSpaceData, &
188           assim_flag_val_opt=obs_assimilated)
189      if ( btCount == 0 ) cycle  sensor_loop
190   
191      allocate (sensorHeaderIndexes (profileCount), stat= allocStatus(1))
192      allocate (profilesdata_tl(profileCount),      stat= allocStatus(2))
193      if (runObsOperatorWithClw_tl) write(*,*) 'tvslin_rttov_tl: using clw_data'
194      if (runObsOperatorWithHydrometeors_tl) write(*,*) 'tvslin_rttov_tl: using hydrometeor data'
195      allocate (surfTypeIsWater(profileCount),stat= allocStatus(3))
196      call utl_checkAllocationStatus(allocStatus, ' tvslin_rttov_tl')
197 
198      sensorHeaderIndexes(:) = 0 
199      
200      surfTypeIsWater(:) = .false.
201
202      ! allocate profiledata_tl structures
203      asw = 1 ! 1 to allocate
204      call rttov_alloc_tl(                     &
205              allocStatus(1),                  &
206              asw,                             &
207              nprofiles=profileCount,          &
208              nchanprof=btCount,               &
209              nlevels=nlv_T,                   &
210              chanprof=chanprof,               &
211              opts=tvs_opts(sensorIndex),      &
212              profiles_tl=profilesdata_tl,     &
213              coefs=tvs_coefs(sensorIndex),    &
214              transmission=transmission,       &
215              transmission_tl=transmission_tl, &
216              radiance=radiancedata_d,         &
217              radiance_tl=radiancedata_tl,     &
218              calcemis=calcemis,               &
219              emissivity=emissivity_local,     &
220              emissivity_tl=emissivity_tl,     &
221              init=.true.)
222      if (runObsOperatorWithHydrometeors_tl) then
223        allocate(cld_profiles_tl(profileCount))
224        call rttov_alloc_scatt_prof ( allocStatus(2),   &
225                                      profileCount,     &
226                                      cld_profiles_tl,  &
227                                      nlv_T,            &
228                                      nhydro=5,         &
229                                      nhydro_frac=1,    &
230                                      asw=asw,          &
231                                      init=.false.,     &  
232                                      flux_conversion=[1,2,0,0,0])
233      end if
234    
235      call utl_checkAllocationStatus(allocStatus(1:2), ' tvslin_rtttov_tl rttov_alloc_tl 1')
236
237      profileCount = 0
238
239      obs_loop: do tovsIndex = 1, nobmax
240        if (tvs_lsensor(tovsIndex) /= sensorIndex) cycle obs_loop
241        headerIndex = tvs_headerIndex(tovsIndex)
242        profileCount = profileCount + 1
243        surfTypeIsWater(profileCount) = ( tvs_ChangedStypValue(obsSpaceData,headerIndex) == surftype_sea )
244        sensorHeaderIndexes(profileCount) = headerIndex
245      end do obs_loop
246
247      do  profileIndex = 1 , profileCount
248        profilesdata_tl(profileIndex) % gas_units       = gas_unit_specconc ! all gas profiles should be provided in kg/kg
249        profilesdata_tl(profileIndex) % nlevels         =  nlv_T
250        profilesdata_tl(profileIndex) % nlayers         =  nlv_T - 1
251        if (tvs_coefs(sensorIndex)%coef%nozone > 0) then
252          if (tvs_useO3Climatology) then
253            profilesdata_tl(profileIndex) % o3(:) =  0.0d0
254          else
255            delO3 => col_getColumn(columnAnlInc,sensorHeaderIndexes(profileIndex),trim(ozoneVarName))
256            profilesdata_tl(profileIndex) % o3(1:nlv_T) =  delO3(1:nlv_T) * 1.0d-9 ! Assumes model ozone in ug/kg
257            profilesdata_tl(profileIndex) % s2m % o  = col_getElem(columnAnlInc,ilowlvl_T,sensorHeaderIndexes(profileIndex),trim(ozoneVarName)) * 1.0d-9 ! Assumes model ozone in ug/kg
258          end if
259        end if
260
261        ! using the zero CLW value for land FOV
262        if (runObsOperatorWithClw_tl) then 
263          if (surfTypeIsWater(profileIndex)) then
264            delCLW => col_getColumn(columnAnlInc,sensorHeaderIndexes(profileIndex),'LWCR')
265            profilesdata_tl(profileIndex) % clw(1:nlv_T)  = delCLW(:)
266          else
267            profilesdata_tl(profileIndex) % clw(1:nlv_T)  = 0.d0
268          end if
269        end if
270
271        if (runObsOperatorWithHydrometeors_tl) then 
272          if (surfTypeIsWater(profileIndex)) then
273            ! rain flux
274            if (col_varExist(columnAnlInc,'RF')) then
275              delRF => col_getColumn(columnAnlInc,sensorHeaderIndexes(profileIndex),'RF')
276              cld_profiles_tl(profileIndex) % hydro(1:nlv_T,1) = delRF(:)
277            else
278              cld_profiles_tl(profileIndex) % hydro(1:nlv_T,1) = 0.0d0
279            end if
280
281            ! snow flux
282            if (col_varExist(columnAnlInc,'SF')) then
283              delSF => col_getColumn(columnAnlInc,sensorHeaderIndexes(profileIndex),'SF')
284              cld_profiles_tl(profileIndex) % hydro(1:nlv_T,2)  = delSF(:)
285            else
286              cld_profiles_tl(profileIndex) % hydro(1:nlv_T,2) = 0.0d0
287            end if
288
289            ! graupel
290            cld_profiles_tl(profileIndex) % hydro(1:nlv_T,3)  = 0.d0 ! no information for graupel
291
292            ! cloud liquid water content
293            delCLW => col_getColumn(columnAnlInc,sensorHeaderIndexes(profileIndex),'LWCR')
294            cld_profiles_tl(profileIndex) % hydro(1:nlv_T,4) = delCLW(:)
295
296            ! cloud ice water content
297            delCIW => col_getColumn(columnAnlInc,sensorHeaderIndexes(profileIndex),'IWCR')
298            cld_profiles_tl(profileIndex) % hydro(1:nlv_T,5)  = delCIW(:)
299          else
300            cld_profiles_tl(profileIndex) % hydro(1:nlv_T,1:5)  = 0.d0
301          end if ! surfTypeIsWater
302
303          cld_profiles_tl(profileIndex) % hydro_frac(1:nlv_T,1) = 0.d0   ! no perturbation on cloud fraction as it is a binary variable (or or 1.0) in this implementation
304        end if ! runObsOperatorWithHydrometeors_tl
305        
306        profilesdata_tl(profileIndex) % ctp             = 0.0d0
307        profilesdata_tl(profileIndex) % cfraction       = 0.0d0
308        profilesdata_tl(profileIndex) % zenangle        = 0.0d0
309        profilesdata_tl(profileIndex) % azangle         = 0.0d0
310        profilesdata_tl(profileIndex) % skin % surftype = 0
311        profilesdata_tl(profileIndex) % skin % t        = col_getElem(columnAnlInc,1,sensorHeaderIndexes(profileIndex),'TG')
312        profilesdata_tl(profileIndex) % skin % fastem(:)= 0.0d0
313        profilesdata_tl(profileIndex) % skin % salinity = 0.0d0
314        profilesdata_tl(profileIndex) % s2m % t         = col_getElem(columnAnlInc,ilowlvl_T,sensorHeaderIndexes(profileIndex),'TT')        
315        profilesdata_tl(profileIndex) % s2m % q         = 0.d0
316
317        profilesdata_tl(profileIndex) % s2m % p         = col_getElem(columnAnlInc,1,sensorHeaderIndexes(profileIndex),'P0')*MPC_MBAR_PER_PA_R8
318        profilesdata_tl(profileIndex) % s2m % u         = col_getElem(columnAnlInc,ilowlvl_M,sensorHeaderIndexes(profileIndex),'UU')
319        profilesdata_tl(profileIndex) % s2m % v         = col_getElem(columnAnlInc,ilowlvl_M,sensorHeaderIndexes(profileIndex),'VV')
320
321        delP => col_getColumn(columnAnlInc,sensorHeaderIndexes(profileIndex),'P_T')
322        profilesdata_tl(profileIndex) % p(1:nlv_T)    = delP(:) * MPC_MBAR_PER_PA_R8
323        delTT => col_getColumn(columnAnlInc,sensorHeaderIndexes(profileIndex),'TT')
324        profilesdata_tl(profileIndex) % t(1:nlv_T)    = delTT(:)
325        delHU => col_getColumn(columnAnlInc,sensorHeaderIndexes(profileIndex),'HU')
326        profilesdata_tl(profileIndex) % q(1:nlv_T)    = delHU(:)
327        if (runObsOperatorWithHydrometeors_tl) then
328          cld_profiles_tl(profileIndex) % ph (1) = 0.d0
329          cld_profiles_tl(profileIndex) % cfrac = 0.d0
330          do levelIndex = 1, nlv_T - 1
331            cld_profiles_tl(profileIndex) % ph (levelIndex+1) = 0.5d0 * (profilesdata_tl(profileIndex) % p(levelIndex) + profilesdata_tl(profileIndex) % p(levelIndex+1) )
332          end do
333          cld_profiles_tl(profileIndex) % ph (nlv_T+1) = profilesdata_tl(profileIndex) % s2m % p
334        end if
335      end do
336
337      deallocate (sensorHeaderIndexes,  stat= allocStatus(1) )
338      deallocate (surfTypeIsWater,stat= allocStatus(2)) 
339      call utl_checkAllocationStatus(allocStatus, ' tvslin_rttov_tl', .false.)
340
341      !  set nthreads to actual number of threads which will be used.
342
343      nthreads = min(max_nthreads, profileCount)  
344
345      !   2.2  Prepare all input variables required by rttov.
346     
347      allocate ( surfem1(btCount)           ,stat=allocStatus(1))
348      allocate ( sensorBodyIndexes(btCount) ,stat=allocStatus(2))
349      if (runObsOperatorWithHydrometeors_tl) then
350        allocate (frequencies(btCount), stat=allocStatus(3))
351      end if
352      call utl_checkAllocationStatus(allocStatus(1:3), ' tvslin_rtttov_tl')
353    
354      !    get Hyperspecral IR emissivities
355      if ( tvs_isInstrumHyperSpectral(instrum) ) call tvs_getHIREmissivities(sensorTovsIndexes(1:profileCount), &
356           obsSpaceData, surfem1)      
357      allocate( lchannel_subset(profileCount,tvs_nchan(sensorIndex)) )
358      call tvs_getChanprof(sensorTovsIndexes(1:profileCount), obsSpaceData, chanprof, &
359           iptobs_cma_opt=sensorBodyIndexes, lchannel_subset_opt = lchannel_subset)
360      if (runObsOperatorWithHydrometeors_tl) then
361        call rttov_scatt_setupindex (       &
362              errorStatus,                  &
363              profileCount,                 & ! number of profiles
364              tvs_nchan(sensorIndex),       & ! number of channels 
365              tvs_coefs(sensorIndex),       & ! coef structure read in from rttov coef file
366              tvs_coef_scatt(sensorIndex),  & ! coef structure read in from rttov coef file
367              btcount,                      & ! number of calculated channels
368              chanprof,                     & ! channels and profile numbers
369              frequencies,                  & ! array, frequency number for each channel
370              lchannel_subset )               ! OPTIONAL array of logical flags to indicate a subset of channels
371        if (errorStatus /= errorStatus_success) then
372          write(*,*) 'tvslin_rttov_tl: fatal error in rttov_scatt_setupindex ', errorStatus
373          call utl_abort('tvslin_rttov_tl')
374        end if
375      end if
376      deallocate( lchannel_subset )
377      
378      call tvs_getOtherEmissivities(chanprof, sensorTovsIndexes, sensorType, instrum, surfem1, calcemis)
379
380      if (sensorType == sensor_id_mw) then
381        call tvs_getMWemissivityFromAtlas(surfem1(1:btcount), emissivity_local, sensorIndex, chanprof, sensorTovsIndexes(1:profileCount))
382      else
383        emissivity_local(:)%emis_in = surfem1(:)
384      end if
385 
386      !  2.3  Compute tl radiance with rttov_tl
387      
388      errorStatus = errorStatus_success
389      emissivity_tl(:)%emis_in = 0.0d0
390
391      if (runObsOperatorWithHydrometeors_tl) then
392        call rttov_scatt_tl(                                &
393            errorStatus,                                    & ! out
394            tvs_opts_scatt(sensorIndex),                    & ! in
395            nlv_T,                                          & ! in
396            chanprof,                                       & ! in
397            frequencies,                                    & ! in
398            profiles(sensorTovsIndexes(1:profileCount)),    & ! in  
399            cld_profiles(sensorTovsIndexes(1:profileCount)),& ! in
400            tvs_coefs(sensorIndex),                         & ! in
401            tvs_coef_scatt(sensorIndex),                    & ! in
402            calcemis,                                       & ! in
403            emissivity_local,                               & ! inout
404            profilesdata_tl,                                & ! in
405            cld_profiles_tl,                                & ! in
406            emissivity_tl,                                  & ! inout
407            radiancedata_d,                                 & ! inout
408            radiancedata_tl)                                  ! inout 
409      else
410        call rttov_parallel_tl(                             &
411            errorStatus,                                    & ! out
412            chanprof,                                       & ! in
413            tvs_opts(sensorIndex),                          & ! in
414            profiles(sensorTovsIndexes(1:profileCount)),    & ! in
415            profilesdata_tl,                                & ! inout
416            tvs_coefs(sensorIndex),                         & ! in
417            transmission,                                   & ! inout
418            transmission_tl,                                & ! inout
419            radiancedata_d,                                 & ! inout
420            radiancedata_tl,                                & ! inout
421            calcemis=calcemis,                              & ! in
422            emissivity=emissivity_local,                    & ! in
423            emissivity_tl=emissivity_tl,                    & ! inout
424            nthreads=nthreads )                               ! in
425      end if
426      
427      if (errorStatus /= errorStatus_success) then
428        write(*,*) 'Error in rttov_parallel_tl', errorStatus
429        write(*,*) 'temperature           profile=',profiles(sensorTovsIndexes(1)) % t(:)
430        write(*,*) 'temperature increment profile=',profilesdata_tl(1) % t(:)
431        call utl_abort('tvslin_rttov_tl')
432      end if
433
434      !  2.4  Store hx in obsSpaceData,OBS_WORK
435      
436      do btIndex = 1, btCount
437        
438        bodyIndex = sensorBodyIndexes(btIndex)
439        call obs_bodySet_r(obsSpaceData,OBS_WORK,bodyIndex, &
440             radiancedata_tl % bt(btIndex) )
441        if ( tvs_debug ) then
442          obsOMP = obs_bodyElem_r(obsSpaceData,OBS_OMP,bodyIndex)
443          write(*,'(a,i4,2f8.2)') ' ichn,sim,obs= ', &
444               chanprof(btIndex)%chan, radiancedata_tl % bt(btIndex), obsOMP
445        end if
446        
447      end do
448 
449      ! de-allocate memory
450      
451      asw = 0 ! 0 to deallocate
452      if (runObsOperatorWithHydrometeors_tl) then
453        call rttov_alloc_scatt_prof (allocStatus(1),              &
454                                     profileCount,                &
455                                     cld_profiles_tl,             &
456                                     nlv_T,                       &
457                                     nhydro=5,                    &
458                                     nhydro_frac=1,               &
459                                     asw=asw,                     &   
460                                     flux_conversion=[1,2,0,0,0])
461        deallocate(cld_profiles_tl)
462      end if
463      call rttov_alloc_tl(                   &
464           allocStatus(2),                   &
465           asw,                              &
466           nprofiles=profileCount,           &
467           nchanprof=btCount,                &
468           nlevels=nlv_T,                    &
469           chanprof=chanprof,                &
470           opts=tvs_opts(sensorIndex),       &
471           profiles_tl=profilesdata_tl,      &
472           coefs=tvs_coefs(sensorIndex),     &
473           transmission=transmission,        &
474           transmission_tl=transmission_tl,  &
475           radiance=radiancedata_d,          &
476           radiance_tl=radiancedata_tl,      &
477           calcemis=calcemis,                &
478           emissivity=emissivity_local,      &
479           emissivity_tl=emissivity_tl )
480
481      deallocate ( surfem1,          stat=allocStatus(3) )
482      if (allocated(frequencies)) deallocate(frequencies, stat=allocStatus(4))
483      deallocate ( sensorBodyIndexes,stat=allocStatus(5) )
484      call utl_checkAllocationStatus(allocStatus(1:5), ' tvslin_rtttov_tl', .false.)
485      
486    end do sensor_loop
487
488    deallocate ( sensorTovsIndexes )
489    nullify( profiles )
490    write(*,*) 'tvslin_rttov_tl: Finished'
491
492  end subroutine tvslin_rttov_tl
493
494  !--------------------------------------------------------------------------
495  !  tvslin_rttov_ad
496  !--------------------------------------------------------------------------
497  subroutine tvslin_rttov_ad( columnAnlInc, columnTrlOnAnlIncLev, obsSpaceData )
498    !
499    !:Purpose: Adjoint of computation of radiance with rttov_ad
500    !
501
502    implicit none
503
504    ! Arguments:
505    type(struct_columnData) :: columnAnlInc
506    type(struct_columnData) :: columnTrlOnAnlIncLev
507    type(struct_obs)        :: obsSpaceData
508
509    ! locals
510    type(struct_vco), pointer :: vco_anl
511    integer, allocatable :: sensorTovsIndexes(:) 
512    integer, allocatable :: sensorHeaderIndexes(:) 
513
514    integer :: allocStatus(17)
515    integer :: omp_get_num_threads, nthreads
516    integer :: nobmax
517    integer :: sensorIndex, tovsIndex
518    integer :: ilowlvl_T,ilowlvl_M,profileCount,headerIndex,nlv_M,nlv_T
519    integer :: profileIndex, levelIndex
520    integer :: Vcode
521    real(8), allocatable :: tt_ad(:,:)
522    real(8), allocatable :: hu_ad(:,:)
523    real(8), allocatable :: pressure_ad(:,:)  
524    real(8), allocatable :: ozone_ad(:,:)
525    character(len=4) :: ozoneVarName
526    real(8), allocatable :: clw_ad(:,:)
527    real(8), allocatable :: ciw_ad(:,:), rf_ad(:,:), sf_ad(:,:)
528    logical, allocatable :: surfTypeIsWater(:), lchannel_subset(:,:)
529
530    real(8), pointer :: uu_column(:),vv_column(:),tt_column(:),hu_column(:),ps_column(:),  &
531                        tg_column(:),p_column(:),o3_column(:),clw_column(:), &
532                        ciw_column(:), rf_column(:),sf_column(:)
533
534    integer :: btCount
535    integer :: max_nthreads
536    integer :: instrum
537    integer :: btIndex, bodyIndex
538    integer :: sensorType   ! sensor type (1=infrared; 2=microwave; 3=high resolution, 4=polarimetric)
539    
540    integer, allocatable :: sensorBodyIndexes(:)
541    integer :: errorStatus 
542    
543    real(8), allocatable :: surfem1(:)
544    integer, allocatable :: frequencies(:)
545    type(rttov_emissivity), pointer :: emissivity_local(:)
546    type(rttov_emissivity), pointer :: emissivity_ad(:)
547    type(rttov_transmission) :: transmission,transmission_ad
548    type(rttov_radiance) :: radiancedata_ad, radiancedata_d
549    
550    type(rttov_profile), pointer  :: profilesdata_ad(:) ! ad profiles buffer used in rttov calls
551    type(rttov_profile), pointer  :: profiles(:)
552    type(rttov_profile_cloud), pointer  :: cld_profiles(:)
553    type(rttov_profile_cloud), pointer  :: cld_profiles_ad(:)
554    type(rttov_chanprof), pointer :: chanprof(:)
555    integer :: asw
556    logical, pointer :: calcemis  (:)
557    logical :: runObsOperatorWithClw_ad
558    logical :: runObsOperatorWithHydrometeors_ad
559         
560    if (tvs_nobtov == 0) return      ! exit if there are not tovs data
561    write(*,*) 'tvslin_rttov_ad: Starting'
562
563    call tvs_getProfile(profiles, 'tlad', cld_profiles)
564
565    if (.not. tvs_useO3Climatology .and. .not. col_varExist(columnTrlOnAnlIncLev,'TO3') .and. .not.  col_varExist(columnTrlOnAnlIncLev,'O3L') ) then
566      call utl_abort('tvslin_rttov_ad: if tvs_useO3Climatology is set to .true. the ozone variable must be included as an analysis variable in NAMSTATE.')
567    else if (.not.tvs_useO3Climatology) then 
568      if (col_varExist(columnTrlOnAnlIncLev,'TO3')) then
569        ozoneVarName = 'TO3'
570      else
571        ozoneVarName = 'O3L'
572      end if 
573    end if
574
575    !     1.    Set index for model's lowest level and model top
576
577    nlv_M = col_getNumLev(columnTrlOnAnlIncLev,'MM')
578    nlv_T = col_getNumLev(columnTrlOnAnlIncLev,'TH')
579
580    if (  col_getPressure(columnTrlOnAnlIncLev,1,1,'TH') < col_getPressure(columnTrlOnAnlIncLev,nlv_T,1,'TH') ) then
581      ilowlvl_M = nlv_M
582      ilowlvl_T = nlv_T
583    else
584      ilowlvl_M = 1
585      ilowlvl_T = 1
586    end if
587
588    vco_anl => col_getVco(columnTrlOnAnlIncLev)
589    Vcode = vco_anl%Vcode
590
591    !     1.  Get number of threads available and allocate memory for some variables
592 
593    !$omp parallel 
594    max_nthreads = omp_get_num_threads()
595    !$omp end parallel
596
597    allocate ( sensorTovsIndexes(tvs_nobtov) )
598
599    !     2.  Computation of adjoint hx for tovs data only
600
601    ! Loop over all sensors specified by user
602
603    sensor_loop:do  sensorIndex = 1, tvs_nsensors
604
605      runObsOperatorWithClw_ad = col_varExist(columnTrlOnAnlIncLev,'LWCR') .and. &
606                                 tvs_isInstrumUsingCLW(tvs_instruments(sensorIndex)) .and. &
607                                 tvs_mwInstrumUsingCLW_tl
608      runObsOperatorWithHydrometeors_ad = col_varExist(columnTrlOnAnlIncLev,'LWCR') .and. &
609                                          col_varExist(columnTrlOnAnlIncLev,'IWCR') .and. &
610                                          tvs_isInstrumUsingHydrometeors(tvs_instruments(sensorIndex)) .and. &
611                                          tvs_mwInstrumUsingHydrometeors_tl
612     
613      sensorType = tvs_coefs(sensorIndex) % coef% id_sensor
614      instrum = tvs_coefs(sensorIndex) % coef% id_inst
615
616      profileCount = 0
617      do tovsIndex = 1, tvs_nobtov
618        !    Currently processed sensor?
619        if ( tvs_lsensor(tovsIndex) == sensorIndex ) then
620          profileCount = profileCount + 1
621          sensorTovsIndexes(profileCount) = tovsIndex
622        end if
623      end do
624
625      if (profileCount == 0) cycle sensor_loop
626
627      nobmax = sensorTovsIndexes(profileCount)
628
629      !  compute the number of radiances/tbs to be calculated
630      btCount = tvs_countRadiances(sensorTovsIndexes(1:profileCount), obsSpaceData)
631
632      if (btCount == 0) cycle sensor_loop
633     
634      allocStatus(:) = 0
635      allocate (sensorHeaderIndexes(profileCount),       stat= allocStatus(1))
636      allocate (tt_ad              (nlv_T,profileCount), stat= allocStatus(2))
637      allocate (hu_ad              (nlv_T,profileCount), stat= allocStatus(3))
638      allocate (pressure_ad        (nlv_T,profileCount), stat= allocStatus(4))
639      if (.not. tvs_useO3Climatology) then
640        if (tvs_coefs(sensorIndex) %coef %nozone > 0) then
641          allocate (ozone_ad(nlv_T,profileCount),   stat= allocStatus(5))
642        end if
643      end if
644      if (runObsOperatorWithClw_ad .or. runObsOperatorWithHydrometeors_ad) then
645        allocate (clw_ad(nlv_T,profileCount), stat= allocStatus(6))
646      end if
647      allocate (surfTypeIsWater(profileCount),stat= allocStatus(7))
648      surfTypeIsWater(:) = .false.
649      if (runObsOperatorWithHydrometeors_ad) then
650        allocate (ciw_ad(nlv_T,profileCount), stat= allocStatus(8))
651        allocate (rf_ad(nlv_T,profileCount),  stat= allocStatus(9))
652        allocate (sf_ad(nlv_T,profileCount), stat= allocStatus(10))
653      end if
654
655      call utl_checkAllocationStatus(allocStatus, ' tvslin_rttov_ad')
656
657      !  loop over all obs.
658      profileCount = 0 
659      
660      ! loop over all obs.
661      obs_loop: do tovsIndex = 1, nobmax
662        if (tvs_lsensor(tovsIndex)/=sensorIndex) cycle obs_loop
663        headerIndex = tvs_headerIndex(tovsIndex)
664        profileCount = profileCount + 1
665        sensorHeaderIndexes(profileCount) = headerIndex
666      end do obs_loop
667     
668      !  2.1  Calculate the actual number of threads which will be used.
669
670      nthreads = min(max_nthreads, profileCount )  
671
672      !  2.2  Prepare all input variables required by rttov_ad.
673
674      asw = 1 ! 1 for allocation, 0 for deallocation
675
676      call rttov_alloc_ad(                  &
677           allocStatus(1),                  &
678           asw,                             &
679           profileCount,                    &
680           btCount,                         &
681           nlv_T,                           &
682           chanprof,                        &
683           opts=tvs_opts(sensorIndex),      &
684           profiles_ad=profilesdata_ad,     &
685           coefs=tvs_coefs(sensorIndex),    &
686           transmission= transmission,      &
687           transmission_ad= transmission_ad,&
688           radiance=radiancedata_d,         &
689           radiance_ad=radiancedata_ad,     &
690           calcemis=calcemis,               &
691           emissivity=emissivity_local,     &
692           emissivity_ad=emissivity_ad,     &
693           init=.true.)
694
695      allocate (surfem1(btCount), stat=allocStatus(2))
696      if (runObsOperatorWithHydrometeors_ad) allocate ( frequencies(btCount), stat=allocStatus(3))
697      allocate (sensorBodyIndexes(btCount), stat=allocStatus(4))
698      if (runObsOperatorWithHydrometeors_ad) then
699        allocate(cld_profiles_ad(profileCount), stat=allocStatus(5))
700        call rttov_alloc_scatt_prof (allocStatus(6),   &
701                                     profileCount,     &
702                                     cld_profiles_ad,  &
703                                     nlv_T,            &
704                                     nhydro=5,         &
705                                     nhydro_frac=1,    &
706                                     asw=asw,          &     
707                                     flux_conversion=[1,2,0,0,0])
708      end if
709      call utl_checkAllocationStatus(allocStatus(1:6), ' tvslin_rttov_ad')
710      
711      !  get Hyperspectral IR emissivities
712      
713      if ( tvs_isInstrumHyperSpectral(instrum) ) call tvs_getHIREmissivities(sensorTovsIndexes(1:profileCount), obsSpaceData, surfem1)
714
715      ! Build the list of channels/profiles indices
716      allocate( lchannel_subset(profileCount,tvs_nchan(sensorIndex)) )
717      call tvs_getChanprof(sensorTovsIndexes(1:profileCount), obsSpaceData, chanprof, &
718         iptobs_cma_opt = sensorBodyIndexes, lchannel_subset_opt = lchannel_subset)
719      if (runObsOperatorWithHydrometeors_ad) then
720        call rttov_scatt_setupindex (       &
721              errorStatus,                  &
722              profileCount,                 &  ! number of profiles
723              tvs_nchan(sensorIndex),       &  ! number of channels 
724              tvs_coefs(sensorIndex),       &  ! coef structure read in from rttov coef file
725              tvs_coef_scatt(sensorIndex),  &  ! coef structure read in from rttov coef file
726              btcount,                      &  ! number of calculated channels
727              chanprof,                     &  ! channels and profile numbers
728              frequencies,                  &  ! array, frequency number for each channel
729              lchannel_subset )                ! OPTIONAL array of logical flags to indicate a subset of channels
730        if (errorStatus /= errorStatus_success) then
731          write(*,*) 'tvslin_rttov_ad: fatal error in rttov_scatt_setupindex ', errorStatus
732          call utl_abort('tvslin_rttov_ad')
733        end if
734      end if
735      deallocate( lchannel_subset )
736      !     get non Hyperspectral IR emissivities
737      call tvs_getOtherEmissivities(chanprof, sensorTovsIndexes, sensorType, instrum, surfem1, calcemis)
738
739      if (sensorType == sensor_id_mw) then
740        call tvs_getMWemissivityFromAtlas(surfem1(1:btcount), emissivity_local, sensorIndex, chanprof, sensorTovsIndexes(1:profileCount))
741      else
742        emissivity_local(:)%emis_in = surfem1(:)
743      end if
744        
745      do btIndex = 1, btCount
746        bodyIndex = sensorBodyIndexes(btIndex)
747        radiancedata_ad % bt( btIndex ) = obs_bodyElem_r(obsSpaceData,OBS_WORK,bodyIndex)
748      end do
749
750      !  2.3  Compute ad radiance with rttov_ad
751
752      errorStatus = errorStatus_success
753      emissivity_ad(:) % emis_in = 0.0d0
754      emissivity_ad(:) % emis_out = 0.0d0
755  
756      if (runObsOperatorWithHydrometeors_ad) then
757        call rttov_scatt_ad(                                & 
758            errorStatus,                                    &! out
759            tvs_opts_scatt(sensorIndex),                    &! in
760            nlv_T,                                          &! in
761            chanprof,                                       &! in
762            frequencies,                                    &! in
763            profiles(sensorTovsIndexes(1:profileCount)),    &! in
764            cld_profiles(sensorTovsIndexes(1:profileCount)),&! in
765            tvs_coefs(sensorIndex),                         &! in
766            tvs_coef_scatt(sensorIndex),                    &! in
767            calcemis,                                       &! in
768            emissivity_local,                               &! inout
769            profilesdata_ad,                                &! inout
770            cld_profiles_ad,                                &! inout
771            emissivity_ad,                                  &! inout
772            radiancedata_d,                                 &! inout
773            radiancedata_ad)                                 ! inout
774      else
775        call rttov_parallel_ad(                             &
776            errorstatus,                                    &! out
777            chanprof,                                       &! in
778            tvs_opts(sensorIndex),                          &! in
779            profiles(sensorTovsIndexes(1:profileCount)),    &! in
780            profilesdata_ad,                                &! in
781            tvs_coefs(sensorIndex),                         &! in
782            transmission,                                   &! inout
783            transmission_ad,                                &! inout
784            radiancedata_d,                                 &! inout
785            radiancedata_ad,                                &! inout
786            calcemis=calcemis,                              &! in
787            emissivity=emissivity_local,                    &! inout
788            emissivity_ad=emissivity_ad,                    &! inout
789            nthreads = nthreads )
790      end if
791      if (errorStatus /= errorStatus_success) then
792        write(*,*) 'Error in rttov_parallel_ad', errorStatus
793        call utl_abort('tvslin_rttov_ad')
794      end if
795
796      !   2.0  Store adjoints in columnData object
797      tt_ad(:,:) = 0.d0
798      hu_ad(:,:) = 0.d0
799      pressure_ad(:,:) = 0.d0
800      if (.not. tvs_useO3Climatology) then
801        if (tvs_coefs(sensorIndex) %coef %nozone > 0) ozone_ad(:,:) = 0.d0
802      endif
803      if (runObsOperatorWithClw_ad .or. runObsOperatorWithHydrometeors_ad) clw_ad(:,:) = 0.d0
804      if (runObsOperatorWithHydrometeors_ad) then
805        ciw_ad(:,:) = 0.d0
806        rf_ad(:,:) = 0.d0
807        sf_ad(:,:) = 0.d0
808      end if
809      
810      do btIndex = 1, btCount
811        profileIndex = chanprof(btIndex)%prof
812        headerIndex = sensorHeaderIndexes(profileIndex)
813
814        ps_column => col_getColumn(columnAnlInc,headerIndex,'P0')
815        p_column  => col_getColumn(columnAnlInc,headerIndex,'P_T')
816        tg_column => col_getColumn(columnAnlInc,headerIndex,'TG')
817        tt_column => col_getColumn(columnAnlInc,headerIndex,'TT')
818        hu_column => col_getColumn(columnAnlInc,headerIndex,'HU')
819        uu_column => col_getColumn(columnAnlInc,headerIndex,'UU')
820        vv_column => col_getColumn(columnAnlInc,headerIndex,'VV')
821
822        tt_ad(:,profileIndex) =  profilesdata_ad(profileIndex) % t(:)
823        hu_ad(:,profileIndex) = profilesdata_ad(profileIndex) % q(:)
824        pressure_ad(:,profileIndex)   =  profilesdata_ad(profileIndex) % p(:)
825        tg_column(1) = profilesdata_ad(profileIndex) % skin % t 
826        tt_column(ilowlvl_T) = profilesdata_ad(profileIndex) % s2m % t
827        ps_column(1) = profilesdata_ad(profileIndex) % s2m % p * MPC_MBAR_PER_PA_R8
828        hu_column(ilowlvl_T) = 0.d0 
829        uu_column(ilowlvl_M) = profilesdata_ad(profileIndex) % s2m % u
830        vv_column(ilowlvl_M) = profilesdata_ad(profileIndex) % s2m % v
831
832        if (.not. tvs_useO3Climatology) then
833          if (tvs_coefs(sensorIndex) %coef %nozone > 0) then
834            ! This step is just to transfer the value for ilowlvl_T to the memory space defined by 'col_getColumn(...trim(ozoneVarName))  
835            o3_column => col_getColumn(columnAnlInc,headerIndex,trim(ozoneVarName))
836            o3_column(ilowlvl_T) =  profilesdata_ad(profileIndex) % s2m % o * 1.0d-9
837            ozone_ad(:,profileIndex) = profilesdata_ad(profileIndex) % o3(:)
838          end if
839        end if
840
841        if (runObsOperatorWithClw_ad) then
842          clw_ad(:,profileIndex) = profilesdata_ad(profileIndex) % clw(:)
843        end if
844
845        if (runObsOperatorWithHydrometeors_ad) then
846          rf_ad(:,profileIndex)  = cld_profiles_ad(profileIndex) % hydro(:,1)
847          sf_ad(:,profileIndex)  = cld_profiles_ad(profileIndex) % hydro(:,2)
848          clw_ad(:,profileIndex) = cld_profiles_ad(profileIndex) % hydro(:,4)
849          ciw_ad(:,profileIndex) = cld_profiles_ad(profileIndex) % hydro(:,5)
850        end if
851      end do
852
853      !     .  2.1  Store adjoints in columnData object
854      !     .       -----------------------------------
855
856      do  profileIndex = 1 , profileCount 
857        ps_column => col_getColumn(columnAnlInc, sensorHeaderIndexes(profileIndex), 'P0')
858        p_column  => col_getColumn(columnAnlInc, sensorHeaderIndexes(profileIndex), 'P_T')
859        tt_column => col_getColumn(columnAnlInc, sensorHeaderIndexes(profileIndex), 'TT')
860        hu_column => col_getColumn(columnAnlInc, sensorHeaderIndexes(profileIndex), 'HU')
861        
862        do levelIndex = 1, nlv_T
863          p_column(levelIndex) = p_column(levelIndex)  + pressure_ad  (levelIndex,profileIndex) * MPC_MBAR_PER_PA_R8
864          tt_column(levelIndex) = tt_column(levelIndex) + tt_ad  (levelIndex,profileIndex)
865          hu_column(levelIndex) = hu_column(levelIndex) + hu_ad (levelIndex,profileIndex)
866        end do
867      end do
868
869      if (.not. tvs_useO3Climatology) then
870        if (tvs_coefs(sensorIndex) %coef %nozone > 0) then
871          do  profileIndex = 1 , profileCount 
872            o3_column => col_getColumn(columnAnlInc, sensorHeaderIndexes(profileIndex),trim(ozoneVarName))
873            do levelIndex = 1, col_getNumLev(columnAnlInc,'TH')
874              o3_column(levelIndex) = o3_column(levelIndex) +  ozone_ad(levelIndex,profileIndex) * 1.0d-9
875            end do
876          end do
877        end if
878      end if
879
880      if (runObsOperatorWithClw_ad) then
881        do  profileIndex = 1 , profileCount 
882          surfTypeIsWater(profileIndex) = ( tvs_ChangedStypValue(obsSpaceData,sensorHeaderIndexes(profileIndex)) == surftype_sea )
883          if (surfTypeIsWater(profileIndex)) then
884            clw_column => col_getColumn(columnAnlInc, sensorHeaderIndexes(profileIndex),'LWCR')
885            do levelIndex = 1, col_getNumLev(columnAnlInc,'TH')
886              clw_column(levelIndex) = clw_column(levelIndex) + &
887                                       clw_ad(levelIndex,profileIndex)
888            end do
889          end if
890        end do
891      end if
892
893      if (runObsOperatorWithHydrometeors_ad) then
894        do  profileIndex = 1 , profileCount 
895          surfTypeIsWater(profileIndex) = (tvs_ChangedStypValue(obsSpaceData,sensorHeaderIndexes(profileIndex)) == surftype_sea)
896          if (surfTypeIsWater(profileIndex)) then
897            ! rain flux
898            if (col_varExist(columnAnlInc,'RF')) then
899              rf_column => col_getColumn(columnAnlInc, sensorHeaderIndexes(profileIndex),'RF')
900              do levelIndex = 1, col_getNumLev(columnAnlInc,'TH')
901                rf_column(levelIndex) = rf_column(levelIndex) + rf_ad(levelIndex,profileIndex)
902              end do
903            end if
904
905            ! snow flux
906            if (col_varExist(columnAnlInc,'SF')) then
907              sf_column => col_getColumn(columnAnlInc, sensorHeaderIndexes(profileIndex),'SF')
908              do levelIndex = 1, col_getNumLev(columnAnlInc,'TH')
909                sf_column(levelIndex) = sf_column(levelIndex) + sf_ad(levelIndex,profileIndex)
910              end do
911            end if
912
913            ! cloud liquid/ice water content
914            clw_column => col_getColumn(columnAnlInc, sensorHeaderIndexes(profileIndex),'LWCR')
915            ciw_column => col_getColumn(columnAnlInc, sensorHeaderIndexes(profileIndex),'IWCR')
916            do levelIndex = 1, col_getNumLev(columnAnlInc,'TH')
917              clw_column(levelIndex) = clw_column(levelIndex) + clw_ad(levelIndex,profileIndex)
918              ciw_column(levelIndex) = ciw_column(levelIndex) + ciw_ad(levelIndex,profileIndex)
919            end do
920          end if ! surfTypeIsWater
921        end do ! profileIndex
922      end if ! runObsOperatorWithHydrometeors_ad
923
924      deallocate (sensorHeaderIndexes, stat= allocStatus(1) )
925      deallocate (tt_ad,               stat= allocStatus(2) )
926      deallocate (hu_ad,               stat= allocStatus(3) )
927      deallocate (pressure_ad,         stat= allocStatus(4) )
928      if (.not. tvs_useO3Climatology) then
929        if (tvs_coefs(sensorIndex) %coef %nozone > 0) then
930          deallocate (ozone_ad,        stat= allocStatus(5) )
931        end if 
932      end if
933      if ( allocated(clw_ad) ) then
934        deallocate (clw_ad,stat=allocStatus(6))
935      end if
936      deallocate (surfTypeIsWater,stat=allocStatus(7))
937      if ( allocated(ciw_ad) ) then
938        deallocate (ciw_ad, stat= allocStatus(8))
939        deallocate (rf_ad,  stat= allocStatus(9))
940        deallocate (sf_ad,  stat= allocStatus(10))
941      end if
942      
943      call utl_checkAllocationStatus(allocStatus, ' tvslin_fill_profiles_ad', .false.)
944    
945      !     de-allocate memory
946
947      asw = 0 ! 0 to deallocate
948      if (runObsOperatorWithHydrometeors_ad) then
949        call rttov_alloc_scatt_prof (allocStatus(1),   &
950                                     profileCount,     &
951                                     cld_profiles_ad,  &
952                                     nlv_T,            &
953                                     nhydro=5,         &
954                                     nhydro_frac=1,    &
955                                     asw=asw,          &
956                                     flux_conversion= [1,2,0,0,0])
957        deallocate(cld_profiles_ad, stat=allocStatus(2))
958      end if
959      call rttov_alloc_ad(                  &
960           allocStatus(3),                  &
961           asw,                             &
962           profileCount,                    &
963           btCount,                         &
964           nlv_T,                           &
965           chanprof,                        &
966           opts=tvs_opts(sensorIndex),      &
967           profiles_ad=profilesdata_ad,     &
968           coefs=tvs_coefs(sensorIndex),    &
969           transmission= transmission,      &
970           transmission_ad= transmission_ad,&
971           radiance=radiancedata_d,         &
972           radiance_ad=radiancedata_ad,     &
973           calcemis=calcemis,               &
974           emissivity=emissivity_local,     &
975           emissivity_ad=emissivity_ad )
976     
977      
978      deallocate ( surfem1,           stat=allocStatus(4))
979      deallocate ( sensorBodyIndexes, stat=allocStatus(5))
980      if (allocated(frequencies)) deallocate ( frequencies, stat=allocStatus(6))
981      call utl_checkAllocationStatus(allocStatus(1:6), ' tvslin_rttov_ad', .false.)
982     
983    end do sensor_loop
984
985    ! 3.  Close up
986
987    deallocate ( sensorTovsIndexes )
988    nullify( profiles )
989    write(*,*) 'tvslin_rttov_ad: Finished'
990
991  end subroutine tvslin_rttov_ad
