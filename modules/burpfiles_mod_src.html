 45  !--------------------------------------------------------------------------
 46  ! brpf_getDataStamp
 47  !--------------------------------------------------------------------------
 48  subroutine brpf_getDateStamp(datestamp, burpFileName)
 49    implicit none
 50
 51    ! arguments
 52    integer :: dateStamp
 53    character(len=*), intent(in) :: burpFileName
 54    ! locals
 55    integer :: ier, inblks, nulburp, fnom, fclos, numblks
 56    logical :: isExist_L 
 57    integer :: ktime, kdate, kdate_recv, ktime_recv, ihandl, ilong
 58    integer :: itime, iflgs, idburp, ilat, ilon, idx, idy
 59    integer :: ialt, idelay, idate, irs, irunn, inblk, isup, ixaux
 60    integer :: insup, inxaux
 61    integer, allocatable :: ibuf(:)
 62    integer :: inrecs, mrfcls, mrfopn, mrfopc, mrbhdr, mrfloc, mrfget, mrfmxl
 63    integer :: istampobs, inewhh, newdate, nresume, ivals
 64    real(8) :: delhh
 65    character(len=9) :: clstnid
 66
 67    !
 68    !- Get the date from the burp files
 69    !
 70
 71    ier = mrfopc('MSGLVL','FATAL')
 72
 73    ivals = 8
 74    kdate = -9999
 75    ktime = -9999
 76    nresume = 0
 77    nulburp = 0
 78    inquire(file=trim(burpFileName),exist=isExist_L)
 79    if ( isExist_L ) then
 80      ier = fnom(nulburp,trim(burpFileName),'RND+OLD',0)
 81      write(*,*)' Open File : ',trim(burpFileName)
 82      if ( ier == 0 ) then
 83        inblks = -1
 84        inblks = numblks(nulburp)
 85        if ( inblks > 0 ) then
 86          inrecs= mrfopn(nulburp,'READ')
 87          ilong = mrfmxl(nulburp)
 88          allocate(ibuf(ilong + 20))
 89          ibuf(1) = ilong + 20
 90          ihandl  = mrfloc(nulburp,0,'>>*******',-1,-1,-1,-1,-1,-1,0)
 91          if ( ihandl < 0 ) then
 92            ihandl=mrfloc(nulburp,0,'*********',-1,-1,-1,-1,-1,-1,0)
 93          else
 94            nresume=nresume+1
 95          end if
 96          if ( ihandl < 0 ) then
 97            write(*,*) 'AUCUN ENREGISTREMENT VALIDE DANS LE FICHIER BURP'
 98          else
 99            if ((kdate < 0.and.ktime < 0).or.nresume == 1) then 
100              insup=0
101              inxaux=0
102              ier=mrfget(ihandl,ibuf)
103              ier=mrbhdr(ibuf,itime,iflgs,clstnid,idburp,ilat,   &
104                   ilon,idx,idy, ialt,idelay,idate,irs,irunn,inblk, &
105                   isup,insup,ixaux,inxaux)
106              ktime=itime
107              kdate=idate
108              if (nresume == 1) nresume=2
109            end if
110          end if
111          deallocate(ibuf)
112          ier=mrfcls(nulburp)
113        end if
114        end if
115      ier= fclos(nulburp)
116    end if
117
118    !
119    !- Set reference datestamp
120    !
121    ! Make sure all mpi tasks have a valid date (important for split burp files)
122    call rpn_comm_allreduce(kdate,kdate_recv,1,"MPI_INTEGER","MPI_MAX","GRID",ier)
123    call rpn_comm_allreduce(ktime,ktime_recv,1,"MPI_INTEGER","MPI_MAX","GRID",ier)
124    kdate = kdate_recv
125    ktime = ktime_recv
126    if (nresume >= 1 ) then  
127      ier = newdate(datestamp,kdate,ktime*10000,3)
128    else
129      ! Assumes 6-hour windows with reference times being synoptic times.
130      ! Does not require kdate and ktime to be from a resume record.
131      ier = newdate(istampobs,kdate,ktime*10000,3)
132      delhh = 3.0d0
133      call incdatr (datestamp, istampobs, delhh)
134      ier = newdate(datestamp,kdate,inewhh,-3)
135      ktime = ktime/100
136      if (ktime >= 21 .or. ktime < 3) then
137        ktime = 0
138      else if(ktime >= 3 .and. ktime < 9) then
139        ktime = 6
140      else if(ktime >= 9 .and. ktime < 15) then
141        ktime = 12
142      else
143        ktime = 18
144      end if
145      ier = newdate(datestamp,kdate,ktime*1000000,3)
146      ktime = ktime*100
147    end if
148
149    write(*,*)' BURP FILES VALID DATE (YYYYMMDD) : ', kdate
150    write(*,*)' BURP FILES VALID TIME     (HHMM) : ', ktime
151    write(*,*)' BURP FILES DATESTAMP             : ', datestamp
152
153  end subroutine brpf_getDateStamp

158  subroutine brpf_readFile(obsdat,fileName,familyType,fileIndex)
159    implicit none
160
161    ! arguments
162    type (struct_obs), intent(inout) :: obsdat
163    character(len=*) :: fileName
164    character(len=*) :: familyType
165    integer :: fileIndex
166
167    ! locals
168    integer :: bodyIndex, bodyIndexBegin, bodyIndexEnd, headerIndexBegin, headerIndexEnd, headerIndex
169    integer :: numBody, numHeader
170    logical :: burp_chem
171    real(pre_obsReal)  :: missingValue
172
173    write(*,*) ' '
174    write(*,*) 'brpf_readFile: Starting'
175    write(*,*) ' '
176    missingValue = real(MPC_missingValue_R8,pre_obsReal)
177    
178    bodyIndexBegin   = obs_numbody(obsdat) + 1
179    headerIndexBegin = obs_numheader(obsdat) + 1
180    call brpr_readBurp(obsdat,                         & ! INOUT
181                       familyType, fileName, fileIndex)  ! IN
182    bodyIndexEnd   = obs_numbody(obsdat)
183    headerIndexEnd = obs_numheader(obsdat)
184 
185    burp_chem = trim(familyType) == 'CH'
186
187    if ( trim(familyType) /= 'TO' .and. .not.burp_chem ) then
188
189      call ovt_transformObsValues      (obsdat, headerIndexBegin, headerIndexEnd )
190      call ovt_adjustHumGZ             (obsdat, headerIndexBegin, headerIndexEnd )
191      call obsu_computeVertCoordSurfObs(obsdat, headerIndexBegin, headerIndexEnd )
192
193    end if  
194
195    do headerIndex = headerIndexBegin, headerIndexEnd
196
197      call obs_headSet_i(obsdat, OBS_OTP, headerIndex, fileIndex)
198      call obs_setFamily(obsdat, trim(familyType), headerIndex)
199
200      ! For CH family, apply scaling from the element BUFR_SCALE_EXPONENT when present.
201      if (burp_chem) call brpf_setScaleCH(obsdat, headerIndex, forward=.true.)
202
203    end do
204
205    ! initializations
206    do bodyIndex = bodyIndexBegin, bodyIndexEnd
207
208      if ( obs_columnActive_RB(obsdat, OBS_OMA) )  call obs_bodySet_r(obsdat, OBS_OMA , bodyIndex, missingValue )
209      if ( obs_columnActive_RB(obsdat, OBS_OMA0))  call obs_bodySet_r(obsdat, OBS_OMA0, bodyIndex, missingValue )
210      if ( obs_columnActive_RB(obsdat, OBS_OMP) )  call obs_bodySet_r(obsdat, OBS_OMP , bodyIndex, missingValue )
211      if ( obs_columnActive_RB(obsdat, OBS_OMP6))  call obs_bodySet_r(obsdat, OBS_OMP6, bodyIndex, missingValue )
212      if ( obs_columnActive_RB(obsdat, OBS_OER) )  call obs_bodySet_r(obsdat, OBS_OER , bodyIndex, missingValue )
213      if ( obs_columnActive_RB(obsdat, OBS_HPHT))  call obs_bodySet_r(obsdat, OBS_HPHT, bodyIndex, missingValue )
214      if ( obs_columnActive_RB(obsdat, OBS_HAHT))  call obs_bodySet_r(obsdat, OBS_HAHT, bodyIndex, missingValue )
215      if ( obs_columnActive_RB(obsdat, OBS_WORK))  call obs_bodySet_r(obsdat, OBS_WORK, bodyIndex, missingValue )
216      if ( obs_columnActive_RB(obsdat, OBS_SIGI))  call obs_bodySet_r(obsdat, OBS_SIGI, bodyIndex, missingValue )
217      if ( obs_columnActive_RB(obsdat, OBS_SIGO))  call obs_bodySet_r(obsdat, OBS_SIGO, bodyIndex, missingValue )
218      if ( obs_columnActive_RB(obsdat, OBS_ZHA ))  call obs_bodySet_r(obsdat, OBS_ZHA , bodyIndex, missingValue )
219
220    end do
221
222    ! For GP family, initialize OBS_OER to element 15032 (ZTD formal error) 
223    ! for all ZTD data (element 15031)
224    if ( trim(familyType) == 'GP') then
225      write(*,*)' Initializing OBS_OER for GB-GPS ZTD to formal error (ele 15032)'
226      call obsu_setGbgpsError(obsdat, headerIndexBegin, headerIndexEnd )
227    end if
228
229    numHeader = obs_numHeader(obsdat)
230    numBody   = obs_numBody(obsdat)
231    write(*,*) 'brpf_readFile: obs_numheader', numHeader
232    write(*,*) 'brpf_readFile: obs_numbody  ', numBody
233
234  end subroutine brpf_readFile

239  subroutine brpf_updateFile(obsSpaceData,fileName,familyType,fileIndex)
240    implicit none
241
242    ! arguments
243    type (struct_obs), intent(inout) :: obsSpaceData
244    character(len=*) :: fileName
245    character(len=*) :: familyType
246    integer :: fileIndex
247
248    ! locals
249    integer :: headerIndex
250
251    call utl_tmg_start(12,'----UpdateBurpFile')
252
253    write(*,*)
254    write(*,*) 'brpf_updateFile: Starting'
255
256    ! CH family: Scaling of the obs related values to be stored in the BURP files
257    if (familytype == 'CH') then  
258      call obs_set_current_header_list(obsSpaceData,'CH')
259      HEADER: do
260        headerIndex = obs_getHeaderIndex(obsSpaceData)
261        if (headerIndex < 0) exit HEADER
262        call brpf_setScaleCH(obsSpaceData,headerIndex,forward=.false.)
263      end do HEADER
264    end if
265    
266    call brpr_updateBurp(obsSpaceData,familyType,fileName,fileIndex)
267
268    write(*,*) 'brpf_updateFile: Done'
269    write(*,*)
270
271    call utl_tmg_stop(12)
272
273  end subroutine brpf_updateFile

275  !--------------------------------------------------------------------------
276  ! brpf_setScaleCH
277  !--------------------------------------------------------------------------
278  subroutine brpf_setScaleCH(obsdat,headerIndex,forward)
279    !
280    !:Purpose:  Apply or unapply scaling to CH observations  by multiplying
281    !            (or dividing) with 10^{exponent} where the exponent is from
282    !            element BUFR_SCALE_EXPONENT if provided.           
283    !
284    !
285      implicit none
286
287      ! Arguments      
288      type (struct_obs), intent(inout):: obsdat ! struct_obs instance
289      integer, intent(in) :: headerIndex ! header index in obsdat
290      logical, intent(in) :: forward     ! applies scaling if .true., unapplies scaling if .false.
291
292      ! Locals
293      integer  :: bodyIndex,rln,nlv
294
295      real(pre_obsReal) :: obsv
296      real(pre_obsReal) :: vomp, voma, voer, vhpht, scale
297      integer        :: nexp,iobs,iexp
298
299      real(pre_obsReal), allocatable :: expnt(:)
300
301      rln = obs_headElem_i(obsdat,OBS_RLN,headerIndex)
302      nlv = obs_headElem_i(obsdat,OBS_NLV,headerIndex)
303
304      allocate(expnt(nlv))
305
306      ! Count number of power of 10 exponents
307      nexp = 0
308      do bodyIndex = rln, nlv + rln -1
309         if (obs_bodyElem_i(obsdat,OBS_VNM,bodyIndex) == bufr_scale_exponent) then
310            nexp = nexp + 1
311            expnt(nexp) = obs_bodyElem_r(obsdat,OBS_VAR,bodyIndex)
312         end if
313      end do
314
315      if (nexp == 0) then
316         deallocate(expnt)
317         return
318      end if
319
320      if (nexp*2 /= nlv) then
321         ! Skip over obs assuming mantissa was filtered out in brpr_readBurp 
322         ! (not inserted in obsSpaceData) due to quality flags.
323         ! Set exponent quality flag to that of a 'Suspicious element' 
324         
325         do bodyIndex = RLN, NLV + RLN -1
326            call obs_bodySet_r(obsdat,OBS_VAR,bodyIndex, 0.0D0 )
327            call obs_bodySet_i(obsdat,OBS_FLG,bodyIndex, ibset(obs_bodyElem_i(obsdat,OBS_FLG,bodyIndex),02) )
328            call obs_bodySet_i(obsdat,OBS_FLG,bodyIndex, ibset(obs_bodyElem_i(obsdat,OBS_FLG,bodyIndex),04) )
329            call obs_bodySet_i(obsdat,OBS_FLG,bodyIndex, ibset(obs_bodyElem_i(obsdat,OBS_FLG,bodyIndex),09) )
330         end do
331              
332         ! write(*,*) 'NLV =',nlv,' Nexp=',nexp    
333         ! call utl_abort('brpf_setScaleCH: Inconsistent number of exponents')
334         deallocate(expnt)
335         return
336      end if
337
338      if (forward) then
339         
340         ! Apply power of 10 exponents if present
341         iobs = 0
342         do bodyIndex = RLN, NLV + RLN -1
343            if (obs_bodyElem_i(obsdat,OBS_VNM,bodyIndex) /= bufr_scale_exponent) then
344               iobs = iobs + 1
345               obsv = obs_bodyElem_r(obsdat,OBS_VAR,bodyIndex)
346               call obs_bodySet_r(obsdat,OBS_VAR,bodyIndex,obsv*10**(expnt(iobs)) )
347            end if
348         end do
349      
350      else
351             
352         ! Unapply power of 10 exponents if present
353         iobs=0
354         iexp=0
355         do bodyIndex = RLN, NLV + RLN -1
356            if (obs_bodyElem_i(obsdat,OBS_VNM,bodyIndex) == bufr_scale_exponent) then
357               ! Store scaling exponents
358               iexp = iexp + 1
359               call obs_bodySet_r(obsdat,OBS_OMP,bodyIndex,expnt(iexp))
360               call obs_bodySet_r(obsdat,OBS_OMA,bodyIndex,expnt(iexp))
361               call obs_bodySet_r(obsdat,OBS_OER,bodyIndex,expnt(iexp))
362               call obs_bodySet_r(obsdat,OBS_HPHT,bodyIndex,expnt(iexp))
363            else
364               iobs=iobs+1
365               obsv=obs_bodyElem_r(obsdat,OBS_VAR,bodyIndex)
366               vomp=obs_bodyElem_r(obsdat,OBS_OMP,bodyIndex)
367               voma=obs_bodyElem_r(obsdat,OBS_OMA,bodyIndex)
368               voer=obs_bodyElem_r(obsdat,OBS_OER,bodyIndex)
369               vhpht=obs_bodyElem_r(obsdat,OBS_HPHT,bodyIndex)
370               scale=10**(-expnt(iobs))
371               call obs_bodySet_r(obsdat,OBS_VAR,bodyIndex,obsv*scale )
372               call obs_bodySet_r(obsdat,OBS_OMP,bodyIndex,vomp*scale )
373               call obs_bodySet_r(obsdat,OBS_OMA,bodyIndex,voma*scale )
374               call obs_bodySet_r(obsdat,OBS_OER,bodyIndex,voer*scale )
375               call obs_bodySet_r(obsdat,OBS_HPHT,bodyIndex,vhpht*scale )
376            end if
377         end do
378                 
379      end if
380
381  end subroutine brpf_setScaleCH

386  function brpf_obsSub_read(filename,stnid,varno,nlev,ndim,block_type,bkstp_opt, &
387                            match_nlev_opt,codtyp_opt,numColumns_opt) result(burp_out)
388    !
389    !:Purpose: To retrieve information from observation BURP file. Returns the
390    !          data in a struct_oss_obsdata object. Can retrieve either 1D or 2D
391    !          data from a report.
392    !
393    !:Comments:
394    !
395    !    - BUFR power 10 exponent element (i.e. data with BUFR number
396    !      BUFR_SCALE_EXPONENT) will be applied only to 1D data if present.
397    !    - As burp_out is for a specific input stnid, burp_out%code contains
398    !      only the (lat/long and time coord.) with 22 characters.
399    !    - Exponent BUFR data (i.e. data with BUFR number BUFR_SCALE_EXPONENT)
400    !      will be applied only to 1D data.
401    !
402    !:Arguments:
403    !   :filename:        observation family name
404    !   :stnid:           station ID of observation
405    !   :varno:           BUFR code (if <=0, search through all codes to obtain first
406    !                     between 10000 and 16000)
407    !   :nlev:            number of levels in the observation (number of rows) 
408    !   :ndim:            number of dimensions for the retrieved data in each report
409    !                     (e.g. ndim=1 for std, ndim=2 for averaging kernels)
410    !   :numColumns_opt:  Number of columns (if different from nlev and for ndim=2)
411    !   :block_type:      block type indicated by the two rightmost bits of bknat.
412    !                     Valid values are 'DATA', 'INFO', '3-D', and 'MRQR'.
413    !   :match_nlev_opt: =.true. (default) causes filtering out of report if
414    !                    the report number of levels is different from the input
415    !                    argument nlev
416    !
417    implicit none
418
419    ! Arguments:
420    character(len=*), intent(in)  :: filename ! BURP file name
421    character(len=9), intent(in)  :: stnid ! station ID of observation
422    integer, intent(in)           :: varno
423    integer, intent(in)           :: nlev  ! number of levels in the observation
424    integer, intent(in)           :: ndim
425    character(len=4), intent(in)  :: block_type
426    integer, intent(in), optional :: numColumns_opt ! Number of columns (if different from nlev and for ndim=2)
427    integer, intent(in), optional :: bkstp_opt ! bkstp number of requested block
428    logical, intent(in), optional :: match_nlev_opt
429    integer, intent(in), optional :: codtyp_opt(:) ! optional CODTYP list for search
430    type(struct_oss_obsdata) :: burp_out   ! struct_oss_obsdata object
431    
432    ! Locals:
433    character(len=9)  :: rep_stnid
434    type(burp_file)   :: brp
435    type(burp_rpt)    :: rep
436    type(burp_block)  :: blk
437    integer           :: error,ref_rpt,nrep,ref_blk,varno_ivar
438    integer           :: ref_bkstp,nval,ivar,ilev,icount,icodtyp
439    integer           :: date,time,ilat,ilon,iele,nele,icol,ivar_previous
440    logical           :: match_nlev
441    real(8)           :: exponent
442
443    if (present(match_nlev_opt)) then
444       match_nlev = match_nlev_opt
445    else
446       match_nlev = .true.
447    end if
448
449    ! initialize burp file, report, and block
450    call BURP_Init(brp, iostat=error)
451    call BURP_Init(rep, iostat=error)
452    call BURP_Init(blk, iostat=error)
453
454    ! open the burp file
455    call BURP_New(brp, FILENAME=filename, MODE=FILE_ACC_READ, IOSTAT=error)
456    if (error /= 0) call utl_abort('brpf_obsSub_read: Could not find/open BURP file: ' // trim(filename))
457
458    write(*,*) "brpf_obsSub_read: Reading file " // trim(filename)
459    write(*,*) "brpf_obsSub_read: Selecting STNID = ",stnid," BUFR = ",varno," block type = ",block_type
460    write(*,*) "brpf_obsSub_read:           nlev = ",nlev," match_nlev = ",match_nlev
461    if (present(bkstp_opt))  write(*,*) "brpf_obsSub_read:           bkstp  = ",bkstp_opt
462    if (present(codtyp_opt)) write(*,*) "brpf_obsSub_read:           codtyp = ",codtyp_opt(:)
463
464    ! get number of reports in file
465    call BURP_Get_Property(brp, NRPTS=nrep)
466
467    ! allocate memory
468    if (ndim == 1) then
469      call oss_obsdata_alloc(burp_out,nrep,dim1=nlev)
470    else
471      if (present(numColumns_opt)) then
472        call oss_obsdata_alloc(burp_out,nrep,dim1=nlev,dim2_opt=numColumns_opt)
473      else
474        call oss_obsdata_alloc(burp_out,nrep,dim1=nlev,dim2_opt=nlev)
475      end if
476    end if
477    
478    icount = 0  ! counter of reports with same stnid, number of levels, and varno as input 
479    ref_rpt = 0
480    
481    ! loop through reports    
482    REPORTS: do
483
484       ref_rpt = BURP_Find_Report(brp, REPORT=rep, SEARCH_FROM=ref_rpt, IOSTAT=error)
485
486       if (ref_rpt<0) exit REPORTS
487       
488       call BURP_Get_Property(rep, STNID=rep_stnid, DATE=date, TEMPS=time, LATI=ilat, LONG=ilon, IDTYP=icodtyp) 
489
490       if (present(codtyp_opt)) then
491          if (.not.any(codtyp_opt(:) == icodtyp)) cycle REPORTS
492       end if
493
494       if (.not.utl_stnid_equal(stnid,rep_stnid)) cycle REPORTS
495
496       ! loop through blocks
497       ref_blk = 0
498       BLOCKS: do
499          
500          ref_blk = BURP_Find_Block(rep, BLOCK=blk, SEARCH_FROM=ref_blk, IOSTAT=error)          
501          if (ref_blk<0) exit BLOCKS
502          
503          call BURP_Get_Property(blk, NELE=nele, NVAL=nval, BKSTP=ref_bkstp, IOSTAT=error)
504
505          if (.not.IS_Burp_Btyp(trim(block_type),BLOCK=blk)) cycle BLOCKS
506          if (match_nlev .and. nval /= nlev) cycle BLOCKS
507          if (present(bkstp_opt)) then
508             if (bkstp_opt /= ref_bkstp) cycle BLOCKS
509          end if
510
511          if (varno > 0) then
512             ivar = BURP_Find_Element(blk, ELEMENT=varno, IOSTAT=error)
513             if (ivar < 0) cycle BLOCKS
514          else 
515             ! Search for first data element within elements 10000 and 16000.
516             varno_ivar=-1
517             do ivar=1,nele
518                varno_ivar=BURP_Get_Element(blk, INDEX=ivar, IOSTAT=error)
519                if (varno_ivar >= 10000.and.varno_ivar < 16000) exit
520             end do
521             if (varno_ivar < 10000.or.varno_ivar >= 16000) call utl_abort('brpf_obsSub_read: No valid element found for STNID ' // rep_stnid )
522          end if
523
524          ! required block found if code reaches this point, retrieve data and store in burp_out
525          
526          if (nval > nlev) call utl_abort('brpf_obsSub_read: number of levels in the report (' // trim(utl_str(nval)) // &
527                                         ') exceeds the specified maximum number of levels (' // trim(utl_str(nlev)) // &
528                                         ') for STNID ' // rep_stnid )
529
530          icount=icount+1
531          burp_out%code(icount) = oss_obsdata_get_header_code(ilon,ilat,date,time,rep_stnid)  ! this code is a unique identifier for this report
532
533          if (ndim == 1) then
534             ! retrieve 1D data
535
536             do ilev=1,nval                   
537               burp_out%data1d(ilev,icount) = BURP_Get_Rval(blk, NELE_IND=ivar, NVAL_IND=ilev, NT_IND=1, IOSTAT=error)                
538             end do
539             
540             if (ivar < nele) then             
541               if (BURP_Get_Element(blk, INDEX=ivar+1, IOSTAT=error) == BUFR_SCALE_EXPONENT) then
542                 ! Apply exponent
543                 do ilev=1,nval                   
544                   exponent = BURP_Get_Rval(blk, NELE_IND=ivar+1, NVAL_IND=ilev, NT_IND=1, IOSTAT=error)
545                   burp_out%data1d(ilev,icount) = burp_out%data1d(ilev,icount) * 10**exponent
546                 end do
547               end if
548             end if
549                   
550          else if (ndim == 2) then
551             ! retrieve 2D data
552
553             icol = 0
554             ivar_previous=0
555             do iele=1,nele
556                ivar = BURP_Get_Element(blk, INDEX=iele, IOSTAT=error)
557                if (ivar == varno) then
558                   icol = icol+1
559                   ivar_previous=ivar
560                   do ilev=1,nval
561                      burp_out%data2d(ilev,icol,icount) = BURP_Get_Rval(blk, NELE_IND=iele, NVAL_IND=ilev, NT_IND=1, IOSTAT=error)
562                   end do                   
563                else if (ivar_previous == varno .and. ivar == BUFR_SCALE_EXPONENT) then
564                   ivar_previous=0
565                   do ilev=1,nval
566                      exponent = BURP_Get_Rval(blk, NELE_IND=iele, NVAL_IND=ilev, NT_IND=1, IOSTAT=error)
567                      burp_out%data2d(ilev,icol,icount) = burp_out%data2d(ilev,icol,icount) * 10**exponent
568                   end do   
569                else
570                   ivar_previous=0                
571                end if  
572             end do
573             if (present(numColumns_opt)) then
574               if (icol /= numColumns_opt) call utl_abort('brpf_obsSub_read: number of columns (' // trim(utl_str(icol)) // &
575                                         ') is not equal to the required number (' // trim(utl_str(numColumns_opt)) // &
576                                         ') for STNID ' // rep_stnid )
577             else
578               if (icol > nlev ) call utl_abort('brpf_obsSub_read: number of columns (' // trim(utl_str(icol)) // &
579                                         ') exceeds the maximum number (' // trim(utl_str(nlev)) // &
580                                         ') for STNID ' // rep_stnid )
581             end if
582          end if
583
584          exit BLOCKS
585          
586       end do BLOCKS
587      
588    end do REPORTS
589
590    ! resize first dimension of data arrays from length of nrep to icount
591    call utl_resize(burp_out%code,icount)
592    if (ndim == 1) then
593       call utl_resize(burp_out%data1d,nlev,icount)
594    else if (ndim == 2) then
595       call utl_resize(burp_out%data2d,nlev,nlev,icount)
596    end if
597
598    burp_out%nrep = icount
599
600    write(*,*) "brpf_obsSub_read: Reading of file complete. Number of reports found: ",burp_out%nrep
601    
602    ! deallocate
603    Call BURP_Free(brp,iostat=error)
604    Call BURP_Free(rep,iostat=error)
605    Call BURP_Free(blk,iostat=error)
606    
607  end function brpf_obsSub_read
608
609  !--------------------------------------------------------------------------
610  ! brpf_obsSub_update
611  !--------------------------------------------------------------------------
612  function brpf_obsSub_update(obsdata,filename,varno,block_type,bkstp_opt, &
613                              multi_opt) result(nrep_modified)
614    !
615    !:Purpose: To add or modify data in BURP files from data stored in a
616    !          struct_oss_obsdata object. Provided data can be either 1D or 2D.
617    !
618    !:Comments:
619    !   - Currently assumes that all elements of varno(:) are distinct from
620    !     each other.
621    !   - Blocks with new data will overwrite any existing data of the same
622    !     varno. Otherwise the new data will be appended to the block.
623    !
624    !:Arguments:
625    !   :varno:        BUFR descriptors. Number of elements must be 
626    !                  max(1,obsdata%dim2)
627    !   :block_type:   block type indicated by the two rightmost bits of bknat.
628    !                  Valid values are 'DATA', 'INFO', '3-D', and 'MRQR'.
629    !   :multi_opt:    Indicates if intended report are for 'UNI' or 'MULTI'
630    !                  level data (description is not accurate)
631    !     
632    implicit none
633
634    ! Arguments:
635    type(struct_oss_obsdata), intent(inout) :: obsdata ! Input struct_oss_obsdata object for varno
636    character(len=*), intent(in) :: filename ! BURP file name
637    character(len=4), intent(in) :: block_type
638    integer, intent(in) :: varno(:)
639    integer, intent(in), optional :: bkstp_opt ! bkstp number of requested block
640    character(len=*), intent(in), optional :: multi_opt
641    !Output:
642    integer :: nrep_modified ! Number of modified reports
643
644    ! Locals:
645    integer :: ncount
646    logical :: blk_found
647    integer, parameter :: LNMX=100000, code_len=90
648
649    character(len=9)  :: stnid
650    character(len=code_len) :: code    ! Must be at least as large as burp_code_len
651    type(burp_file)   :: brp
652    type(burp_rpt)    :: rep,rep_new
653    type(burp_block)  :: blk
654    integer           :: error,ref_rpt,nrep,ref_blk,ndim,dim1,dim2
655    integer           :: ref_bkstp,nval,ivar,ilev,istat
656    integer           :: date,time,ilat,ilon,iele,nele,k
657    integer, allocatable :: address(:)
658    real(4), allocatable :: new_vals(:,:,:)
659    logical, allocatable :: modify(:)
660    
661    ! Check presence of data to update
662    if (obsdata%nrep <= 0) then
663       write(*,*) 'brpf_obsSub_update: Skipped due to absence of data to update.'
664       return
665    end if
666    
667    ! Identify dimensions for the input data    
668    ndim=obsdata%ndim
669    dim1=obsdata%dim1
670    if (ndim == 1) then
671       dim2=1
672    else
673       dim2=obsdata%dim2
674    end if
675    
676    if (size(varno) < dim2) call utl_abort('brpf_obsSub_update: Number of BUFR elements not sufficient. ' // &
677                                          trim(utl_str(size(varno))) // ' vs ' // trim(utl_str(dim2)))
678
679    if (code_len < oss_obsdata_code_len()) call utl_abort('brpf_obsSub_update: Length of code string' &
680                                          // ' needs to be increased to ' // trim(utl_str(oss_obsdata_code_len())))
681     
682    ! initialize burp file, report, and block system resources
683    call BURP_Init(brp, iostat=error)
684    call BURP_Init(rep, R2=rep_new, iostat=error)
685    call BURP_Init(blk, iostat=error)
686
687    ! open the burp file in append mode (to replace or add data in a block)
688    call BURP_New(brp, FILENAME=filename, MODE=FILE_ACC_APPEND, IOSTAT=error)
689    if (error /= 0) call utl_abort('brpf_obsSub_update: Could not open BURP file: ' // trim(filename))
690
691    ! get number of reports in file
692    call BURP_Get_Property(brp, NRPTS=nrep)
693
694    allocate(address(nrep),modify(nrep),new_vals(dim1,dim2,nrep))
695    address(:)=0
696    modify(:)=.false.
697    new_vals(:,:,:)=0.
698
699    ! First loop through reports to identify addresses of original file as well as identify if new
700    ! information should be included to that report.
701    ! NOTE: The addresses of all reports have to be saved in their original order to ensure the
702    !       order of the reports in the file is unchanged.
703    ref_rpt=0
704    ncount=0
705    obsdata%irep=1
706    REPORTS1: do
707
708       ref_rpt = BURP_Find_Report(brp, REPORT=rep, SEARCH_FROM=ref_rpt, IOSTAT=error)
709       if (ref_rpt<0) exit REPORTS1
710
711       ncount=ncount+1
712       address(ncount)=ref_rpt
713
714       call BURP_Get_Property(rep, STNID=stnid, DATE=date, TEMPS=time, LATI=ilat, LONG=ilon)
715
716       if (stnid(1:2) == '>>') cycle REPORTS1
717
718       ! Get unique identifier for search from input data
719       code = oss_obsdata_get_header_code(ilon,ilat,date,time,stnid)
720       
721       ! Determine if replacement/additional data likely present for this report
722       if (dim1 == 1.and.dim2 == 1) then
723          new_vals(1,1,ncount)=real(oss_obsdata_get_element(obsdata,trim(code),1,stat_opt=istat))
724       else if (dim2 == 1) then
725          new_vals(:,1,ncount)=real(oss_obsdata_get_array1d(obsdata,trim(code),stat_opt=istat))
726       else 
727          new_vals(:,:,ncount)=real(oss_obsdata_get_array2d(obsdata,trim(code),stat_opt=istat))
728       end if
729
730       if (istat == 0) then
731          if (present(multi_opt)) then
732             ! loop through blocks to find first data block
733             ref_blk = 0
734             BLOCKS1: do
735                ref_blk = BURP_Find_Block(rep, BLOCK=blk, SEARCH_FROM=ref_blk, IOSTAT=error)          
736                if (ref_blk<0) exit BLOCKS1
737                if (IS_Burp_Btyp('DATA',BLOCK=blk)) then
738                   if (IS_Burp_Btyp(trim(multi_opt),BLOCK=blk)) modify(ncount) = .true.
739                   exit BLOCKS1
740                end if
741             end do BLOCKS1
742          else
743             modify(ncount) = .true.
744          end if
745       end if
746
747    end do REPORTS1
748    
749    nrep_modified = count(modify)   ! number of reports with same code and, possibly, same number of obs data levels
750
751    ! Generate new report
752    Call BURP_New(rep_new, ALLOC_SPACE=10*LNMX, IOSTAT=error)
753
754    ! second loop through reports to include the new information to the file    
755    REPORTS2: do k=1,ncount
756    
757       call BURP_Get_Report(brp, REPORT=rep, REF=address(k), IOSTAT=error)
758       
759       ! Copy report header
760       Call BURP_Copy_Header(TO=rep_new,FROM=rep)
761       Call BURP_Init_Report_Write(brp,rep_new,IOSTAT=error)
762
763       ! loop through blocks
764       ref_blk = 0
765       BLOCKS: do
766          
767          ref_blk = BURP_Find_Block(rep, BLOCK=blk, SEARCH_FROM=ref_blk, IOSTAT=error)          
768          if (ref_blk<0) exit BLOCKS
769          
770          if (modify(k)) then
771
772             call BURP_Get_Property(blk, NELE=nele, NVAL=nval, BKSTP=ref_bkstp, IOSTAT=error)
773
774             blk_found = IS_Burp_Btyp(trim(block_type),BLOCK=blk) .and. dim1 == nval
775             if (present(bkstp_opt)) blk_found = blk_found .and. bkstp_opt == ref_bkstp
776
777             if (blk_found) then
778                ! Block to be modified has been found, add new data to block.
779                ! If the varno is already in the block, the new data will overwrite the
780                ! existing data, otherwise will append the new data to the block.
781
782                do iele=1,dim2
783                   ivar = BURP_Find_Element(blk, ELEMENT=varno(iele), IOSTAT=error)           
784                   if (ivar < 0) then
785                      ivar=nele+1
786                      call BURP_Resize_Block(blk,ADD_NELE=1,IOSTAT=error)
787                      call BURP_Set_Element(blk,NELE_IND=ivar,ELEMENT=varno(iele),IOSTAT=error)
788                   end if
789                
790                   do ilev=1,nval 
791                      call BURP_Set_Rval(blk,NELE_IND=ivar,NVAL_IND=ilev,NT_IND=1,RVAL=new_vals(ilev,iele,k),IOSTAT=error)                 
792                   end do
793                end do
794        
795             end if
796          end if
797          
798          ! The call to BURP_Write_Block has ENCODE_BLOCK and CONVERT_BLOCK set
799          ! to .true. in all cases, including when the block has not been
800          ! modified, due to problems that can occur when writing blocks
801          ! containing negative integers with datyp=4.
802          call BURP_Write_Block(rep_new, BLOCK=blk, ENCODE_BLOCK=.true., CONVERT_BLOCK=.true., IOSTAT=error)
803         
804       end do BLOCKS
805       
806       call BURP_Delete_Report(brp,rep,IOSTAT=error)
807       call BURP_Write_Report(brp,rep_new,IOSTAT=error) 
808  
809    end do REPORTS2
810        
811    ! deallocate
812    deallocate(address,modify,new_vals)
813    Call BURP_Free(brp,iostat=error)
814    Call BURP_Free(rep,R2=rep_new,iostat=error)
815    Call BURP_Free(blk,iostat=error)
816    
817  end function brpf_obsSub_update
818
