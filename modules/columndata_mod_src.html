 74  !--------------------------------------------------------------------------
 75  ! col_setup
 76  !--------------------------------------------------------------------------
 77  subroutine col_setup
 78    implicit none
 79    integer :: varIndex, loopIndex
 80    integer :: fnom,fclos,nulnam,ierr
 81    integer :: numVar3D, numVar2D, numVarOther
 82
 83    ! Namelist variables (local)
 84    character(len=4) :: anlvar(vnl_numvarmax)           ! list of state variable names
 85    character(len=8) :: anltime_bin                     ! can be 'MIDDLE', 'FIRST' or 'LAST'
 86    logical          :: conversionVarKindCHtoMicrograms ! activate unit conversion for CH variables
 87    logical          :: abortOnMpiImbalance             ! choose to abort program when MPI imbalance is too large
 88
 89    namelist /namstate/anlvar,rhumin,anltime_bin,addHeightSfcOffset,conversionVarKindCHtoMicrograms, &
 90                       minValVarKindCH, abortOnMpiImbalance
 91
 92    if(mmpi_myid == 0) write(*,*) 'col_setup: List of known (valid) variable names'
 93    if(mmpi_myid == 0) write(*,*) 'col_setup: varNameList3D=',vnl_varNameList3D
 94    if(mmpi_myid == 0) write(*,*) 'col_setup: varNameList2D=',vnl_varNameList2D
 95    if(mmpi_myid == 0) write(*,*) 'col_setup: varNameList  =',vnl_varNameList
 96
 97    ! Read NAMELIST NAMSTATE to find which fields are needed
 98
 99    anlvar(:) = '    '
100    rhumin = MPC_MINIMUM_HU_R8
101    anltime_bin = 'MIDDLE'
102    addHeightSfcOffset = .false.
103    conversionVarKindCHtoMicrograms = .false.
104    minValVarKindCH(:) = mpc_missingValue_r8
105    abortOnMpiImbalance = .true.
106
107    nulnam=0
108    ierr=fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
109    read(nulnam,nml=namstate,iostat=ierr)
110    if(ierr.ne.0) call utl_abort('col_setup: Error reading namelist')
111    if(mmpi_myid == 0) write(*,nml=namstate)
112    ierr=fclos(nulnam)
113
114    col_rhumin = rhumin
115    col_minValVarKindCH(:)=minValVarKindCH(:)
116
117    if( varneed('Z_T') .or. varneed('Z_M') ) then
118      call utl_abort('col_setup: height can not be specified as analysis variable in namelist!')
119    end if
120    if( varneed('P_T') .or. varneed('P_M') ) then
121      call utl_abort('col_setup: pressure can not be specified as analysis variable in namelist!')
122    end if
123
124    numVar3D    = 0
125    numVar2D    = 0
126    numVarOther = 0
127
128    do varIndex = 1, vnl_numvarmax3D
129      if (varneed(vnl_varNameList3D(varIndex))) then
130        varExistList(varIndex) = .true.
131        numVar3D = numVar3D + 1
132      else
133        varExistList(varIndex) = .false.
134      end if
135    end do
136
137    do varIndex = 1, vnl_numvarmax2D
138      if (varneed(vnl_varNameList2D(varIndex))) then
139        varExistList(varIndex+vnl_numvarmax3D) = .true.
140        numVar2D = numVar2D + 1
141      else
142        varExistList(varIndex+vnl_numvarmax3D) = .false.
143      end if
144    end do
145
146    do varIndex = 1, vnl_numvarmaxOther
147      if (varneed(vnl_varNameListOther(varIndex))) then
148        varExistList(varIndex+vnl_numvarmax3D+vnl_numvarmax2D) = .true.
149        numVarOther = numVarOther + 1
150      else
151        varExistList(varIndex+vnl_numvarmax3D+vnl_numvarmax2D) = .false.
152      end if
153    end do
154
155    ! Setup to assign min values to apply
156    
157    ! Check for input values only for variables of CH kind
158    do varIndex = 1, vnl_numvarmax
159      if ( trim(AnlVar(varIndex)) == '' ) exit
160      if ( vnl_varKindFromVarname(AnlVar(varIndex)) == 'CH' ) then
161        if ( minValVarKindCH(varIndex) < 0.99d0 * MPC_missingValue_R8 ) then
162          if ( trim(AnlVar(varIndex)) == 'AF' .or. trim(AnlVar(varIndex)) == 'AC' ) then
163            ! Set for particulate matter in micrograms/cm^3
164            minValVarKindCH(varIndex) = MPC_MINIMUM_PM_R8
165          else
166            ! Set for concentrations in micrograms/kg
167            minValVarKindCH(varIndex) = MPC_MINIMUM_CH_R8
168          end if
169        end if
170      end if
171    end do
172
173    ! Assign min values to apply
174    col_minValVarKindCH(:) = MPC_missingValue_R8
175    do varIndex = 1, vnl_numvarmax
176      if ( varExistList(varIndex) ) then
177        do loopIndex = 1, vnl_numvarmax
178          if ( trim(AnlVar(loopIndex)) == '' ) exit
179          if ( trim(vnl_varNameList(varIndex)) == trim(AnlVar(loopIndex)) ) &
180             col_minValVarKindCH(varIndex) = minValVarKindCH(loopIndex)
181        end do
182      end if 
183    end do
184
185    if(mmpi_myid == 0) write(*,*) 'col_setup: numVar3D (no Z_T/Z_M/P_T/P_M included), numVar2D, numVarOther = ', numVar3D, numVar2D, numVarOther
186    if(mmpi_myid == 0) write(*,*) 'col_setup: varExistList (no Z_T/Z_M/P_T/P_M included) = ',varExistList
187
188    contains
189
190      logical function varneed(varName)
191        character(len=*) :: varName
192        integer :: jvar
193 
194        varneed = .false.
195        do jvar = 1, vnl_numVarMax
196          if (trim(varName) == trim(anlvar(jvar))) then
197            varneed = .true.
198          end if
199        end do
200
201      end function varneed
202
203  end subroutine col_setup

219  !--------------------------------------------------------------------------
220  ! col_allocate
221  !--------------------------------------------------------------------------
222  subroutine col_allocate(column, numCol, mpiLocal_opt, beSilent_opt, setToZero_opt, varNames_opt)
223    implicit none
224
225      ! arguments
226    type(struct_columnData)   :: column
227    integer, intent(in)       :: numCol
228    logical, optional         :: mpiLocal_opt
229    logical, optional         :: beSilent_opt
230    logical, optional         :: setToZero_opt
231    character(len=*),optional :: varNames_opt(:)
232
233    ! locals
234    integer :: iloc, varIndex, varIndex2, numVar
235    logical :: beSilent, setToZero
236
237    if ( present(beSilent_opt) ) then
238      beSilent = beSilent_opt
239    else
240      beSilent = .false.
241    end if
242
243    if ( present(setToZero_opt) ) then
244      setToZero = setToZero_opt
245    else
246      setToZero = .true.
247    end if
248
249    if ( present(varNames_opt) ) then      
250      column%varExistList(:) = .false.
251      numVar = size( varNames_opt ) 
252      do varIndex2 = 1, numVar
253        varIndex = vnl_varListIndex(varNames_opt(varIndex2))
254        column%varExistList(varIndex) = .true.
255      end do
256    else
257      ! set the variable list using the global ExistList
258      column%varExistList(:) = varExistList(:)
259    end if
260
261    if ( column%varExistList(vnl_varListIndex('TT')) .and. &
262         column%varExistList(vnl_varListIndex('HU')) .and. &
263         column%varExistList(vnl_varListIndex('P0')) ) then
264      if ( col_getNumLev(column,'TH') > 0 ) column%varExistList(vnl_varListIndex('Z_T')) = .true.
265      if ( col_getNumLev(column,'MM') > 0 ) column%varExistList(vnl_varListIndex('Z_M')) = .true.
266    end if
267
268    if ( column%varExistList(vnl_varListIndex('P0')) ) then
269      if ( col_getNumLev(column,'TH') > 0 ) column%varExistList(vnl_varListIndex('P_T')) = .true.
270      if ( col_getNumLev(column,'MM') > 0 ) column%varExistList(vnl_varListIndex('P_M')) = .true.
271    end if
272
273    ! add P0LS to the varExistList if vcode=5100
274    if (column%vco%vcode == 5100) then
275      column%varExistList(vnl_varListIndex('P0LS')) = .true.
276    end if
277
278    column%numCol = numCol
279    if(present(mpiLocal_opt)) then
280      column%mpi_local=mpiLocal_opt
281    else
282      column%mpi_local=.true.
283      if (mmpi_myid == 0 .and. .not.beSilent) write(*,*) 'col_allocate: assuming columnData is mpi-local'
284    end if
285
286    if(.not.column%vco%initialized) then
287      call utl_abort('col_allocate: VerticalCoord has not been initialized!')
288    end if
289
290    allocate(column%varOffset(vnl_numvarmax))
291    column%varOffset(:)=0
292    allocate(column%varNumLev(vnl_numvarmax))
293    column%varNumLev(:)=0
294
295    iloc = 0
296    do varIndex = 1, vnl_numvarmax3d
297      if(column%varExistList(varIndex)) then
298        column%varOffset(varIndex) = iloc
299        column%varNumLev(varIndex) = col_getNumLev(column,vnl_varLevelFromVarname(vnl_varNameList(varIndex)))
300        iloc = iloc + column%varNumLev(varIndex)
301      end if
302    end do
303    do varIndex2 = 1, vnl_numvarmax2d
304      varIndex = varIndex2+vnl_numvarmax3d
305      if(column%varExistList(varIndex)) then
306        column%varOffset(varIndex) = iloc
307        column%varNumLev(varIndex) = 1
308        iloc = iloc + 1
309      end if
310    end do
311    do varIndex2 = 1, vnl_numvarmaxOther
312      varIndex = varIndex2+vnl_numvarmax3d+vnl_numvarmax2d
313      if(column%varExistList(varIndex)) then
314        column%varOffset(varIndex) = iloc
315        column%varNumLev(varIndex) = col_getNumLev(column,'OT',vnl_varNameListOther(varIndex2))
316        iloc = iloc + column%varNumLev(varIndex)
317      end if
318    end do
319
320    if (iloc == 0) then
321      call utl_abort('col_allocate: Nothing to allocate')
322    end if
323
324    column%nk = iloc
325
326    if(column%numCol.le.0) then
327      if ( .not.beSilent ) write(*,*) 'col_allocate: number of columns is zero, not allocated'
328    else         
329      allocate(column%all(column%nk,column%numCol))
330      if ( setToZero ) column%all(:,:)=0.0d0
331
332      allocate(column%heightSfc(column%numCol))
333      column%heightSfc(:)=0.0d0
334
335      allocate(column%oltv(2,col_getNumLev(column,'TH'),numCol))
336      if ( setToZero ) column%oltv(:,:,:)=0.0d0
337
338      allocate(column%lat(numCol))
339      if ( setToZero ) column%lat(:)=0.0d0
340    end if
341 
342    if(mmpi_myid == 0 .and. .not.beSilent) write(*,*) 'col_allocate: column%nk = ', column%nk
343    if(mmpi_myid == 0 .and. .not.beSilent) write(*,*) 'col_allocate: varOffset=',column%varOffset
344    if(mmpi_myid == 0 .and. .not.beSilent) write(*,*) 'col_allocate: varNumLev=',column%varNumLev
345
346    column%addHeightSfcOffset = addHeightSfcOffset
347
348    column%allocated=.true.
349
350  end subroutine col_allocate

376  recursive function col_varExist(column_opt,varName) result(varExist)
377    implicit none
378    type(struct_columnData), optional :: column_opt
379    character(len=*), intent(in)      :: varName
380    logical                           :: varExist 
381
382    if (varName == 'Z_*') then
383      varExist =  col_varExist(column_opt, 'Z_T') .and. &
384                  col_varExist(column_opt, 'Z_M')
385    else if (varName == 'P_*') then
386      varExist =  col_varExist(column_opt, 'P_T') .and. &
387                  col_varExist(column_opt, 'P_M')
388    else
389      if ( present(column_opt) ) then
390        if ( column_opt%varExistList(vnl_varListIndex(varName)) ) then
391          varExist = .true.
392        else
393          varExist = .false.
394        end if
395      else
396        if ( varExistList(vnl_varListIndex(varName)) ) then
397          varExist = .true.
398        else
399          varExist = .false.
400        end if
401      end if
402
403      if (present(column_opt)) then
404        varExist = column_opt % varExistList(vnl_varListIndex(varName))
405      else
406        varExist = varExistList(vnl_varListIndex(varName))
407      end if
408    end if
409  
410  end function col_varExist

430  function col_getLevIndexFromVarLevIndex(column, varLevIndex) result(levIndex)
431    implicit none
432    type(struct_columnData) :: column
433    integer, intent(in)     :: varLevIndex
434    integer                 :: levIndex
435    integer                 :: varIndex
436
437    do varIndex = 1, vnl_numvarmax
438      if ( column%varExistList(varIndex) ) then
439        if ( (varLevIndex >= (column%varOffset(varIndex) + 1)) .and.  &
440            (varLevIndex <= (column%varOffset(varIndex) + column%varNumLev(varIndex))) ) then
441          levIndex = varLevIndex - column%varOffset(varIndex)
442          return
443        end if
444      end if
445    end do
446
447    write(*,*) 'col_getLevIndexFromVarLevIndex: varLevIndex out of range: ', varLevIndex
448    call utl_abort('col_getLevIndexFromVarLevIndex')
449

575  function col_getColumn(column,headerIndex,varName_opt) result(onecolumn)
576    implicit none
577    type(struct_columnData), intent(in)    :: column
578    integer, intent(in)                    :: headerIndex
579    character(len=*), intent(in), optional :: varName_opt
580    real(8), pointer                       :: onecolumn(:)
581    integer                                :: ilev1,ilev2
582
583    if(present(varName_opt)) then
584      if(col_varExist(column,varName_opt)) then
585        ilev1 = column%varOffset(vnl_varListIndex(varName_opt))+1
586        ilev2 = ilev1 - 1 + column%varNumLev(vnl_varListIndex(varName_opt))
587        onecolumn => column%all(ilev1:ilev2,headerIndex)
588      else
589        call utl_abort('col_getColumn: Unknown variable name! ' // varName_opt)
590      end if
591    else
592      onecolumn => column%all(:,headerIndex)
