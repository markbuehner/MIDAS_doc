 56  !--------------------------------------------------------------------------
 57  ! loc_setup
 58  !--------------------------------------------------------------------------
 59  subroutine loc_setup(loc, cvDim_out, hco_loc, vco_loc, nEns, vertLocation, ntrunc, locType, &
 60                       locMode, horizLengthScale1, horizLengthScale2, vertLengthScale)
 61    implicit none
 62
 63    type(struct_loc) :: loc
 64    integer, intent(out) :: cvDim_out
 65
 66    type(struct_hco), pointer, intent(in) :: hco_loc
 67    type(struct_vco), pointer, intent(in) :: vco_loc
 68
 69    integer, intent(in) :: nEns
 70    integer, intent(in) :: nTrunc
 71
 72    real(8), intent(in) :: vertLocation(:)
 73    real(8), intent(in) :: horizLengthScale1
 74    real(8), intent(in) :: horizLengthScale2
 75    real(8), intent(in) :: vertLengthScale
 76
 77    character(len=*), intent(in) :: locType, locMode
 78
 79    integer :: nEnsOverDimension, nLev
 80
 81    if (verbose) write(*,*) 'Entering loc_Setup'
 82
 83    !
 84    !- 2.  Setup
 85    !
 86    loc%locType = trim(locType)
 87    loc%hco => hco_loc
 88    loc%vco => vco_loc
 89
 90    if ( loc%vco%Vcode == 5002 .or. loc%vco%Vcode == 5005 ) then
 91      if (loc%vco%nLev_M > 0) then
 92        nLev = loc%vco%nLev_M
 93      else
 94        nLev = loc%vco%nLev_T
 95      end if
 96    else if ( loc%vco%nLev_depth > 0 ) then
 97      nLev = loc%vco%nLev_depth
 98    else
 99      nLev = 1
100    end if
101
102    select case (trim(loc%locType))
103    case('spectral')
104       if (mmpi_myid == 0) write(*,*)
105       if (mmpi_myid == 0) write(*,*) 'loc_setup: LocType = ', trim(locType)
106       call lsp_setup(hco_loc, nEns, nLev, vertLocation, ntrunc, locType,          & ! IN
107                      locMode, horizLengthScale1, horizLengthScale2, vertLengthScale, & ! IN
108                      cvDim_out, loc%lsp, nEnsOverDimension)                            ! OUT
109    case default
110       write(*,*)
111       write(*,*) 'locType = ', trim(locType)
112       call utl_abort('loc_setup: unknown locType')
113    end select
114
115    loc%cvDim = cvDim_out
116    loc%nEnsOverDimension = nEnsOverDimension 
117
118    !
119    !- 3.  Ending
120    !
121    loc%initialized = .true.
122
123  end subroutine loc_setup

128  subroutine loc_Lsqrt(loc, controlVector, ensAmplitude, stepIndex)
129    implicit none

131    type(struct_loc)     :: loc
132
133    integer, intent(in)  :: stepIndex
134    real(8), intent(in)  :: controlVector(:)
135    type(struct_ens)     :: ensAmplitude
136
137    if (verbose) write(*,*) 'Entering loc_Lsqrt'
138
139    select case (trim(loc%locType))
140    case('spectral')
141      call lsp_Lsqrt(loc%lsp, controlVector, & ! IN
142                     ensAmplitude,          & ! OUT
143                     stepIndex)               ! IN
144    case default
145       call utl_abort('loc_Lsqrt: unknown locType')
146    end select
