  93  subroutine vco_SetupFromFile(vco, templatefile, etiket_opt, beSilent_opt)
  94    ! 
  95    !:Purpose: Initialize vertical coordinate object with information from 
  96    !           a standard file.
  97    !
  98    implicit none
  99
 100    ! arguments:
 101    type(struct_vco),pointer,   intent(inout) :: vco          ! Vertical coordinate object 
 102    character(len=*),           intent(in)    :: templatefile ! Template file
 103    character(len=*), optional, intent(in)    :: etiket_opt   ! Optional argument etiket
 104    logical, optional,          intent(in)    :: beSilent_opt ! Optional argument beSilent
 105
 106    ! locals:
 107    logical           :: beSilent
 108    character(len=12) :: etiket
 109    integer :: nultemplate,ierr
 110    integer, parameter :: maxNumRecords = 1000
 111    integer :: recordIndex, numRecords, ikeys(maxNumRecords)
 112    integer :: fnom,fstouv,fstfrm,fclos,fstprm,fstinl
 113    integer :: ip1_sfc
 114    character(len=10) :: blk_S
 115    logical :: fileExists, atmFieldFound, sfcFieldFound, oceanFieldFound
 116    integer :: IP1kind
 117    real :: vertCoordValue
 118    integer :: ideet, inpas, dateStamp_origin, ini, inj, ink, inbits, idatyp
 119    integer :: ip1, ip2, ip3, ig1, ig2, ig3, ig4, iswa, ilng, idltf, iubc
 120    integer :: iextra1, iextra2, iextra3
 121    character(len=2)  :: typvar
 122    character(len=4)  :: nomvar
 123    character(len=1)  :: grtyp
 124
 125    if ( associated(vco) ) then
 126      call utl_abort('vco_setupFromFile: the supplied vco pointer is not null!')
 127    end if
 128
 129    allocate(vco)
 130    call convip(vco%ip1_seaLevel, 0.0, 0, 2, blk_s, .false.) 
 131
 132    if (present(etiket_opt)) then
 133      etiket = etiket_opt
 134    else
 135      etiket = ' '
 136    end if
 137
 138    if (present(beSilent_opt)) then
 139      beSilent = beSilent_opt
 140    else
 141      beSilent = .false.
 142    end if
 143
 144    ! Check if template file exists
 145    if (mmpi_myid == 0 .and. .not. beSilent) then
 146      write(*,*) 'vco_setupFromFile: Template File = ', trim(templatefile)
 147    end if
 148    inquire(file=templatefile,exist=fileExists)
 149    if ( .not. fileExists )then
 150      write(*,*) 'vco_setupFromFile: Template File = ', trim(templatefile)
 151      call utl_abort('vco_setupFromFile: CANNOT FIND TEMPLATE FILE!')
 152    end if
 153
 154    ! First priority, check if vgrid descriptor record present - if so, atmospheric fields
 155    atmFieldFound = utl_varNamePresentInFile('!!',fileName_opt=trim(templatefile))
 156
 157    ! If not atmospheric field, we need to examine data records in the template file
 158    if (.not. atmFieldFound) then
 159
 160      if (.not. beSilent) then
 161        write(*,*) 'vco_setupFromFile: No vgrid descriptor found, examine data records in template file'
 162      end if
 163
 164      ! open the template file
 165      nultemplate=0
 166      ierr=fnom(nultemplate,templatefile,'RND+OLD+R/O',0)
 167      if ( ierr == 0 ) then
 168        ierr =  fstouv(nultemplate,'RND+OLD')
 169      else
 170        write(*,*) 'vco_setupFromFile: Template File = ', trim(templatefile)
 171        call utl_abort('vco_setupFromFile: CANNOT OPEN TEMPLATE FILE!')
 172      end if
 173
 174      ierr = fstinl(nultemplate,ini,inj,ink,-1,etiket,-1,-1,-1,' ', &
 175                    ' ',ikeys,numRecords,maxNumRecords)
 176      if (ikeys(1) <= 0) then
 177        call utl_abort('vco_setupFromFile: Could not find any records in the supplied file')
 178      end if
 179      if (.not. beSilent) then
 180        write(*,*) 'vco_setupFromFile: number of records found = ', numRecords
 181      end if
 182
 183      ! check records for ocean or surface fields
 184      sfcFieldFound   = .false.
 185      oceanFieldFound = .false.
 186      record_loop: do recordIndex = 1, numRecords
 187        ierr = fstprm(ikeys(recordIndex), dateStamp_origin, ideet, inpas, ini, inj, &
 188                      ink, inbits, idatyp, ip1, ip2, ip3, &
 189                      typvar, nomvar, etiket, grtyp, ig1, ig2, ig3, ig4, &
 190                      iswa, ilng, idltf, iubc, iextra1, iextra2, iextra3)
 191
 192        ! ignore any variables not present in varnamelist_mod
 193        if (.not. vnl_varnameIsValid(trim(nomvar))) cycle record_loop
 194
 195        ! check for record with ocean data
 196        call convip(ip1, vertCoordValue, Ip1Kind, -1, blk_s, .false.) 
 197        if (Ip1Kind == 0 .and. &
 198            vnl_varKindFromVarname(trim(nomvar)) == 'OC') then
 199          oceanFieldFound = .true.
 200          exit record_loop
 201        end if
 202
 203        ! check for record with surface data (and that are not ocean variables)
 204        call convip(ip1_sfc, 1.0, 5, 2, blk_s, .false.) 
 205        if (ip1 == 0 .or. ip1 == ip1_sfc .or. ip1 == vco%ip1_seaLevel .or. Ip1Kind == 3) then
 206          sfcFieldFound = .true.
 207          exit record_loop
 208        end if
 209
 210        ! something else was found, abort (not sure abort is necessary)
 211        write(*,*) 'vco_setupFromFile: found record with unknown vertical coordinate'
 212        write(*,*) 'varName = ', trim(nomvar), ' typvar = ', typvar, ' ip1 = ', ip1
 213        call utl_abort('vco_setupFromFile: found a non-surface field')
 214
 215      end do record_loop
 216
 217      ierr =  fstfrm(nultemplate)
 218      ierr =  fclos (nultemplate)
 219
 220    end if
 221
 222    ! call appropriate setup routine based on what was found in template file
 223    if (atmFieldFound) then
 224      call vco_setupAtmFromFile(vco, templatefile, etiket, beSilent)
 225    else if (oceanFieldFound) then
 226      call vco_setupOceanFromFile(vco, templatefile, etiket, beSilent)
 227    else if (sfcFieldFound) then
 228      call vco_setupSfcFromFile(vco, beSilent)
 229    else
 230      call utl_abort('vco_setupFromFile: could not setup vco from template file')
 231    end if
 232
 233  end subroutine vco_setupFromFile

 235  !--------------------------------------------------------------------------
 236  ! vco_setupAtmFromFile
 237  !--------------------------------------------------------------------------
 238  subroutine vco_setupAtmFromFile(vco, templatefile, etiket, beSilent)
 239    ! 
 240    !:Purpose: Initialize vertical coordinate object with information from 
 241    !           a standard file. Use vgrid descriptor for atmospheric fields.
 242    !
 243    implicit none
 244
 245    ! arguments:
 246    type(struct_vco), pointer, intent(inout) :: vco          ! Vertical coordinate object 
 247    character(len=*),          intent(in)    :: templatefile ! Template file
 248    character(len=*),          intent(in)    :: etiket
 249    logical,                   intent(in)    :: beSilent
 250
 251    ! locals:
 252    integer :: Vcode, jlev, nlevMatched, stat, nultemplate, ierr, ikey
 253    integer :: fnom, fstouv, fstfrm, fclos, fstinf
 254    integer :: vgd_nlev_M, vgd_nlev_T, ip1_sfc
 255    integer :: ni, nj, nk, varListIndex, IP1kind
 256    integer,   pointer :: vgd_ip1_M(:), vgd_ip1_T(:)
 257    logical :: ip1_found
 258    character(len=10) :: blk_S
 259    character(len=4) :: nomvar_T, nomvar_M, nomvar_Other
 260    character(len=10) :: IP1string
 261    real :: otherVertCoordValue
 262
 263    ! Open the template file
 264    nultemplate = 0
 265    ierr = fnom(nultemplate,templatefile,'RND+OLD+R/O',0)
 266    if (ierr == 0) then
 267      ierr = fstouv(nultemplate,'RND+OLD')
 268    else
 269      write(*,*) 'vco_setupFromFile: Template File = ', trim(templatefile)
 270      call utl_abort('vco_setupAtmFromFile: CANNOT OPEN TEMPLATE FILE!')
 271    end if
 272
 273    ! Try creating vgrid descriptor
 274    stat = vgd_new(vco%vgrid, unit = nultemplate, format = "fst", ip1 = -1, ip2 = -1)
 275    if (stat == VGD_OK) then
 276      vco%vgridPresent = .true.
 277    else
 278      call utl_abort('vco_setupAtmFromFile: !! record exists, but not able to create descriptor object')
 279    end if
 280
 281    ! Print out vertical structure 
 282    if (mmpi_myid == 0 .and. .not. beSilent) then
 283      call flush(6) ! possibly needed so vgd_print output appears correctly in listing
 284      stat = vgd_print(vco%vgrid)
 285      if (stat /= VGD_OK)then
 286        call utl_abort('vco_setupAtmFromFile: ERROR with vgd_print')
 287      end if
 288    end if
 289
 290    ! Get version of the vertical coordinate
 291    stat = vgd_get(vco%vgrid, key = 'ig_1 - vertical coord code', value = Vcode)
 292    if ( stat /= VGD_OK ) then
 293      call utl_abort('vco_setupAtmFromFile: problem with vgd_get: key= ig_1 - vertical coord code')
 294    end if
 295    if (Vcode /= 5002 .and. Vcode /= 5005 .and. Vcode /= 5100 .and. Vcode /= 21001) then
 296      call utl_abort('vco_setupAtmFromFile: Invalid Vcode. Currently only 5002, 5005, 5100 and 21001 supported.')
 297    end if
 298    vco%Vcode = Vcode
 299
 300    ! Get vgrid values for ip1
 301    stat = vgd_get(vco%vgrid, key='vipm - vertical levels (m)', value = vgd_ip1_m)
 302    stat = vgd_get(vco%vgrid, key='vipt - vertical ip1 levels (t)', value = vgd_ip1_t)
 303
 304    vgd_nlev_M = size(vgd_ip1_M)
 305    vgd_nlev_T = size(vgd_ip1_T)
 306
 307    ! Set the number of vertical levels and allocate vco arrays
 308    vco%nlev_T = 0
 309    nomvar_T = 'TH  '
 310    do jlev = 1, vgd_nlev_T
 311      ikey = fstinf(nultemplate, ni, nj, nk, -1 ,etiket, vgd_ip1_T(jlev), -1, -1, ' ', nomvar_T)
 312      if (ikey > 0) vco%nlev_T = vco%nlev_T + 1
 313    end do
 314    if (vco%nlev_T == 0) then
 315      if (mmpi_myid == 0 .and. .not. beSilent) then
 316        write(*,*) 'vco_setupAtmFromFile: TH not found looking for TT to get nlev_T'
 317      end if
 318      nomvar_T = 'TT  '
 319      do jlev = 1, vgd_nlev_T
 320        ikey = fstinf(nultemplate, ni, nj, nk, -1 ,etiket, vgd_ip1_T(jlev), -1, -1, ' ', nomvar_T)
 321        if (ikey > 0) vco%nlev_T = vco%nlev_T + 1
 322      end do
 323    end if
 324    if (vco%nlev_T == 0 .and. .not. beSilent) then
 325      write(*,*) 
 326      write(*,*) 'vco_setupAtmFromFile: Could not find a valid thermodynamic variable in the template file!'
 327    else if (vco%nlev_T > vco_maxNumLevels) then
 328      write(*,*)
 329      write(*,*) 'nlev_T           = ',vco%nlev_T
 330      write(*,*) 'vco_maxNumLevels = ',vco_maxNumLevels
 331      call utl_abort('vco_setupAtmFromFile: nlev_T is greater than vco_maxNumLevels!')
 332    end if
 333
 334    vco%nlev_M = 0
 335    nomvar_M = 'MM  '
 336    do jlev = 1, vgd_nlev_M
 337      ikey = fstinf(nultemplate, ni, nj, nk, -1 ,etiket, vgd_ip1_M(jlev), -1, -1, ' ', nomvar_M)
 338      if (ikey > 0) vco%nlev_M = vco%nlev_M + 1
 339    end do
 340    if (vco%nlev_M == 0) then
 341      if (mmpi_myid == 0 .and. .not. beSilent) then
 342        write(*,*) 'vco_setupAtmFromFile: MM not found looking for UU to get nlev_M'
 343      end if
 344      nomvar_M = 'UU  '
 345      do jlev = 1, vgd_nlev_M
 346        ikey = fstinf(nultemplate, ni, nj, nk, -1 ,etiket, vgd_ip1_M(jlev), -1, -1, ' ', nomvar_M)
 347        if (ikey > 0) vco%nlev_M = vco%nlev_M + 1
 348      end do
 349    end if
 350    if (vco%nlev_M == 0) then
 351      if (mmpi_myid == 0 .and. .not. beSilent) then
 352        write(*,*) 'vco_setupAtmFromFile: UU not found looking for PP to get nlev_M'
 353      end if
 354      nomvar_M = 'PP  '
 355      do jlev = 1, vgd_nlev_M
 356        ikey = fstinf(nultemplate, ni, nj, nk, -1 ,etiket, vgd_ip1_M(jlev), -1, -1, ' ', nomvar_M)
 357        if (ikey > 0) vco%nlev_M = vco%nlev_M + 1
 358      end do
 359    end if
 360    if (vco%nlev_M == 0 .and. .not. beSilent) then
 361      write(*,*) 
 362      write(*,*) 'vco_setupAtmFromFile: Could not find a valid momentum variable in the template file!'
 363    else if (vco%nlev_M > vco_maxNumLevels) then
 364      write(*,*)
 365      write(*,*) 'nlev_M           = ',vco%nlev_M
 366      write(*,*) 'vco_maxNumLevels = ',vco_maxNumLevels
 367      call utl_abort('vco_setupAtmFromFile: nlev_M is greater than vco_maxNumLevels!')
 368    end if
 369
 370    do jlev = 1, maxNumOtherLevels
 371      otherVertCoordValue = real(jlev)
 372      IP1kind = 3
 373      call convip_plus(vco_ip1_other(jlev), otherVertCoordValue, IP1kind, +2, IP1string, .false.)
 374    end do
 375    vco%nlev_Other(:) = 0
 376    do varListIndex = 1, vnl_numvarmaxOther
 377      nomvar_Other = vnl_varNameListOther(varListIndex)
 378      do jlev = 1, maxNumOtherLevels
 379        ikey = fstinf(nultemplate, ni, nj, nk, -1 ,etiket, vco_ip1_other(jlev), -1, -1, ' ', nomvar_Other)
 380        if (ikey > 0) vco%nlev_Other(varListIndex) = vco%nlev_Other(varListIndex) + 1
 381      end do
 382      if (mmpi_myid == 0 .and. .not. beSilent) then
 383        if (vco%nlev_Other(varListIndex) == 0) then
 384          write(*,*) 
 385          write(*,*) 'vco_setupAtmFromFile: Found no levels in template file for OTHER type variable ', nomvar_Other
 386        else
 387          write(*,*) 'vco_setupAtmFromFile: Found ', vco%nlev_Other(varListIndex),  &
 388               ' levels in template file for OTHER type variable ', nomvar_Other
 389        end if
 390      end if
 391    end do
 392
 393    if (vco%nlev_M == 0 .and. vco%nlev_T == 0) then
 394      call utl_abort('vco_setupAtmFromFile: they were no valid momentum and thermodynamic variables in the template file!')
 395    end  if
 396
 397    if (mmpi_myid == 0 .and. .not.beSilent) then
 398      write(*,*) 'vco_setupAtmFromFile: nlev_M, nlev_T=',vco%nlev_M,vco%nlev_T
 399    end if
 400    
 401    call vco_allocateIp1(vco)
 402
 403    ! Match up ip1 values from file and vgrid for momentum levels
 404    nlevMatched = 0
 405    do jlev = 1, vgd_nlev_M
 406      ikey = fstinf(nultemplate, ni, nj, nk, -1 ,etiket, vgd_ip1_M(jlev), -1, -1, ' ', nomvar_M)
 407      if (ikey > 0) then
 408        nlevMatched = nlevMatched + 1
 409        if (nlevMatched > vco%nlev_M) then
 410          call utl_abort('vco_setupAtmFromFile: Problem with consistency between vgrid descriptor and template file (momentum)')
 411        end if
 412        vco%ip1_M(nlevMatched) = vgd_ip1_M(jlev)
 413      end if
 414    end do
 415    if (nlevMatched /= vco%nlev_M) then
 416      write(*,*) 'vco_setupAtmFromFile: nlevMatched = ', nlevMatched, ', nlev_M = ', vco%nlev_M
 417      call utl_abort('vco_setupAtmFromFile: Problem with consistency between vgrid descriptor and template file (momentum)')
 418    end if
 419
 420    ! Match up ip1 values from file and vgrid for thermo levels
 421    nlevMatched = 0
 422    do jlev = 1, vgd_nlev_T
 423      ikey = fstinf(nultemplate, ni, nj, nk, -1 ,etiket, vgd_ip1_T(jlev), -1, -1, ' ', nomvar_T)
 424      if (ikey > 0) then
 425        nlevMatched = nlevMatched + 1
 426        if (nlevMatched > vco%nlev_T) then
 427          call utl_abort('vco_setupAtmFromFile: Problem with consistency between vgrid descriptor and template file (thermo)')
 428        end if
 429        vco%ip1_T(nlevMatched) = vgd_ip1_T(jlev)
 430      end if
 431    end do
 432    if (nlevMatched /= vco%nlev_T) then
 433      write(*,*) 'vco_setupAtmFromFile: nlevMatched = ', nlevMatched, ', nlev_T = ', vco%nlev_T
 434      call utl_abort('vco_setupAtmFromFile: Problem with consistency between vgrid descriptor and template file (thermo)')
 435    end if
 436
 437    ! determine IP1 of sfc (hyb=1.0)
 438    if (Vcode == 5002 .or. Vcode == 5005 .or. Vcode == 5100) then
 439      call convip(ip1_sfc, 1.0, 5, 2, blk_s, .false.)
 440    else if (Vcode == 21001) then
 441      call convip(ip1_sfc, 0.0, 21, 2, blk_s, .false.)
 442    end if
 443    ip1_found = .false.
 444    do jlev = 1, vgd_nlev_T
 445      if (ip1_sfc == vgd_ip1_T(jlev)) then
 446        ip1_found = .true.
 447        vco%ip1_sfc = vgd_ip1_T(jlev)
 448      end if
 449    end do
 450    if (.not. ip1_found) then
 451      write(*,*) 'vco_setupAtmFromFile: Could not find IP1=',ip1_sfc
 452      call utl_abort('vco_setupAtmFromFile: No surface level found in Vgrid!!!')
 453    else
 454      if (mmpi_myid == 0 .and. .not. beSilent) write(*,*) 'vco_setupAtmFromFile: Set surface level IP1=', vco%ip1_sfc
 455    end if
 456
 457    ! determine IP1s of 2m and 10m levels
 458    call set_2m_10m_levels(vco)
 459
 460    vco%initialized = .true.
 461
 462    ierr =  fstfrm(nultemplate)
 463    ierr =  fclos (nultemplate)
 464
 465  end subroutine vco_setupAtmFromFile

 468  ! vco_setupOceanFromFile
 469  !--------------------------------------------------------------------------
 470  subroutine vco_setupOceanFromFile(vco,templatefile,etiket,beSilent)
 471    ! 
 472    !:Purpose: Initialize vertical coordinate object with information from 
 473    !           a standard file. For Ocean fields on depth levels.
 474    !
 475    implicit none
 476
 477    ! arguments:
 478    type(struct_vco), pointer, intent(inout) :: vco          ! Vertical coordinate object 
 479    character(len=*),          intent(in)    :: templatefile ! Template file
 480    character(len=*),          intent(in)    :: etiket
 481    logical,                   intent(in)    :: beSilent
 482
 483    ! locals:
 484    integer :: nultemplate, ierr
 485    integer, parameter :: maxNumRecords = 500
 486    integer :: recordIndex, numRecords, ikeys(maxNumRecords)
 487    integer :: fnom,fstouv,fstfrm,fclos,fstprm,fstinl
 488    character(len=10) :: blk_S
 489    integer :: IP1kind
 490    real :: vertCoordValue
 491    integer, parameter :: maxNumDepthLevels = 200
 492    real(8)            :: depths(maxNumDepthLevels)
 493    integer            :: ip1_depth(maxNumDepthLevels)
 494
 495    integer :: ideet, inpas, dateStamp_origin, ini, inj, ink, inbits, idatyp
 496    integer :: ip1, ip2, ip3, ig1, ig2, ig3, ig4, iswa, ilng, idltf, iubc
 497    integer :: iextra1, iextra2, iextra3
 498    character(len=2)  :: typvar
 499    character(len=4)  :: nomvar
 500    character(len=1)  :: grtyp
 501
 502    if (.not. beSilent) write(*,*) 'vco_setupOceanFromFile: found ocean fields'
 503
 504    ! initialize some components of vco
 505    vco%vgridPresent = .false.
 506    vco%nlev_T = 0
 507    vco%nlev_M = 0
 508    vco%Vcode  = 0
 509    vco%initialized = .true.
 510
 511    ! open the template file
 512    nultemplate = 0
 513    ierr = fnom(nultemplate,templatefile,'RND+OLD+R/O',0)
 514    if ( ierr == 0 ) then
 515      ierr =  fstouv(nultemplate,'RND+OLD')
 516    else
 517      write(*,*) 'vco_setupFromFile: Template File = ', trim(templatefile)
 518      call utl_abort('vco_setupOceanFromFile: CANNOT OPEN TEMPLATE FILE!')
 519    end if
 520
 521    ierr = fstinl(nultemplate,ini,inj,ink,-1,etiket,-1,-1,-1,' ', &
 522                  ' ',ikeys,numRecords,maxNumRecords)
 523    if ( ikeys(1) <= 0 ) then
 524      call utl_abort('vco_setupOceanFromFile: Could not find any records ' //  &
 525                     'in the supplied file')
 526    end if
 527    record_loop: do recordIndex = 1, numRecords
 528      ierr = fstprm(ikeys(recordIndex), dateStamp_origin, ideet, inpas, ini, inj, &
 529                    ink, inbits, idatyp, ip1, ip2, ip3, &
 530                    typvar, nomvar, etiket, grtyp, ig1, ig2, ig3, ig4, &
 531                    iswa, ilng, idltf, iubc, iextra1, iextra2, iextra3)
 532
 533      ! ignore any variables not present in varnamelist_mod
 534      if (.not. vnl_varnameIsValid(trim(nomvar))) cycle record_loop
 535
 536      ! check for record with ocean data on depth levels
 537      call convip(ip1, vertCoordValue, Ip1Kind, -1, blk_s, .false.) 
 538      if ( Ip1Kind == 0 .and. vertCoordValue >= 0.0 .and. &
 539           vnl_varKindFromVarname(trim(nomvar)) == 'OC' .and. &
 540           vnl_varLevelFromVarname(trim(nomvar)) == 'DP' ) then
 541        ! check if we've NOT already recorded this depth level
 542        if ( .not. any(vertCoordValue == depths(1:vco%nLev_depth)) ) then
 543          vco%nLev_depth = vco%nLev_depth + 1
 544          depths(vco%nLev_depth) = vertCoordValue
 545          ip1_depth(vco%nLev_depth) = ip1
 546          if (mmpi_myid == 0 .and. .not.beSilent) then
 547            write(*,*) 'vco_setupOceanFromFile: found ocean record: nLev_depth = ', &
 548                 vco%nLev_depth, 'varName = ', trim(nomvar), &
 549                 ', value = ', depths(vco%nLev_depth)
 550          end if
 551        end if
 552        cycle record_loop
 553      end if
 554
 555    end do record_loop
 556
 557    ierr =  fstfrm(nultemplate)
 558    ierr =  fclos (nultemplate)
 559
 560    ! Allocate object arrays and copy in depth information
 561    allocate(vco%depths(vco%nLev_depth))
 562    allocate(vco%ip1_depth(vco%nLev_depth))
 563    vco%depths(:)    = depths(1:vco%nLev_depth)
 564    vco%ip1_depth(:) = ip1_depth(1:vco%nLev_depth)
 565
 566    ! Check if ocean depth levels are in correct order (ascending in value)
 567    if ( vco%nLev_depth > 1 ) then
 568      if ( any(vco%depths(2:vco%nLev_depth)-vco%depths(1:(vco%nLev_depth-1)) < 0.0) ) then
 569        call utl_abort('vco_setupOceanFromFile: some depth levels not in ascending order')
 570      end if
 571    end if
 572    
 573  end subroutine vco_setupOceanFromFile

 668  !--------------------------------------------------------------------------
 669  ! vco_mpiBcast
 670  !--------------------------------------------------------------------------
 671  subroutine vco_mpiBcast(vco)
 672    !
 673    !:Purpose: MPI broadcast of vertical coordinate object
 674    !
 675    implicit none
 676
 677    ! arguments:
 678    type(struct_vco), pointer, intent(inout) :: vco ! vertical coordinate object
 679
 680    ! locals:
 681    integer :: ierr, vgd_nlev_M, vgd_nlev_T
 682    integer :: vgdig1, vgdig2, vgdig3, vgdig4, vgdip1, vgdip2, vgdip3, vgddate
 683    integer :: vgdtable_dim1, vgdtable_dim2, vgdtable_dim3
 684    character(len=12) :: vgdetik
 685    real(8), pointer :: vgdtable(:,:,:)
 686
 687    nullify(vgdtable)
 688
 689    write(*,*) 'vco_mpiBcast: starting'
 690
 691    if (mmpi_myid > 0) then
 692      if (.not.associated(vco)) then
 693        allocate(vco)
 694      else 
 695        call utl_abort('vco_mpiBcast: vco must be nullified for mpi task id > 0')
 696      end if
 697    end if
 698
 699    call rpn_comm_bcast(vco%initialized , 1, 'MPI_LOGICAL', 0, 'GRID', ierr)
 700    call rpn_comm_bcast(vco%vgridPresent, 1, 'MPI_LOGICAL', 0, 'GRID', ierr)
 701    call rpn_comm_bcast(vco%nlev_T      , 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 702    call rpn_comm_bcast(vco%nlev_M      , 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 703    call rpn_comm_bcast(vco%ip1_sfc     , 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 704    call rpn_comm_bcast(vco%ip1_T_2m    , 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 705    call rpn_comm_bcast(vco%ip1_M_10m   , 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 706    call rpn_comm_bcast(vco%nlev_depth  , 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 707    call rpn_comm_bcast(vco%Vcode       , 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 708    call rpn_comm_bcast(vco%nlev_other, vnl_numvarmaxOther, 'MPI_INTEGER', 0, 'GRID', ierr)
 709    if (vco%nLev_depth > 0) then
 710      if (mmpi_myid > 0) then
 711        allocate(vco%ip1_depth(vco%nlev_depth))
 712        allocate(vco%depths(vco%nlev_depth))
 713      end if
 714      call rpn_comm_bcast(vco%ip1_depth , vco%nlev_depth, 'MPI_INTEGER', 0, 'GRID', ierr)
 715      call rpn_comm_bcast(vco%depths    , vco%nlev_depth, 'MPI_REAL8'  , 0, 'GRID', ierr)
 716    end if
 717    if (vco%vgridPresent) then
 718      if (mmpi_myid == 0) then
 719        vgd_nlev_M = size(vco%ip1_M)
 720        vgd_nlev_T = size(vco%ip1_T)
 721      end if
 722      call rpn_comm_bcast(vgd_nlev_M, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 723      call rpn_comm_bcast(vgd_nlev_T, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 724      if (mmpi_myid > 0) then
 725        allocate(vco%ip1_M(vgd_nlev_M))
 726        allocate(vco%ip1_T(vgd_nlev_T))
 727      end if
 728      if (vgd_nlev_M > 0) call rpn_comm_bcast(vco%ip1_M, vgd_nlev_M, 'MPI_INTEGER', 0, 'GRID', ierr)
 729      if (vgd_nlev_T > 0) call rpn_comm_bcast(vco%ip1_T, vgd_nlev_T, 'MPI_INTEGER', 0, 'GRID', ierr)
 730    end if
 731
 732    ! now do bcast for vgrid object
 733    if (vco%vgridPresent) then
 734
 735      if (mmpi_myid == 0) then
 736        ierr = vgd_get(vco%vgrid,'VTBL',vgdtable)
 737        vgdtable_dim1 = size(vgdtable,1)
 738        vgdtable_dim2 = size(vgdtable,2)
 739        vgdtable_dim3 = size(vgdtable,3)
 740        ierr = vgd_get(vco%vgrid,'DATE',vgddate)
 741        ierr = vgd_get(vco%vgrid,'ETIK',vgdetik)
 742        ierr = vgd_get(vco%vgrid,'IG_1',vgdig1)
 743        ierr = vgd_get(vco%vgrid,'IG_2',vgdig2)
 744        ierr = vgd_get(vco%vgrid,'IG_3',vgdig3)
 745        ierr = vgd_get(vco%vgrid,'IG_4',vgdig4)
 746        ierr = vgd_get(vco%vgrid,'IP_1',vgdip1)
 747        ierr = vgd_get(vco%vgrid,'IP_2',vgdip2)
 748        ierr = vgd_get(vco%vgrid,'IP_3',vgdip3)
 749      end if
 750
 751      ! 3D table of real*8
 752      call rpn_comm_bcast(vgdtable_dim1, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 753      call rpn_comm_bcast(vgdtable_dim2, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 754      call rpn_comm_bcast(vgdtable_dim3, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 755      if (mmpi_myid > 0) allocate(vgdtable(vgdtable_dim1, vgdtable_dim2, vgdtable_dim3)) 
 756      call rpn_comm_bcast(vgdtable, size(vgdtable), 'MPI_REAL8', 0, 'GRID', ierr)
 757      ! others
 758      call rpn_comm_bcast(vgddate, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 759      call rpn_comm_bcastc(vgdetik, len(vgdetik), 'MPI_CHARACTER', 0, 'GRID', ierr)
 760      call rpn_comm_bcast(vgdig1, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 761      call rpn_comm_bcast(vgdig2, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 762      call rpn_comm_bcast(vgdig3, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 763      call rpn_comm_bcast(vgdig4, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 764      call rpn_comm_bcast(vgdip1, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 765      call rpn_comm_bcast(vgdip2, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 766      call rpn_comm_bcast(vgdip3, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
 767      
 768      if (mmpi_myid > 0) then
 769        ierr = vgd_new(vco%vgrid,vgdtable)
 770        ierr = vgd_put(vco%vgrid,'DATE',vgddate)
 771        ierr = vgd_put(vco%vgrid,'ETIK',vgdetik)
 772        ierr = vgd_put(vco%vgrid,'IG_1',vgdig1)
 773        ierr = vgd_put(vco%vgrid,'IG_2',vgdig2)
 774        ierr = vgd_put(vco%vgrid,'IG_3',vgdig3)
 775        ierr = vgd_put(vco%vgrid,'IG_4',vgdig4)
 776        ierr = vgd_put(vco%vgrid,'IP_1',vgdip1)
 777        ierr = vgd_put(vco%vgrid,'IP_2',vgdip2)
 778        ierr = vgd_put(vco%vgrid,'IP_3',vgdip3)
 779      end if
 780      
 781      deallocate(vgdtable) 
 782
 783    end if
 784
 785    write(*,*) 'vco_mpiBcast: done'
 786
 787  end subroutine vco_mpiBcast

 789  !--------------------------------------------------------------------------
 790  ! vco_equal
 791  !--------------------------------------------------------------------------
 792  function vco_equal(vco1,vco2) result(equal)
 793    !
 794    !:Purpose: Compare two vertical grid object and provide a logical result if they are equal or not
 795    !
 796    implicit none
 797  
 798    ! arguments:
 799    type(struct_vco), pointer, intent(in) :: vco1 ! vertical coordinate object one
 800    type(struct_vco), pointer, intent(in) :: vco2 ! vertical coordinate object two
 801    ! result:
 802    logical                   :: equal
 803
 804    equal = .true.
 805
 806    equal = equal .and. (vco1%Vcode == vco2%Vcode)
 807    if (.not. equal) then
 808      write(*,*) 'vco_equal: Vcode not equal'
 809      return
 810    end if
 811    if ( vco1%vgridPresent .and. vco2%vgridPresent ) then
 812       equal = equal .and. (vco1%vgrid == vco2%vgrid)
 813       if (.not. equal) then
 814          write(*,*) 'vco_equal: vgrid not equal'
 815          return
 816       end if
 817    end if
 818
 819    ! Even if vgrid defined, not enough just to compare vgrid, must compare everything
 820    equal = equal .and. (vco1%nlev_T == vco2%nlev_T)
 821    if (.not. equal) then
 822       write(*,*) 'vco_equal: nlev_T not equal', vco1%nlev_T, vco2%nlev_T
 823       return
 824    end if
 825    equal = equal .and. (vco1%nlev_M == vco2%nlev_M)
 826    if (.not. equal) then
 827       write(*,*) 'vco_equal: nlev_M not equal', vco1%nlev_M, vco2%nlev_M
 828       return
 829    end if
 830    if (vco1%vgridPresent .and. vco2%vgridPresent .and. &
 831        vco1%nlev_T > 0 .and. vco2%nlev_T > 0) then
 832      equal = equal .and. all(vco1%ip1_T(:) == vco2%ip1_T(:))
 833      if (.not. equal) then
 834        write(*,*) 'vco_equal: ip1_T not equal'
 835        return
 836      end if
 837      equal = equal .and. all(vco1%ip1_M(:) == vco2%ip1_M(:))
 838      if (.not. equal) then
 839        write(*,*) 'vco_equal: ip1_M not equal'
 840        return
 841      end if
 842      equal = equal .and. (vco1%ip1_sfc == vco2%ip1_sfc)
 843      if (.not. equal) then
 844        write(*,*) 'vco_equal: ip1_sfc not equal'
 845        return
 846      end if
 847      if (vco1%Vcode == 5002 .or. vco1%Vcode == 5005 .or. vco1%Vcode == 5100) then
 848        equal = equal .and. hybridCoefEqualOrNot(vco1, vco2)
 849        if (.not. equal) then
 850          write(*,*) 'vco_equal: hybrid parameters are not equal'
 851          return
 852        end if
 853      end if
 854    end if
 855
 856    ! For ocean fields, check depth levels
 857    if (vco1%nLev_depth > 0) then
 858      equal = equal .and. all(vco1%depths(:) == vco2%depths(:))
 859      if (.not. equal) then
 860        write(*,*) 'vco_equal: ocean depth levels are not equal'
 861        return
 862      end if
 863    end if
 864
 865  end function vco_equal

 867  !--------------------------------------------------------------------------
 868  ! vco_subsetOrNot
 869  !--------------------------------------------------------------------------
 870  function vco_subsetOrNot(vco_template, vco_full) result(subset)
 871    !
 872    !:Purpose: This function determines if vco_template is a subset of vco_full.
 873    !
 874    implicit none
 875    
 876    ! arguments:
 877    type(struct_vco), pointer, intent(in)  :: vco_full     ! vertical coordinate object full
 878    type(struct_vco), pointer, intent(in)  :: vco_template ! vertical coordinate object template
 879    ! result:
 880    logical :: subset
 881
 882    ! locals:
 883    integer, allocatable :: THlevelWanted(:), MMlevelWanted(:)
 884
 885    !
 886    !- Compare the vCode
 887    !
 888    if (vco_template%Vcode /= vco_full%Vcode) then
 889      subset = .false.
 890      return
 891    end if
 892
 893    !
 894    !- Check if there are any thermo or momentum levels
 895    !
 896    if ( (vco_template%nlev_T == 0) .and. (vco_template%nlev_M == 0) ) then
 897      subset = .false.
 898      return
 899    end if
 900
 901    !
 902    !- Compare the IP1s
 903    !
 904    allocate(THlevelWanted(vco_template%nlev_T))
 905    allocate(MMlevelWanted(vco_template%nlev_M))
 906
 907    call vco_levelMatchingList( THlevelWanted, MMlevelWanted, & ! OUT
 908                                vco_template, vco_full )        ! IN
 909
 910    if (any(THlevelWanted == -1) .or. any(MMlevelWanted == -1)) then
 911      subset = .false.
 912      return
 913    end if
 914
 915    deallocate(MMlevelWanted)
 916    deallocate(THlevelWanted)
 917
 918    !
 919    !- For hybrid coordinates, compare additional grid parameters
 920    !
 921    if ( .not. hybridCoefEqualOrNot(vco_template, vco_full) ) then
 922      subset = .false.
 923      return
 924    end if
 925
 926    !
 927    !- When reaching this point, we assume that we have a subset
 928    !
 929    subset = .true.
 930
 931  end function vco_subsetOrNot
 932
 933  !--------------------------------------------------------------------------
 934  ! hybridCoefEqualOrNot
 935  !--------------------------------------------------------------------------
 936  function hybridCoefEqualOrNot(vco1, vco2) result(equal)
 937    !
 938    !:Purpose: To compare two vertical coordinate hybrid coefficient object 
 939    !
 940    implicit none
 941
 942    ! arguments:
 943    type(struct_vco), pointer, intent(in)  :: vco1 ! vertical coordinate object one
 944    type(struct_vco), pointer, intent(in)  :: vco2 ! vertical coordinate object two
 945    ! result:
 946    logical :: equal
 947
 948    ! locals:
 949    real(8) :: ptop1, ptop2
 950    real(8), pointer :: coefA1(:), coefA2(:)
 951    real(8), pointer :: coefB1(:), coefB2(:)
 952    real :: coefR11, coefR12
 953    real :: coefR21, coefR22
 954    integer :: stat, levIndex
 955
 956    if (vco1%Vcode == 5002) then
 957      !- Ptop
 958      stat = vgd_get(vco1%vgrid,key='PTOP - top level pressure',value=ptop1)
 959      stat = vgd_get(vco2%vgrid,key='PTOP - top level pressure',value=ptop2)
 960      if (ptop1 /= ptop2) then
 961        equal = .false.
 962        return
 963      end if
 964    end if
 965
 966    if (vco1%Vcode == 5002 .or. vco1%Vcode == 5005 .or. vco1%Vcode == 5100) then
 967
 968      !- Pref
 969      stat = vgd_get(vco1%vgrid,key='PREF - reference pressure',value=ptop1)
 970      stat = vgd_get(vco2%vgrid,key='PREF - reference pressure',value=ptop2)
 971      if (ptop1 /= ptop2) then
 972        equal = .false.
 973        return
 974      end if
 975      !- R-coef 1
 976      stat = vgd_get(vco1%vgrid,key='RC_1 - first R-coef value',value=coefR11)
 977      stat = vgd_get(vco2%vgrid,key='RC_1 - first R-coef value',value=coefR12)
 978      if (coefR11 /= coefR12) then
 979        equal = .false.
 980        return
 981      end if
 982      !- R-coef 2
 983      stat = vgd_get(vco1%vgrid,key='RC_2 - second R-coef value',value=coefR21)
 984      stat = vgd_get(vco2%vgrid,key='RC_2 - second R-coef value',value=coefR22)
 985      if (coefR21 /= coefR22) then
 986        equal = .false.
 987        return
 988      end if
 989      !- A
 990      nullify(coefA1)
 991      nullify(coefA2)
 992      stat = vgd_get(vco1%vgrid,key='CA_M - vertical A coefficient (m)',value=coefA1)
 993      stat = vgd_get(vco2%vgrid,key='CA_M - vertical A coefficient (m)',value=coefA2)
 994      if ( size(coefA1) /= size(coefA2) ) then
 995        equal = .false.
 996        return
 997      end if
 998      do levIndex = 1, size(coefA1)
 999        if (coefA1(levIndex) /= coefA2(levIndex)) then
1000          equal = .false.
1001          return
1002        end if
1003      end do
1004      !- B
1005      nullify(coefB1)
1006      nullify(coefB2)
1007      stat = vgd_get(vco1%vgrid,key='CB_M - vertical B coefficient (m)',value=coefB1)
1008      stat = vgd_get(vco2%vgrid,key='CB_M - vertical B coefficient (m)',value=coefB2)
1009      if ( size(coefB1) /= size(coefB2) ) then
1010        equal = .false.
1011        return
1012      end if
1013      do levIndex = 1, size(coefB1)
1014        if (coefB1(levIndex) /= coefB2(levIndex)) then
1015          equal = .false.
1016          return
