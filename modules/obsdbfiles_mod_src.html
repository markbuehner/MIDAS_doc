  98  subroutine readNml()
  99    !
 100    !:Purpose: Read the namelist for obsDB files
 101    !
 102    implicit none
 103
 104    ! locals
 105    integer            :: nulfile, ierr
 106    integer, external  :: fnom, fclos
 107    logical, save      :: alreadyRead = .false.
 108    integer            :: elementIndex
 109
 110    namelist /namobsdb/ numElemIdList, elemIdList
 111
 112    if ( alreadyRead ) return
 113
 114    alreadyRead = .true.
 115
 116    ! default values
 117    numElemIdList = MPC_missingValue_INT
 118    elemIdList(:) = 0
 119
 120    if ( .not. utl_isNamelistPresent('NAMOBSDB','./flnml') ) then
 121      if ( mmpi_myid == 0 ) then
 122        write(*,*) 'readNml (odbf): namObsDB is missing in the namelist.'
 123        write(*,*) '                The default values will be taken.'
 124      end if
 125    else
 126      ! reading namelist variables
 127      nulfile = 0
 128      ierr = fnom(nulfile,'./flnml','FTN+SEQ+R/O',0)
 129      read(nulfile, nml=namobsdb, iostat=ierr)
 130      if ( ierr /= 0 ) call utl_abort('readNml (odbf): Error reading namelist')
 131      ierr = fclos(nulfile)
 132      if (numElemIdList /= MPC_missingValue_INT) then
 133        call utl_abort('readNml (odbf): check namobsdb namelist section: numElemIdList should be removed')
 134      end if
 135      numElemIdList = 0
 136      do elementIndex = 1, maxElementNumber
 137        if (elemIdList(elementIndex) == 0) exit
 138        numElemIdList = numElemIdList + 1
 139      end do
 140    end if
 141    if ( mmpi_myid == 0 ) write(*, nml=namObsDb)
 142
 143    if (numElemIdList==0) then
 144      call utl_abort('odbf_setup: element list is empty')
 145    end if
 146
 147  end subroutine readNml

 152  subroutine odbf_setup()
 153    !
 154    !:Purpose: Read the namelist for obsDB files and read the obsDB column table
 155    !
 156    implicit none
 157
 158    ! locals
 159    
 160    integer            :: nulfile, ierr
 161    integer, external  :: fnom, fclos
 162    logical, save      :: alreadyRead = .false.
 163    
 164    character(len=512), parameter :: obsDbColumnFile = 'ObsDBColumnTable.dat' 
 165
 166    character(len=512) :: readLine
 167    character(len=lenSqlName) :: readDBColumn, readObsSpaceColumn, readBufrColumn  ! Strings to temporary assign 
 168                                                                                ! read header and body column names
 169    
 170    integer            :: headerTableRow, bodyTableRow, midasBodyTableRow, midasHeadTableRow ! Counters to determine the size of 
 171                                                                         ! header and body table
 172    
 173    integer            :: countRow, countMatchRow, countVarRow                        
 174    integer            :: stringFoundIndex
 175    logical            :: obsColumnIsValid
 176                                                                       
 177    if ( alreadyRead ) return
 178
 179    alreadyRead = .true.
 180
 181    call readNml()
 182
 183    ! initialize obsDb columns names to be consistent with MIDAS obsSpaceData Report column names
 184
 185    nulfile = 0 
 186    ierr = fnom(nulfile,trim(obsDbColumnFile),'FTN+SEQ+R/O',0)
 187    if ( ierr /= 0 ) call utl_abort('odbf_setup: Error reading ObsDBColumnTable file') 
 188  
 189    ! Get number of rows in ObsDBColumnTable file
 190    headerTableRow = 0
 191    bodyTableRow = 0
 192    midasBodyTableRow = 0
 193    midasHeadTableRow = 0
 194    numHeadMatch = 0
 195    numBodyMatch = 0 
 196    numMidasBodyMatch = 0
 197    numMidasHeadMatch = 0
 198    numVarNo = 0
 199    do while(ierr == 0)
 200      read(nulfile, '(A)' ,iostat = ierr) readline
 201
 202      if (index(trim(readline),'OBS_HEADER TABLE INFO BEGINS') > 0) then
 203        ! Found the start of header table in the file
 204        do while (ierr == 0)
 205          read(nulfile,'(A)' ,iostat = ierr) readline
 206          ! Stop search at the end of header table
 207          if (index(trim(readline),'OBS_HEADER TABLE INFO ENDS') > 0) exit
 208          headerTableRow = headerTableRow +1
 209          if (index(trim(readline),'[') == 0 .and. index(trim(readline),']') == 0) then 
 210            ! Count obsSpaceData header columns  
 211            numHeadMatch = numHeadMatch + 1
 212          end if
 213        end do
 214
 215      else if (index(trim(readline),'OBS_BODY TABLE INFO BEGINS') > 0) then  
 216        ! Found the start of the body table
 217        do while (ierr == 0)
 218          read(nulfile,'(A)' ,iostat = ierr) readline
 219          ! Stop search at the end of body table
 220          if (index(trim(readline),'OBS_BODY TABLE INFO ENDS') > 0) exit
 221          bodyTableRow = bodyTableRow + 1
 222          if (index(trim(readline),'[') == 0 .and. index(trim(readline),']') == 0) then 
 223            ! Count obsSpaceData body columns
 224            numBodyMatch = numBodyMatch + 1 
 225            if (index(trim(readline),'VAR') /= 0) then
 226              ! Count VarNo columns
 227              numVarNo = numVarNo + 1
 228            end if
 229          end if
 230        end do
 231
 232      else if (index(trim(readline),'MIDAS_HEADER TABLE INFO BEGINS') > 0) then
 233        ! Found the start of the MIDAS_HEADER table
 234        do while (ierr == 0)
 235          read(nulfile,'(A)' ,iostat = ierr) readline
 236          ! Stop search at the end of MIDAS_HEADER table
 237          if (index(trim(readline),'MIDAS_HEADER TABLE INFO ENDS') > 0) exit
 238          midasHeadTableRow = midasHeadTableRow + 1
 239          ! Count obsSpaceData  MIDAS_HEADER columns
 240          if (index(trim(readline),'[') == 0 .and. index(trim(readline),']') == 0) then 
 241            numMidasHeadMatch = numMidasHeadMatch + 1
 242          end if
 243        end do
 244
 245      else if (index(trim(readline),'MIDAS_BODY TABLE INFO BEGINS') > 0) then
 246        ! Found the start of the MIDAS_BODY table
 247        do while (ierr == 0)
 248          read(nulfile,'(A)' ,iostat = ierr) readline
 249          ! Stop search at the end of MIDAS_BODY table
 250          if (index(trim(readline),'MIDAS_BODY TABLE INFO ENDS') > 0) exit
 251          midasBodyTableRow = midasBodyTableRow + 1
 252          ! Count obsSpaceData MIDAS_BODY columns
 253          if (index(trim(readline),'[') == 0 .and. index(trim(readline),']') == 0) then 
 254            numMidasBodyMatch = numMidasBodyMatch + 1
 255          end if
 256        end do
 257      end if
 258    end do
 259
 260    write(*,*) 'odbf_setup: Number of obs header columns in ObsDBColumnTable', headerTableRow
 261    write(*,*) 'odbf_setup: Number of obs body columns in ObsDBColumnTable', bodyTableRow
 262    write(*,*) 'odbf_setup: Number of midas header columns in ObsDBColumnTable', midasHeadTableRow
 263    write(*,*) 'odbf_setup: Number of midas body columns in ObsDBColumnTable', midasBodyTableRow
 264    
 265    write(*,*) 'odbf_setup: numHeadMatch', numHeadMatch
 266    write(*,*) 'odbf_setup: numBodyMatch', numBodyMatch
 267    write(*,*) 'odbf_setup: numMidasHeadMatch',numMidasHeadMatch
 268    write(*,*) 'odbf_setup: numMidasBodyMatch', numMidasBodyMatch
 269
 270    ! Read Report obsdb and MIDAS obsSpaceData columns name
 271    allocate(headMatchList(2,numHeadMatch))
 272    allocate(headBufrList(numHeadMatch))
 273    allocate(bodyMatchList(2,numBodyMatch))
 274    allocate(bodyBufrList(numBodyMatch))
 275    allocate(midasBodyNamesList(2,numMidasBodyMatch))
 276    allocate(midasHeadNamesList(2,numMidasHeadMatch))
 277    allocate(varNoList(2,numVarNo))
 278
 279    rewind(nulfile)
 280
 281    ierr = 0
 282    do while (ierr == 0)
 283      read(nulfile, '(A)' ,iostat = ierr) readline
 284
 285      if (index(trim(readline),'OBS_HEADER TABLE INFO BEGINS') > 0) then
 286        ! Found the start of Obs Header table in the file  
 287        countMatchRow = 0
 288        do countRow = 1, headerTableRow
 289          ! Read all header table rows
 290          read(nulfile,*,iostat = ierr) readDBColumn, readObsSpaceColumn, readBufrColumn      
 291          select case (trim(readObsSpaceColumn))
 292            ! Assign read values to appropriate variable
 293            case ('[headTableName]')
 294              headTableName = trim(readDBColumn)
 295              ! Remove the brackets from column name string 
 296              headTableName = headTableName(1:len(headTableName)-1)
 297            case ('[headPrimaryKey]')
 298              obsHeadKeySqlName = trim(readDBColumn)
 299              ! Remove the brackets from column name string 
 300              obsHeadKeySqlName = obsHeadKeySqlName(1:len(obsHeadKeySqlName)-1)
 301            case ('[headDateSqlName]')
 302              headDateSqlName = trim(readDBColumn)
 303              ! Remove the brackets from column name string  
 304              headDateSqlName = headDateSqlName(1:len(headDateSqlName)-1)
 305            case default
 306              countMatchRow = countMatchRow + 1
 307              headMatchList(1,countMatchRow) = trim(readDBColumn)
 308              headMatchList(2,countMatchRow) = trim(readObsSpaceColumn)
 309              headBufrList(countMatchRow) = trim(readBufrColumn)
 310          end select
 311        end do
 312
 313      else if (index(trim(readline),'OBS_BODY TABLE INFO BEGINS') > 0) then
 314        ! Found the start of Obs BODY table in the file  
 315        countMatchRow = 0
 316        countVarRow = 0 
 317        do countRow = 1, bodyTableRow
 318          ! Read all body table rows 
 319          read(nulfile,*,iostat = ierr) readDBColumn, readObsSpaceColumn, readBufrColumn      
 320          select case (trim(readObsSpaceColumn))
 321            ! Assign read values to appropriate variable
 322            case ('[bodyTableName]')
 323              bodyTableName = trim(readDBColumn)
 324              ! Remove the brackets from column name string
 325              bodyTableName = bodyTableName(1:len(bodyTableName)-1)
 326            case ('[bodyPrimaryKey]')
 327              obsBodyKeySqlName = trim(readDBColumn)
 328              ! Remove the brackets from column name string
 329              obsBodyKeySqlName = obsBodyKeySqlName(1:len(obsBodyKeySqlName)-1)
 330            case default
 331              countMatchRow = countMatchRow + 1
 332              bodyMatchList(1,countMatchRow) = trim(readDBColumn)
 333              bodyMatchList(2,countMatchRow) = trim(readObsSpaceColumn)
 334              bodyBufrList(countMatchRow) = trim(readBufrColumn)
 335
 336              ! Assign varNoList if appropriate
 337              if (bodyMatchList(2,countMatchRow) == 'VAR') then
 338                countVarRow = countVarRow + 1 
 339                varNoList(1, countVarRow) = bodyMatchList(1,countMatchRow) 
 340                varNoList(2, countVarRow) = bodyBufrList(countMatchRow)
 341              end if
 342          end select
 343        end do
 344
 345      else if (index(trim(readline),'MIDAS_HEADER TABLE INFO BEGINS') > 0) then
 346        ! Found the start of MIDAS Header table in the file
 347        countMatchRow = 0
 348        do countRow = 1, midasHeadTableRow
 349          !Read all header table rows
 350          read(nulfile,*,iostat = ierr) readDBColumn, readObsSpaceColumn, readBufrColumn
 351          select case (trim(readObsSpaceColumn))
 352            case ('[midasHeadTableName]')
 353              midasHeadTableName = trim(readDBColumn)
 354              midasHeadTableName = midasHeadTableName(1:len(midasHeadTableName)-1)
 355            case ('[midasHeadPrimaryKey]')
 356              midasHeadKeySqlName = trim(readDBColumn)
 357              ! Remove the brackets from column name string
 358              midasHeadKeySqlName = midasHeadKeySqlName(1:len(midasHeadKeySqlName)-1)
 359            case default
 360              countMatchRow = countMatchRow + 1
 361              midasHeadNamesList(1,countMatchRow) = trim(readDBColumn)              
 362              midasHeadNamesList(2,countMatchRow) = trim(readObsSpaceColumn)
 363          end select
 364        end do
 365
 366      else if (index(trim(readline),'MIDAS_BODY TABLE INFO BEGINS') > 0) then
 367        ! Found the start of MIDAS Body table in the file
 368        numBodyMidasTableRequired = 0
 369        countMatchRow = 0
 370        do countRow = 1, midasBodyTableRow
 371          ! Read all body table rows
 372          read(nulfile,*,iostat = ierr) readDBColumn, readObsSpaceColumn, readBufrColumn      
 373          select case (trim(readObsSpaceColumn))
 374            ! Assign read values to appropriate variable
 375            case ('[midasBodyTableName]')
 376              midasBodyTableName = trim(readDBColumn)
 377              ! Remove the brackets from column name string
 378              midasBodyTableName = midasBodyTableName(1:len(midasBodyTableName)-1)
 379            case ('[midasBodyPrimaryKey]')
 380              midasBodyKeySqlName = trim(readDBColumn)
 381              ! Remove the brackets from column name string
 382              midasBodyKeySqlName = midasBodyKeySqlName(1:len(midasBodyKeySqlName)-1)
 383            case default
 384              countMatchRow = countMatchRow + 1
 385              midasBodyNamesList(1,countMatchRow) = trim(readDBColumn)              
 386              midasBodyNamesList(2,countMatchRow) = trim(readObsSpaceColumn)
 387             
 388              ! Check if the ObsSpaceData column name is mandatory 
 389              stringFoundIndex = index(midasBodyNamesList(2,countMatchRow),'*')
 390              if (stringFoundIndex > 0) then
 391                midasBodyNamesList(2,countMatchRow) = trim(midasBodyNamesList(2,countMatchRow)(1:stringFoundIndex-1))
 392                numBodyMidasTableRequired = numBodyMidasTableRequired + 1
 393              end if
 394          end select
 395        end do
 396      end if
 397    end do
 398
 399    ierr = fclos(nulfile)
 400 
 401    ! Check if the header, body and MIDAS table column names are read correctly 
 402    if (len(trim(headTableName)) == 0) call utl_abort('odbf_setup: headTableName is incorrectly defined or missing')
 403    if (len(trim(obsHeadKeySqlName)) == 0) call utl_abort('odbf_setup: obsHeadKeySqlName is incorrectly defined or missing')
 404    if (len(trim(headDateSqlName)) == 0) call utl_abort('odbf_setup: headDateSqlName is incorrectly defined or missing')
 405    if (len(trim(bodyTableName)) == 0) call utl_abort('odbf_setup: bodyTableName is incorrectly defined or missing')
 406    if (len(trim(obsBodyKeySqlName)) == 0) call utl_abort('odbf_setup: obsBodyKeySqlName is incorrectly defined or missing')
 407    if (len(trim(midasBodyTableName)) == 0) call utl_abort('odbf_setup: midasBodyTableName is incorrectly defined or missing')
 408    if (len(trim(midasBodyKeySqlName)) == 0) call utl_abort('odbf_setup: midasBodyKeySqlName is incorrectly defined or missing')
 409    if (len(trim(midasHeadKeySqlName)) == 0) call utl_abort('odbf_setup: midasHeadKeySqlName is incorrectly defined or missing')
 410
 411    do countRow = 1, numHeadMatch
 412      obsColumnIsValid = obs_isColumnNameValid(trim(headMatchList(2,countRow)))
 413      if (.not. obsColumnIsValid) call utl_abort('odbf_setup: Column in Header Table does not exist in ObsSpaceData: '&
 414             //trim(headMatchList(2,countRow)))
 415    end do
 416      
 417    do countRow = 1, numBodyMatch 
 418      obsColumnIsValid = obs_isColumnNameValid(trim(bodyMatchList(2,countRow)))
 419      if (.not. obsColumnIsValid) call utl_abort('odbf_setup: Column in Body Table does not exist in ObsSpaceData: '&
 420             //trim(bodyMatchList(2,countRow)))
 421    end do
 422
 423    do countRow = 1, numMidasBodyMatch 
 424      obsColumnIsValid = obs_isColumnNameValid(trim(midasBodyNamesList(2,countRow)))
 425      if (.not. obsColumnIsValid) call utl_abort('odbf_setup: Column in MIDAS Body Table does not exist in ObsSpaceData: '&
 426             //trim(midasBodyNamesList(2,countRow)))
 427    end do
 428
 429    do countRow = 1, numMidasHeadMatch 
 430      obsColumnIsValid = obs_isColumnNameValid(trim(midasHeadNamesList(2,countRow)))
 431      if (.not. obsColumnIsValid) call utl_abort('odbf_setup: Column in MIDAS Header Table does not exist in ObsSpaceData: '&
 432             //trim(midasHeadNamesList(2,countRow)))
 433    end do
 434
 435  end subroutine odbf_setup

 440  subroutine odbf_getDateStamp(dateStamp, fileName)
 441    !
 442    ! Purpose: get dateStamp from an obsDB file
 443    !
 444    implicit none
 445    
 446    ! arguments
 447    integer         , intent(out) :: dateStamp
 448    character(len=*), intent(in)  :: fileName
 449    
 450    ! locals
 451    integer,    allocatable :: headDateValues(:), headTimeValues(:)
 452    integer                 :: ier, imode, validTime, validDate, validDateRecv, validTimeRecv
 453    integer                 :: newdate
 454
 455    call odbf_setup()
 456
 457    call sqlu_getColumnValuesDateStr(headDateValues, headTimeValues, fileName=trim(fileName), &
 458                                     tableName=headTableName, sqlColumnName=headDateSqlName)
 459
 460    validDate = MPC_missingValue_INT 
 461    validTime = MPC_missingValue_INT 
 462
 463    call tim_getValidDateTimeFromList(headDateValues, headTimeValues, validDate, validTime)
 464
 465    ! Make sure all mpi tasks have a valid date (important for split sqlite files)
 466    call rpn_comm_allreduce(validDate, validDateRecv, 1, "MPI_INTEGER", "MPI_MAX", "GRID", ier)
 467    call rpn_comm_allreduce(validTime, validTimeRecv, 1, "MPI_INTEGER", "MPI_MAX", "GRID", ier)
 468    
 469    if (validDateRecv == MPC_missingValue_INT .or. validTimeRecv == MPC_missingValue_INT) then
 470      call utl_abort('odbf_getDateStamp: Error in getting valid date and time!')
 471    end if
 472    
 473    ! printable to stamp, validTime must be multiplied with 1e6 to make newdate work
 474    imode = 3
 475    ier = newdate(dateStamp, validDateRecv, validTimeRecv * 1000000, imode)
 476    write(*,*)'odbf_getDateStamp: obsDB files valid date (YYYYMMDD): ', validDateRecv
 477    write(*,*)'odbf_getDateStamp: obsDB files valid time       (HH): ', validTimeRecv
 478    write(*,*)'odbf_getDateStamp: obsDB files dateStamp            : ', datestamp
 479
 480    deallocate(headDateValues)
 481    deallocate(headTimeValues)
 482
 483  end subroutine odbf_getDateStamp

 488  subroutine odbf_readFile(obsdat, fileName, familyType, fileIndex)
 489    !
 490    !:Purpose: Read the contents of an obsDB file and put in obsSpaceData
 491    !
 492    implicit none
 493
 494    ! arguments:
 495    type (struct_obs), intent(inout) :: obsdat
 496    character(len=*)                 :: fileName
 497    character(len=*)                 :: familyType
 498    integer                          :: fileIndex
 499
 500    ! locals:
 501    integer :: bodyIndex, bodyIndexBegin, bodyIndexEnd, headIndexBegin
 502    integer :: headIndexEnd, headIndex, obsRln
 503    integer :: numBody, numHead, columnIndex
 504    integer :: headTableIndex, numRowsHeadTable, bodyTableIndex, numRowsBodyTable
 505    character(len=lenSqlName), allocatable :: headCharSqlNames(:)
 506    character(len=lenSqlName), allocatable :: headSqlNames(:),    bodySqlNames(:)
 507    character(len=50),         allocatable :: headCharValues(:,:)
 508    integer,                   allocatable :: headDateValues(:), headTimeValues(:)
 509    real(8),                   allocatable :: headValues(:,:), bodyValues(:,:)
 510    integer(8),                allocatable :: headPrimaryKey(:)
 511    integer(8),                allocatable :: bodyPrimaryKey(:), bodyHeadKey(:)
 512
 513    write(*,*)
 514    write(*,*) 'odbf_readFile: Starting'
 515    write(*,*)
 516    write(*,*) 'odbf_readFile: FileName   : ', trim(FileName)
 517    write(*,*) 'odbf_readFile: FamilyType : ', FamilyType
 518
 519    !- 0.0 Some initialization
 520    call odbf_setup()
 521    call ovt_setup(elemIdList(1:numElemIdList))
 522
 523    !- 1.0 Determine names of columns present in obsDB file
 524
 525    call sqlu_getSqlColumnNames(headCharSqlNames, fileName=trim(fileName), &
 526                                tableName=headTableName, dataType='varchar')
 527    call sqlu_getSqlColumnNames(headSqlNames, fileName=trim(fileName), &
 528                                tableName=headTableName, dataType='numeric' )
 529    call sqlu_getSqlColumnNames(bodySqlNames, fileName=trim(fileName), &
 530                                tableName=bodyTableName, dataType='numeric')
 531
 532    ! Print all of the column names to the listing
 533    do columnIndex = 1, size(headCharSqlNames)
 534      write(*,*) 'odbf_readFile: headCharSqlNames =', columnIndex, &
 535                 trim(headCharSqlNames(columnIndex))
 536    end do
 537    do columnIndex = 1, size(headSqlNames)
 538      write(*,*) 'odbf_readFile: headSqlNames =', columnIndex, &
 539                 trim(headSqlNames(columnIndex))
 540    end do
 541    do columnIndex = 1, size(bodySqlNames)
 542      write(*,*) 'odbf_readFile: bodySqlNames   =', columnIndex, &
 543                 trim(bodySqlNames(columnIndex))
 544    end do
 545
 546    !- 1.1 Read most of the contents of the file into local tables
 547
 548    call odbf_getPrimaryKeys(headPrimaryKey, bodyPrimaryKey, bodyHeadKey, &
 549                             fileName=trim(fileName))
 550

 678  ! odbf_readMidasBodyTable
 679  !--------------------------------------------------------------------------
 680  subroutine odbf_readMidasBodyTable(obsdat, fileName, familyType, fileIndex)
 681    !
 682    !:Purpose: Read values from any column found the MIDAS table, if it
 683    !           already exists in the file. This will replace any existing
 684    !           values read from the original obs-DB tables (e.g. the obs
 685    !           value).
 686    !
 687    implicit none
 688
 689    ! arguments:
 690    type(struct_obs), intent(inout) :: obsdat
 691    character(len=*), intent(in)    :: fileName   
 692    character(len=*), intent(in)    :: familyType
 693    integer,          intent(in)    :: fileIndex
 694
 695    ! locals:
 696    type(fSQL_STATUS)    :: stat ! sqlite error status
 697    type(fSQL_DATABASE)  :: db   ! sqlite file handle
 698    type(fSQL_STATEMENT) :: stmt ! precompiled sqlite statements
 699    integer(8)           :: obsIdd
 700    integer              :: obsIdf, obsVarNo, sqlNameIndex, matdata_int(1,1)
 701    integer              :: headIndex, bodyIndex, bodyIndexBegin, bodyIndexEnd
 702    integer              :: obsSpaceColIndex, ierr, numRows, numColumns
 703    real(8)              :: obsPPP, matdata_r8(1,1)
 704    character(len=4)     :: obsSpaceColumnName
 705    character(len=lenSqlName) :: sqlColumnName, vnmSqlName, pppSqlName, varSqlName
 706    character(len=3000)  :: query
 707    logical              :: midasBodyTableExists
 708    logical, allocatable :: midasColumnExists(:)
 709
 710    write(*,*)
 711    write(*,*) 'odbf_readMidasBodyTable: Starting'
 712    write(*,*)
 713    write(*,*) 'odbf_readMidasBodyTable: FileName   : ', trim(FileName)
 714    write(*,*) 'odbf_readMidasBodyTable: FamilyType : ', FamilyType
 715
 716    ! check if midasTable already exists in the file
 717    midasBodyTableExists = sqlu_sqlTableExists(fileName, midasBodyTableName)
 718
 719    if (.not. midasBodyTableExists) then
 720      write(*,*) 'odbf_readMidasBodyTable: MIDAS table not present in file'
 721      return
 722    else
 723      write(*,*) 'odbf_readMidasBodyTable: MIDAS table present in file, will read contents'
 724    end if
 725
 726    ! some sql column names
 727    vnmSqlName = odbf_midasTabColFromObsSpaceName('VNM', midasBodyNamesList)
 728    pppSqlName = odbf_midasTabColFromObsSpaceName('PPP', midasBodyNamesList)
 729    varSqlName = odbf_midasTabColFromObsSpaceName('VAR', midasBodyNamesList)
 730
 731    ! check which columns exist in the MIDAS output table
 732    allocate(midasColumnExists(numMidasBodyMatch))
 733    do sqlNameIndex = 1, numMidasBodyMatch
 734      sqlColumnName = midasBodyNamesList(1,sqlNameIndex)
 735      midasColumnExists(sqlNameIndex) = &
 736           sqlu_sqlColumnExists(fileName, midasBodyTableName, sqlColumnName)
 737    end do
 738
 739    ! open the obsDB file
 740    call fSQL_open( db, trim(fileName), stat )
 741    if ( fSQL_error(stat) /= FSQL_OK ) then
 742      write(*,*) 'odbf_readMidasBodyTable: fSQL_open: ', fSQL_errmsg(stat)
 743      call utl_abort( 'odbf_readMidasBodyTable: fSQL_open' )
 744    end if
 745    
 746    ! read the contents of the MIDAS table, one column at a time
 747    SQLNAME: do sqlNameIndex = 1, numMidasBodyMatch
 748
 749      ! skip this sql column name if it is not present in the file
 750      if (.not. midasColumnExists(sqlNameIndex)) cycle SQLNAME
 751
 752      ! get obsSpaceData column name and index corresponding to sql column
 753      sqlColumnName      = midasBodyNamesList(1,sqlNameIndex)
 754      obsSpaceColumnName = midasBodyNamesList(2,sqlNameIndex)
 755      ierr = clib_toUpper(obsSpaceColumnName)
 756      obsSpaceColIndex = obs_columnIndexFromName(trim(obsSpaceColumnName))
 757
 758      ! skip this sql column name if the related obsSpaceData column is not active
 759      if (obs_columnDataType(obsSpaceColIndex) == 'real') then
 760        if (.not. obs_columnActive_RB(obsdat, obsSpaceColIndex)) cycle SQLNAME
 761      else
 762        if (.not. obs_columnActive_IB(obsdat, obsSpaceColIndex)) cycle SQLNAME
 763      end if
 764
 765      write(*,*) 'odbf_readMidasBodyTable: reading midasTable column: ', &
 766                 trim(sqlColumnName)
 767      write(*,*) 'odbf_readMidasBodyTable: to update obsSpaceData column: ', &
 768                 trim(obsSpaceColumnName)
 769
 770      ! prepare sql update query
 771      query = 'select ' // trim(sqlColumnName) // &
 772              ' from ' // trim(midasBodyTableName) // &
 773              ' where ' // &
 774              trim(obsBodyKeySqlName) // ' = ? and '   // &
 775              trim(vnmSqlName)     // ' = ? and '   // &
 776              trim(pppSqlName)     // ' = ? ;'
 777      write(*,*) 'odbf_readMidasBodyTable: query ---> ', trim(query)
 778
 779      call fSQL_prepare( db, query , stmt, stat )
 780      if ( fSQL_error(stat) /= FSQL_OK ) then
 781        write(*,*) 'odbf_readMidasBodyTable: fSQL_prepare: ', fSQL_errmsg(stat)
 782        call utl_abort( 'odbf_readMidasBodyTable: fSQL_prepare' )
 783      end if
 784
 785      call fSQL_begin(db)
 786      HEADER2: do headIndex = 1, obs_numHeader(obsdat)
 787
 788        obsIdf = obs_headElem_i( obsdat, OBS_IDF, headIndex )
 789        if ( obsIdf /= fileIndex ) cycle HEADER2
 790
 791        bodyIndexBegin = obs_headElem_i( obsdat, OBS_RLN, headIndex )
 792        bodyIndexEnd = bodyIndexBegin + &
 793                       obs_headElem_i( obsdat, OBS_NLV, headIndex ) - 1
 794
 795        BODY2: do bodyIndex = bodyIndexBegin, bodyIndexEnd
 796
 797          ! execute the sql query to select the desired value
 798          obsIdd  = obs_bodyPrimaryKey( obsdat, bodyIndex )
 799          call fSQL_bind_param(stmt, PARAM_INDEX=1, INT8_VAR=obsIdd)
 800          obsVarNo  = obs_bodyElem_i( obsdat, obs_vnm, bodyIndex )
 801          call fSQL_bind_param(stmt, PARAM_INDEX=2, INT_VAR=obsVarNo)
 802          obsPPP  = obs_bodyElem_r( obsdat, obs_ppp, bodyIndex )
 803          call fSQL_bind_param(stmt, PARAM_INDEX=3, REAL8_VAR=obsPPP)
 804          call fSQL_exec_stmt(stmt)
 805
 806          ! read the real or integer value
 807          if (obs_columnDataType(obsSpaceColIndex) == 'real') then
 808            call fSQL_get_many( stmt, nrows=numRows, ncols=numColumns, &
 809                 mode=FSQL_REAL8, status=stat )
 810            if ( fSQL_error(stat) /= FSQL_OK ) then
 811              write(*,*) 'odbf_readMidasBodyTable: fSQL_get_many: ', fSQL_errmsg(stat)
 812              call utl_abort('odbf_readMidasBodyTable: problem with fSQL_get_many')
 813            end if
 814            if (numRows /= 1 .or. numColumns /= 1) then
 815              write(*,*) 'odbf_readMidasBodyTable: numRows, numColumns =', numRows, numColumns
 816              call utl_abort('odbf_readMidasBodyTable: sql query did not return 1 value')
 817            end if
 818            call fSQL_fill_matrix ( stmt, matdata_r8 )
 819            call obs_bodySet_r(obsdat,obsSpaceColIndex,bodyIndex,matdata_r8(1,1))
 820          else
 821            call fSQL_get_many( stmt, nrows=numRows, ncols=numColumns, &
 822                                mode=FSQL_INT, status=stat )
 823            if ( fSQL_error(stat) /= FSQL_OK ) then
 824              write(*,*) 'odbf_readMidasBodyTable: fSQL_get_many: ', fSQL_errmsg(stat)
 825              call utl_abort('odbf_readMidasBodyTable: problem with fSQL_get_many')
 826            end if
 827            if (numRows /= 1 .or. numColumns /= 1) then
 828              write(*,*) 'odbf_readMidasBodyTable: numRows, numColumns =', numRows, numColumns
 829              call utl_abort('odbf_readMidasBodyTable: sql query did not return 1 value')
 830            end if
 831            call fSQL_fill_matrix ( stmt, matdata_int )
 832            call obs_bodySet_i(obsdat,obsSpaceColIndex,bodyIndex,matdata_int(1,1))
 833          end if
 834
 835          call fSQL_free_mem( stmt )
 836
 837        end do BODY2
 838
 839      end do HEADER2
 840
 841      call fSQL_finalize( stmt )
 842      call fSQL_commit( db )
 843
 844    end do SQLNAME
 845
 846    ! close the obsDB file
 847    call fSQL_close( db, stat ) 
 848
 849    deallocate(midasColumnExists)
 850
 851    write(*,*)
 852    write(*,*) 'odbf_readMidasBodyTable: finished'
 853    write(*,*)
 854
 855  end subroutine odbf_readMidasBodyTable
