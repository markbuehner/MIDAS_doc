 44  subroutine vqc_setup(obsSpaceData)
 45    !
 46    !:Purpose: To set certain parameters for the asymmetric check
 47    !           and for variational quality control
 48    !
 49
 50    implicit none
 51
 52    type(struct_obs) :: obsSpaceData ! obsSpaceData object
 53
 54    integer jdata, jjo, idata, idatend, idburp
 55    integer ityp, iass, iother, jj, istyp, ilev
 56
 57    real(8) zagz, zahu, zatt, zauv, zabt, zabtb, zapn, zaps, zazd, zach, zatm, zaice, zapr
 58    real(8) zdgz, zdhu, zdtt, zduv, zdbt, zdbtb, zdpn, zdps, zdzd, zdch, zdtm, zdice, zdpr
 59    real(8) zattra, zauvra, zattsym, zdvis, zavis
 60    real(8) zslev, zlev, zval, zspdo, zspdf, zofcst, zoval, zdiff, zaasym, zoer
 61    real(8) zfcst, zlat, zlon, zprior
 62
 63    logical llok
 64
 65    !
 66    !_____prior probabilities for winds:ZAUV
 67    !     prior probabilities for scalar variables: zagz and zahu
 68    !     standard deviation multiple for background check for winds: zduv
 69    !     standard deviation multiple for background check for heights: zdgz
 70    !     standard deviation multiple for background check for humidity: zdhu
 71    !
 72    zagz  = 1.0d-12
 73!    zahu  = 1.0d-2
 74    zahu  = 5.0d-2
 75    zatt  = 5.0d-2
 76!    zauv  = 1.0d-2
 77    zauv  = 2.0d-2
 78    zabt  = 1.0d-1
 79    zabtb = 1.0d-1
 80    zapn  = 1.0d-4
 81    zaps  = 1.0d-4
 82    zazd  = 2.0d-2
 83    zach  = 1.0d-3
 84    zatm  = 1.0d-12
 85    zaice = 0.0d0
 86    zapr  = 0.0d0
 87
 88    zattra = 0.005d0
 89    zattsym = 1.d-1
 90    zauvra = 1.d-5
 91    zavis= 1.0d-3
 92
 93    zdgz  = 5.d0
 94    zdhu  = 5.d0
 95    zdtt  = 5.d0
 96    zduv  = 5.d0
 97    zdbt  = 3.d0
 98    zdbtb = 3.d0
 99    zdpn  = 5.d0
100    zdps  = 5.d0
101    zdzd  = 3.d0
102    zdch  = 1.d1
103    zdtm  = 5.d0
104    zdice = 5.d2
105    zdpr  = 5.d2
106    zdvis= 5.d0
107
108    do jjo = 1, obs_numheader(obsSpaceData)
109
110       IDATA   = obs_headElem_i( obsSpaceData, OBS_RLN, jjo )
111       IDATEND = obs_headElem_i( obsSpaceData, OBS_NLV, jjo ) + IDATA - 1
112       IDBURP  = obs_headElem_i( obsSpaceData, OBS_ITY, jjo )
113       ZLAT    = obs_headElem_r( obsSpaceData, OBS_LAT, jjo ) * MPC_DEGREES_PER_RADIAN_R8
114       ZLON    = obs_headElem_r( obsSpaceData, OBS_LON, jjo ) * MPC_DEGREES_PER_RADIAN_R8
115
116       do JDATA = IDATA, IDATEND
117
118          ityp  = obs_bodyElem_i( obsSpaceData, OBS_VNM, JDATA )
119          IASS  = obs_bodyElem_i( obsSpaceData, OBS_ASS, JDATA )
120          ZLEV  = obs_bodyElem_r( obsSpaceData, OBS_PPP, JDATA ) * MPC_MBAR_PER_PA_R8
121          ZOER  = obs_bodyElem_r( obsSpaceData, OBS_OER, JDATA )
122          ZVAL  = obs_bodyElem_r( obsSpaceData, OBS_VAR, JDATA )
123
124          ZFCST = ZVAL - obs_bodyElem_r( obsSpaceData, OBS_OMP,JDATA)
125
126          if (ityp == BUFR_NETS .or. ityp == BUFR_NEPS .or.  &
127              ityp == BUFR_NEPN .or. ityp == BUFR_NESS .or.  &
128              ityp == BUFR_NEUS .or. ityp == BUFR_NEVS .or.  &
129              ityp == BUFR_NEZD .or. ityp == bufr_vis  .or.  &
130              ityp == bufr_logVis .or. ityp == bufr_gust .or. &
131              ityp == bufr_radarPrecip .or. ityp == bufr_logRadarPrecip ) then
132             LLOK = (obs_bodyElem_i(obsSpaceData,OBS_ASS,JDATA) == obs_assimilated)
133          else
134             LLOK = (IASS == obs_assimilated) .and. ((obs_bodyElem_i(obsSpaceData,OBS_XTR,JDATA) ==0) &
135                .or. ((obs_bodyElem_i(obsSpaceData,OBS_XTR,JDATA) == 2) .and. &
136                      (obs_bodyElem_i(obsSpaceData,OBS_VNM,JDATA) == BUFR_NEGZ)))
137          end if
138
139          if (LLOK) then
140             if (ityp == BUFR_NEUU .or. ityp == BUFR_NEVV .or.  &
141                 ityp == BUFR_NEUS .or. ityp == BUFR_NEVS) then
142                ZAASYM = 1.0d0
143                IOTHER = -1
144                if (ityp == BUFR_NEUU .or. ityp == BUFR_NEUS) then
145                  do JJ = IDATA, JDATA
146                    ISTYP = obs_bodyElem_i( obsSpaceData, OBS_VNM, JJ )
147                    ZSLEV = obs_bodyElem_r( obsSpaceData, OBS_PPP, JJ ) * MPC_MBAR_PER_PA_R8
148                    if ((ISTYP == BUFR_NEVV .or. ISTYP == BUFR_NEVS)  &
149                         .and. ZLEV == ZSLEV) then
150                      IOTHER = JJ
151                    end if
152                  end do
153                else
154                  do JJ=IDATA,JDATA
155                    ISTYP = obs_bodyElem_i(obsSpaceData,OBS_VNM,JJ)
156                    ZSLEV = obs_bodyElem_r(obsSpaceData,OBS_PPP,JJ)*MPC_MBAR_PER_PA_R8
157                    if ((ISTYP == BUFR_NEUU .or. ISTYP == BUFR_NEUS)  &
158                         .and. ZLEV == ZSLEV) then
159                      IOTHER = JJ
160                    end if
161                  end do
162                end if
163                if (IOTHER /= -1) then
164                  ZOER = obs_bodyElem_r(obsSpaceData,OBS_OER,JDATA)
165                  ZOVAL = obs_bodyElem_r(obsSpaceData,OBS_VAR,IOTHER)
166                  ZOFCST = ZOVAL-obs_bodyElem_r(obsSpaceData,OBS_OMP,IOTHER)
167                  ZSPDO = SQRT(ZOVAL*ZOVAL + ZVAL*ZVAL)
168                  ZSPDF = SQRT(ZOFCST*ZOFCST + ZFCST*ZFCST)
169                  ZDIFF = ZSPDO - ZSPDF
170                  ILEV = NINT(ZLEV)
171                  !
172                  !___tighten rejection criterion for satob winds
173                  !
174                  if (IDBURP == 88 .or. IDBURP == 188) then
175                    if (ZDIFF < 0.0d0 .and. ABS(ZLAT) > 20.d0 .and.  &
176                         ILEV < 550) then
177                       ZAASYM = 0.7d0*MAX(ZSPDF,1.0d0)
178                       ZPRIOR = ZAASYM*ZAUV
179                       if (ZPRIOR > 0.99d0) ZAASYM = 0.99d0/ZAUV
180                    else
181                       !
182                       !___zaasym used to specify new criterion for satwinds
183                       !   than are not included in the asymmetric test
184                       !
185                       ZAASYM = 10.d0
186                    end if
187                  end if
188
189                end if
190                !
191                !__INITIAL VALUE OF GAMMA FOR QCVAR (WINDS)
192                !
193                call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,(1.d0 -  &
194                   (1.d0-(ZAUV*ZAASYM))*(1.d0-(ZAUV*ZAASYM)))*(2.d0*MPC_PI_r8)/  &
195                  ((1.d0-(ZAUV*ZAASYM))*(1.d0-(ZAUV*ZAASYM))*  &
196                                    (2.d0*ZDUV)*(2.d0*ZDUV)))
197                if ((IDBURP >= 32  .and. IDBURP <= 38) .or.  &
198                   (IDBURP >= 135 .and. IDBURP <= 142) .or.  &
199                   (IDBURP == 132) )  &
200                call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,  &
201                       (1.d0 - (1.d0-ZAUVRA)*(1.d0-ZAUVRA))  &
202                       *(2.d0*MPC_PI_R8)/((1.d0-ZAUVRA)*(1.d0-ZAUVRA)*  &
203                                   (2.d0*ZDUV)*(2.d0*ZDUV)))
204             else if (ityp == BUFR_NEGZ) then
205                !
206                ! INITIAL VALUE OF GAMMA FOR QCVAR (HEIGHTS)
207                !
208                call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,(ZAGZ*  &
209                    SQRT(2.d0*MPC_PI_R8))/((1.d0-ZAGZ)*(2.d0*ZDGZ)))
210             else if (ityp == BUFR_NETT) then
211                !
212                ! INITIAL VALUE OF GAMMA FOR QCVAR (TEMPERATURES)
213                !
214                call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,(ZATT*  &
215                    SQRT(2.d0*MPC_PI_R8))/((1.d0-ZATT)*(2.d0*ZDTT)))
216                if ((IDBURP >= 32  .and. IDBURP <= 38) .or.  &
217                    (IDBURP >= 135 .and. IDBURP <= 142) .or.  &
218                    (IDBURP == 132) )  &
219                  call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,(ZATTRA*  &
220                    SQRT(2.d0*MPC_PI_R8))/((1.d0-ZATTRA)*(2.d0*ZDTT)))
221             else if (ityp == BUFR_NETS) then
222                !
223                ! INITIAL VALUE OF GAMMA FOR QCVAR (SCREEN-LEVEL TEMPERATURES)
224                !
225                call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,(ZATT*  &
226                    SQRT(2.d0*MPC_PI_R8))/((1.d0-ZATT)*(2.d0*ZDTT)))
227                !
228                ! ASYMMETRIC TEST FOR SHIP TEMPERATURES
229                !
230                if ((IDBURP == 13) .and. (ZVAL > ZFCST)) then
231                  call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,(ZATTSYM*  &
232                      SQRT(2.d0*MPC_PI_R8))/((1.d0-ZATTSYM)*(2.d0*ZDTT)))
233                end if
234             else if (ityp == BUFR_NEPN) then
235                !
236                ! INITIAL VALUE OF GAMMA FOR QCVAR (MSL PRESSURE)
237                !
238                call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,(ZAPN*  &
239                    SQRT(2.d0*MPC_PI_R8))/((1.d0-ZAPN)*(2.d0*ZDPN)))
240             else if (ityp == BUFR_NEPS) then
241                !
242                ! INITIAL VALUE OF GAMMA FOR QCVAR (STATION PRESSURE)
243                !
244                call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,(ZAPS*  &
245                    SQRT(2.d0*MPC_PI_R8))/((1.d0-ZAPS)*(2.d0*ZDPS)))
246             else if ( ityp == 12062 .or. ityp == 12063 .or.  &
247                      ityp == 12163) then
248                !
249                ! INITIAL VALUE OF GAMMA FOR BRIGHTNESS TEMPERATURES
250                ! TOVS AMSU-A + AIRS + IASI + GEORAD !!!!! 
251                !
252                call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,(ZABT*  &
253                    SQRT(2.d0*MPC_PI_R8))/((1.d0-ZABT)*(2.d0*ZDBT)))
254                if (IDBURP == 181 .or. IDBURP == 182 .or. &
255                    IDBURP == 200) then
256                  !
257                  ! INITIAL VALUE OF GAMMA FOR TOVS AMSU-B (181) AND MHS (182)
258                  ! AND MWHS2(200)
259                  !
260                  call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,(ZABTB*  &
261                        SQRT(2.d0*MPC_PI_R8))/((1.d0-ZABTB)*(2.d0*ZDBTB)))
262                end if
263             else if (ityp == BUFR_NEZD) then
264                !
265                ! INITIAL VALUE OF GAMMA FOR QCVAR (GPS ZENITH DELAY)
266                !
267                call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,(ZAZD*  &
268                    SQRT(2.d0*MPC_PI_R8))/((1.d0-ZAZD)*(2.d0*ZDZD)))
269             else if (bufr_IsAtmosConstituent(ityp)) then
270                !
271                ! INITIAL VALUE OF GAMMA FOR THE CH (constituents) family
272                !
273                  call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,(ZACH* &
274                       SQRT(2.d0*MPC_PI_R8))/((1.d0-ZACH)*(2.d0*ZDCH)))
275             else if (ityp == bufr_sst ) then
276                !
277                ! INITIAL VALUE OF GAMMA FOR SST
278                !
279                call obs_bodySet_r( obsSpaceData, OBS_POB, JDATA, ( zatm *  &
280                    sqrt( 2.d0 * MPC_PI_R8 )) / (( 1.d0 - zatm ) * ( 2.d0 * zdtm )))
281             else if (ityp == BUFR_ICEC ) then
282                !
283                ! INITIAL VALUE OF GAMMA FOR SEA ICE
284                !
285                call obs_bodySet_r( obsSpaceData, OBS_POB, JDATA, ( zaice *  &
286                    sqrt( 2.d0 * MPC_PI_R8 )) / (( 1.d0 - zaice ) * ( 2.d0 * zdice )))
287             else if (ityp == BUFR_radarPrecip .or. ityp == BUFR_logRadarPrecip ) then
288                !
289                ! INITIAL VALUE OF GAMMA FOR RADAR PRECIPITATION
290                !
291                call obs_bodySet_r( obsSpaceData, OBS_POB, JDATA, ( zapr *  &
292                    sqrt( 2.d0 * MPC_PI_R8 )) / (( 1.d0 - zapr ) * ( 2.d0 * zdpr )))
293             else if (ityp == bufr_gust ) then
294                !
295                ! INITIAL VALUE OF GAMMA FOR WIND GUST
296                !
297                call obs_bodySet_r( obsSpaceData, OBS_POB, JDATA, ( zauv *  &
298                    sqrt( 2.d0 * MPC_PI_R8 )) / (( 1.d0 - zauv ) * ( 2.d0 * zduv )))
299              else if (ityp == bufr_vis .or. ityp == bufr_logVis) then
300                !
301                ! INITIAL VALUE OF GAMMA FOR VISIBILITY
302                !
303                call obs_bodySet_r( obsSpaceData, OBS_POB, JDATA, ( zavis *  &
304                    sqrt( 2.d0 * MPC_PI_R8 )) / (( 1.d0 - zavis ) * ( 2.d0 * zdvis )))
305             else
306                !
307                ! INITIAL VALUE OF GAMMA FOR QCVAR (OTHERS)
308                !
309                call obs_bodySet_r(obsSpaceData,OBS_POB,JDATA,(ZAHU*  &
310                    SQRT(2.d0*MPC_PI_R8))/((1.d0-ZAHU)*(2.d0*ZDHU)))
311             end if
312          end if
313       end do
314
315    end do
316
317  end subroutine vqc_setup

320  subroutine vqc_NlTl(obsSpaceData)
321    !
322    !:Purpose: 1) Modify Jo [OBS_JOBS] according to
323    !              Andersson and Jarvinen 1999, Variational quality control,
324    !              Q.J.R., 125, pp. 697-722.
325    !           2) Save the values of (1-Wqc) in OBS_QCV
326    !              for gradient factorization and postalt flag criterion.
327    !
328    implicit none
329
330    type(struct_obs) :: obsSpaceData ! obsSpaceData object
331
332    integer :: index_body,istyp,jj,index_header,ityp,index_body_start
333    real*8 :: zgami,zjon,zqcarg,zppost,zlev,zslev
334    logical :: lluv
335    logical :: includeFlag
336
337    BODY: do index_body = 1, obs_numbody(obsSpaceData)
338      includeFlag = (obs_bodyElem_i(obsSpaceData,OBS_ASS,index_body) == obs_assimilated).and.  &
339                    (obs_getFamily(obsSpaceData,bodyIndex=index_body).ne.'RO')
340      ! pas de qcvar pour  les radiances en mode matrice R non diagonale
341      if (rmat_lnondiagr) includeFlag = includeFlag .and.  &
342        (obs_getFamily(obsSpaceData,bodyIndex=index_body) /= 'TO') 
343
344      if (includeFlag) then
345        index_header = obs_bodyElem_i(obsSpaceData,OBS_HIND,index_body)
346        index_body_start = obs_headElem_i(obsSpaceData,OBS_RLN,INDEX_HEADER)
347        ZLEV = obs_bodyElem_r(obsSpaceData,OBS_PPP,INDEX_BODY)
348        zgami = obs_bodyElem_r(obsSpaceData,OBS_POB,index_body)
349        ityp = obs_bodyElem_i(obsSpaceData,OBS_VNM,INDEX_BODY)
350        LLUV = ((ityp == BUFR_NEUU .or. ityp == BUFR_NEUS) .and.  &
351               col_varExist(varName='UU')) .or. ((ityp == BUFR_NEVV .or.  &
352               ityp == BUFR_NEVS).and.col_varExist(varName='VV'))
353        if (LLUV) then
354          if (ityp == BUFR_NEUU .or. ityp == BUFR_NEUS)then
355            !
356            ! In order to calculate the contribution to Jo from a wind, the o-a
357            ! must be available for both u and v components. Hence, loop over only
358            ! data for which o-a has already been calculated
359            !
360            do JJ=INDEX_BODY_START, INDEX_BODY
361              ISTYP = obs_bodyElem_i(obsSpaceData,OBS_VNM,JJ)
362              ZSLEV = obs_bodyElem_r(obsSpaceData,OBS_PPP,JJ)
363              if ((ISTYP == BUFR_NEVV .or.  &
364                   ISTYP == BUFR_NEVS) .and.  &
365                   ZSLEV == ZLEV) then
366                ZJON=obs_bodyElem_r(obsSpaceData,OBS_JOBS,INDEX_BODY)+  &
367                     obs_bodyElem_r(obsSpaceData,OBS_JOBS,JJ)
368                ZQCARG = ZGAMI + EXP(-1.0D0*ZJON)
369                ZPPOST = ZGAMI/ZQCARG
370                !
371                ! Store the value of o-a multiplied by one minus the posterior
372                ! probability of gross error (needed for the adjoint calculations)
373                !
374                call obs_bodySet_r(obsSpaceData,OBS_QCV,INDEX_BODY, ZPPOST)
375                call obs_bodySet_r(obsSpaceData,OBS_QCV,JJ, ZPPOST)
376
377                call obs_bodySet_r(obsSpaceData,OBS_JOBS,INDEX_BODY,-LOG(ZQCARG/(ZGAMI+1.D0))/2.D0)
378                call obs_bodySet_r(obsSpaceData,OBS_JOBS,JJ, -LOG(ZQCARG/(ZGAMI+1.D0))/2.D0)
379              end if
380            end do
381          else ! ityp
382            do JJ=INDEX_BODY_START, INDEX_BODY
383              ISTYP = obs_bodyElem_i(obsSpaceData,OBS_VNM,JJ)
384              ZSLEV = obs_bodyElem_r(obsSpaceData,OBS_PPP,JJ)
385              if ((ISTYP == BUFR_NEUU .or.  &
386                   ISTYP == BUFR_NEUS) .and.  &
387                   ZSLEV == ZLEV) then
388                ZJON=obs_bodyElem_r(obsSpaceData,OBS_JOBS,INDEX_BODY)+  &
389                     obs_bodyElem_r(obsSpaceData,OBS_JOBS,JJ)
390                ZQCARG = ZGAMI + EXP(-1.0D0*ZJON)
391                ZPPOST = ZGAMI/ZQCARG
392                call obs_bodySet_r(obsSpaceData,OBS_QCV,INDEX_BODY, ZPPOST)
393                call obs_bodySet_r(obsSpaceData,OBS_QCV,JJ, ZPPOST)
394                call obs_bodySet_r(obsSpaceData,OBS_JOBS,INDEX_BODY,-LOG(ZQCARG/(ZGAMI+1.D0))/2.D0)
395                call obs_bodySet_r(obsSpaceData,OBS_JOBS,JJ, -LOG(ZQCARG/(ZGAMI+1.D0))/2.D0)
396              end if
397            enddo
398          endif !ityp
399        else ! LLUV
400          zjon = obs_bodyElem_r(obsSpaceData,OBS_JOBS,index_body)
401          zqcarg = zgami + exp(-1.0D0*zjon)
402          zppost = zgami/zqcarg
403          call obs_bodySet_r(obsSpaceData,OBS_QCV,index_body, zppost)
404          call obs_bodySet_r(obsSpaceData,OBS_JOBS,INDEX_BODY, - log(zqcarg/(zgami+1.D0)))
405        endif ! LLUV
406
407      endif ! includeFlag
408
409    enddo BODY
410
411  end subroutine vqc_NlTl

414  subroutine vqc_ad(obsSpaceData)
415    !
416    !:Purpose: Factorizes Grad(Jo) according to Andersson and Jarvinen
417    !           1999, Variational quality control, Q.J.R., 125, pp. 697-722.
418    !           It uses the value of (1-Wqc) saved in OBS_QCV in vqc_NlTl
419    implicit none
420
421    type(struct_obs) :: obsSpaceData ! obsSpaceData object
422
423    integer :: index_body
424    logical :: includeFlag
425
426    do index_body=1,obs_numbody(obsSpaceData)
427      includeFlag = (obs_bodyElem_i(obsSpaceData,OBS_ASS,index_body) == obs_assimilated) .and.  &
428                    (obs_getFamily(obsSpaceData,bodyIndex=index_body) /= 'RO')
429      ! pas de qcvar pour les radiances en mode matrice R non diagonale
430      if (rmat_lnondiagr) includeFlag = includeFlag .and.  &
431         (obs_getFamily(obsSpaceData,bodyIndex=index_body) /= 'TO')
432
433      if (includeFlag) then
434        call obs_bodySet_r(obsSpaceData,OBS_WORK,index_body,  &
435               obs_bodyElem_r(obsSpaceData,OBS_WORK,index_body)  &
436               *(1.d0 - obs_bodyElem_r(obsSpaceData,OBS_QCV,index_body)))
437      endif
438    enddo
439
440  end subroutine vqc_ad

443  subroutine vqc_listrej(lobsSpaceData)
444    !
445    !:Purpose: List all observations rejected by the variational QC
446    !           Set QC flags consistent with VARQC decisions
447    !           Set global flag indicating report contains rejected observations
448    !           as required
449
450    implicit none
451
452    type(struct_obs) :: lobsSpaceData
453
454    integer, parameter :: numitem = 16
455    integer :: icount( numitem, ofl_numFamily )
456
457    integer :: jfam, jitem, headerIndex
458    integer :: bodyIndex, bodyIndex2, ISTART,ityp,ISTYP
459    integer :: ISPDO,IDIRO,ISPDF,IDIRF,ISPDA,IDIRA,ILEV
460    integer :: IDBURP,IOTHER,obsONM
461    real(8) :: ZVAR,ZFCST,ZANA,ZPOST,ZLAT,ZLON,ZUU,ZVV
462    real(8) :: SPD,DEG,ZLEV,ZSLEV,ZCUT
463    character(len=4)  :: CLITM(NUMITEM),CLDESC
464    character(len=2)  :: CLUNITS
465    character(len=21) :: CODTYPNAME
466    character(len=12) :: stnId
467    logical :: LLOK,LLELREJ
468    !
469    !  ------NOTE----------
470    ! CURRENTLY SUPPORTED FAMILIES OF DATA 'UA' 'AI' 'SF' 'HU' 'TO' 'GO' 'GP'
471    !
472    DATA ZCUT /0.75D0/
473    DATA CLITM(1), CLITM(2), CLITM(3), CLITM(4), CLITM(5), CLITM(6)  &
474           /'WND',    'WND',    'HGT',    'TMP',    'DPD',   'STNP'/
475    DATA CLITM(7), CLITM(8), CLITM(9), CLITM(10), CLITM(11), CLITM(12)  &
476          /'MSLP',   'TSFC',   'SDPD',   'SWND',   'SWND',   'BTMP'/
477    DATA CLITM(13), CLITM(14), CLITM(15) , CLITM(16)  &
478           /'ZTD',   'CHM',     'SST',    'ICEC'/
479
480
481    do jfam = 1, ofl_numFamily
482      do jitem = 1, numitem
483        icount( jitem, jfam ) = 0
484      end do
485    end do
486    write(*,*) 'LIST OF DATA REJECTED BY VARIATIONAL QUALITY CONTROL'
487    !
488    !_____LOOP OVER ALL REPORTS, ONE FAMILY AT A TIME
489    !
490    FAMILY: do jfam = 1, ofl_numFamily
491
492      write(*,*) ' '
493      if (ofl_familyList(jfam) /= 'TO' ) then
494        write(*,*) 'IDENT     TYPE DESCRIPTION           LAT   LONG    LEVEL  ITEM   OBSVD     BKGND     ANAL  POST PROB REPORT'
495      else
496        write(*,*) 'IDENT     TYPE DESCRIPTION           LAT   LONG    CHNL   ITEM   OBSVD     BKGND     ANAL  POST PROB REPORT'
497      end if
498
499      ! loop over all header indices of the family
500      call obs_set_current_header_list( lobsSpaceData, ofl_familyList( jfam ) )
501
502      HEADER: do
503        headerIndex = obs_getHeaderIndex( lobsSpaceData )
504        if (headerIndex < 0) exit HEADER
505
506        llelrej = .false.
507        idburp  = obs_headElem_i( lobsSpaceData, OBS_ITY, headerIndex )
508        zlat    = obs_headElem_r( lobsSpaceData, OBS_LAT, headerIndex ) * MPC_DEGREES_PER_RADIAN_R8
509        zlon    = obs_headElem_r( lobsSpaceData, OBS_LON, headerIndex ) * MPC_DEGREES_PER_RADIAN_R8
510
511        ! loop over all body indices for this headerIndex
512        call obs_set_current_body_list( lobsSpaceData, headerIndex )
513
514        BODY: do 
515           bodyIndex = obs_getBodyIndex( lobsSpaceData )
516           if ( bodyIndex < 0) exit BODY
517           ityp = obs_bodyElem_i( lobsSpaceData, OBS_VNM, bodyIndex )
518           if (ityp == BUFR_NETS .or. ityp == BUFR_NEPS .or.  &
519               ityp == BUFR_NEPN .or. ityp == BUFR_NESS .or.  &
520               ityp == BUFR_NEUS .or. ityp == BUFR_NEVS .or.  &
521               ityp == BUFR_NEZD .or. ityp == bufr_sst  .or. ityp == BUFR_ICEC .or.  &
522               ityp == bufr_vis  .or. ityp == bufr_logVis  .or. ityp == bufr_gust .or.  &
523               ityp == bufr_radarPrecip .or. ityp == bufr_logRadarPrecip ) then
524              llok = (obs_bodyElem_i( lobsSpaceData, OBS_ASS, bodyIndex ) == obs_assimilated )
525           else
526              llok = (obs_bodyElem_i( lobsSpaceData, OBS_ASS, bodyIndex ) == obs_assimilated .and.  &
527                      obs_bodyElem_i( lobsSpaceData, OBS_XTR, bodyIndex ) == 0) .or.  &
528                     (obs_bodyElem_i( lobsSpaceData, OBS_XTR, bodyIndex ) == 2 .and.  &
529                      obs_bodyElem_i( lobsSpaceData, OBS_VNM, bodyIndex ) == BUFR_NEGZ)
530           end if
531           if ( llok ) then
532             zpost = obs_bodyElem_r( lobsSpaceData, OBS_QCV, bodyIndex )
533             zlev  = obs_bodyElem_r( lobsSpaceData, OBS_PPP, bodyIndex ) * MPC_MBAR_PER_PA_R8
534             if ( obs_bodyElem_i( lobsSpaceData, OBS_VCO, bodyIndex ) == 2) then
535               zlev = obs_bodyElem_r( lobsSpaceData, OBS_PPP, bodyIndex ) * MPC_MBAR_PER_PA_R8
536               CLUNITS = 'MB'
537             else if (obs_bodyElem_i(lobsSpaceData,OBS_VCO,bodyIndex) == 1) then
538               !
539               ! VERTICAL COORDINATE IS HEIGHT
540               !
541               ZLEV = obs_bodyElem_r(lobsSpaceData,OBS_PPP,bodyIndex)
542               CLUNITS = ' M'
543               if (ofl_familyList(JFAM)=='CH') then
544                  ZLEV = ZLEV*0.01
545                  CLUNITS = 'HM'
546               end if
547             else if (obs_bodyElem_i(lobsSpaceData,OBS_VCO,bodyIndex) == -1) then
548               !
549               ! TOVS CHANNEL NUMBER
550               !
551               ZLEV = obs_bodyElem_r(lobsSpaceData,OBS_PPP,bodyIndex)
552               CLUNITS = '  '
553             else 
554               ZLEV = obs_bodyElem_r(lobsSpaceData,OBS_PPP,bodyIndex)
555               CLUNITS = '  '
556             end if
557
558             ZVAR = obs_bodyElem_r(lobsSpaceData,OBS_VAR,bodyIndex)
559
560             ZFCST= ZVAR - obs_bodyElem_r(lobsSpaceData,OBS_OMP,bodyIndex)
561             ZANA = ZVAR - obs_bodyElem_r(lobsSpaceData,OBS_OMA,bodyIndex)
562             !
563             !_____________TREAT WINDS AS SPECIAL CASE
564             !              BUFR_NEUU       = 011003 (U COMPONENT)           (m/s)
565             !              BUFR_NEVV       = 011004 (V COMPONENT)           (m/s)
566             !              BUFR_NEUS       = 011215 (U COMPONENT AT 10 M)   (m/s)
567             !              BUFR_NEVS       = 011216 (V COMPONENT AT 10 M)   (m/s)
568             !
569             if (((ityp==BUFR_NEUU .or. ityp == BUFR_NEUS) .and.  &
570                  col_varExist(varName='UU')).or.  &
571                 ((ityp==BUFR_NEVV .or. ityp == BUFR_NEVS) .and.  &
572                  col_varExist(varName='VV'))) then
573               IOTHER = -1
574               if (ityp == BUFR_NEUU .or. ityp == BUFR_NEUS) then
575                 ISTART=obs_headElem_i(lobsSpaceData,OBS_RLN,headerIndex)
576                 do bodyIndex2=ISTART,bodyIndex
577                   ISTYP = obs_bodyElem_i(lobsSpaceData,OBS_VNM,bodyIndex2)
578                   if (obs_bodyElem_i(lobsSpaceData,OBS_VCO,bodyIndex2) == 2) then
579                     ZSLEV = obs_bodyElem_r(lobsSpaceData,OBS_PPP,bodyIndex2)*MPC_MBAR_PER_PA_R8
580                   end if
581                   if (obs_bodyElem_i(lobsSpaceData,OBS_VCO,bodyIndex2) == 1) then
582                     ZSLEV = obs_bodyElem_r(lobsSpaceData,OBS_PPP,bodyIndex2)
583                   end if
584                   if ((ISTYP == BUFR_NEVV .or. ISTYP == BUFR_NEVS) .and. &
585                        ZLEV == ZSLEV) then
586                     IOTHER = bodyIndex2
587                   end if
588                 end do
589               else
590                 ISTART=obs_headElem_i(lobsSpaceData,OBS_RLN,headerIndex)
591                 do bodyIndex2=ISTART,bodyIndex
592                   ISTYP = obs_bodyElem_i(lobsSpaceData,OBS_VNM,bodyIndex2)
593                   if (obs_bodyElem_i(lobsSpaceData,OBS_VCO,bodyIndex2) == 2) then
594                     ZSLEV = obs_bodyElem_r(lobsSpaceData,OBS_PPP,bodyIndex2)*MPC_MBAR_PER_PA_R8
595                   end if
596                   if (obs_bodyElem_i(lobsSpaceData,OBS_VCO,bodyIndex2) == 1) then
597                     ZSLEV = obs_bodyElem_r(lobsSpaceData,OBS_PPP,bodyIndex2)
598                   end if
599                   if ((ISTYP == BUFR_NEUU .or. ISTYP == BUFR_NEUS) .and.  &
600                        ZLEV == ZSLEV) then
601                     IOTHER = bodyIndex2
602                   end if
603                 end do
604               end if
605               if (IOTHER /= -1) then
606                 if (ityp == BUFR_NEVV .or. ityp == BUFR_NEUS) then
607                   ZUU = ZVAR
608                   ZVV = obs_bodyElem_r(lobsSpaceData,OBS_VAR,IOTHER)
609                 else
610                   ZVV = ZVAR
611                   ZUU = obs_bodyElem_r(lobsSpaceData,OBS_VAR,IOTHER)
612                 end if
613                 SPD = SQRT(ZUU*ZUU + ZVV*ZVV)
614                 ISPDO = NINT(SPD)
615                 if (ZUU==0.D0 .and. ZVV==0.D0)then
616                   IDIRO = 999
617                 else
618                   DEG = 270. - ATAN2(ZVV,ZUU)*MPC_DEGREES_PER_RADIAN_R8
619                   if (DEG > 360.D0)DEG = DEG - 360.D0
620                   if (DEG < 0.D0)  DEG = DEG + 360.D0
621                   IDIRO = NINT(DEG)
622                 end if
623                 if (ityp == BUFR_NEUU .or. ityp == BUFR_NEUS) then
624                   ZUU = ZFCST
625                   ZVV = obs_bodyElem_r(lobsSpaceData,OBS_VAR,IOTHER) - &
626                         obs_bodyElem_r(lobsSpaceData,OBS_OMP,IOTHER)
627                 else
628                   ZVV = ZFCST
629                   ZUU = obs_bodyElem_r(lobsSpaceData,OBS_VAR,IOTHER) - &
630                         obs_bodyElem_r(lobsSpaceData,OBS_OMP,IOTHER)
631                 end if
632                 SPD=SQRT(ZUU*ZUU + ZVV*ZVV)
633                 ISPDF = NINT(SPD)
634                 if (ZUU == 0.D0 .and. ZVV == 0.D0) then
635                   IDIRF = 999
636                 else
637                   DEG = 270.D0 - ATAN2(ZVV,ZUU)*MPC_DEGREES_PER_RADIAN_R8
638                   if (DEG > 360.D0)DEG = DEG - 360.D0
639                   if (DEG < 0.D0)  DEG = DEG + 360.D0
640                   IDIRF = NINT(DEG)
641                 end if
642                 if (ityp == BUFR_NEUU .or. ityp == BUFR_NEUS) then
643                   ZUU = ZANA
644                   ZVV = obs_bodyElem_r(lobsSpaceData,OBS_VAR,IOTHER) -  &
645                         obs_bodyElem_r(lobsSpaceData,OBS_OMA,IOTHER)
646                 else
647                   ZVV = ZANA
648                   ZUU = obs_bodyElem_r(lobsSpaceData,OBS_VAR,IOTHER) -  &
649                         obs_bodyElem_r(lobsSpaceData,OBS_OMA,IOTHER)
650                 end if
651                 SPD=SQRT(ZUU*ZUU + ZVV*ZVV)
652                 ISPDA = NINT(SPD)
653                 if (ZUU==0.D0 .and. ZVV==0.D0) then
654                   IDIRA = 999
655                 else
656                   DEG = 270.D0 - ATAN2(ZVV,ZUU)*MPC_DEGREES_PER_RADIAN_R8
657                   if (DEG > 360.D0)DEG = DEG - 360.D0
658                   if (DEG < 0.D0)  DEG = DEG + 360.D0
659                   IDIRA = NINT(DEG)
660                 end if
661                 ILEV = NINT(ZLEV)
662                 if (ZPOST > ZCUT) then
663                   LLELREJ = .TRUE.
664                   call obs_bodySet_i(lobsSpaceData,OBS_FLG,bodyIndex,  &
665                     IBSET(obs_bodyElem_i(lobsSpaceData,OBS_FLG,bodyIndex),9))
666                   call obs_bodySet_i(lobsSpaceData,OBS_FLG,IOTHER,  &
667                     IBSET(obs_bodyElem_i(lobsSpaceData,OBS_FLG,IOTHER),9))
668                   call obs_bodySet_i(lobsSpaceData,OBS_FLG,bodyIndex,  &
669                     IBSET(obs_bodyElem_i(lobsSpaceData,OBS_FLG,bodyIndex),17))
670                   call obs_bodySet_i(lobsSpaceData,OBS_FLG,IOTHER,  &
671                     IBSET(obs_bodyElem_i(lobsSpaceData,OBS_FLG,IOTHER),17))
672                   if (ityp == BUFR_NEUU .or.  &
673                       ityp == BUFR_NEVV) then
674                     ICOUNT(1,JFAM) = ICOUNT(1,JFAM) + 1
675                   end if
676                   if (ityp == BUFR_NEUS .or. ityp == BUFR_NEVS) then
677                     ICOUNT(10,JFAM) = ICOUNT(10,JFAM) + 1
678                   end if
679                   codtypname=codtyp_get_name(IDBURP)
680                   stnId = obs_elem_c(lobsSpaceData,'STID',headerIndex)
681                   obsONM = obs_headElem_i(lobsSpaceData,OBS_ONM,headerIndex)
682                   write(*,620) stnId,IDBURP,codtypname,ZLAT,ZLON,  &
683                     ILEV,CLUNITS,IDIRO,ISPDO,IDIRF,ISPDF,IDIRA,  &
684                     ISPDA,ZPOST,obsONM
685 620               FORMAT(A9,1X,i3,1x,A21,1X,F5.1,2X,F5.1,2X,I4,A2,2X,  &
686                     'WND',3X,I3,'/',I3,3X,I3,'/',I3,3X,I3,  &
687                     '/',I3,2X,F7.4,1X,I8)
688
689                 end if
690               end if
691             else
692               ILEV = NINT(ZLEV)
693               if (ZPOST > ZCUT) then
694                 LLELREJ = .TRUE.
695                 if (ofl_familyList(JFAM)=='CH') then
696                    ICOUNT(14,JFAM)=ICOUNT(14,JFAM)+1
697                    CLDESC=CLITM(14)
698                    if (obs_headElem_i(lobsSpaceData,OBS_CHM,headerIndex).ge.0) then
699
700                       ! CONVERT TO PERCENTAGE DIFFERENCE FROM THE FORECAST
701
702                       if (abs(ZFCST).lt.1.D-10*max(1.D-30,abs(ZVAR-ZFCST))) then
703                          write(*,*) 'vqc_listrej: WARNING for CH obs. BKGND and or OBSVD value out of bounds for output.'
704                          write(*,*) 'OBSVD and ANAL set to 0.0 below in addition to BKGND.'
705                          ZVAR=0.0D0
706                          ZANA=0.0D0
707                       else
708                          ZVAR=(ZVAR-ZFCST)/ZFCST*100.
709                          ZANA=(ZANA-ZFCST)/ZFCST*100.
710                       end if
711                       ZFCST=0.0
712                       CLDESC=vnl_varnameFromVarnum(ityp,obs_headElem_i(lobsSpaceData,OBS_CHM,headerIndex))
713                         
714                    else if (ityp == BUFR_NETT) then
715
716                       ! CONVERT FROM KELVIN TO CELCIUS
717
718                       CLDESC = CLITM(4)
719                       ZVAR = ZVAR - MPC_K_C_DEGREE_OFFSET_R8
720                       ZFCST = ZFCST - MPC_K_C_DEGREE_OFFSET_R8
721                       ZANA  = ZANA - MPC_K_C_DEGREE_OFFSET_R8
722                    end if
723                 else if (ityp == BUFR_NEGZ) then
724                   CLDESC = CLITM(3)
725                   ICOUNT(3,JFAM) = ICOUNT(3,JFAM) + 1
726                   !
727                   ! CONVERT FROM GEOPOTENTIAL TO GEOPOTENTIAL HEIGHT
728                   !
729                   ZVAR = ZVAR/ec_rg
730                   ZFCST = ZFCST/ec_rg
731                   ZANA  = ZANA/ec_rg
732                 else if (ityp == BUFR_NETT) then
733                   CLDESC = CLITM(4)
734                   ICOUNT(4,JFAM) = ICOUNT(4,JFAM) + 1
735                   !
736                   ! CONVERT FROM KELVIN TO CELCIUS
737                   !
738                   ZVAR = ZVAR - MPC_K_C_DEGREE_OFFSET_R8
739                   ZFCST = ZFCST - MPC_K_C_DEGREE_OFFSET_R8
740                   ZANA  = ZANA - MPC_K_C_DEGREE_OFFSET_R8
741                 else if (ityp == BUFR_NEES) then
742                   CLDESC = CLITM(5)
743                   ICOUNT(5,JFAM) = ICOUNT(5,JFAM) + 1
744                 else if (ityp == BUFR_NEPS) then
745                   CLDESC = CLITM(6)
746                   ICOUNT(6,JFAM) = ICOUNT(6,JFAM) + 1
747                   !
748                   ! CONVERT FROM PASCALS TO MILLIBARS
749                   !
750                   ZVAR = ZVAR*MPC_MBAR_PER_PA_R8
751                   ZANA = ZANA*MPC_MBAR_PER_PA_R8
752                   ZFCST = ZFCST*MPC_MBAR_PER_PA_R8
753                 else if (ityp == BUFR_NEPN) then
754                   CLDESC = CLITM(7)
755                   ICOUNT(7,JFAM) = ICOUNT(7,JFAM) + 1
756                   !
757                   ! CONVERT FROM PASCALS TO MILLIBARS
758                   !
759                   ZVAR = ZVAR*MPC_MBAR_PER_PA_R8
760                   ZANA = ZANA*MPC_MBAR_PER_PA_R8
761                   ZFCST = ZFCST*MPC_MBAR_PER_PA_R8
762                 else if (ityp == BUFR_NETS) then
763                   CLDESC = CLITM(8)
764                   ICOUNT(8,JFAM) = ICOUNT(8,JFAM) + 1
765                   !
766                   ! CONVERT FROM KELVIN TO CELCIUS
767                   !
768                   ZVAR = ZVAR - MPC_K_C_DEGREE_OFFSET_R8
769                   ZFCST = ZFCST - MPC_K_C_DEGREE_OFFSET_R8
770                   ZANA  = ZANA - MPC_K_C_DEGREE_OFFSET_R8
771                 else if (ityp == BUFR_NESS) then
772                   CLDESC = CLITM(9)
773                   ICOUNT(9,JFAM) = ICOUNT(9,JFAM) + 1
774                 else if ( ityp==BUFR_NBT1 .or. ityp==BUFR_NBT2 .or.  &
775                      ityp==BUFR_NBT3      ) then
776                   CLDESC = CLITM(12)
777                   ICOUNT(12,JFAM) = ICOUNT(12,JFAM) + 1
778                 else if (ityp == BUFR_NEZD) then
779                   CLDESC = CLITM(13)
780                   ICOUNT(13,JFAM) = ICOUNT(13,JFAM) + 1
781                   !
782                   ! CONVERT ZTD FROM METRES TO MILLIMETRES
783                   !
784                   ZVAR = ZVAR * 1000.D0
785                   ZFCST = ZFCST * 1000.D0
786                   ZANA  = ZANA * 1000.D0
787                 else if (ityp == bufr_sst) then
788                   CLDESC = CLITM(15)
789                   ICOUNT(15,JFAM) = ICOUNT(15,JFAM) + 1
790                 else if (ityp == BUFR_ICEC) then
791                   CLDESC = CLITM(16)
792                   ICOUNT(16,JFAM) = ICOUNT(16,JFAM) + 1
793                 end if
794
795                 call obs_bodySet_i(lobsSpaceData,OBS_FLG,bodyIndex,  &
796                   IBSET(obs_bodyElem_i(lobsSpaceData,OBS_FLG,bodyIndex),9))
797                 call obs_bodySet_i(lobsSpaceData,OBS_FLG,bodyIndex,  &
798                   IBSET(obs_bodyElem_i(lobsSpaceData,OBS_FLG,bodyIndex),17))
799                 codtypname=codtyp_get_name(IDBURP)
800                 stnId = obs_elem_c(lobsSpaceData,'STID',headerIndex)
801                 obsONM = obs_headElem_i(lobsSpaceData,OBS_ONM,headerIndex)
802                 write(*,630) stnId,IDBURP,  &
803                   codtypname,ZLAT,ZLON,ILEV,CLUNITS,CLDESC, &
804                   ZVAR,ZFCST,ZANA,ZPOST,obsONM
805 630               FORMAT(A9,1X,I3,1X,A21,1X,F5.1,2X,F5.1,2X,I4,A2,2X,  &
806                    A4,2X,F7.1,3X,F7.1,3X,F7.1,2X,F7.4,1X,I8,1X)
807               end if
808             end if
809           end if
810        end do BODY
811        !
812        ! NOW SET THE GLOBAL FLAGS IN THE BURP RECORD HEADER
813        !
814        if (LLELREJ) then
815          call obs_headSet_i(lobsSpaceData,OBS_ST1,headerIndex, IBSET( obs_headElem_i(lobsSpaceData,OBS_ST1,headerIndex), 06))
816        end if
817      end do HEADER
818    end do FAMILY
819
820
821    write(*,640)
822 640  FORMAT(//)
823    write(*,*) ' REJECTED DATA ACCORDING TO FAMILY OF REPORT.'
824    write(*,670)(ofl_familyList(JFAM),JFAM=1,ofl_numFamily)
825 670  FORMAT(5X,11(7X,A2))
826    do JITEM=1,NUMITEM
827      if ( .NOT. (JITEM == 2 .or. JITEM == 11)) then
828        write(*,680)CLITM(JITEM),(ICOUNT(JITEM,JFAM),JFAM=1,ofl_numFamily)
829 680      FORMAT(1X,A4,11(4X,I5))
830      end if
831    end do
832    write(*,640)
833
834  end subroutine vqc_listrej
