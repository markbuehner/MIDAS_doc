  84  subroutine eob_allocate(ensObs, numMembers, numObs, obsSpaceData, &
  85                          fileMemberIndex1_opt, edaObsImpact_opt)
  86    !
  87    !:Purpose: Allocate an ensObs object
  88    !
  89    implicit none
  90
  91    ! arguments
  92    type(struct_eob)        , intent(inout) :: ensObs
  93    integer                 , intent(in)    :: numMembers
  94    integer                 , intent(in)    :: numObs
  95    type(struct_obs), target, intent(in)    :: obsSpaceData
  96    integer, optional       , intent(in)    :: fileMemberIndex1_opt
  97    logical, optional       , intent(in)    :: edaObsImpact_opt
  98
  99    ! locals
 100    logical :: edaObsImpact
 101
 102    if ( ensObs%allocated ) then
 103      write(*,*) 'eob_allocate: this object is already allocated, deallocating first.'
 104      call eob_deallocate( ensObs )
 105    end if
 106
 107    if ( present(fileMemberIndex1_opt) ) ensObs%fileMemberIndex1 = fileMemberIndex1_opt
 108
 109    if ( present(edaObsImpact_opt) ) then
 110      edaObsImpact = edaObsImpact_opt
 111    else
 112      edaObsImpact = .false.
 113    end if
 114
 115    ensObs%obsSpaceData  => obsSpaceData
 116    ensObs%numMembers    = numMembers
 117    ensObs%numObs        = numObs
 118
 119    allocate( ensObs%lat(ensObs%numObs) )
 120    allocate( ensObs%lon(ensObs%numObs) )
 121    allocate( ensObs%vertLocation(ensObs%numObs) )
 122    allocate( ensObs%obsValue(ensObs%numObs) )
 123    allocate( ensObs%obsErrInv(ensObs%numObs) )
 124    if ( edaObsImpact ) allocate( ensObs%obsErrInv_eda(ensObs%numObs) )
 125    allocate( ensObs%Yb_r4(ensObs%numMembers,ensObs%numObs) )
 126    allocate( ensObs%meanYb(ensObs%numObs) )
 127    allocate( ensObs%deterYb(ensObs%numObs) )
 128    allocate( ensObs%assFlag(ensObs%numObs) )
 129
 130    ensObs%allocated = .true.
 131
 132    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 133
 134  end subroutine eob_allocate

 220  !--------------------------------------------------------------------------
 221  ! eob_clean (private routine)
 222  !--------------------------------------------------------------------------
 223  subroutine eob_clean(ensObs,ensObsClean)
 224    !
 225    !:Purpose: Remove all obs from the ensObs object that are not 
 226    !           flagged for assimilation. Put the cleaned result in the
 227    !           locally created output object.
 228    !
 229    implicit none
 230
 231    ! arguments
 232    type(struct_eob), intent(in)  :: ensObs
 233    type(struct_eob), intent(out) :: ensObsClean
 234
 235    ! locals
 236    integer :: obsIndex, obsCleanIndex, numObsClean
 237
 238    call eob_setAssFlag(ensObs)
 239
 240    numObsClean = 0
 241    do obsIndex = 1, ensObs%numObs
 242      if (ensObs%assFlag(obsIndex) == 1) numObsClean = numObsClean + 1
 243    end do
 244
 245    write(*,*) 'eob_clean: reducing numObs from ', ensObs%numObs, ' to ', numObsClean
 246    call eob_allocate(ensObsClean, ensObs%numMembers, numObsClean, ensObs%obsSpaceData)
 247    if ( allocated(ensObs%obsErrInv_eda) ) then
 248      allocate(ensObsClean%obsErrInv_eda(numObsClean))
 249    end if
 250    if ( allocated(ensObs%Ya_r4) ) then
 251      allocate(ensObsClean%Ya_r4(ensObs%numMembers,numObsClean))
 252    end if
 253    if ( allocated(ensObs%randPert_r4) ) then
 254      allocate(ensObsClean%randPert_r4(ensObs%numMembers,numObsClean))
 255    end if
 256
 257    obsCleanIndex = 0
 258    do obsIndex = 1, ensObs%numObs
 259      if (ensObs%assFlag(obsIndex) == 1) then
 260        obsCleanIndex = obsCleanIndex + 1
 261        ensObsClean%lat(obsCleanIndex)           = ensObs%lat(obsIndex)
 262        ensObsClean%lon(obsCleanIndex)           = ensObs%lon(obsIndex)
 263        ensObsClean%vertLocation(obsCleanIndex)  = ensObs%vertLocation(obsIndex)
 264        ensObsClean%obsErrInv(obsCleanIndex)     = ensObs%obsErrInv(obsIndex)
 265        if ( allocated(ensObs%obsErrInv_eda) ) then
 266          ensObsClean%obsErrInv_eda(obsCleanIndex) = ensObs%obsErrInv_eda(obsIndex)
 267        end if 
 268        ensObsClean%Yb_r4(:,obsCleanIndex)       = ensObs%Yb_r4(:,obsIndex)
 269        if ( allocated(ensObs%Ya_r4) ) then
 270          ensObsClean%Ya_r4(:,obsCleanIndex) = ensObs%Ya_r4(:,obsIndex)
 271        end if
 272        if ( allocated(ensObs%randPert_r4) ) then
 273          ensObsClean%randPert_r4(:,obsCleanIndex) = ensObs%randPert_r4(:,obsIndex)
 274        end if
 275        ensObsClean%meanYb(obsCleanIndex)        = ensObs%meanYb(obsIndex)
 276        ensObsClean%deterYb(obsCleanIndex)       = ensObs%deterYb(obsIndex)
 277        ensObsClean%obsValue(obsCleanIndex)      = ensObs%obsValue(obsIndex)
 278        ensObsClean%assFlag(obsCleanIndex)       = ensObs%assFlag(obsIndex)
 279      end if
 280    end do
 281
 282  end subroutine eob_clean

 321  subroutine eob_allGather(ensObs,ensObs_mpiglobal)
 322    !
 323    !:Purpose: Collect obs information distributed over all mpi tasks and
 324    !           make it available on all mpi tasks. The output ensObs object
 325    !           will be allocated within this subroutine.
 326    !
 327    implicit none
 328
 329    ! arguments
 330    type(struct_eob), intent(in)  :: ensObs
 331    type(struct_eob), intent(out) :: ensObs_mpiglobal
 332
 333    ! locals
 334    type(struct_eob) :: ensObsClean
 335    integer :: ierr, procIndex, memberIndex, numObs_mpiglobal
 336    integer :: allNumObs(mmpi_nprocs), displs(mmpi_nprocs)
 337
 338    write(*,*) 'eob_allGather: starting'
 339    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 340
 341    call utl_tmg_start(10,'--Observations')
 342    call utl_tmg_start(24,'----Eob_AllGather')
 343
 344    ! refresh assimilation flag and then clean ensObs before communicating and writing
 345    call eob_setAssFlag(ensObs)
 346    call eob_clean(ensObs,ensObsClean)
 347
 348    call rpn_comm_allgather( ensObsClean%numObs, 1, 'mpi_integer',  &
 349                             allNumObs, 1, 'mpi_integer', &
 350                             'GRID', ierr )
 351    numObs_mpiglobal = sum(allNumObs(:))
 352
 353    if (ensObs_mpiglobal%allocated) then
 354      call utl_abort('eob_allGather: output ensObs object must not be already allocated')
 355    end if
 356    call eob_allocate(ensObs_mpiglobal, ensObsClean%numMembers, numObs_mpiglobal, ensObsClean%obsSpaceData, &
 357                      fileMemberIndex1_opt=ensObs%fileMemberIndex1)
 358    if ( allocated(ensObsClean%Ya_r4) ) then
 359      allocate(ensObs_mpiglobal%Ya_r4(ensObsClean%numMembers,numObs_mpiglobal))
 360    end if
 361    if ( allocated(ensObsClean%obsErrInv_eda) ) then
 362      allocate(ensObs_mpiglobal%obsErrInv_eda(numObs_mpiglobal))
 363    end if  
 364    if ( allocated(ensObsClean%randPert_r4) ) then
 365      allocate(ensObs_mpiglobal%randPert_r4(ensObsClean%numMembers,numObs_mpiglobal))
 366    end if
 367    ensObs_mpiglobal%typeVertCoord = ensObsClean%typeVertCoord
 368
 369    if ( mmpi_myid == 0 ) then
 370      displs(1) = 0
 371      do procIndex = 2, mmpi_nprocs
 372        displs(procIndex) = displs(procIndex-1) + allNumObs(procIndex-1)
 373      end do
 374    else
 375      displs(:) = 0
 376    end if
 377
 378    call rpn_comm_gatherv( ensObsClean%lat, ensObsClean%numObs, 'mpi_real8', &
 379                           ensObs_mpiglobal%lat, allNumObs, displs, 'mpi_real8',  &
 380                           0, 'GRID', ierr )
 381    call rpn_comm_gatherv( ensObsClean%lon, ensObsClean%numObs, 'mpi_real8', &
 382                           ensObs_mpiglobal%lon, allNumObs, displs, 'mpi_real8',  &
 383                           0, 'GRID', ierr )
 384    call rpn_comm_gatherv( ensObsClean%vertLocation, ensObsClean%numObs, 'mpi_real8', &
 385                           ensObs_mpiglobal%vertLocation, allNumObs, displs, 'mpi_real8',  &
 386                           0, 'GRID', ierr )
 387    call rpn_comm_gatherv( ensObsClean%obsValue, ensObsClean%numObs, 'mpi_real8', &
 388                           ensObs_mpiglobal%obsValue, allNumObs, displs, 'mpi_real8',  &
 389                           0, 'GRID', ierr )
 390    call rpn_comm_gatherv( ensObsClean%obsErrInv, ensObsClean%numObs, 'mpi_real8', &
 391                           ensObs_mpiglobal%obsErrInv, allNumObs, displs, 'mpi_real8',  &
 392                           0, 'GRID', ierr )
 393    call rpn_comm_gatherv( ensObsClean%meanYb, ensObsClean%numObs, 'mpi_real8', &
 394                           ensObs_mpiglobal%meanYb, allNumObs, displs, 'mpi_real8',  &
 395                           0, 'GRID', ierr )
 396    call rpn_comm_gatherv( ensObsClean%deterYb, ensObsClean%numObs, 'mpi_real8', &
 397                           ensObs_mpiglobal%deterYb, allNumObs, displs, 'mpi_real8',  &
 398                           0, 'GRID', ierr )
 399    call rpn_comm_gatherv( ensObsClean%assFlag, ensObsClean%numObs, 'mpi_integer', &
 400                           ensObs_mpiglobal%assFlag, allNumObs, displs, 'mpi_integer',  &
 401                           0, 'GRID', ierr )
 402    if ( allocated(ensObsClean%obsErrInv_eda) ) then
 403      call rpn_comm_gatherv( ensObsClean%obsErrInv_eda, ensObsClean%numObs, 'mpi_real8', &
 404                             ensObs_mpiglobal%obsErrInv_eda, allNumObs, displs, 'mpi_real8',  &
 405                             0, 'GRID', ierr )
 406    end if
 407    do memberIndex = 1, ensObsClean%numMembers
 408      call rpn_comm_gatherv( ensObsClean%Yb_r4(memberIndex,:), ensObsClean%numObs, 'mpi_real4', &
 409                             ensObs_mpiglobal%Yb_r4(memberIndex,:), allNumObs, displs, 'mpi_real4',  &
 410                             0, 'GRID', ierr )
 411      if ( allocated(ensObsClean%Ya_r4) ) then
 412        call rpn_comm_gatherv( ensObsClean%Ya_r4(memberIndex,:), ensObsClean%numObs, 'mpi_real4', &
 413                              ensObs_mpiglobal%Ya_r4(memberIndex,:), allNumObs, displs, 'mpi_real4',  &
 414                              0, 'GRID', ierr )
 415      end if
 416      if ( allocated(ensObsClean%randPert_r4) ) then
 417        call rpn_comm_gatherv( ensObsClean%randPert_r4(memberIndex,:), ensObsClean%numObs, 'mpi_real4', &
 418                              ensObs_mpiglobal%randPert_r4(memberIndex,:), allNumObs, displs, 'mpi_real4',  &
 419                              0, 'GRID', ierr )
 420      end if
 421    end do
 422
 423    call rpn_comm_bcast(ensObs_mpiglobal%lat, ensObs_mpiglobal%numObs, 'mpi_real8',  &
 424                        0, 'GRID', ierr)
 425    call rpn_comm_bcast(ensObs_mpiglobal%lon, ensObs_mpiglobal%numObs, 'mpi_real8',  &
 426                        0, 'GRID', ierr)
 427    call rpn_comm_bcast(ensObs_mpiglobal%vertLocation, ensObs_mpiglobal%numObs, 'mpi_real8',  &
 428                        0, 'GRID', ierr)
 429    call rpn_comm_bcast(ensObs_mpiglobal%obsValue, ensObs_mpiglobal%numObs, 'mpi_real8',  &
 430                        0, 'GRID', ierr)
 431    call rpn_comm_bcast(ensObs_mpiglobal%obsErrInv, ensObs_mpiglobal%numObs, 'mpi_real8',  &
 432                        0, 'GRID', ierr)
 433    if ( allocated(ensObs_mpiglobal%obsErrInv_eda) ) then
 434      call rpn_comm_bcast(ensObs_mpiglobal%obsErrInv_eda, ensObs_mpiglobal%numObs, 'mpi_real8',  &
 435                          0, 'GRID', ierr)
 436    end if 
 437    call rpn_comm_bcast(ensObs_mpiglobal%meanYb, ensObs_mpiglobal%numObs, 'mpi_real8',  &
 438                        0, 'GRID', ierr)
 439    call rpn_comm_bcast(ensObs_mpiglobal%deterYb, ensObs_mpiglobal%numObs, 'mpi_real8',  &
 440                        0, 'GRID', ierr)
 441    call rpn_comm_bcast(ensObs_mpiglobal%assFlag, ensObs_mpiglobal%numObs, 'mpi_integer',  &
 442                        0, 'GRID', ierr)
 443    do memberIndex = 1, ensObsClean%numMembers
 444      call rpn_comm_bcast(ensObs_mpiglobal%Yb_r4(memberIndex,:), ensObs_mpiglobal%numObs, 'mpi_real4',  &
 445                          0, 'GRID', ierr)
 446      if ( allocated(ensObs_mpiglobal%Ya_r4) ) then
 447        call rpn_comm_bcast(ensObs_mpiglobal%Ya_r4(memberIndex,:), ensObs_mpiglobal%numObs, 'mpi_real4',  &
 448                            0, 'GRID', ierr)
 449      end if
 450      if ( allocated(ensObs_mpiglobal%randPert_r4) ) then
 451        call rpn_comm_bcast(ensObs_mpiglobal%randPert_r4(memberIndex,:), ensObs_mpiglobal%numObs, 'mpi_real4',  &
 452                            0, 'GRID', ierr)        
 453      end if
 454    end do
 455
 456    call eob_deallocate( ensObsClean )
 457
 458    write(*,*) 'eob_allGather: total number of obs to be assimilated =', sum(ensObs_mpiglobal%assFlag(:))
 459
 460    call utl_tmg_stop(24)
 461    call utl_tmg_stop(10)
 462
 463    write(*,*) 'eob_allGather: finished'
 464    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 465
 466  end subroutine eob_allGather

 471  subroutine eob_writeToFiles(ensObs, outputFilenamePrefix, writeObsInfo, &
 472                              numGroupsToDivideMembers_opt, &
 473                              maxNumMembersPerGroup_opt)
 474    !
 475    !:Purpose: Write the contents of an ensObs mpi local object to files
 476    !
 477    implicit none
 478
 479    ! arguments
 480    type(struct_eob),  intent(in) :: ensObs
 481    character(len=*),  intent(in) :: outputFilenamePrefix
 482    logical,           intent(in) :: writeObsInfo
 483    integer, optional, intent(in) :: numGroupsToDivideMembers_opt
 484    integer, optional, intent(in) :: maxNumMembersPerGroup_opt
 485
 486    ! locals
 487    integer :: unitNum, ierr, obsIndex, memberIndex
 488    integer :: obsVcoCode(ensObs%numObs), obsAssFlag(ensObs%numObs)
 489    integer :: obsFlag(ensObs%numObs)
 490    integer, allocatable :: memberIndexArray(:)
 491    character(len=40) :: fileName
 492    character(len=4)  :: myidxStr, myidyStr
 493    character(len=30) :: fileNameExtention
 494    integer :: fnom, fclos
 495    logical :: fileExists
 496
 497    if (.not. ensObs%allocated) then

 565  subroutine eob_readFromFiles(ensObs, numMembersToRead, inputFilenamePrefix, &
 566                               readObsInfo)
 567    !
 568    !:Purpose: Read mpi local ensObs%Yb object from file. Several files in separate subdirectories 
 569    !           can be read. Some examples of path+filename are:
 570    !           ensObs_0001/eob_HX_0001_0001
 571    !           ensObs_0002/eob_HX_0001_0001
 572    !
 573    implicit none
 574
 575    ! arguments
 576    type(struct_eob), intent(inout) :: ensObs
 577    integer         ,    intent(in) :: numMembersToRead
 578    character(len=*),    intent(in) :: inputFilenamePrefix
 579    logical,             intent(in) :: readObsInfo
 580    
 581    ! locals
 582    real(8) :: latFromFile(ensObs%numObs), lonFromFile(ensObs%numObs)
 583    real(8) :: obsValueFromFile(ensObs%numObs)
 584    integer :: obsVcoCode(ensObs%numObs), obsVcoCodeFromFile(ensObs%numObs)
 585    integer :: obsFlag(ensObs%numObs), assFlagFrom1File(ensObs%numObs)
 586    integer :: assFlagFromAllFiles(ensObs%numObs)
 587    integer :: unitNum, ierr, memberIndex, obsIndex, numObsFromFile
 588    integer :: numMembersFromFile, fnom, fclos
 589    integer :: fileIndex, numMembersAlreadyRead
 590    integer, allocatable :: memberIndexFromFile(:)
 591    logical :: fileExists
 592    character(len=256) :: fileName
 593    character(len=100) :: fileBaseName
 594    character(len=4)   :: myidxStr, myidyStr
 595    character(len=3)   :: fileIndexStr
 596    character(len=30)  :: fileNameExtention
 597
 598    if ( .not. ensObs%allocated ) then
 599      call utl_abort('eob_readFromFiles: this object is not allocated')
 600    end if
 601
 602    call obs_extractObsIntBodyColumn(obsVcoCode, ensObs%obsSpaceData, OBS_VCO)
 603
 604    write(myidxStr,'(I4.4)') (mmpi_myidx + 1)
 605    write(myidyStr,'(I4.4)') (mmpi_myidy + 1)
 606    fileNameExtention = trim(myidxStr) // '_' // trim(myidyStr)
 607
 608    if (readObsInfo) then
 609      ! loop on all file from different directories, read obsInfo and check they match ensObs
 610      fileBaseName = 'eob_obsInfo_' // trim(fileNameExtention)
 611
 612      fileIndex = 0
 613      numMembersAlreadyRead = 0
 614      do while (numMembersAlreadyRead < numMembersToRead)
 615        fileIndex = fileIndex + 1
 616        write(fileIndexStr,'(i3.3)') fileIndex
 617        fileName = './ensObs_' // fileIndexStr // '/' // fileBaseName
 618
 619        write(*,*) 'eob_readFromFiles: reading ',trim(fileName)
 620        inquire(file=trim(fileName),exist=fileExists)
 621        if (.not. fileExists) then
 622          write(*,*) 'fileName=', fileName
 623          call utl_abort('eob_readFromFiles: file does not exist')
 624        end if
 625
 626        unitNum = 0
 627        ierr = fnom(unitNum,trim(fileName),'FTN+SEQ+UNF',0)
 628        read(unitNum) numMembersFromFile, numObsFromFile
 629        if (ensObs%numObs /= numObsFromFile) then
 630          call utl_abort('eob_readFromFiles: ensObs%numObs does not match with that of file')
 631        end if
 632
 633        read(unitNum) (latFromFile(obsIndex), obsIndex = 1, ensObs%numObs)
 634        read(unitNum) (lonFromFile(obsIndex), obsIndex = 1, ensObs%numObs)
 635        read(unitNum) (obsVcoCodeFromFile(obsIndex), obsIndex = 1, ensObs%numObs)
 636        read(unitNum) (obsValueFromFile(obsIndex), obsIndex = 1, ensObs%numObs)
 637        
 638        if (maxval(abs(latFromFile(:) - ensObs%lat(:))) > 1.0d-5 .or. &
 639            maxval(abs(lonFromFile(:) - ensObs%lon(:))) > 1.0d-5 .or. &
 640            maxval(abs(obsValueFromFile(:) - ensObs%obsValue(:))) > 1.0d-7 .or. &
 641            .not. all(obsVcoCodeFromFile(:) == obsVcoCode(:))) then
 642
 643          call utl_abort('eob_readFromFiles: obsInfo file do not match ensObs')
 644        end if
 645      
 646        ! Read assimilation flag for of all files and apply a "logical or" to get the value 
 647        !   to put in obsSpaceData. Read obs flag only on the first file.
 648        read(unitNum) (assFlagFrom1File(obsIndex), obsIndex = 1, ensObs%numObs)
 649        if (numMembersAlreadyRead == 0) then
 650          read(unitNum) (obsFlag(obsIndex), obsIndex = 1, ensObs%numObs)
 651        end if
 652
 653        if (numMembersAlreadyRead == 0) assFlagFromAllFiles(:) = assFlagFrom1File(:)
 654        where (assFlagFrom1File(:) == obs_notAssimilated .and. numMembersAlreadyRead > 0)
 655          assFlagFromAllFiles(:) = obs_notAssimilated
 656        end where
 657
 658        ierr = fclos(unitNum)
 659
 660        numMembersAlreadyRead = numMembersAlreadyRead + numMembersFromFile
 661      end do
 662
 663      ! update assimilation flag in obsSpaceData
 664      do obsIndex = 1, ensObs%numObs
 665        ! skip this obs it is already set to be assimilated
 666        if (assFlagFromAllFiles(obsIndex) == obs_assimilated) cycle
 667
 668        call obs_bodySet_i(ensObs%obsSpaceData, OBS_ASS, obsIndex, obs_notAssimilated)
 669        call obs_bodySet_i(ensObs%obsSpaceData, OBS_FLG, obsIndex, obsFlag(obsIndex))
 670      end do
 671    end if
 672
 673    ! loop on all files from different directories to read ensObs%Yb for all members
 674    fileBaseName = trim(inputFilenamePrefix) // '_' // trim(fileNameExtention)
 675
 676    fileIndex = 0
 677    numMembersAlreadyRead = 0
 678    do while (numMembersAlreadyRead < numMembersToRead)
 679      fileIndex = fileIndex + 1
 680      write(fileIndexStr,'(i3.3)') fileIndex
 681      fileName = './ensObs_' // fileIndexStr // '/' // fileBaseName
 682
 683      write(*,*) 'eob_readFromFiles: reading ',trim(fileName)
 684      inquire(file=trim(fileName),exist=fileExists)
 685      if (.not. fileExists) then
 686        write(*,*) 'fileName=', fileName
 687        call utl_abort('eob_readFromFiles: file does not exist')
 688      end if
 689      
 690      unitNum = 0
 691      ierr = fnom(unitNum,trim(fileName),'FTN+SEQ+UNF',0)
 692      read(unitNum) numMembersFromFile
 693      allocate(memberIndexFromFile(numMembersFromFile))  
 694      read(unitNum) (memberIndexFromFile(memberIndex), memberIndex = 1, numMembersFromFile)
 695      do memberIndex = 1, numMembersFromFile
 696        read(unitNum) (ensObs%Yb_r4(memberIndexFromFile(memberIndex),obsIndex), obsIndex = 1, ensObs%numObs)
 697      end do
 698      ierr = fclos(unitNum)
 699
 700      deallocate(memberIndexFromFile)
 701
 702      numMembersAlreadyRead = numMembersAlreadyRead + numMembersFromFile
 703    end do
 704
 705  end subroutine eob_readFromFiles

 708  ! eob_getLocalBodyIndices
 709  !--------------------------------------------------------------------------
 710  function eob_getLocalBodyIndices(ensObs,localBodyIndices,distances,lat,lon,vertLocation,  &
 711                                   hLocalize,vLocalize,numLocalObsFound) result(numLocalObs)
 712    !
 713    !:Purpose: Return a list of values of bodyIndex for all observations within 
 714    !           the local volume around the specified lat/lon used for assimilation
 715    !           (as defined by h/vLocalize). The kdtree2 module is used to efficiently
 716    !           perform this task. The kdtree itself is constructed on the first call.
 717    !
 718    implicit none
 719
 720    ! arguments
 721    integer                       :: numLocalObs ! function output
 722    type(struct_eob), intent(in)  :: ensObs
 723    integer         , intent(out) :: localBodyIndices(:)
 724    real(8)         , intent(out) :: distances(:)
 725    real(8)         , intent(in)  :: lat, lon, vertLocation, hLocalize, vLocalize
 726    integer         , intent(out) :: numLocalObsFound
 727
 728    ! locals
 729    integer :: bodyIndex, numLocalObsFoundSearch, maxNumLocalObs, localObsIndex
 730    real(8) :: distance
 731    real(kdkind), allocatable         :: positionArray(:,:)
 732    type(kdtree2_result), allocatable :: searchResults(:)
 733    real(kdkind)                      :: maxRadius
 734    real(kdkind)                      :: refPosition(3)
 735
 736    ! create the kdtree on the first call
 737    if (.not. associated(tree)) then
 738      write(*,*) 'eob_getLocalBodyIndices: start creating kdtree'
 739      write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 740      allocate(positionArray(3,ensObs%numObs))
 741      do bodyIndex = 1, ensObs%numObs
 742        positionArray(1,bodyIndex) = ec_ra * sin(ensObs%lon(bodyIndex)) * cos(ensObs%lat(bodyIndex))
 743        positionArray(2,bodyIndex) = ec_ra * cos(ensObs%lon(bodyIndex)) * cos(ensObs%lat(bodyIndex))
 744        positionArray(3,bodyIndex) = ec_ra *                              sin(ensObs%lat(bodyIndex))
 745      end do
 746      tree => kdtree2_create(positionArray, sort=.true., rearrange=.true.) 
 747      write(*,*) 'eob_getLocalBodyIndices: done creating kdtree'
 748      write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 749    end if
 750
 751    ! do the search
 752    maxNumLocalObs = size(localBodyIndices)
 753    maxRadius = hLocalize**2
 754    refPosition(1) = ec_ra * sin(lon) * cos(lat)
 755    refPosition(2) = ec_ra * cos(lon) * cos(lat)
 756    refPosition(3) = ec_ra *            sin(lat)
 757    allocate(searchResults(maxNumLocalObsSearch))
 758    call kdtree2_r_nearest(tp=tree, qv=refPosition, r2=maxRadius, nfound=numLocalObsFoundSearch,&
 759                           nalloc=maxNumLocalObsSearch, results=searchResults)
 760    if (numLocalObsFoundSearch > maxNumLocalObsSearch) then
 761      call utl_abort('eob_getLocalBodyIndices: the parameter maxNumLocalObsSearch must be increased')
 762    end if
 763
 764    if ( vLocalize > 0.0d0 .and. vertLocation /= MPC_missingValue_R8 ) then
 765      ! copy search results to output vectors, only those within vertical localization distance
 766      numLocalObsFound = 0
 767      numLocalObs = 0
 768      do localObsIndex=1, numLocalObsFoundSearch
 769        distance = abs( vertLocation - ensObs%vertLocation(searchResults(localObsIndex)%idx) )
 770        if (distance <= vLocalize .and. ensObs%assFlag(searchResults(localObsIndex)%idx)==1) then
 771          numLocalObsFound = numLocalObsFound + 1
 772          if (numLocalObs < maxNumLocalObs) then
 773            numLocalObs = numLocalObs + 1
 774            localBodyIndices(numLocalObs) = searchResults(localObsIndex)%idx
 775            distances(numLocalObs) = sqrt(searchResults(localObsIndex)%dis)
 776          end if
 777        end if
 778      end do
 779    else
 780      ! no vertical location, so just copy results
 781      numLocalObsFound = 0
 782      numLocalObs = 0
 783      do localObsIndex=1, numLocalObsFoundSearch
 784        if (ensObs%assFlag(searchResults(localObsIndex)%idx)==1) then
 785          numLocalObsFound = numLocalObsFound + 1
 786          if (numLocalObs < maxNumLocalObs) then
 787            numLocalObs = numLocalObs + 1
 788            localBodyIndices(numLocalObs) = searchResults(localObsIndex)%idx
 789            distances(numLocalObs) = sqrt(searchResults(localObsIndex)%dis)
 790          end if
 791        end if
 792      end do      
 793    end if
 794    deallocate(searchResults)
 795
 796  end function eob_getLocalBodyIndices

 872  !--------------------------------------------------------------------------
 873  ! eob_setVertLocation
 874  !--------------------------------------------------------------------------
 875  subroutine eob_setVertLocation(ensObs, columnMeanTrl)
 876    !
 877    !:Purpose: Set the vertical location value for each observation that 
 878    !           will be used when doing vertical localization. For
 879    !           radiance observations, the level of the maximum value
 880    !           of the derivative of transmission is used. This value
 881    !           is also written in obsSpaceData in the OBS_ZHA column.
 882    !
 883    implicit none
 884
 885    ! arguments
 886    type(struct_eob)       , intent(inout) :: ensObs
 887    type(struct_columnData), intent(in)    :: columnMeanTrl
 888
 889    ! locals
 890    integer          :: obsIndex, headerIndex, channelIndex, tovsIndex, numTovsLevels, nosensor
 891    integer          :: levIndex, levIndexBelow, levIndexAbove, nLev_M
 892    integer          :: varNumber(ensObs%numObs), obsVcoCode(ensObs%numObs), codType(ensObs%numObs)
 893    real(8)          :: obsHeight, interpFactor, obsPPP(ensObs%numObs)
 894    real(8), pointer :: sfcPres_ptr(:,:), presM_ptr(:,:), heightM_ptr(:,:)
 895    type(rttov_profile), pointer :: profiles(:)
 896    logical          :: verbose = .false.
 897
 898    call eob_setAssFlag(ensObs)
 899
 900    call obs_extractObsRealBodyColumn(obsPPP, ensObs%obsSpaceData, OBS_PPP)
 901    call obs_extractObsIntBodyColumn(varNumber, ensObs%obsSpaceData, OBS_VNM)
 902    call obs_extractObsIntBodyColumn(obsVcoCode, ensObs%obsSpaceData, OBS_VCO)
 903    call obs_extractObsIntHeaderColumn(codType, ensObs%obsSpaceData, OBS_ITY)
 904
 905    if (ensObs%typeVertCoord == 'logPressure') then
 906
 907      presM_ptr   => col_getAllColumns(columnMeanTrl,'P_M')
 908      heightM_ptr => col_getAllColumns(columnMeanTrl,'Z_M')
 909      sfcPres_ptr => col_getAllColumns(columnMeanTrl,'P0')
 910      nLev_M = col_getNumLev(columnMeanTrl,'MM')
 911
 912      call tvs_getProfile(profiles,'nl')
 913
 914    end if
 915
 916    OBS_LOOP: do obsIndex = 1, ensObs%numObs
 917      headerIndex = obs_bodyElem_i(ensObs%obsSpaceData,OBS_HIND,obsIndex)
 918
 919      if( varNumber(obsIndex) == BUFR_NETS .or. varNumber(obsIndex) == BUFR_NEPS   .or.  &
 920          varNumber(obsIndex) == BUFR_NEUS .or. varNumber(obsIndex) == BUFR_NEVS   .or.  &
 921          varNumber(obsIndex) == BUFR_NESS .or. varNumber(obsIndex) == BUFR_NEPN   .or.  &
 922          varNumber(obsIndex) == BUFR_VIS  .or. varNumber(obsIndex) == BUFR_LOGVIS .or.  &
 923          varNumber(obsIndex) == BUFR_GUST .or.  &
 924          varNumber(obsIndex) == BUFR_radarPrecip .or. varNumber(obsIndex) == BUFR_logRadarPrecip ) then
 925
 926        ! all surface observations
 927        if (ensObs%typeVertCoord == 'logPressure') then
 928          ensObs%vertLocation(obsIndex) = log(sfcPres_ptr(1,headerIndex))
 929        else if (ensObs%typeVertCoord == 'depth') then
 930          ensObs%vertLocation(obsIndex) = 0.0D0
 931        else
 932          call utl_abort('eob_setVertLocation: unknown typeVertCoord:' // trim(ensObs%typeVertCoord))
 933        end if
 934
 935      else if (varNumber(obsIndex) == BUFR_NEZD) then
 936
 937        ! ZTD observation, try 0.7*Psfc (i.e. ~700hPa when Psfc=1000hPa)
 938        if (ensObs%typeVertCoord == 'logPressure') then
 939          ensObs%vertLocation(obsIndex) = log(0.7D0 * sfcPres_ptr(1,headerIndex))
 940        else
 941          call utl_abort('eob_setVertLocation: ZTD obs only compatible with logPressure coordinate')
 942        end if
 943
 944      else if (obsPPP(obsIndex) > 0.0d0 .and. obsVcoCode(obsIndex)==2) then
 945
 946        ! all pressure level observations
 947        if (ensObs%typeVertCoord == 'logPressure') then
 948          ensObs%vertLocation(obsIndex) = log(obsPPP(obsIndex))
 949        else
 950          call utl_abort('eob_setVertLocation: pressure obs only compatible with logPressure coordinate')
 951        end if
 952
 953      else if(obsVcoCode(obsIndex) == 1 .and. .not.bufr_isOceanObs(varNumber(obsIndex))) then
 954
 955        if (ensObs%typeVertCoord /= 'logPressure') then
 956          ! skip this obs if it will not be assimilated
 957          if (ensObs%assFlag(obsIndex) == 0) cycle OBS_LOOP
 958          ! otherwise, abort
 959          write(*,*) 'eob_setVertLocation: varNum = ', varNumber(obsIndex)
 960          call utl_abort('eob_setVertLocation: height level obs only compatible with logPressure coordinate')
 961        end if
 962
 963        ! all height level observations (not including surface obs)
 964        obsHeight = obsPPP(obsIndex)
 965
 966        ! find level just below the observation
 967        levIndexBelow = 0
 968        LEV_LOOP: do levIndex = 1, nLev_M
 969          if (obsHeight > heightM_ptr(levIndex,headerIndex)) then
 970            levIndexBelow = levIndex
 971            exit LEV_LOOP
 972          end if
 973        end do LEV_LOOP
 974
 975        ! set the log pressure for observation
 976        if (levIndexBelow == 1) then
 977          ! above top level, use top level pressure
 978          ensObs%vertLocation(obsIndex) = log(presM_ptr(1,headerIndex))
 979        else if (levIndexBelow == 0) then
 980          ! below bottom level, use surface pressure
 981          ensObs%vertLocation(obsIndex) = log(sfcPres_ptr(1,headerIndex))
 982        else
 983          ! interpolate
 984          levIndexAbove = levIndexBelow - 1
 985          interpFactor = ( obsHeight                              - heightM_ptr(levIndexBelow,headerIndex) ) /  &
 986                         ( heightM_ptr(levIndexAbove,headerIndex) - heightM_ptr(levIndexBelow,headerIndex) )
 987          ensObs%vertLocation(obsIndex) = interpFactor           * log(presM_ptr(levIndexAbove,headerIndex)) +  &
 988                                          (1.0D0 - interpFactor) * log(presM_ptr(levIndexBelow,headerIndex))
 989        end if
 990
 991      else if(tvs_isIdBurpTovs(codType(obsIndex))) then
 992
 993        if (ensObs%typeVertCoord /= 'logPressure') then
 994          call utl_abort('eob_setVertLocation: radiance obs only compatible with logPressure coordinate')
 995        end if
 996
 997        tovsIndex = tvs_tovsIndex(headerIndex)
 998        nosensor = tvs_lsensor(tovsIndex)
 999        numTovsLevels   = size(tvs_transmission(tovsIndex)%tau_levels,1)
1000        channelIndex = nint(obsPPP(obsIndex))

1246  ! eob_backgroundCheck
1247  !--------------------------------------------------------------------------
1248  subroutine eob_backgroundCheck(ensObs)
1249    !
1250    !:Purpose: Apply additional background using the ensemble spread.
1251    !
1252    implicit none
1253
1254    ! arguments:
1255    type(struct_eob), intent(inout) :: ensObs
1256
1257    ! locals:
1258    integer :: bodyIndexBeg, bodyIndexEnd, headerIndex, ivar, bodyIndex
1259    integer :: numRejected, numRejectedMpiGlobal, ierr, windCount
1260    real    :: sigo, sigb, omp, sig, reject_limit
1261    logical :: reject_wind
1262
1263    numRejected = 0
1264    reject_limit = 5.0
1265    do headerIndex = 1, obs_numheader(ensObs%obsSpaceData)
1266      bodyIndexBeg = obs_headElem_i(ensObs%obsSpaceData, OBS_RLN, headerIndex)
1267      bodyIndexEnd = obs_headElem_i(ensObs%obsSpaceData, OBS_NLV, headerIndex) + bodyIndexBeg - 1
1268      reject_wind = .false.
1269
1270      do bodyIndex = bodyIndexBeg, bodyIndexEnd
1271        if (obs_bodyElem_i(ensObs%obsSpaceData, OBS_ASS, bodyIndex) == obs_notAssimilated) cycle
1272        sigo = obs_bodyElem_r(ensObs%obsSpaceData, OBS_OER, bodyIndex)
1273        sigb = obs_bodyElem_r(ensObs%obsSpaceData, OBS_HPHT, bodyIndex)
1274        ! cut off at reject_limit standard deviations
1275        sig = reject_limit*(sigo**2 + sigb**2)**0.5
1276        omp = abs(obs_bodyElem_r(ensObs%obsSpaceData, OBS_OMP, bodyIndex))
1277        if (omp > sig) then
1278          call obs_bodySet_i(ensObs%obsSpaceData, OBS_ASS, bodyIndex, obs_notAssimilated)
1279          call obs_bodySet_i(ensObs%obsSpaceData, OBS_FLG, bodyIndex,  &
1280                             IBSET(obs_bodyElem_i(ensObs%obsSpaceData, OBS_FLG, bodyIndex),9))
1281          ivar = obs_bodyElem_i(ensObs%obsSpaceData, OBS_VNM, bodyIndex)
1282          reject_wind = bufr_isWindComponent(ivar)
1283          write(*,*) 'eob_backgroundCheck: headerIndex, omp, sig, ivar, reject_wind', &
1284                                           headerIndex, omp, sig, ivar, reject_wind
1285          numRejected = numRejected + 1
1286        end if
1287      end do
1288
1289      ! take the same reject decision for both components of the
1290      ! wind vector (and any other winds for that station).
1291      ! N.B.: This seems to assume only one level per header, this is generally 
1292      ! the case for wind observations currently, since radiosondes are 4D
1293      if (reject_wind) then 
1294        ! first count how many wind observations we have for this station
1295        windCount = 0
1296        do bodyIndex = bodyIndexBeg, bodyIndexEnd
1297          if (bufr_isWindComponent(obs_bodyElem_i(ensObs%obsSpaceData, OBS_VNM, bodyIndex))) then
1298            windCount = windCount + 1
1299          end if
1300        end do
1301        if (windCount > 2) then
1302          write(*,*) 'eob_backgroundCheck: WARNING' 
1303          write(*,*) 'Station ',headerIndex,' has ',windCount,' wind observations '
1304          write(*,*) 'Perhaps old radiosonde format - std dev not changed for other wind component'
1305        else
1306          do bodyIndex = bodyIndexBeg, bodyIndexEnd
1307            if (obs_bodyElem_i(ensObs%obsSpaceData, OBS_ASS, bodyIndex) == obs_notassimilated) cycle
1308            ivar = obs_bodyElem_i(ensObs%obsSpaceData, OBS_VNM, bodyIndex)
1309            if (bufr_isWindComponent(ivar) .and.  &
1310                obs_bodyElem_i(ensObs%obsSpaceData, OBS_ASS, bodyIndex) == obs_assimilated) then
1311              call obs_bodySet_i(ensObs%obsSpaceData, OBS_ASS, bodyIndex, obs_notAssimilated)
1312              call obs_bodySet_i(ensObs%obsSpaceData, OBS_FLG, bodyIndex,  &
1313                                 IBSET(obs_bodyElem_i(ensObs%obsSpaceData, OBS_FLG, bodyIndex),9))
1314              write(*,*) 'eob_backgroundCheck: other wind component headerIndex, ivar', &
1315                                               headerIndex, ivar
1316              numRejected = numRejected + 1
1317            end if
1318          end do
1319        end if
1320      end if
1321    end do
1322
1323    call rpn_comm_allreduce(numRejected,numRejectedMpiGlobal,1,'mpi_integer','mpi_sum','GRID',ierr)
1324    write(*,*)
1325    write(*,*) 'eob_backgroundCheck: number of observations rejected (local) =', numRejected
1326    write(*,*) 'eob_backgroundCheck: number of observations rejected (global)=', numRejectedMpiGlobal
1327
1328  end subroutine eob_backgroundCheck

1421  ! eob_huberNorm
1422  !--------------------------------------------------------------------------
1423  subroutine eob_huberNorm(ensObs)
1424    !
1425    !:Purpose: Apply huber norm quality control procedure. This modifies
1426    !           the OBS_OER value.
1427    !
1428    implicit none
1429
1430    ! arguments
1431    type(struct_eob), intent(inout) :: ensObs
1432
1433    ! locals
1434    integer           :: huberCount, huberCountMpiGlobal, ivar, windCount, ierr
1435    integer           :: bodyIndex, bodyIndexBeg, bodyIndexEnd, headerIndex
1436    real(pre_obsReal) :: c_limit, sig, sigo, sigb, omp, sigo_hub, sigo_hub_wind
1437    logical           :: reject_wind
1438
1439    c_limit = 2.0
1440    huberCount = 0
1441    do headerIndex = 1, obs_numheader(ensObs%obsSpaceData)
1442      bodyIndexBeg = obs_headElem_i(ensObs%obsSpaceData, OBS_RLN, headerIndex)
1443      bodyIndexEnd = obs_headElem_i(ensObs%obsSpaceData, OBS_NLV, headerIndex) + bodyIndexBeg - 1
1444      reject_wind = .false.
1445      sigo_hub_wind = 0.0
1446      do bodyIndex = bodyIndexBeg, bodyIndexEnd
1447        sigo = obs_bodyElem_r(ensObs%obsSpaceData, OBS_OER, bodyIndex)
1448        sigb = obs_bodyElem_r(ensObs%obsSpaceData, OBS_HPHT, bodyIndex)
1449        ! cut off at reject_limit standard deviations
1450        sig = c_limit*(sigo**2 + sigb**2)**0.5
1451        omp = abs(obs_bodyElem_r(ensObs%obsSpaceData, OBS_OMP, bodyIndex))
1452        if (omp > sig) then
1453          ! redefining the observational error such that the innovation
1454          ! is at exactly c_limit standard deviations.
1455          huberCount = huberCount + 1
1456          sigo_hub = ((omp/c_limit)**2-sigb**2)**0.5
1457          ivar = obs_bodyElem_i(ensObs%obsSpaceData, OBS_VNM, bodyIndex)
1458          if (bufr_isWindComponent(ivar)) then
1459            ! the other wind components will be changed at the same time (next if)
1460            reject_wind = .true.
1461            call obs_bodySet_r(ensObs%obsSpaceData, OBS_OER, bodyIndex, sigo_hub)
1462            ! this is for the special case both components of the wind innovation
1463            ! suggest using the Huber norm. 
1464            if (sigo_hub > sigo_hub_wind) then
1465              sigo_hub_wind = sigo_hub
1466            end if
1467          else
1468            call obs_bodySet_r(ensObs%obsSpaceData, OBS_OER, bodyIndex, sigo_hub)
1469          end if
1470        end if
1471      end do
1472
1473      ! Give the same inflated observation error to both wind components.
1474      ! this part assumes that modern sondes are used (at most two
1475      ! wind observations per station.
1476      if (reject_wind) then
1477        ! first count how many wind observations we have for this station
1478        windCount = 0
1479        do bodyIndex = bodyIndexBeg, bodyIndexEnd
1480          if (bufr_isWindComponent(obs_bodyElem_i(ensObs%obsSpaceData, OBS_VNM, bodyIndex))) then
1481            windCount = windCount + 1
1482          end if
1483        end do
1484        if (windCount > 2) then
1485          write(*,*) 'Warning Hubernorm' 
1486          write(*,*) 'Station ',headerIndex,' has ',windCount,' wind observations '
1487          write(*,*) 'Perhaps old radiosonde format - std dev not changed for other wind component'
1488        else
1489          huberCount = huberCount + 1
1490          do bodyIndex = bodyIndexBeg, bodyIndexEnd
1491            if (bufr_isWindComponent(obs_bodyElem_i(ensObs%obsSpaceData, OBS_VNM, bodyIndex))) then
1492              call obs_bodySet_r(ensObs%obsSpaceData, OBS_OER, bodyIndex, sigo_hub_wind)
1493            end if
1494          end do
1495        end if
1496      end if 
1497    end do
1498
1499    call rpn_comm_allreduce(huberCount, huberCountMpiGlobal, 1, 'mpi_integer', 'mpi_sum', 'GRID', ierr)
1500    write(*,*)
1501    write(*,*) 'eob_huberNorm: number of obs with increased error stddev (local) = ', huberCount
1502    write(*,*) 'eob_huberNorm: number of obs with increased error stddev (global)= ', huberCountMpiGlobal
1503
1504  end subroutine eob_huberNorm

1557                                        numGroupsToDivideMembers_opt, &
1558                                        maxNumMembersPerGroup_opt)
1559    !
1560    !:Purpose: get memberIndex array corresponding to the full ensemble set. This 
1561    !           is useful when ensObs is a subset of full ensemble members. 
1562    !           If first member in ensObs is member 6, to get the full ensemble set equivalent of ensObs members:
1563    !           a) When members are not grouped (numGroupsToDivideMembers=1), all members are offset by 
1564    !              memberIndexOffset (e.g. 6, 7, ..., 6+ensObs%numMermbers)
1565    !           b) When members are grouped (numGroupsToDivideMembers/=1), members within each group 
1566    !              are offset by memberIndexOffset but there is increment of maxNumMembersPerGroup_opt 
1567    !              to jump to the next group (e.g. if maxNumMembersPerGroup_opt=10, for first group 6, 7, 8, 9, 
1568    !              for second group 6+10, 7+10, 8+10, 9+10, and so on)
1569    !
1570    implicit none
1571
1572    ! arguments
1573    type(struct_eob),  intent(in) :: ensObs
1574    integer,        intent(inout) :: memberIndexArray(:)
1575    integer, optional, intent(in) :: numGroupsToDivideMembers_opt
1576    integer, optional, intent(in) :: maxNumMembersPerGroup_opt
1577
1578    ! locals
1579    integer :: memberIndex, groupIndex, memberIndexOffset, memberIndexInGroup
1580    integer :: numGroupsToDivideMembers, numMembersPerGroup
1581
1582    if (present(numGroupsToDivideMembers_opt)) then
1583      numGroupsToDivideMembers = numGroupsToDivideMembers_opt
1584    else
1585      numGroupsToDivideMembers = 1
1586    end if
1587
1588    memberIndexOffset = ensObs%fileMemberIndex1
1589
1590    if (numGroupsToDivideMembers == 1) then
1591      do memberIndex = 1, ensObs%numMembers
1592        memberIndexArray(memberIndex) = memberIndex + memberIndexOffset - 1
1593      end do
1594    else
1595      if (.not. present(maxNumMembersPerGroup_opt)) then
1596        call utl_abort('getMemberIndexInFullEnsSet: maxNumMembersPerGroup_opt input argument missing')
1597      end if
1598
1599      ! divide members into groups
1600      numMembersPerGroup = ensObs%numMembers / numGroupsToDivideMembers
1601      if (numMembersPerGroup > maxNumMembersPerGroup_opt) then
1602        call utl_abort('getMemberIndexInFullEnsSet: numMembersPerGroup > maxNumMembersPerGroup_opt')
1603      end if
1604
1605      memberIndex = 0
1606      do groupIndex = 1, numGroupsToDivideMembers 
1607        do memberIndexInGroup = 1, numMembersPerGroup
1608          memberIndex = memberIndex + 1
1609          memberIndexArray(memberIndex) = (groupIndex - 1) * maxNumMembersPerGroup_opt + &
1610                                          memberIndexInGroup + memberIndexOffset - 1
1611        end do
1612      end do
1613
1614    end if

1621  subroutine max_transmission(transmission, numLevels, transIndex, rttovPres, maxLnP)
1622    !
1623    !:Purpose: Determine the height in log pressure where we find the maximum 
1624    !           value of the first derivative of transmission with respect to 
1625    !           log pressure
1626    !
1627    implicit none
1628
1629    ! arguments
1630    type(rttov_transmission), intent(in)  :: transmission ! transmission (rttov type)
1631    integer(kind=jpim)      , intent(in)  :: numLevels    ! number of RTTOV levels
1632    integer                 , intent(in)  :: transIndex   ! index of transmission%tau_levels
1633    real(kind=jprb), pointer, intent(in)  :: rttovPres(:) ! pressure of RTTOV levels
1634    real(8)                 , intent(out) :: maxLnP       ! log pressure of maximum
1635
1636    ! locals
1637    integer :: levIndex
1638    real(8) :: lnPres(numLevels), avgPres(numLevels-1)
1639    real(8) :: diffTau, derivTau(numLevels), maxDeriv
1640    integer :: nAvgLev, maxIndex
1641
1642    nAvgLev = numLevels - 1
1643    lnPres(:) = log(rttovPres(:)*MPC_PA_PER_MBAR_R8)
1644    ! calculate the first derivative of transmission with respect to log pressure
1645    ! and find the level index for its maximum
1646    maxDeriv = -0.1d0
1647    derivTau(1) = 0.0d0
1648    maxIndex = numLevels
1649    do levIndex = 2, numLevels
1650      avgPres(levIndex-1) = 0.5d0*(lnPres(levIndex)+lnPres(levIndex-1))
1651      diffTau = transmission%tau_levels(levIndex-1,transIndex) - transmission%tau_levels(levIndex,transIndex)
1652      derivTau(levIndex) = diffTau / (lnPres(levIndex)-lnPres(levIndex-1))
1653      if (derivTau(levIndex)>maxDeriv) then
1654        maxDeriv = derivTau(levIndex)
1655        maxIndex = levIndex
1656      end if
1657    end do
1658
1659    ! get the height in log pressure for the level index (maxIndex) found above
1660    if (maxIndex==1) maxIndex = maxIndex + 1
1661    if ((maxIndex==2).or.(maxIndex==numLevels)) then
1662      maxLnP = avgPres(maxIndex-1)
1663    else
1664      call get_peak(maxIndex,nAvgLev,avgPres,derivTau,maxLnP)
1665    end if
1666
1667  end subroutine max_transmission

1671  !--------------------------------------------------------------------------
1672  subroutine get_peak(maxIndex,nlev,lnp,deriv,maxLnP)
1673    !
1674    !:Purpose: Do quadratic interpolation to find pressure of peak transmission.
1675    !
1676    implicit none
1677
1678    ! arguments
1679    integer, intent(in)    :: maxIndex, nlev
1680    real(8), intent(in)    :: lnp(nlev), deriv(nlev+1)
1681    real(8), intent(inout) :: maxLnP
1682
1683    ! locals
1684    external :: dgesv
1685    integer, parameter :: N=3
1686    integer :: info
1687    integer, parameter :: lda=N, ldb=N, nrhs=1
1688    integer :: ipiv(N)
1689    real(8) :: A(lda,N),B(ldb,nrhs)
1690    integer :: index1, index2
1691
1692    index2 = 0
1693    do index1=maxIndex-1,maxIndex+1
1694      index2 = index2 + 1
1695      A(index2,1) = lnp(index1-1)*lnp(index1-1)
1696      A(index2,2) = lnp(index1-1)
1697      A(index2,3) = 1.0d0
1698      B(index2,1) = deriv(index1)
1699    end do
1700
1701    call dgesv(N,nrhs,A,lda,ipiv,B,ldb,info)
1702
1703    if (info==0) then
1704      maxLnP = -0.5*(B(2,1)/B(1,1))
1705    else
1706      maxLnP = lnp(maxIndex-1)
1707    end if
1708
1709  end subroutine get_peak
