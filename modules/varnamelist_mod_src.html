136    function vnl_varListIndex3d(varName) result(listIndex)
137      !
138      !:Purpose: To get the 3d list index from the variable name
139      !
140
141      implicit none
142
143      ! Arguments:
144      character(len=*), intent(in) :: varName
145      integer                      :: listIndex
146      
147      !Local:
148      integer                      :: jvar
149
150      listIndex=-1
151      do jvar=1,vnl_numvarmax3D
152        if( varName == vnl_varNameList3d(jvar)) then
153          listIndex=jvar
154          exit
155        end if
156      end do
157
158      if( listIndex <= 0 ) then
159        call utl_abort('vnl_varListIndex3D: Unknown variable name! ' // varName)
160      end if

286    function vnl_varnameFromVarnum( varNumber, varNumberChm_opt, modelName_opt ) result(varName)
287      !
288      !:Purpose: To get the variable name from the variable number
289      !
290
291      implicit none
292
293      !Arguments:
294      integer, intent(in) :: varNumber
295      integer, intent(in), optional :: varNumberChm_opt
296      character(len=*), intent(in), optional :: modelName_opt
297      character(len=4)    :: varName
298
299      varName='    '
300      select case (varNumber)
301      case ( BUFR_NEUU, BUFR_NEUS, BUFR_NEAL )
302        varName='UU'
303      case( BUFR_NEVV, BUFR_NEVS )
304        varName='VV'
305      case( BUFR_NETT, BUFR_NETS )
306        varName='TT'
307      case( BUFR_NEDZ, BUFR_NEGZ )
308        varName='Z_T'
309      case( BUFR_NEHU, BUFR_NEHS, BUFR_NEES, BUFR_NESS )
310        varName='HU'
311      case( BUFR_NEPS, BUFR_NEPN )
312        varName='P0'
313      case ( BUFR_NERF, BUFR_NEBD, BUFR_NEZD )
314        varName='TT'   ! temporarily associate refractivity and ZTD with temperature
315      case ( BUFR_NEDW )
316        varName='DW'
317      case ( BUFR_SST )
318        varname='TG'
319      case ( BUFR_ICEC, BUFR_ICEP, BUFR_ICEV, BUFR_ICES )
320        varname='GL'
321      case ( bufr_vis )
322        varname='VIS'
323      case ( bufr_logVis )
324        varname='LVIS'
325      case ( bufr_radarPrecip )
326        varname='PR'
327      case ( bufr_logRadarPrecip )
328        varname='LPR'
329      case ( bufr_gust )
330        varname='WGE'
331      case ( bufr_riverFlow )
332        varname='QO1'
333      case ( BUFR_NEFS, bufr_radvel) 
334        varname='UV'
335      case default
336        !
337        ! Search for constituents. Identification depends on value and presence of second parameter.
338        !
339        if (present(varNumberChm_opt)) then 
340           select case (varNumberChm_opt)
341              case(BUFR_NECH_O3)
342                 varname='TO3' 
343              case(BUFR_NECH_H2O)
344                 varname='HU'
345              case(BUFR_NECH_CH4)
346                 varname='TCH4'
347              case(BUFR_NECH_CO2)
348                 varname='TCO2'
349              case(BUFR_NECH_CO)
350                 varname='TCO'
351              case(BUFR_NECH_NO2)
352                 varname='TNO2'
353              case(BUFR_NECH_N2O)
354                 varname='TN2O' 
355              case(BUFR_NECH_NO)
356                 varname='TNO'
357              case(BUFR_NECH_HCHO)
358                 varname='THCH'
359              case(BUFR_NECH_SO2)
360                 varname='TSO2'
361              case(BUFR_NECH_NH3)
362                 varname='TNH3'
363              case(BUFR_NECH_PM25)
364                 varname='AF'
365              case(BUFR_NECH_PM10)
366                 varname='AC'
367              case default
368                 call utl_abort( 'vnl_varnameFromVarnum: Unknown variable number! ' // &
369                   utl_str(varNumber) // ', ' // utl_str(varNumberChm_opt) )
370           end select
371           if (present(modelName_opt)) then
372             if (trim(modelName_opt) == 'GEM') then
373               select case (varNumberChm_opt)
374                 case(BUFR_NECH_O3)
375                   varname='O3L'  
376                 case(BUFR_NECH_CH4)
377                   varname='CH4L'
378                 case(BUFR_NECH_N2O)
379                   varname='N2OL'
380                 case default
381                   call utl_abort( 'vnl_varnameFromVarnum: Unknown variable number or model! ' // &
382                      utl_str(varNumber) // ', ' // utl_str(varNumberChm_opt) // ', ' // trim(modelName_opt) )
383               end select
384             end if
385           end if      
386        else
387           write(*,*) 'vnl_varnameFromVarnum: Unknown variable number! ',varNumber
388           call utl_abort('vnl_varnameFromVarnum')
389        end if 
390      end select
391
392    end function vnl_varnameFromVarnum

397    function vnl_varnumFromVarName(varName,varKind_opt) result(varNumber)
398      !
399      !:Purpose: Identifies varNumber from varName for use in assimilating
400      !           obs in the CH family.   
401      !           Here, for weather variables, there is a 1-1 association between
402      !           a variable name and an observation unit.
403      !           So one must provide the name directly associated to a single
404      !           BUFR code.
405      !           As such, weather variable varNames may not necessarily be a
406      !           member of the vnl_varNameList for this routine only.
407      !   
408      !           For constituents, the varNumber refers only to the field/
409      !           variable and not units. As consequence, there is a unique
410      !           pairing of varNumbers with the varNames from vnl_VarNameList.
411      !
412
413      implicit none
414
415      !Arguments:
416      character(len=*),  intent(in) :: varName
417      character(len=*),  intent(in), optional :: varKind_opt
418      integer    :: varNumber
419      
420      varNumber=0
421      select case (varName)
422      
423      ! Weather variables. Must provide name directly associated to a single
424      ! BUFR code. As such, the varName may not necessarily be a member of the
425      ! vnl_varNameList for this routine only.
426
427      case('UU')
428        varNumber=BUFR_NEUU
429      case('US')
430        varNumber=BUFR_NEUS
431      case('VV')
432        varNumber=BUFR_NEVV
433      case('VS')
434        varNumber=BUFR_NEVS
435      case('TT')
436        varNumber=BUFR_NETT
437      case('TS')
438        varNumber=BUFR_NETS
439      case('RF')            
440        varNumber=BUFR_NERF
441      case('BD')
442        varNumber=BUFR_NEBD
443      case('ZD')
444        varNumber=BUFR_NEZD
445      case('GZ')
446        varNumber=BUFR_NEGZ
447      case('DZ')
448        varNumber=BUFR_NEDZ
449      case('UV')
450        varNumber=BUFR_NEFS 
451      case('HU')
452        if (present(varKind_opt)) then
453           if (varKind_opt == 'CH') then
454              varNumber=BUFR_NECH_H2O
455           else
456              varNumber=BUFR_NEHU
457           end if
458         else
459            varNumber=BUFR_NEHU
460         end if
461      case('HS')
462        varNumber=BUFR_NEHS
463      case('SS')
464        varNumber=BUFR_NESS
465      case('P0','PS')
466        varNumber=BUFR_NEPS
467      case('PN')
468        varNumber=BUFR_NEPN
469      case('DW')
470        varNumber=BUFR_NEDW
471      case('WGE')
472        varNumber=bufr_gust
473      case('LVIS')
474        varNumber=bufr_logVis
475      case('VIS')
476        varNumber=bufr_vis
477
478      ! Atmospheric constituents other than HU
479      case('TO3','O3L')
480        varNumber=BUFR_NECH_O3
481      case('TH2O')
482        varNumber=BUFR_NECH_H2O
483      case('TCH4','CH4L')
484        varNumber=BUFR_NECH_CH4
485      case('TCO2')
486        varNumber=BUFR_NECH_CO2
487      case('TCO','ECO')
488        varNumber=BUFR_NECH_CO
489      case('TNO2','ENO2')
490        varNumber=BUFR_NECH_NO2
491      case('TN2O','N2OL')
492        varNumber=BUFR_NECH_N2O
493      case('TNO')
494        varNumber=BUFR_NECH_NO
495      case('THCH','EHCH')
496        varNumber=BUFR_NECH_HCHO
497      case('TSO2','ESO2')
498        varNumber=BUFR_NECH_SO2
499      case('TNH3','ENH3')
500        varNumber=BUFR_NECH_NH3
501      case('AF')
502        varNumber=BUFR_NECH_PM25
503      case('AC')
504        varNumber=BUFR_NECH_PM10
505        
506      case default
507         call utl_abort('vnl_varnumFromVarName: Unknown variable name ' // trim(varName) )
508      end select
509      
510    end function vnl_varnumFromVarname

513   ! vnl_varLevelFromVarname
514   !--------------------------------------------------------------------------
515    function vnl_varLevelFromVarname(varName) result(varLevel)
516      !
517      !:Purpose: To get variable level list from variable name 
518      !
519      implicit none
520
521      !Arguments:
522      character(len=*), intent(in)   :: varName
523      character(len=4)               :: varLevel
524
525      !Locals:
526      integer                :: nulnam, ierr
527      integer, external      :: fnom, fclos
528      logical, save          :: firstTime = .true.
529
530      ! Namelist variables
531      character(len=4), save :: forceSfcOnly(vnl_numVarMax) ! List of 3D variable names only allocated at the surface
532
533      NAMELIST /namvnl/forceSfcOnly
534
535      if (firstTime) then
536        firstTime = .false.
537        ! default values (not a valid variable name)
538        forceSfcOnly(:) = 'XXXX'
539
540        if (utl_isNamelistPresent('namvnl','./flnml')) then
541          nulnam = 0
542          ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
543          read (nulnam, nml = NAMVNL, iostat = ierr)
544          if ( ierr /= 0 ) call utl_abort('vnl_varLevelFromVarname: Error reading namelist')
545          if ( mmpi_myid == 0 ) write(*,nml=namvnl)
546          ierr = fclos(nulnam)
547        else
548          write(*,*)
549          write(*,*) 'vnl_varLevelFromVarname: namvnl is missing in the namelist. The default value will be taken.'
550        end if
551      end if
552
553      varLevel = varLevelList(vnl_varListIndex(varName))
554      if (any(forceSfcOnly(:) == varName)) then
555        if (varLevel == 'TH') then
556          varLevel = 'SFTH'
557        else if (varLevel == 'MM') then
558          varLevel = 'SFMM'
559        else
560          call utl_abort('vnl_varLevelFromVarname: something is wrong')
561        end if
562      end if
563
564    end function vnl_varLevelFromVarname
