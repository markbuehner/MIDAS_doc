  90  subroutine csg_setup(nens_in, hco_in, vco_in)
  91    !
  92    !:Purpose: Main setup routine for this module
  93    !
  94    implicit none
  95    ! arguments:
  96    integer, intent(in)                     :: nens_in
  97    type(struct_vco), pointer, intent(in)   :: vco_in
  98    type(struct_hco), pointer, intent(in)   :: hco_in
  99    ! locals:
 100    integer :: nulnam, ierr, waveBandIndex, memberIndex
 101    integer :: fclos, fnom
 102    real(8) :: zps
 103
 104    NAMELIST /NAMCALCSTATS_GLB/ntrunc,waveBandPeaks
 105
 106    write(*,*)
 107    write(*,*) 'csg_setup: Starting...'
 108
 109    nens=nens_in
 110    allocate(cflensin(nens))
 111    do memberIndex = 1, nEns
 112      call fln_ensfileName(cflensin(memberIndex), 'ensemble', memberIndex_opt=memberIndex)
 113    end do
 114    
 115    if ( mmpi_myid == 0 ) then
 116      call mpc_printConstants(6)
 117      call pre_printPrecisions
 118    end if
 119
 120    ! parameters from namelist (date in filename should come directly from sequencer?)
 121    ntrunc=108
 122    waveBandPeaks(:) = -1.0d0
 123
 124    nulnam=0
 125    ierr=fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
 126    read(nulnam,nml=NAMCALCSTATS_GLB)
 127    if (ierr /= 0) call utl_abort('csg_setup: Error reading namelist NAMCALCSTATS_GLB')
 128    if (mmpi_myid == 0) write(*,nml=NAMCALCSTATS_GLB)
 129    ierr=fclos(nulnam)
 130
 131    !- Setup horizontal grid
 132    hco_ens => hco_in
 133    ni=hco_in%ni
 134    nj=hco_in%nj
 135
 136    gridSpacingInKm = ec_ra * hco_in%dlon / 1000.d0
 137    write(*,*) 'Grid Spacing in Km = ', gridSpacingInKm
 138
 139    ! setup mpi local grid parameters
 140    call mmpi_setup_latbands(nj, latPerPE, latPerPEmax, myLatBeg, myLatEnd)
 141    call mmpi_setup_lonbands(ni, lonPerPE, lonPerPEmax, myLonBeg, myLonEnd)
 142    
 143    !- Setup vertical levels
 144    vco_ens => vco_in
 145    nLevEns_M=vco_in%nlev_M
 146    nLevEns_T=vco_in%nlev_T
 147    nLevPtot=nLevEns_M-1 ! ignore streamfunction at hyb=1, since highly correlated with next level
 148    varLevOffset(1) = 0
 149    varLevOffset(2) = 1*nLevEns_M
 150    varLevOffset(3) = 2*nLevEns_M
 151    varLevOffset(4) = 2*nLevEns_M+1*nLevEns_T
 152    varLevOffset(5) = 2*nLevEns_M+2*nLevEns_T
 153    nkgdimEns=nLevEns_M*2+nLevEns_T*2+1 ! NO TG !!!
 154    nla=(ntrunc+1)*(ntrunc+2)/2
 155
 156    !- Setup the global spectral transform 
 157    gstID_nkgdimEns = gst_setup(ni,nj,ntrunc,nkgdimEns)
 158    gstID_nLevEns_M = gst_setup(ni,nj,ntrunc,nLevEns_M)
 159    gstID_nLevEns_T_P1 = gst_setup(ni,nj,ntrunc,nLevEns_T+1)
 160
 161    ! setup mpi local spectral vector parameters
 162    call mmpi_setup_m(ntrunc, mymBeg, mymEnd, mymSkip, mymCount)
 163    call mmpi_setup_n(ntrunc, mynBeg, mynEnd, mynSkip, mynCount)
 164    call gst_ilaList_mpiglobal(ilaList_mpiglobal, nla_mpilocal, maxMyNla,  &
 165         gstID_nkgdimEns, mymBeg, mymEnd, mymSkip, mynBeg, mynEnd, mynSkip)
 166    call gst_ilaList_mpilocal(ilaList_mpilocal,  &
 167         gstID_nkgdimEns, mymBeg, mymEnd, mymSkip, mynBeg, mynEnd, mynSkip)
 168
 169    ! setup ensemble members mpi partinionning (when working with struct_ens)
 170    call mmpi_setup_levels(nEns,myMemberBeg,myMemberEnd,myMemberCount)
 171    call rpn_comm_allreduce(myMemberCount, maxMyMemberCount, &
 172                            1,"MPI_INTEGER","MPI_MAX","GRID",ierr)
 173    nEnsOverDimension = mmpi_npex * maxMyMemberCount
 174    
 175    !- Setup ip1s
 176    allocate(nip1_M(nLevEns_M))
 177    nip1_M(:)=vco_in%ip1_M(:)
 178    allocate(nip1_T(nLevEns_T))
 179    nip1_T(:)=vco_in%ip1_T(:)
 180
 181    !- Estimate the pressure profile for each vertical grid    
 182    zps = 101000.D0
 183    call czp_fetch1DLevels(vco_in, zps, &
 184                           profM_opt=pressureProfile_M, profT_opt=pressureProfile_T)
 185
 186    !
 187    !- Setup variable names
 188    !
 189    nomvar3d(1,modelSpace)='UU'
 190    nomvar3d(2,modelSpace)='VV'
 191    nomvar3d(3,modelSpace)='TT'
 192    nomvar3d(4,modelSpace)='LQ'
 193    nomvar2d(1,modelSpace)='P0'
 194    
 195    nomvar3d(1,cvSpace)='PP'
 196    nomvar3d(2,cvSpace)='CC'
 197    nomvar3d(3,cvSpace)='TT'
 198    nomvar3d(4,cvSpace)='LQ'
 199    nomvar2d(1,cvSpace)='P0'
 200    
 201    nomvar3d(1,cvUnbalSpace)='PP'
 202    nomvar3d(2,cvUnbalSpace)='UC'
 203    nomvar3d(3,cvUnbalSpace)='UT'
 204    nomvar3d(4,cvUnbalSpace)='LQ'
 205    nomvar2d(1,cvUnbalSpace)='UP'
 206
 207    nomvar(1:4,:) = nomvar3d(1:4,:)
 208    nomvar(5,:)   = nomvar2d(1,:)
 209    
 210    !
 211    !- Wave band decomposition option (available in )
 212    !
 213    nWaveBand = count(waveBandPeaks .ge. 0)
 214    if ( nWaveBand < 1 ) then
 215       nWaveBand = 1
 216    else if (nWaveBand == 1) then
 217       write(*,*) 'You have specified only ONE waveBandPeaks'
 218       call utl_abort('calbmatrix_glb')
 219    else
 220       write(*,*)
 221       write(*,*) 'WaveBand decomposition is ACTIVATED'
 222    end if
 223    
 224    ! Make sure that the wavenumbers are in the correct (decreasing) order
 225    do waveBandIndex = 1, nWaveBand-1
 226       if ( waveBandPeaks(waveBandIndex)-waveBandPeaks(waveBandIndex+1) <= 0 ) then
 227          write(*,*) 'csg_setup: waveBandPeaks are not in decreasing wavenumber order'
 228          call utl_abort('calbmatrix_glb')
 229       end if
 230    end do
 231
 232    ! Make sure the truncation is compatible with the waveBandPeaks
 233    if ( ntrunc < nj-1 .and. nWaveBand > 1 ) then
 234       write(*,*) 'csg_setup: The truncation is not compatible with wave band decomposition'
 235       write(*,*) '                 ntrunc should = ', nj-1
 236       call utl_abort('calbmatrix_glb')
 237    end if
 238
 239    !
 240    !- Ending
 241    !
 242    initialized = .true.
 243
 244    write(*,*)
 245    write(*,*) 'csg_setup: Done!'
 246
 247  end subroutine csg_setup

 253    !
 254    !:Purpose: Master routine for Bhi computation in global mode
 255    !
 256    implicit none
 257    ! locals:
 258    integer :: ierr, nulnam
 259    integer :: fclos, fnom
 260    character(len=12) :: formulation ! Bhi formulation

 266

 268    ierr=fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)

 277    case ('latbands')
 278      call csg_computeBhiLatBands
 279    case default
 280      write(*,*)
 281      write(*,*) 'csg_computeBhi: Unknown value of FORMULATION for Bhi computation: ',formulation
 282      write(*,*) 'Please select legacy or latbands'
 283      call utl_abort('csg_computeBhi')
 284    end select

 286  end subroutine csg_computeBhi

 292    !
 293    !:Purpose: Computation of Bhi using the legacy formulation
 294    !
 295    implicit none
 296    real(4), pointer :: ensPerturbations(:,:,:,:), ens_ptr(:,:,:,:)
 297    real(4), pointer :: ensBalPerturbations(:,:,:,:)
 298    real(8), pointer :: stddev3d(:,:,:), stddev3dBal(:,:,:), stddev3dUnbal(:,:,:)
 299    real(8), pointer :: stddev3d_ptr(:,:,:)
 300    real(8), pointer :: stddevZonAvg(:,:), stddevZonAvgBal(:,:), stddevZonAvgUnbal(:,:)
 301    real(8), allocatable :: PtoT(:,:,:),theta1(:,:),theta2(:,:)
 302    real(8), allocatable :: corns(:,:,:),rstddev(:,:)
 303
 304    integer :: variableType 
 305
 306    allocate(ensPerturbations(myLonBeg:myLonEnd, myLatBeg:myLatEnd, nkgdimEns, nens))
 307    allocate(ensBalPerturbations(myLonBeg:myLonEnd, myLatBeg:myLatEnd, nLevEns_T+1, nens))
 308    allocate(theta1(nlevEns_M,nj))
 309    allocate(stddev3d(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nkgdimEns))
 310    allocate(stddevZonAvg(nkgdimEns,nj))
 311    allocate(PtoT(nlevEns_T+1,nlevEns_M,nj))
 312    allocate(theta2(nlevEns_M,nj))
 313    allocate(stddev3dBal(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nLevEns_T+1))
 314    allocate(stddev3dUnbal(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nkgdimEns))
 315    allocate(stddevZonAvgBal(nLevEns_T+1,nj))
 316    allocate(stddevZonAvgUnbal(nkgdimEns,nj))
 317    allocate(corns(nkgdimEns,nkgdimEns,0:ntrunc))
 318    allocate(rstddev(nkgdimEns,0:ntrunc))
 319
 320    write(*,*) 'Initializing ensemble arrays to claim memory'
 321    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 322    ensPerturbations(:,:,:,:) = 0.0
 323    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 324    ensBalPerturbations(:,:,:,:) = 0.0
 325    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 326
 327    call readEnsemble(ensPerturbations)
 328    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 329
 330    call removeMean(ensPerturbations)
 331
 332    call uv_to_psichi(ensPerturbations)
 333
 334    call calcStddev3d(ensPerturbations,stddev3d,nkgdimens)
 335
 336    call calcZonAvg(stddevZonAvg,stddev3d,nkgdimens)
 337
 338    call calcTheta(ensPerturbations,theta1) ! theta1 is put in glbcov and used for analysis!
 339    if (mmpi_myid == 0) write(301,*) theta1
 340
 341    call removeBalancedChi(ensPerturbations,theta1)
 342
 343    call normalize3d(ensPerturbations,stddev3d)
 344
 345    call calcPtoT(ensPerturbations,PtoT)
 346    if (mmpi_myid == 0) write(303,*) PTOT(:,:,1)
 347    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 348
 349!    call calcTheta(ensPerturbations,theta2) ! theta2 is used previously for computing unbalanced Chi!
 350!    if (mmpi_myid == 0) write(302,*) theta2
 351
 352    call removeBalancedT_Ps(ensPerturbations,ensBalPerturbations,PtoT)
 353    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 354
 355!    call removeBalancedChi(ensPerturbations,theta2)
 356
 357    call multiply3d(ensPerturbations,stddev3d,nkgdimens)
 358
 359    ens_ptr(myLonBeg:,myLatBeg:,1:,1:) => ensBalPerturbations(:,:,1:nLevEns_T,:)
 360    stddev3d_ptr(myLonBeg:,myLatBeg:,1:) => stddev3d(:,:,(2*nLevEns_M+1):(2*nLevEns_M+nLevEns_T))
 361    call multiply3d(ens_ptr, stddev3d_ptr, nLevEns_T)
 362
 363    ens_ptr(myLonBeg:,myLatBeg:,1:,1:) => ensBalPerturbations(:,:,(nLevEns_T+1):(nLevEns_T+1),:)
 364    stddev3d_ptr(myLonBeg:,myLatBeg:,1:) => stddev3d(:,:,(2*nLevEns_M+2*nLevEns_T+1):(2*nLevEns_M+2*nLevEns_T+1))
 365    call multiply3d(ens_ptr, stddev3d_ptr, 1)

 369    call spectralFilter(ensBalPerturbations,nLevEns_T+1)
 370
 371    call calcStddev3d(ensPerturbations,stddev3dUnbal,nkgdimens)
 372
 373    call calcStddev3d(ensBalPerturbations,stddev3dBal,nLevEns_T+1)
 374
 375    call calcZonAvg(stddevZonAvgUnbal,stddev3dUnbal,nkgdimens)
 376
 377    call calcZonAvg(stddevZonAvgBal,stddev3dBal,nLevEns_T+1)
 378
 379    call normalize3d(ensPerturbations,stddev3dUnbal)
 380
 381    call removeGlobalMean(ensPerturbations)
 382
 383    call calcCorrelations(ensPerturbations,corns,rstddev)
 384
 385    variableType = cvUnbalSpace
 386    call writeStats(corns,rstddev,ptot,theta1)

 393    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 394
 395    if (mmpi_myid == 0) then
 396      write(200,*) stddevZonAvg(1:nlevEns_M,:)
 397      write(201,*) stddevZonAvg((1+1*nlevEns_M):(2*nlevEns_M),:)
 398      write(202,*) stddevZonAvg((1+2*nlevEns_M):(3*nlevEns_T),:)
 399      write(203,*) stddevZonAvg((1+2*nlevEns_M+1*nlevEns_T):(2*nlevEns_M+2*nlevEns_T),:)
 400      write(204,*) stddevZonAvg((1+2*nlevEns_M+2*nlevEns_T),:)/1.0d2
 401
 402      write(400,*) stddevZonAvgUnbal(1:nlevEns_M,:)

 406      write(404,*) stddevZonAvgUnbal((1+2*nlevEns_M+2*nlevEns_T),:)/1.0d2
 407    end if
 408

 414  subroutine csg_computeBhiLatBands
 415    !
 416    !:Purpose: Computation of Bhi on a set of latitude bands
 417    !
 418    implicit none
 419    
 420    integer :: variableType, latIndex, jlatband, lat1, lat2, lat3
 421    real(4), pointer     :: ensPerturbations(:,:,:,:)
 422    real(8), pointer     :: stddev3d(:,:,:)
 423    real(8), pointer     :: stddevZonAvg(:,:)
 424    real(8), allocatable :: corns(:,:,:),rstddev(:,:)
 425    real(8) :: latMask(nj)
 426
 427    allocate(ensPerturbations(ni,nj,nkgdimEns,nens))
 428    allocate(stddev3d(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nkgdimEns))
 429    allocate(stddevZonAvg(nkgdimEns,nj))
 430    allocate(corns(nkgdimEns,nkgdimEns,0:ntrunc))
 431    allocate(rstddev(nkgdimEns,0:ntrunc))
 432
 433    call readEnsemble(ensPerturbations)
 434    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 435
 436    call removeMean(ensPerturbations)
 437
 438    call uv_to_psichi(ensPerturbations)
 439
 440    call calcStddev3d(ensPerturbations,stddev3d,nkgdimens)
 441
 442    call calcZonAvg(stddevZonAvg,stddev3d,nkgdimens)
 443
 444    call normalize3d(ensPerturbations,stddev3d)
 445
 446    call removeGlobalMean(ensPerturbations)
 447
 448    do jlatband = 1, 3
 449      write(*,*) 'csg_computeBhiLatBands: selected LATBAND = ',jlatband
 450      lat1=nj/4
 451      lat2=nj/2
 452      lat3=3*nj/4
 453      write(*,*) 'lat1,2,3=',lat1,lat2,lat3
 454      if(jlatband==1) then
 455        ! Southern extratropics
 456        latMask(1:lat1) = 1.0d0
 457        do latIndex = lat1, lat2
 458          !latMask(latIndex) = sqrt(dble((lat2-latIndex)*4)/dble(nj))
 459          latMask(latIndex) = sqrt(0.5d0*(1.0d0+cos(dble((latIndex-lat1)*4)*MPC_PI_R8/dble(nj))))
 460        end do
 461        latMask(lat2:nj) = 0.0d0
 462      else if(jlatband==2) then
 463        ! Tropics
 464        !latMask(1:lat1) = 0.0d0
 465        !do latIndex = lat1, lat2
 466        !  latMask(latIndex) = sqrt(dble((latIndex-lat1)*4)/dble(nj))
 467        !end do
 468        !do latIndex = lat2,lat3
 469        !  latMask(latIndex) = sqrt(dble((lat3-latIndex)*4)/dble(nj))
 470        !end do
 471        !latMask(lat3:nj) = 0.0d0
 472
 473        ! NOTE: use much broader band for tropics to avoid shortening horizontal correlations
 474        ! ok, since the masks do not have to sum to one for calculation, but they do
 475        ! when used in bmatrixhi_mod
 476        ! Tropics
 477        do latIndex = 1, lat1
 478          !latMask(latIndex) = sqrt(dble((latIndex-1)*4)/dble(nj))
 479          latMask(latIndex) = sqrt(0.5d0*(1.0d0+cos(dble((lat1-latIndex)*4)*MPC_PI_R8/dble(nj))))
 480        end do
 481        latMask(lat1:lat3) = 1.0d0
 482        do latIndex = lat3,nj
 483          !latMask(latIndex) = sqrt(dble((nj-latIndex)*4)/dble(nj))
 484          latMask(latIndex) = sqrt(0.5d0*(1.0d0+cos(dble((latIndex-lat3)*4)*MPC_PI_R8/dble(nj))))
 485        end do
 486      else if(jlatband==3) then
 487        ! Northern extratropics
 488        latMask(1:lat2) = 0.0d0
 489        do latIndex = lat2, lat3
 490          !latMask(latIndex) = sqrt(dble((latIndex-lat2)*4)/dble(nj))
 491          latMask(latIndex) = sqrt(0.5d0*(1.0d0+cos(dble((lat3-latIndex)*4)*MPC_PI_R8/dble(nj))))
 492        end do
 493        latMask(lat3:nj) = 1.0d0
 494      end if

 518    !
 519    !:Purpose: High-level routine to do a variety of diagnostic operations
 520    !          on the ensemble of error samples
 521    !
 522    implicit none
 523
 524    ! NOTE: The diagnostic computed here are in model variable space 
 525    !       (no variable transform!!!)
 526
 527    integer :: waveBandIndex
 528    integer :: nulnam, ierr, fclos, fnom, numStep
 529
 530    real(8), allocatable :: corns(:,:,:), rstddev(:,:), powerSpec(:,:)
 531
 532    integer, allocatable :: dateStampList(:)
 533    
 534    type(struct_ens), target  :: ensPerts, ensPertsFilt
 535
 536    type(struct_ens), pointer :: ensPerts_ptr
 537    
 538    type(struct_gsv) :: statevector_template
 539
 540    type(struct_gbi) :: gbi_zonalMean
 541    
 542    integer :: variableType
 543 
 544    logical :: ensContainsFullField
 545    logical :: makeBiPeriodic
 546    logical :: doSpectralFilter
 547    
 548    character(len=60) :: tool
 549
 550    NAMELIST /NAMTOOLBOX/tool, ensContainsFullField, doSpectralFilter

 553    write(*,*) 'csg_toolbox'
 554    write(*,*)
 555

 559    variableType = modelSpace     ! hardwired
 560    ensContainsFullField = .true. ! default value
 561    doSpectralFilter     = .true.
 562

 564    ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)

 567    if (mmpi_myid == 0) write(*,nml=NAMTOOLBOX)
 568    ierr = fclos(nulnam)
 569

 571    !- Read ensemble
 572    !

 579      call ens_allocate(ensPertsFilt, nEns, numStep, hco_ens, vco_ens, dateStampList)
 580      ensPerts_ptr => ensPertsFilt
 581    else
 582      ensPerts_ptr => ensPerts
 583    end if

 585    makeBiPeriodic = .false.
 586    call ens_readEnsemble(ensPerts, './ensemble', makeBiPeriodic, &
 587                          containsFullField_opt=ensContainsFullField)
 588
 589    !
 590    !- Compute and remove the ensemble mean; compute the stdDev
 591    !
 592    call ens_computeMean(ensPerts)
 593    call ens_removeMean (ensPerts)

 595    !
 596    !- Tool selection
 597    !
 598    select case(trim(tool))
 599    case ('HVCORREL_HI')
 600      write(*,*)
 601      write(*,*) 'Computing Homogeneous and Isotropic Correlation'
 602
 603      if (mmpi_nprocs > 1) then
 604        call utl_abort('csg_toolbox: this tool is not yet MPI capable') ! only due to horizCorrelFunction
 605      end if
 606       
 607      call spectralFilter2(ensPerts,          & ! IN
 608                           ensPerts_ptr,      & ! OUT
 609                           waveBandIndex_opt=1) ! IN
 610      call ens_computeStdDev(ensPerts_ptr)
 611      call ens_normalize(ensPerts_ptr)
 612      call ens_removeGlobalMean(ensPerts_ptr)
 613
 614      allocate(corns(nkgdimEns,nkgdimEns,0:ntrunc))
 615      allocate(rstddev(nkgdimEns,0:ntrunc))
 616

 618                             corns,        & ! OUT (vertical correlation in spectral space)
 619                             rstddev)        ! OUT ( sqrt(normalized power spectrum) )
 620
 621      call writeStats(corns,rstddev,waveBandIndex_opt=1) ! IN
 622      call calcHorizScale(rstddev,variableType,waveBandIndex_opt=1) ! IN
 623      call horizCorrelFunction(rstddev,variableType,waveBandIndex_opt=1) ! IN
 624
 625      deallocate(rstddev)
 626      deallocate(corns)
 627       
 628    case ('HVCORREL_LOCAL')
 629      write(*,*)
 630      write(*,*) 'Computing Local Correlation'
 631
 632      call ens_removeGlobalMean(ensPerts)
 633      call spectralFilter2(ensPerts,          & ! IN
 634                           ensPerts_ptr,      & ! OUT
 635                           waveBandIndex_opt=1) ! IN
 636      call ens_computeStdDev(ensPerts_ptr)
 637      call ens_normalize(ensPerts_ptr)
 638      call calcLocalCorrelations(ensPerts_ptr) ! IN
 639
 640    case ('VCORRMATRIX')
 641      write(*,*)
 642      write(*,*) 'Computing Local Correlation'
 643
 644      if (doSpectralFilter) then
 645        call ens_removeGlobalMean(ensPerts)
 646        call spectralFilter2(ensPerts,          & ! IN
 647                             ensPerts_ptr,      & ! OUT
 648                             waveBandIndex_opt=1) ! IN
 649      end if
 650      call ens_computeStdDev(ensPerts_ptr)
 651      call ens_normalize(ensPerts_ptr)
 652      call calcLocalVertCorrMatrix(ensPerts_ptr) ! IN
 653
 654    case ('LOCALIZATIONRADII')
 655      write(*,*)
 656      write(*,*) 'Estimating the optimal covariance localization radii'
 657
 658      call ens_removeGlobalMean(ensPerts)
 659
 660      do waveBandIndex = 1, nWaveBand
 661        
 662        call spectralFilter2(ensPerts,                      & ! IN
 663                             ensPerts_ptr,                  & ! OUT
 664                             waveBandIndex_opt=waveBandIndex) ! IN
 665
 666        call ens_computeStdDev(ensPerts_ptr)
 667       
 668        if (waveBandIndex == 1) then
 669          call ens_copyEnsStdDev(ensPerts_ptr, statevector_template) ! IN
 670          call bmd_setup(statevector_template, hco_ens, nEns, pressureProfile_M, & ! IN
 671                         pressureProfile_T, nWaveBand)                             ! IN
 672        end if
 673         
 674        call bmd_localizationRadii(ensPerts_ptr, waveBandIndex_opt=waveBandIndex) ! IN
 675
 676      end do
 677      
 678    case ('STDDEV')
 679      write(*,*)
 680      write(*,*) 'Computing standard deviations using PSI/CHI for winds'
 681
 682      ! u,v to psi,chi 
 683      call gvt_setup(hco_ens, hco_ens, vco_ens)
 684      call gvt_transform(ensPerts_ptr, 'UVtoPsiChi')
 685      if (.not. ens_varExist(ensPerts_ptr,'PP') .and. &
 686          .not. ens_varExist(ensPerts_ptr,'CC') ) then
 687        call ens_modifyVarName(ensPerts_ptr, 'UU', 'PP')
 688        call ens_modifyVarName(ensPerts_ptr, 'VV', 'CC')
 689      end if
 690
 691      ! Compute the grid point std dev
 692      call ens_computeStdDev(ensPerts_ptr)
 693      call ens_copyEnsStdDev(ensPerts_ptr, statevector_template) ! IN
 694      call gio_writeToFile(statevector_template, './stddev.fst', 'STDDEV_GRIDP', &
 695                           typvar_opt = 'E', numBits_opt = 32)
 696
 697      ! Compute the zonal std dev
 698      call gbi_setup(gbi_zonalMean, 'YrowBand', statevector_template, hco_ens)
 699      call gbi_stdDev(gbi_zonalMean, ensPerts_ptr, & ! IN
 700                      statevector_template)          ! OUT
 701      call gio_writeToFile(statevector_template, './stddev.fst', 'STDDEV_ZONAL', &
 702                           typvar_opt = 'E', numBits_opt = 32)
 703
 704    case ('POWERSPEC')
 705      write(*,*)
 706      write(*,*) 'Computing power spectra'
 707
 708      call calcPowerSpec(ensPerts_ptr, & ! IN
 709                         powerSpec)      ! OUT
 710      call writePowerSpec(powerSpec, modelSpace) ! IN
 711      
 712    case default
 713      write(*,*)
 714      write(*,*) 'Unknown TOOL in csg_toolbox : ', trim(tool)
 715      call utl_abort('csg_toolbox')
 716    end select
 717
 718    !
 719    !- Write the estimated pressure profiles
 720    !
 721    if (vco_ens%vgridPresent) then
 722      call writePressureProfiles
 723    end if
 724
 725    !
 726    !- Ending
 727    !
 728    call ens_deallocate(ensPerts)
 729
 730    write(*,*)
 731    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 732    write(*,*) 'csl_toolbox: Done!'
 733
 734  end subroutine csg_toolbox

 853  !--------------------------------------------------------------------------
 854  subroutine removeBalancedT_Ps(ensPerturbations,ensBalPerturbations,PtoT)
 855    !
 856    !:Purpose: Subtract the balanced components of temperature and surface
 857    !          pressure from the full variables
 858    !
 859    implicit none
 860    real(4),pointer :: ensPerturbations(:,:,:,:)
 861    real(4),pointer :: ensBalPerturbations(:,:,:,:)
 862    real(8) :: PtoT(:,:,:)
 863
 864    real(4),pointer :: tt_ptr(:,:,:), ps_ptr(:,:,:), ttb_ptr(:,:,:), psb_ptr(:,:,:)
 865    real(8) :: spectralState(nla_mpilocal,2,nLevEns_M), spBalancedP(nla_mpilocal,2,nlevEns_M)
 866    real(8) :: balancedP(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nlevEns_M)
 867    real(8) :: psi(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nLevEns_M)
 868    integer :: ensIndex, latIndex, lonIndex, jk1, jk2
 869
 870    do ensIndex=1,nens
 871
 872      write(*,*) 'removing balanced T and Ps for member ', ensIndex
 873
 874      psi(:,:,:) = ensPerturbations(:,:,1:nlevEns_M,ensIndex)
 875      call gst_setID(gstID_nLevEns_M)
 876      call gst_reespe(spectralState,psi)
 877      call calcBalancedP(spectralState,spBalancedP)
 878      call gst_speree(spBalancedP,balancedP)
 879
 880      tt_ptr(myLonBeg:,myLatBeg:,1:)  => ensPerturbations(:,:,(1+2*nLevEns_M):(2*nLevEns_M+1*nLevEns_T),ensIndex)
 881      ps_ptr(myLonBeg:,myLatBeg:,1:)  => ensPerturbations(:,:,(1+2*nLevEns_M+2*nLevEns_T):(1+2*nLevEns_M+2*nLevEns_T),ensIndex)
 882      ttb_ptr(myLonBeg:,myLatBeg:,1:) => ensBalPerturbations(:,:,1:nLevEns_T,ensIndex)
 883      psb_ptr(myLonBeg:,myLatBeg:,1:) => ensBalPerturbations(:,:,(1+nLevEns_T):(1+nLevEns_T),ensIndex)
 884
 885      ttb_ptr(:,:,:)=0.0d0
 886      psb_ptr(:,:,:)=0.0d0
 887
 888      !$OMP PARALLEL DO PRIVATE (latIndex,lonIndex,jk1,jk2)
 889      do latIndex = myLatBeg, myLatEnd
 890        do lonIndex = myLonBeg, myLonEnd
 891          do jk1 = 1, nLevEns_T
 892            do jk2 = 1, nlevptot
 893              ttb_ptr(lonIndex,latIndex,jk1) = ttb_ptr(lonIndex,latIndex,jk1) + PtoT(jk1,jk2,latIndex)*balancedP(lonIndex,latIndex,jk2)
 894            end do
 895            tt_ptr(lonIndex,latIndex,jk1) = tt_ptr(lonIndex,latIndex,jk1) - ttb_ptr(lonIndex,latIndex,jk1)

 899          end do
 900          ps_ptr(lonIndex,latIndex,1) = ps_ptr(lonIndex,latIndex,1) - psb_ptr(lonIndex,latIndex,1)
 901        end do
 902      end do
 903      !$OMP END PARALLEL DO
 904
 905    end do
 906
 907    write(*,*) 'finished removing balanced T and Ps...'

 911  !--------------------------------------------------------------------------
 912  ! CALCCORRELATIONS
 913  !--------------------------------------------------------------------------
 914  subroutine calcCorrelations(ensPerturbations,corns,rstddev,latMask_opt)
 915    !
 916    !:Purpose: Calculate the homogeneous and isotropic correlations in spectral space
 917    !
 918    implicit none
 919    real(4),pointer :: ensPerturbations(:,:,:,:)
 920    real(8) :: corns(nkgdimEns,nkgdimEns,0:ntrunc),corns_mpiglobal(nkgdimEns,nkgdimEns,0:ntrunc)
 921    real(8) :: rstddev(nkgdimEns,0:ntrunc)
 922    real(8),  optional :: latMask_opt(:)
 923
 924    real(8) :: spectralState(nla_mpilocal,2,nkgdimEns)
 925    real(8) :: gridState(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nkgdimEns)
 926    real(8) :: dfact,dfact2,dsummed
 927    integer :: ensIndex,ila_mpilocal,ila_mpiglobal,jn,jm,jk1,jk2,latIndex,nsize,ierr
 928
 929    call utl_tmg_start(120,'--CalcStats_Corr')
 930
 931    corns(:,:,:) = 0.0d0
 932    do ensIndex = 1, nens
 933
 934      write(*,*) 'calcCorrelations: processing member ',ensIndex
 935
 936      gridState(:,:,:) = ensPerturbations(:,:,:,ensIndex)
 937      if(present(latMask_opt)) then
 938        do latIndex = myLatBeg, myLatEnd
 939          gridState(:,latIndex,:) = latMask_opt(latIndex)*gridState(:,latIndex,:)
 940        end do
 941      end if
 942      call gst_setID(gstID_nkgdimEns)
 943      call gst_reespe(spectralState,gridState)
 944
 945      !$OMP PARALLEL DO PRIVATE (jn,jm,dfact,ila_mpilocal,ila_mpiglobal,jk1,jk2)
 946      do jn = mynBeg, mynEnd, mynSkip
 947        do jm = mymBeg, mymEnd, mymSkip
 948          if(jm.le.jn) then
 949            dfact = 2.0d0
 950            if (jm.eq.0) dfact = 1.0d0
 951            ila_mpiglobal = gst_getNind(jm,gstID_nkgdimEns) + jn - jm
 952            ila_mpilocal = ilaList_mpilocal(ila_mpiglobal)
 953            do jk1 = 1, nkgdimEns
 954              do jk2 = 1, nkgdimEns
 955                corns(jk1,jk2,jn) = corns(jk1,jk2,jn) +     &
 956                     dfact*( spectralState(ila_mpilocal,1,jk1)*spectralState(ila_mpilocal,1,jk2) +   &
 957                     spectralState(ila_mpilocal,2,jk1)*spectralState(ila_mpilocal,2,jk2)  )
 958              end do
 959            end do
 960          end if
 961        end do
 962      end do
 963      !$OMP END PARALLEL DO
 964
 965    end do
 966
 967    ! communicate between all tasks
 968    nsize = nkgdimEns*nkgdimEns*(1+ntrunc)
 969    call rpn_comm_allreduce(corns,corns_mpiglobal,nsize,"mpi_double_precision","mpi_sum","GRID",ierr)
 970    corns(:,:,:) = corns_mpiglobal(:,:,:)
 971    
 972    !$OMP PARALLEL DO PRIVATE (jn,jk1)
 973    do jn = 0, ntrunc
 974      do jk1 = 1, nkgdimEns
 975        if(abs(corns(jk1,jk1,jn)).gt.0.0d0) then
 976          rstddev(jk1,jn) = dsqrt(abs(corns(jk1,jk1,jn)))
 977        else
 978          rstddev(jk1,jn) = 0.0d0
 979        end if
 980      end do
 981    end do
 982    !$OMP END PARALLEL DO
 983
 984    !$OMP PARALLEL DO PRIVATE (jn,jk1,jk2)
 985    do jn = 0, ntrunc
 986      do jk1 = 1, nkgdimEns
 987        do jk2 = 1, nkgdimEns
 988          if(rstddev(jk1,jn).ne.0..and.rstddev(jk2,jn).ne.0.) then
 989            corns(jk1,jk2,jn) =  corns(jk1,jk2,jn)/(rstddev(jk1,jn)*rstddev(jk2,jn))
 990          else
 991            corns(jk1,jk2,jn) = 0.0d0
 992          end if
 993        end do
 994      end do
 995    end do
 996    !$OMP END PARALLEL DO
 997
 998    dfact2 = 1.0d0/sqrt(dble(nens-1))
 999    do jn = 0, ntrunc
1000      dfact = 1.0d0/sqrt(2.0d0*dble(jn) + 1.0d0)
1001      do jk1 = 1, nkgdimEns
1002        rstddev(jk1,jn) = rstddev(jk1,jn)*dfact2*dfact
1003      end do
1004    end do
1005
1006    ! Normalize to ensure correlations in horizontal and Multiply by sqrt(0.5) to make valid for m.ne.0
1007    !$OMP PARALLEL DO PRIVATE (jk1,jn,dsummed)
1008    do jk1 = 1, nkgdimEns
1009      dsummed=0.0d0
1010      do jn = 0, ntrunc
1011        dsummed=dsummed + (rstddev(jk1,jn)**2)*((2.0d0*dble(jn))+1.0d0)/2.0d0
1012      end do
1013      do jn = 0, ntrunc
1014        if(dsummed.gt.0.0d0) rstddev(jk1,jn)=rstddev(jk1,jn)*sqrt(0.5d0/dsummed)
1015      end do

1027  subroutine calcCorrelations2(ensPerts,corns,rstddev,latMask_opt)
1028    !
1029    !:Purpose: Calculate the homogeneous and isotropic correlations in spectral space
1030    !
1031    implicit none
1032
1033    type(struct_ens) :: ensPerts
1034    real(8), intent(out)          :: corns(nkgdimEns,nkgdimEns,0:ntrunc)
1035    real(8), intent(out)          :: rstddev(nkgdimEns,0:ntrunc)
1036    real(8), optional, intent(in) :: latMask_opt(:)
1037
1038    real(4), pointer :: ptr4d_r4(:,:,:,:)
1039    
1040    real(8) :: corns_mpiglobal(nkgdimEns,nkgdimEns,0:ntrunc)
1041    real(8) :: spectralState(nla_mpilocal,2,nkgdimEns)
1042    real(8) :: gridState(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nkgdimEns)
1043    real(8) :: dfact, dfact2, dsummed
1044    
1045    integer :: ensIndex, ila_mpilocal, ila_mpiglobal, jn, jm, jk1, jk2
1046    integer :: levIndex, latIndex, nsize, ierr
1047
1048    call utl_tmg_start(120,'--CalcStats_Corr')
1049
1050    corns(:,:,:) = 0.0d0
1051    do ensIndex = 1, nens
1052
1053      write(*,*) 'calcCorrelations: processing member ',ensIndex
1054
1055      !- 2.1 Extract fields from ensPerturbations
1056      do levIndex = 1, ens_getNumK(ensPerts)
1057        ptr4d_r4 => ens_getOneLev_r4(ensPerts,levIndex)
1058        gridState(:,:,levIndex) = real(ptr4d_r4(ensIndex,1,:,:),8)
1059      end do
1060      if ( present(latMask_opt) ) then
1061        do latIndex = myLatBeg, myLatEnd
1062          gridState(:,latIndex,:) = latMask_opt(latIndex)*gridState(:,latIndex,:)
1063        end do
1064      end if
1065      call gst_setID(gstID_nkgdimEns)
1066      call gst_reespe(spectralState,gridState)
1067
1068      !$OMP PARALLEL DO PRIVATE (jn,jm,dfact,ila_mpilocal,ila_mpiglobal,jk1,jk2)
1069      do jn = mynBeg, mynEnd, mynSkip
1070        do jm = mymBeg, mymEnd, mymSkip
1071          if(jm.le.jn) then
1072            dfact = 2.0d0
1073            if (jm.eq.0) dfact = 1.0d0
1074            ila_mpiglobal = gst_getNind(jm,gstID_nkgdimEns) + jn - jm
1075            ila_mpilocal = ilaList_mpilocal(ila_mpiglobal)
1076            do jk1 = 1, nkgdimEns
1077              do jk2 = 1, nkgdimEns
1078                corns(jk1,jk2,jn) = corns(jk1,jk2,jn) +     &
1079                     dfact*( spectralState(ila_mpilocal,1,jk1)*spectralState(ila_mpilocal,1,jk2) +   &
1080                     spectralState(ila_mpilocal,2,jk1)*spectralState(ila_mpilocal,2,jk2)  )
1081              end do
1082            end do
1083          end if
1084        end do
1085      end do
1086      !$OMP END PARALLEL DO
1087
1088    end do

1091    nsize = nkgdimEns*nkgdimEns*(1+ntrunc)

1093    corns(:,:,:) = corns_mpiglobal(:,:,:)
1094    
1095    !$OMP PARALLEL DO PRIVATE (jn,jk1)
1096    do jn = 0, ntrunc
1097      do jk1 = 1, nkgdimEns
1098        if(abs(corns(jk1,jk1,jn)).gt.0.0d0) then
1099          rstddev(jk1,jn) = dsqrt(abs(corns(jk1,jk1,jn)))
1100        else
1101          rstddev(jk1,jn) = 0.0d0
1102        end if
1103      end do
1104    end do
1105    !$OMP END PARALLEL DO
1106
1107    !$OMP PARALLEL DO PRIVATE (jn,jk1,jk2)
1108    do jn = 0, ntrunc
1109      do jk1 = 1, nkgdimEns
1110        do jk2 = 1, nkgdimEns
1111          if(rstddev(jk1,jn).ne.0..and.rstddev(jk2,jn).ne.0.) then
1112            corns(jk1,jk2,jn) =  corns(jk1,jk2,jn)/(rstddev(jk1,jn)*rstddev(jk2,jn))
1113          else
1114            corns(jk1,jk2,jn) = 0.0d0
1115          end if
1116        end do
1117      end do
1118    end do
1119    !$OMP END PARALLEL DO
1120
1121    dfact2 = 1.0d0/sqrt(dble(nens-1))
1122    do jn = 0, ntrunc
1123      dfact = 1.0d0/sqrt(2.0d0*dble(jn) + 1.0d0)
1124      do jk1 = 1, nkgdimEns
1125        rstddev(jk1,jn) = rstddev(jk1,jn)*dfact2*dfact
1126      end do
1127    end do
1128
1129    ! Normalize to ensure correlations in horizontal and Multiply by sqrt(0.5) to make valid for m.ne.0
1130    !$OMP PARALLEL DO PRIVATE (jk1,jn,dsummed)
1131    do jk1 = 1, nkgdimEns
1132      dsummed=0.0d0
1133      do jn = 0, ntrunc
1134        dsummed=dsummed + (rstddev(jk1,jn)**2)*((2.0d0*dble(jn))+1.0d0)/2.0d0
1135      end do
1136      do jn = 0, ntrunc
1137        if(dsummed.gt.0.0d0) rstddev(jk1,jn)=rstddev(jk1,jn)*sqrt(0.5d0/dsummed)
1138      end do
1139    end do
1140    !$OMP END PARALLEL DO
1141
1142    call utl_tmg_stop(120)
1143    write(*,*) 'finished computing correlations...'
1144
1145  end subroutine calcCorrelations2

1150  subroutine calcPowerSpec(ensPerts,powerSpec)
1151    !
1152    !:Purpose: Calculate the horizontal power spectrum of the ensemble
1153    !          of error samples
1154    !
1155    implicit none
1156    type(struct_ens)           :: ensPerts
1157    real(8), allocatable       :: powerSpec(:,:)
1158
1159    real(8), allocatable :: ensPertSP(:,:,:)
1160    real(8), allocatable :: ensPertGD(:,:,:)
1161    real(4), pointer     :: ptr4d_r4(:,:,:,:)
1162    real(8) :: dfact, dfact2
1163
1164    integer :: gstPowerSpecID
1165    integer :: memberIndex, levIndex, latIndex, lonIndex
1166    integer :: jn, jm, ila_mpilocal, ila_mpiglobal
1167
1168    allocate(powerSpec          (ens_getNumK(ensPerts),0:ntrunc))
1169
1170    !
1171    !- Spectral decomposition and spectral coefficient summation
1172    !
1173    gstPowerSpecID = gst_setup(ni,nj,nTrunc,nEnsOverDimension)
1174    
1175    allocate(ensPertSP(nla_mpilocal,2,nEnsOverDimension))
1176    allocate(ensPertGD(nEnsOverDimension,myLonBeg:myLonEnd,myLatBeg:myLatEnd))
1177
1178    powerSpec(:,:)=0.0d0
1179    
1180    do levIndex = 1, ens_getNumK(ensPerts) ! Loop on variables and vertical levels
1181      
1182      ptr4d_r4 => ens_getOneLev_r4(ensPerts,levIndex)
1183
1184      do latIndex = myLatBeg, myLatEnd
1185        ensPertGD(:,:,latIndex) = 0.0d0
1186      end do
1187
1188      !$OMP PARALLEL DO PRIVATE (memberIndex,latIndex,lonIndex)
1189      do latIndex = myLatBeg, myLatEnd
1190        do lonIndex = myLonBeg, myLonEnd
1191          do memberIndex = 1, nEns
1192            ensPertGD(memberIndex,lonIndex,latIndex) = dble(ptr4d_r4(memberIndex,1,lonIndex,latIndex))
1193          end do
1194        end do
1195      end do
1196      !$OMP END PARALLEL DO
1197      
1198      !- GridPoint space -> Spectral Space
1199      call gst_setID(gstPowerSpecID) ! IN
1200      call gst_reespe_kij(ensPertSP, & ! OUT
1201                          ensPertGD)   ! IN
1202
1203      !$OMP PARALLEL DO PRIVATE (memberIndex,jn,jm,dfact,ila_mpilocal,ila_mpiglobal)
1204      do jn = mynBeg, mynEnd, mynSkip
1205        do jm = mymBeg, mymEnd, mymSkip
1206          if (jm .le. jn) then
1207            dfact = 2.0d0
1208            if (jm.eq.0) dfact = 1.0d0
1209            ila_mpiglobal = gst_getNind(jm,gstPowerSpecID) + jn - jm
1210            ila_mpilocal = ilaList_mpilocal(ila_mpiglobal)
1211            do memberIndex = 1, nEns
1212              powerSpec(levIndex,jn) = powerSpec(levIndex,jn) +     &
1213                   dfact*( ensPertSP(ila_mpilocal,1,memberIndex)**2 + ensPertSP(ila_mpilocal,2,memberIndex)**2 )
1214            end do
1215          end if
1216        end do
1217      end do
1218      !$OMP END PARALLEL DO
1219
1220    end do
1221
1222    !
1223    !- Communicate between all tasks
1224    !
1225    call mmpi_allreduce_sumR8_2d(powerSpec, "GRID")
1226
1227    !
1228    !- Apply the appropriate scaling
1229    !
1230    dfact2 = 1.0d0/sqrt(dble(nens-1))
1231    do jn = 0, ntrunc
1232      dfact = 1.0d0/sqrt(2.0d0*dble(jn) + 1.0d0)
1233      do levIndex = 1, ens_getNumK(ensPerts)
1234        powerSpec(levIndex,jn) = powerSpec(levIndex,jn)*dfact2*dfact
1235      end do
1236    end do
1237
1238    write(*,*) 'finished computing power spectrum...'
1239
1240  end subroutine calcPowerSpec

1843  !--------------------------------------------------------------------------
1844  subroutine calcTheta(ensPerturbations,theta)
1845    !
1846    !:Purpose: Calculate the Theta turning angle according to Ekman balance
1847    !
1848    implicit none
1849    real(4),pointer :: ensPerturbations(:,:,:,:)
1850    real(8) :: theta(:,:)
1851    real(8) :: zchipsi(nLevEns_M,nj), zpsipsi(nLevEns_M,nj)
1852    real(8) :: zchipsi_mpiglobal(nLevEns_M,nj), zpsipsi_mpiglobal(nLevEns_M,nj)
1853    real(4), pointer :: psi_ptr(:,:,:), chi_ptr(:,:,:)
1854    integer :: latIndex,lonIndex,levIndex,ensIndex, ierr, nsize
1855
1856    theta(:,:) = 0.0d0
1857    zchipsi(:,:) = 0.0d0
1858    zpsipsi(:,:) = 0.0d0
1859    zchipsi_mpiglobal(:,:) = 0.0d0
1860    zpsipsi_mpiglobal(:,:) = 0.0d0
1861
1862    do ensIndex = 1,nens
1863      psi_ptr(myLonBeg:,myLatBeg:,1:) => ensPerturbations(:,:,1:nlevEns_M,ensIndex)
1864      chi_ptr(myLonBeg:,myLatBeg:,1:) => ensPerturbations(:,:,(nlevEns_M+1):(2*nlevEns_M),ensIndex)
1865
1866      ! update zchipsi and zpsipsi covariances
1867      do latIndex = myLatBeg, myLatEnd
1868        do lonIndex = myLonBeg, myLonEnd
1869          do levIndex = 1, nLevEns_M
1870            zpsipsi(levIndex,latIndex) = zpsipsi(levIndex,latIndex) +  &
1871                 psi_ptr(lonIndex,latIndex,levIndex) * psi_ptr(lonIndex,latIndex,levIndex)
1872            zchipsi(levIndex,latIndex) = zchipsi(levIndex,latIndex) +  &
1873                 chi_ptr(lonIndex,latIndex,levIndex) * psi_ptr(lonIndex,latIndex,levIndex)
1874          end do
1875        end do
1876      end do
1877    end do
1878
1879    nsize = nLevEns_M*nj
1880    call rpn_comm_allreduce(zchipsi,zchipsi_mpiglobal,nsize,"mpi_double_precision","mpi_sum","GRID",ierr)
1881    call rpn_comm_allreduce(zpsipsi,zpsipsi_mpiglobal,nsize,"mpi_double_precision","mpi_sum","GRID",ierr)
1882
1883    !  calculate THETA
1884    do latIndex = 1, nj
1885      do levIndex = 1, nLevEns_M
1886        theta(levIndex,latIndex) = atan(-zchipsi_mpiglobal(levIndex,latIndex) /  &
1887                                         zpsipsi_mpiglobal(levIndex,latIndex))
1888      end do
1889    end do
1890
1891    write(*,*) 'finished computing theta...'
1892

1896  ! CALCPTOT
1897  !--------------------------------------------------------------------------
1898  subroutine calcPtoT(ensPerturbations,PtoT)
1899    !
1900    !:Purpose: Calculate the "P" to Temperature transform matrix using
1901    !          a regression analysis of the "P" and temperature samples
1902    !
1903    implicit none
1904    real(4), pointer :: ensPerturbations(:,:,:,:)
1905    real(8)  :: PtoT(:,:,:)
1906
1907    real(8) :: spectralState(nla_mpilocal,2,nLevEns_M), spBalancedP(nla_mpilocal,2,nlevEns_M)
1908    real(8) :: balancedP(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nlevEns_M)
1909    real(8) :: psi(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nLevEns_M)
1910    real(4), pointer :: tt_ptr(:,:,:),ps_ptr(:,:)
1911    INTEGER :: ensIndex, JK1, JK2, nsize
1912    INTEGER :: IERR, JK, latIndex, lonIndex, JB, JPNLATBND
1913    PARAMETER (JPNLATBND = 3)
1914    REAL(8) :: ZFACT,zlat(nj)
1915    REAL(8) :: ZFACTTOT
1916    REAL(8) :: ZM1(NLEVENS_T+1,NLEVENS_M,JPNLATBND), ZM2(NLEVPTOT,NLEVPTOT,JPNLATBND)
1917    REAL(8) :: ZM1_mpiglobal(NLEVENS_T+1,NLEVENS_M,JPNLATBND), ZM2_mpiglobal(NLEVPTOT,NLEVPTOT,JPNLATBND)
1918    REAL(8) :: ZPTOTBND(NLEVENS_T+1,NLEVENS_M)
1919    REAL(8) :: ZM2INV(NLEVPTOT,NLEVPTOT,JPNLATBND)
1920    REAL(8) :: DLA2, DL1SA2
1921    REAL(8) :: DLLATMIN(JPNLATBND), DLLATMAX(JPNLATBND)
1922    real(8) :: zeigwrk(4*nlevPtoT),zeigen(nlevPtoT,nlevPtoT),zeigenv(nlevPtoT)
1923    real(8) :: zeigenvi(nlevPtoT)
1924    integer :: iwork,info
1925
1926    DATA DLLATMIN / -60.0D0, -30.0D0, 30.0D0 /
1927    DATA DLLATMAX / -30.0D0,  30.0D0, 60.0D0 /
1928
1929    DLA2 = ec_ra * ec_ra
1930    DL1SA2 = 1.D0/DLA2
1931
1932    ! 1. Initialize P_to_T, ZM1, ZM2
1933
1934    ZFACTTOT = 0.0D0
1935    DO latIndex = 1, nj
1936      ZFACTTOT = ZFACTTOT + cos(GST_GETRLATI(latIndex))
1937    END DO
1938    ZFACTTOT = NJ/ZFACTTOT
1939
1940    PtoT(:,:,:) = 0.0d0
1941    ZM1(:,:,:) = 0.0d0
1942    ZM2(:,:,:) = 0.0d0
1943    ZPTOTBND(:,:) = 0.0d0
1944    ZM1_mpiglobal(:,:,:) = 0.0d0
1945    ZM2_mpiglobal(:,:,:) = 0.0d0
1946
1947    do ensIndex = 1, nens
1948
1949      write(*,*) 'calcPtoT: processing member ',ensIndex
1950
1951      psi(:,:,:) = ensPerturbations(:,:,1:nlevEns_M,ensIndex)
1952      call gst_setID(gstID_nLevEns_M)
1953      call gst_reespe(spectralState,psi)
1954      call calcBalancedP(spectralState,spBalancedP)
1955      call gst_speree(spBalancedP,balancedP)
1956
1957      tt_ptr(myLonBeg:,myLatBeg:,1:) => ensPerturbations(:,:,(2*nLevEns_M+1):(2*nLevEns_M+nLevEns_T),ensIndex)
1958      ps_ptr(myLonBeg:,myLatBeg:)    => ensPerturbations(:,:,2*nLevEns_M+2*nLevEns_T+1,ensIndex)
1959
1960      do latIndex = myLatBeg, myLatEnd
1961        zlat(latIndex)=GST_GETRLATI(latIndex)
1962      end do
1963
1964      !$OMP PARALLEL DO PRIVATE (JK1,JB,latIndex,ZFACT,lonIndex,JK2)
1965      DO JK1 = 1, (nLevEns_T+1)
1966        DO JB=1,JPNLATBND
1967          DO latIndex = myLatBeg, myLatEnd
1968            if ((ZLAT(latIndex) .gt. 2.D0*MPC_PI_R8*DLLATMIN(JB)/360.D0) .and.  &
1969                (ZLAT(latIndex) .le. 2.D0*MPC_PI_R8*DLLATMAX(JB)/360.D0)) then
1970              ZFACT = cos(ZLAT(latIndex))*ZFACTTOT
1971              DO lonIndex = myLonBeg, myLonEnd
1972
1973                ! update ZM1 = sum_over_t_x_y[vec(T lnPs) vec(P_b)^T]
1974                DO JK2 = 1, nLevEns_M
1975                  IF(JK1.LE.nLevEns_T) THEN
1976                    zm1(jk1,jk2,jb) = zm1(jk1,jk2,jb) + zfact * tt_ptr(lonIndex,latIndex,jk1) * balancedP(lonIndex,latIndex,jk2)
1977                  ELSE
1978                    zm1(jk1,jk2,jb) = zm1(jk1,jk2,jb) + zfact * ps_ptr(lonIndex,latIndex) * balancedP(lonIndex,latIndex,jk2)
1979                  END IF
1980                END DO
1981
1982                ! update ZM2 = sum_over_t_x_y[vec(P_b) vec(P_b)^T]
1983                IF(JK1.LE.NLEVPTOT) THEN
1984                  DO JK2 = 1, NLEVPTOT
1985                    zm2(jk1,jk2,jb) = zm2(jk1,jk2,jb) + zfact * balancedP(lonIndex,latIndex,jk1) * balancedP(lonIndex,latIndex,jk2)
1986                  END DO
1987                END IF
1988
1989              END DO
1990            end if
1991          END DO
1992        END DO ! Loop on JPNLATBND
1993      END DO  ! Loop on JK1
1994      !$OMP END PARALLEL DO
1995
1996    END DO
1997
1998    ! communicate matrices to have global result on all tasks
1999    nsize = (NLEVENS_T+1)*NLEVENS_M*JPNLATBND
2000    call rpn_comm_allreduce(zm1,zm1_mpiglobal,nsize,"mpi_double_precision","mpi_sum","GRID",ierr)

2195  !--------------------------------------------------------------------------
2196  subroutine calcBalancedP(sppsi,spgz)
2197    !
2198    !:Purpose: Calculate the balanced "P" variable using geostrophy
2199    !
2200    implicit none
2201
2202    real(8) :: sppsi(:,:,:),spgz(:,:,:)
2203    real(8) :: spvor_mpiglobal(nla,2,nlevEns_M)
2204    real(8) :: spvor_mpiglobal2(nla,2,nlevEns_M)
2205    real(8) :: spgz_mpiglobal(nla,2,nlevEns_M)
2206    integer :: ia, ib, ji, jm, levIndex, jla_mpilocal, ila_mpiglobal
2207    integer :: ierr, nsize
2208    real(8) :: zn, zm, zenm, zenmp1, zcon, dl1sa2
2209    
2210    ! convert PSI to vorticity 
2211    dl1sa2   = 1.0d0/(ec_ra*ec_ra)
2212    spvor_mpiglobal(:,:,:) = 0.0d0
2213    spvor_mpiglobal2(:,:,:) = 0.0d0
2214    do levIndex = 1, nlevEns_M
2215      do jla_mpilocal = 1, nla_mpilocal
2216        ila_mpiglobal = ilaList_mpiglobal(jla_mpilocal)
2217        spvor_mpiglobal(ila_mpiglobal,1,levIndex) = sppsi(jla_mpilocal,1,levIndex)*dl1sa2*gst_getRnnp1(ila_mpiglobal)
2218        spvor_mpiglobal(ila_mpiglobal,2,levIndex) = sppsi(jla_mpilocal,2,levIndex)*dl1sa2*gst_getRnnp1(ila_mpiglobal)
2219      end do
2220      ! ensure input field is zero for spectral component (0,0)
2221      spvor_mpiglobal(1,1,levIndex) = 0.0D0
2222      spvor_mpiglobal(1,2,levIndex) = 0.0D0
2223    end do
2224
2225    nsize = nla*2*nlevEns_M
2226    call rpn_comm_allreduce(spvor_mpiglobal,spvor_mpiglobal2,nsize,"mpi_double_precision","mpi_sum","GRID",ierr)
2227    
2228    ! initialize output field to zero
2229    spgz_mpiglobal(:,:,:)=0.0d0
2230
2231    ! loop over levels and zonal wavenumbers
2232    ! n.b.: at the tip of the triangle, no contributions
2233    
2234    zcon = -2.D0*ec_ROmega*ec_ra**2
2235    do levIndex = 1, nlevEns_M
2236
2237      ! the base address ia will point to the spherical harmonic
2238      ! coefficient (m,m), in the input field
2239      ia = 1
2240      do jm = 0, ntrunc-1
2241        ib = ia + ntrunc - jm
2242        zm = dble(jm)
2243
2244        ! at the base, contributions from n+1 coeff only
2245        zn = zm
2246        zenmp1 = sqrt ( ((zn+1)**2-zm**2)/(4.D0*(zn+1)**2-1.D0) )
2247        spgz_mpiglobal(ia,1,levIndex)=zcon*spvor_mpiglobal2(ia+1,1,levIndex)*zenmp1/((zn+1.0D0)**2)
2248        spgz_mpiglobal(ia,2,levIndex)=zcon*spvor_mpiglobal2(ia+1,2,levIndex)*zenmp1/((zn+1.0D0)**2)
2249
2250        zn = zn+1
2251        do ji = ia+1, ib-1
2252          zenm = sqrt ( (zn**2-zm**2)/(4.D0*zn**2-1.D0) )
2253          zenmp1 = sqrt ( ((zn+1)**2-zm**2)/(4.D0*(zn+1)**2-1.D0) )
2254          spgz_mpiglobal(ji,1,levIndex)=spvor_mpiglobal2(ji-1,1,levIndex)*zenm/(zn**2)
2255          spgz_mpiglobal(ji,2,levIndex)=spvor_mpiglobal2(ji-1,2,levIndex)*zenm/(zn**2)
2256          spgz_mpiglobal(ji,1,levIndex)=zcon*(spgz_mpiglobal(ji,1,levIndex)+spvor_mpiglobal2(ji+1,1,levIndex)*zenmp1/((zn+1.0D0)**2))
2257          spgz_mpiglobal(ji,2,levIndex)=zcon*(spgz_mpiglobal(ji,2,levIndex)+spvor_mpiglobal2(ji+1,2,levIndex)*zenmp1/((zn+1.0D0)**2))
2258          zn = zn + 1.0D0
2259        end do
2260
2261        ! at the top, contributions from n-1 coeff only
2262        zenm = sqrt ( (zn**2-zm**2)/(4.D0*zn**2-1.D0) )
2263        spgz_mpiglobal(ib,1,levIndex) = zcon*spvor_mpiglobal2(ib-1,1,levIndex)*zenm/(zn**2)
2264        spgz_mpiglobal(ib,2,levIndex) = zcon*spvor_mpiglobal2(ib-1,2,levIndex)*zenm/(zn**2)
2265        ia = ib + 1
2266      end do
2267    end do
2268
2269    ! ensure correct value for mass spectral-coefficient for m=n=0
2270    do levIndex = 1, nlevens_M
2271      spgz_mpiglobal(1,1,levIndex) = 0.0D0
2272      spgz_mpiglobal(1,2,levIndex) = 0.0D0
2273    end do
2274
2275    ! copy to mpilocal output array
2276    do levIndex = 1, nlevEns_M
2277      do jla_mpilocal = 1, nla_mpilocal
2278        ila_mpiglobal = ilaList_mpiglobal(jla_mpilocal)
2279        spgz(jla_mpilocal,1,levIndex) = spgz_mpiglobal(ila_mpiglobal,1,levIndex)
2280        spgz(jla_mpilocal,2,levIndex) = spgz_mpiglobal(ila_mpiglobal,2,levIndex)
2281      end do
2282    end do
2283
2284  end subroutine calcBalancedP
