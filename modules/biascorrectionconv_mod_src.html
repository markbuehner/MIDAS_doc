 114  subroutine bcc_readConfig()
 115    !
 116    !:Purpose: Read NAMBIASCONV namelist section and NAMSONDETYPES section if uaBiasActive=true
 117    !
 118    implicit none
 119
 120    !Locals:
 121    integer  :: ierr, nulnam, sondeIndex
 122    
 123    if ( initialized ) then
 124      write(*,*) "bcc_readConfig has already been called. Returning..."
 125      return
 126    end if
 127  
 128    ! set default values for namelist variables
 129    aiBiasActive = .false.  ! bias correct AI data (TT)
 130    gpBiasActive = .false.  ! bias correct GP data (ZTD)
 131    uaBiasActive = .false.  ! bias correct UA data (TT,ES)
 132    aiRevOnly    = .false.  ! AI: don't apply new correction but simply reverse any old corrections
 133    gpRevOnly    = .false.  ! GP: don't apply new correction but simply reverse any old corrections
 134    uaRevOnly    = .false.  ! UA: don't apply new correction but simply reverse any old corrections
 135    uaRejUnBcor  = .false.  ! UA: Set DATA QC flag bit 11 on to exclude uncorrected UA observations from assimilation
 136    uaNprofsMin  = 100      ! UA: If Nprofs for a given stn/stype < uaNprofsMin, flag the bcors as unusable
 137    uaNbiasCat   = 1        ! UA: Number of bias profile categories in UA bcor files: 1, or 2 for "asc" and "desc" categories
 138    uaNlatBands  = 1        ! UA: Number of latitude bands in ua_bcors_stype file (1 or 5): 1 = no bands (global biases).
 139    ! read in the namelist NAMBIASCONV
 140    if ( utl_isNamelistPresent('nambiasconv','./flnml') ) then
 141      nulnam = 0
 142      ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
 143      read(nulnam,nml=nambiasconv,iostat=ierr)
 144      if ( ierr /= 0 )  call utl_abort('bcc_readConfig: Error reading namelist section NAMBIASCONV')
 145      if ( mmpi_myid == 0 ) write(*,nml=nambiasconv)
 146      ierr = fclos(nulnam)
 147    else
 148      write(*,*)
 149      write(*,*) 'bcc_readconfig: NAMBIASCONV section is missing in the namelist. The default values will be used.'
 150    end if
 151        
 152    bcc_aiBiasActive = aiBiasActive
 153    bcc_gpBiasActive = gpBiasActive
 154    bcc_uaBiasActive = uaBiasActive
 155    
 156    ! read in the namelist NAMSONDETYPES
 157    if ( uaBiasActive .and. .not.uaRevOnly ) then
 158      if ( utl_isNamelistPresent('namsondetypes','./flnml') ) then
 159        nlNbSondes = MPC_missingValue_INT
 160        nlSondeTypes(:)   = 'empty'
 161        nlSondeCodes(:,:) = MPC_missingValue_INT
 162        nulnam = 0
 163        ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
 164        read(nulnam,nml=namsondetypes,iostat=ierr)
 165        if ( ierr /= 0 )  call utl_abort('bcc_readConfig: Error reading namelist section NAMSONDETYPES')
 166        if (nlNbSondes /= MPC_missingValue_INT) then
 167          call utl_abort('bcc_readConfig: check namsondetypes namelist section, you should remove nlNbSondes')
 168        end if
 169        nlNbSondes = 0
 170        do sondeIndex = 1, nSondesMax
 171          if (trim(nlSondeTypes(sondeIndex)) == 'empty') exit
 172          nlNbSondes = nlNbSondes + 1
 173        end do
 174        if ( mmpi_myid == 0 ) write(*,nml=namsondetypes)
 175        ierr = fclos(nulnam)
 176      else 
 177        write(*,*)
 178        call utl_abort('bcc_readconfig: NAMSONDETYPES section is missing in the namelist!')
 179      end if
 180      if ( nlNbSondes > nSondesMax ) call utl_abort('bcc_readconfig: Number of sonde types in namelist exceeds nSondesMax!')
 181      if ( nlNbSondes <= 0 )         call utl_abort('bcc_readconfig: Number of sonde types in namelist <= 0!')
 182      allocate( rsTypes(nlNbSondes) )
 183      do sondeIndex = 1, nlNbSondes
 184        rsTypes(sondeIndex)%name     = nlSondeTypes(sondeIndex)
 185        rsTypes(sondeIndex)%codes(:) = nlSondeCodes(sondeIndex,:)
 186      end do
 187    end if
 188    
 189    initialized = .true.
 190    
 191  end subroutine bcc_readConfig

 196  function bcc_UACorrection(timeOfDayX,corrNight,corrDay) result(uaCorrection)
 197    !:Purpose: Returns a UA bias correction from day and night corrections
 198    
 199    implicit none
 200    
 201    real(8) :: uaCorrection
 202    
 203    !Arguments:
 204    real(8), intent(in)  :: timeOfDayX ! 0.0 (night) <= timeOfDayX <= 1.0 (day), depends on solar_elev
 205    real(8), intent(in)  :: corrNight  ! night bias correction
 206    real(8), intent(in)  :: corrDay    ! day bias correction
 207    
 208    uaCorrection = MPC_missingValue_R8
 209    
 210    if ( timeOfDayX == 0.0d0 ) then
 211      uaCorrection = corrNight
 212    else if ( timeOfDayX == 1.0d0 ) then
 213      uaCorrection = corrDay
 214    else
 215      if ( corrNight /= MPC_missingValue_R8 .and. corrDay /= MPC_missingValue_R8 ) then
 216        uaCorrection = timeOfDayX*corrDay + (1.0d0-timeOfDayX)*corrNight
 217      end if
 218    end if
 219    
 220  end function bcc_UACorrection

 222  !-----------------------------------------------------------------------
 223  ! bcc_GetUACorrection
 224  !-----------------------------------------------------------------------
 225  subroutine bcc_GetUACorrection(varName,stnIndex,sondeTypeIndex,sondeType,biasProfileCategory,timeOfDayX,latband,obsPressure,corr,sourceCorr)
 226    !
 227    !:Purpose: Return a TT or TD bias correction (corr) 
 228    
 229    !
 230    !      varName        = (str) 'TT' or 'TD'
 231    !      stnIndex       = (int) station index (in uaStations) = -1 if station was not found in bcor file
 232    !      sondeTypeIndex = (int) sonde type index (in rsTypes) =  0 if sonde-type code is not associated with any type
 233    !      sondeType      = (str) sonde-type from rsTypes
 234    !      biasProfileCategory = (int) bias profile category = 1 (ascent/none) or 3 (descent)
 235    !      timeOfDayX     = (float)  0.0 (night) <= TimeOfDayX <= 1.0 (day), depends on solar_elev
 236    !      latband        = (int) 1-5
 237    !      obsPressure    = (float) pressure level (mb) of observation
 238    !
 239    !  Also returns info regarding the source of the correction returned:
 240    !      sourceCorr     = (str) 'none', 'stn', or 'stype'
 241    !
 242    ! Requires TT and TD correction profiles read from UA bcor file ua_bcors_stn and stored in
 243    !    ttCorrectionsStn(stnIndex,sondeTypeIndex,j,MandLev)
 244    !    tdCorrectionsStn(stnIndex,sondeTypeIndex,j,MandLev)
 245    ! with backup corrections by sonde-type read from UA bcor file ua_bcors_stype and stored in
 246    !    ttCorrections(sondeTypeIndex,latband,j,MandLev)
 247    !    tdCorrections(sondeTypeIndex,latband,j,MandLev)
 248    ! where MandLev = 1 (1000 mb) to 16 (10 mb)
 249    ! If uaNbiasCat = 1 (biasProfileCategory = 1)
 250    !       j = 1 (night)
 251    !           2 (day)
 252    ! If uaNbiasCat = 2 (biasProfileCategory = 1 [ascent] or 3 [descent])
 253    !       j = 1 (night-ascent)
 254    !           2 (day-ascent)
 255    !           3 (night-descent)
 256    !           4 (day-descent)
 257    !
 258    ! Interpolation of the correction profiles on mandatory levels is used to get the 
 259    ! correction at the observation level (obsPressure).
 260    ! Persistence is applied for observations outside the range of the mandatory levels.
 261    !
 262    implicit none
 263
 264    !Arguments:
 265    integer, intent(in)           ::  stnIndex
 266    integer, intent(in)           ::  sondeTypeIndex
 267    integer, intent(in)           ::  biasProfileCategory
 268    integer, intent(in)           ::  latband
 269    real(8), intent(in)           ::  timeOfDayX
 270    real(8), intent(in)           ::  obsPressure
 271    character(len=*), intent(in)  ::  varName
 272    character(len=*), intent(in)  ::  sondeType
 273    real(8), intent(out)          ::  corr
 274    character(len=*), intent(out) ::  sourceCorr
 275
 276    !Locals:
 277    real(8) ::  corrProfileStnDay(16), corrProfileStnNight(16), corrProfileStypeDay(16), corrProfileStypeNight(16)
 278    real(8) ::  corrDay, corrNight
 279    real(8) ::  weight1, weight2, deltaPressure, pressureAbove, pressureBelow, corrAbove, corrBelow
 280    real(8) ::  corrAboveNight, corrBelowNight, corrAboveDay, corrBelowDay
 281    integer ::  level, levelIndex
 282    logical ::  profileExistsStn, profileExistsStype, doInterp
 283
 284    sourceCorr = "none"
 285    
 286    ! Bias correction by station and sonde-type
 287    if ( stnIndex == -1 ) then
 288      profileExistsStn = .false.
 289    else
 290      if ( sondeTypeIndex > 0 ) then
 291        ! Check that correction profile for this station, sonde-type and TimeOfDay is available for use
 292        profileExistsStn = biasCorrPresentStn(stnIndex,sondeTypeIndex,biasProfileCategory)
 293      else ! There are no bias profiles for this station/stype combination
 294        profileExistsStn = .false.
 295      end if
 296    end if
 297
 298    ! BACKUP: Bias correction by sonde-type
 299    profileExistsStype = .false.
 300    if ( trim(sondeType) /= 'unknown' .and. trim(sondeType) /= 'Others' .and. trim(sondeType) /= 'None' ) then
 301      ! Check if correction profile for this sonde-type,latband,and TimeOfDay is available for use
 302      profileExistsStype = biasCorrPresentStype(sondeTypeIndex,latband,biasProfileCategory)
 303    end if
 304       
 305    if ( .not.profileExistsStn  .and. .not.profileExistsStype ) then
 306      corr = MPC_missingValue_R8
 307      return
 308    end if
 309    
 310    ! Fill the night and day bias correction profiles
 311    if ( varName == 'TT' ) then
 312      if ( profileExistsStn ) then
 313        corrProfileStnNight(:)   = ttCorrectionsStn(stnIndex,sondeTypeIndex,biasProfileCategory,:)
 314        corrProfileStnDay(:)     = ttCorrectionsStn(stnIndex,sondeTypeIndex,biasProfileCategory+1,:)
 315      end if
 316      if ( profileExistsStype ) then
 317        corrProfileStypeNight(:) = ttCorrections(sondeTypeIndex,latband,biasProfileCategory,:)
 318        corrProfileStypeDay(:)   = ttCorrections(sondeTypeIndex,latband,biasProfileCategory+1,:)
 319      end if
 320    else if ( varName == 'TD' ) then
 321      if ( profileExistsStn ) then
 322         corrProfileStnNight(:)   = tdCorrectionsStn(stnIndex,sondeTypeIndex,biasProfileCategory,:)
 323         corrProfileStnDay(:)     = tdCorrectionsStn(stnIndex,sondeTypeIndex,biasProfileCategory+1,:)
 324      end if
 325      if ( profileExistsStype ) then
 326         corrProfileStypeNight(:) = tdCorrections(sondeTypeIndex,latband,biasProfileCategory,:)
 327         corrProfileStypeDay(:)   = tdCorrections(sondeTypeIndex,latband,biasProfileCategory+1,:)
 328      end if
 329    else 
 330      call utl_abort('bcc_GetUACorrection: Unsupported varName '//varName)
 331    end if
 332    
 333    corr = MPC_missingValue_R8
 334    doInterp = .true.
 335
 336    !--------------------------------------------------------------------------------------
 337    ! Get the correction at the observation level (obsPressure)
 338    !--------------------------------------------------------------------------------------
 339    
 340    ! Check if obsPressure is outside range of levels (no interpolation possible)
 341    if ( obsPressure >= mandLevs(1) ) then
 342      levelIndex = 1
 343    else if ( obsPressure <= mandLevs(nMandLevs) ) then 
 344      levelIndex = nMandLevs
 345    else
 346      levelIndex = 0
 347    end if
 348    
 349    ! Check if obsPressure is close to one of the 16 mandatory levels (no interpolation needed)
 350    if ( levelIndex == 0 ) then
 351      do level = 1, nMandLevs
 352        if ( abs(obsPressure-mandLevs(level)) <= tolPress(level) ) then
 353          levelIndex = level
 354          exit
 355        end if
 356      end do
 357    end if
 358    
 359    ! If obsPressure is close to one of the 16 mandatory levels get the correction for the mandatory level: 
 360    ! Use correction by station with correction by stype as backup
 361    if ( levelIndex > 0 ) then
 362      if ( profileExistsStn ) then
 363        corrNight = corrProfileStnNight(levelIndex)
 364        corrDay   = corrProfileStnDay(levelIndex)
 365        sourceCorr = "stn"
 366        corr = bcc_UACorrection(timeOfDayX,corrNight,corrDay)
 367        if ( corr == MPC_missingValue_R8 ) then
 368          sourceCorr = "none"
 369          if ( profileExistsStype ) then
 370            corrNight = corrProfileStypeNight(levelIndex)
 371            corrDay   = corrProfileStypeDay(levelIndex)
 372            sourceCorr = "stype"
 373            corr = bcc_UACorrection(timeOfDayX,corrNight,corrDay)
 374            if ( corr == MPC_missingValue_R8 ) sourceCorr = "none"
 375          end if
 376        end if
 377      else
 378        if ( profileExistsStype ) then
 379          corrNight = corrProfileStypeNight(levelIndex)
 380          corrDay   = corrProfileStypeDay(levelIndex)
 381          sourceCorr = "stype"
 382          corr = bcc_UACorrection(timeOfDayX,corrNight,corrDay)
 383          if ( corr == MPC_missingValue_R8 ) sourceCorr = "none"
 384        end if
 385      end if
 386      doInterp = .false.
 387    end if
 388    
 389    ! or interpolate to get correction for observation level obsPressure:
 390    ! Use corrections by station with corrections by stype as backup
 391    if ( doInterp ) then
 392      corrAbove = MPC_missingValue_R8
 393      corrBelow = MPC_missingValue_R8
 394      
 395      if ( profileExistsStn ) then
 396        do level = 1, nMandLevs-1
 397          if ( obsPressure <= mandLevs(level) .and. obsPressure > mandLevs(level+1) ) then
 398            pressureBelow = mandLevs(level)
 399            pressureAbove = mandLevs(level+1)
 400            corrBelowNight = corrProfileStnNight(level)
 401            corrAboveNight = corrProfileStnNight(level+1)
 402            corrBelowDay   = corrProfileStnDay(level)
 403            corrAboveDay   = corrProfileStnDay(level+1)
 404            exit
 405          end if
 406        end do
 407        sourceCorr = "stn"
 408        if ( timeOfDayX == 0.0d0 ) then
 409          corrBelow = corrBelowNight
 410          corrAbove = corrAboveNight 
 411        else if ( timeOfDayX == 1.0d0 ) then
 412          corrBelow = corrBelowDay
 413          corrAbove = corrAboveDay
 414        else 
 415          if ( corrBelowNight /= MPC_missingValue_R8 .and. corrBelowDay /= MPC_missingValue_R8 ) then
 416            corrBelow = timeOfDayX*corrBelowDay + (1.0d0-timeOfDayX)*corrBelowNight
 417          end if
 418          if ( corrAboveNight /= MPC_missingValue_R8 .and. corrAboveDay /= MPC_missingValue_R8 ) then
 419            corrAbove = timeOfDayX*corrAboveDay + (1.0d0-timeOfDayX)*corrAboveNight
 420          end if
 421        end if
 422      end if
 423      
 424      if ( corrAbove == MPC_missingValue_R8 .or. corrBelow == MPC_missingValue_R8 ) then
 425        if ( profileExistsStype ) then
 426          do level = 1, nMandLevs-1
 427            if ( obsPressure <= mandLevs(level) .and. obsPressure > mandLevs(level+1) ) then
 428              pressureBelow = mandLevs(level)
 429              pressureAbove = mandLevs(level+1)
 430              corrBelowNight = corrProfileStypeNight(level)
 431              corrAboveNight = corrProfileStypeNight(level+1)
 432              corrBelowDay   = corrProfileStypeDay(level)
 433              corrAboveDay   = corrProfileStypeDay(level+1)
 434              exit
 435            end if
 436          end do
 437          sourceCorr = "stype"
 438          if ( timeOfDayX == 0.0d0 ) then
 439            corrBelow = corrBelowNight
 440            corrAbove = corrAboveNight 
 441          else if ( timeOfDayX == 1.0d0 ) then
 442            corrBelow = corrBelowDay
 443            corrAbove = corrAboveDay
 444          else 
 445            if ( corrBelowNight /= MPC_missingValue_R8 .and. corrBelowDay /= MPC_missingValue_R8 ) then
 446              corrBelow = timeOfDayX*corrBelowDay + (1.0d0-timeOfDayX)*corrBelowNight
 447            end if
 448            if ( corrAboveNight /= MPC_missingValue_R8 .and. corrAboveDay /= MPC_missingValue_R8 ) then
 449              corrAbove = timeOfDayX*corrAboveDay + (1.0d0-timeOfDayX)*corrAboveNight
 450            end if
 451          end if
 452        end if
 453      end if
 454      deltaPressure = log10(pressureBelow)-log10(pressureAbove)
 455      weight1 = (log10(pressureBelow) - log10(obsPressure)) / deltaPressure
 456      weight2 = (log10(obsPressure) - log10(pressureAbove)) / deltaPressure
 457      if ( corrAbove /= MPC_missingValue_R8 .and. corrBelow /= MPC_missingValue_R8 ) then
 458        corr = weight1*corrAbove + weight2*corrBelow
 459      else if ( corrAbove /= MPC_missingValue_R8 .and. max(weight1,weight2) == weight1 ) then
 460        corr = corrAbove
 461      else if ( corrBelow /= MPC_missingValue_R8 .and. max(weight1,weight2) == weight2 ) then
 462        corr = corrBelow
 463      else
 464        corr = MPC_missingValue_R8
 465        sourceCorr = "none"
 466      end if
 467    end if
 468  
 469  end subroutine bcc_GetUACorrection

 474  function bcc_StationIndex(station) result(stationIndexOut)
 475    !
 476    !:Purpose: Return the station index (order in array uaStations) corresponding to station
 477    !           Returns -1 if station is not found in uaStations.
 478    !
 479    implicit none
 480    
 481    integer  :: stationIndexOut
 482     
 483    !Arguments:
 484    character(len=*), intent(in)  :: station
 485
 486    !Locals:
 487    integer    :: stationIndex
 488

 590  !-----------------------------------------------------------------------
 591  subroutine bcc_GetSolarElevation(lat,lon,date,time,solarElev)
 592    !
 593    !:Purpose: Returns the solar elevation angle (degrees) given lat,lon,date(yyyymmdd),time(hhmm)
 594    !
 595    
 596    !    lat,  lon  = obsSpaceData header column OBS_LAT, OBS_LON (radians)
 597    !    date, time = obsSpaceData header column OBS_DAT (yyyymmdd), OBS_ETM (hhmm)  -or-
 598    !       datestamp = tim_getDatestamp()   -- date stamp for central (analysis) time
 599    !       ier  = newdate(datestamp,date,time,-3)
 600    !       time = time/10000
 601    !
 602    implicit none
 603
 604    !Arguments:
 605    integer, intent(in)  :: date          ! yyyymmdd
 606    integer, intent(in)  :: time          ! hhmm
 607    real(8), intent(in)  :: lat           ! radians
 608    real(8), intent(in)  :: lon           ! radians
 609    real(8), intent(out) :: solarElev     ! degrees
 610
 611    !Locals:
 612    integer :: days(13) = (/0,31,28,31,30,31,30,31,31,30,31,30,31/)
 613    integer :: leap_years(7) = (/2016,2020,2024,2028,2032,2036,2040/)
 614    integer :: yy, mmdd, mm, dd, hh, nn, doy
 615    real(8) :: timeUTC, timeLCL, solarDec, hourAngle, csz, sza
 616
 617    yy   = date/10000
 618    mmdd = date - yy*10000
 619    mm   = mmdd/100
 620    dd   = mmdd - mm*100 
 621    hh   = time/100
 622    nn   = time - hh*100
 623    
 624    if ( ANY(leap_years == yy) ) days(3) = 29
 625    doy = SUM(days(1:mm)) + dd
 626    timeUTC = float(hh) + float(nn)/60.0d0
 627    timeLCL = timeUTC + (lon*MPC_DEGREES_PER_RADIAN_R8)/15.0d0
 628    if ( timeLCL < 0.0d0  ) timeLCL = 24.0d0 + timeLCL
 629    if ( timeLCL > 24.0d0 ) timeLCL = timeLCL - 24.0d0
 630    solarDec = 0.4093d0 * sin(((2.0d0*MPC_PI_R8)/365.0d0)*(284.0d0 + float(doy)))
 631    hourAngle = 15.0d0*(timeLCL-12.0d0) * MPC_RADIANS_PER_DEGREE_R8
 632    csz = sin(solarDec)*sin(lat) + cos(solarDec)*cos(lat)*cos(hourAngle)
 633    sza = MPC_DEGREES_PER_RADIAN_R8 * acos(csz)
 634    solarElev = 90.0d0 - sza
 635  
 636  end subroutine bcc_GetSolarElevation

 725    !           The first line of the file is the number of aircraft plus one.

 789  !-----------------------------------------------------------------------
 790  ! bcc_applyAIBcor
 791  !-----------------------------------------------------------------------
 792  subroutine bcc_applyAIBcor(obsSpaceData)
 793    !
 794    !:Purpose:  to apply aircraft (AI) bias corrections to observations in ObsSpaceData
 795    !
 796    implicit none
 797
 798    !Arguments:
 799    type(struct_obs)        :: obsSpaceData
 800
 801    !Locals:
 802    integer  :: headerIndex, bodyIndex, codtyp
 803    integer  :: flag, phase, bufrCode
 804    integer  :: phaseIndex, levelIndex, stationIndex, stationNumber
 805    integer  :: countTailCorrections,  countBulkCorrections
 806    integer  :: headerFlag
 807    real(8)  :: corr, tt, oldCorr, pressure
 808    character(len=9) :: stnid, stnId1, stnId2
 809
 810    if (.not.initialized) call bcc_readConfig()
 811    
 812    if ( .not. aiBiasActive ) return
 813
 814    write(*,*) "bcc_applyAIBcor: start"
 815
 816    if ( .not.aiRevOnly ) call bcc_readAIBiases(aiBcFile)
 817    
 818    countTailCorrections = 0
 819    countBulkCorrections = 0
 820
 821    call obs_set_current_header_list(obsSpaceData,'AI')
 822    HEADER: do
 823      headerIndex = obs_getHeaderIndex(obsSpaceData)
 824      if ( headerIndex < 0 ) exit HEADER
 825      
 826      headerFlag = obs_headElem_i(obsSpaceData, OBS_ST1, headerIndex )
 827      
 828      call obs_set_current_body_list(obsSpaceData, headerIndex)
 829
 830      BODY: do
 831
 832        bodyIndex = obs_getBodyIndex(obsSpaceData)
 833        if ( bodyIndex < 0 ) exit BODY
 834
 835        if ( obs_bodyElem_i(obsSpaceData,OBS_ASS,bodyIndex) /= obs_assimilated ) cycle BODY 
 836
 837        bufrCode = obs_bodyElem_i(obsSpaceData, OBS_VNM, bodyIndex )
 838
 839        if ( bufrCode == BUFR_NETT ) then
 840          tt      = obs_bodyElem_r(obsSpaceData, OBS_VAR, bodyIndex )
 841          flag    = obs_bodyElem_i(obsSpaceData, OBS_FLG, bodyIndex )
 842          oldCorr = obs_bodyElem_r(obsSpaceData, OBS_BCOR, bodyIndex )
 843          corr = MPC_missingValue_R8
 844          
 845          if ( tt /= MPC_missingValue_R8 ) then
 846          
 847            if ( btest(flag, 6) .and. oldCorr /= MPC_missingValue_R8 ) then
 848              if ( btest(headerFlag, 15) ) then
 849                tt = tt - oldCorr
 850              else
 851                tt = tt + oldCorr
 852              end if
 853              flag = ibclr(flag, 6)
 854            end if
 855            if ( aiRevOnly ) corr = 0.0D0
 856             
 857            if ( .not.aiRevOnly ) then
 858                
 859              pressure = obs_bodyElem_r(obsSpaceData, OBS_PPP, bodyIndex ) * MPC_MBAR_PER_PA_R8
 860
 861              ! Get level index and current (mar 2020) static bulk corrections applied to AI TT data at derivate stage
 862              if ( (pressure <= 1100.d0) .and. (pressure > 700.d0) ) then
 863                corr = 0.0D0
 864                levelIndex = 5
 865              else if ( (pressure <= 700.d0)  .and. (pressure > 500.d0) ) then
 866                corr = -0.1D0
 867                levelIndex = 4
 868              else if ( (pressure <= 500.d0)  .and. (pressure > 400.d0) ) then
 869                corr = -0.2D0
 870                levelIndex = 3
 871              else if ( (pressure <= 400.d0)  .and. (pressure > 300.d0) ) then
 872                corr = -0.3D0
 873                levelIndex = 2
 874              else if ( (pressure <= 300.d0)  .and. (pressure > 100.d0) ) then
 875                corr = -0.5D0
 876                levelIndex = 1
 877              else 
 878                levelIndex = 0
 879                corr = 0.0D0
 880              end if
 881 
 882              codtyp = obs_headElem_i(obsSpaceData, OBS_ITY, headerIndex)
 883
 884              ! Default bulk corrections read from bcor file (applied if dynamic corrections are not availble for the aircraft)
 885              select case(  trim( codtyp_get_name(codtyp) ) )
 886                case('airep','ads')
 887                  phaseIndex = phaseAscentIndex
 888                case('amdar','acars')
 889                  phaseIndex = phaseDescentIndex
 890                case default
 891                  write(*,*) 'bcc_applyAIBcor: codtyp=', codtyp
 892                  call utl_abort('bcc_applyAIBcor: unknown codtyp') 
 893              end select
 894
 895              if ( levelIndex /= 0 ) then
 896                if ( AIttCorrections(1,phaseIndex,levelIndex) /= MPC_missingValue_R8 ) corr = AIttCorrections(1,phaseIndex,levelIndex)
 897                countBulkCorrections = countBulkCorrections + 1
 898              end if
 899
 900              headerIndex = obs_bodyElem_i(obsSpaceData, OBS_HIND, bodyIndex  )
 901              stnid = trim( obs_elem_c(obsSpaceData,'STID',headerIndex) )
 902
 903              ! on verifie si la station est dans le dictionnaire du fichier de correction de biais
 904              !---------------------------------------------------------------------------------
 905              stationNumber = 0
 906              stnId2 = trim(stnid)
 907              do stationIndex = 1, nbAircrafts
 908                stnId1 = trim(aircraftIds(stationIndex))
 909                if ( stnId2(2:9) == stnId1(1:8) ) stationNumber = stationIndex
 910              end do
 911            
 912              phase =  obs_headElem_i(obsSpaceData, OBS_PHAS, headerIndex  )
 913
 914              ! If the aircraft is in the bias correction file, get the new correction
 915              ! AIttCorrections(stationNumber,phaseIndex,levelIndex) where
 916              !     stationNumber = index for this AC in bias correction file (0 if not found)
 917              !     phaseIndex   = index for the 3 phases of flight (level, asc, desc)
 918              !     levelIndex   = index for the 5 layers (100-300, 300-400,400-500,500-700,700-1100)
 919              !  and use it instead of bulk value (if it is not missing value).
 920              if ( stationNumber /= 0 ) then 
 921                phaseIndex = 0
 922                if ( phase == phaseLevel   ) phaseIndex = phaseLevelIndex
 923                if ( phase == phaseAscent  ) phaseIndex = phaseAscentIndex
 924                if ( phase == phaseDescent ) phaseIndex = phaseDescentIndex
 925                if ( levelIndex /= 0 .and. phaseIndex /= 0 ) then
 926                  if ( AIttCorrections(stationNumber,phaseIndex,levelIndex) /= MPC_missingValue_R8 ) then
 927                    corr = AIttCorrections(stationNumber,phaseIndex,levelIndex)
 928                    countTailCorrections = countTailCorrections + 1
 929                    countBulkCorrections = countBulkCorrections - 1
 930                  end if
 931                end if
 932              end if
 933            
 934              ! Apply the bias correction (bulk or new) and set the "bias corrected" bit in TT data flag ON
 935              tt = tt + corr
 936              flag = ibset(flag, 6)
 937            
 938            end if
 939            
 940          end if
 941
 942          call obs_bodySet_r( obsSpaceData, OBS_BCOR, bodyIndex, corr )
 943          call obs_bodySet_r( obsSpaceData, OBS_VAR , bodyIndex, tt   )
 944          call obs_bodySet_i( obsSpaceData, OBS_FLG , bodyIndex, flag )        
 945
 946        end if
 947        
 948      end do BODY
 949      
 950      ! Set header flag bit 15 on to indicate that bias correction has been ADDED to raw value.
 951      ! In older versions of this routine, such as the version used in IC-3/GDPS v8.0, the correction was SUBTRACTED.
 952      headerFlag = ibset(headerFlag, 15)
 953      call obs_headSet_i( obsSpaceData, OBS_ST1, headerIndex, headerFlag )
 954      
 955    end do HEADER
 956    
 957    if ( countBulkCorrections + countTailCorrections /= 0 ) then
 958      write (*, '(a58, i10)' ) "bcc_applyAIBcor: Number of obs with TT bulk correction  = ", countBulkCorrections
 959      write (*, '(a58, i10)' ) "bcc_applyAIBcor: Number of obs with TT tail correction  = ", countTailCorrections
 960    else
 961      write(*,*) "bcc_applyAIBcor: No AI data found"
 962    end if
 963    
 964    if ( allocated(AIttCorrections) ) deallocate(AIttCorrections)
 965    if ( allocated(aircraftIds)   ) deallocate(aircraftIds)
 966    
 967    write(*,*) "bcc_applyAIBcor: end"
 968    
 969  end subroutine bcc_applyAIBcor
