 44  !--------------------------------------------------------------------------
 45  ! cfn_calcJo
 46  !--------------------------------------------------------------------------
 47  subroutine cfn_calcJo(lobsSpaceData)
 48    !
 49    !:Purpose: To compute JO contribution of each assimilated and diagnosed
 50    !          datum, and to store the result in OBS_JOBS
 51    implicit none
 52
 53    ! Arguments:
 54    type(struct_obs) :: lobsSpaceData
 55
 56    ! Locals:
 57    integer :: bodyIndex
 58
 59!$OMP PARALLEL DO PRIVATE(bodyIndex)
 60    do bodyIndex=1,obs_numbody(lobsSpaceData)
 61
 62      if ( obs_bodyElem_i( lobsSpaceData, OBS_ASS, bodyIndex) == obs_assimilated) then
 63        call obs_bodySet_r( lobsSpaceData, OBS_JOBS, bodyIndex, &
 64             ( obs_bodyElem_r( lobsSpaceData, OBS_WORK, bodyIndex ) &
 65             * obs_bodyElem_r( lobsSpaceData, OBS_WORK, bodyIndex ) &
 66             ) / 2.d0 )
 67      else
 68        call obs_bodySet_r(lobsSpaceData,OBS_JOBS,bodyIndex, 0.0d0)
 69      end if
 70
 71    end do
 72!$OMP END PARALLEL DO
 73
 74  end subroutine cfn_calcJo

 76  !--------------------------------------------------------------------------
 77  ! cfn_sumJo
 78  !--------------------------------------------------------------------------
 79  subroutine cfn_sumJo( lobsSpaceData, pjo, beSilent_opt )
 80    !
 81    !:Purpose: To compute the sum of Jo contributions saved in OBS_JOBS. Also,
 82    !          to compute contribution of each family of observation (for
 83    !          diagnostic purposes)
 84    implicit none
 85
 86    ! Arguments:
 87    type(struct_obs) :: lobsSpaceData
 88    real(8), intent(out) :: pjo ! Total observation cost function
 89    logical, optional :: beSilent_opt
 90
 91    ! Locals:
 92    integer :: bodyIndex, tovsIndex, sensorIndex, headerIndex, bodyIndexBeg, bodyIndexEnd
 93    integer :: channelNumber, channelNumberIndexInListFound, channelIndex
 94    integer :: sensorIndexInList, sensorIndexInListFound
 95    logical :: beSilent
 96
 97    real(8) :: dljoraob, dljoairep, dljosatwind, dljoscat, dljosurfc, dljotov, dljosst, dljoice
 98    real(8) :: dljoprof, dljogpsro, dljogpsztd, dljochm, pjo_1, dljoaladin, dljohydro, dljoradar
 99    real(8) :: dljotov_sensors( tvs_nsensors )
100    real(8) :: joTovsPerChannelSensor(tvs_maxNumberOfChannels,tvs_nsensors)
101    character(len=15) :: lowerCaseName
102
103    logical :: printJoTovsPerChannelSensor
104  
105    real(8), allocatable :: joSSTInstrument(:)
106    integer, allocatable :: nobsInstrument(:), nobsInstrumentGlob(:)
107    integer :: SSTdatasetIndex, codeType, ierr
108
109    if ( present(beSilent_opt) ) then
110      beSilent = beSilent_opt
111    else
112      beSilent = .false.
113    end if
114
115    if (.not. allocated(channelNumberList)) then
116      allocate(channelNumberList(tvs_maxNumberOfChannels,tvs_nsensors))
117    end if
118    if (.not. allocated(sensorNameList)) then
119      allocate(sensorNameList(tvs_nsensors))
120    end if
121    
122    if(oer_getSSTdataParam_int('numberSSTDatasets') > 0) then
123      allocate(joSSTInstrument(oer_getSSTdataParam_int('numberSSTDatasets')))
124      allocate(nobsInstrument(oer_getSSTdataParam_int('numberSSTDatasets')))
125      allocate(nobsInstrumentGlob(oer_getSSTdataParam_int('numberSSTDatasets')))
126    end if
127
128    call readNameList
129    printJoTovsPerChannelSensor = .false.
130    if (any(sensorNameList(:) /= '') .and. any(channelNumberList(:,:) /= 0)) then
131      printJoTovsPerChannelSensor = .true.
132    end if
133
134    dljogpsztd = 0.d0
135    dljoraob = 0.d0
136    dljoairep = 0.d0
137    dljosatwind = 0.d0
138    dljosurfc = 0.d0
139    dljoscat = 0.d0
140    dljotov = 0.d0
141    dljogpsro = 0.d0
142    dljoprof = 0.d0
143    dljochm = 0.d0
144    dljosst = 0.0d0
145    dljoaladin = 0.d0
146    dljoice = 0.0d0
147    dljotov_sensors(:) = 0.d0
148    joTovsPerChannelSensor(:,:) = 0.0d0
149    dljohydro = 0.0d0
150    dljoradar = 0.0d0
151    joSSTInstrument(:) = 0.0d0
152    nobsInstrumentGlob(:) = 0
153    nobsInstrument(:) = 0
154
155    pjo = 0.0d0
156
157    do bodyIndex = 1, obs_numbody(lobsSpaceData)
158
159      pjo_1 = obs_bodyElem_r(lobsSpaceData, OBS_JOBS, bodyIndex)
160
161      ! total observation cost function
162      pjo   = pjo + pjo_1
163
164      ! subcomponents of observation cost function (diagnostic only)
165      select case(obs_getFamily(lobsSpaceData, bodyIndex = bodyIndex))
166      case('UA')
167        dljoraob    = dljoraob    + pjo_1
168      case('AI')
169        dljoairep   = dljoairep   + pjo_1
170      case('SW')
171        dljosatwind = dljosatwind + pjo_1
172      case('SF')
173        dljosurfc   = dljosurfc   + pjo_1
174      case('SC')
175        dljoscat    = dljoscat    + pjo_1
176      case('TO')
177        dljotov     = dljotov     + pjo_1
178      case('RO')
179        dljogpsro   = dljogpsro   + pjo_1
180      case('PR')
181        dljoprof    = dljoprof    + pjo_1
182      case('GP')
183        dljogpsztd  = dljogpsztd  + pjo_1
184      case('CH')
185        dljochm     = dljochm     + pjo_1
186      case('TM')
187        dljosst     = dljosst     + pjo_1
188      case('AL')
189        dljoaladin  = dljoaladin  + pjo_1
190      case('GL')
191        dljoice     = dljoice     + pjo_1
192      case('HY')
193        dljohydro   = dljohydro   + pjo_1
194      case('RA')
195        dljoradar   = dljoradar   + pjo_1
196      end select
197    enddo
198
199    do tovsIndex = 1, tvs_nobtov
200      headerIndex = tvs_headerIndex( tovsIndex )
201      if ( headerIndex > 0 ) then
202        bodyIndexBeg = obs_headElem_i(lobsSpaceData, OBS_RLN, headerIndex)
203        bodyIndexEnd = obs_headElem_i(lobsSpaceData, OBS_NLV, headerIndex) + bodyIndexBeg - 1
204        sensorIndex = tvs_lsensor (tovsIndex)
205
206        sensorIndexInListFound = 0
207        if ( printJoTovsPerChannelSensor ) then
208          loopSensor1: do sensorIndexInList = 1, tvs_nsensors
209            call up2low(sensorNameList(sensorIndexInList),lowerCaseName)
210
211            if ( trim(lowerCaseName) == trim(inst_name(tvs_instruments(sensorIndex))) ) then
212              sensorIndexInListFound = sensorIndexInList
213              exit loopSensor1
214            end if
215
216          end do loopSensor1
217        end if
218
219        do bodyIndex = bodyIndexBeg, bodyIndexEnd
220          pjo_1 = obs_bodyElem_r(lobsSpaceData, OBS_JOBS, bodyIndex)
221          dljotov_sensors(sensorIndex) =  dljotov_sensors(sensorIndex) + pjo_1
222
223          if ( printJoTovsPerChannelSensor .and. &
224               sensorIndexInListFound > 0 ) then
225            call tvs_getChannelNumIndexFromPPP( lobsSpaceData, headerIndex, bodyIndex, &
226                                                channelNumber, channelIndex )
227            channelNumberIndexInListFound = utl_findloc(channelNumberList(:,sensorIndexInListFound), &
228                                                        channelNumber)
229
230            if ( channelNumberIndexInListFound > 0 ) then
231              joTovsPerChannelSensor(channelNumberIndexInListFound,sensorIndexInListFound) = &
232                        joTovsPerChannelSensor(channelNumberIndexInListFound,sensorIndexInListFound) + &
233                        pjo_1
234            end if
235
236          end if
237
238        end do
239      end if
240    end do
241
242    if(oer_getSSTdataParam_int('numberSSTDatasets') > 0) then
243      do headerIndex = 1, obs_numheader(lobsSpaceData)
244        codeType     = obs_headElem_i(lobsSpaceData, OBS_ITY, headerIndex)
245        bodyIndexBeg = obs_headElem_i(lobsSpaceData, OBS_RLN, headerIndex)
246        bodyIndexEnd = obs_headElem_i(lobsSpaceData, OBS_NLV, headerIndex) + bodyIndexBeg - 1
247        do bodyIndex = bodyIndexBeg, bodyIndexEnd
248          pjo_1 = obs_bodyElem_r(lobsSpaceData, OBS_JOBS, bodyIndex)
249          dataset_loop: do SSTdatasetIndex = 1, oer_getSSTdataParam_int('numberSSTDatasets')
250            if (codeType == oer_getSSTdataParam_int('codeType', SSTdatasetIndex) .and. codeType /= codtyp_get_codtyp('satob')) then
251              joSSTInstrument(SSTdatasetIndex) = joSSTInstrument(SSTdatasetIndex) + pjo_1
252              nobsInstrument(SSTdatasetIndex) = nobsInstrument(SSTdatasetIndex) + 1
253              exit dataset_loop
254            else
255              if (obs_elem_c(lobsSpaceData, 'STID', headerIndex) == oer_getSSTdataParam_char('sensor', SSTdatasetIndex)) then
256                joSSTInstrument(SSTdatasetIndex) = joSSTInstrument(SSTdatasetIndex) + pjo_1
257                nobsInstrument(SSTdatasetIndex) = nobsInstrument(SSTdatasetIndex) + 1
258                exit dataset_loop
259              end if
260            end if
261          end do dataset_loop
262        end do
263      end do
264    end if
265
266    call mmpi_allreduce_sumreal8scalar( pjo, "GRID" )
267    call mmpi_allreduce_sumreal8scalar( dljoraob, "GRID" )
268    call mmpi_allreduce_sumreal8scalar( dljoairep, "GRID" )
269    call mmpi_allreduce_sumreal8scalar( dljosatwind, "GRID" )
270    call mmpi_allreduce_sumreal8scalar( dljosurfc, "GRID" )
271    call mmpi_allreduce_sumreal8scalar( dljoscat, "GRID" )
272    call mmpi_allreduce_sumreal8scalar( dljotov, "GRID" )
273    call mmpi_allreduce_sumreal8scalar( dljogpsro, "GRID" )
274    call mmpi_allreduce_sumreal8scalar( dljoprof, "GRID" )
275    call mmpi_allreduce_sumreal8scalar( dljogpsztd, "GRID" )
276    call mmpi_allreduce_sumreal8scalar( dljochm, "GRID" )
277    call mmpi_allreduce_sumreal8scalar( dljosst, "GRID" )
278    call mmpi_allreduce_sumreal8scalar( dljoaladin,"GRID")
279    call mmpi_allreduce_sumreal8scalar( dljoice, "GRID" )
280    call mmpi_allreduce_sumreal8scalar( dljohydro, "GRID" )
281    call mmpi_allreduce_sumreal8scalar( dljoradar, "GRID" )
282    do sensorIndex = 1, tvs_nsensors
283      call mmpi_allreduce_sumreal8scalar(dljotov_sensors(sensorIndex), "GRID")
284    end do
285    if (printJoTovsPerChannelSensor) then
286      loopSensor2: do sensorIndex = 1, tvs_nsensors
287        if (trim(sensorNameList(sensorIndex)) == '') cycle loopSensor2
288
289        call mmpi_allreduce_sumR8_1d(joTovsPerChannelSensor(:,sensorIndex), "GRID")
290      end do loopSensor2
291    end if
292
293    ! SST data per instrument
294    do SSTdatasetIndex = 1, oer_getSSTdataParam_int('numberSSTDatasets')
295      call mmpi_allreduce_sumreal8scalar(joSSTInstrument(SSTdatasetIndex), "grid")
296      call rpn_comm_allreduce(nobsInstrument(SSTdatasetIndex), nobsInstrumentGlob(SSTdatasetIndex), &
297                              1, "mpi_integer", "mpi_sum", "grid", ierr)
298    end do
299
300    if ( mmpi_myid == 0 .and. .not. beSilent ) then
301      write(*,'(a15,f30.17)') 'Jo(UA)   = ', dljoraob
302      write(*,'(a15,f30.17)') 'Jo(AI)   = ', dljoairep
303      write(*,'(a15,f30.17)') 'Jo(SF)   = ', dljosurfc
304      write(*,'(a15,f30.17)') 'Jo(SC)   = ', dljoscat
305      write(*,'(a15,f30.17)') 'Jo(TO)   = ', dljotov
306      write(*,'(a15,f30.17)') 'Jo(SW)   = ', dljosatwind
307      write(*,'(a15,f30.17)') 'Jo(PR)   = ', dljoprof
308      write(*,'(a15,f30.17)') 'Jo(RO)   = ', dljogpsro
309      write(*,'(a15,f30.17)') 'Jo(GP)   = ', dljogpsztd
310      write(*,'(a15,f30.17)') 'Jo(CH)   = ', dljochm
311      write(*,'(a15,f30.17)') 'Jo(TM)   = ', dljosst
312      write(*,'(a15,f30.17)') 'Jo(AL)   = ', dljoaladin
313      write(*,'(a15,f30.17)') 'Jo(GL)   = ', dljoice
314      write(*,'(a15,f30.17)') 'Jo(HY)   = ', dljohydro
315      write(*,'(a15,f30.17)') 'Jo(RA)   = ', dljoradar
316      write(*,*) ' '
317      if ( tvs_nsensors > 0 ) then
318        write(*,'(1x,a)') 'For TOVS decomposition by sensor:'
319        write(*,'(1x,a)') '#  plt sat ins    Jo'
320        do sensorIndex = 1, tvs_nsensors
321          write(*,'(i2,1x,a,1x,a,1x,i2,1x,f30.17)') sensorIndex, inst_name(tvs_instruments(sensorIndex)), &
322                                                    platform_name(tvs_platforms(sensorIndex)), &
323                                                    tvs_satellites(sensorIndex), &
324                                                    dljotov_sensors(sensorIndex)
325        end do
326        write(*,*) ' '
327      end if
328
329      ! print per channel information
330      if ( tvs_nsensors > 0 .and. printJoTovsPerChannelSensor ) then
331        write(*,'(1x,a)') 'For TOVS decomposition by sensor/channel:'
332        write(*,'(1x,a)') 'index  sensorName  channel  Jo'
333        loopSensor3: do sensorIndex = 1, tvs_nsensors
334        if ( trim(sensorNameList(sensorIndex)) == '' ) cycle loopSensor3
335
336          loopChannel: do channelIndex = 1, tvs_maxNumberOfChannels
337            if ( channelNumberList(channelIndex,sensorIndex) == 0 ) cycle loopChannel
338
339            write(*,'(i2,1x,a,1x,i4,1x,f30.17)') sensorIndex, &
340                                                 sensorNameList(sensorIndex), &
341                                                 channelNumberList(channelIndex,sensorIndex), &
342                                                 joTovsPerChannelSensor(channelIndex,sensorIndex)
343          end do loopChannel
344
345        end do loopSensor3
346        write(*,*) ' '
347      end if
348
349      ! print SST data per instrument
350      if(oer_getSSTdataParam_int('numberSSTDatasets') > 0) then
351        write(*,*) 'cfn_sumJo: SST data by data type:'
352        write(*,'(a5,a15,a10,a30,a10, a15)') 'index', ' instrument', ' sensor', ' Jo', ' nobs', ' Jo/nobs'
353        do SSTdatasetIndex = 1, oer_getSSTdataParam_int('numberSSTDatasets')
354          if (nobsInstrumentGlob(SSTdatasetIndex) > 0) then
355            write(*,'(i5,a15,a10,f30.17,i15,f10.5)') SSTdatasetIndex, &
356                                                     oer_getSSTdataParam_char('instrument', SSTdatasetIndex),&
357                                                     oer_getSSTdataParam_char('sensor', SSTdatasetIndex), &
358                                                     joSSTInstrument(SSTdatasetIndex), &
359                                                     nobsInstrumentGlob(SSTdatasetIndex),&
360                                                     joSSTInstrument(SSTdatasetIndex) / &
361                                                     real(nobsInstrumentGlob(SSTdatasetIndex))
362          end if    
363        end do
364      end if
365
366    end if
367    
368    if(oer_getSSTdataParam_int('numberSSTDatasets') > 0) then
369      deallocate(joSSTInstrument)
370      deallocate(nobsInstrument)
371      deallocate(nobsInstrumentGlob)
372    end if
373
374  end subroutine cfn_sumJo

379  subroutine readNameList
380    !
381    !:Purpose: Reading NAMCFN namelist by any subroutines in costfunction_mod module.
382    !
383    implicit none
384
385    integer :: nulnam, ierr
386    integer, external :: fnom, fclos
387    logical, save :: nmlAlreadyRead = .false.
388    NAMELIST /NAMCFN/ sensorNameList, channelNumberList
389
390    if ( .not. nmlAlreadyRead ) then
391      nmlAlreadyRead = .true.
392
393      !- Setting default values
394      sensorNameList(:) = ''
395      channelNumberList(:,:) = 0
396
397      if ( .not. utl_isNamelistPresent('NAMCFN','./flnml') ) then
398        if ( mmpi_myid == 0 ) then
399          write(*,*) 'NAMCFN is missing in the namelist. The default values will be taken.'
400        end if
401
402      else
403        ! Reading the namelist
404        nulnam = 0
405        ierr = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
406        read(nulnam, nml=namcfn, iostat=ierr)
407        if ( ierr /= 0) call utl_abort('costfunction_mod: Error reading namelist')
408        ierr = fclos(nulnam)
409
410        call sortChannelNumbersInNml
411      end if
412      if ( mmpi_myid == 0 ) write(*,nml=namcfn)
413    end if
414
415  end subroutine readNameList

417  !--------------------------------------------------------------------------
418  ! sortChannelNumbersInNml
419  !--------------------------------------------------------------------------
420  subroutine sortChannelNumbersInNml
421    !
422    !:Purpose: Sort channelNumbers in NAMCFN namelist. This involves removing
423    !          the duplicates and combine channelNumbers of same sensor
424    !          prescribed on different lines.
425    !
426    implicit none
427
428    ! Arguments:
429
430    ! Locals:
431    integer :: channelIndex
432    integer :: channelIndex1, channelIndex2
433    integer :: sensorIndexInList, sensorIndexInList1, sensorIndexInList2
434
435    character(len=15) :: sensorName1LowerCase, sensorName2LowerCase
436
437    if ( mmpi_myid == 0 ) then
438      write(*,*) 'costfunction_mod: sortChannelNumbersInNml START'
439    end if
440
441    ! set duplicate channelNumber for each sensor to zero
442    loopSensor3: do sensorIndexInList = 1, tvs_nsensors
443      if ( trim(sensorNameList(sensorIndexInList)) == '' ) cycle loopSensor3
444
445      loopChannel2: do channelIndex = tvs_maxNumberOfChannels, 2, -1
446        if ( channelNumberList(channelIndex,sensorIndexInList) == 0 ) cycle loopChannel2
447
448        if ( any(channelNumberList(1:channelIndex-1,sensorIndexInList) == &
449                 channelNumberList(channelIndex    ,sensorIndexInList)) ) then
450          channelNumberList(channelIndex,sensorIndexInList) = 0
451        end if
452      end do loopChannel2
453    end do loopSensor3
454
455    ! remove later duplicates of sensor/channel pairs
456    loopSensor4: do sensorIndexInList1 = 1, tvs_nsensors-1
457      if ( trim(sensorNameList(sensorIndexInList1)) == '' ) cycle loopSensor4
458
459      call up2low(sensorNameList(sensorIndexInList1),sensorName1LowerCase)
460
461      loopSensor5: do sensorIndexInList2 = sensorIndexInList1+1, tvs_nsensors
462        if ( trim(sensorNameList(sensorIndexInList2)) == '' ) cycle loopSensor5
463
464        call up2low(sensorNameList(sensorIndexInList2),sensorName2LowerCase)
465
466        if ( trim(sensorName2LowerCase) == trim(sensorName1LowerCase) ) then
467          loopChannel3: do channelIndex2 = 1, tvs_maxNumberOfChannels
468            if ( channelNumberList(channelIndex2,sensorIndexInList2) == 0 ) cycle loopChannel3
469
470            if ( any(channelNumberList(:           ,sensorIndexInList1) == &
471                     channelNumberList(channelIndex2,sensorIndexInList2)) ) then
472              ! set second occurance of channelNumber to zero
473              channelNumberList(channelIndex2,sensorIndexInList2) = 0
474            else
475
476              ! replace the first zero channelNumber of first sensor with new non-zero channelNumber
477              do channelIndex1 = 1, tvs_maxNumberOfChannels
478                if ( channelNumberList(channelIndex1,sensorIndexInList1) == 0 ) then
479                  channelNumberList(channelIndex1,sensorIndexInList1) = &
480                        channelNumberList(channelIndex2,sensorIndexInList2)
481                  channelNumberList(channelIndex2,sensorIndexInList2) = 0
482                  exit
483                end if
484              end do
485
486            end if
487
488          end do loopChannel3
489        end if
490      end do loopSensor5
491    end do loopSensor4
492
493    ! if all entries for a sensor are zero, remove the sensor from namelist.
494    ! otherwise sort in ascending order
495    loopSensor6: do sensorIndexInList = 1, tvs_nsensors
496      if ( trim(sensorNameList(sensorIndexInList)) == '' ) cycle loopSensor6
497
498      if ( all(channelNumberList(:,sensorIndexInList) == 0) ) then
499        sensorNameList(sensorIndexInList) = ''
500      else
501        call isort(channelNumberList(:,sensorIndexInList),tvs_maxNumberOfChannels)
502      end if
503    end do loopSensor6
504
505    if ( mmpi_myid == 0 ) then
506      write(*,*) 'costfunction_mod: sortChannelNumbersInNml END'
507    end if
508
509  end subroutine sortChannelNumbersInNml
