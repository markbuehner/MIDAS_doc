 73  !--------------------------------------------------------------------------
 74  ! sstb_computeBias
 75  !--------------------------------------------------------------------------
 76  subroutine sstb_computeBias(obsData, hco, vco)
 77    !
 78    !:Purpose: compute bias for SST satellite data with respect to insitu data 
 79    !  
 80    implicit none
 81    
 82    ! Arguments: 
 83    type(struct_obs), intent(inout)          :: obsData              ! obsSpaceData
 84    type(struct_hco), intent(inout), pointer :: hco                  ! horizontal grid structure
 85    type(struct_vco), intent(in)   , pointer :: vco                  ! vertical grid structure
 86      
 87    ! locals
 88    integer                     :: sensorIndex, productIndex
 89    real(8)                     :: insituGrid   (hco%ni, hco%nj)
 90    real(8)                     :: satelliteGrid(hco%ni, hco%nj)
 91    logical                     :: mask(hco%ni, hco%nj), openWater(hco%ni, hco%nj) 
 92    type(struct_ocm)            :: oceanMask
 93    integer                     :: numberOpenWaterPoints, lonIndex, latIndex
 94    integer                     :: nobsFoundInsituGlob, nobsFoundInsituLoc
 95    integer                     :: nobsFoundSatGlob, nobsFoundSatLoc
 96    type(struct_gsv)            :: stateVector_ice
 97    real(4), pointer            :: seaice_ptr(:,:,:)
 98    integer         , parameter :: numberProducts = 2  ! day and night
 99    character(len=*), parameter :: listProducts(numberProducts)= (/'day', 'night'/)
100
101    write(*,*) 'sstb_computeBias: Starting...'
102    write(*,*) 'sstb_computeBias: Sea-ice Fraction threshold: ', iceFractionThreshold
103
104    ! read the namelist
105    call readNml()
106
107    ! get mpi topology
108    call mmpi_setup_lonbands(hco%ni, lonPerPE, lonPerPEmax, myLonBeg, myLonEnd)
109    call mmpi_setup_latbands(hco%nj, latPerPE, latPerPEmax, myLatBeg, myLatEnd)
110
111    ! get latest sea-ice analysis
112    call gsv_allocate(stateVector_ice, 1, hco, vco, dataKind_opt = 4, &
113                      datestamp_opt = -1, mpi_local_opt = .false.,    &
114                      varNames_opt = (/'LG'/), hInterpolateDegree_opt ='LINEAR')
115    call gio_readFromFile(stateVector_ice, './seaice_analysis', ' ','A', &
116                           unitConversion_opt=.false., containsFullField_opt=.true.)
117    call gsv_getField(stateVector_ice, seaice_ptr)
118    
119    ! Get land mask from analysisgrid file (1=water, 0=land) 
120    ! and the number of open water points
121    call ocm_readMaskFromFile(oceanMask, hco, vco, './analysisgrid')
122
123    numberOpenWaterPoints = 0
124    mask(:, :) = .false.
125    openWater(:, :) = .false.
126    do latIndex = 1, hco%nj
127      do lonIndex = 1, hco%ni
128        if (oceanMask%mask(lonIndex, latIndex, 1)) then
129          mask(lonIndex, latIndex) = .true.
130          if (seaice_ptr(lonIndex, latIndex, 1) <= iceFractionThreshold) then
131            openWater(lonIndex, latIndex) = .true.
132            numberOpenWaterPoints = numberOpenWaterPoints + 1
133          end if
134        end if
135      end do
136    end do
137    call ocm_deallocate(oceanMask)
138    call gsv_deallocate(stateVector_ice)
139    write(*,*) 'sstb_computeBias: computing bias for ', numberOpenWaterPoints, ' open water points'
140    
141    insituGrid(:, :) = MPC_missingValue_R8
142
143    call sstb_getGriddedObs(obsData, insituGrid, nobsFoundInsituGlob, &
144                            nobsFoundInsituLoc, hco, openWater, 'insitu')
145
146    do sensorIndex = 1, numberSensors 
147      do productIndex = 1, numberProducts
148        if (nobsFoundInsituGlob > 0) then
149          satelliteGrid(:, :) = MPC_missingValue_R8
150          call sstb_getGriddedObs(obsData, satelliteGrid(:, :), nobsFoundSatGlob, &
151                                  nobsFoundSatLoc, hco, openWater, sensorList(sensorIndex), &
152                                  dayOrNight_opt = listProducts(productIndex))
153          if (nobsFoundSatGlob > 0) then
154            call sstb_getGriddedBias(satelliteGrid(:, :), insituGrid, nobsFoundSatGlob, &
155                                     nobsFoundSatLoc, hco, vco, mask, openWater, &
156                                     sensorList(sensorIndex), numberOpenWaterPoints, &
157                                     listProducts(productIndex))
158          else
159            write(*,*) 'sstb_computeBias: WARNING: missing ', trim(sensorList(sensorIndex)), ' ', &
160                       trim(listProducts(productIndex)),' data.' 
161            write(*,*) 'Bias estimate will be read from the previous state...'
162            call sstb_getBiasFromPreviousState(hco, vco, &
163                                               sensorList(sensorIndex), &
164                                               listProducts(productIndex)) 
165          end if
166        else
167          write(*,*) 'sstb_computeBias: WARNING: missing insitu data.'
168          write(*,*) 'sstb_computeBias: bias estimates for all sensors will be read from the previous state...'
169          call sstb_getBiasFromPreviousState(hco, vco, &
170                                             sensorList(sensorIndex), &
171                                             listProducts(productIndex)) 
172        end if
173      end do
174    end do
175      
176    write(*,*) 'sstb_computeBias: done.'
177  
178  end subroutine sstb_computeBias

182  !--------------------------------------------------------------------------
183  subroutine sstb_getGriddedObs(obsData, obsGrid, countObsGlob, countObsLoc, &
184                                hco, openWater, instrument, dayOrNight_opt)
185    !
186    !:Purpose: put observations of a given family on the grid
187    !           
188    implicit none
189    
190    ! Arguments: 
191    type(struct_obs), intent(inout)          :: obsData        ! obsSpaceData
192    real(8)         , intent(inout)          :: obsGrid(:,:)   ! observations on the grid
193    integer         , intent(out)            :: countObsGlob   ! global number of data found (all procs)
194    integer         , intent(out)            :: countObsLoc    ! number of data found (current MPI proc)
195    type(struct_hco), intent(in)   , pointer :: hco            ! horizontal grid structure
196    logical         , intent(in)             :: openWater(:,:) ! open water points (.true.)
197    character(len=*), intent(in)             :: instrument     ! name of instrument
198    character(len=*), intent(in), optional   :: dayOrNight_opt ! look for daytime or nighttime obs
199    
200    ! locals
201    integer, parameter        :: maxPointsSearch = 200000
202    real(4), parameter        :: solarZenithThreshold = 90.0 ! to distinguish day and night
203    type(kdtree2), pointer    :: tree => null() 
204    real(kdkind), allocatable :: positionArray(:,:)
205    type(kdtree2_result)      :: searchResults(maxPointsSearch)
206    real(kdkind)              :: refPosition(3)
207    real(kdkind)              :: lon_grd, lat_grd
208    real(pre_obsReal)         :: lon_obs, lat_obs
209    integer                   :: lonIndex, latIndex
210    integer                   :: bodyIndex, headerIndex, ierr, headerCounter, codtyp
211    integer                   :: localObsIndex
212    integer                   :: ndataFoundGridLoc(hco%ni, hco%nj)  ! kd-tree output: number of data found within the search radius for every grid point
213    integer                   :: ndataFoundGridGlob(hco%ni, hco%nj) ! to compute mpi_allreduce of ndataFoundGridLoc 
214    real(kdkind)              :: searchRadiusSquared
215    integer, allocatable      :: headerIndexes(:)
216    real(8)                   :: currentObs
217    character(len=50)         :: instrumentString 
218
219    countObsLoc = 0
220    
221    ! count local observations for the given instrument
222    if (trim(instrument) == 'insitu') then
223      instrumentString = trim(instrument)
224      do headerIndex = 1, obs_numheader(obsData)
225        codtyp = obs_headElem_i(obsData, obs_ity, headerIndex)
226        if (codtyp == codtyp_get_codtyp('shipnonauto') .or. &
227             codtyp == codtyp_get_codtyp('drifter')     .or. &
228             codtyp == codtyp_get_codtyp('ashipauto')) then
229          countObsLoc = countObsLoc + 1
230        end if
231      end do
232    else
233      instrumentString = trim(instrument)//' '//dayOrNight_opt
234      do headerIndex = 1, obs_numheader(obsData)
235        if (obs_elem_c(obsData, 'STID' , headerIndex) == trim(instrument)) then
236          if (trim(dayOrNight_opt) == 'day') then
237            if (obs_headElem_r(obsData, obs_sun, headerIndex) <  solarZenithThreshold) &
238              countObsLoc = countObsLoc + 1
239          else if (trim(dayOrNight_opt) == 'night') then
240            if (obs_headElem_r(obsData, obs_sun, headerIndex) >= solarZenithThreshold) &
241              countObsLoc = countObsLoc + 1
242          end if
243        end if
244      end do 
245    end if  
246    write(*,*) ''
247    write(*,"(a, i10, a)") 'sstb_getGriddedObs: found ', countObsLoc, ' '//trim(instrumentString)//' data'
248
249    call rpn_comm_allreduce(countObsLoc, countObsGlob, 1, "mpi_integer", "mpi_sum", "grid", ierr)
250
251    obsGrid(:, :) = 0.0d0
252    ndataFoundGridLoc(:,:) = 0
253    ndataFoundGridGlob(:,:) = 0
254    
255    POSITIVECOUNTOBSLOC: if (countObsLoc > 0) then
256
257      write(*,*)'sstb_getGriddedObs: define kd-tree using data positions...'    
258      allocate(positionArray(3, countObsLoc))
259      allocate(headerIndexes(countObsLoc))
260
261      headerCounter = 0
262      do headerIndex = 1, obs_numheader(obsData)
263        if (trim(instrument) == 'insitu') then
264          codtyp = obs_headElem_i(obsData, obs_ity, headerIndex)
265	  if (codtyp == codtyp_get_codtyp('shipnonauto') .or. &
266              codtyp == codtyp_get_codtyp('drifter')     .or. &
267              codtyp == codtyp_get_codtyp('ashipauto')) then
268            lon_obs = obs_headElem_r(obsData, obs_lon, headerIndex)
269            lat_obs = obs_headElem_r(obsData, obs_lat, headerIndex)
270            headerCounter = headerCounter + 1
271            positionArray(:, headerCounter) = kdtree2_3dPosition(lon_obs, lat_obs)
272            headerIndexes(headerCounter) = headerIndex
273          end if
274        else
275          if (obs_elem_c(obsData, 'STID' , headerIndex) == trim(instrument)) then
276            if (trim(dayOrNight_opt) == 'day') then
277              if (obs_headElem_r(obsData, obs_sun, headerIndex) <  solarZenithThreshold) then
278                lon_obs = obs_headElem_r(obsData, obs_lon, headerIndex)
279                lat_obs = obs_headElem_r(obsData, obs_lat, headerIndex)
280                headerCounter = headerCounter + 1
281                positionArray(:, headerCounter) = kdtree2_3dPosition(lon_obs, lat_obs)
282                headerIndexes(headerCounter) = headerIndex
283              end if
284            else if (trim(dayOrNight_opt) == 'night') then 
285              if (obs_headElem_r(obsData, obs_sun, headerIndex) >= solarZenithThreshold) then
286                lon_obs = obs_headElem_r(obsData, obs_lon, headerIndex)
287                lat_obs = obs_headElem_r(obsData, obs_lat, headerIndex)
288                headerCounter = headerCounter + 1
289                positionArray(:, headerCounter) = kdtree2_3dPosition(lon_obs, lat_obs)
290                headerIndexes(headerCounter) = headerIndex
291              end if
292            end if  
293          end if
294        end if 
295      end do
296    
297      nullify(tree)
298      tree => kdtree2_create(positionArray, sort=.true., rearrange=.true.)
299      
300      ! do the search
301      write(*, '(a,f5.1,a)') 'sstb_getGriddedObs: Collocation radius: ', searchRadius, ' km'
302      searchRadiusSquared = (1.1d0 * searchRadius * 1000.d0)**2 ! convert from km to m2
303
304      write(*,*) 'sstb_getGriddedObs: computing the sum of data values and '//&
305                 'their number within the collocation radius for every grid point...'
306      do latIndex = 1, hco%nj
307        do lonIndex = 1, hco%ni 
308
309          ! compute gridded obs for every open water point
310          OPENWATERPTS: if (openWater(lonIndex, latIndex)) then 
311        
312            lon_grd = real(hco % lon2d_4 (lonIndex, latIndex), 8)
313            lat_grd = real(hco % lat2d_4 (lonIndex, latIndex), 8)
314            refPosition(:) = kdtree2_3dPosition(lon_grd, lat_grd)
315            call kdtree2_r_nearest(tp = tree, qv = refPosition, r2 = searchRadiusSquared, &
316                                   nfound = ndataFoundGridLoc(lonIndex, latIndex), & 
317                                   nalloc = maxPointsSearch, results = searchResults)
318            if (ndataFoundGridLoc(lonIndex, latIndex) > maxPointsSearch) &
319            call utl_abort('sstb_getGriddedObs: the parameter maxPointsSearch must be increased')
320
321            if (ndataFoundGridLoc(lonIndex, latIndex) > 0) then
322              do localObsIndex = 1, ndataFoundGridLoc(lonIndex, latIndex)
323                bodyIndex  = obs_headElem_i(obsData, obs_rln, headerIndexes(searchResults(localObsIndex)%idx))
324                currentObs = obs_bodyElem_r(obsData, obs_var, bodyIndex)
325                obsGrid(lonIndex, latIndex) = obsGrid(lonIndex, latIndex) + currentObs 
326              end do
327	    end if
328	    
329	  end if OPENWATERPTS
330        end do
331      end do
332
333      deallocate(headerIndexes)
334      deallocate(positionArray)
335      
336    end if POSITIVECOUNTOBSLOC
337
338    POSITIVECOUNTOBSGLOB: if (countObsGlob > 0) then
339      write(*,*) 'sstb_getGriddedObs: computing average values for every grid point...'
340      do latIndex = 1, hco%nj
341        do lonIndex = 1, hco%ni 
342          ! summing the values over all mpi tasks and sending them back to all tasks preserving the order of summation
343          call mmpi_allreduce_sumreal8scalar(obsGrid(lonIndex, latIndex), "grid")
344          call rpn_comm_allreduce(ndataFoundGridLoc(lonIndex, latIndex), &
345                                  ndataFoundGridGlob(lonIndex, latIndex), 1, &
346                                  'mpi_integer', 'mpi_sum', 'grid', ierr)
347	  if (ndataFoundGridGlob(lonIndex, latIndex) > 0) then
348	    obsGrid(lonIndex, latIndex) = obsGrid(lonIndex, latIndex) / &
349                                          real(ndataFoundGridGlob(lonIndex, latIndex))
350	  else
351	    obsGrid(lonIndex, latIndex) = MPC_missingValue_R8
352	  end if
353          if (.not.openWater(lonIndex, latIndex)) obsGrid(lonIndex, latIndex) = MPC_missingValue_R8
354        end do
355      end do
356      call rpn_comm_barrier('GRID', ierr)
357      write(*,*) 'sstb_getGriddedObs: gridding for '//trim(instrumentString)//' data completed'
358    end if POSITIVECOUNTOBSGLOB
359
360  end subroutine sstb_getGriddedObs

364  !--------------------------------------------------------------------------
365  subroutine sstb_getGriddedBias(satelliteGrid, insituGrid, nobsGlob, nobsLoc, &
366                                 hco, vco,  mask, openWater, sensor, &
367                                 numberOpenWaterPoints, dayOrNight)
368    !
369    !:Purpose: compute the satellite SST data bias estimation field on a grid
370    !           
371    implicit none
372    
373    ! Arguments: 
374    real(8)         , intent(inout)          :: satelliteGrid(:,:)   ! gridded satellite data
375    real(8)         , intent(inout)          :: insituGrid(:,:)      ! gridded insitu data
376    integer         , intent(in)             :: nobsGlob             ! number of data on all procs 
377    integer         , intent(in)             :: nobsLoc              ! number of data on the current MPI proc 
378    type(struct_hco), intent(in)   , pointer :: hco                  ! horizontal grid structure
379    type(struct_vco), intent(in)   , pointer :: vco                  ! vertical grid structure
380    logical         , intent(in)             :: mask(:,:)            ! land-ocean mask
381    logical         , intent(in)             :: openWater(:,:)       ! open water points (.true.)
382    character(len=*), intent(in)             :: sensor               ! current sensor name
383    integer         , intent(in)             :: numberOpenWaterPoints! number of open water points to allocate kd-tree work arrays 
384    character(len=*), intent(in)             :: dayOrNight           ! look for daytime or nighttime obs
385
386    ! locals
387    real(4), parameter          :: solarZenithThreshold = 90.0       ! to distinguish day and night
388    type(kdtree2), pointer      :: tree => null() 
389    real(kdkind), allocatable   :: positionArray(:,:)
390    integer, parameter          :: maxPointsSearch = 200000    
391    type(kdtree2_result)        :: searchResults(maxPointsSearch)
392    real(kdkind)                :: refPosition(3)
393    real(kdkind)                :: lon_grd, lat_grd
394    integer                     :: lonIndex, latIndex
395    integer                     :: ierr
396    integer                     :: localIndex, indexCounter, localLonIndex, localLatIndex
397    integer                     :: numPointsFound
398    real(kdkind)                :: searchRadiusSquared
399    type(struct_gsv)            :: stateVector                      ! state vector containing bias estimation field
400    type(struct_gsv)            :: stateVector_searchRadius, stateVector_previous
401    type(struct_gsv)            :: stateVectorNobs, stateVectorWeight    
402    real(4), pointer            :: griddedBias_r4_ptr(:,:,:), searchRadius_ptr(:,:,:)
403    real(4), pointer            :: nobsField_r4_ptr(:,:,:), weightField_r4_ptr(:,:,:)
404    real(4), pointer            :: griddedBias_r4_previous_ptr(:,:,:)
405    integer, allocatable        :: gridPointIndexes(:,:)
406    real(8)                     :: weight, distance, correlation, lengthscale, difference, numberPoints
407    character(len=1)            :: extension
408    character(len=*), parameter :: outputFileName = './satellite_bias.fst'
409    character(len=*), parameter :: outputAuxFileName = './auxOutput.fst' ! auxiliary file name to store nobs^a and weight
410    
411    write(*,*) ''
412    write(*,*) 'sstb_getGriddedBias: computing bias for: '//trim(sensor)//' '//trim(dayOrNight)
413    write(*,*) 'sstb_getGriddedBias: the current processor contains ', nobsLoc, ' data out of ', nobsGlob  
414    
415    if (dayOrNight == 'day') then
416      extension = 'D'
417    else if (dayOrNight == 'night') then
418      extension = 'N'	  
419    else  
420      call utl_abort('sstb_getGriddedBias: wrong extension: '//trim(extension)) 
421    end if  
422    
423    POSITIVEOBSNUMBER: if (nobsLoc > 0) then
424
425      write(*,*) 'sstb_getGriddedBias: compute kd-tree with all the grid points...'
426      allocate(positionArray(3, numberOpenWaterPoints))
427      allocate(gridPointIndexes(2, numberOpenWaterPoints))
428    
429      call lfn_setup('FifthOrder')
430
431      indexCounter = 0
432      do latIndex = 1, hco%nj
433        do lonIndex = 1, hco%ni 
434          if (openWater(lonIndex, latIndex)) then
435            indexCounter = indexCounter + 1
436            lon_grd = real(hco%lon2d_4(lonIndex, latIndex), 8)
437            lat_grd = real(hco%lat2d_4(lonIndex, latIndex), 8)
438            positionArray(:, indexCounter) = kdtree2_3dPosition(lon_grd, lat_grd)
439            gridPointIndexes(1, indexCounter) = lonIndex
440            gridPointIndexes(2, indexCounter) = latIndex
441          end if  
442        end do
443      end do
444    
445      nullify(tree)
446      tree => kdtree2_create(positionArray, sort=.true., rearrange=.true.)
447 
448    end if POSITIVEOBSNUMBER 
449
450    ! get search radius field
451    call gsv_allocate(stateVector_searchRadius, 1, hco, vco, dataKind_opt = 4, &
452                      datestamp_opt = -1, mpi_local_opt = .true., varNames_opt = (/'TM'/))
453    call gio_readFromFile(stateVector_searchRadius, './searchRadius', 'RADIUS','A', &
454                          unitConversion_opt=.false., containsFullField_opt=.true.)
455    call gsv_getField(stateVector_searchRadius, searchRadius_ptr)
456
457    if (saveAuxFields) then
458      ! output nobs state vector
459      call gsv_allocate(stateVectorNobs, 1, hco, vco, dataKind_opt = 4, &
460                        datestamp_opt = tim_getDateStamp(), mpi_local_opt = .true., &
461                        varNames_opt = (/'TM'/))
462      ! pointer for nobs stateVector
463      call gsv_getField(stateVectorNobs, nobsField_r4_ptr)
464      ! output weight state vector
465      call gsv_allocate(stateVectorWeight, 1, hco, vco, dataKind_opt = 4, &
466                        datestamp_opt = tim_getDateStamp(), mpi_local_opt = .true., &
467                        varNames_opt = (/'TM'/))
468      ! pointer for weight stateVector
469      call gsv_getField(stateVectorWeight, weightField_r4_ptr)
470      nobsField_r4_ptr(:,:,1) = 0.0d0
471      weightField_r4_ptr(:,:,1) = 0.0d0
472    end if
473    
474    ! previous bias estimation
475    call gsv_allocate(stateVector_previous, 1, hco, vco, dataKind_opt = 4, &
476                      datestamp_opt = -1, mpi_local_opt = .true., varNames_opt = (/'TM'/))
477    call gio_readFromFile(stateVector_previous, './trlm_01', 'B_'//trim(sensor)//'_'//trim(extension), &
478                          ' ', unitConversion_opt=.false., containsFullField_opt=.true.)
479    call gsv_getField(stateVector_previous, griddedBias_r4_previous_ptr) 
480       
481    ! resulting bias estimation state vector
482    call gsv_allocate(stateVector, 1, hco, vco, dataKind_opt = 4, &
483                      datestamp_opt = tim_getDateStamp(), mpi_local_opt = .true., varNames_opt = (/'TM'/))
484    ! pointer for bias estimation stateVector
485    call gsv_getField(stateVector, griddedBias_r4_ptr)
486    
487    griddedBias_r4_ptr(:,:,1) = 0.0d0
488
489    if (nobsLoc > 0) then
490      write(*,*) 'sstb_getGriddedBias: do the search for '//trim(sensor)//' '//trim(dayOrNight)//'...'
491    else
492      write(*,*) 'sstb_getGriddedBias: no '//trim(sensor)//' '&
493                 //trim(dayOrNight)//' data on the current processor' 
494      write(*,*) 'sstb_getGriddedBias: previous estimation state will be used.'
495    end if
496
497    do latIndex = myLatBeg, myLatEnd
498      do lonIndex = myLonBeg, myLonEnd 
499
500        POSITIVEOBSNUMBER2: if (nobsLoc > 0) then
501    
502	  numberPoints = 0.0d0
503          LANDSEAMASK: if (mask(lonIndex, latIndex)) then
504            lon_grd = real(hco%lon2d_4(lonIndex, latIndex), 8)
505            lat_grd = real(hco%lat2d_4(lonIndex, latIndex), 8)
506            refPosition(:) = kdtree2_3dPosition(lon_grd, lat_grd)
507            ! convert from km to m2
508            searchRadiusSquared = (1.1d0 * searchRadius_ptr(lonIndex, latIndex, 1) * 1000.d0)**2
509            call kdtree2_r_nearest(tp = tree, qv = refPosition, &
510                                   r2 = searchRadiusSquared, &
511                                   nfound = numPointsFound, &
512                                   nalloc = maxPointsSearch, &
513                                   results = searchResults)
514            POSITIVENPTSFOUND: if (numPointsFound > 0) then
515              do localIndex = 1, numPointsFound
516                localLonIndex = gridPointIndexes(1, searchResults(localIndex)%idx)
517                localLatIndex = gridPointIndexes(2, searchResults(localIndex)%idx)
518                difference = satelliteGrid(localLonIndex, localLatIndex) - &
519                             insituGrid(localLonIndex, localLatIndex)
520                if (insituGrid   (localLonIndex, localLatIndex) /= MPC_missingValue_R8 .and. &
521                    satelliteGrid(localLonIndex, localLatIndex) /= MPC_missingValue_R8 .and. &
522                    abs(difference) < maxBias) then
523                  distance = sqrt(searchResults(localIndex)%dis)
524                  lengthscale = 1000.d0 * searchRadius_ptr(lonIndex, latIndex, 1)
525                  correlation = lfn_response(distance, lengthscale)
526                  griddedBias_r4_ptr(lonIndex, latIndex, 1) = griddedBias_r4_ptr(lonIndex, latIndex, 1) + &
527                                                              correlation * difference 
528                  numberPoints = numberPoints + correlation
529                end if
530              end do
531    
532              if (numberPoints > 0.0d0) &
533              griddedBias_r4_ptr(lonIndex, latIndex, 1) = griddedBias_r4_ptr(lonIndex, latIndex, 1) / numberPoints
534            end if POSITIVENPTSFOUND   
535          end if LANDSEAMASK
536
537          weight = numberPoints / (numberPoints + numberPointsBG)
538          if (weight < weightMin) weight = weightMin
539          if (weight > weightMax) weight = weightMax
540       
541	  if (saveAuxFields) then
542            weightField_r4_ptr(lonIndex, latIndex, 1) = weight
543	    nobsField_r4_ptr(lonIndex, latIndex, 1) = numberPoints
544	  end if
545	  
546	  ! computation of the bias:
547          griddedBias_r4_ptr(lonIndex, latIndex, 1) = (1.0d0 - weight) * bgTermZeroBias * &
548                                                      griddedBias_r4_previous_ptr(lonIndex, latIndex, 1) + &
549                                                      weight * griddedBias_r4_ptr(lonIndex, latIndex, 1)
550        else ! no data on the current processor   
551       
552	  ! the bias estimation on the current processor is the estimation from previous state:
553          griddedBias_r4_ptr(lonIndex, latIndex, 1) = griddedBias_r4_previous_ptr(lonIndex, latIndex, 1)
554
555        end if POSITIVEOBSNUMBER2
556      end do
557    end do
558    
559    call rpn_comm_barrier('GRID', ierr)
560    call gio_writeToFile(stateVector, outputFileName, 'B_'//trim(sensor)//'_'//trim(extension))
561
562    if (nobsLoc > 0) then
563      deallocate(gridPointIndexes)
564      deallocate(positionArray)
565    end if
566
567    call gsv_deallocate(stateVector_searchRadius)
568    call gsv_deallocate(stateVector_previous)
569    call gsv_deallocate(stateVector)
570    if (saveAuxFields) then
571      call gio_writeToFile(stateVectorNobs, outputAuxFileName, 'N_'//trim(sensor)//'_'//trim(extension))
572      call gio_writeToFile(stateVectorWeight, outputAuxFileName, 'W_'//trim(sensor)//'_'//trim(extension))
573      call gsv_deallocate(stateVectorNobs)
574      call gsv_deallocate(stateVectorWeight)
575    end if
576    
577    write(*,*) 'sstb_getGriddedBias: completed for: '//trim(sensor)//' '//trim(dayOrNight)
578
579  end subroutine sstb_getGriddedBias

584  subroutine sstb_getBiasCorrection(stateVector, column, obsData, hco, sensor, dayOrNight)
585    !
586    !:Purpose: To compute bias correction and put it into obsSpace data. 
587    !          Columns from input field are interpolated to obs location
588    !     
589    implicit none
590    
591    ! arguments
592    type(struct_gsv)       , intent(inout)       :: stateVector       ! state vector containing bias estimation field    
593    type(struct_columnData), intent(in)          :: column            ! column data
594    type(struct_obs)       , intent(inout)       :: obsData           ! obsSpaceData
595    type(struct_hco)       , intent(in), pointer :: hco               ! horizontal grid
596    character(len=*)       , intent(in)          :: sensor            ! current sensor name
597    character(len=*)       , intent(in)          :: dayOrNight        ! look for daytime or nighttime obs
598
599    ! locals
600    real(4), parameter :: solarZenithThreshold = 90.0 ! to distinguish day and night
601    integer            :: bodyIndex, headerIndex
602    real(8)            :: currentObs
603
604    write(*,*)
605    write(*,*) 'sstb_getBiasCorrection: computing bias correction for ', trim(sensor), ' ', trim(dayOrNight), 'time ************'
606
607    call s2c_nl(stateVector, obsData, column, hco, timeInterpType = timeInterpType_nl, &
608                moveObsAtPole_opt = .true., numObsBatches_opt = numObsBatches, dealloc_opt = .false.)
609
610    do headerIndex = 1, obs_numheader(obsData)
611      
612      if (obs_elem_c(obsData, 'STID' , headerIndex) == trim(sensor)) then
613        
614        bodyIndex  = obs_headElem_i(obsData, obs_rln, headerIndex)
615	currentObs = obs_bodyElem_r(obsData, obs_var, bodyIndex)
616	
617	if (trim(dayOrNight) == 'day') then
618          if (obs_headElem_r(obsData, obs_sun, headerIndex) <  solarZenithThreshold) then
619            call obs_bodySet_r(obsData, obs_bcor, bodyIndex, col_getElem(column, 1, headerIndex, 'TM'))
620            call obs_bodySet_r(obsData, obs_var , bodyIndex, currentObs - col_getElem(column, 1, headerIndex, 'TM'))
621	  end if  
622	else if (trim(dayOrNight) == 'night') then
623          if (obs_headElem_r(obsData, obs_sun, headerIndex) >= solarZenithThreshold) then
624            call obs_bodySet_r(obsData, obs_bcor, bodyIndex, col_getElem(column, 1, headerIndex, 'TM'))
625            call obs_bodySet_r(obsData, obs_var , bodyIndex, currentObs - col_getElem(column, 1, headerIndex, 'TM'))
626          end if
627	end if
628      end if
629      
630    end do 
631 
632    write(*,*) 'sstb_getBiasCorrection: END'
633
634  end subroutine sstb_getBiasCorrection

637  ! readNml (private subroutine)
638  !--------------------------------------------------------------------------
639  subroutine readNml()
640    !
641    !:Purpose: Read the namelist `namSSTbiasEstimate`
642    !
643    implicit none
644
645    ! locals
646    integer :: ierr, nulnam, sensorIndex
647    
648    namelist /namSSTbiasEstimate/ searchRadius, maxBias, iceFractionThreshold, numberPointsBG, &
649                                  timeInterpType_nl, numObsBatches, numberSensors, sensorList, &
650                                  weightMin, weightMax, saveAuxFields, bgTermZeroBias
651
652    ! Setting default namelist variable values
653    searchRadius = 10.            
654    maxBias = 1.                  
655    iceFractionThreshold   = 0.6 
656    numberSensors = MPC_missingValue_INT
657    numberPointsBG = 0            
658    timeInterpType_nl = 'NEAREST'
659    numObsBatches = 20
660    sensorList(:) = ''
661    weightMin = 0.0
662    weightMax = 1.0
663    saveAuxFields = .False.
664    bgTermZeroBias = 1.0
665    
666    ! Read the namelist
667    nulnam = 0
668    ierr = fnom( nulnam, './flnml', 'FTN+SEQ+R/O', 0 )
669    read(nulnam, nml = namSSTbiasEstimate, iostat = ierr )
670    if (ierr /= 0) call utl_abort('readNml (sstb): Error reading namelist')
671    if (mmpi_myid == 0) write(*, nml = namSSTbiasEstimate )
672    ierr = fclos( nulnam )
673
674    if (numberSensors /= MPC_missingValue_INT) then
675      call utl_abort('readNml (sstb): check namSSTbiasEstimate namelist section: numberSensors should be removed')
676    end if
677
678    numberSensors = 0
679    do sensorIndex = 1, maxNumberSensors
680      if (trim(sensorList(sensorIndex))=='') exit
681      numberSensors = numberSensors + 1
682    end do
683    if (numberSensors == 0) call utl_abort('readNml (sstb): check namSSTbiasEstimate namelist section: empty sensorList')
684
685    if (mmpi_myid == 0) then
686      write(*,*)
687      write(*,*) 'readNml (sstb): sensors to treat: '
688      do sensorIndex = 1, numberSensors
689        write(*,*) 'readNml (sstb): sensor index: ', sensorIndex, ', sensor: ', sensorList( sensorIndex )
690      end do
691      write(*,*) 'readNml (sstb): interpolation type: ', timeInterpType_nl
692      write(*,*) 'readNml (sstb): number obs batches: ', numObsBatches
693      write(*,*) 'readNml (sstb): weight limits for current bias estimate: ', weightMin, weightMax
694    end if
695
696  end subroutine readNml

699  ! sstb_applySatelliteSSTBiasCorrection
700  !--------------------------------------------------------------------------
701  subroutine sstb_applySatelliteSSTBiasCorrection(obsData, hco, vco, column)
702    !
703    !:Purpose: To apply bias satellite SST data bias correction and put it into obsSpace data. 
704    !          Columns from input field are interpolated to obs location
705    !     
706    implicit none
707    
708    ! arguments
709    type(struct_obs)       , intent(inout)          :: obsData ! obsSpaceData
710    type(struct_hco)       , intent(inout), pointer :: hco     ! horizontal grid structure
711    type(struct_vco)       , intent(in)   , pointer :: vco     ! vertical grid structure
712    type(struct_columnData), intent(in)             :: column  ! column data 
713
714    ! locals
715    type(struct_gsv)            :: stateVector
716    integer         , parameter :: numberProducts = 2            ! day and night
717    character(len=*), parameter :: listProducts(numberProducts)= (/'day', 'night'/) ! day and night biases
718    integer                     :: sensorIndex, productIndex
719    character(len=1)            :: extension
720    character(len=*), parameter :: biasFileName = './satellite_bias.fst'
721
722    ! read the namelist
723    call readNml()
724
725    ! allocate state vector for bias estimation field
726    call gsv_allocate(stateVector, 1, hco, vco, dataKind_opt = 4, hInterpolateDegree_opt = 'LINEAR', &
727                      datestamp_opt = -1, mpi_local_opt = .true., varNames_opt = (/'TM'/))
728
729    do sensorIndex = 1, numberSensors 
730      do productIndex = 1, numberProducts
731      
732        if (trim(listProducts(productIndex)) == 'day') then
733          extension = 'D'
734        else if (trim(listProducts(productIndex)) == 'night') then
735          extension = 'N'
736        else
737          call utl_abort('sstb_applySatelliteSSTBiasCorrection: wrong extension: '//trim(extension)) 
738        end if
739	
740        call gio_readFromFile(stateVector, biasFileName, 'B_'//trim(sensorList(sensorIndex))//'_'//trim(extension), &
741                              ' ', unitConversion_opt = .false., containsFullField_opt = .true.)
742        call sstb_getBiasCorrection(stateVector, column, obsData, hco, trim(sensorList(sensorIndex)), &
743                                    trim(listProducts(productIndex)))
744      end do
745    end do
746    				    
747    call gsv_deallocate(stateVector)
748			    
749  end subroutine sstb_applySatelliteSSTBiasCorrection

752  ! sstb_getBiasFromPreviousState
753  !--------------------------------------------------------------------------
754  subroutine sstb_getBiasFromPreviousState(hco, vco, sensor, dayOrNight)
755    !
756    !:Purpose: to get a satellite SST data bias estimate from the previous state if data is missing.
757    !          or there are no insitu data for the current dateStamp,
758    !          hence, unable to compute bias estimates for the satellite data.
759 
760    implicit none
761    
762    ! arguments
763    type(struct_hco), intent(inout), pointer :: hco        ! horizontal grid structure
764    type(struct_vco), intent(in)   , pointer :: vco        ! vertical grid structure
765    character(len=*), intent(in)             :: sensor     ! sensor name
766    character(len=*), intent(in)             :: dayOrNight ! look for daytime or nighttime bias estimation
767
768    ! locals
769    type(struct_gsv)            :: stateVector                  ! state vector containing current  bias estimation field
770    type(struct_gsv)            :: stateVector_previous         ! state vector containing previous bias estimation field
771    real(4), pointer            :: griddedBias_r4_ptr(:, :, :)
772    real(4), pointer            :: griddedBias_r4_previous_ptr(:, :, :)
773    character(len=1)            :: extension
774    character(len=*), parameter :: outputFileName = './satellite_bias.fst'
775
776    write(*,*) ''
777    write(*,*) 'sstb_getBiasFromPreviousState: for ', trim(sensor), ' ', trim(dayOrNight), ' data...'
778
779    if (trim(dayOrNight) == 'day') then
780      extension = 'D'
781    else if (trim(dayOrNight) == 'night') then
782      extension = 'N'	  
783    else  
784      call utl_abort('sstb_getBiasFromPreviousState: wrong extension: '//trim(extension))
785    end if
786
787    ! read previous bias estimation
788    call gsv_allocate(stateVector_previous, 1, hco, vco, dataKind_opt = 4, &
789                       datestamp_opt = -1, mpi_local_opt = .true., varNames_opt = (/'TM'/))
790    call gio_readFromFile(stateVector_previous, './trlm_01', 'B_'//trim(sensor)//'_'//trim(extension), &
791                          ' ', unitConversion_opt=.false., containsFullField_opt=.true.)
792    call gsv_getField(stateVector_previous, griddedBias_r4_previous_ptr) 
793
794    ! resulting bias estimation state vector
795    call gsv_allocate(stateVector, 1, hco, vco, dataKind_opt = 4, &
796                      datestamp_opt = tim_getDateStamp(), mpi_local_opt = .true., varNames_opt = (/'TM'/))
797    ! pointer for bias estimation stateVector
798    call gsv_getField(stateVector, griddedBias_r4_ptr)
799        
800    griddedBias_r4_ptr(:, :, :) = griddedBias_r4_previous_ptr(:, :, :)
801    call gio_writeToFile(stateVector, outputFileName, 'B_'//trim(sensor)//'_'//trim(extension))
802
803    call gsv_deallocate(stateVector)
804    call gsv_deallocate(stateVector_previous)
805
806    write(*,*) 'sstb_getBiasFromPreviousState: done ', trim(sensor), ' ', trim(dayOrNight), ' data.'
807
808  end subroutine sstb_getBiasFromPreviousState
