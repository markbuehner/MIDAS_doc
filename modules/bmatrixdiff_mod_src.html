 75  !--------------------------------------------------------------------------
 76  ! bdiff_setup
 77  !--------------------------------------------------------------------------
 78  subroutine bdiff_setup ( hco_in, vco_in, cvDim_out, mode_opt )
 79    !
 80    !:Purpose: Setup the diffusion B matrix
 81    !
 82    implicit none
 83
 84    ! Arguments:
 85    type(struct_hco), intent(inout), pointer  :: hco_in
 86    type(struct_vco), intent(inout), pointer  :: vco_in
 87    integer         , intent(out)             :: cvDim_out
 88    character(len=*), intent(in)   , optional :: mode_opt
 89
 90    ! Locals:
 91    character(len=15)         :: bdiff_mode
 92    type(struct_vco), pointer :: vco_anl
 93    integer                   :: nulnam, ierr, fnom, fclos
 94    integer                   :: variableIndex, latIndex, latIndexIgnore
 95    real(8)                   :: maxDistance
 96    real(8), allocatable      :: distance(:)
 97    character(len=*), parameter :: myName = 'bdiff_setup'
 98
 99    ! namelist variables
100    real    :: corr_len( maxNumVars )  ! Horizontal correlation length scale (km)
101    real    :: stab( maxNumVars )      ! Stability criteria (definitely < 0.5)
102    integer :: nsamp(maxNumVars)       ! Number of samples in the estimation of the normalization factors by randomization
103    real(8) :: latIgnoreFraction       ! Relative zonal grid spacing limit where lats near each numerical pole are ignored
104    logical :: useImplicit(maxNumVars) ! choose to use implicit formulation of diffusion operator (.true.) or explicit version (.false.)
105    
106    NAMELIST /NAMBDIFF/ corr_len, stab, nsamp, useImplicit, scaleFactor, stddevMode, homogeneous_std, latIgnoreFraction
107
108    call utl_tmg_start(65,'----B_DIFF_Setup')
109    if(mmpi_myid == 0) write(*,*) myName//': starting'
110    if(mmpi_myid == 0) write(*,*) myName//': Memory Used: ',get_max_rss()/1024,'Mb'
111
112    ! default values for namelist variables
113    corr_len(:) = 10.0
114    stab(:)     = 0.2
115    nsamp(:)    = 10000
116    useImplicit(:) = .false.
117    scaleFactor(:) = 0.0d0
118    stddevMode  = 'GD2D'
119    homogeneous_std(:) = -1.0d0
120    latIgnoreFraction = 1.0d6 ! large value so that nothing is ignored by default
121
122    if ( .not. utl_isNamelistPresent('NAMBDIFF','./flnml') ) then
123      if ( mmpi_myid == 0 ) then
124        write(*,*) 'bdiff_setup: nambdiff is missing in the namelist.'
125        write(*,*) '             The default values will be taken.'
126      end if
127    else
128      nulnam = 0
129      ierr = fnom( nulnam,'./flnml','FTN+SEQ+R/O',0)
130      read( nulnam, nml = nambdiff, iostat = ierr )
131      if ( ierr /= 0 ) call utl_abort( myName//': Error reading namelist')
132      if ( mmpi_myid == 0) write( *, nml = nambdiff )
133      ierr = fclos( nulnam )
134    end if
135
136    if ( sum(scaleFactor(:) ) == 0.0d0 ) then
137      if( mmpi_myid == 0) write(*,*) myName//': scaleFactor=0, skipping rest of setup'
138      cvdim_out = 0
139      call utl_tmg_stop(65)
140      return
141    end if
142
143    if ( present( mode_opt ) ) then
144      if ( trim( mode_opt ) == 'Analysis' .or. trim( mode_opt ) == 'BackgroundCheck' ) then
145        bdiff_mode = trim( mode_opt )
146        if( mmpi_myid == 0 ) write(*,*)
147        if( mmpi_myid == 0 ) write(*,*) myName//': Mode activated = ', trim(bdiff_mode)
148      else
149        write(*,*)
150        write(*,*)  myName//'mode = ', trim(mode_opt)
151        call utl_abort( myName//': unknown mode' )
152      end if
153    else
154      bdiff_mode = 'Analysis'
155      if( mmpi_myid == 0 ) write(*,*)
156      if( mmpi_myid == 0 ) write(*,*) myName//': analysis mode activated (by default)'
157    end if
158
159    vco_anl => vco_in
160    if (vco_anl%Vcode  /= 5002 .and. vco_anl%Vcode /= 5005 .and. vco_anl%Vcode /= 0 ) then
161      write(*,*)  myName//'vco_anl%Vcode = ', vco_anl%Vcode
162      call utl_abort( myName//': unknown vertical coordinate type!')
163    end if
164
165    numvar2d = 0
166
167    allocate( bdiff_varNameList( vnl_numvarmax ) )
168    bdiff_varNameList( : )=''
169    allocate( nsposit( vnl_numvarmax + 1 ) )
170    nsposit(1) = 1
171
172    ! Find the 2D variables (within NAMSTATE namelist)
173
174    if ( gsv_varExist( varName = 'GL  ' )) then
175
176      numvar2d = numvar2d + 1
177      nsposit( numvar2d + 1 ) = nsposit( numvar2d ) + 1
178      bdiff_varNameList( numvar2d ) = 'GL  '
179
180    end if
181   
182    if ( gsv_varExist( varName = 'TM  ')) then
183
184      numvar2d = numvar2d + 1
185      nsposit( numvar2d + 1 ) = nsposit( numvar2d ) + 1
186      bdiff_varNameList( numvar2d ) = 'TM  '
187
188    end if
189
190    if ( numvar2d == 0) then
191       
192      if ( mmpi_myid == 0) then
193        write(*,*) myName//': Bdiff matrix not produced.'
194        write(*,*) myName//': END'
195      end if
196      call utl_tmg_stop(65)
197      cvdim_out = 0
198      return
199      
200    else if (mmpi_myid == 0) then
201
202      write(*,*) myName//': number of 2D variables', numvar2d, bdiff_varNameList( 1 : numvar2d )
203
204    end if
205
206    if ( trim(bdiff_mode) == 'BackgroundCheck' ) then
207      cvDim_out = 9999 ! Dummy value > 0 to indicate to the background check (s/r ose_compute_HBHT_ensemble) 
208                       ! that Diff is used
209      call utl_tmg_stop(65)
210      return
211    end if
212
213    ! Assumes the input 'scalefactor' is a scaling factor of the variances.
214
215    do variableIndex = 1, numvar2d
216      if ( scaleFactor( variableIndex ) > 0.0d0 ) then 
217        scaleFactor_sigma( variableIndex ) = sqrt( scaleFactor( variableIndex ) )
218      else
219        scaleFactor_sigma( variableIndex ) = 0.0d0
220      end if
221    end do
222
223    ni_l = hco_in%ni
224    nj_l = hco_in%nj
225
226    ! Compute latIndexIgnore from latIgnoreFraction
227    if (latIgnoreFraction < 1.0) then
228      allocate(distance(nj_l))
229      do latIndex = 1, nj_l
230        distance(latIndex) = &
231             phf_calcDistance(real(hco_in%lat2d_4(ni_l/2,latIndex),8), real(hco_in%lon2d_4((ni_l/2)+1,latIndex),8), &
232                              real(hco_in%lat2d_4(ni_l/2,latIndex),8), real(hco_in%lon2d_4((ni_l/2)  ,latIndex),8))
233      end do
234      maxDistance = maxval(distance(:))
235      if (mmpi_myid==0) write(*,*) myName//': maxDistance = ', maxDistance
236      latIndexIgnore = 0
237      latLoop: do latIndex = 1, nj_l
238        if (distance(latIndex)/maxDistance > latIgnoreFraction) then
239          if (mmpi_myid==0) then
240            write(*,*) '   latIndex-1, distance, fraction = ', latIndex-1, distance(latIndex-1), distance(latIndex-1)/maxDistance 
241            write(*,*) '***latIndex  , distance, fraction = ', latIndex  , distance(latIndex  ), distance(latIndex  )/maxDistance 
242            write(*,*) '   latIndex+1, distance, fraction = ', latIndex+1, distance(latIndex+1), distance(latIndex+1)/maxDistance
243          end if
244          latIndexIgnore = latIndex
245          exit latLoop
246        end if
247      end do latLoop
248      deallocate(distance)
249    else
250      latIndexIgnore = 0
251    end if
252    if (mmpi_myid==0) write(*,*) myName//': latIndexIgnore = ', latIndexIgnore
253
254    allocate( diffID( numvar2d ) )
255    do variableIndex = 1, numvar2d
256      write(*,*) myName//': setup the diffusion operator for the variable ', bdiff_varNameList( variableIndex ) 
257      diffID( variableIndex ) = diff_setup ( variableIndex, bdiff_varNameList(1:numvar2d), hco_in, vco_in, corr_len( variableIndex ), &
258                                             stab( variableIndex ), nsamp( variableIndex ), useImplicit( variableIndex ), &
259                                             latIndexIgnore )
260    end do
261
262    call mmpi_setup_latbands( nj_l, latPerPE, latPerPEmax, myLatBeg, myLatEnd )
263    call mmpi_setup_lonbands( ni_l, lonPerPE, lonPerPEmax, myLonBeg, myLonEnd )
264
265    ! compute mpilocal control vector size
266    cvDim_mpilocal = lonPerPE * latPerPE * numvar2d
267    cvDim_out = cvDim_mpilocal
268
269    ! also compute mpiglobal control vector dimension
270    call rpn_comm_allreduce( cvDim_mpilocal, cvDim_mpiglobal, 1, "mpi_integer", "mpi_sum", "GRID", ierr )
271
272    allocate(stddev(myLonBeg:myLonEnd, myLatBeg:myLatEnd, numvar2d))
273
274    call bdiff_rdstats( hco_in, vco_in )
275
276    if(mmpi_myid == 0) write(*,*) myName//': Memory Used: ',get_max_rss()/1024,'Mb'
277
278    if(mmpi_myid == 0) write(*,*) myName//': END'
279
280    initialized = .true.
281
282    call utl_tmg_stop(65)
283
284  end subroutine bdiff_setup

312  subroutine bdiff_rdstats( hco_in, vco_in )
313    !
314    !:Purpose: To read background-error stats file.
315    !
316    implicit none
317    
318    ! Arguments:
319    type(struct_hco), pointer :: hco_in
320    type(struct_vco), pointer :: vco_in
321    
322    ! Locals:
323    integer :: ierr, nmax, fnom, fstouv, fstfrm, fclos
324    integer :: variableIndex
325    logical :: lExists
326    character(len=12) :: bFileName1 = './bgstddev'
327    character(len=8)  :: bFileName2 = './bgcov'
328    character(len=*), parameter :: myName = 'bdiff_rdstats'
329    
330    write(*,*) myName//': stddevMode is ', stddevMode
331    write(*,*) myName//': Number of 2D variables', numvar2d, bdiff_varNameList( 1 : numvar2d )
332    
333    if ( stddevMode == 'GD2D' ) then
334
335      inquire( file = bFileName1, exist = lExists )
336       
337      if ( lexists ) then
338        ierr = fnom(nulbgst, bFileName1, 'RND+OLD+R/O', 0)
339        if ( ierr == 0 ) then
340          nmax = fstouv(nulbgst, 'RND+OLD')
341        else
342          call utl_abort( myName//': error opening file '//trim(bFileName1))
343        end if
344      else
345        ! Assume background-error stats in file bgcov. 
346        inquire( file = bFileName2, exist = lExists )  
347        if ( lexists ) then 
348          ierr = fnom( nulbgst, bFileName2, 'RND+OLD+R/O', 0 )
349          if ( ierr == 0 ) then
350            nmax = fstouv(nulbgst, 'RND+OLD')
351          else
352            call utl_abort( myName//': error opening file '//trim(bFileName2) )
353          end if
354        else
355          call utl_abort( myName//': no background error statistics file found!!' )
356        end if
357      end if
358
359      call bdiff_readBGstdField( hco_in, vco_in )
360
361      ierr = fstfrm(nulbgst)
362      ierr = fclos(nulbgst)
363
364    else if ( stddevMode == 'HOMO' ) then
365
366      do variableIndex = 1, numvar2d
367        write(*,*) myName//': stdev = ', homogeneous_std( variableIndex ), ' for variable ', bdiff_varNameList( variableIndex )    
368        stddev( :, :, variableIndex) = homogeneous_std( variableIndex )
369      end do
370       
371    else
372
373      call utl_abort( myName//': unknown stddevMode: '//trim(stddevMode) )
374
375    end if
376
377    call bdiff_scalestd
378
379    write(*,*) myName//': END '
380    
381  end subroutine bdiff_rdstats

386  subroutine bdiff_readBGstdField( hco_in, vco_in )
387    !
388    !:Purpose: to read 2D background error standard deviation field
389    !          stored on Z, U or G grid and interpolate it to the analysis grid
390    !
391    implicit none
392    
393    ! Arguments:
394    type(struct_hco), pointer, intent(in) :: hco_in 
395    type(struct_vco), pointer, intent(in) :: vco_in
396
397    ! Locals:
398    integer                     :: variableIndex, ierr
399    type(struct_gsv)            :: statevector
400    real(4), pointer            :: field3D_r4_ptr(:,:,:)
401    real(8)                     :: minStddev, maxStddev
402    character(len=*), parameter :: myName = 'bdiff_readBGstdField'
403
404    write(*,*) myName//': Reading 2D fields from ./bgstddev...'
405    write(*,*) myName//': Number of 2D variables', numvar2d, bdiff_varNameList( 1 : numvar2d )
406
407    call gsv_allocate( statevector, 1, hco_in, vco_in, dateStamp_opt=-1, &
408                       dataKind_opt=4, mpi_local_opt=.true., &
409                       hInterpolateDegree_opt='LINEAR', &
410                       varNames_opt=bdiff_varNameList(1:numvar2d) )
411    call gsv_zero( statevector )
412    call gio_readFromFile(statevector, './bgstddev', 'STDDEV', ' ', unitConversion_opt = .false. )
413
414    do variableIndex = 1, numvar2d
415      call gsv_getField( statevector, field3D_r4_ptr, bdiff_varNameList( variableIndex ) )
416      stddev( :, :, variableIndex ) = dble( field3D_r4_ptr( :, :, 1 ) )
417      if (mmpi_nprocs > 1) then
418        call rpn_comm_allreduce(minval(stddev(:,:,variableIndex)),minStddev,1,'mpi_real8','mpi_min','GRID',ierr)
419        call rpn_comm_allreduce(maxval(stddev(:,:,variableIndex)),maxStddev,1,'mpi_real8','mpi_max','GRID',ierr)
420      else
421        minStddev = minval(stddev(:,:,variableIndex))
422        maxStddev = maxval(stddev(:,:,variableIndex))
423      end if
424      write(*,*) myName//': variable ', bdiff_varNameList( variableIndex ),' min/max: ', &
425                 minStddev, maxStddev
426    end do
427   
428    call gsv_deallocate( statevector )
429
430  end subroutine bdiff_readBGstdField

458  subroutine bdiff_bSqrt( controlVector_in, statevector )
459    !
460    !:Purpose: Apply the sqrt of the B matrix 
461    !    
462    implicit none
463
464    ! Arguments:
465    real(8),          intent(in)    :: controlVector_in( cvDim_mpilocal )
466    type(struct_gsv), intent(inout) :: statevector
467
468    ! Locals:
469    real(8) :: gd_in( myLonBeg:myLonEnd, myLatBeg:myLatEnd, numvar2d)
470    real(8) :: gd_out(myLonBeg:myLonEnd, myLatBeg:myLatEnd, numvar2d)
471
472    integer :: variableIndex
473    character(len=*), parameter :: myName = 'bdiff_bSqrt'
474    
475    if( .not. initialized) then
476      if( mmpi_myid == 0 ) write(*,*) myName//': bMatrixDIFF not initialized'
477      return
478    end if
479
480    if(mmpi_myid == 0) write(*,*) myName//': starting'
481
482    call bdiff_cain( controlVector_in, gd_in )
483
484    do variableIndex = 1, numvar2d
485
486      ! Apply square root of the diffusion operator.
487      call diff_Csqrt( diffID( variableIndex ), gd_in( :, :, variableIndex ), gd_out( :, :, variableIndex ) )
488
489      ! Multiply by the diagonal matrix of background error standard deviations.
490      gd_out( :, :, variableIndex ) = gd_out( :, :, variableIndex ) * stddev( :, :, variableIndex )
491
492    end do
493
494    call bdiff_copyToStatevector( statevector, gd_out )
495
496    if(mmpi_myid == 0) write(*,*) myName//': Memory Used: ',get_max_rss()/1024,'Mb'
497    if(mmpi_myid == 0) write(*,*) myName//': done'
498
499  end subroutine bdiff_bSqrt

502  ! bdiff_bSqrtAd
503  !--------------------------------------------------------------------------
504  subroutine bdiff_bSqrtAd( statevector, controlVector_out )
505    !
506    !:Purpose: Apply the adjoint (i.e. transpose) of the sqrt of the B matrix
507    !    
508    implicit none
509
510    ! Arguments:
511    type(struct_gsv), intent(in)  :: statevector
512    real(8),          intent(out) :: controlVector_out(cvDim_mpilocal)
513
514    ! Locals:
515    real(8) :: gd_in( myLonBeg:myLonEnd, myLatBeg:myLatEnd, numvar2d)
516    real(8) :: gd_out(myLonBeg:myLonEnd, myLatBeg:myLatEnd, numvar2d)
517
518    integer :: variableIndex
519    character(len=*), parameter :: myName = 'bdiff_bSqrtAd'
520
521    if ( .not. initialized ) then
522      if ( mmpi_myid == 0 ) write(*,*) myName//': bMatrixDIFF not initialized'
523      return
524    end if
525
526    if(mmpi_myid == 0) write(*,*)  myName//': starting'
527
528    call bdiff_copyFromStatevector( statevector, gd_in )
529
530    do variableIndex = 1, numvar2d
531
532      ! Multiply by the diagonal matrix of background-error standard deviations.
533      gd_in( :, :, variableIndex ) = gd_in( :, :, variableIndex ) * stddev( :, :, variableIndex )
534
535      ! Apply the adjoint of the square root of the diffusion operator.
536      call diff_Csqrtadj( diffID( variableIndex ), gd_in( :, :, variableIndex ), gd_out( :, :, variableIndex) )
537
538    end do
539
540    call bdiff_cainad(gd_out, controlVector_out)
541
542    if ( mmpi_myid == 0) write(*,*) myName//': Memory Used: ', get_max_rss()/1024,'Mb'
543    if ( mmpi_myid == 0) write(*,*) myNAme//': done'
544
545  end subroutine bdiff_bSqrtAd

547  !--------------------------------------------------------------------------
548  ! bdiff_copyToStatevector
549  !--------------------------------------------------------------------------
550  subroutine bdiff_copyToStatevector ( statevector, gd )
551    !
552    !:Purpose: Copy the working array to a statevector object 
553    !    
554    implicit none
555
556    ! Arguments:
557    real(8),          intent(in)    :: gd(myLonBeg:myLonEnd, myLatBeg:myLatEnd, numvar2d)
558    type(struct_gsv), intent(inout) :: statevector
559
560    ! Locals:
561    integer :: jlon, jlev, jlev2, jlat, variableIndex, ilev1, ilev2
562    real(4), pointer :: field_r4(:,:,:)
563    real(8), pointer :: field_r8(:,:,:)
564    character(len=*), parameter :: myName = 'bdiff_copyToStatevector'
565    
566    do variableIndex = 1, numvar2d
567
568      ilev1 = nsposit( variableIndex )
569      ilev2 = nsposit( variableIndex + 1 ) - 1 
570
571      if ( mmpi_myid == 0) write(*,*) myName//': ',bdiff_varNameList( variableIndex )
572      
573      if (gsv_getDataKind(statevector) == 4) then
574        call gsv_getField( statevector, field_r4, bdiff_varNameList( variableIndex ) )
575        do jlev = ilev1, ilev2
576          jlev2 = jlev-ilev1+1
577          do jlat = myLatBeg, myLatEnd
578            do jlon = myLonBeg, myLonEnd
579              field_r4(jlon,jlat,jlev2) = gd(jlon,jlat,jlev)
580            end do
581          end do
582        end do
583      else
584        call gsv_getField( statevector, field_r8, bdiff_varNameList( variableIndex ) )
585        do jlev = ilev1, ilev2
586          jlev2 = jlev-ilev1+1
587          do jlat = myLatBeg, myLatEnd
588            do jlon = myLonBeg, myLonEnd
589              field_r8(jlon,jlat,jlev2) = gd(jlon,jlat,jlev)
590            end do
591          end do
592        end do
593      end if
594
595    end do

654    !:Purpose: Extract the subset of the global control vector needed for local MPI task

658    ! Arguments:

664    real(8), allocatable :: gd_mpiGlobal(:,:,:)
665
666    allocate(gd_mpiGlobal(ni_l,nj_l,numvar2d))

669    jn = 0
670    if (mmpi_myid == 0) then
671      do jlev = 1, numvar2d
672        do jlat = 1, nj_l
673          do jlon = 1, ni_l
674            jn = jn + 1
675            gd_mpiGlobal( jlon, jlat, jlev ) = cv_mpiglobal( jn )
676          end do
677        end do
678      end do
679    end if 
680    call rpn_comm_bcast(gd_mpiGlobal, size(gd_mpiGlobal), 'MPI_REAL8', 0, 'GRID', ierr)
681    
682    jn = 0
683    do jlev = 1, numvar2d
684      do jlat = myLatBeg, myLatEnd
685        do jlon = myLonBeg, myLonEnd
686          jn = jn + 1
687          cv_mpilocal(jn) = gd_mpiGlobal(jlon,jlat,jlev)
688        end do
689      end do
690    end do
