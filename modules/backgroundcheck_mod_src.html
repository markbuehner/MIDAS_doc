 38  ! public procedures
 39  public :: bgck_bgcheck_conv

 41  integer,          parameter :: numFamilyToProcess = 11
 42  character(len=2), parameter :: familyListToProcess(numFamilyToProcess)= (/ &
 43                                 'UA','AI','SF','SC','SW','PR','RO','GP','RA', &
 44                                 'CH','AL' /)

 48  !--------------------------------------------------------------------------
 49  ! bgck_bgcheck_conv
 50  !--------------------------------------------------------------------------
 51  subroutine bgck_bgcheck_conv( columnTrlOnAnlIncLev, columnTrlOnTrlLev, hco_anl, obsSpaceData )
 52    !
 53    !:Purpose: Do background check on all conventional observations
 54    !
 55    implicit none
 56
 57    ! Arguments:
 58    type(struct_obs)         , intent(inout) :: obsSpaceData         ! Observation-related data
 59    type(struct_columnData)  , intent(inout) :: columnTrlOnAnlIncLev ! column data on analysis levels
 60    type(struct_columnData)  , intent(inout) :: columnTrlOnTrlLev    ! column data on trial levels
 61    type(struct_hco), pointer, intent(in)    :: hco_anl              ! horizontal grid structure
 62
 63    ! Locals:
 64    integer                     :: familyIndex
 65    integer                     :: nulNam, ier, fnom, fclos
 66    logical                     :: noObsToProcess
 67    character(len=*), parameter :: myName = 'bgck_bgcheck_conv'
 68
 69    ! namelist variables
 70    logical                     :: new_bgck_sw ! choose to use the 'new' background check for SW obs
 71    namelist /NAMBGCKCONV/ new_bgck_sw
 72
 73    write(*,*) myName//' begin conventional data background check'
 74
 75    ! Check if any observations are present for conventional background check
 76    noObsToProcess = .true.
 77    do familyIndex = 1, numFamilyToProcess
 78      if (obs_famExist(obsSpaceData,familyListToProcess(familyIndex))) then
 79        noObsToProcess = .false.
 80      end if
 81    end do
 82    if (noObsToProcess) then
 83      write(*,*) myName//': No observations to process'
 84      return
 85    end if
 86     
 87    call utl_tmg_start(117,'--BgckConventional')
 88
 89    new_bgck_sw = .false.
 90
 91    nulNam=0
 92    ier = fnom( nulNam, 'flnml', 'r/o', 0 )
 93    read( nulNam, nml = NAMBGCKCONV, IOSTAT = ier )
 94    if ( ier /= 0 ) write(*,*) myName//': no valid namelist NAMBGCKCONV found, default values will be taken:'
 95    write(*,*) myName//': new_bgck_sw = ',new_bgck_sw
 96    ier = fclos(nulNam)
 97     
 98 
 99    ! Obtain or calc OmP-error std dev when requested and possible.
100    ! Otherwise calc HBHT contribution (sigma_B in observation space)  
101    ! -------------------------------------------------------------------- 
102    call ose_computeStddev( columnTrlOnAnlIncLev, hco_anl, & ! IN
103                            obsSpaceData )                   ! INOUT
104
105    ! DO A BACKGROUND CHECK ON ALL THE OBSERVATIONS
106    ! ----------------------------------------------
107
108    do familyIndex = 1, ofl_numFamily
109      ! For SW only, old and new background check schemes controlled by "new_bgck_sw"
110      if ( obs_famExist( obsSpaceData, ofl_familyList( familyIndex )) ) then
111        call bgck_data( ofl_familyList(familyIndex), obsSpaceData, new_bgck_sw )
112      end if
113    end do
114
115    if (obs_famExist( obsSpaceData, 'RO' )) call bgck_gpsro( columnTrlOnTrlLev , obsSpaceData )
116
117    ! Conduct obs-space post-processing diagnostic tasks (some diagnostic 
118    ! computations controlled by NAMOSD namelist in flnml)
119
120    call osd_ObsSpaceDiag( obsSpaceData, columnTrlOnAnlIncLev, hco_anl, analysisMode_opt = .false. )
121
122    call utl_tmg_stop(117)
123
124  end subroutine bgck_bgcheck_conv

126  !--------------------------------------------------------------------------
127  ! bgck_data
128  !--------------------------------------------------------------------------
129  subroutine bgck_data( obsFamily, obsData, new_bgck_sw )
130    !
131    !:Purpose:  Calculate a background check for a data family and set the
132    !           appropriate quality-control flags in obsSpaceData
133   
134    implicit none
135
136    ! NOTE 1: gps_gb_YSFERRWGT IN MODGPSZTD_MOD (FROM NML FILE) IS USED HERE FOR ERROR WEIGHTING
137    !         OF TIME SERIES (FGAT) GPS MET OBSERVATIONS PS, TS, DPDS. IT IS APPLIED
138    !         (ALONG WITH gps_gb_YZDERRWGT FOR ZTD) IN S/R SETERR AS A MULT. FACTOR TO ERRORS.
139    !         gps_gb_YZDERRWGT AND gps_gb_YSFERRWGT = 1 FOR NORMAL 3D-VAR (3D-THINNING).
140
141    ! Arguments
142    character(len=2), intent(in)    :: obsFamily   ! current observation family
143    type(struct_obs), intent(inout) :: obsData     ! obsSpaceData
144    logical         , intent(in)    :: new_bgck_sw
145    
146    real(8) :: averageFGE, averageOER
147    integer :: obsFlag, obsVarno, headerIndex, codeType, bodyIndex, obsCount
148    integer :: numberObs, numberObsRejected, INZOBS, INZREJ
149    integer :: INPOBS, INTOBS, INDOBS, INPREJ, INTREJ, INDREJ
150    real(8) :: ZOER,ZOMP,ZFGE,ZOMPER,ZBGCHK,ZVAR,ZLEV,ZLAT,ZLON,ZSOP
151    logical :: LLOK, LLZD
152    character(len=12) :: stnid
153    integer :: i_ass, i_vco, i_oth, bodyIndex_u, bodyIndex_v, bodyIndex_start
154    real(8) :: uu_d, uu_r, uu_f, vv_d, vv_r, vv_f, duv2, duv2_lim, zslev
155    logical :: found_u, found_v
156    character(len=*), parameter :: myName = 'bgck_data'
157
158    write(*,*)' '
159    write(*,*) ' ------------------------------'
160    write(*,*) myName//'  background check for', obsFamily, ' data'
161    write(*,*) ' ------------------------------'
162    write(*,*) ' '
163    write(*,'(a55,a74)')'  STNID     LATITU LONGITU  ID Elem        Level        ',  &
164    ' Value        Sigmao       Sigmap         O-P       SigmaOP         qcflag  '
165    write(*,'(a55,a74)')'  -----     ------ -------  -- ----        -----        ',  &
166    ' -----        ------       ------         ---       -------         ------  '
167
168    obsCount = 0
169    averageFGE = 0.d0
170    averageOER=0.d0
171
172    numberObs = 0
173    numberObsRejected = 0
174
175    ! Initialize counters for GP family observations
176
177    if ( obsFamily == 'GP' ) then
178      INZOBS=0
179      INPOBS=0
180      INTOBS=0
181      INDOBS=0
182      INZREJ=0
183      INPREJ=0
184      INTREJ=0
185      INDREJ=0
186    end if
187
188    if ( .not. new_bgck_sw .or. obsFamily /= 'SW' ) then
189
190      ! loop over all header indices of the current observation family
191      call obs_set_current_header_list( obsData, obsFamily )
192      HEADER: do
193        headerIndex = obs_getHeaderIndex( obsData )
194        if ( headerIndex < 0 ) exit HEADER
195
196        stnid = obs_elem_c( obsData, 'STID', headerIndex )
197
198        ! 1. Computation of (HX - Z)**2/(SIGMAo**2 +SIGMAp**2)
199        ! ----------------------------------------------------
200          
201        ! loop over all body indices for the current headerIndex
202        call obs_set_current_body_list( obsData, headerIndex )
203        BODY: do 
204          bodyIndex = obs_getBodyIndex( obsData )
205          if (bodyIndex < 0 ) exit BODY
206
207          obsVarno = obs_bodyElem_i( obsData, OBS_VNM, bodyIndex )
208          LLOK = ( obs_bodyElem_i( obsData, OBS_ASS, bodyIndex ) == obs_assimilated )
209          if ( LLOK ) then
210            numberObs = numberObs + 1
211            if ( obsFamily == 'GP' ) then
212              if ( obsVarno == BUFR_NEZD ) INZOBS = INZOBS+1
213              if ( obsVarno == BUFR_NEPS ) INPOBS = INPOBS+1
214              if ( obsVarno == BUFR_NETS ) INTOBS = INTOBS+1
215              if ( obsVarno == BUFR_NESS ) INDOBS = INDOBS+1
216            end if
217            ZVAR = obs_bodyElem_r( obsData, OBS_VAR, bodyIndex )
218            ZLEV = obs_bodyElem_r( obsData, OBS_PPP, bodyIndex )              
219            ZLAT  = obs_headElem_r( obsData, OBS_LAT, headerIndex ) * MPC_DEGREES_PER_RADIAN_R8
220            ZLON  = obs_headElem_r( obsData, OBS_LON, headerIndex ) * MPC_DEGREES_PER_RADIAN_R8
221
222            ! BACKGROUND CHECK
223
224            ZOMP  = obs_bodyElem_r( obsData, OBS_OMP, bodyIndex )
225
226            ! Get error std dev
227              
228            ! std(O-P) (valid/available if > 0.0)
229            zomper = obs_bodyElem_r( obsData, OBS_OMPE, bodyIndex )
230                                         
231            ! std(O)
232            zoer = obs_bodyElem_r( obsData, OBS_OER, bodyIndex )
233            ! std(P)
234            zfge  = obs_bodyElem_r( obsData, OBS_HPHT, bodyIndex )
235            ! NOTE: For GB-GPS ZTD observations (GP family), ZFGE is not the FGE but
236            !       rather Std(O-P) set in s/r SETERRGPSGB.
237            if ( obsFamily == 'GP' .and. obsVarno == BUFR_NEZD ) then
238              if ( zomper <= 0.0d0 .and. zfge > 0.0d0 ) zomper = zfge
239            end if
240              
241            ! Protect against error std dev values that are too small
242            if ( obsFamily == 'GP' ) then
243              if ( obsVarno == BUFR_NEZD ) then
244                ZOER = ZOER / gps_gb_YZDERRWGT
245              else
246                ZOER = ZOER / gps_gb_YSFERRWGT
247              end if
248              if ( ZOER < 1.0d-3 .and. obsVarno /= BUFR_NEZD ) then
249                write(*,*)' Problem for GP STNID ZOER= ' , stnid, ZOER
250                call utl_abort( myName//': PROBLEM WITH OER.')
251              end if
252              IF ( ZFGE < 1.0d-3 ) then
253                write(*,*)' Problem for GP STNID FGE= ', stnid, ZFGE
254                ZFGE=1.0d-3
255              end if
256              if ( zomper > 0.0d0 .and. zomper < 1.0d-3 ) zomper = 1.0d-3
257            else if ( obsFamily == 'CH' ) then
258              if ( ZFGE**2 + ZOER**2 < 1.0d-60) then
259                write(*,*) ' Problem for STNID FGE ZOER=',  stnid, ZFGE, ZOER
260                ZFGE=1.0d30
261                ZOER=1.0d-30
262              end if
263              if ( zomper > 0.0 .and. zomper < 1.0d-30 ) zomper = 1.0d-30
264            else
265              if ( zfge < 0.0d0 ) zfge = 0.0d0
266              if ( ZFGE**2 + ZOER**2 < 1.0d-5)then
267                write(*,*) ' Problem for STNID FGE ZOER=', stnid, ZFGE, ZOER
268                ZFGE=1.0d-5
269                ZOER=1.0d-5
270              end if
271              if ( zomper > 0.0 .and. zomper < 1.0D-5 ) zomper = 1.0d-5
272            end if
273              
274            ! Calculate zbgchk
275              
276            if ( zomper > 0.0d0 ) then
277              zbgchk = (zomp**2)/(zomper**2) 
278              zsop = zomper
279            else
280              ZBGCHK=(ZOMP)**2/(ZFGE**2 + ZOER**2)
281              ZSOP = SQRT(ZFGE**2 + ZOER**2)
282            end if
283              
284            ! UPDATE QUALITY CONTROL FLAGS, based on zbgchk
285            ! ( ELEMENT FLAGS + GLOBAL HEADER FLAGS)
286            obsVarno = obs_bodyElem_i( obsData, OBS_VNM, bodyIndex )
287            if( obsVarno == bufr_nees .and. obs_bodyElem_i( obsData, OBS_XTR, bodyIndex ) == 0 ) then
288              averageFGE = averageFGE + zfge * zfge
289              averageOER = averageOER + zoer * zoer
290              obsCount   = obsCount + 1
291            end if
292            codeType = obs_headElem_i( obsData, OBS_ITY, headerIndex )
293            obsFlag = ISETFLAG( obsFamily, codeType, obsVarno, ZBGCHK )
294
295            ! CONVERT ZTD VALUES FROM M TO MM FOR PRINTOUT
296
297            LLZD = .FALSE.
298            if ( obsFamily == 'GP' .and. obsVarno == BUFR_NEZD ) then
299              ZVAR = ZVAR * 1000.0D0
300              ZOER = ZOER * 1000.0D0
301              ZFGE = ZFGE * 1000.0D0
302              ZOMP = ZOMP * 1000.0D0
303              ZSOP = ZSOP * 1000.0D0
304              LLZD = .TRUE.
305            end if
306
307            stnid = obs_elem_c( obsData, 'STID', headerIndex )
308            if ( obsFlag >= 2 .or. ( LLZD .and. gps_gb_LTESTOP )) then
309              if ( obsFamily /= 'CH' ) then 
310                write(*,122)  &
311                   stnid, zlat, zlon, codeType, obsVarno, ZLEV, ZVAR, ZOER,  &
312                   ZFGE, ZOMP, ZSOP, ZBGCHK, obsFlag
313              else 
314                write(*,124)  &
315                   stnid, zlat, zlon, codeType, obsVarno, ZLEV, ZVAR, ZOER,  &
316                   ZFGE, ZOMP, ZSOP, ZBGCHK, obsFlag
317              end if
318              if ( obsFlag >= 2 ) numberObsRejected = numberObsRejected + 1
319              if ( obsFlag >= 2 .and. obsFamily == 'GP' ) then
320              if ( obsVarno == BUFR_NEZD ) INZREJ = INZREJ+1
321              if ( obsVarno == BUFR_NEPS ) INPREJ = INPREJ+1
322              if ( obsVarno == BUFR_NETS ) INTREJ = INTREJ+1
323              if ( obsVarno == BUFR_NESS ) INDREJ = INDREJ+1
324            end if
325          end if
326
327          if ( obsFlag == 1 ) then
328            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex ), 13 ))
329          else if ( obsFlag == 2 ) then
330            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex ), 14 ))
331            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex ), 16 ))
332            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex ), 09 ))
333            call obs_headSet_i( obsData, OBS_ST1, headerIndex, ibset( obs_headElem_i( obsData, OBS_ST1, headerIndex ), 06 ))
334
335          else if ( obsFlag == 3 ) then
336            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex), 15 ))
337            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex), 16 ))
338            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex), 09 ))
339            call obs_headSet_i( obsData, OBS_ST1, headerIndex, ibset( obs_headElem_i( obsData, OBS_ST1, headerIndex), 06 ))
340          end if
341        end if
342      end do BODY
343
344124       FORMAT(2x,a9,1x,f6.2,1x,f7.2,1x,I3,1x,I5,7(2x,G11.2),I3 )
345122       FORMAT(2x,a9,1x,f6.2,1x,f7.2,1x,I3,1x,I5,7(2x,F11.2),I3 )
346
347    end do HEADER
348
349    else !IF (.not. new_bgck_sw .or. obsFamily .ne. 'SW') then
350
351      call obs_set_current_body_list( obsData, 'SW' )
352      bodyuv: do
353        bodyIndex = obs_getBodyIndex(obsData)
354        if ( bodyIndex < 0 ) exit bodyuv
355
356        ! Only process pressure level observations flagged to be assimilated
357        i_ass = obs_bodyElem_i ( obsData, OBS_ASS, bodyIndex )
358        i_vco = obs_bodyElem_i ( obsData, OBS_VCO, bodyIndex )
359
360        if( i_ass /= obs_assimilated .or. i_vco /= 2 ) cycle bodyuv
361 
362        headerIndex     = obs_bodyElem_i( obsData, OBS_HIND, bodyIndex )
363        bodyIndex_start = obs_headElem_i( obsData, OBS_RLN, headerIndex )
364
365        obsVarno = obs_bodyElem_i( obsData, OBS_VNM, bodyIndex )
366        zlev  = obs_bodyElem_r( obsData, OBS_PPP, bodyIndex )
367
368        found_u = .false.
369        found_v = .false.
370
371        if ( obsVarno == BUFR_NEUU ) then
372      
373          bodyIndex_u = bodyIndex
374          found_u = .true.
375
376          do i_oth = bodyIndex_start, bodyIndex
377            obsVarno = obs_bodyElem_i( obsData, OBS_VNM, i_oth )
378            zslev = obs_bodyElem_r( obsData, OBS_PPP, i_oth )
379            if ( obsVarno == BUFR_NEVV .and. zslev == zlev ) then
380              bodyIndex_v = i_oth
381              found_v = .true.
382            end if
383          end do
384
385        else
386
387          bodyIndex_v = bodyIndex
388          found_v = .true.
389 
390          do i_oth = bodyIndex_start, bodyIndex
391            obsVarno = obs_bodyElem_i( obsData, OBS_VNM, i_oth )
392            zslev = obs_bodyElem_r( obsData, OBS_PPP, i_oth )
393            if ( obsVarno == BUFR_NEUU .and. zslev == zlev ) then
394              bodyIndex_u = i_oth
395              found_u = .true.
396            end if
397          end do
398
399        end if
400
401        if ( found_u .and. found_v ) then
402
403          uu_d = obs_bodyElem_r( obsData, OBS_OMP, bodyIndex_u )
404          vv_d = obs_bodyElem_r( obsData, OBS_OMP, bodyIndex_v )
405          uu_r = obs_bodyElem_r( obsData, OBS_OER, bodyIndex_u )
406          vv_r = obs_bodyElem_r( obsData, OBS_OER, bodyIndex_v )
407          uu_f = obs_bodyElem_r( obsData, OBS_HPHT, bodyIndex_u )
408          vv_f = obs_bodyElem_r( obsData, OBS_HPHT, bodyIndex_v )
409
410          duv2 = uu_d**2 + vv_d**2
411
412          obsFlag = 0
413          duv2_lim = (uu_r**2 + uu_f**2 + vv_r**2 + vv_f**2)*1
414          if (duv2 > duv2_lim) then
415            obsFlag = 1
416          end if
417          duv2_lim = (uu_r**2 + uu_f**2 + vv_r**2 + vv_f**2)*2
418          if(duv2 > duv2_lim) then
419            obsFlag = 2
420          end if
421          duv2_lim = (uu_r**2 + uu_f**2 + vv_r**2 + vv_f**2)*4
422          if(duv2 > duv2_lim) then
423            obsFlag = 3
424          end if
425
426          numberObs = numberObs + 2
427          if ( obsFlag >= 2 ) numberObsRejected = numberObsRejected + 2
428
429          if ( obsFlag == 1 ) then
430            call obs_bodySet_i( obsData,OBS_FLG,bodyIndex_u, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_u ), 13 ))
431            call obs_bodySet_i( obsData,OBS_FLG,bodyIndex_v, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_v ), 13 ))
432
433          else if ( obsFlag == 2 ) then
434            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex_u, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_u ), 14 ))
435            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex_u, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_u ), 16 ))
436            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex_u, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_u ), 09 ))
437            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex_v, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_v ), 14 ))
438            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex_v, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_v ), 16 ))
439            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex_v, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_v ), 09 ))
440            call obs_headSet_i( obsData, OBS_ST1, headerIndex, ibset( obs_headElem_i( obsData, OBS_ST1, headerIndex ), 06 ))
441
442          else if ( obsFlag == 3 ) then
443            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex_u, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_u ), 15 ))
444            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex_u, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_u ), 16 ))
445            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex_u, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_u ), 09 ))
446            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex_v, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_v ), 15 ))
447            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex_v, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_v ), 16 ))
448            call obs_bodySet_i( obsData, OBS_FLG, bodyIndex_v, ibset( obs_bodyElem_i( obsData, OBS_FLG, bodyIndex_v ), 09 ))
449            call obs_headSet_i( obsData, OBS_ST1, headerIndex, ibset( obs_headElem_i( obsData, OBS_ST1, headerIndex ), 06 ))
450          end if
451
452        end if !if(found_u .and. found_v) 
453
454      end do bodyuv
455
456    end if !IF (.not. new_bgck_sw .or. obsFamily .ne. 'SW')
457
458    if ( numberObs > 0 ) then
459      write(*,*)' '
460      write(*,*)   myName//': FINISHED BGCHECK OF ', obsFamily, ' DATA'
461      write(*,123) myName//':   ', numberObsRejected, ' OBSERVATIONS REJECTED OUT OF ', numberObs
462      write(*,*)' '
463    end if
464
465    if ( numberObs > 0 .and. obsFamily == 'GP' ) then
466      write(*,*)' '
467      write(*,*) '  BGCDATA:    REPORT FOR GP FAMILY OF OBSERVATIONS'
468      write(*,123) 'BGCDATA:   ',INZREJ, ' ZTD  OBSERVATIONS REJECTED OUT OF ', INZOBS
469      write(*,123) 'BGCDATA:   ',INPREJ, ' PSFC OBSERVATIONS REJECTED OUT OF ', INPOBS
470      write(*,123) 'BGCDATA:   ',INTREJ, ' TSFC OBSERVATIONS REJECTED OUT OF ', INTOBS
471      write(*,123) 'BGCDATA:   ',INDREJ, ' DPDS OBSERVATIONS REJECTED OUT OF ', INDOBS

492  !--------------------------------------------------------------------------
493  subroutine bgck_gpsro(columnTrlOnTrlLev,obsData)
494      !
495      !:Purpose: Set background-check flag on GPSRO data if ABS(O-P)/P is too
496      !          large
497      !
498      implicit none
499
500      type(struct_columnData) :: columnTrlOnTrlLev
501      type(struct_obs) :: obsData
502      type(struct_vco), pointer :: vco_trl
503      real(8) :: HNH1, ZOBS, ZMHX, ZOMF, ZREF, ZOER, Rad
504
505      integer :: headerIndex
506      integer :: IDATYP, iProfile, varNum
507      integer :: IDATA   , IDATEND, bodyIndex
508      integer :: NGPSLEV
509      integer :: iversion
510
511      vco_trl => col_getVco(columnTrlOnTrlLev)
512      iversion = vco_trl%vCode
513      
514      write(*,*)'ENTER BGCSGPSRO'
515      
516      ! 1.  Initializations
517      !     ---------------
518
519      NGPSLEV=col_getNumLev(columnTrlOnTrlLev,'TH')
520
521      ! Loop over all files
522
523      ! loop over all header indices of the 'RO' family
524      call obs_set_current_header_list( obsData, 'RO' )
525      HEADER: do
526         headerIndex = obs_getHeaderIndex(obsData)
527         if (headerIndex < 0) exit HEADER
528     
529         ! Process only refractivity data (codtyp 169)
530
531         IDATYP = obs_headElem_i(obsData,OBS_ITY,headerIndex)
532         IF ( IDATYP == 169 ) THEN
533            iProfile = gps_iprofile_from_index(headerIndex)
534            varNum = gps_vRO_IndexPrf(iProfile, 2)
535
536            ! Basic geometric variables of the profile
537            
538            Rad  = obs_headElem_r(obsData,OBS_TRAD,headerIndex)
539
540            ! Loops over data in the observation
541            
542            IDATA   = obs_headElem_i(obsData,OBS_RLN,headerIndex)
543            IDATEND = obs_headElem_i(obsData,OBS_NLV,headerIndex) + IDATA - 1
544
545            ! Scan for requested assimilations, and count them
546
547            do bodyIndex= IDATA, IDATEND
548               if ( obs_bodyElem_i( obsData, OBS_ASS, bodyIndex ) == obs_assimilated ) then
549                  HNH1 = obs_bodyElem_r( obsData, OBS_PPP, bodyIndex )
550                  if ( varNum == bufr_nebd ) HNH1 = HNH1-Rad
551
552                  ! Increment OMF = Y - H(x)
553
554                  ZOMF = obs_bodyElem_r( obsData, OBS_OMP, bodyIndex )
555
556                  ! Observation value    Y
557
558                  ZOBS = obs_bodyElem_r( obsData, OBS_VAR, bodyIndex )
559                  ZOER = obs_bodyElem_r( obsData, OBS_OER, bodyIndex )
560                  ZMHX = ZOBS-ZOMF
561
562                  ! Reference order of magnitude value:
563
564                  if ( varNum == bufr_nebd ) then
565                     ZREF = 0.025d0*exp(-HNH1/6500.d0)
566                  else
567                     if ( .not. gps_roBNorm ) then
568                       ZREF = 300.d0*exp(-HNH1/6500.d0)
569                     else
570                       ZREF = ZMHX
571                     end if
572                  end if
573                           
574                  ! OMF Tested criteria:
575
576                  if ( .not. gps_roBNorm ) then
577                    if (DABS(ZOMF)/ZREF > gps_BgckBand .or. DABS(ZOMF)/ZOER > 3.d0) then
578                      call obs_bodySet_i(obsData,OBS_FLG,bodyIndex,ibset(obs_bodyElem_i(obsData,OBS_FLG,bodyIndex),16))
579                      call obs_bodySet_i(obsData,OBS_FLG,bodyIndex,ibset(obs_bodyElem_i(obsData,OBS_FLG,bodyIndex),9))
580                    end if
581                  else
582                    if ( DABS(ZOMF)/ZREF > gps_BgckBand .or. DABS(ZOMF)/ZOER > gps_roNsigma) then
583                      call obs_bodySet_i(obsData,OBS_FLG,bodyIndex,ibset(obs_bodyElem_i(obsData,OBS_FLG,bodyIndex),16))
584                      call obs_bodySet_i(obsData,OBS_FLG,bodyIndex,ibset(obs_bodyElem_i(obsData,OBS_FLG,bodyIndex),9))
585                      write(*,'(A40,F10.0,3F12.4)') ' REJECT BGCSGPSRO H  O  P (O-P/ZREF) =',HNH1,ZOBS,ZMHX,(ZOMF)/ZREF
586                    end if                  
587                  end if
588                  
589               end if
590            end do
591         end if
592
593      end do HEADER
594
595      write(*,*)'EXIT BGCSGPSRO'
596      RETURN
597      
598  end subroutine bgck_gpsro
