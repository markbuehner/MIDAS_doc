 39  !--------------------------------------------------------------------------
 40  ! obsu_updateSourceVariablesFlag
 41  !--------------------------------------------------------------------------
 42  subroutine obsu_updateSourceVariablesFlag(obsSpaceData)
 43    implicit none
 44    ! argument
 45    type(struct_obs) :: obsSpaceData
 46
 47    ! locals
 48    integer          :: transformBufrCode, transformBufrCodeExtra
 49    integer          :: flag, flagExtra, mergedFlag
 50    integer          :: sourceBufrCode, sourceBufrCodeExtra
 51    integer          :: headerIndex, bodyIndexStart, bodyIndexEnd, bodyIndex, bodyIndex2
 52    real(pre_obsReal)   :: level
 53
 54    body : do bodyIndex = 1, obs_numBody(obsSpaceData) 
 55
 56      if (ovt_isTransformedVariable(obs_bodyElem_i(obsSpaceData,OBS_VNM,bodyIndex))) then
 57
 58        ! We have found a transformed variable
 59        transformBufrCode = obs_bodyElem_i(obsSpaceData,OBS_VNM,bodyIndex)
 60
 61        ! We will assign his assimilation flag to its source variable
 62        sourceBufrCode = ovt_getSourceBufrCode(transformBufrCode)
 63
 64        headerIndex      = obs_bodyElem_i(obsSpaceData, OBS_HIND, bodyIndex   )
 65        bodyIndexStart   = obs_headElem_i(obsSpaceData, OBS_RLN , headerIndex )
 66        bodyIndexEnd     = obs_headElem_i(obsSpaceData, OBS_NLV , headerIndex ) + bodyIndexStart - 1
 67        level            = obs_bodyElem_r(obsSpaceData, OBS_PPP , bodyIndex   )
 68        flag             = obs_bodyElem_i(obsSpaceData, OBS_FLG , bodyIndex   )
 69
 70        if (ovt_isWindObs(sourceBufrCode)) then
 71
 72           ! Get the flag of the companion transformed variable
 73          transformBufrCodeExtra = ovt_getDestinationBufrCode(sourceBufrCode, &
 74                                                                   extra_opt=.true.)
 75
 76          flagExtra = -999
 77          do bodyIndex2 = bodyIndexStart, bodyIndexEnd
 78            if ( obs_bodyElem_i(obsSpaceData, OBS_VNM, bodyIndex2 ) == transformBufrCodeExtra .and. &
 79                 obs_bodyElem_r(obsSpaceData, OBS_PPP, bodyIndex2 ) == level ) then
 80              flagExtra = obs_bodyElem_i(obsSpaceData, OBS_FLG , bodyIndex2)
 81            end if
 82          end do
 83
 84          ! Combine flags
 85          if (flagExtra /= -999) then
 86            mergedFlag = ior(flag,flagExtra)
 87          else
 88            call utl_abort('obsu_updateSourceVariablesFlag: could not find the wind companion variable')
 89          end if
 90          
 91          ! Find sourceBufrCode and sourceBufrCodeExtra and update their flag
 92          sourceBufrCodeExtra = ovt_getSourceBufrCode(transformBufrCode,&
 93                                                                      extra_opt=.true.)
 94          do bodyIndex2 = bodyIndexStart, bodyIndexEnd
 95            if ( obs_bodyElem_i(obsSpaceData, OBS_VNM, bodyIndex2 ) == sourceBufrCode .and. &
 96                 obs_bodyElem_r(obsSpaceData, OBS_PPP, bodyIndex2 ) == level ) then
 97              call obs_bodySet_i(obsSpaceData, OBS_FLG, bodyIndex2, mergedFlag ) 
 98            end if
 99            if ( obs_bodyElem_i(obsSpaceData, OBS_VNM, bodyIndex2 ) == sourceBufrCodeExtra .and. &
100                 obs_bodyElem_r(obsSpaceData, OBS_PPP, bodyIndex2 ) == level ) then
101              call obs_bodySet_i(obsSpaceData, OBS_FLG, bodyIndex2, mergedFlag ) 
102            end if
103          end do
104          
105        else
106          
107          ! Find sourceBufrCode and update its flag
108          do bodyIndex2 = bodyIndexStart, bodyIndexEnd
109            if ( obs_bodyElem_i(obsSpaceData, OBS_VNM, bodyIndex2 ) == sourceBufrCode .and. &
110                 obs_bodyElem_r(obsSpaceData, OBS_PPP, bodyIndex2 ) == level ) then
111              call obs_bodySet_i(obsSpaceData, OBS_FLG, bodyIndex2, flag ) 
112            end if
113          end do
114
115        end if
116        
117      end if
118      
119    end do body
120
121  end subroutine obsu_updateSourceVariablesFlag

126  subroutine obsu_setassflg(obsSpaceData)
127    !
128    !:Purpose: Set banco quality control bit #12 for all data assimilated
129    !           by current analysis.
130    !
131    implicit none
132    ! argument
133    type(struct_obs) :: obsSpaceData
134
135    ! local
136    integer :: bodyIndex
137
138    ! Process all data
139    BODY: do bodyIndex = 1, obs_numBody(obsSpaceData)
140
141      if (obs_bodyElem_i(obsSpaceData, OBS_ASS, bodyIndex ) == obs_assimilated ) &
142        call obs_bodySet_i( obsSpaceData, OBS_FLG, bodyIndex, &
143                            ibset( obs_bodyElem_i(obsSpaceData, OBS_FLG,bodyIndex), 12 ))
144    end do BODY
145
146  end subroutine obsu_setassflg

149  ! surfvcord
150  !--------------------------------------------------------------------------
151  real function surfvcord( varno, codtyp )
152
153    implicit none
154    ! Arguments
155    integer    :: varno
156    integer    :: codtyp
157    ! locals
158    character(len=9)  :: family
159
160    family = codtypfam(codtyp)
161    surfvcord = 0.0
162
163    select case(family)
164      case ('synop')
165        select case(varno)
166          case (bufr_neds, bufr_nefs, bufr_neus, bufr_nevs, bufr_gust)
167            surfvcord = 10.0
168          case (bufr_nets, bufr_nees, bufr_ness, bufr_vis, bufr_logVis)
169            surfvcord = 1.5
170        end select
171      case ('ship')
172        select case(varno)
173          case (bufr_neds, bufr_nefs, bufr_neus, bufr_nevs, bufr_gust)
174            surfvcord = 20.0
175          case (bufr_nets, bufr_nees, bufr_ness, bufr_vis, bufr_logVis)
176            surfvcord = 11.5
177        end select
178      case ('upairland')
179        select case(varno)
180          case (bufr_neds, bufr_nefs, bufr_neus, bufr_nevs, bufr_gust)
181            surfvcord = 10.0
182          case (bufr_nets, bufr_ness, bufr_vis, bufr_logVis)
183            surfvcord = 1.5
184        end select
185      case ('upairship')
186        select case(varno)
187          case (bufr_neds, bufr_nefs, bufr_neus, bufr_nevs, bufr_gust)
188            surfvcord = 20.0
189          case (bufr_nets, bufr_ness, bufr_vis, bufr_logVis)
190            surfvcord = 1.5
191        end select
192      case ('scatwinds')
193        select case(varno)
194          case (bufr_neds, bufr_nefs, bufr_neus, bufr_nevs)
195            surfvcord = 10.0
196        end select
197    end select
198
199  end function  surfvcord

201  !--------------------------------------------------------------------------
202  ! codtypfam
203  !--------------------------------------------------------------------------
204  function codtypfam(codtyp) result(family)
205    implicit none
206    integer          :: codtyp
207    character(len=9) :: family
208
209    if (       codtyp == codtyp_get_codtyp('synopnonauto')   .or. codtyp == codtyp_get_codtyp('synopmobil')      &
210       .or.    codtyp == codtyp_get_codtyp('asynopauto') ) then
211       family = 'synop'
212    else if (  codtyp == codtyp_get_codtyp('shipnonauto')    .or. codtyp == codtyp_get_codtyp('drifter')         &
213       .or.    codtyp == codtyp_get_codtyp('synoppatrol')    .or. codtyp == codtyp_get_codtyp('ashipauto') ) then
214       family = 'ship'
215    else if (  codtyp == codtyp_get_codtyp('temppilot')      .or. codtyp == codtyp_get_codtyp('tempsynop')       &
216       .or.    codtyp == codtyp_get_codtyp('pilotsynop')     .or. codtyp == codtyp_get_codtyp('temppilotsynop')  &
217       .or.    codtyp == codtyp_get_codtyp('pilot')          .or. codtyp == codtyp_get_codtyp('pilotmobil')      &
218       .or.    codtyp == codtyp_get_codtyp('temp')           .or. codtyp == codtyp_get_codtyp('tempdrop')        &
219       .or.    codtyp == codtyp_get_codtyp('tempmobil')      .or. codtyp == codtyp_get_codtyp('temppilotmobil')  &
220       .or.    codtyp == codtyp_get_codtyp('tempsynopmobil') .or. codtyp == codtyp_get_codtyp('pilotsynopmobil') &
221       .or.    codtyp == codtyp_get_codtyp('temppilotsynopmobil') ) then
222       family = 'upairland'
223    else if (  codtyp == codtyp_get_codtyp('temppilotship')  .or. codtyp ==codtyp_get_codtyp('tempshipship')     &
224       .or.    codtyp == codtyp_get_codtyp('tempsshipship')  .or. codtyp ==codtyp_get_codtyp('pilotshipship')    &
225       .or.    codtyp == codtyp_get_codtyp('pilotship')      .or. codtyp ==codtyp_get_codtyp('tempship') ) then
226       family = 'upairship'
227    else if (  codtyp == codtyp_get_codtyp('ascat') ) then
228       family = 'scatwinds'
229    else
230       family = 'other'
231    end if
232
233  end function codtypfam

237  !--------------------------------------------------------------------------
238  subroutine  obsu_computeVertCoordSurfObs(obsdat, headerIndexStart, headerIndexEnd )
239    !
240    implicit none
241
242    ! Arguments
243    type (struct_obs), intent(inout) :: obsdat
244    integer          , intent(in)    :: headerIndexStart
245    integer          , intent(in)    :: headerIndexEnd
246
247    ! locals
248    integer  :: bodyIndex, headerIndex, bodyIndexStart, bodyIndexEnd
249    integer  :: varno, codtyp
250    real           :: sfc_vco, elev
251    real(pre_obsReal) :: ppp
252
253    HEADER: do headerIndex = headerIndexStart, headerIndexEnd
254        
255      bodyIndexStart = obs_headElem_i(obsdat, OBS_RLN, headerIndex )
256      bodyIndexEnd   = obs_headElem_i(obsdat, OBS_NLV, headerIndex ) + bodyIndexStart - 1
257      codtyp = obs_headElem_i(obsdat, OBS_ITY, headerIndex )
258      elev = obs_headElem_r(obsdat, OBS_ALT, headerIndex )
259
260      BODY: do bodyIndex = bodyIndexStart, bodyIndexEnd
261        varno = obs_bodyElem_i(obsdat, OBS_VNM, bodyIndex )
262
263        select case(varno)
264          case(bufr_neds, bufr_nefs, bufr_neus, bufr_nevs, bufr_nets, bufr_ness, bufr_nepn, bufr_neps, bufr_nehs, &
265               bufr_nezd, bufr_vis, bufr_logVis, bufr_gust )
266            sfc_vco = surfvcord(varno, codtyp )
267            if ( varno /= bufr_nepn) then
268              ppp = elev + sfc_vco
269              call obs_bodySet_r(obsdat, OBS_PPP, bodyIndex, ppp )
270              call obs_bodySet_i(obsdat, OBS_VCO, bodyIndex, 1 )
271            else
272              ppp = 0.
273              call obs_bodySet_r(obsdat,OBS_PPP, bodyIndex, ppp )
274              call obs_bodySet_i(obsdat,OBS_VCO, bodyIndex, 1 )
275            end if
276        end select
277
278      end do BODY
279
280    end do HEADER
281
282  end subroutine obsu_computeVertCoordSurfObs

287  subroutine  obsu_setGbgpsError(obsdat, headerIndexStart, headerIndexEnd )
288    !
289    implicit none
290
291    ! Arguments
292    type (struct_obs), intent(inout) :: obsdat
293    integer          , intent(in)    :: headerIndexStart
294    integer          , intent(in)    :: headerIndexEnd
295
296    ! locals
297    real(pre_obsReal)    :: obsv
298    integer  :: bodyIndex, headerIndex,bodyIndexStart, bodyIndexEnd
299    integer  :: varno
300
301    HEADER: do headerIndex = headerIndexStart, headerIndexEnd
302 
303      bodyIndexStart = obs_headElem_i(obsdat, OBS_RLN, headerIndex )
304      bodyIndexEnd   = obs_headElem_i(obsdat, OBS_NLV, headerIndex ) + bodyIndexStart - 1
305      obsv = real(MPC_missingValue_R8, pre_obsReal)
306      
307      BODY: do bodyIndex = bodyIndexStart, bodyIndexEnd 
308
309        varno = obs_bodyElem_i(obsdat,OBS_VNM,bodyIndex )
310
311        if ( varno == bufr_nefe ) then
312          obsv = obs_bodyElem_r(obsdat,OBS_VAR, bodyIndex )
313          call obs_bodySet_i(obsdat, OBS_VNM, bodyIndex, 999 )
314          exit BODY
315        end if
316
317      end do BODY
318
319      BODY2: do bodyIndex = bodyIndexStart, bodyIndexEnd
320
321        varno = obs_bodyElem_i(obsdat, OBS_VNM, bodyIndex )
322
323        if ( varno == bufr_nezd .and. obsv /= real(MPC_missingValue_R8, pre_obsReal)) then
324          call obs_bodySet_r(obsdat, OBS_OER, bodyIndex, obsv )
325          exit BODY2
326        end if
327
328      end do BODY2
329
330    end do HEADER
331
332  end subroutine obsu_setGbgpsError

335  ! obsu_cvt_obs_instrum
336  !--------------------------------------------------------------------------
337  integer function obsu_cvt_obs_instrum(sensor)
338    !
339    !:Purpose: Map burp satellite sensor indicator (element #2048) to
340    !           burp satellite instrument (element #2019). This is a more
341    !           complete common element, allowing for future expansion.
342    !
343    !:Table of BURP satellite sensor indicator element #002048:
344    ! ==================  =============================== 
345    ! Satellite sensor    BURP satellite sensor indicator
346    ! ==================  =============================== 
347    !               HIRS                 0
348    !                MSU                 1
349    !                SSU                 2
350    !              AMSUA                 3
351    !              AMSUB                 4
352    !              AVHRR                 5
353    !               SSMI                 6
354    !              NSCAT                 7
355    !           SEAWINDS                 8
356    !           Reserved                9-14
357    !      Missing value                15
358    ! ==================  =============================== 
359    !
360    implicit none
361    integer :: sensor      ! BURP satellite sensor indicator (element #2048)
362    integer :: instrument  ! BURP satellite instrument       (element #2019)
363
364    select case (sensor)
365      case (000);   instrument = 606  ! HIRS
366      case (001);   instrument = 623  ! MSU
367      case (002);   instrument = 627  ! SSU
368      case (003);   instrument = 570  ! AMSUA
369      case (004);   instrument = 574  ! AMSUB
370      case (005);   instrument = 591  ! AVHRR
371      case (006);   instrument = 905  ! SSMI
372      case (007);   instrument = 312  ! NSCAT
373      case (008);   instrument = 313  ! SEAWINDS
374      case (015);   instrument = 2047 ! Missing value
375      case default; instrument = 2047 ! Unrecognized value
376    end select
377
378    obsu_cvt_obs_instrum = instrument
379
380  end function obsu_cvt_obs_instrum
