  22  !           non-extended).

  36  public :: agd_SetupFromHCO, agd_mach, agd_mach_r4

  40  ! Definition of some parameters characterizing the geometry of
  41  ! the Limited-Area (LA) analysis grid and associated metric factors
  42  type :: struct_glmf

  44     real(8), allocatable :: rlon   (:) ! Longitudes of Scalar gridpoints

  47     real(8), allocatable :: cos2 (:)   ! Grid metric for Psi-Chi to U-V
  48     real(8), allocatable :: cos2h(:)   ! Grid metric for Psi-Chi to U-V
  49     real(8), allocatable :: cos2vd (:) ! Grid metric for U-V to Vort-Div
  50     real(8), allocatable :: cos2hvd(:) ! Grid metric for U-V to Vort-Div

  77  !--------------------------------------------------------------------------
  78  ! agd_SetupFromHCO
  79  !--------------------------------------------------------------------------
  80  subroutine agd_SetupFromHCO( hco_ext_in, hco_core_opt )
  81    implicit none
  82
  83    type(struct_hco), pointer :: hco_ext_in
  84    type(struct_hco), pointer, optional :: hco_core_opt
  85
  86    real(8), allocatable :: rlath  (:) ! Latitudes of half grid-points of gridpoints in lat-direction
  87    real(8), allocatable :: rrcos  (:) ! 1.0/Cos(Latitudes of gridpoints)
  88    !real(8), allocatable :: rrcosh (:) ! 1.0/Cos(Half-Latitudes of gridpoints)
  89    real(8), allocatable :: rdmu   (:) ! Differences of mu=sin(lat)
  90    real(8), allocatable :: rdmuh  (:) ! Differences of muh=sin(lath)
  91    real(8), allocatable :: r1mmu2 (:) ! (1.-mu**2)
  92    real(8), allocatable :: r1mmu2h(:) ! (1.-muh**2)
  93
  94    real(8) :: dlon_test, dlat_test, dlon_ref, dlat_ref
  95
  96    integer :: i, j
  97
  98    logical, save :: firstCall = .true.
  99
 100    ! Ensure subroutine only runs one time during program execution
 101    if (firstCall) then
 102      firstCall = .false.
 103    else
 104      return
 105    end if
 106
 107    write(*,*)
 108    write(*,*) 'agd_SetupFromHCO: Starting...'
 109
 110    initialized = .true.
 111
 112    !
 113    !- 1.  Check the hco structures
 114    !
 115    hco_ext  => hco_ext_in
 116    if( present(hco_core_opt) ) then
 117      hco_core => hco_core_opt
 118    else
 119      hco_core => hco_ext_in
 120    end if
 121
 122    if ( (.not. hco_core%initialized) .or.  & 
 123         (.not. hco_ext%initialized) ) then
 124      write(*,*)
 125      write(*,*) 'agd_SetupFromHCO: At least one hco structure was not initilzed'
 126      write(*,*) 'hco_core = ', hco_core%initialized
 127      write(*,*) 'hco_ext = ', hco_ext%initialized
 128      call utl_abort('agd_SetupFromHCO: abort')
 129    end if
 130
 131    if ( (hco_core%global) .or.  & 
 132         (hco_ext%global) ) then
 133      write(*,*)
 134      write(*,*) 'agd_SetupFromHCO: At least one hco structure is from a global grid, skipping rest of setup'
 135      write(*,*) 'hco_core = ', hco_core%global
 136      write(*,*) 'hco_ext = ', hco_ext%global
 137      return
 138    end if
 139
 140    !
 141    !- 2.  Dimension settings and Memory allocation
 142    !
 143
 144    !- 2.1 Dimensions
 145    ni_core = hco_core%ni
 146    nj_core = hco_core%nj
 147
 148    ni_ext = hco_ext%ni
 149    nj_ext = hco_ext%nj
 150
 151    ext_i  = ni_ext - ni_core
 152    ext_j  = nj_ext - nj_core
 153
 154    if ( ext_i == 0 .and. ext_j == 0 ) then
 155      write(*,*)
 156      write(*,*) 'agd_SetupFromHCO: LAM core and extended grids are identical'
 157    else if ( ext_i < 10 .or. ext_j < 10 ) then
 158      write(*,*)
 159      write(*,*) 'agd_SetupFromHCO: LAM domain extension is less than 10 gridpoints'
 160      write(*,*) ' ext_i = ', ext_i,' ext_j = ', ext_j
 161    end if
 162
 163    ni_ext_per = ni_ext + 1  ! Fields will be periodic (i = 1 repeated) at ni_ext+1
 164    nj_ext_per = nj_ext + 1  ! Fields will be periodic (j = 1 repeated) at nj_ext+1
 165
 166    !- 2.2 First Make sure we have a uniform grid in x and y direction
 167
 168    !- 2.2.1 Analysis core grid
 169    dlon_ref = hco_core%lon(2) - hco_core%lon(1)
 170    do i = 2, ni_core
 171      dlon_test = hco_core%lon(i) - hco_core%lon(i-1)
 172      if ( (dlon_test - dlon_ref) > dlon_ref/100.0d0 ) then
 173        write(*,*)
 174        write(*,*) 'agd_SetupFromHCO: Core grid spacing is not uniform in x-direction'
 175        write(*,*) ' i         = ', i
 176        write(*,*) ' dlon      = ', dlon_test
 177        write(*,*) ' dlon ref  = ', dlon_ref
 178        call utl_abort('agd_SetupFromHCO')
 179      end if
 180    end do
 181
 182    dlat_ref = hco_core%lat(2) - hco_core%lat(1)
 183    do j = 2, nj_core
 184      dlat_test = hco_core%lat(j) - hco_core%lat(j-1) 
 185      if ( (dlat_test - dlat_ref) > dlat_ref/100.0d0 ) then
 186        write(*,*)
 187        write(*,*) 'agd_SetupFromHCO: Core grid spacing is not uniform in x-direction'
 188        write(*,*) ' j         = ', j
 189        write(*,*) ' dlat      = ', dlon_test
 190        write(*,*) ' dlat ref  = ', dlon_ref
 191        call utl_abort('agd_SetupFromHCO') 
 192      end if
 193    end do
 194
 195    !- 2.2.2 Extended Analysis grid
 196    dlon_ref = hco_ext%lon(2) - hco_ext%lon(1)
 197    do i = 2, ni_ext
 198      dlon_test = hco_ext%lon(i) - hco_ext%lon(i-1)
 199      if ( (dlon_test - dlon_ref) > dlon_ref/100.0d0 ) then
 200        write(*,*)
 201        write(*,*) 'agd_SetupFromHCO: Extended grid spacing is not uniform in x-direction'
 202        write(*,*) ' i         = ', i
 203        write(*,*) ' dlon      = ', dlon_test
 204        write(*,*) ' dlon ref  = ', dlon_ref
 205        call utl_abort('agd_SetupFromHCO')
 206      end if
 207    end do
 208
 209    dlat_ref = hco_ext%lat(2) - hco_ext%lat(1)
 210    do j = 2, nj_ext
 211      dlat_test = hco_ext%lat(j) - hco_ext%lat(j-1)
 212      if ( (dlat_test - dlat_ref) > dlat_ref/100.0d0 ) then
 213        write(*,*)
 214        write(*,*) 'agd_SetupFromHCO: Extended grid spacing is not uniform in x-direction'
 215        write(*,*) ' j         = ', j
 216        write(*,*) ' dlat      = ', dlon_test
 217        write(*,*) ' dlat ref  = ', dlon_ref
 218        call utl_abort('agd_SetupFromHCO') 
 219      end if
 220    end do
 221
 222    !- 2.3 Dimensions for variables needed to be symmetric
 223    istart = -4          ! 5 gridpoints padding in West direction
 224    iend   = ni_ext + 4  ! 4 gridpoints padding in East direction
 225    jstart = -4          ! 5 gridpoints padding in South direction
 226    jend   = nj_ext + 4  ! 4 gridpoints padding in North direction
 227
 228    !- 2.4 Allocations
 229    allocate(glmf%rlon(istart:iend))
 230    allocate(glmf%rlat(jstart:jend))
 231    allocate(rlath    (jstart:jend))
 232    allocate(rrcos    (jstart:jend))
 233    !allocate(rrcosh   (jstart:jend))
 234    allocate(rdmu     (jstart:jend))
 235    allocate(rdmuh    (jstart:jend))
 236    allocate(r1mmu2   (jstart:jend))
 237    allocate(r1mmu2h  (jstart:jend))
 238
 239    !
 240    !- 3.  Set Lat-Lon of the computational grid
 241    !
 242
 243    !  3.1 Set (constant) Grid spacing
 244    glmf%rdlon = (hco_ext%lon(2) - hco_ext%lon(1))
 245    glmf%rdlat = (hco_ext%lat(2) - hco_ext%lat(1))
 246
 247    !- 3.2 Lat-Lon
 248
 249    !  3.2.1 Extract the lat-lon from the extended grid
 250    glmf%rlon(1:ni_ext) = hco_ext%lon(1:ni_ext)
 251    glmf%rlat(1:nj_ext) = hco_ext%lat(1:nj_ext)
 252
 253    !  3.2.2 Extend to the full computational grid (larger than the extended grid!)
 254
 255    ! West
 256    do i = istart, 0
 257      glmf%rlon(i) = glmf%rlon(1) + (i-1) * glmf%rdlon
 258    end do
 259    ! East
 260    do i = ni_ext+1, iend
 261      glmf%rlon(i) = glmf%rlon(ni_ext) + (i-ni_ext) * glmf%rdlon
 262    end do
 263    ! North
 264    do j = nj_ext+1, jend
 265      glmf%rlat(j) = glmf%rlat(nj_ext) + (j-nj_ext) * glmf%rdlat
 266    end do
 267    ! South
 268    do j = jstart, 0
 269      glmf%rlat(j) = glmf%rlat(1) + (j-1) * glmf%rdlat
 270    end do
 271
 272    !- 3.2 Half Lat-Lon
 273    do j = jstart+1, jend-1
 274      rlath(j) = ( glmf%rlat(j) + glmf%rlat(j+1) ) / 2.0d0
 275    end do
 276
 277    !
 278    !- 4. Set Metric Factors
 279    !
 280
 281    !- 4.1  Compute local factors
 282    do j = jstart+1, jend-2
 283      rrcos  (j) = 1.0d0 / cos(glmf%rlat (j))
 284      !rrcosh (j) = 1.0d0 / cos(rlath(j))
 285      rdmu   (j) = sin(glmf%rlat (j+1)) - sin(glmf%rlat (j))
 286      rdmuh  (j) = sin(rlath(j+1)) - sin(rlath(j))
 287      r1mmu2 (j) = (cos(glmf%rlat (j)))**2
 288      r1mmu2h(j) = (cos(rlath(j)))**2
 289    end do
 290
 291    !- 4.2  Bi-periodize and symmetrize Metric coefficients
 292    call agd_mach(rdmu   (1:nj_ext), & ! INOUT
 293                   1, nj_ext,1)        ! IN
 294    call agd_mach(rdmuh  (1:nj_ext), & ! INOUT
 295                   1, nj_ext,1)        ! IN 
 296    call agd_mach(r1mmu2 (1:nj_ext), & ! INOUT
 297                   1, nj_ext,1)        ! IN
 298    call agd_mach(r1mmu2h(1:nj_ext), & ! INOUT
 299                   1, nj_ext,1)        ! IN
 300
 301    call symmetrize_coef(rdmu   ) ! INOUT
 302    call symmetrize_coef(rdmuh  ) ! INOUT
 303    call symmetrize_coef(r1mmu2 ) ! INOUT
 304    call symmetrize_coef(r1mmu2h) ! INOUT
 305
 306    !- 4.3 Compute global factors
 307    glmf%dx  = 1.d0 / (ec_ra * glmf%rdlon)
 308
 309    allocate(glmf%cos2   ( 0:nj_ext+1))
 310    allocate(glmf%cos2h  ( 0:nj_ext+1))
 311    do j = 0, nj_ext+1
 312      glmf%cos2 (j) = r1mmu2 (j) / (ec_ra * rdmuh(j-1))
 313      glmf%cos2h(j) = r1mmu2h(j) / (ec_ra * rdmu (j  ))
 314    end do
 315
 316    allocate(glmf%idmuh  (0:nj_ext-1))
 317    allocate(glmf%idmu   (1:nj_ext  ))
 318    allocate(glmf%cos2vd (1:nj_ext  ))
 319    allocate(glmf%cos2hvd(1:nj_ext  ))
 320
 321    do j = 1, nj_ext
 322      glmf%idmuh  (j-1) = 1.d0 / rdmuh(j-1)
 323      glmf%idmu   (j)   = 1.d0 / rdmu   (j)
 324      glmf%cos2vd (j)   = 1.d0 / r1mmu2 (j)
 325      glmf%cos2hvd(j)   = 1.d0 / r1mmu2h(j)
 326    end do
 327
 328    !
 329    ! Conversion of wind images to physical winds and vice-versa
 330    ! N.B.: Those are geometrical factors of the COMPUTATIONAL grid 
 331    !        ==> only computational latitude variation...
 332    !
 333    allocate(glmf%conphy(nj_ext))
 334    allocate(glmf%conima(nj_ext))
 335
 336    call agd_mach(rrcos(1:nj_ext),  & ! INOUT
 337                  1, nj_ext,1)        ! IN
 338
 339    do j = 1, nj_ext
 340      glmf%conphy(j) = rrcos(j)                ! to go from Wind Images to true winds
 341      glmf%conima(j) = 1.0d0 / glmf%conphy(j)  ! to go from true winds to Wind Images
 342    end do
 343
 344    !
 345    !- 5. MPI partitionning
 346    !
 347    if ( mmpi_nprocs /= 0 ) then
 348       call mmpi_setup_lonbands(ni_ext,                      & ! IN
 349                                lonPerPE, lonPerPEmax, myLonBeg, myLonEnd ) ! OUT
 350
 351       call mmpi_setup_latbands(nj_ext,                      & ! IN
 352                                latPerPE, latPerPEmax, myLatBeg, myLatEnd ) ! OUT
 353    else
 354       ! This option is needed for the biper program
 355       lonPerPE = ni_ext
 356       myLonBeg = 1
 357       myLonEnd = ni_ext
 358       latPerPE = nj_ext
 359       myLatBeg = 1
 360       myLatEnd = nj_ext
 361       write(*,*)
 362       write(*,*) 'WARNING: the module will be executed in NO-MPI MODE!'
 363    end if
 364
 365    !
 366    !- 6.  Ending
 367    !
 368    deallocate(rlath  )
 369    deallocate(rrcos  )
 370    !deallocate(rrcosh )
 371    deallocate(rdmu   )
 372    deallocate(rdmuh  )
 373    deallocate(r1mmu2 )
 374    deallocate(r1mmu2h)
 375
 376    write(*,*)
 377    write(*,*) 'agd_SetupFromHCO: Done!'
 378
 379  end subroutine agd_SetupFromHCO

 382  ! symmetrize_coef
 383  !--------------------------------------------------------------------------
 384  subroutine symmetrize_coef(coef_inout)
 385    !
 386    !:Purpose: Extend symmetrically Metric coefficients.
 387    !
 388    implicit none
 389
 390    real(8), intent(inout) :: coef_inout(jstart:jend)
 391
 392    integer j
 393    
 394    do j = jstart, 0
 395      coef_inout(j) = coef_inout(nj_ext+j)
 396    end do
 397
 398    do j = nj_ext+1, jend
 399      coef_inout(j) = coef_inout(j-nj_ext) 
 400    end do
 401
 402  end subroutine symmetrize_coef

 404  !--------------------------------------------------------------------------
 405  ! agd_PsiChiToUV
 406  !--------------------------------------------------------------------------
 407  subroutine agd_PsiChiToUV(psi, chi, uphy, vphy, nk)
 408    implicit none
 409
 410    integer,          intent(in)  :: nk
 411
 412    real(8),          intent(in)  :: psi(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nk)
 413    real(8),          intent(in)  :: chi(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nk)
 414
 415    real(8),          intent(out) :: uphy(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nk)
 416    real(8),          intent(out) :: vphy(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nk)
 417
 418    real(8), allocatable :: psi_ext(:,:,:)
 419    real(8), allocatable :: chi_ext(:,:,:)
 420    real(8), allocatable :: uimg(:,:,:)
 421    real(8), allocatable :: vimg(:,:,:)
 422    real(8), allocatable :: uimgs(:,:,:)
 423    real(8), allocatable :: vimgs(:,:,:)
 424
 425    integer :: i,j,k
 426
 427    if ( hco_ext%global ) then
 428      call utl_abort('agd_PsiChiToUV: Not compatible with global grid')
 429    endif
 430
 431    if ( .not. initialized ) then
 432      call utl_abort('agd_PsiChiToUV: AnalysisGrid not initialized')
 433    endif
 434
 435    allocate(psi_ext( (myLonBeg-1):(myLonEnd+1), (myLatBeg-1):(myLatEnd+1), nk))
 436    allocate(chi_ext( (myLonBeg-1):(myLonEnd+1), (myLatBeg-1):(myLatEnd+1), nk))
 437    allocate(uimg   ( myLonBeg:myLonEnd        , myLatBeg:myLatEnd        , nk))
 438    allocate(vimg   ( myLonBeg:myLonEnd        , myLatBeg:myLatEnd        , nk))
 439    allocate(uimgs  ( (myLonBeg-1):myLonEnd    , myLatBeg:(myLatEnd+1)    , nk))
 440    allocate(vimgs  ( myLonBeg:(myLonEnd+1)    , (myLatBeg-1):myLatEnd    , nk))
 441
 442    !
 443    !- 1.  Symmetrize
 444    !
 445    call symmetrize( psi_ext,                                        & ! OUT
 446                     psi, myLonBeg, myLonEnd, myLatBeg, myLatEnd, nk ) ! IN
 447    call symmetrize( chi_ext,                                        & ! OUT
 448                     chi, myLonBeg, myLonEnd, myLatBeg, myLatEnd, nk ) ! IN
 449
 450    !
 451    !- 2.  Compute Wind on staggered grid
 452    !
 453
 454    !$OMP PARALLEL DO PRIVATE (k,j,i)
 455    do k = 1, nk
 456      !- 2.1 u-wind component
 457       do j = myLatBeg, myLatEnd+1
 458        do i = myLonBeg-1, myLonEnd
 459           uimgs(i,j,k) =   glmf%dx      * ( chi_ext(i+1,j,k) - chi_ext(i,j,k)   ) &
 460                          - glmf%cos2(j) * ( psi_ext(i,j,k)   - psi_ext(i,j-1,k) )
 461        end do
 462      end do
 463      !- 2.2 v-wind component
 464      do j = myLatBeg-1, myLatEnd
 465        do i = myLonBeg, myLonEnd+1
 466           vimgs(i,j,k) =   glmf%cos2h(j) * ( chi_ext(i,j+1,k) - chi_ext(i,j,k)   ) &
 467                          + glmf%dx       * ( psi_ext(i,j,k)   - psi_ext(i-1,j,k) )
 468        end do
 469      end do
 470    end do
 471    !$OMP END PARALLEL DO
 472
 473    !
 474    !- 3.  Move to collocated (scalar) grid
 475    !
 476    call uvStagToColloc( uimgs, vimgs,                              & ! IN
 477                         uimg , vimg ,                              & ! OUT
 478                         myLonBeg, myLonEnd, myLatBeg, myLatEnd, nk ) ! IN
 479
 480    !
 481    !- 4.  Convert Wind images to Physical (true) winds
 482    !
 483    !$OMP PARALLEL DO PRIVATE (j)
 484    do j = myLatBeg, myLatEnd
 485      uphy(:,j,:) =  glmf%conphy(j) * uimg(:,j,:)
 486      vphy(:,j,:) =  glmf%conphy(j) * vimg(:,j,:)
 487    end do
 488    !$OMP END PARALLEL DO
 489
 490    deallocate(psi_ext)
 491    deallocate(chi_ext)
 492    deallocate(uimg)
 493    deallocate(vimg)
 494    deallocate(uimgs)
 495    deallocate(vimgs)
 496
 497  end subroutine agd_PsiChiToUV

 499  !--------------------------------------------------------------------------
 500  ! agd_PsiChiToUVAdj
 501  !--------------------------------------------------------------------------
 502  subroutine agd_PsiChiToUVAdj(psi, chi, uphy, vphy, nk)
 503    implicit none
 504
 505    integer,          intent(in)  :: nk
 506
 507    real(8),          intent(out) :: psi(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nk)
 508    real(8),          intent(out) :: chi(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nk)
 509    
 510    real(8),          intent(in)  :: uphy(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nk)
 511    real(8),          intent(in)  :: vphy(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nk)
 512
 513    real(8), allocatable :: psi_ext(:,:,:)
 514    real(8), allocatable :: chi_ext(:,:,:)
 515    real(8), allocatable :: uimg(:,:,:)
 516    real(8), allocatable :: vimg(:,:,:)
 517    real(8), allocatable :: uimgs(:,:,:)
 518    real(8), allocatable :: vimgs(:,:,:)
 519
 520    integer :: i,j,k
 521
 522    if ( hco_ext%global ) then
 523      call utl_abort('agd_PsiChiToUVAdj: Not compatible with global grid')
 524    endif
 525
 526    if ( .not. initialized ) then
 527      call utl_abort('agd_PsiChiToUV: AnalysisGrid not initialized')
 528    endif
 529
 530    allocate(psi_ext( (myLonBeg-1):(myLonEnd+1), (myLatBeg-1):(myLatEnd+1), nk))
 531    allocate(chi_ext( (myLonBeg-1):(myLonEnd+1), (myLatBeg-1):(myLatEnd+1), nk))
 532    allocate(uimg   ( myLonBeg:myLonEnd        , myLatBeg:myLatEnd        , nk))
 533    allocate(vimg   ( myLonBeg:myLonEnd        , myLatBeg:myLatEnd        , nk))
 534    allocate(uimgs  ( (myLonBeg-1):myLonEnd    , myLatBeg:(myLatEnd+1)    , nk))
 535    allocate(vimgs  ( myLonBeg:(myLonEnd+1)    , (myLatBeg-1):myLatEnd    , nk))
 536
 537    !
 538    !- 4.  Convert Physical (true) winds to Wind images
 539    !
 540    !$OMP PARALLEL DO PRIVATE (j)
 541    do j = myLatBeg, myLatEnd
 542      uimg(:,j,:) =  glmf%conphy(j) * uphy(:,j,:)
 543      vimg(:,j,:) =  glmf%conphy(j) * vphy(:,j,:)
 544    end do
 545    !$OMP END PARALLEL DO
 546
 547    !
 548    !- 3.  Move to stagerred grid
 549    !
 550    uimgs(:,:,:) = 0.d0
 551    vimgs(:,:,:) = 0.d0
 552
 553    call uvStagToCollocAdj( uimgs, vimgs,                              & ! OUT
 554                            uimg , vimg ,                              & ! IN
 555                            myLonBeg, myLonEnd, myLatBeg, myLatEnd, nk ) ! IN
 556
 557    !
 558    !- 2.  Compute Psi-Chi
 559    !
 560    chi_ext(:,:,:) = 0.d0
 561    psi_ext(:,:,:) = 0.d0
 562
 563    !$OMP PARALLEL DO PRIVATE (k,j,i)
 564    do k = 1, nk
 565      !- 2.2 from v-wind component
 566      do j = myLatEnd, myLatBeg-1, -1
 567        do i = myLonEnd+1, myLonBeg, -1
 568          chi_ext(i  ,j+1,k) = chi_ext(i  ,j+1,k) + vimgs(i,j,k) * glmf%cos2h(j)
 569          chi_ext(i  ,j  ,k) = chi_ext(i  ,j  ,k) - vimgs(i,j,k) * glmf%cos2h(j)
 570          psi_ext(i  ,j  ,k) = psi_ext(i  ,j  ,k) + vimgs(i,j,k) * glmf%dx
 571          psi_ext(i-1,j  ,k) = psi_ext(i-1,j  ,k) - vimgs(i,j,k) * glmf%dx
 572        end do
 573      end do
 574      !- 2.1 from u-wind component
 575      do j = myLatEnd+1, myLatBeg, -1
 576        do i = myLonEnd, myLonBeg-1, -1
 577          chi_ext(i+1,j  ,k) = chi_ext(i+1,j  ,k) + uimgs(i,j,k) * glmf%dx
 578          chi_ext(i  ,j  ,k) = chi_ext(i  ,j  ,k) - uimgs(i,j,k) * glmf%dx
 579          psi_ext(i  ,j  ,k) = psi_ext(i  ,j  ,k) - uimgs(i,j,k) * glmf%cos2(j)
 580          psi_ext(i  ,j-1,k) = psi_ext(i  ,j-1,k) + uimgs(i,j,k) * glmf%cos2(j)
 581        end do
 582     end do
 583    end do
 584    !$OMP END PARALLEL DO
 585
 586    !
 587    !- 1.  De-Symmetrize
 588    !
 589    chi(:,:,:) = 0.d0
 590    psi(:,:,:) = 0.d0
 591
 592    call symmetrizeAdj( psi_ext,                                   & ! IN
 593                        psi,                                       & ! OUT
 594                        myLonBeg, myLonEnd, myLatBeg, myLatEnd, nk ) ! IN
 595    call symmetrizeAdj( chi_ext,                                   & ! IN
 596                        chi,                                       & ! OUT
 597                        myLonBeg, myLonEnd, myLatBeg, myLatEnd, nk ) ! IN
 598
 599    deallocate(psi_ext)
 600    deallocate(chi_ext)
 601    deallocate(uimg)
 602    deallocate(vimg)
 603    deallocate(uimgs)
 604    deallocate(vimgs)
 605
 606  end subroutine agd_PsiChiToUVAdj

 667  subroutine symmetrize(field_out, field_in, iBeg, iEnd, jBeg, jEnd, nk)
 668    !
 669    !:Purpose: Extend symmetrically outside 1 grid point all around LAM-boundary
 670    !          ready for finite differences
 671    !
 672    implicit none
 673    integer, intent(in) :: iBeg, iEnd, jBeg, jEnd, nk
 674    real(8), intent(out):: field_out(iBeg-1:iEnd+1,jBeg-1:jEnd+1, nk)

 677    real(8), allocatable :: field_8(:,:,:)
 678
 679    integer :: ni,nj

 682    nj = jEnd-jBeg+1
 683

 685    field_8(:,:,:) = 0.d0
 686
 687    field_8(1:ni,1:nj,:) = field_in(iBeg:iEnd,jBeg:jEnd,:)

 690                             0,ni+1,0,nj+1,ni,nj,nk, & ! IN
 691                             1,1,.true.,.true.,ni,0)   ! IN
 692
 693    field_out(iBeg-1:iEnd+1,jBeg-1:jEnd+1,:) = field_8(0:ni+1,0:nj+1,:)
 694    deallocate(field_8)
 695
 696  end subroutine symmetrize

 700  !--------------------------------------------------------------------------
 701  subroutine symmetrizeAdj(field_in, field_out, iBeg, iEnd, jBeg, jEnd, nk)
 702    !
 703    !:Purpose: Adjoint of sub. symmetrize.
 704    !
 705    implicit none
 706    integer, intent(in)  :: iBeg, iEnd, jBeg, jEnd, nk
 707    real(8), intent(in)  :: field_in(iBeg-1:iEnd+1,jBeg-1:jEnd+1, nk)
 708    real(8), intent(out) :: field_out(iBeg:iEnd,jBeg:jEnd,nk)

 711
 712    integer :: i,j,k,ni,nj
 713
 714    ni = iEnd-iBeg+1
 715    nj = jEnd-jBeg+1
 716
 717    allocate(field_8(0:ni+1,0:nj+1, nk))
 718    field_8(0:ni+1,0:nj+1,:) = field_in(iBeg-1:iEnd+1,jBeg-1:jEnd+1,:)
 719
 720    call RPN_COMM_adj_halo8(field_8,                 & ! INOUT
 721                            0,ni+1,0,nj+1,ni,nj,nk,  & ! IN
 722                            1,1,.true.,.true.,ni,0)    ! IN

 725    do k = 1, nk
 726       do j = jBeg, jEnd
 727          do i = iBeg, iEnd
 728             field_out(i,j,k) = field_out(i,j,k) + field_8(i-iBeg+1,j-jBeg+1,k)
 729          end do
 730       end do
 731    end do
 732    !$OMP END PARALLEL DO
 733
 734    deallocate(field_8)
 735
 736  end subroutine symmetrizeAdj

 739  ! agd_Mach
 740  !--------------------------------------------------------------------------
 741  subroutine agd_mach(gd,ni,nj,nk)
 742    !
 743    !:Purpose:  [to be completed]
 744    !
 745    !:Arguments:
 746    !       :ni: Maximum I-dimension where the input array is assumed to carry
 747    !            information.  Will be used as I-limit where backward
 748    !            derivatives will be evaluated
 749    !
 750    !       :nj: Maximum J-dimension where the input array is assumed to carry
 751    !            information. Will be used as J-limit where backward derivatives
 752    !            will be evaluated
 753    !
 754    implicit none
 755
 756    integer,          intent(in)    :: ni, nj, nk
 757    real(8),          intent(inout) :: gd(ni,nj,nk)
 758
 759    integer :: istart, jstart
 760    integer :: i,j,k
 761
 762    real(8) :: con, xp, yp, a0, a1, b1, b2
 763    real(8) :: deriv_istart, deriv_jstart, deriv_i0, deriv_j0, del
 764
 765    if ( hco_ext%global ) then
 766      call utl_abort('agd_Mach: Not compatible with global grid')
 767    endif
 768
 769    if ( .not. initialized ) then
 770      call utl_abort('agd_Mach: AnalysisGrid not initialized')
 771    endif
 772
 773    if ( (ni /= 1 .and. ni /= ni_ext) .or. &
 774         (nj /= 1 .and. nj /= nj_ext) ) then
 775      call utl_abort('agd_Mach: Invalid Dimensions')
 776    end if
 777
 778    !$OMP PARALLEL
 779    !$OMP DO PRIVATE (k,j,i,istart,jstart,con,a0,a1,b1,b2,del,deriv_istart,deriv_i0,deriv_jstart,deriv_j0,xp,yp)
 780    do k = 1, nk
 781    !
 782    !- 1.  Periodicized in x-direction from ni_core to ni
 783    !    
 784    if ( ni > 1 ) then
 785      istart = ni_core - 2  ! I-limit where backward derivatives will be evaluated
 786      con = 1.d0 / ( glmf%rdlon * MPC_DEGREES_PER_RADIAN_R8 * 111.d+3)
 787      do j = 1, nj
 788        a0           = 0.5d0 * ( gd(istart,j,k)  + gd(1,j,k)       )
 789        a1           = 0.5d0 * ( gd(istart,j,k)  - gd(1,j,k)       )
 790        deriv_istart = con   * ( gd(istart,j,k)  - gd(istart-1,j,k))
 791        deriv_i0     = con   * ( gd(2,j,k)       - gd(1,j,k)       )
 792        b1           = 0.5d0 * ( deriv_istart  - deriv_i0       )
 793        b2           = 0.25d0* ( deriv_istart  + deriv_i0       )
 794        del          = real(ni_ext_per-istart,8)
 795        do i = istart, ni
 796          xp = MPC_PI_R8 * real(i-istart,8) / del
 797          gd(i,j,k) = a0 + a1*cos(xp) + b1*sin(xp) + b2*sin(2.d0*xp)
 798        end do
 799      end do
 800    end if
 801    !
 802    !- 2.  Periodicized in y-direction from nj_core to nj
 803    !
 804    if ( nj > 1 ) then
 805      jstart = nj_core - 2
 806      con = 1.d0 / (glmf%rdlat * MPC_DEGREES_PER_RADIAN_R8 * 111.d+3)
 807      do i = 1, ni
 808        a0           = 0.5d0 * ( gd(i,jstart,k) + gd(i,1,k)       )
 809        a1           = 0.5d0 * ( gd(i,jstart,k) - gd(i,1,k)       )
 810        deriv_jstart = con   * ( gd(i,jstart,k) - gd(i,jstart-1,k))
 811        deriv_j0     = con   * ( gd(i,2,k)      - gd(i,1,k)       )
 812        b1           = 0.5d0 * ( deriv_jstart - deriv_j0     )
 813        b2           = 0.25d0* ( deriv_jstart + deriv_j0     )
 814        del          = real(nj_ext_per-jstart,8)
 815        do j = jstart , nj
 816          yp = MPC_PI_R8 * real(j-jstart,8) / del
 817          gd(i,j,k) = a0 + a1*cos(yp) + b1*sin(yp) + b2*sin(2.d0*yp)
 818        end do
 819      end do
 820    end if
 821
 822    end do
 823    !$OMP END DO
 824    !$OMP END PARALLEL
 825
 826  end subroutine agd_mach

 916  !--------------------------------------------------------------------------
 917  ! agd_UVToVortDiv
 918  !--------------------------------------------------------------------------
 919  subroutine agd_UVToVortDiv(Vorticity, Divergence, uphy, vphy, nk)
 920    implicit none
 921
 922    integer,          intent(in)  :: nk
 923
 924    real(8),          intent(out)  :: Vorticity (myLonBeg:myLonEnd,myLatBeg:myLatEnd,nk)
 925    real(8),          intent(out)  :: Divergence(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nk)
 926
 927    real(8),          intent(in) :: uphy(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nk)
 928    real(8),          intent(in) :: vphy(myLonBeg:myLonEnd,myLatBeg:myLatEnd,nk)
 929
 930    real(8), allocatable :: uimg(:,:,:)
 931    real(8), allocatable :: vimg(:,:,:)
 932    real(8), allocatable :: uimg_sym(:,:,:)
 933    real(8), allocatable :: vimg_sym(:,:,:)
 934
 935    integer :: i,j,k
 936
 937    if ( hco_ext%global ) then
 938      call utl_abort('agd_UVToVortDiv: Not compatible with global grid')
 939    endif
 940
 941    if ( .not. initialized ) then
 942      call utl_abort('agd_UVToVortDiv: AnalysisGrid not initialized')
 943    endif
 944
 945    allocate(uimg_sym( (myLonBeg-1):(myLonEnd+1), (myLatBeg-1):(myLatEnd+1), nk))
 946    allocate(vimg_sym( (myLonBeg-1):(myLonEnd+1), (myLatBeg-1):(myLatEnd+1), nk))
 947    allocate(uimg    ( myLonBeg:myLonEnd        , myLatBeg:myLatEnd        , nk))
 948    allocate(vimg    ( myLonBeg:myLonEnd        , myLatBeg:myLatEnd        , nk))
 949
 950    !
 951    !- 2.  Convert Physical (true) winds to Wind images
 952    !
 953    !$OMP PARALLEL DO PRIVATE (j)
 954    do j = myLatBeg, myLatEnd
 955          uimg(:,j,:) =  glmf%conima(j) * uphy(:,j,:)
 956          vimg(:,j,:) =  glmf%conima(j) * vphy(:,j,:)
 957    end do
 958    !$OMP END PARALLEL DO
 959
 960    !
 961    !- 3.  Symmetrize
 962    !
 963    call symmetrize( uimg_sym,                                        & ! OUT
 964                     uimg, myLonBeg, myLonEnd, myLatBeg, myLatEnd, nk ) ! IN
 965    call symmetrize( vimg_sym,                                        & ! OUT
 966                     vimg, myLonBeg, myLonEnd, myLatBeg, myLatEnd, nk ) ! IN
 967    deallocate(uimg,vimg)
 968
 969    !
 970    !- 4.  Compute Vorticity and Divergence
 971    !
 972
 973    !$OMP PARALLEL DO PRIVATE (k,j,i)
 974     do k = 1, nk
 975       do j = myLatBeg, myLatEnd
 976         do i = myLonBeg, myLonEnd
 977           vorticity (i,j,k) = &
 978                glmf%cos2hvd(j) * glmf%dx       * (vimg_sym(i+1,j,k)-vimg_sym(i,j,k)) - &
 979                       ec_r1sa  * glmf%idmu(j)  * (uimg_sym(i,j+1,k)-uimg_sym(i,j,k))
 980
 981           divergence(i,j,k) = &
 982                glmf%cos2vd(j) * glmf%dx        * (uimg_sym(i,j,k)-uimg_sym(i-1,j,k)) + &
 983                       ec_r1sa * glmf%idmuh(j-1)* (vimg_sym(i,j,k)-vimg_sym(i,j-1,k))
 984         end do
 985       end do

 997  subroutine agd_createLamTemplateGrids(templateFileName, hco_core, vco, &
 998                                        grd_ext_x, grd_ext_y)
 999    implicit none
1000
1001    character(len=*), intent(in) :: templateFileName
1002    type(struct_hco) :: hco_core
1003    type(struct_vco) :: vco
1004    integer         , intent(in) :: grd_ext_x
1005    integer         , intent(in) :: grd_ext_y
1006
1007    integer :: ni_ext, nj_ext, i, j, lev, ni, nj, nk
1008    integer :: iun = 0
1009    integer :: ier, fnom, fstouv, fstfrm, fclos, fstecr
1010
1011    real(8), allocatable :: Field2d(:,:)
1012    real(8), allocatable :: lat_ext(:)
1013    real(8), allocatable :: lon_ext(:)
1014
1015    real(4), allocatable :: dummy2D(:,:)
1016
1017    real(8) :: dlat, dlon
1018    real(4) :: work
1019
1020    integer :: dateo,npak,status
1021    integer :: ip1,ip2,ip3,deet,npas,datyp,ig1,ig2,ig3,ig4
1022    integer :: ig1_tictac,ig2_tictac,ig3_tictac,ig4_tictac
1023
1024    character(len=1)  :: grtyp
1025    character(len=2)  :: typvar
1026    character(len=12) :: etiket
1027
1028    !
1029    !- 1.  Opening the output template file
1030    !
1031    ier = fnom(iun, trim(templateFileName), 'RND', 0)
1032    ier = fstouv(iun, 'RND')
1033
1034    npak     = -32
1035
1036    !
1037    !- 2.  Writing the core grid (Ensemble) template
1038    !
1039
1040    !- 2.1 Tic-Tac
1041    deet     =  0
1042    ip1      =  hco_core%ig1
1043    ip2      =  hco_core%ig2
1044    ip3      =  hco_core%ig3
1045    npas     =  0
1046    datyp    =  1
1047    grtyp    =  hco_core%grtypTicTac
1048    typvar   = 'X'
1049    etiket   = 'COREGRID'
1050    dateo =  0
1051
1052    call cxgaig ( grtyp,                                          & ! IN
1053         ig1_tictac, ig2_tictac, ig3_tictac, ig4_tictac, & ! OUT
1054         real(hco_core%xlat1), real(hco_core%xlon1),   & ! IN
1055         real(hco_core%xlat2), real(hco_core%xlon2)  )   ! IN
1056
1057    ig1      =  ig1_tictac
1058    ig2      =  ig2_tictac
1059    ig3      =  ig3_tictac
1060    ig4      =  ig4_tictac
1061
1062    ier = utl_fstecr(hco_core%lon*MPC_DEGREES_PER_RADIAN_R8, npak, &
1063         iun, dateo, deet, npas, hco_core%ni, 1, 1, ip1,    &
1064         ip2, ip3, typvar, '>>', etiket, grtyp, ig1,          &
1065         ig2, ig3, ig4, datyp, .true.)
1066
1067    ier = utl_fstecr(hco_core%lat*MPC_DEGREES_PER_RADIAN_R8, npak, &
1068         iun, dateo, deet, npas, 1, hco_core%nj, 1, ip1,    &
1069         ip2, ip3, typvar, '^^', etiket, grtyp, ig1,          &
1070         ig2, ig3, ig4, datyp, .true.)
1071
1072    !- 2.2 2D Field
1073    allocate(Field2d(hco_core%ni,hco_core%nj))
1074    Field2d(:,:) = 10.d0
1075
1076    deet      =  0
1077    ip1       =  0
1078    ip2       =  0
1079    ip3       =  0
1080    npas      =  0
1081    datyp     =  1
1082    grtyp     =  hco_core%grtyp
1083    typvar    = 'A'
1084    etiket    = 'COREGRID'
1085    dateo     =  0
1086    ig1       =  hco_core%ig1
1087    ig2       =  hco_core%ig2
1088    ig3       =  hco_core%ig3
1089    ig4       =  hco_core%ig4
1090
1091    ier = utl_fstecr(Field2d, npak,                                    &
1092         iun, dateo, deet, npas, hco_core%ni, hco_core%nj, 1, ip1, &
1093         ip2, ip3, typvar, 'P0', etiket, grtyp, ig1,                &
1094         ig2, ig3, ig4, datyp, .true.)
1095
1096    deallocate(Field2d)
1097
1098    !
1099    !- 3.  Create and Write the extended grid (Analysis) template
1100    !
1101    ni_ext = hco_core%ni + grd_ext_x
1102    nj_ext = hco_core%nj + grd_ext_y
1103
1104    !- 3.1 Tic-Tac
1105    allocate(lon_ext(ni_ext))
1106    allocate(lat_ext(nj_ext))
1107
1108    !- Copy core grid info
1109    lon_ext(1:hco_core%ni) = hco_core%lon(:) 
1110    lat_ext(1:hco_core%nj) = hco_core%lat(:)
1111
1112    !- Extend the lat lon
1113    dlon = hco_core%lon(2) - hco_core%lon(1) 
1114    do i = hco_core%ni + 1, ni_ext
1115      lon_ext(i) = lon_ext(hco_core%ni) + (i - hco_core%ni) * dlon
1116    end do
1117
1118    dlat = hco_core%lat(2) - hco_core%lat(1) 
1119    do j = hco_core%nj + 1, nj_ext
1120      lat_ext(j) = lat_ext(hco_core%nj) + (j - hco_core%nj) * dlat
1121    end do
1122
1123    !- Write
1124    deet     =  0
1125    ip1      =  hco_core%ig1 + 100 ! Must be different from the core grid
1126    ip2      =  hco_core%ig2 + 100 ! Must be different from the core grid
1127    if (hco_core%ig3 > 0) then
1128      ip3      =  hco_core%ig3 + 100 ! Must be different from the core grid
1129    else
1130      ip3      =  0
1131    end if
1132    npas     =  0
1133    datyp    =  1
1134    grtyp    =  hco_core%grtypTicTac
1135    typvar   = 'X'
1136    etiket   = 'ANALYSIS'
1137    dateo    =  0
1138    ig1      =  ig1_tictac
1139    ig2      =  ig2_tictac
1140    ig3      =  ig3_tictac
1141    ig4      =  ig4_tictac
1142
1143    ier = utl_fstecr(lon_ext*MPC_DEGREES_PER_RADIAN_R8, npak, &
1144         iun, dateo, deet, npas, ni_ext, 1, 1, ip1,  &
1145         ip2, ip3, typvar, '>>', etiket, grtyp, ig1,    &
1146         ig2, ig3, ig4, datyp, .true.)
1147
1148    ier = utl_fstecr(lat_ext*MPC_DEGREES_PER_RADIAN_R8, npak, &
1149         iun, dateo, deet, npas, 1, nj_ext, 1, ip1,  &
1150         ip2, ip3, typvar, '^^', etiket, grtyp, ig1,    &
1151         ig2, ig3, ig4, datyp, .true.)
1152
1153    deallocate(lon_ext)
1154    deallocate(lat_ext)
1155
1156    !- 3.2 2D Field
1157    allocate(Field2d(ni_ext,nj_ext))
1158    Field2d(:,:) = 10.d0
1159
1160    deet      =  0
1161    ip1       =  0
1162    ip2       =  0
1163    ip3       =  0
1164    npas      =  0
1165    datyp     =  1
1166    grtyp     =  hco_core%grtyp
1167    typvar    = 'A'
1168    etiket    = 'ANALYSIS'
1169    dateo     =  0
1170    ig1       =  hco_core%ig1 + 100 ! Must be different from the core grid
1171    ig2       =  hco_core%ig2 + 100 ! Must be different from the core grid
1172    if (hco_core%ig3 > 0) then
1173      ig3      =  hco_core%ig3 + 100 ! Must be different from the core grid
1174    else
1175      ig3      =  0
1176    end if
1177    ig4       =  0
1178
1179    ier = utl_fstecr(Field2d, npak,                                  &
1180         iun, dateo, deet, npas, ni_ext, nj_ext, 1, ip1, &
1181         ip2, ip3, typvar, 'P0', etiket, grtyp, ig1,     &
1182         ig2, ig3, ig4, datyp, .true.)
1183
1184    deallocate(Field2d)
1185
1186    !
1187    !- 4. Write the vertical grid description
1188    !
1189
1190    if (vco%vgridPresent) then
1191      !- 4.1 Write the toc-toc
1192      status = vgd_write(vco%vgrid,iun,'fst')
1193      
1194      if ( status /= VGD_OK ) then
1195        call utl_abort('createLamTemplateGrids: ERROR with vgd_write')
1196      end if
1197      
1198      !- 4.2 Write a dummy 2D field for each MM and TH levels
1199      npak   = -12
1200      dateo  = 0
1201      deet   = 0
1202      npas   = 0
1203      ni     = 4
1204      nj     = 2
1205      nk     = 1
1206      ip2    = 0
1207      ip3    = 0
1208      typvar = 'A'
1209      etiket = 'VERTICALGRID'
1210      grtyp  = 'G'
1211      ig1    = 0
1212      ig2    = 0
1213      ig3    = 0
1214      ig4    = 0
1215      datyp  = 1
1216      
1217      allocate(dummy2D(ni,nj))
1218      dummy2D(:,:) = 0.0
1219      
1220      do lev = 1, vco%nlev_M
1221        ip1 = vco%ip1_M(lev)
1222        ier = fstecr(dummy2D, work, npak, iun, dateo, deet, npas, ni, nj, &
1223                     nk, ip1, ip2, ip3, typvar, 'MM', etiket, grtyp,              &
1224                     ig1, ig2, ig3, ig4, datyp, .true.)
1225      end do
1226      do lev = 1, vco%nlev_T
1227        ip1 = vco%ip1_T(lev)
1228        ier = fstecr(dummy2D, work, npak, iun, dateo, deet, npas, ni, nj, &
1229                     nk, ip1, ip2, ip3, typvar, 'TH', etiket, grtyp,              &
1230                     ig1, ig2, ig3, ig4, datyp, .true.)
1231      end do
1232
1233      deallocate(dummy2D)
1234    end if ! vco%vgridPresent
1235
1236    !
1237    !- 5.  Closing the output template file
1238    !
1239    ier = fstfrm(iun)
1240    ier = fclos (iun)

1242  end subroutine agd_createLamTemplateGrids
