  95  subroutine lbhi_Setup( hco_anl_in, hco_core_in, vco_anl_in, cvDim_out )
  96    implicit none
  97
  98    type(struct_hco), pointer, intent(in)    :: hco_anl_in
  99    type(struct_hco), pointer, intent(in)    :: hco_core_in
 100    type(struct_vco), pointer, intent(in)    :: vco_anl_in
 101    integer,          intent(out)   :: cvDim_out
 102
 103    integer  :: var
 104    integer  :: ntrunc
 105
 106    integer  :: iu_bstats = 0
 107    integer  :: iu_flnml = 0
 108
 109    integer  :: ier, fnom, fstouv, fstfrm, fclos, levIndex, nLev
 110
 111    logical  :: FileExist
 112
 113    type(struct_vco), pointer :: vco_file => null()
 114
 115    !namelist
 116    NAMELIST /NAMBHI/ntrunc,scaleFactor
 117
 118    write(*,*)
 119    write(*,*) 'lbhi_Setup: Starting...'
 120
 121    !
 122    !- 0.  Read namelist options
 123    !
 124    ntrunc         = 75     ! default values
 125    scaleFactor(:) =  0.0d0 ! default values
 126
 127    ier = fnom(iu_flnml,'./flnml','FTN+SEQ+R/O',0)
 128    write(*,*)
 129    write(*,*) 'lbhi_setup: Reading namelist, ier = ',ier
 130    read(iu_flnml,nml=nambhi)
 131    write(*,nml=nambhi)
 132    ier = fclos(iu_flnml)
 133
 134    nLev = max(max(vco_anl_in%nlev_M,vco_anl_in%nlev_T),1)
 135
 136    do levIndex = 1, nLev
 137      if ( scaleFactor(levIndex) > 0.0d0 ) then 
 138        scaleFactor(levIndex) = sqrt(scaleFactor(levIndex))
 139      else
 140        scaleFactor(levIndex) = 0.0d0
 141      end if
 142    end do
 143
 144    write(*,*) ' sum(scaleFactor) : ',sum(scaleFactor(1:nLev))
 145    if ( sum(scaleFactor(1:nLev)) == 0.0d0 ) then
 146      write(*,*) 'lambmatrixHI: scaleFactor=0, skipping rest of setup'
 147      cvDim_out   = 0
 148      return
 149    end if
 150
 151    !- Setup the LAM analysis grid metrics
 152    call agd_SetupFromHCO( hco_anl_in, hco_core_in ) ! IN
 153    
 154    trunc = ntrunc
 155    write(*,*)
 156    write(*,*) 'Spectral TRUNCATION = ', trunc
 157
 158    !
 159    !- 1.  Open the background stats file
 160    !
 161    inquire(file=trim(BStatsFilename), exist=FileExist)
 162
 163    if ( FileExist ) then
 164      ier = fnom(iu_bstats,trim(BStatsFilename),'RND+OLD+R/O',0)
 165      if ( ier == 0 ) then
 166        write(*,*)
 167        write(*,*) 'Background Stats File :', trim(BStatsFilename)
 168        write(*,*) 'opened as unit file ',iu_bstats
 169        ier = fstouv(iu_bstats,'RND+OLD')
 170      else
 171        write(*,*)
 172        write(*,*) 'lbhi_Setup: Error in opening the background stats file'
 173        write(*,*) trim(BStatsFilename)
 174        call utl_abort('lbhi_Setup')
 175      end if
 176    else
 177      write(*,*)
 178      write(*,*) 'lbhi_Setup: The background stats file DOES NOT EXIST'
 179      write(*,*) trim(BStatsFilename)
 180      call utl_abort('lbhi_Setup')
 181    end if
 182
 183    ! Check if analysisgrid and covariance file have the same vertical levels
 184    call vco_SetupFromFile( vco_file,      & ! OUT
 185                            BStatsFilename ) ! IN
 186    if (.not. vco_equal(vco_anl_in,vco_file)) then
 187      call utl_abort('lamBmatrixHI: vco from analysisgrid and cov file do not match')
 188    end if
 189
 190    !
 191    !- 2.  Set some variables
 192    !
 193    hco_bhi => hco_anl_in
 194    nlev_M  = vco_anl_in%nlev_M
 195    nlev_T  = vco_anl_in%nlev_T
 196
 197    !- Read variables and vertical grid info from the background stats file
 198    call lbhi_GetControlVariableInfo( iu_bstats ) ! IN
 199    call lbhi_GetHorizGridInfo()
 200
 201    nkgdim = 0
 202    do var = 1, nControlVariable
 203      allocate( ControlVariable(var)%GpStdDev (1:hco_bhi%ni, 1:hco_bhi%nj, 1:ControlVariable(var)%nlev) )
 204      allocate( ControlVariable(var)%ip1 (1:ControlVariable(var)%nlev) )
 205
 206      if (ControlVariable(var)%GridType == 'TH') then
 207         ControlVariable(var)%ip1(:) = vco_anl_in%ip1_T(:)
 208      elseif (ControlVariable(var)%GridType == 'MM') then
 209         ControlVariable(var)%ip1(:) = vco_anl_in%ip1_M(:)
 210      else
 211        ControlVariable(var)%ip1(:) = 0
 212      end if
 213
 214      ControlVariable(var)%kDimStart = nkgdim + 1
 215      nkgdim = nkgdim + ControlVariable(var)%nlev
 216      ControlVariable(var)%kDimEnd    = nkgdim
 217    end do
 218
 219    nksdim = nkgdim ! + nlev
 220
 221    allocate( bsqrt  (1:nksdim, 1:nksdim ,0:trunc) )
 222
 223    !- 2.2 Initialized the LAM spectral transform
 224    call mmpi_setup_lonbands(hco_bhi%ni,                  & ! IN
 225                               lonPerPE, lonPerPEmax, myLonBeg, myLonEnd ) ! OUT
 226
 227    call mmpi_setup_latbands(hco_bhi%nj,                  & ! IN
 228                               latPerPE, latPerPEmax, myLatBeg, myLatEnd ) ! OUT
 229
 230    call lst_Setup(lst_bhi,                         & ! OUT
 231                   hco_bhi%ni, hco_bhi%nj,          & ! IN
 232                   hco_bhi%dlon, trunc,             & ! IN
 233                   'LatLonMN', maxlevels_opt=nksdim)  ! IN
 234
 235    cvDim     = nkgdim * lst_bhi%nla * lst_bhi%nphase
 236    cvDim_out = cvDim
 237
 238    ! also compute mpiglobal control vector dimension
 239    call rpn_comm_allreduce(cvDim,cvDim_mpiglobal,1,"mpi_integer","mpi_sum","GRID",ier)
 240
 241    !- 2.3 Initialized the Wind spectral transform
 242    if ( trim(WindTransform) == 'VortDiv' ) then
 243       call lst_Setup(lst_wind,                       & ! OUT
 244                      hco_bhi%ni, hco_bhi%nj,         & ! IN
 245                      hco_bhi%dlon, trunc,            & ! IN
 246                      'LatLonMN', maxlevels_opt=nlev_M) ! IN
 247    end if
 248
 249    !
 250    !- 3.  Read info from the background error statistics file
 251    !
 252    call lbhi_ReadStats( iu_bstats )  ! IN
 253
 254    !
 255    !- 4.  Close the background stats files
 256    !
 257    ier = fstfrm(iu_bstats)
 258    ier = fclos (iu_bstats)
 259
 260    !
 261    !- 6.  Ending
 262    !
 263    initialized = .true.
 264
 265    write(*,*)
 266    write(*,*) 'lbhi_Setup: Done!'
 267
 268  end subroutine lbhi_Setup

 273  subroutine lbhi_GetControlVariableInfo( iu_bstats )
 274    implicit none
 275
 276    integer, intent(in) :: iu_bstats
 277
 278    integer :: key, fstinf, fstlir, fstlir_s
 279    integer :: ni, nj, nlev
 280    integer :: dateo, nk
 281    integer :: ip1, ip2, ip3
 282    integer :: var
 283
 284    character(len=4 )      :: nomvar
 285    character(len=2 )      :: typvar
 286    character(len=12)      :: etiket
 287
 288    character(len=4), allocatable :: ControlModelVarnameList(:)
 289    character(len=4), allocatable :: ControlBhiVarnameList  (:)
 290    character(len=2), allocatable :: ControlVarGridTypeList (:)
 291    integer, allocatable          :: ControlVarNlevList     (:)
 292
 293    !
 294    !- 1.  How Many Control Variables do we have?
 295    !
 296    dateo  = -1
 297    etiket = 'NLEV'
 298    ip1    = -1
 299    ip2    = -1
 300    ip3    = -1
 301    typvar = ' '
 302    nomvar = 'CVL'
 303
 304    key = fstinf( iu_bstats,                                  & ! IN
 305                  ni, nj, nk,                                 & ! OUT
 306                  dateo, etiket, ip1, ip2, ip3, typvar, nomvar )! IN
 307
 308    if (key < 0) then
 309      write(*,*)
 310      write(*,*) 'lbhi_GetControlVariableInfo: Unable to find variable =',nomvar
 311      call utl_abort('lbhi_GetControlVariableInfo')
 312    end if
 313
 314    nControlVariable = ni
 315    write(*,*)
 316    write(*,*) 'Number of Control Variables found = ', nControlVariable
 317
 318    allocate(ControlModelVarnameList(nControlVariable))
 319    allocate(ControlBhiVarnameList  (nControlVariable))
 320    allocate(ControlVarGridTypeList (nControlVariable))
 321    allocate(ControlVarNlevList     (nControlVariable))
 322
 323    !
 324    !- 2. Read the info from the input file
 325    !
 326    nomvar = 'CVN'
 327
 328    etiket = 'MODEL'
 329    key = fstlir_s(ControlModelVarnameList,                    & ! OUT 
 330                   iu_bstats,                                  & ! IN
 331                   ni, nj, nlev,                               & ! OUT
 332                   dateo, etiket, ip1, ip2, ip3, typvar,nomvar)  ! IN
 333    if (key < 0) then
 334      write(*,*)
 335      write(*,*) 'lbhi_GetControlVariableInfo: Cannot find variable ', nomvar 
 336      call utl_abort('lbhi_GetControlVariableInfo') 
 337    end if
 338
 339    etiket = 'B_HI'
 340    key = fstlir_s(ControlBhiVarnameList,                      & ! OUT 
 341                   iu_bstats,                                  & ! IN
 342                   ni, nj, nlev,                               & ! OUT
 343                   dateo, etiket, ip1, ip2, ip3, typvar,nomvar)  ! IN
 344    if (key < 0) then
 345      write(*,*)
 346      write(*,*) 'lbhi_GetControlVariableInfo: Cannot find variable ', nomvar
 347      call utl_abort('lbhi_GetControlVariableInfo') 
 348    end if
 349    
 350
 351    nomvar = 'CVL'
 352
 353    etiket = 'NLEV'
 354    key = fstlir  (ControlVarNlevList,                         & ! OUT 
 355                   iu_bstats,                                  & ! IN
 356                   ni, nj, nlev,                               & ! OUT
 357                   dateo, etiket, ip1, ip2, ip3, typvar,nomvar)  ! IN
 358    if (key < 0) then
 359      write(*,*)
 360      write(*,*) 'lbhi_GetControlVariableInfo: Cannot find variable ', nomvar
 361      call utl_abort('lbhi_GetControlVariableInfo') 
 362    end if
 363
 364    etiket = 'LEVTYPE'
 365    key = fstlir_s(ControlVarGridTypeList,                     & ! OUT 
 366                   iu_bstats,                                  & ! IN
 367                   ni, nj, nlev,                               & ! OUT
 368                   dateo, etiket, ip1, ip2, ip3, typvar,nomvar)  ! IN
 369    if (key < 0) then
 370      write(*,*)
 371      write(*,*) 'lbhi_GetControlVariableInfo: Cannot find variable ', nomvar
 372      call utl_abort('lbhi_GetControlVariableInfo') 
 373    end if
 374
 375    !
 376    !- 3. Introduce the info in the ControlVariable structure
 377    !
 378    do var = 1, nControlVariable
 379       ControlVariable(var)%nomvar(cv_model)= trim(ControlModelVarnameList(var))
 380       ControlVariable(var)%nomvar(cv_bhi)  = trim(ControlBhiVarnameList(var))
 381       ControlVariable(var)%nlev            = ControlVarNlevList(var)
 382       ControlVariable(var)%GridType        = trim(ControlVarGridTypeList(var))
 383
 384       if      (trim(ControlVariable(var)%nomvar(cv_model)) == 'UU' ) then 
 385          UWindID = var
 386       else if (trim(ControlVariable(var)%nomvar(cv_model)) == 'VV' ) then
 387          VWindID = var
 388       end if
 389
 390       if ( trim(ControlVariable(var)%nomvar(cv_model)) == 'LQ' ) then
 391         ControlVariable(var)%nomvar(cv_model) = 'HU'  ! PATCH because gridStateVector uses HU
 392       end if
 393
 394       write(*,*)
 395       write(*,*) 'nomvar(cv_model) = ', ControlVariable(var)%nomvar(cv_model)
 396       write(*,*) 'nomvar(cv_bhi)   = ', ControlVariable(var)%nomvar(cv_bhi)
 397       write(*,*) 'nlev             = ', ControlVariable(var)%nlev
 398       write(*,*) 'gridtype         = ', ControlVariable(var)%GridType
 399    end do
 400
 401    deallocate(ControlModelVarnameList)
 402    deallocate(ControlBhiVarnameList)
 403    deallocate(ControlVarGridTypeList)
 404    deallocate(ControlVarNlevList)
 405
 406    !
 407    !- 4. Error traps
 408    !
 409
 410    !- Make sure that all the control variables are present in GridStateVector
 411    do var = 1, nControlVariable
 412       if ( .not. gsv_varExist(varName=ControlVariable(var)%nomvar(cv_model)) ) then
 413          write(*,*)
 414          write(*,*) 'lbhi_GetControlVariableInfo: The following variable is MISSING in GridStateVector'
 415          write(*,*) trim(ControlVariable(var)%nomvar(cv_model))
 416          call utl_abort('lbhi_GetControlVariableInfo')
 417       end if
 418    end do
 419
 420    !
 421    !- 5.  Set the type of momentum control variables
 422    !
 423    if (UWindID /= -1 .and. VWindID /= -1) then
 424       if ( trim(ControlVariable(UWindID)%nomvar(cv_model)) == 'UU' .and. &
 425            trim(ControlVariable(UWindID)%nomvar(cv_bhi))   == 'PP' .and. &
 426            trim(ControlVariable(VWindID)%nomvar(cv_model)) == 'VV' .and. &
 427            trim(ControlVariable(VWindID)%nomvar(cv_bhi))   == 'CC' ) then
 428          write(*,*)
 429          write(*,*) 'Momentum Control Variables = Psi-Chi'
 430          WindTransform = 'PsiChi'
 431       else if ( trim(ControlVariable(UWindID)%nomvar(cv_model)) == 'UU' .and. &
 432            trim(ControlVariable(UWindID)%nomvar(cv_bhi))   == 'QR' .and. &
 433            trim(ControlVariable(VWindID)%nomvar(cv_model)) == 'VV' .and. &
 434            trim(ControlVariable(VWindID)%nomvar(cv_bhi))   == 'DD' ) then
 435          write(*,*)
 436          write(*,*) 'Momentum Control Variables = Vort-Div'
 437          WindTransform = 'VortDiv'
 438       else if ( trim(ControlVariable(UWindID)%nomvar(cv_model)) == 'UU' .and. &
 439            trim(ControlVariable(UWindID)%nomvar(cv_bhi))   == 'UU' .and. &
 440            trim(ControlVariable(VWindID)%nomvar(cv_model)) == 'VV' .and. &
 441            trim(ControlVariable(VWindID)%nomvar(cv_bhi))   == 'VV' ) then
 442          write(*,*)
 443          write(*,*) 'Momentum Control Variables = U-V '
 444          WindTransform = 'UV'
 445       else
 446          call utl_abort('lbhi_GetControlVariableInfo: Unkown Wind Tranform')
 447       end if
 448    end if
 449
 450  end subroutine lbhi_GetControlVariableInfo

 508  !--------------------------------------------------------------------------
 509  ! lbhi_ReadBSqrt
 510  !--------------------------------------------------------------------------
 511  subroutine lbhi_ReadBSqrt( iu_bstats )
 512    implicit none
 513
 514    integer, intent(in) :: iu_bstats
 515
 516    real(8), allocatable :: bsqrt2d  (:,:)
 517
 518    integer :: key, fstinf, fstinl, totwvnb, infon
 519    integer, parameter :: nmax=2000
 520    integer :: liste(nmax)
 521
 522    integer                     :: ip1, ip2, ip3
 523    integer                     :: ni_t, nj_t, nlev_t, dateo  
 524    character(len=4 )           :: nomvar
 525    character(len=2 )           :: typvar
 526    character(len=12)           :: etiket
 527
 528    dateo  = -1
 529    etiket = 'B_SQUAREROOT'
 530    ip1    = -1
 531    ip3    = -1
 532    typvar = ' '
 533    nomvar = 'ZN'
 534
 535    !
 536    !- 1.  Find the truncation in the stats file
 537    !
 538    ip2    = -1
 539
 540    key = fstinl(iu_bstats,                                   & ! IN
 541                ni_t, nj_t, nlev_t,                           & ! OUT
 542                dateo, etiket, ip1, ip2, ip3, typvar, nomvar, & ! IN
 543                liste, infon,                                 & ! OUT
 544                nmax )                                          ! IN
 545
 546    if (key >= 0) then
 547      !- 1.2 Ensure spectral trunctation are the same
 548      if ( infon - 1  /= trunc ) then
 549        write(*,*)
 550        write(*,*) 'lbhi_ReadBSqrt: Truncation here and on stats file different'
 551        write(*,*) 'VAR truncation        = ', trunc
 552        write(*,*) 'Stats file truncation = ', infon-1
 553        call utl_abort('lbhi_ReadBSqrt')
 554      end if
 555    else
 556      write(*,*)
 557      write(*,*) 'lbhi_ReadBSqrt: Cannot find B square-root ', nomvar 
 558      call utl_abort('lbhi_ReadBSqrt')
 559    end if
 560
 561    !
 562    !- 2.   Read B^0.5
 563    !
 564    allocate( bsqrt2d  (1:nksdim, 1:nksdim) )
 565
 566    do totwvnb = 0, trunc
 567
 568      ip2    = totwvnb
 569
 570      !- 2.1 Check if field exists and its dimensions
 571      key = fstinf( iu_bstats,                                  & ! IN
 572                    ni_t, nj_t, nlev_t,                         & ! OUT
 573                    dateo, etiket, ip1, ip2, ip3, typvar,nomvar)  ! IN
 574
 575      if (key >= 0) then
 576        !- 2.2 Ensure that the number of vertical levels are compatible
 577        if ( ni_t /= nksdim .or. nj_t /= nksdim  ) then
 578          write(*,*)
 579          write(*,*) 'lbhi_ReadBSqrt: BG stat levels inconsitencies'
 580          write(*,*) 'for BSQRT: ni_t, nj_t, nksdim =', ni_t, nj_t, nksdim
 581          call utl_abort('lbhi_ReadBSqrt')
 582        endif
 583
 584        !- 2.3 Reading
 585        key = utl_fstlir( bsqrt2d,                                    & ! OUT 
 586                          iu_bstats,                                  & ! IN
 587                          ni_t, nj_t, nlev_t,                         & ! OUT
 588                          dateo, etiket, ip1, ip2, ip3, typvar,nomvar)  ! IN
 589      else
 590        write(*,*)
 591        write(*,*) 'lbhi_ReadBSqrt: Cannot find BSQRT for totwvnb = ', totwvnb
 592        call utl_abort('lbhi_ReadBSqrt')
 593      end if
 594
 595      !- 2.4 Transfer to a 3D array
 596      bsqrt(:,:,totwvnb) = bsqrt2d(:,:)
 597
 598    end do
 599
 600    deallocate( bsqrt2d )
 601
 602  end subroutine lbhi_ReadBSqrt

 830  ! lbhi_cv2gd
 831  !--------------------------------------------------------------------------
 832  subroutine lbhi_cv2gd(hiControlVector_in, gd_out)
 833    implicit none
 834
 835    real(8), intent(inout) :: hiControlVector_in(lst_bhi%nla, lst_bhi%nphase, nksdim)
 836    real(8), intent(out)   :: gd_out(myLonBeg:myLonEnd  ,myLatBeg:myLatEnd  ,1:nksdim)
 837
 838    real(8), allocatable :: uphy(:,:,:)
 839    real(8), allocatable :: vphy(:,:,:)
 840    real(8), allocatable :: psi(:,:,:)
 841    real(8), allocatable :: chi(:,:,:)
 842
 843    integer :: kstart, kend, var
 844
 845    character(len=19)   :: kind
 846
 847    !
 848    !- 1. B^1/2 * xi (in spectral space)
 849    !
 850    call lbhi_bSqrtXi(hiControlVector_in)    ! INOUT
 851
 852    !
 853    !- 2. Spectral Space -> Grid Point Space
 854    !
 855    kind = 'SpectralToGridPoint'
 856    call lst_VarTransform(lst_bhi,            & ! IN
 857                          hiControlVector_in, & ! IN
 858                          gd_out,             & ! OUT
 859                          kind, nksdim )        ! IN
 860
 861    !
 862    !- 3.  Multiply by the grid point standard deviations
 863    !
 864    !$OMP PARALLEL DO PRIVATE(var,kstart,kend)
 865    do var = 1, nControlVariable
 866      kstart = ControlVariable(var)%kDimStart
 867      kend   = ControlVariable(var)%kDimEnd
 868      gd_out(:,:,kstart:kend) = gd_out(:,:,kstart:kend) * ControlVariable(var)%GpStdDev(myLonBeg:myLonEnd,myLatBeg:myLatEnd,:)
 869    end do
 870    !$OMP END PARALLEL DO
 871
 872    !
 873    !- 4.  Momentum variables transform
 874    !
 875    if ( UWindID /= -1 .and. VWindID /= -1) then
 876       if ( ControlVariable(UWindID)%nlev /= nlev_M .or. &
 877            ControlVariable(VWindID)%nlev /= nlev_M  ) then
 878          call utl_abort('lbhi_cv2gd: Error in Wind related parameters')
 879       end if
 880    
 881       if ( trim(WindTransform) /= 'UV') then
 882
 883          allocate(uphy(myLonBeg:myLonEnd,myLatBeg:myLatEnd,1:nlev_M))
 884          allocate(vphy(myLonBeg:myLonEnd,myLatBeg:myLatEnd,1:nlev_M))
 885          allocate(psi (myLonBeg:myLonEnd,myLatBeg:myLatEnd,1:nlev_M))
 886          allocate(chi (myLonBeg:myLonEnd,myLatBeg:myLatEnd,1:nlev_M))
 887
 888          !- 4.1 Vort / Div -> Psi / Chi
 889          if ( trim(WindTransform) == 'VortDiv') then
 890
 891             !  4.1.1 Putting vorticity in psi array and divergence in chi array
 892             psi(:,:,:) = gd_out(:,:,ControlVariable(UWindID)%kDimStart:ControlVariable(UWindID)%kDimEnd)
 893             chi(:,:,:) = gd_out(:,:,ControlVariable(VWindID)%kDimStart:ControlVariable(VWindID)%kDimEnd)
 894             
 895             !  4.1.2 Vort -> Psi
 896             call lst_Laplacian(lst_wind,           & ! IN
 897                                Psi,                & ! INOUT
 898                                'Inverse', nlev_M)    ! IN    
 899
 900             !  4.1.3 Div -> Chi
 901             call lst_Laplacian(lst_wind,          & ! IN
 902                                Chi,               & ! INOUT
 903                                'Inverse', nlev_M)   ! IN
 904
 905          end if
 906
 907          !- 4.2 Psi / Chi -> U-wind / V-wind
 908          if ( trim(WindTransform) == 'PsiChi') then
 909             psi(:,:,:) = gd_out(:,:,ControlVariable(UWindID)%kDimStart:ControlVariable(UWindID)%kDimEnd)
 910             chi(:,:,:) = gd_out(:,:,ControlVariable(VWindID)%kDimStart:ControlVariable(VWindID)%kDimEnd)
 911          end if
 912
 913          !  4.2.1 Do Transform
 914          call agd_PsiChiToUV(psi, chi,           & ! IN
 915                              uphy, vphy,         & ! OUT
 916                              nlev_M)               ! IN
 917
 918          !  4.2.2 Insert results in gd_out and deallocate memories
 919          gd_out(:,:,1       :  nlev_M) = uphy(:,:,:)
 920          gd_out(:,:,nlev_M+1:2*nlev_M) = vphy(:,:,:)
 921          
 922          deallocate(chi)
 923          deallocate(psi)
 924          deallocate(vphy)
 925          deallocate(uphy)
 926          
 927       end if
 928
 929    end if
 930
 931  end subroutine lbhi_cv2gd

1034  !--------------------------------------------------------------------------
1035  ! lbhi_bSqrtXi
1036  !--------------------------------------------------------------------------
1037  subroutine lbhi_bSqrtXi(hiControlVector_in)
1038    implicit none
1039
1040    real(8), intent(inout) :: hiControlVector_in(lst_bhi%nla, lst_bhi%nphase, nksdim)
1041
1042    real(8), allocatable :: sp_in (:,:,:)
1043    real(8), allocatable :: sp_out(:,:,:)
1044
1045    integer :: totwvnb, e, k, ila
1046    integer :: m, n, lda, ldb, ldc
1047
1048    !
1049    !- 1. B^1/2 * xi (in spectral space)
1050    !
1051    do totwvnb = 0, trunc
1052
1053      if ( lst_bhi%nePerK(totwvnb) == 0 ) then
1054         cycle
1055      end if
1056
1057      allocate( sp_in (nksdim,lst_bhi%nphase,lst_bhi%nePerK(totwvnb)) )
1058      allocate( sp_out(nksdim,lst_bhi%nphase,lst_bhi%nePerK(totwvnb)) )
1059
1060      !- 1.1 Select spectral elements associated with the total wavenumber
1061      !$OMP PARALLEL DO PRIVATE(e,ila,k)
1062      do e = 1, lst_bhi%nePerK(totwvnb)
1063        ila = lst_bhi%ilaFromEK(e,totwvnb)
1064        do k = 1, nksdim
1065          sp_in(k,1:lst_bhi%nphase,e) = hiControlVector_in(ila,1:lst_bhi%nphase,k)
1066        end do
1067      end do
1068      !$OMP END PARALLEL DO
1069
1070      !- 1.2 Compute bsqrt * sp_in using DGEMM 
1071
1072      ! For documentation on dgemm, see: http://www.netlib.org/blas/dgemm.f
1073      ! Matrix A = BSQRT(:,:,totwvnb)
1074      ! Matrix B = SP_IN
1075      ! Matrix C = SP_OUT
1076      m   = nksdim
1077      n   = lst_bhi%nphase * lst_bhi%nePerK(totwvnb)
1078      k   = nksdim
1079      lda = nksdim
1080      ldb = nksdim
1081      ldc = nksdim
1082
1083      call dgemm( 'N', 'N', m, n, k, 1.d0,                   &  ! IN
1084                  bsqrt(:,:,totwvnb), lda, sp_in, ldb, 0.d0, &  ! IN
1085                  sp_out,                                    &  ! OUT
1086                  ldc )                                         ! IN
1087
1088      !- 1.3 Replace sp values with output matrix
1089      !$OMP PARALLEL DO PRIVATE(e,ila,k)
1090      do e = 1, lst_bhi%nePerK(totwvnb)
1091        ila = lst_bhi%ilaFromEK(e,totwvnb)
1092        do k = 1, nksdim
1093          hiControlVector_in(ila,1:lst_bhi%nphase,k) = sp_out(k,1:lst_bhi%nphase,e)
1094        end do
1095      end do
1096      !$OMP END PARALLEL DO
1097
1098      deallocate(sp_in)
1099      deallocate(sp_out)
1100
1101    end do ! Total Wavenumber
1102
1103  end subroutine lbhi_bSqrtXi
