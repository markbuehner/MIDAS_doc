 17module getGridPosition_mod
 18  ! MODULE getGridPosition_mod (prefix='gpos' category='8. Low-level utilities and constants')
 19  !
 20  !:Purpose: A place to collect numerous interpolation related routines.
 21  !           The main task of the module is to compute the grid XY position from a lat-lon.
 22  !           This simply calls the ezsint routine gdxyfll for simple grids. For
 23  !           Yin-Yan grids it calls the function gpos_xyfll_yinYangGrid
 24  !           (see below in this module). There is also support for
 25  !           RPN Y grids, in which case it calls the subroutine gpos_xyfll_unstructGrid.
 26  !
 27  use kdtree2_mod
 28  use mathPhysConstants_mod
 29  use physicsFunctions_mod
 30  use utilities_mod
 31
 32  implicit none
 33  save
 34  private
 35
 36  ! public procedures
 37  public :: gpos_getPositionXY
 38
 39  integer, parameter :: maxNumLocalGridPointsSearch = 300
 40  integer, external  :: get_max_rss
 41
 42  type(kdtree2), pointer :: tree => null()
 43
 44contains

 46  !---------------------------------------------------------
 47  ! gpos_getPositionXY
 48  !---------------------------------------------------------
 49  function gpos_getPositionXY( gdid, xpos_r4, ypos_r4, xpos2_r4, ypos2_r4,  &
 50                          lat_deg_r4, lon_deg_r4, subGridIndex ) result(ierr)
 51    !
 52    !:Purpose: Compute the grid XY position from a lat-lon. This
 53    !           simply calls the ezsint routine gdxyfll for simple grids. For
 54    !           Yin-Yan grids it calls the function gpos_xyfll_yinYangGrid
 55    !           (see below in this module). There is also support for
 56    !           RPN Y grids, in which case it calls the subroutine gpos_xyfll_unstructGrid.
 57    !
 58    implicit none
 59
 60    ! arguments
 61    integer :: ierr  ! returned value of function
 62    integer, intent(in) :: gdid
 63    integer, intent(out) :: subGridIndex
 64    real(4), intent(out) :: xpos_r4
 65    real(4), intent(out) :: ypos_r4
 66    real(4), intent(out) :: xpos2_r4
 67    real(4), intent(out) :: ypos2_r4
 68    real(4), intent(in) :: lat_deg_r4
 69    real(4), intent(in) :: lon_deg_r4
 70
 71    ! locals
 72    integer :: numSubGrids
 73    integer :: ezget_nsubGrids, gdxyfll, ezgprm
 74    character(len=1) :: grtyp
 75    integer :: ni, nj, ig1, ig2, ig3, ig4
 76
 77    numSubGrids = ezget_nsubGrids(gdid)
 78    xpos2_r4 = -999.0
 79    ypos2_r4 = -999.0
 80
 81    if ( numSubGrids == 1 ) then
 82
 83      ierr = ezgprm(gdid, grtyp, ni, nj, ig1, ig2, ig3, ig4)
 84
 85      if (grtyp == 'Y') then
 86
 87        !$omp critical
 88        ierr = gpos_xyfll_unstructGrid(gdid, xpos_r4, ypos_r4, lat_deg_r4, lon_deg_r4)
 89        !$omp end critical
 90
 91      else
 92
 93        ! Not a Yin-Yang grid, call the standard ezscint routine
 94        ierr = gdxyfll(gdid, xpos_r4, ypos_r4, lat_deg_r4, lon_deg_r4, 1)
 95
 96      end if
 97
 98      subGridIndex = 1
 99
100    else
101
102      ! This is a Yin-Yang grid, do something different
103      !$omp critical
104      ierr = gpos_xyfll_yinYangGrid(gdid, xpos_r4, ypos_r4, xpos2_r4, ypos2_r4, &
105                                    lat_deg_r4, lon_deg_r4, subGridIndex)
106      !$omp end critical
107
108    end if    
109
110    if ( subGridIndex /= 3 ) then
111      ! when only returning 1 position, copy values to pos2
112      xpos2_r4 = xpos_r4
113      ypos2_r4 = ypos_r4
114    end if
115
116  end function gpos_getPositionXY

118  !---------------------------------------------------------
119  ! gpos_xyfll_yinYangGrid
120  !---------------------------------------------------------
121  function gpos_xyfll_yinYangGrid( gdid, xpos_r4, ypos_r4, xpos2_r4, ypos2_r4,  &
122                          lat_deg_r4, lon_deg_r4, subGridIndex ) result(ierr)
123
124    !:Purpose: Compute the grid XY position from a lat-lon for a Yin-Yang grid.
125    !           It returns locations from both the Yin and Yang
126    !           subgrids when in the overlap region, depending on the logical 
127    !           variable `useSingleValueOverlap`.
128    implicit none
129
130    ! arguments
131    integer :: ierr  ! returned value of function
132    integer, intent(in)  :: gdid
133    integer, intent(out) :: subGridIndex
134    real(4), intent(in)  :: lat_deg_r4, lon_deg_r4
135    real(4), intent(out) :: xpos_r4,  ypos_r4
136    real(4), intent(out) :: xpos2_r4, ypos2_r4
137
138    ! Local Variables
139    integer :: ezget_subGridids, gdgaxes, gdxyfll, ezgprm
140    integer :: EZscintIDvec(2)
141    integer, save :: EZscintIDvec1_old = -999
142    integer :: lonIndex, latIndex
143    real :: lonrot, latrot
144    real, allocatable, save :: ax_yin(:), ay_yin(:), ax_yan(:), ay_yan(:)
145    logical :: axesDifferent
146    character(len=1) :: grtyp
147    integer :: ni, nj, ig1, ig2, ig3, ig4
148
149    ! this controls which approach to use for interpolation within the YIN-YAN overlap
150    logical :: useSingleValueOverlap = .true.  
151
152    ierr = ezget_subGridids(gdid, EZscintIDvec)
153    ! get ni nj of subGrid, assume same for both YIN and YANG
154    ierr = ezgprm(EZscintIDvec(1), grtyp, ni, nj, ig1, ig2, ig3, ig4)
155
156    ! first check YIN
157    ierr = gdxyfll(EZscintIDvec(1), xpos_r4, ypos_r4, lat_deg_r4, lon_deg_r4, 1)
158
159    ! compute rotated lon and lat at obs location
160    axesDifferent = (EZscintIDvec1_old /= EZscintIDvec(1))
161    if (axesDifferent) then
162      write(*,*) 'gpos_getPositionXY: axesDifferent, compute needed parameters'
163      if (allocated(ax_yin)) deallocate(ax_yin,ay_yin)
164      allocate(ax_yin(ni),ay_yin(nj))
165      ierr = gdgaxes(EZscintIDvec(1), ax_yin, ay_yin)
166      EZscintIDvec1_old = EZscintIDvec(1)
167    end if
168    lonIndex = floor(xpos_r4)
169    if ( lonIndex >= 1 .and. (lonIndex+1) <= ni ) then
170      lonrot = ax_yin(lonIndex) + (ax_yin(lonIndex+1) - ax_yin(lonIndex)) *  &
171           (xpos_r4 - lonIndex)
172    else
173      lonrot = -999.0
174    end if
175    latIndex = floor(ypos_r4)
176    if ( latIndex >= 1 .and. (latIndex+1) <= nj ) then
177      latrot = ay_yin(latIndex) + (ay_yin(latIndex+1) - ay_yin(latIndex)) *  &
178           (ypos_r4 - latIndex)
179    else
180      latrot = -999.0
181    end if
182    subGridIndex = 1
183
184    if ( useSingleValueOverlap ) then
185
186      ! this approach is most similar to how ezsint works, preferentially take YIN
187
188      if ( lonrot < 45.0 .or. lonrot > 315.0 .or. latrot < -45.0 .or. latrot > 45.0 ) then
189        ! Outside YIN, therefore use YANG (assume it is inside YANG)
190        ierr = gdxyfll(EZscintIDvec(2), xpos_r4, ypos_r4, lat_deg_r4, lon_deg_r4, 1)
191        ypos_r4 = ypos_r4 + real(nj) ! shift from YANG position to Supergrid position
192        subGridIndex = 2
193      else
194        subGridIndex = 1
195      end if
196
197    else ! not useSingleValueOverlap
198
199      ! this approach returns both the YIN and YAN locations when point is inside both
200
201      if ( lonrot < 45.0 .or. lonrot > 315.0 .or. latrot < -45.0 .or. latrot > 45.0 ) then
202        ! Outside YIN, therefore use YANG (assume it is inside YANG)
203        ierr = gdxyfll(EZscintIDvec(2), xpos_r4, ypos_r4, lat_deg_r4, lon_deg_r4, 1)
204        ypos_r4 = ypos_r4 + real(nj) ! shift from YANG position to Supergrid position
205        subGridIndex = 2
206      else
207        ! inside YIN, check if also inside YANG
208        allocate(ax_yan(ni),ay_yan(nj))
209        ierr = gdgaxes(EZscintIDvec(2), ax_yan, ay_yan)
210        ierr = gdxyfll(EZscintIDvec(2), xpos2_r4, ypos2_r4, lat_deg_r4, lon_deg_r4, 1)
211        if ( lonIndex >= 1 .and. (lonIndex+1) <= ni ) then
212          lonrot = ax_yan(lonIndex) + (ax_yan(lonIndex+1) - ax_yan(lonIndex)) *  &
213               (xpos2_r4 - lonIndex)
214        else
215          lonrot = -999.0
216        end if
217        latIndex = floor(ypos2_r4)
218        if ( latIndex >= 1 .and. (latIndex+1) <= nj ) then
219          latrot = ay_yan(latIndex) + (ay_yan(latIndex+1) - ay_yan(latIndex)) *  &
220               (ypos2_r4 - latIndex)
221        else
222          latrot = -999.0
223        end if
224        deallocate(ax_yan,ay_yan)
225        if ( lonrot < 45.0 .or. lonrot > 315.0 .or. latrot < -45.0 .or. latrot > 45.0 ) then
226          ! outside YANG, only inside YIN
227          xpos2_r4 = -999.0
228          ypos2_r4 = -999.0
229          subGridIndex = 1
230        else
231          ! inside both YIN and YANG
232          ypos2_r4 = ypos2_r4 + real(nj) ! shift from YANG position to Supergrid position
233          subGridIndex = 3
234        end if
235      end if
236
237    end if
238
239  end function gpos_xyfll_yinYangGrid

241  !---------------------------------------------------------
242  ! gpos_xyfll_unstructGrid
243  !---------------------------------------------------------
244  function gpos_xyfll_unstructGrid( gdid, xpos_r4, ypos_r4, lat_deg_r4, lon_deg_r4 ) result(ierr)
245
246    !:Purpose: This function is used to interpolate from a RPN grid to a location
247    !           specified by a latitude and longitude.
248    !           It has special treatment for the ORCA12 and ORCA025 tri-polar grids
249    !           used for sea ice and ocean models.
250    !           The kdtree2 module is used to efficiently
251    !           perform this task. The kdtree itself is constructed on the first call.
252    implicit none
253
254    ! arguments
255    integer :: ierr  ! returned value of function
256    integer, intent(in)  :: gdid
257    real(4), intent(in)  :: lat_deg_r4, lon_deg_r4
258    real(4), intent(out) :: xpos_r4, ypos_r4
259
260    ! Local Variables
261
262    real(8) :: lon_rad_r8, lat_rad_r8, pertGridLonRad, pertGridLatRad
263    real(8), parameter :: deltaGrid = 0.00001d0
264    real(8) :: pertPosition(3)
265    real(8) :: gridSpacing
266    real(8) :: gridSpacingSquared, lowerLeftCornerDistSquared, lowerRightCornerDistSquared, upperLeftCornerDistSquared
267
268    integer, save :: gdidOld = -999
269    integer :: nx, ny
270    integer, save :: ni, nj
271    real(8), allocatable, save :: grid_lon_rad(:,:), grid_lat_rad(:,:)
272    real(4), save :: maxGridSpacing
273    integer, save :: startXIndex, startYIndex, endXIndex, endYIndex
274
275    character(len=1) :: grtyp
276    integer :: ig1, ig2, ig3, ig4
277    real(4), allocatable :: grid_lat_deg_r4(:,:), grid_lon_deg_r4(:,:)
278    integer :: closePointsIndex
279    integer :: gridIndex
280    integer :: xIndex, yIndex, xIndexMin, xIndexMax, yIndexMin, yIndexMax
281
282    ! Functions
283    integer :: ezgprm, gdll
284
285    integer                   :: numLocalGridPointsFound
286    real(kdkind), allocatable :: positionArray(:,:)
287    type(kdtree2_result)      :: searchResults(maxNumLocalGridPointsSearch)
288    real(kdkind)              :: maxRadiusSquared
289    real(kdkind)              :: refPosition(3)
290
291    if ( gdid /= gdidOld .and. gdidOld > 0) then
292      write(*,*) 'gpos_xyfll_unstructGrid: gdid gdidOld = ',gdid, gdidOld
293      call utl_abort('gpos_xyfll_unstructGrid: only one grid expected. Change code !')
294    end if
295
296    ! create the kdtree on the first call
297    if (.not. associated(tree)) then
298      write(*,*) 'gpos_xyfll_unstructGrid: start creating kdtree'
299      write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
300      ierr = ezgprm(gdid, grtyp, ni, nj, ig1, ig2, ig3, ig4)
301
302      startXIndex = 1
303      startYIndex = 1
304      endXIndex = ni
305      endYIndex = nj
306      maxGridSpacing = 10000.0
307
308      if (allocated(grid_lat_rad)) deallocate(grid_lat_rad, grid_lon_rad)
309      allocate(grid_lat_rad(ni,nj), grid_lon_rad(ni,nj))
310      allocate(grid_lat_deg_r4(ni,nj), grid_lon_deg_r4(ni,nj))
311      ierr = gdll(gdid, grid_lat_deg_r4, grid_lon_deg_r4)
312      grid_lat_rad(:,:) = real(grid_lat_deg_r4(:,:),8)*MPC_RADIANS_PER_DEGREE_R8
313      grid_lon_rad(:,:) = real(grid_lon_deg_r4(:,:),8)*MPC_RADIANS_PER_DEGREE_R8
314      deallocate(grid_lat_deg_r4, grid_lon_deg_r4)
315      gdidOld = gdid
316
317      if( (ni == 4322 .and. nj == 3059) .or. (ni == 1442 .and. nj == 1021) ) then
318        ! This is orca12 or orca025
319        ! The last 2 columns (i=4321 and 4322 in orca12) are repetitions of the first 2 columns (i=1 and 2)
320        ! so both ends are removed from the search area.
321        startXIndex = 2
322        endXIndex = ni - 1
323        ! The last line (j=3059 in orca12) is a repetition in reverse order of the line (j=3057 in orca12)
324        ! and is thus eliminated from the search domain.
325        endYIndex = nj - 1
326        ! Check that is true
327        do yIndex = 1, nj
328          if(grid_lat_rad(1,yIndex) /= grid_lat_rad(ni-1,yIndex) .or. grid_lat_rad(2,yIndex) /= grid_lat_rad(ni,yIndex) .or. &
329             grid_lon_rad(1,yIndex) /= grid_lon_rad(ni-1,yIndex) .or. grid_lon_rad(2,yIndex) /= grid_lon_rad(ni,yIndex)) then
330            write(*,*) 'yIndex = ',yIndex
331            write(*,*) grid_lat_rad(1,yIndex), grid_lat_rad(ni-1,yIndex), grid_lat_rad(2,yIndex), grid_lat_rad(ni,yIndex), &
332                       grid_lon_rad(1,yIndex), grid_lon_rad(ni-1,yIndex), grid_lon_rad(2,yIndex), grid_lon_rad(ni,yIndex)
333            call utl_abort('gpos_xyfll_unstructGrid: Assumptions about the grid not verified.')
334          end if
335        end do
336      end if
337      if(ni == 1442 .and. nj == 1021) then
338        ! This is orca025
339        maxGridSpacing = 40000.0
340      end if
341
342      nx = endXIndex - startXIndex + 1
343      ny = endYIndex - startYIndex + 1
344
345      allocate(positionArray(3,nx*ny))
346
347      gridIndex = 0
348      do yIndex = startYIndex, endYIndex
349        do xIndex = startXIndex, endXIndex
350          gridIndex = gridIndex + 1
351          positionArray(:,gridIndex) = kdtree2_3dPosition(grid_lon_rad(xIndex,yIndex), grid_lat_rad(xIndex,yIndex))
352        end do
353      end do
354      tree => kdtree2_create(positionArray, sort=.true., rearrange=.true.) 
355      write(*,*) 'gpos_xyfll_unstructGrid: done creating kdtree'
356      write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
357
358    else
359
360      ierr = 0
361      nx = endXIndex - startXIndex + 1
362
363    end if
364
365    ! do the search
366    maxRadiusSquared = maxGridSpacing**2
367    lon_rad_r8 = real(lon_deg_r4,8)*MPC_RADIANS_PER_DEGREE_R8
368    lat_rad_r8 = real(lat_deg_r4,8)*MPC_RADIANS_PER_DEGREE_R8
369    refPosition(:) = kdtree2_3dPosition(lon_rad_r8, lat_rad_r8)
370
371    call kdtree2_r_nearest(tp=tree, qv=refPosition, r2=maxRadiusSquared, nfound=numLocalGridPointsFound,&
372                           nalloc=maxNumLocalGridPointsSearch, results=searchResults)
373    if (numLocalGridPointsFound > maxNumLocalGridPointsSearch) then
374      call utl_abort('gpos_xyfll_unstructGrid: the parameter maxNumLocalGridPointsSearch must be increased')
375    end if
376
377    if (numLocalGridPointsFound < 4) then
378      write(*,*) 'gpos_xyfll_unstructGrid: numLocalGridPointsFound = ',numLocalGridPointsFound
379      write(*,*) 'gpos_xyfll_unstructGrid: search radius           = ',maxGridSpacing,' meters'
380      write(*,*) 'gpos_xyfll_unstructGrid: obs lon, lat = ', lon_deg_r4, lat_deg_r4
381      call utl_abort('gpos_xyfll_unstructGrid: the search did not find 4 close points.')
382    end if
383
384    if (searchResults(1)%dis > maxRadiusSquared) then
385      write(*,*) 'gpos_xyfll_unstructGrid: No grid point found within ',maxGridSpacing,' meters'
386      write(*,*) 'of the reference location lat-lon (degrees): ',lat_deg_r4,lon_deg_r4
387      xpos_r4 = -999.0
388      ypos_r4 = -999.0
389      return
390    end if
391
392    ! We found the closest grid point to the reference point.
393    ! Now we need to determine in which of the 4 quadrants around the grid point the reference point lies.
394
395    closePointsIndex = 1
396
397    gridIndex = searchResults(closePointsIndex)%idx
398    yIndex = (gridIndex-1)/nx + startYIndex
399    xIndex = gridIndex - (yIndex-startYIndex)*nx + startXIndex - 1
400
401    ! Perturbing the closest grid point in the x direction
402
403    if ( xIndex < ni ) then
404
405      if ( grid_lon_rad(xIndex+1,yIndex) - grid_lon_rad(xIndex,yIndex) > 5.0d0 ) then
406        pertGridLonRad = grid_lon_rad(xIndex,yIndex) + (grid_lon_rad(xIndex+1,yIndex) - (grid_lon_rad(xIndex,yIndex) + 2.d0 * MPC_PI_R8))*deltaGrid
407      else if ( grid_lon_rad(xIndex+1,yIndex) - grid_lon_rad(xIndex,yIndex) < -5.0d0 ) then
408        pertGridLonRad = grid_lon_rad(xIndex,yIndex) + ((grid_lon_rad(xIndex+1,yIndex) + 2.d0 * MPC_PI_R8) - grid_lon_rad(xIndex,yIndex))*deltaGrid
409      else
410        pertGridLonRad = grid_lon_rad(xIndex,yIndex) + (grid_lon_rad(xIndex+1,yIndex) - grid_lon_rad(xIndex,yIndex))*deltaGrid
411      end if
412      pertGridLatRad = grid_lat_rad(xIndex,yIndex) + (grid_lat_rad(xIndex+1,yIndex) - grid_lat_rad(xIndex,yIndex))*deltaGrid
413
414      ! Calculate the distance between the reference point and the perturbed grid point.
415      pertPosition(:) = kdtree2_3dPosition(pertGridLonRad, pertGridLatRad)
416
417      gridSpacingSquared = sum( (pertPosition(:) - refPosition(:))**2 )
418
419      if (gridSpacingSquared < searchResults(closePointsIndex)%dis) then
420        xIndexMin = xIndex
421        xIndexMax = xIndex + 1
422      else
423        if ( xIndex > 1 ) then
424          xIndexMin = xIndex - 1
425          xIndexMax = xIndex
426        else
427          write(*,*) 'xIndex yIndex ', xIndex, yIndex
428          write(*,*) 'Closest distance = ',sqrt(searchResults(closePointsIndex)%dis)
429          write(*,*) 'Perturbed distance = ',sqrt(gridSpacingSquared)
430          call utl_abort('gpos_xyfll_unstructGrid: 1. the reference point is outside the grid.')
431        end if
432      end if
433
434    else
435
436      if ( grid_lon_rad(xIndex-1,yIndex) - grid_lon_rad(xIndex,yIndex) > 5.0d0 ) then
437        pertGridLonRad = grid_lon_rad(xIndex,yIndex) + (grid_lon_rad(xIndex-1,yIndex) - (grid_lon_rad(xIndex,yIndex) + 2.d0 * MPC_PI_R8))*deltaGrid
438      else if ( grid_lon_rad(xIndex-1,yIndex) - grid_lon_rad(xIndex,yIndex) < -5.0d0 ) then
439        pertGridLonRad = grid_lon_rad(xIndex,yIndex) + ((grid_lon_rad(xIndex-1,yIndex) + 2.d0 * MPC_PI_R8) - grid_lon_rad(xIndex,yIndex))*deltaGrid
440      else
441        pertGridLonRad = grid_lon_rad(xIndex,yIndex) + (grid_lon_rad(xIndex-1,yIndex) - grid_lon_rad(xIndex,yIndex))*deltaGrid
442      end if
443      pertGridLatRad = grid_lat_rad(xIndex,yIndex) + (grid_lat_rad(xIndex-1,yIndex) - grid_lat_rad(xIndex,yIndex))*deltaGrid
444
445      ! Calculate the distance between the reference point and the perturbed grid point.
446      pertPosition(:) = kdtree2_3dPosition(pertGridLonRad, pertGridLatRad)
447
448      gridSpacingSquared = sum( (pertPosition(:) - refPosition(:))**2 )
449
450      if (gridSpacingSquared < searchResults(closePointsIndex)%dis) then
451        xIndexMin = xIndex - 1
452        xIndexMax = xIndex
453      else
454        write(*,*) 'xIndex yIndex ', xIndex, yIndex
455        write(*,*) 'Closest distance = ',sqrt(searchResults(closePointsIndex)%dis)
456        write(*,*) 'Perturbed distance = ',sqrt(gridSpacingSquared)
457        call utl_abort('gpos_xyfll_unstructGrid: 2. the reference point is outside the grid.')
458      end if
459
460    end if
461
462    ! Perturbing the closest grid point in the y direction
463
464    if ( yIndex < nj ) then
465
466      if ( grid_lon_rad(xIndex,yIndex+1) - grid_lon_rad(xIndex,yIndex) > 5.0d0 ) then
467        pertGridLonRad = grid_lon_rad(xIndex,yIndex) + (grid_lon_rad(xIndex,yIndex+1) - (grid_lon_rad(xIndex,yIndex) + 2.d0 * MPC_PI_R8))*deltaGrid
468      else if ( grid_lon_rad(xIndex,yIndex+1) - grid_lon_rad(xIndex,yIndex) < -5.0d0 ) then
469        pertGridLonRad = grid_lon_rad(xIndex,yIndex) + ((grid_lon_rad(xIndex,yIndex+1) + 2.d0 * MPC_PI_R8) - grid_lon_rad(xIndex,yIndex))*deltaGrid
470      else
471        pertGridLonRad = grid_lon_rad(xIndex,yIndex) + (grid_lon_rad(xIndex,yIndex+1) - grid_lon_rad(xIndex,yIndex))*deltaGrid
472      end if
473      pertGridLatRad = grid_lat_rad(xIndex,yIndex) + (grid_lat_rad(xIndex,yIndex+1) - grid_lat_rad(xIndex,yIndex))*deltaGrid
474
475      ! Calculate the distance between the reference point and the perturbed grid point.
476      pertPosition(:) = kdtree2_3dPosition(pertGridLonRad, pertGridLatRad)
477
478      gridSpacingSquared = sum( (pertPosition(:) - refPosition(:))**2 )
479
480      if (gridSpacingSquared < searchResults(closePointsIndex)%dis) then
481        yIndexMin = yIndex
482        yIndexMax = yIndex + 1
483      else
484        if ( yIndex > 1 ) then
485          yIndexMin = yIndex - 1
486          yIndexMax = yIndex
487        else
488          write(*,*) 'xIndex yIndex ', xIndex, yIndex
489          write(*,*) 'Closest distance = ',sqrt(searchResults(closePointsIndex)%dis)
490          write(*,*) 'Perturbed distance = ',sqrt(gridSpacingSquared)
491          call utl_abort('gpos_xyfll_unstructGrid: 3. the reference point is outside the grid.')
492        end if
493      end if
494
495    else
496
497      if ( grid_lon_rad(xIndex,yIndex-1) - grid_lon_rad(xIndex,yIndex) > 5.0d0 ) then
498        pertGridLonRad = grid_lon_rad(xIndex,yIndex) + (grid_lon_rad(xIndex,yIndex-1) - (grid_lon_rad(xIndex,yIndex) + 2.d0 * MPC_PI_R8))*deltaGrid
499      else if ( grid_lon_rad(xIndex,yIndex-1) - grid_lon_rad(xIndex,yIndex) < -5.0d0 ) then
500        pertGridLonRad = grid_lon_rad(xIndex,yIndex) + ((grid_lon_rad(xIndex,yIndex-1) + 2.d0 * MPC_PI_R8) - grid_lon_rad(xIndex,yIndex))*deltaGrid
501      else
502        pertGridLonRad = grid_lon_rad(xIndex,yIndex) + (grid_lon_rad(xIndex,yIndex-1) - grid_lon_rad(xIndex,yIndex))*deltaGrid
503      end if
504      pertGridLatRad = grid_lat_rad(xIndex,yIndex) + (grid_lat_rad(xIndex,yIndex-1) - grid_lat_rad(xIndex,yIndex))*deltaGrid
505
506      ! Calculate the distance between the reference point and the perturbed grid point.
507      pertPosition(:) = kdtree2_3dPosition(pertGridLonRad, pertGridLatRad)
508
509      gridSpacingSquared = sum( (pertPosition(:) - refPosition(:))**2 )
510
511      if (gridSpacingSquared < searchResults(closePointsIndex)%dis) then
512        yIndexMin = yIndex - 1
513        yIndexMax = yIndex
514      else
515        write(*,*) 'xIndex yIndex ', xIndex, yIndex
516        write(*,*) 'Closest distance = ',sqrt(searchResults(closePointsIndex)%dis)
517        write(*,*) 'Perturbed distance = ',sqrt(gridSpacingSquared)
518        call utl_abort('gpos_xyfll_unstructGrid: 4. the reference point is outside the grid.')
519      end if
520
521    end if
522
523    ! Calculate real x and y position in the grid.
524    ! kdtree returns distance in square meters !
525
526    gridSpacingSquared = phf_calcDistance(grid_lat_rad(xIndexMin,yIndexMin), grid_lon_rad(xIndexMin,yIndexMin), &
527           grid_lat_rad(xIndexMax,yIndexMin), grid_lon_rad(xIndexMax,yIndexMin))**2
528
529    lowerLeftCornerDistSquared = phf_calcDistance(grid_lat_rad(xIndexMin,yIndexMin), grid_lon_rad(xIndexMin,yIndexMin), &
530           lat_rad_r8, lon_rad_r8)**2
531
532    lowerRightCornerDistSquared = phf_calcDistance(grid_lat_rad(xIndexMax,yIndexMin), grid_lon_rad(xIndexMax,yIndexMin), &
533           lat_rad_r8, lon_rad_r8)**2
534
535    xpos_r4 = real(xIndexMin) + (lowerLeftCornerDistSquared + gridSpacingSquared - lowerRightCornerDistSquared)/(2.0*(gridSpacingSquared))
536
537    gridSpacingSquared = phf_calcDistance(grid_lat_rad(xIndexMin,yIndexMin), grid_lon_rad(xIndexMin,yIndexMin), &
538           grid_lat_rad(xIndexMin,yIndexMax), grid_lon_rad(xIndexMin,yIndexMax))**2
539
540    upperLeftCornerDistSquared = phf_calcDistance(grid_lat_rad(xIndexMin,yIndexMax), grid_lon_rad(xIndexMin,yIndexMax), &
541           lat_rad_r8, lon_rad_r8)**2
542
543    ypos_r4 = real(yIndexMin) + (lowerLeftCornerDistSquared + gridSpacingSquared  - upperLeftCornerDistSquared)/(2.0*(gridSpacingSquared))
544
545    if ( abs(ypos_r4 - yIndexMin) > 1.1 .or. abs(xpos_r4 - xIndexMin) > 1.1 ) then
546      write(*,*) 'xpos_r4 = ',xpos_r4
547      write(*,*) 'ypos_r4 = ',ypos_r4
548      write(*,*) 'xIndexMin = ',xIndexMin
549      write(*,*) 'yIndexMin = ',yIndexMin
550      do closePointsIndex = 1,min(numLocalGridPointsFound,5)
551        gridIndex = searchResults(closePointsIndex)%idx
552        yIndex = (gridIndex-1)/nx + startYIndex
553        xIndex = gridIndex - (yIndex-startYIndex)*nx
554        write(*,*) 'closePointsIndex: ', closePointsIndex
555        write(*,*) 'xIndex yIndex ', xIndex, yIndex
556        write(*,*) 'idx: ',searchResults(closePointsIndex)%idx
557        write(*,*) 'distance (meters): ',sqrt(searchResults(closePointsIndex)%dis)
558      end do
559      write(*,*) 'lowerLeftCornerDist = ',sqrt(lowerLeftCornerDistSquared)
560      write(*,*) 'lowerRightCornerDist = ',sqrt(lowerRightCornerDistSquared)
561      write(*,*) 'upperLeftCornerDist = ',sqrt(upperLeftCornerDistSquared)
562      write(*,*) 'y gridSpacing = ',sqrt(gridSpacingSquared)
563      gridSpacing = phf_calcDistance(grid_lat_rad(xIndexMin,yIndexMin), grid_lon_rad(xIndexMin,yIndexMin), &
564           grid_lat_rad(xIndexMax,yIndexMin), grid_lon_rad(xIndexMax,yIndexMin))
565      write(*,*) 'x gridSpacing = ',gridSpacing
566      write(*,*) 'lon-lat (deg):'
567      write(*,*) 'reference point: ',lon_deg_r4, lat_deg_r4
568      write(*,*) xIndexMin,yIndexMin,grid_lon_rad(xIndexMin,yIndexMin)* MPC_DEGREES_PER_RADIAN_R8,grid_lat_rad(xIndexMin,yIndexMin)* MPC_DEGREES_PER_RADIAN_R8
569      write(*,*) xIndexMax,yIndexMin,grid_lon_rad(xIndexMax,yIndexMin)* MPC_DEGREES_PER_RADIAN_R8,grid_lat_rad(xIndexMax,yIndexMin)* MPC_DEGREES_PER_RADIAN_R8
570      write(*,*) xIndexMax+1,yIndexMin,grid_lon_rad(xIndexMax+1,yIndexMin)* MPC_DEGREES_PER_RADIAN_R8,grid_lat_rad(xIndexMax+1,yIndexMin)* MPC_DEGREES_PER_RADIAN_R8
571      write(*,*) xIndexMin,yIndexMax,grid_lon_rad(xIndexMin,yIndexMax)* MPC_DEGREES_PER_RADIAN_R8,grid_lat_rad(xIndexMin,yIndexMax)* MPC_DEGREES_PER_RADIAN_R8
572      write(*,*) xIndexMax,yIndexMax,grid_lon_rad(xIndexMax,yIndexMax)* MPC_DEGREES_PER_RADIAN_R8,grid_lat_rad(xIndexMax,yIndexMax)* MPC_DEGREES_PER_RADIAN_R8
573      write(*,*) xIndexMax+1,yIndexMax,grid_lon_rad(xIndexMax+1,yIndexMax)* MPC_DEGREES_PER_RADIAN_R8,grid_lat_rad(xIndexMax+1,yIndexMax)* MPC_DEGREES_PER_RADIAN_R8
574      write(*,*) xIndexMin,yIndexMax+1,grid_lon_rad(xIndexMin,yIndexMax+1)* MPC_DEGREES_PER_RADIAN_R8,grid_lat_rad(xIndexMin,yIndexMax+1)* MPC_DEGREES_PER_RADIAN_R8
575      write(*,*) xIndexMax,yIndexMax+1,grid_lon_rad(xIndexMax,yIndexMax+1)* MPC_DEGREES_PER_RADIAN_R8,grid_lat_rad(xIndexMax,yIndexMax+1)* MPC_DEGREES_PER_RADIAN_R8
576      write(*,*) xIndexMax+1,yIndexMax+1,grid_lon_rad(xIndexMax+1,yIndexMax+1)* MPC_DEGREES_PER_RADIAN_R8,grid_lat_rad(xIndexMax+1,yIndexMax+1)* MPC_DEGREES_PER_RADIAN_R8
577      call utl_abort('gpos_xyfll_unstructGrid: Check code !')
578    end if
579
580  end function gpos_xyfll_unstructGrid
