  56  subroutine gio_readFromFile(statevector_out, fileName, etiket_in, typvar_in, &
  57                              stepIndex_opt, unitConversion_opt, &
  58                              statevectorRef_opt, readHeightSfc_opt, &
  59                              containsFullField_opt, vcoFileIn_opt)
  60    !
  61    !:Purpose: Read an RPN standard file and put the contents into a
  62    !           stateVector object. Main high level wrapper subroutine.
  63    !
  64    implicit none
  65
  66    ! arguments
  67    type(struct_gsv),                    intent(inout) :: statevector_out
  68    character(len=*),                    intent(in)    :: fileName
  69    character(len=*),                    intent(in)    :: etiket_in
  70    character(len=*),                    intent(in)    :: typvar_in
  71    integer,          optional,          intent(in)    :: stepIndex_opt
  72    logical,          optional,          intent(in)    :: unitConversion_opt
  73    type(struct_gsv), optional,          intent(in)    :: statevectorRef_opt ! Reference statevector providing optional fields (P0, TT, HU)
  74    logical,          optional,          intent(in)    :: readHeightSfc_opt
  75    logical,          optional,          intent(in)    :: containsFullField_opt
  76    type(struct_vco), optional, pointer, intent(in)    :: vcoFileIn_opt
  77
  78    ! locals
  79    logical           :: doHorizInterp, doVertInterp, unitConversion
  80    logical           :: readHeightSfc, containsFullField
  81    logical           :: foundVarNameInFile 
  82    integer           :: stepIndex, varIndex
  83    character(len=4)  :: varName
  84
  85    type(struct_vco), pointer :: vco_file
  86    type(struct_hco), pointer :: hco_file
  87
  88    nullify(vco_file, hco_file)
  89
  90    write(*,*)
  91    write(*,*) 'gio_readFromFile: START'
  92    call utl_tmg_start(160,'low-level--gsv_readFromFile')
  93
  94    if ( present(stepIndex_opt) ) then
  95      stepIndex = stepIndex_opt
  96    else
  97      stepIndex = statevector_out%anltime
  98    end if
  99    if (stepIndex > stateVector_out%numStep .or. stepIndex < 1) then
 100      write(*,*) 'stepIndex = ', stepIndex
 101      call utl_abort('gio_readFromFile: invalid value for stepIndex')
 102    end if
 103
 104    if ( present(unitConversion_opt) ) then
 105      unitConversion = unitConversion_opt
 106    else
 107      unitConversion = .true.
 108    end if
 109
 110    if (present(containsFullField_opt)) then
 111      containsFullField = containsFullField_opt
 112    else
 113      containsFullField = .true.
 114    end if
 115    write(*,*) 'gio_readFromFile: containsFullField = ', containsFullField
 116
 117    if ( present(readHeightSfc_opt) ) then
 118      readHeightSfc = readHeightSfc_opt
 119    else
 120      readHeightSfc = .false.
 121    end if
 122
 123    ! set up vertical and horizontal coordinate for input file
 124    if (present(vcoFileIn_opt)) then
 125      vco_file => vcoFileIn_opt
 126      write(*,*)
 127      write(*,*) 'gio_readFromFile: vertical levels defined in user-supplied vco object will be read'
 128    else
 129      call vco_setupFromFile(vco_file,trim(fileName), beSilent_opt=.true.)
 130      write(*,*)
 131      write(*,*) 'gio_readFromFile: all the vertical levels will be read from ', trim(fileName) 
 132    end if
 133
 134    foundVarNameInFile = .false.
 135
 136    do varIndex = 1, vnl_numvarmax
 137      varName = vnl_varNameList(varIndex)
 138
 139      if (.not. gsv_varExist(statevector_out,varName)) cycle
 140
 141      ! make sure variable is in the file
 142      if (.not. utl_varNamePresentInFile(varName,fileName_opt=trim(fileName))) cycle
 143
 144      ! adopt a variable on the full/dynamic LAM grid
 145      if (.not. statevector_out%hco%global .and. (trim(varName) == 'TM' .or. trim(varName) == 'MG')) cycle
 146
 147      foundVarNameInFile = .true.
 148
 149      exit
 150
 151    end do
 152
 153    ! special case when only TM (Surface Temperature) is in the file:
 154    if (.not. foundVarNameInFile) then
 155      varname = 'TM'
 156      if (gsv_varExist( statevector_out, varname) .and. &
 157          utl_varNamePresentInFile( varname, fileName_opt = trim( fileName ))) &
 158        foundVarNameInFile = .true.
 159    end if   
 160
 161    ! to be safe for situations where, e.g. someone wants to only read MG from a file
 162    if (.not. foundVarNameInFile) then
 163      varname = 'P0'
 164      if (utl_varNamePresentInFile( varname, fileName_opt = trim( fileName))) &
 165        foundVarNameInFile = .true.
 166    end if   
 167
 168    if (.not. foundVarNameInFile) call utl_abort('gio_readFromFile: NO variables found in the file!!!')
 169
 170    write(*,*) 'gio_readFromFile: defining hco by varname= ', varName
 171
 172    call hco_setupFromFile(hco_file, trim(fileName), etiket_in, gridName_opt='FILEGRID', varName_opt = varName)
 173
 174    ! test if horizontal and/or vertical interpolation needed for statevector grid
 175    doVertInterp = .not.vco_equal(vco_file,statevector_out%vco)
 176    doHorizInterp = .not.hco_equal(hco_file,statevector_out%hco)
 177    write(*,*) 'gio_readFromFile: doVertInterp = ', doVertInterp, ', doHorizInterp = ', doHorizInterp
 178
 179    ! call appropriate subroutine to do actual work
 180    if ((doVertInterp .or. doHorizInterp) .and. statevector_out%mpi_distribution=='Tiles') then
 181      call readFromFileAndInterpToTiles(statevector_out, fileName,  &
 182             vco_file, hco_file, etiket_in, typvar_in, stepIndex, unitConversion,  &
 183             readHeightSfc, containsFullField, statevectorRef_opt=statevectorRef_opt)
 184    else if ((doVertInterp .or. doHorizInterp) .and. .not.stateVector_out%mpi_local) then
 185      call readFromFileAndInterp1Proc(statevector_out, fileName,  &
 186             vco_file, hco_file, etiket_in, typvar_in, stepIndex, unitConversion,  &
 187             readHeightSfc, containsFullField)
 188    else if (.not.(doVertInterp .or. doHorizInterp) .and. stateVector_out%mpi_local) then
 189      call readFromFileAndTransposeToTiles(statevector_out, fileName,  &
 190             etiket_in, typvar_in, stepIndex, unitConversion,  &
 191             readHeightSfc, containsFullField)
 192    else
 193      call readFromFileOnly(statevector_out, fileName,  &
 194                                etiket_in, typvar_in, stepIndex, unitConversion,  &
 195                                readHeightSfc, containsFullField)
 196    end if
 197
 198    call utl_tmg_stop(160)
 199    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 200    write(*,*) 'gio_readFromFile: END'
 201
 202  end subroutine gio_readFromFile

 207  subroutine readFromFileAndInterpToTiles(statevector_out, fileName,  &
 208             vco_file, hco_file, etiket_in, typvar_in, stepIndex, unitConversion,  &
 209             readHeightSfc, containsFullField, statevectorRef_opt)
 210    !
 211    !:Purpose: Read an RPN standard file and put the contents into a
 212    !           stateVector object. Wrapper subroutine that also proceed with
 213    !           distributed interpolation on MPI tiles.
 214    !
 215    !:Note: this routine currently only works correctly for reading FULL FIELDS,
 216    !        not increments or perturbations... because of the HU -> LQ conversion
 217    implicit none
 218
 219    ! arguments
 220    type(struct_gsv),           intent(inout) :: statevector_out
 221    character(len=*),           intent(in)    :: fileName
 222    type(struct_vco), pointer,  intent(in)    :: vco_file
 223    type(struct_hco), pointer,  intent(in)    :: hco_file
 224    character(len=*),           intent(in)    :: etiket_in
 225    character(len=*),           intent(in)    :: typvar_in
 226    integer,                    intent(in)    :: stepIndex
 227    logical,                    intent(in)    :: unitConversion
 228    logical,                    intent(in)    :: readHeightSfc
 229    logical,                    intent(in)    :: containsFullField
 230    type(struct_gsv), optional, intent(in)    :: statevectorRef_opt ! Reference statevector providing optional fields (P0, TT, HU)
 231
 232    ! locals
 233    real(4), pointer     :: field3d_r4_ptr(:,:,:)
 234    real(8), pointer     :: field_in_ptr(:,:,:,:), field_out_ptr(:,:,:,:)
 235
 236    character(len=4), pointer :: varNamesToRead(:)
 237
 238    type(struct_gsv) :: statevector_file_r4, statevector_tiles 
 239    type(struct_gsv) :: statevector_hinterp_r4, statevector_vinterp
 240
 241    nullify(field3d_r4_ptr, field_in_ptr, field_out_ptr)
 242
 243    write(*,*) ''
 244    write(*,*) 'readFromFileAndInterpToTiles: START'
 245
 246    nullify(varNamesToRead)
 247    call gsv_varNamesList(varNamesToRead, statevector_out)
 248
 249    !-- 1.0 Read the file, distributed over mpi task with respect to variables/levels
 250
 251    ! initialize single precision 3D working copy of statevector for reading file
 252    call gsv_allocate(statevector_file_r4, 1, hco_file, vco_file,                &
 253                      dateStamp_opt=statevector_out%datestamplist(stepIndex),    &
 254                      mpi_local_opt=.true., mpi_distribution_opt='VarsLevs',     &
 255                      dataKind_opt=4, allocHeightSfc_opt=readHeightSfc,          &
 256                      varNames_opt=varNamesToRead,                               &
 257                      hInterpolateDegree_opt=statevector_out%hInterpolateDegree, &
 258                      hExtrapolateDegree_opt=statevector_out%hExtrapolateDegree )
 259
 260    call gio_readFile(statevector_file_r4, filename, etiket_in, typvar_in,  &
 261                      containsFullField, readHeightSfc_opt=readHeightSfc)
 262
 263    !-- 2.0 Horizontal Interpolation
 264
 265    ! initialize single precision 3D working copy of statevector for horizontal interpolation result
 266    call gsv_allocate(statevector_hinterp_r4, 1, statevector_out%hco, vco_file,  &
 267                      dateStamp_opt=statevector_out%datestamplist(stepIndex),    &
 268                      mpi_local_opt=.true., mpi_distribution_opt='VarsLevs',     &
 269                      dataKind_opt=4, allocHeightSfc_opt=readHeightSfc,          &
 270                      varNames_opt=varNamesToRead,                               &
 271                      hInterpolateDegree_opt=statevector_out%hInterpolateDegree, &
 272                      hExtrapolateDegree_opt=statevector_out%hExtrapolateDegree )
 273
 274    call int_hInterp_gsv(statevector_file_r4, statevector_hinterp_r4)
 275
 276    call gsv_deallocate(statevector_file_r4)
 277
 278    !-- 3.0 Unit conversion
 279
 280    if ( unitConversion ) then
 281      call gio_fileUnitsToStateUnits( statevector_hinterp_r4, containsFullField )
 282    end if
 283
 284    !-- 4.0 MPI communication from vars/levels to lat/lon tiles
 285
 286    ! initialize double precision 3D working copy of statevector for mpi communication result
 287    call gsv_allocate(statevector_tiles, 1, statevector_out%hco, vco_file,    &
 288                      dateStamp_opt=statevector_out%datestamplist(stepIndex), &
 289                      mpi_local_opt=.true., mpi_distribution_opt='Tiles',     &
 290                      dataKind_opt=8, allocHeightSfc_opt=readHeightSfc,               &
 291                      varNames_opt=varNamesToRead )
 292
 293    call gsv_transposeVarsLevsToTiles(statevector_hinterp_r4, statevector_tiles)
 294
 295    call gsv_deallocate(statevector_hinterp_r4)
 296
 297    !-- 5.0 Vertical interpolation
 298
 299    ! initialize double precision 3D working copy of statevector for mpi communication result
 300    call gsv_allocate(statevector_vinterp, 1, statevector_out%hco, statevector_out%vco, &
 301                      dateStamp_opt=statevector_out%datestamplist(stepIndex), &
 302                      mpi_local_opt=.true., mpi_distribution_opt='Tiles', dataKind_opt=8, &
 303                      allocHeightSfc_opt=readHeightSfc, varNames_opt=varNamesToRead )
 304
 305    call int_vInterp_gsv( statevector_tiles, statevector_vinterp, & 
 306                          statevectorRef_opt=statevectorRef_opt)
 307
 308    call gsv_deallocate(statevector_tiles)
 309
 310    !-- 6.0 Copy result to output statevector
 311    call gsv_copy(statevector_vinterp, statevector_out, stepIndexOut_opt=stepIndex)
 312
 313    call gsv_deallocate(statevector_vinterp)
 314    deallocate(varNamesToRead)
 315
 316    write(*,*) 'readFromFileAndInterpToTiles: END'
 317
 318  end subroutine readFromFileAndInterpToTiles

 402             vco_file, hco_file, etiket_in, typvar_in, stepIndex, unitConversion,  &
 403             readHeightSfc, containsFullField)
 404    !
 405    !:Purpose: Read an RPN standard file and put the contents into a
 406    !           stateVector object. Wrapper subroutine that also proceed with
 407    !           (serial) interpolation.
 408    !
 409    implicit none
 410
 411    ! arguments
 412    type(struct_gsv),          intent(inout)  :: statevector_out_r4
 413    character(len=*),          intent(in)     :: fileName
 414    type(struct_vco), pointer, intent(in)     :: vco_file
 415    type(struct_hco), pointer, intent(in)     :: hco_file
 416    character(len=*),          intent(in)     :: etiket_in
 417    character(len=*),          intent(in)     :: typvar_in
 418    integer,                   intent(in)     :: stepIndex
 419    logical,                   intent(in)     :: unitConversion
 420    logical,                   intent(in)     :: readHeightSfc
 421    logical,                   intent(in)     :: containsFullField
 422
 423    ! locals
 424    real(4), pointer :: field_in_ptr(:,:,:,:), field_out_ptr(:,:,:,:)
 425
 426    character(len=4), pointer :: varNamesToRead(:)
 427
 428    type(struct_gsv) :: statevector_file_r4, statevector_hinterp_r4
 429    type(struct_gsv) :: statevector_vinterp_r4
 430
 431    nullify(field_in_ptr, field_out_ptr)
 432
 433    write(*,*) ''
 434    write(*,*) 'readFromFileAndInterp1Proc: START'
 435
 436    nullify(varNamesToRead)
 437    call gsv_varNamesList(varNamesToRead, statevector_out_r4)
 438
 439    !-- 1.0 Read the file
 440
 441    ! initialize single precision 3D working copy of statevector for reading file
 442    call gsv_allocate(statevector_file_r4, 1, hco_file, vco_file,                    &
 443                      dateStamp_opt=statevector_out_r4%datestamplist(stepIndex),     &
 444                      mpi_local_opt=.false., dataKind_opt=4,                         &
 445                      allocHeightSfc_opt=readHeightSfc, varNames_opt=varNamesToRead, &
 446                      hInterpolateDegree_opt=statevector_out_r4%hInterpolateDegree,  &
 447                      hExtrapolateDegree_opt=statevector_out_r4%hExtrapolateDegree)
 448
 449    call gio_readFile(statevector_file_r4, filename, etiket_in, typvar_in,  &
 450                      containsFullField, readHeightSfc_opt=readHeightSfc)
 451
 452    !-- 2.0 Horizontal Interpolation
 453
 454    ! initialize single precision 3D working copy of statevector for horizontal interpolation result
 455    call gsv_allocate(statevector_hinterp_r4, 1, statevector_out_r4%hco, vco_file,   &
 456                      dateStamp_opt=statevector_out_r4%datestamplist(stepIndex),     &
 457                      mpi_local_opt=.false., dataKind_opt=4,                         &
 458                      allocHeightSfc_opt=readHeightSfc, varNames_opt=varNamesToRead, &
 459                      hInterpolateDegree_opt=statevector_out_r4%hInterpolateDegree,  &
 460                      hExtrapolateDegree_opt=statevector_out_r4%hExtrapolateDegree)
 461
 462    call int_hInterp_gsv(statevector_file_r4, statevector_hinterp_r4)
 463
 464    call gsv_deallocate(statevector_file_r4)
 465
 466    !-- 3.0 Unit conversion (must come before vertical interp to get Psfc in Pascals)
 467
 468    if ( unitConversion ) then
 469      call gio_fileUnitsToStateUnits( statevector_hinterp_r4, containsFullField )
 470    end if
 471
 472    !-- 4.0 Vertical interpolation
 473
 474    ! initialize double precision 3D working copy of statevector for mpi communication result
 475    call gsv_allocate(statevector_vinterp_r4, 1, statevector_out_r4%hco, statevector_out_r4%vco, &
 476                      dateStamp_opt=statevector_out_r4%datestamplist(stepIndex),                 &
 477                      mpi_local_opt=.false., dataKind_opt=4,                                     &
 478                      allocHeightSfc_opt=readHeightSfc, varNames_opt=varNamesToRead)
 479
 480    call int_vInterp_gsv(statevector_hinterp_r4,statevector_vinterp_r4)
 481
 482    call gsv_deallocate(statevector_hinterp_r4)
 483
 484    !-- 5.0 Copy result to output statevector
 485    call gsv_copy(statevector_vinterp_r4, statevector_out_r4, stepIndexOut_opt=stepIndex)
 486
 487    call gsv_deallocate(statevector_vinterp_r4)
 488    deallocate(varNamesToRead)
 489
 490    write(*,*) 'readFromFileAndInterp1Proc: END'
 491
 492  end subroutine readFromFileAndInterp1Proc

 497  subroutine readFromFileOnly(statevector_out, fileName,  &
 498             etiket_in, typvar_in, stepIndex, unitConversion,  &
 499             readHeightSfc, containsFullField)
 500    !
 501    !:Purpose: Read an RPN standard file and put the contents into a
 502    !           stateVector object.  Wrapper subroutine
 503    !
 504    implicit none
 505
 506    ! arguments
 507    type(struct_gsv), intent(inout) :: statevector_out
 508    character(len=*), intent(in)    :: fileName
 509    character(len=*), intent(in)    :: etiket_in
 510    character(len=*), intent(in)    :: typvar_in
 511    integer,          intent(in)    :: stepIndex
 512    logical,          intent(in)    :: unitConversion
 513    logical,          intent(in)    :: readHeightSfc
 514    logical,          intent(in)    :: containsFullField
 515
 516    write(*,*) ''
 517    write(*,*) 'readFromFileOnly: Do simple reading with no interpolation and no mpi redistribution'
 518
 519    if ( statevector_out%dataKind /= 4) then
 520      call utl_abort('readFromFileOnly: Only compatible with dataKind=4')
 521    end if
 522
 523    call gio_readFile( statevector_out, filename, etiket_in, typvar_in,  &
 524                       containsFullField, readHeightSfc_opt=readHeightSfc, stepIndex_opt=stepIndex )
 525
 526    if ( unitConversion ) then
 527      call gio_fileUnitsToStateUnits( statevector_out, containsFullField, stepIndex_opt=stepIndex )
 528    end if
 529
 530    write(*,*) 'readFromFileOnly: END'
 531
 532  end subroutine readFromFileOnly

 534  !--------------------------------------------------------------------------
 535  ! gio_readFile
 536  !--------------------------------------------------------------------------
 537  subroutine gio_readFile(statevector, filename, etiket_in, typvar_in, &
 538                          containsFullField, readHeightSfc_opt, stepIndex_opt, &
 539                          ignoreDate_opt)
 540    !
 541    !:Purpose: Read an RPN standard file and put the contents into a
 542    !           stateVector object.  Low level subroutine that does the actual
 543    !           file reading.
 544    !
 545    implicit none
 546
 547    ! arguments
 548    type(struct_gsv),  intent(inout) :: statevector
 549    character(len=*),  intent(in)    :: fileName
 550    character(len=*),  intent(in)    :: etiket_in
 551    character(len=*),  intent(in)    :: typvar_in
 552    logical,           intent(in)    :: containsFullField
 553    logical, optional, intent(in)    :: readHeightSfc_opt
 554    integer, optional, intent(in)    :: stepIndex_opt
 555    logical, optional, intent(in)    :: ignoreDate_opt
 556
 557    ! locals
 558    integer :: nulfile, ierr, ip1, ni_file, nj_file, nk_file, kIndex, stepIndex
 559    integer :: ikey, levIndex
 560    integer :: stepIndexBeg, stepIndexEnd, ni_var, nj_var, nk_var
 561    integer :: fnom, fstouv, fclos, fstfrm, fstlir, fstinf
 562    integer :: fstprm, EZscintID_var, ezdefset, ezqkdef
 563
 564    integer :: dateo_var, deet_var, npas_var, nbits_var, datyp_var
 565    integer :: ip1_var, ip2_var, ip3_var, swa_var, lng_var, dltf_var, ubc_var
 566    integer :: extra1_var, extra2_var, extra3_var
 567    integer :: ig1_var, ig2_var, ig3_var, ig4_var
 568    integer :: varIndex, dateStampList(statevector%numStep)
 569
 570    character(len=4 ) :: nomvar_var
 571    character(len=2 ) :: typvar_var
 572    character(len=1 ) :: grtyp_var
 573    character(len=12) :: etiket_var
 574
 575    real(4), pointer :: field_r4_ptr(:,:,:,:)
 576    real(4), pointer :: gd2d_file_r4(:,:), gd2d_r4_UV_ptr(:,:,:)
 577    real(8), pointer :: heightSfc_ptr(:,:)
 578    real(4), allocatable :: gd2d_var_r4(:,:)
 579
 580    character(len=4)  :: varName, varNameToRead
 581    character(len=4)  :: varLevel
 582
 583    type(struct_vco), pointer :: vco_file
 584    type(struct_hco), pointer :: hco_file
 585
 586    logical :: foundVarNameInFile, ignoreDate
 587
 588    write(*,*) 'gio_readFile: starting'
 589    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 590
 591    call readNml()
 592
 593    vco_file => gsv_getVco(statevector)
 594
 595    if ( statevector%mpi_distribution /= 'VarsLevs' .and. &
 596         statevector%mpi_local ) then
 597      call utl_abort('gio_readFile: statevector must have ' //   &
 598                     'complete horizontal fields on each mpi task.')
 599    end if
 600
 601    if ( present(stepIndex_opt) ) then
 602      stepIndexBeg = stepIndex_opt
 603      stepIndexEnd = stepIndex_opt
 604    else
 605      stepIndexBeg = 1
 606      stepIndexEnd = statevector%numStep
 607    end if
 608
 609    if ( present(ignoreDate_opt) ) then
 610      ignoreDate = ignoreDate_opt
 611    else
 612      ignoreDate = .false.
 613    end if
 614
 615    if (.not. associated(statevector%dateStampList)) then
 616      call utl_abort('gio_readFile: dateStampList of statevector is not associated with a target!')
 617    else
 618      dateStampList(:) = statevector%dateStampList(:)
 619      if (ignoreDate) then
 620        write(*,*) 'gio_readFile: as requested, ignoring the date when reading fields'
 621        dateStampList(:) = -1
 622      end if
 623    end if
 624
 625    !- Open input field
 626    nulfile = 0
 627    write(*,*) 'gio_readFile: file name = ',trim(fileName)
 628    ierr = fnom(nulfile,trim(fileName),'RND+OLD+R/O',0)
 629
 630    if (ierr >= 0) then
 631      ierr  =  fstouv(nulfile,'RND+OLD')
 632    else
 633      call utl_abort('gio_readFile: problem opening input file')
 634    end if
 635
 636    if (nulfile == 0) then
 637      call utl_abort('gio_readFile: unit number for input file not valid')
 638    end if
 639
 640    ! Read surface height if requested
 641    if (present(readHeightSfc_opt)) then
 642      if (readHeightSfc_opt .and. gsv_isAssocHeightSfc(statevector)) then
 643        write(*,*) 'gio_readFile: reading the surface height'
 644        varName = 'GZ'
 645        ip1 = statevector%vco%ip1_sfc
 646        typvar_var = typvar_in
 647        ikey = fstinf(nulfile, ni_file, nj_file, nk_file,  &
 648                      -1, etiket_in, &
 649                      -1, -1, -1, typvar_var, varName)
 650
 651        if (ikey < 0) then
 652          if (trim(typvar_in) /= "") then
 653            typvar_var(2:2) = '@'
 654            ikey = fstinf(nulfile, ni_file, nj_file, nk_file,  &
 655                 -1, etiket_in, &
 656                 -1, -1, -1, typvar_var, varName)
 657          end if
 658          if (ikey < 0) then
 659            write(*,*) 'gio_readFile: etiket_in = ', etiket_in
 660            write(*,*) 'gio_readFile: typvar_in = ', typvar_in
 661            call utl_abort('gio_readFile: Problem with reading surface height from file')
 662          end if
 663        end if
 664
 665        if (ni_file /= statevector%hco%ni .or. nj_file /= statevector%hco%nj) then
 666          write(*,*) 'ni, nj in file        = ', ni_file, nj_file
 667          write(*,*) 'ni, nj in statevector = ', statevector%hco%ni, statevector%hco%nj
 668          call utl_abort('gio_readFile: Dimensions of surface height not consistent')
 669        end if
 670
 671        allocate(gd2d_file_r4(ni_file,nj_file))
 672        gd2d_file_r4(:,:) = 0.0d0
 673        ierr = fstlir(gd2d_file_r4(:,:), nulfile, ni_file, nj_file, nk_file,  &
 674                      -1,etiket_in,ip1,-1,-1,  &
 675                      typvar_var,varName)
 676        if (ierr < 0) then
 677          write(*,*) 'ip1 = ', ip1
 678          write(*,*) 'etiket_in = ', etiket_in
 679          write(*,*) 'typvar_var = ', typvar_var
 680          call utl_abort('gio_readFile: Problem with reading surface height from file')
 681        end if
 682        heightSfc_ptr => gsv_getHeightSfc(statevector)
 683        heightSfc_ptr = real(gd2d_file_r4(1:gsv_getHco(statevector)%ni,1:gsv_getHco(statevector)%nj),8)*10.0d0
 684        deallocate(gd2d_file_r4)
 685      end if
 686    end if
 687
 688    nullify(hco_file)
 689    nullify(gd2d_file_r4)
 690    if ( statevector%mykCount > 0 ) then
 691      if (statevector%hco%global) then
 692
 693        foundVarNameInFile = .false.
 694        do varIndex = 1, vnl_numvarmax
 695          varName = vnl_varNameList(varIndex)
 696
 697          if (.not. gsv_varExist(statevector,varName)) cycle
 698
 699          ! make sure variable is in the file
 700          if ( .not. utl_varNamePresentInFile(varName,fileName_opt=trim(fileName)) ) cycle
 701
 702          ! adopt a variable on the full/dynamic LAM grid
 703          if ( (trim(varName) == 'TM'   .or. trim(varName) == 'MG' ) ) cycle
 704
 705          foundVarNameInFile = .true.
 706
 707          exit
 708        end do
 709
 710        ! special case when only TM (Surface Temperature) is in the file:
 711        if ( .not. foundVarNameInFile ) then
 712          varname = 'TM'
 713          if (gsv_varExist( statevector, varname ) .and. &
 714              utl_varNamePresentInFile( varname, fileName_opt = trim( fileName ))) &
 715            foundVarNameInFile = .true.
 716        end if   
 717
 718        ! to be safe for situations where, e.g. someone wants to only read MG from a file
 719        if (.not. foundVarNameInFile) then
 720          varname = 'P0'
 721          if (utl_varNamePresentInFile( varname, fileName_opt = trim( fileName))) &
 722            foundVarNameInFile = .true.
 723        end if
 724
 725        if (.not. foundVarNameInFile) call utl_abort('gio_readFile: NO variable is in the file')
 726
 727        call hco_setupFromFile(hco_file, filename, ' ', 'INPUTFILE', varName_opt=varName)
 728
 729      else
 730        ! In LAM mode, force the input file dimensions to be always identical to the input statevector dimensions
 731        hco_file => statevector%hco
 732
 733        ! Also attempt to set up the physics grid
 734        if (interpToPhysicsGrid) then
 735          var_loop: do varIndex = 1, vnl_numvarmax
 736            varName = vnl_varNameList(varIndex)
 737            if (.not. gsv_varExist(statevector,varName)) cycle var_loop
 738            if (.not. vnl_isPhysicsVar(varName)) cycle var_loop
 739            if (utl_varNamePresentInFile(varName, fileName_opt=filename) .and. &
 740               .not. associated(statevector%hco_physics)) then
 741              write(*,*) 'gio_readFile: set up physics grid using the variable:', varName
 742              call hco_SetupFromFile(statevector%hco_physics, filename, ' ', &
 743                                     'INPUTFILE', varName_opt=varName)
 744              exit var_loop
 745            end if
 746          end do var_loop
 747        end if
 748
 749      end if
 750      allocate(gd2d_file_r4(hco_file%ni,hco_file%nj))
 751      gd2d_file_r4(:,:) = 0.0
 752    end if
 753
 754    ! Read all other fields needed for this MPI task
 755    call gsv_getField(statevector,field_r4_ptr)
 756    do stepIndex = stepIndexBeg, stepIndexEnd
 757      k_loop: do kIndex = statevector%mykBeg, statevector%mykEnd
 758        varName = gsv_getVarNameFromK(statevector,kIndex)
 759        levIndex = gsv_getLevFromK(statevector,kIndex)
 760
 761        if (.not.gsv_varExist(statevector,varName)) cycle k_loop
 762
 763        ! Check that the wanted field is present in the file
 764        if (utl_varNamePresentInFile(varName,fileUnit_opt=nulfile)) then
 765          varNameToRead = varName
 766        else
 767          select case (trim(varName))
 768          case ('LVIS')
 769            varNameToRead = 'VIS'
 770          case ('Z_T','Z_M','P_T','P_M')
 771            cycle k_loop
 772          case ('LPR')
 773            varNameToRead = 'PR'
 774          case default
 775            call utl_abort('gio_readFile: variable '//trim(varName)//' was not found in '//trim(fileName))
 776          end select
 777        end if
 778
 779        varLevel = vnl_varLevelFromVarname(varNameToRead)
 780        if (varLevel == 'MM') then
 781          ip1 = vco_file%ip1_M(levIndex)
 782        else if (varLevel == 'TH') then
 783          ip1 = vco_file%ip1_T(levIndex)
 784        else if (varLevel == 'SF') then
 785          ip1 = -1
 786        else if (varLevel == 'SFTH') then
 787          ip1 = vco_file%ip1_T_2m
 788        else if (varLevel == 'SFMM') then
 789          ip1 = vco_file%ip1_M_10m
 790        else if (varLevel == 'OT') then
 791          ip1 = vco_ip1_other(levIndex)
 792        else if (varLevel == 'DP') then
 793          ip1 = vco_file%ip1_depth(levIndex)
 794        else if (varLevel == 'SS') then
 795          ip1 = -1
 796        else
 797          write(*,*) 'varLevel =', varLevel
 798          call utl_abort('gio_readFile: unknown varLevel')
 799        end if
 800
 801        typvar_var = typvar_in
 802
 803        ! Make sure that the input variable has the same grid size than hco_file
 804        ikey = fstinf(nulfile, ni_var, nj_var, nk_var,         &
 805                      datestamplist(stepIndex), etiket_in, &
 806                      -1, -1, -1, typvar_var, varNameToRead)
 807
 808        if (ikey < 0) then
 809          if (trim(typvar_in) /= '') then
 810            typvar_var(2:2) = '@'
 811            ikey = fstinf(nulfile, ni_var, nj_var, nk_var, &
 812                          datestamplist(stepIndex), etiket_in, &
 813                          -1, -1, -1, typvar_var, varNameToRead)
 814          end if
 815          if (ikey < 0) then
 816            write(*,*) 'gio_readFile: looking for datestamp = ', datestamplist(stepIndex)
 817            write(*,*) 'gio_readFile: etiket_in = ', etiket_in
 818            write(*,*) 'gio_readFile: typvar_in = ', typvar_in
 819            call utl_abort('gio_readFile: cannot find field ' // trim(varNameToRead) // ' in file ' // trim(fileName))
 820          end if
 821        end if
 822
 823        ierr = fstprm(ikey,                                                               & ! IN
 824                      dateo_var, deet_var, npas_var, ni_var, nj_var, nk_var, nbits_var,   & ! OUT
 825                      datyp_var, ip1_var, ip2_var, ip3_var, typvar_var, nomvar_var,       & ! OUT
 826                      etiket_var, grtyp_var, ig1_var, ig2_var, ig3_var, ig4_var, swa_var, & ! OUT
 827                      lng_var, dltf_var, ubc_var, extra1_var, extra2_var, extra3_var )      ! OUT
 828        statevector%deet                      = deet_var
 829        statevector%ip2List(stepIndex)        = ip2_var
 830        statevector%npasList(stepIndex)       = npas_var
 831        statevector%dateOriginList(stepIndex) = dateo_var
 832        statevector%etiket                    = etiket_var
 833
 834        ! Check if we found a mask field by mistake - if yes, need to fix the code!
 835        if (typvar_var == '@@') then
 836          call utl_abort('gio_readFile: read a mask file by mistake - need to modify file or fix the code')
 837        end if
 838
 839        if (ni_var == hco_file%ni .and. nj_var == hco_file%nj) then
 840          ierr = fstlir(gd2d_file_r4(:,:),nulfile,ni_file, nj_file, nk_file,  &
 841                        datestamplist(stepIndex),etiket_in,ip1,-1,-1,  &
 842                        typvar_var,varNameToRead)
 843        else
 844          ! Special cases for variables that are on a different horizontal grid in LAM (e.g. TG)
 845          write(*,*)
 846          write(*,*) 'gio_readFile: variable on a different horizontal grid = ',trim(varNameToRead)
 847          write(*,*) ni_var, hco_file%ni, nj_var, hco_file%nj
 848          if (interpToPhysicsGrid) then
 849            if (associated(statevector%hco_physics)) then
 850              if (ni_var == statevector%hco_physics%ni .and. &
 851                  nj_var == statevector%hco_physics%nj) then
 852                write(*,*) 'gio_readFile: this variable on same grid as other physics variables'
 853                statevector%onPhysicsGrid(vnl_varListIndex(varName)) = .true.
 854              else
 855                call utl_abort('gio_readFile: this variable not on same grid as other physics variables')
 856              end if
 857            else
 858              call utl_abort('gio_readFile: physics grid has not been set up')
 859            end if
 860          end if
 861
 862          if (statevector%hco%global) then
 863            call utl_abort('gio_readFile: This is not allowed in global mode!')
 864          end if
 865
 866          EZscintID_var  = ezqkdef(ni_var, nj_var, grtyp_var, ig1_var, ig2_var, ig3_var, ig4_var, nulfile) ! IN
 867
 868          allocate(gd2d_var_r4(ni_var,nj_var))
 869          gd2d_var_r4(:,:) = 0.0
 870
 871          ierr = fstlir(gd2d_var_r4(:,:),nulfile,ni_var, nj_var, nk_var,  &
 872                        datestamplist(stepIndex),etiket_in,ip1,-1,-1,  &
 873                        typvar_in,varNameToRead)
 874
 875          ierr = ezdefset(hco_file%EZscintID,EZscintID_var)
 876          ierr = int_hInterpScalar( gd2d_file_r4, gd2d_var_r4, &
 877                                    interpDegree='NEAREST', extrapDegree_opt='NEUTRAL' )
 878
 879          ! read the corresponding mask if it exists
 880          if (typvar_var(2:2) == '@') then
 881            write(*,*) 'gio_readFile: read mask that needs interpolation for variable name: ', nomvar_var
 882            call utl_abort('gio_readFile: not implemented yet')
 883          end if
 884
 885          deallocate(gd2d_var_r4)
 886        end if
 887
 888        if (varNameToRead == varName .or. .not. containsFullField) then
 889          field_r4_ptr(:,:,kIndex,stepIndex) = gd2d_file_r4(1:statevector%hco%ni,1:statevector%hco%nj)
 890        else
 891          select case (trim(varName))
 892          case ('LVIS')
 893            field_r4_ptr(:,:,kIndex,stepIndex) = &
 894                 log(max(min(gd2d_file_r4(1:statevector%hco%ni,1:statevector%hco%nj),mpc_maximum_vis_r4),mpc_minimum_vis_r4))
 895          case ('LPR')
 896            field_r4_ptr(:,:,kIndex,stepIndex) = &
 897                 log(mpc_minimum_pr_r4 + max(0.0,gd2d_file_r4(1:statevector%hco%ni,1:statevector%hco%nj)))
 898          case default
 899            call utl_abort('gio_readFile: Oups! This should not happen... Check the code.')
 900          end select
 901        endif
 902
 903        if (ierr < 0)then
 904          write(*,*) varNameToRead, ip1, datestamplist(stepIndex)
 905          call utl_abort('gio_readFile: Problem with reading file')
 906        end if
 907
 908        ! When mpi distribution could put UU on a different mpi task than VV
 909        ! or only one wind component present in statevector
 910        ! then we re-read the corresponding UV component and store it
 911        if (statevector%extraUVallocated) then
 912          if (varName == 'UU') then
 913            ierr = fstlir(gd2d_file_r4(:,:),nulfile, ni_file, nj_file, nk_file,  &
 914                          datestamplist(stepIndex),etiket_in,ip1,-1,-1,  &
 915                          typvar_in,'VV')
 916            call gsv_getFieldUV(statevector, gd2d_r4_UV_ptr, kIndex)
 917            gd2d_r4_UV_ptr(1:gsv_getHco(statevector)%ni,1:gsv_getHco(statevector)%nj, stepIndex) &
 918                 = gd2d_file_r4(1:gsv_getHco(statevector)%ni,1:gsv_getHco(statevector)%nj)
 919          else if (varName == 'VV') then
 920            ierr = fstlir(gd2d_file_r4(:,:),nulfile, ni_file, nj_file, nk_file,  &
 921                          datestamplist(stepIndex),etiket_in,ip1,-1,-1,  &
 922                          typvar_in,'UU')
 923            call gsv_getFieldUV(statevector, gd2d_r4_UV_ptr, kIndex)
 924            gd2d_r4_UV_ptr(1:gsv_getHco(statevector)%ni,1:gsv_getHco(statevector)%nj, stepIndex) &
 925                 = gd2d_file_r4(1:gsv_getHco(statevector)%ni,1:gsv_getHco(statevector)%nj)
 926          end if
 927        end if
 928
 929      end do k_loop
 930    end do
 931
 932    if (statevector%hco%global .and. statevector%mykCount > 0) then
 933      write(*,*) 'deallocating hco_file'
 934      call hco_deallocate(hco_file)
 935    end if
 936
 937    ierr = fstfrm(nulfile)
 938    ierr = fclos(nulfile)        
 939    if ( associated(gd2d_file_r4) ) deallocate(gd2d_file_r4)
 940
 941    ! Read in an oceanMask if it is present in the file
 942    call gio_readMaskFromFile(statevector, trim(filename))
 943
 944    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 945    write(*,*) 'gio_readFile: finished'
 946
 947  end subroutine gio_readFile

1001  !--------------------------------------------------------------------------
1002  ! gio_readTrials
1003  !--------------------------------------------------------------------------
1004  subroutine gio_readTrials(stateVectorTrialIn)
1005    !
1006    !:Purpose: Reading trials
1007    !
1008    implicit none
1009
1010    ! Arguments
1011    type(struct_gsv), target, intent(inout) :: stateVectorTrialIn
1012
1013    ! Locals
1014    logical             :: fileExists, allocHeightSfc
1015    logical             :: useInputStateVectorTrial 
1016    integer, parameter  :: maxNumTrials = 100
1017    integer             :: fnom, fstouv, fclos, fstfrm, fstinf
1018    integer             :: ierr, ikey, stepIndex, stepIndexToRead, trialIndex, nulTrial
1019    integer             :: ni_file, nj_file, nk_file, dateStamp, varNameIndex
1020    integer             :: procToRead, numBatch, batchIndex, stepIndexBeg, stepIndexEnd
1021
1022    character(len=2)          :: fileNumber
1023    character(len=512)        :: fileName

1179      call gsv_deallocate( stateVectorTrial )

1183    write(*,*) 'gio_readTrials: FINISHED'
1184    write(*,*) ''

1798  subroutine gio_fileUnitsToStateUnits(statevector, containsFullField, stepIndex_opt)
1799    !
1800    !:Purpose: Unit conversion needed after reading RPN standard file
1801    !
1802    implicit none
1803
1804    ! Arguments:
1805    type(struct_gsv),  intent(inout)  :: statevector
1806    logical,           intent(in)     :: containsFullField
1807    integer, optional, intent(in)     :: stepIndex_opt
1808
1809    ! Locals:
1810    integer :: stepIndex, stepIndexBeg, stepIndexEnd, kIndex
1811
1812    real(4), pointer :: field_r4_ptr(:,:,:,:), fieldUV_r4_ptr(:,:,:)
1813    real(8)          :: multFactor
1814
1815    character(len=4) :: varName
1816
1817    if ( present(stepIndex_opt) ) then
1818      stepIndexBeg = stepIndex_opt
1819      stepIndexEnd = stepIndex_opt
1820    else
1821      stepIndexBeg = 1
1822      stepIndexEnd = statevector%numStep
1823    end if
1824
1825    call gsv_getField(statevector,field_r4_ptr)
1826
1827    step_loop: do stepIndex = stepIndexBeg, stepIndexEnd
1828
1829      ! Do unit conversion for all variables
1830      do kIndex = statevector%mykBeg, statevector%mykEnd
1831        varName = gsv_getVarNameFromK(statevector,kIndex)
1832
1833        if ( trim(varName) == 'UU' .or. trim(varName) == 'VV') then
1834          multFactor = mpc_m_per_s_per_knot_r8 ! knots -> m/s
1835        else if ( trim(varName) == 'P0' .or. trim(varName) == 'P0LS' ) then
1836          multFactor = mpc_pa_per_mbar_r8 ! hPa -> Pa
1837        else if ( vnl_varKindFromVarname(trim(varName)) == 'CH' ) then 
1838          if ( gsv_conversionVarKindCHtoMicrograms ) then
1839            if ( trim(varName) == 'TO3' .or. trim(varName) == 'O3L' ) then
1840              ! Convert from volume mixing ratio to micrograms/kg
1841              ! Standard ozone input would not require this conversion as it is already in micrograms/kg
1842              multFactor = 1.0d9*vnl_varMassFromVarName(trim(varName)) &
1843                           /mpc_molar_mass_dry_air_r8 ! vmr -> micrograms/kg
1844            else
1845              multFactor = 1.0d0 ! no conversion
1846            end if
1847          else
1848            multFactor = 1.0d0 ! no conversion
1849          end if
1850        else
1851          multFactor = 1.0d0 ! no conversion
1852        end if
1853
1854        if ( multFactor /= 1.0d0 ) then
1855          field_r4_ptr(:,:,kIndex,stepIndex) = real( multFactor * field_r4_ptr(:,:,kIndex,stepIndex), 4 )
1856        end if
1857
1858        if ( trim(varName) == 'TT' .and. containsFullField ) then
1859          field_r4_ptr(:,:,kIndex,stepIndex) = real( field_r4_ptr(:,:,kIndex,stepIndex) +  &

1862

1864          where (field_r4_ptr(:,:,kIndex,stepIndex) < 100.0)
1865            field_r4_ptr(:,:,kIndex,stepIndex) = real( field_r4_ptr(:,:,kIndex,stepIndex) + &
1866                                                       mpc_k_c_degree_offset_r8, 4 )
1867          end where
1868        end if
1869
1870        if ( trim(varName) == 'VIS' .and. containsFullField ) then
1871          field_r4_ptr(:,:,kIndex,stepIndex) = min(field_r4_ptr(:,:,kIndex,stepIndex),mpc_maximum_vis_r4)
1872        end if
1873
1874        if ( vnl_varKindFromVarname(trim(varName)) == 'CH' .and. containsFullField ) then 
1875          if ( gsv_minValVarKindCH(vnl_varListIndex(varName)) > 1.01*mpc_missingValue_r8 ) &
1876            field_r4_ptr(:,:,kIndex,stepIndex) = max( field_r4_ptr(:,:,kIndex,stepIndex), &
1877              real(gsv_minValVarKindCH(vnl_varListIndex(trim(varName)))) )

1886      if ( statevector%extraUVallocated ) then

1889        !$OMP PARALLEL DO PRIVATE (kIndex, fieldUV_r4_ptr)

1893          fieldUV_r4_ptr(:,:,stepIndex) =  &
1894               real( multFactor * fieldUV_r4_ptr(:,:,stepIndex), 4)
