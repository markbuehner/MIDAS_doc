 18  ! MODULE localizationFunction_mod (prefix='lfn' category='2. B and R matrices')
 19  !
 20  !:Purpose: To store various functions for horizontal and vertical covariance
 21  !           localization. Length-scale estimation for function available in
 22  !           this module is also possible through curve fitting using
 23  !           pre-computed optimal separation-distance localization values.
 24  !

 39  !--------------------------------------------------------------------------
 40  ! lfn_Setup
 41  !--------------------------------------------------------------------------
 42  subroutine lfn_Setup(LocFunctionWanted)
 43    implicit none
 44    
 45    character(len=*), intent(in) :: LocFunctionWanted
 46  
 47    select case(trim(LocFunctionWanted))
 48    case ('FifthOrder')
 49       write(*,*)
 50       write(*,*) 'lfn_setup: Using Gaspari and Cohn 5th order piecewise rationale function'
 51       LocFunction='FifthOrder'
 52    case default
 53       write(*,*)
 54       write(*,*) 'Unsupported HORIZONTAL localization function : ', trim(LocFunctionWanted)
 55       call utl_abort('lfn_setup')
 56    end select
 57
 58    initialized  = .true.
 59
 60  end subroutine lfn_Setup

 65  function lfn_Response(distance,lengthscale) result(correlation)
 66    implicit none
 67    
 68    real(8) :: distance, lengthscale
 69    real(8) :: correlation
 70
 71    if ( .not. initialized ) then
 72      write(*,*)

 74      call utl_abort('lfn_horizResponse')

114  !--------------------------------------------------------------------------
115  ! lfn_FifthOrderFunction
116  !--------------------------------------------------------------------------
117  function lfn_FifthOrderFunction(distance,lengthscale) result(correlation)
118    implicit none
119    
120    real(8) :: distance, lengthscale, halflength
121    real(8) :: correlation
122
123    halflength = lengthscale/2.d0
124
125    if ( distance <= halflength ) then
126       correlation = -0.250d0*(distance/halflength)**5  &
127                     +         0.5d0*(distance/halflength)**4  &
128                     +       0.625d0*(distance/halflength)**3  &
129                     - (5.0d0/3.0d0)*(distance/halflength)**2  &
130                     + 1.0d0
131    else if ( distance <= (2.0d0*halflength) ) then
132       correlation =  (1.0d0/12.0d0)*(distance/halflength)**5  &
133                     -         0.5d0*(distance/halflength)**4  &
134                     +       0.625d0*(distance/halflength)**3  &
135                     + (5.0d0/3.0d0)*(distance/halflength)**2  &
136                     -         5.0d0*(distance/halflength)     &
137                     + 4.0d0                                 &
138                     - (2.0d0/3.0d0)*(halflength/distance) 
139    else
140       correlation = 0.d0
141    endif
142
143  end function lfn_FifthOrderFunction

145  !--------------------------------------------------------------------------
146  ! lfn_FifthOrderFunctionGradient
147  !--------------------------------------------------------------------------
148  function lfn_FifthOrderFunctionGradient(distance,lengthscale) result(gradient)
149    implicit none
150
151    real(8) :: distance, lengthscale, halflength
152    real(8) :: gradient
153
154    halflength = lengthscale/2.d0
155
156    ! Compute df/dhalflength
157    if ( distance <= halflength ) then
158       gradient =     5.d0*0.250d0*(distance/halflength)**5 / halflength  &
159                  -     4.d0*0.5d0*(distance/halflength)**4 / halflength  &
160                  -   3.d0*0.625d0*(distance/halflength)**3 / halflength  &
161                  + (10.0d0/3.0d0)*(distance/halflength)**2 / halflength
162    else if ( distance <= (2.0d0*halflength) ) then
163       gradient =  (-5.0d0/12.0d0)*(distance/halflength)**5 / halflength  &
164                  +    4.0d0*0.5d0*(distance/halflength)**4 / halflength  &
165                  -  3.0d0*0.625d0*(distance/halflength)**3 / halflength  &
166                  - (10.0d0/3.0d0)*(distance/halflength)**2 / halflength  &
167                  +          5.0d0*(distance/halflength)    / halflength  &
168                  -   (2.0d0/3.0d0)*(1.0d0/distance) 
169    else
170       gradient = 0.d0
171    endif
172
173    ! Compute df/dlengthscale (= df/dhalflength * dhalflength/dlengthscale)
174    gradient = gradient / 2.d0
175
176  end function lfn_FifthOrderFunctionGradient

178!--------------------------------------------------------------------------
179! lfn_CreateBiPerFunction
180!--------------------------------------------------------------------------
181  SUBROUTINE  lfn_CreateBiPerFunction(gridpoint,CorrelLength,dlon,ni,nj,nk)
182    use earthConstants_mod
183    implicit none
184
185    integer, intent(in)  :: ni,nj,nk
186    real(8), intent(in)  :: dlon             ! in radian
187    real(8), intent(in)  :: CorrelLength(nk) ! in km
188
189    real(8), intent(out) :: gridpoint(ni,nj,nk)
190
191    integer          :: i, j, k, iref, jref
192    real(8)          :: distance, distance_ref
193
194    gridpoint(:,:,:) = 0.d0
195
196    distance_ref = dlon * ec_ra
197
198    !
199    !- Create a bi-periodic correlation function by centering the function in each 4 corners
200    !
201
202    !- Lower-Left Corner (reference corner)
203    iref = 1
204    jref = 1
205    do j = 1, nj
206       do i = 1, ni
207          distance = distance_ref * sqrt( real((i-iref)**2 + (j-jref)**2,8) )
208          do k = 1, nk
209             gridpoint(i,j,k) = gridpoint(i,j,k) + lfn_response(distance,1000.d0*CorrelLength(k)) 
210          enddo
211       enddo
212    enddo
213
214    !- Upper-Left Corner
215    iref = 1
216    jref = nj+1 ! the lam spectral transform is periodic at nj+1
217    do j = 1, nj
218       do i = 1, ni
219          distance = distance_ref * sqrt( real((i-iref)**2 + (j-jref)**2,8) )
220          do k = 1, nk
221             gridpoint(i,j,k) = gridpoint(i,j,k) + lfn_response(distance,1000.d0*CorrelLength(k)) 
222          enddo
223       enddo
224    enddo
225
226    !- Lower-Right Corner
227    iref = ni+1 ! the lam spectral transform is periodic at ni+1
228    jref = 1
229    do j = 1, nj
230       do i = 1, ni
231          distance = distance_ref * sqrt( real((i-iref)**2 + (j-jref)**2,8) )
232          do k = 1, nk
233             gridpoint(i,j,k) = gridpoint(i,j,k) + lfn_response(distance,1000.d0*CorrelLength(k)) 
234          enddo
235       enddo
236    enddo
237
238    !- Upper-Right Corner
239    iref = ni+1 ! the lam spectral transform is periodic at ni+1
240    jref = nj+1 ! the lam spectral transform is periodic at nj+1
241    do j = 1, nj
242       do i = 1, ni
243          distance = distance_ref * sqrt( real((i-iref)**2 + (j-jref)**2,8) )
244          do k = 1, nk
245             gridpoint(i,j,k) = gridpoint(i,j,k) + lfn_response(distance,1000.d0*CorrelLength(k)) 
246          enddo
247       enddo
248    enddo
249
250  END SUBROUTINE Lfn_CreateBiPerFunction

252  !--------------------------------------------------------------------------
253  ! LFN_LENGTHSCALE
254  !--------------------------------------------------------------------------
255  subroutine lfn_lengthscale(lengthscale, rmse, localizationValues, &
256       distance, weight, numbins)
257    implicit none
258
259    integer, intent(in)  :: numBins
260    real(8), intent(in)  :: localizationValues(numBins)
261    real(8), intent(in)  :: distance(numBins)
262    real(8), intent(in)  :: weight(numBins)
263
264    real(8), intent(inout) :: lengthscale
265    real(8), intent(out)   :: rmse
266
267    integer :: ierr, ilist
268    real(8) :: minv, fit
269
270    ! 1.  Initialization
271    ilist       = 1 
272    ierr        = 0
273    minv        = 0.01
274
275    ! 2.  Apply Curve fitting algorithm
276    call lfn_curveFit(numBins, distance,      & ! IN
277                      localizationValues,     & ! IN
278                      weight, numBins,        & ! IN
279                      lengthscale,            & ! INOUT
280                      minv, ilist,            & ! IN
281                      ierr,                   & ! INOUT
282                      fit)                      ! OUT
283
284    ! 3.  root mean squared deviation output as 'degree of fit'
285    rmse = sqrt(fit)
286
287  end subroutine lfn_lengthscale

289  !--------------------------------------------------------------------------
290  ! LFN_CURVEFIT
291  !--------------------------------------------------------------------------
292  subroutine lfn_curveFit(Nn, x, y, w, nmax, param, minv, ilist, ierr, ssq)
293    !
294    !:Purpose: This subroutine computes the lengthscale (sl, here: param) of a
295    !          FUNCTION that best fits (in a least-square perspective) a
296    !          data set. The method follows routine CURVEFIT from:
297    !          Heeswijk, M.V., and C.G. Fox, 1988: Iterative Method and Fortran
298    !          Code for Nonlinear Curve Fitting, Computers and Geosciences, 14,
299    !          4, pp. 489-503.
300    implicit none
301
302    ! Arguments:
303    INTEGER,            INTENT(IN)    :: Nn
304    REAL(8),            INTENT(IN)    :: x(Nn)
305    REAL(8),            INTENT(IN)    :: y(Nn)
306    REAL(8),            INTENT(IN)    :: w(Nn)
307    INTEGER,            INTENT(IN)    :: nmax
308    REAL(8),            INTENT(INOUT) :: param
309    REAL(8),            INTENT(IN)    :: minv
310    INTEGER,            INTENT(IN)    :: ilist
311    INTEGER,            INTENT(INOUT) :: ierr
312    REAL(8),            INTENT(OUT)   :: ssq
313
314    ! Locals:
315    INTEGER, PARAMETER    :: icon = 100   ! max iteration
316    INTEGER, PARAMETER    :: lbis = 10    ! max bisection
317    
318    REAL(8)              :: d(Nn)
319    REAL(8)              :: g(Nn)
320    REAL(8)              :: m
321    REAL(8)              :: gtd
322    REAL(8)              :: gtg
323    REAL(8)              :: gtginv
324    REAL(8)              :: ssqold, rootmsq
325    REAL(8)              :: w_t
326    
327    INTEGER           :: iter,nbis,i                     ! Loop counters.
328    
329    CHARACTER(LEN=60) :: FMT1
330    CHARACTER(LEN=12), PARAMETER :: RoutineName = 'lfn_curveFit'
331    
332    !-------------------------------------------------------------------------------
333    
334    !-------------------------------------------------------------------------------
335    ! 1. Initialize variables
336    !-------------------------------------------------------------------------------
337    ierr   = 0          ! Error flags
338    iter   = 0          ! Iteration counter
339    nbis   = 0          ! Bisection counter
340    m      = 9.9E9      ! dummy value to initialize m
341    ssqold = 9.9E9      ! dummy value to initialize ssqold
342    
343    !-------------------------------------------------------------------------------
344    ! 2. Perform curve fitting using a minimization (least-square) approach
345    !-------------------------------------------------------------------------------
346    !PRINT*, RoutineName, ' fitting ', trim(locFunction)
347    
348    DO WHILE ( (ABS(m) > minv) .AND. (iter <= icon) .AND. (nbis <= lbis) )
349       
350       ! 2.1  Form vector D; i.e. a data point minus the isolated Taylor Series
351       !      term for that data point. In addition calculate the sum of the 
352       !      square of the residuals
353       ssq = 0.d0
354       DO i = 1, nmax
355          d(i) = y(i) - lfn_response(x(i), param)
356          ssq = ssq + d(i)**2 * w(i)
357       END DO
358       rootmsq = sqrt(ssq)
359       
360       ! 2.2  If we are converging
361       IF (ssq < ssqold) THEN
362          
363          ! The matrix formation in the following program segment follows
364          ! equation 3.12 on page 41 of Geophysical Data Analysis: Discrete
365          ! Inverse Theory by Menke (1984)
366          
367          ! 2.2.1  Form matrix G
368          DO i = 1, nmax
369             g(i)=lfn_gradient(x(i), param) ! curve derivative wrt to param
370          END DO
371          
372          ! 2.2.2  Form matrix GTD
373          gtd = 0.d0
374          DO i = 1, nmax
375             gtd = gtd + d(i) * g(i) * w(i)
376          END DO
377          
378          ! 2.2.3  Form matrix GTG
379          gtg = 0.d0
380          DO i = 1, nmax
381             gtg = gtg + g(i) * g(i) * w(i)
382          END DO
383          
384          ! 2.2.4  Find the inverse of matrix GTG
385          if (gtg /= 0.d0) then
386             gtginv = 1.d0 / gtg
387          else
388             write(*,*) 'Lfn_Curvefit: gtg = 0 ! Aborting the iteration process'
389             DO i = 1, nmax
390                write(*,*) g(i), w(i), x(i), param 
391             END DO
392             gtginv = tiny(1.d0)
393             !call utl_abort('Lfn_Curvefit')
394          end if
395          
396          ! 2.2.5  Find vector M (i.e. increment of PARAM)
397          m = gtginv * gtd
398          
399          ssqold = ssq          ! keep
400          
401          iter = iter + 1
402          
403          ! 2.3  If we are diverging
404       ELSE 
405          
406          m = m / 2.d0           ! Bisect the correction
407          nbis = nbis + 1
408          
409       END IF
410       
411       ! 2.4  Update the parameter
412       param = param + m
413       
414       ! 2.5  Print some info
415       FMT1 = "(A12,2X,A2,I6,2X,A2,I3,2X,A2,G10.4,2X,A5,G10.4,2X,A2,G10.4)"
416       
417       IF (ilist == 1) WRITE(*,FMT1) RoutineName,'i=',iter,'n=',nbis,'m=',m,  &
418            'rmsd=',rootmsq,'s=',param
419       
420    END DO
421    
422    PRINT*
423    PRINT*," Estimated lengthscale = ",param
424    
425    !-------------------------------------------------------------------------------
426    ! 3. Compute the final mean squared deviation
427    !-------------------------------------------------------------------------------
428    ssq = 0.d0
429    w_t = 0.d0
430    DO i = 1, nmax
431       d(i) = y(i) - lfn_response(x(i), param)
432       w_t = w_t + w(i)
433       ssq = ssq + w(i) * d(i)**2
434    END DO
435    ssq = ssq / w_t
436    
437  end SUBROUTINE Lfn_Curvefit
