  98  !--------------------------------------------------------------------------
  99  subroutine bmat1D_bsetup(vco_in, hco_in, obsSpaceData)
 100    !
 101    !:Purpose: To initialize the 1Dvar analysis Background term.
 102    !
 103    implicit none
 104
 105    ! arguments:
 106    type(struct_vco), pointer, intent(in) :: vco_in
 107    type(struct_hco), pointer, intent(in) :: hco_in
 108    type (struct_obs),         intent(in) :: obsSpaceData
 109
 110    ! locals:
 111    integer :: cvdim
 112    integer :: masterBmatIndex, bmatIndex
 113    logical :: active
 114    integer :: nulnam, ierr
 115    integer, external ::  fnom, fclos
 116    integer :: varIndex
 117    call utl_tmg_start(50, '--Bmatrix')
 118
 119    ! default values for namelist variables
 120    scaleFactorHI(:) = 0.d0
 121    scaleFactorHIHumidity(:) = 1.d0
 122    scaleFactorEns(:) = 0.d0
 123    scaleFactorEnsHumidity(:) = 1.d0
 124    nEns = -1
 125    vLocalize = -1.d0
 126    includeAnlVar(:)= ''
 127    numIncludeAnlVar = MPC_missingValue_INT
 128
 129    nulnam = 0
 130    ierr = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
 131    read(nulnam, nml=nambmat1D, iostat=ierr)
 132    if ( ierr /= 0 ) call utl_abort( 'bmat1D_bsetup: Error reading namelist' )
 133    if ( mmpi_myid == 0 ) write( *, nml = nambmat1D )
 134    ierr = fclos( nulnam )
 135    if (numIncludeAnlVar /= MPC_missingValue_INT) then
 136      call utl_abort('bmat1D_bsetup: check NAMBMAT1D namelist section: numIncludeAnlVar should be removed')
 137    end if
 138    numIncludeAnlVar = 0
 139    do varIndex = 1, vnl_numvarmax
 140      if (trim(includeAnlVar(varIndex)) == '') exit
 141      numIncludeAnlVar = numIncludeAnlVar + 1
 142    end do
 143    bmat1D_numIncludeAnlVar = numIncludeAnlVar
 144    allocate( bmat1D_includeAnlVar(bmat1D_numIncludeAnlVar) )
 145    bmat1D_includeAnlVar(1:bmat1D_numIncludeAnlVar) = includeAnlVar(1:numIncludeAnlVar)
 146
 147    !
 148    !- 1.  Setup the B matrices
 149    !
 150    do masterBmatIndex = 1, numMasterBmat
 151
 152      select case( trim(masterBmatTypeList(masterBmatIndex)) )
 153      case ('HI')
 154        !- 1.1 Time-Mean Homogeneous and Isotropic...
 155        write(*,*) 'bmat1D_bsetup: Setting up the modular GLOBAL HI 1D covariances...'
 156        call utl_tmg_start(51, '----B_HI_Setup')
 157        call bmat1D_SetupBHi(vco_in, obsSpaceData, cvdim)
 158        call utl_tmg_stop(51)
 159        write(*,*) ' bmat1D_bsetup: cvdim= ', cvdim
 160      case ('ENS')
 161        !- 1.2 ensemble based
 162        write(*,*) 'bmat1D_bsetup: Setting up the ensemble based 1D matrix.'
 163        call utl_tmg_start(54, '----B_ENS_Setup')
 164        call bmat1D_SetupBEns(vco_in, hco_in, obsSpaceData, cvdim)
 165        call utl_tmg_stop(54)
 166        write(*,*) ' bmat1D_bsetup: cvdim= ', cvdim
 167      case default
 168        call utl_abort( 'bmat1D_bSetup: requested bmatrix type does not exist ' // trim(masterBmatTypeList(masterBmatIndex)) )
 169      end select
 170
 171      !- 1.2 Append the info to the B matrix info arrays and setup the proper control sub-vectors
 172      numBmat = numBmat + 1
 173      bmatLabelList (numBmat) = trim(masterbmatLabelList(masterBmatIndex))
 174      bmatTypeList  (numBmat) = masterBmatTypeList(masterBmatIndex)
 175      bmatIs3dList  (numBmat) = masterbmatIs3dList(masterBmatIndex)
 176      call cvm_setupSubVector(bmatLabelList(numBmat), bmatTypeList(numBmat), cvdim)
 177    end do
 178
 179    !
 180    !- 2. Print a summary and set the active B matrices array
 181    !
 182    write(*,*)
 183    write(*,*) 'bmat1D_bsetup SUMMARY, number of B matrices found = ', numBmat
 184    do bmatIndex = 1, numBmat
 185      write(*,*) '  B matrix #', bmatIndex
 186      active = cvm_subVectorExists(bmatLabelList(bmatIndex))
 187      if (active) then
 188        write(*,*) '   ACTIVE'
 189      else
 190        write(*,*) '   NOT USED'
 191      end if
 192      write(*,*) '     -> label       = ', bmatLabelList (bmatIndex)
 193      write(*,*) '     -> type        = ', bmatTypeList  (bmatIndex)
 194      if (active) then
 195        write(*,*) '     -> is 3D       = ', bmatIs3dList  (bmatIndex)
 196      end if
 197      bmatActive(bmatIndex) = active
 198    end do
 199
 200    call utl_tmg_stop(50)
 201
 202  end subroutine bmat1D_bsetup

 207  subroutine bmat1D_setupBHi(vco_in, obsSpaceData, cvDim_out)
 208    !
 209    !:Purpose: to setup bmat1D module
 210    !
 211    implicit none
 212
 213    ! arguments:
 214    type(struct_vco), pointer, intent(in)  :: vco_in
 215    type (struct_obs)        , intent(in)  :: obsSpaceData
 216    integer                  , intent(out) :: cvDim_out
 217
 218    ! locals:
 219    integer :: levelIndex, ierr
 220    integer, external ::  fnom, fclos
 221    integer :: Vcode_anl
 222    logical :: fileExists
 223    integer :: nulbgst=0
 224    type(struct_vco), pointer :: vco_file => null()
 225    type(struct_vco), target  :: vco_1Dvar
 226    type(struct_vco), pointer :: vco_anl
 227    character(len=18) :: oneDBmatLand = './Bmatrix_land.bin'
 228    character(len=17) :: oneDBmatSea  = './Bmatrix_sea.bin'
 229    character(len=4), allocatable :: includeAnlVarHi(:)
 230    integer :: extractDate, locationIndex, varIndex, numIncludeAnlVarHi
 231    integer :: shiftLevel, varLevIndex, varLevIndex1, varLevIndex2
 232    logical, save :: firstCall=.true.
 233    real(8), allocatable :: bMatrix(:,:), multFactor(:)
 234
 235    if (.not. (gsv_varExist(varName='TT') .and.  &
 236               gsv_varExist(varName='UU') .and.  &
 237               gsv_varExist(varName='VV') .and.  &
 238               (gsv_varExist(varName='HU').or.gsv_varExist(varName='LQ')) .and.  &
 239               gsv_varExist(varName='P0')) ) then
 240      call utl_abort('bmat1D_setupBHi: Some or all weather fields are missing. If it is desired to deactivate&
 241           & the weather assimilation, then all entries of the array SCALEFACTORHI in the namelist NAMVAR1D&
 242           & should be set to zero.')
 243    end if
 244
 245    if (.not. gsv_varExist(varName='TG')) then
 246      write(*,*) 'bmat1D_setupBHi: WARNING: The TG field is missing. This must be present when assimilating'
 247      write(*,*) 'radiance observations.'
 248    end if
 249
 250    if (firstCall) then
 251      call var1D_setup(obsSpaceData)
 252      firstCall = .false.
 253    end if
 254
 255    if(mmpi_myid == 0) write(*,*) 'bmat1D_setupBHi: Starting'
 256    if(mmpi_myid == 0) write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
 257
 258    do levelIndex = 1, vco_maxNumLevels
 259      if( scaleFactorHI(levelIndex) > 0.0d0 ) then 
 260        scaleFactorHI(levelIndex) = sqrt( scaleFactorHI(levelIndex))
 261      else
 262        scaleFactorHI(levelIndex) = 0.0d0
 263      end if
 264    end do
 265   
 266    do levelIndex = 1, vco_maxNumLevels
 267      if(scaleFactorHIHumidity(levelIndex) > 0.0d0) then 
 268        scaleFactorHIHumidity(levelIndex) = sqrt(scaleFactorHIHumidity(levelIndex))
 269      else
 270        scaleFactorHIHumidity(levelIndex) = 0.0d0
 271      end if
 272    end do
 273
 274    if ( sum(scaleFactorHI(1:vco_maxNumLevels)) == 0.0d0 ) then
 275      if ( mmpi_myid == 0 ) write(*,*) 'bmat1D_setupBHi: scaleFactorHI=0, skipping rest of setup'
 276      cvDim_out = 0
 277      return
 278    end if
 279
 280    if (mmpi_myid == 0) write(*,*) 'bmat1D_setupBHi: Read 1DVar background statistics'
 281    inquire(file=trim(oneDBmatLand), exist=fileExists)
 282    if ( fileExists ) then
 283      ierr = fnom(nulbgst, trim(oneDBmatLand), 'FTN+SEQ+UNF+OLD+R/O', 0)
 284    else
 285      call utl_abort('bmat1D_setupBHi: No 1DVar BACKGROUND STAT FILE ' // trim(oneDBmatLand))
 286    end if
 287
 288    read(nulbgst) extractDate, vco_1Dvar%nLev_T, vco_1Dvar%nLev_M, vco_1Dvar%Vcode, &
 289         vco_1Dvar%ip1_sfc, vco_1Dvar%ip1_T_2m, vco_1Dvar%ip1_M_10m, numIncludeAnlVarHi, nkgdim, nLonLatPosLand
 290    allocate( vco_1Dvar%ip1_T(vco_1Dvar%nLev_T), vco_1Dvar%ip1_M(vco_1Dvar%nLev_M) )
 291    if (numIncludeAnlVarHi /= bmat1D_numIncludeAnlVar) then
 292      write(*,*) 'numIncludeAnlVarHi, bmat1D_numIncludeAnlVar= ', numIncludeAnlVarHi, bmat1D_numIncludeAnlVar
 293      call utl_abort('bmat1D_setupBHi: incompatible number of 1DVar analyzed variables in ' // trim(oneDBmatLand))
 294    end if
 295
 296    allocate (multFactor(nkgdim))
 297    if(vco_1Dvar%Vcode == 5002) then
 298      shiftLevel = 1
 299    else
 300      shiftLevel = 0
 301    end if
 302    varLevIndex = 0
 303    do varIndex = 1, bmat1D_numIncludeAnlVar
 304      select case(bmat1D_includeAnlVar(varIndex))
 305        case('TT')
 306          do levelIndex = 1, vco_1Dvar%nLev_T
 307            varLevIndex = varLevIndex + 1
 308            multFactor(varLevIndex) = scaleFactorHI(levelIndex)
 309          end do
 310        case('HU')
 311          do levelIndex = 1, vco_1Dvar%nLev_T
 312            varLevIndex = varLevIndex + 1
 313            multFactor(varLevIndex)= scaleFactorHIHumidity(levelIndex) * scaleFactorHI(levelIndex)
 314          end do
 315        case('UU','VV')
 316          do levelIndex = 1, vco_1Dvar%nLev_M
 317            varLevIndex = varLevIndex + 1
 318            multFactor(varLevIndex) = scaleFactorHI(levelIndex+shiftLevel)
 319          end do
 320        case('P0','TG')
 321          varLevIndex = varLevIndex + 1
 322          multFactor(varLevIndex) = scaleFactorHI(max(vco_1Dvar%nLev_T,vco_1Dvar%nLev_M))
 323        case default
 324          call utl_abort('bmat1D_setupBHi: unsupported variable ' // bmat1D_includeAnlVar(varIndex))
 325        end select
 326    end do
 327
 328    allocate( includeAnlVarHi(bmat1D_numIncludeAnlVar) )
 329    allocate( bMatrix(nkgdim,nkgdim) )
 330    allocate( latLand(nLonLatPosLand), lonLand(nLonLatPosLand))       
 331    allocate( bSqrtLand(nLonLatPosLand, nkgdim, nkgdim) )
 332    read(nulbgst) vco_1Dvar%ip1_T(:), vco_1Dvar%ip1_M(:), includeAnlVarHi(:)
 333    if (any(includeAnlVarHi /= bmat1D_includeAnlVar)) then
 334      do varIndex = 1, bmat1D_numIncludeAnlVar
 335        write(*,*) varIndex, includeAnlVarHi(varIndex), bmat1D_includeAnlVar(varIndex)
 336      end do
 337      call utl_abort('bmat1D_setupBHi: incompatible 1DVar analyzed variable list in ' // trim(oneDBmatLand))
 338    end if
 339    deallocate(includeAnlVarHi)
 340
 341    do locationIndex = 1, nLonLatPosLand
 342      read(nulbgst) latLand(locationIndex), lonLand(locationIndex), bMatrix(:,:)
 343      !application of the scaling factor
 344      do varLevIndex2 = 1, nkgdim
 345        do varLevIndex1 = 1, nkgdim
 346          bSqrtLand(locationIndex, varLevIndex1, varLevIndex2) = &
 347               bMatrix(varLevIndex1, varLevIndex2) *  multFactor(varLevIndex1) * multFactor(varLevIndex2)
 348        end do
 349      end do
 350      call utl_matsqrt(bSqrtLand(locationIndex, :, :), nkgdim, 1.d0, printInformation_opt=.false. )
 351    end do
 352    ierr = fclos(nulbgst)
 353
 354    inquire(file=trim(oneDBmatSea), exist=fileExists)
 355    if ( fileExists ) then
 356      ierr = fnom(nulbgst, trim(oneDBmatSea), 'FTN+SEQ+UNF+OLD+R/O', 0)
 357    else
 358      call utl_abort('bmat1D_setupBHi: No 1DVar BACKGROUND STAT FILE ' // trim(oneDBmatSea))
 359    end if
 360    read(nulbgst) extractDate, vco_1Dvar%nLev_T, vco_1Dvar%nLev_M, vco_1Dvar%Vcode, &
 361         vco_1Dvar%ip1_sfc, vco_1Dvar%ip1_T_2m, vco_1Dvar%ip1_M_10m, numIncludeAnlVarHi, nkgdim, nLonLatPosSea
 362    if (numIncludeAnlVarHi /= bmat1D_numIncludeAnlVar) then
 363      write(*,*) 'numIncludeAnlVarHi, bmat1D_numIncludeAnlVar= ', numIncludeAnlVarHi, bmat1D_numIncludeAnlVar
 364      call utl_abort('bmat1D_setupBHi: incompatible number of 1DVar analyzed variables in ' // trim(oneDBmatSea))
 365    end if
 366    allocate( bSqrtSea(nLonLatPosSea, nkgdim, nkgdim) )
 367    allocate( latSea(nLonLatPosSea), lonSea(nLonLatPosSea))
 368    allocate( includeAnlVarHi(bmat1D_numIncludeAnlVar) )
 369    read(nulbgst) vco_1Dvar%ip1_T(:), vco_1Dvar%ip1_M(:), includeAnlVarHi(:)
 370    if (any(includeAnlVarHi /= bmat1D_includeAnlVar)) then
 371      do varIndex = 1, bmat1D_numIncludeAnlVar
 372        write(*,*) varIndex, includeAnlVarHi(varIndex), bmat1D_includeAnlVar(varIndex)
 373      end do
 374      call utl_abort('bmat1D_setupBHi: incompatible 1DVar analyzed variable list in ' // trim(oneDBmatSea))
 375    end if
 376    deallocate(includeAnlVarHi)
 377    do locationIndex = 1, nLonLatPosSea
 378      read(nulbgst) latSea(locationIndex), lonSea(locationIndex), bMatrix(:,:)
 379      !application of the scaling factor
 380      do varLevIndex2 = 1, nkgdim
 381        do varLevIndex1 = 1, nkgdim
 382          bSqrtSea(locationIndex, varLevIndex1, varLevIndex2) = &
 383               bMatrix(varLevIndex1, varLevIndex2) *  multFactor(varLevIndex1) * multFactor(varLevIndex2)
 384        end do
 385      end do
 386      call utl_matsqrt(bSqrtSea(locationIndex,:,:), nkgdim, 1.d0, printInformation_opt=.false. )
 387    end do
 388    ierr = fclos(nulbgst)
 389  
 390    deallocate(bMatrix, multFactor)
 391
 392    vco_1Dvar%initialized = .true.
 393    vco_1Dvar%vGridPresent = .false.
 394    vco_file => vco_1Dvar
 395    vco_anl => vco_in
 396    if (.not. vco_equal(vco_anl,vco_file)) then
 397      call utl_abort('bmat1D_setupBHi: vco from analysisgrid and cov file do not match')
 398    end if
 399    if (mmpi_myid == 0) write(*,*) 'bmat1D_setupBHi: nLev_M, nLev_T=', vco_1Dvar%nLev_M, vco_1Dvar%nLev_T
 400    Vcode_anl = vco_anl%vCode
 401    if(Vcode_anl /= 5002 .and. Vcode_anl /= 5005) then
 402      write(*,*) 'Vcode_anl = ',Vcode_anl
 403      call utl_abort('bmat1D_setupBHi: unknown vertical coordinate type!')
 404    end if
 405    
 406    cvDim_out = nkgdim * var1D_validHeaderCount
 407    cvDim_mpilocal = cvDim_out
 408    initialized = .true.
 409
 410    if(mmpi_myid == 0) write(*,*) 'bmat1D_setupBHi: Exiting'
 411    if(mmpi_myid == 0) write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
 412
 413  end subroutine bmat1D_setupBHi

 418  subroutine bmat1D_setupBEns(vco_in, hco_in, obsSpaceData, cvDim_out)
 419    !
 420    !:Purpose: to setup bmat1D module
 421    !
 422    implicit none
 423
 424    ! arguments
 425    type(struct_vco), pointer, intent(in)  :: vco_in
 426    type(struct_hco), pointer, intent(in)  :: hco_in
 427    type (struct_obs)        , intent(in)  :: obsSpaceData
 428    integer                  , intent(out) :: cvDim_out
 429
 430    ! locals:
 431    character(len=256) :: ensPathName = 'ensemble'
 432    character(len=256) :: ensFileName
 433    type(struct_vco), pointer :: vco_file => null()
 434    type(struct_vco), pointer :: vco_ens => null()
 435    type(struct_hco), pointer :: hco_ens => null()
 436    type(struct_gsv)          :: stateVector, stateVectorMean
 437    integer, allocatable :: dateStampList(:)
 438    character(len=12) :: hInterpolationDegree ! select degree of horizontal interpolation (if needed)
 439    integer :: memberIndex, columnIndex, headerIndex, varIndex, levIndex
 440    integer :: levIndex1
 441    integer :: varLevIndex, varLevIndex1, varLevIndex2 
 442    integer :: numStep, levIndexColumn
 443    real(8), allocatable :: scaleFactor_M(:), scaleFactor_T(:)
 444    real(8) :: scaleFactor_SF, ZR
 445    logical :: useAnlLevelsOnly, EnsTopMatchesAnlTop
 446    real(8), pointer :: pressureProfileFile_M(:), pressureProfileInc_M(:)
 447    real(8) :: pSurfRef
 448    integer :: nj, latPerPE, latPerPEmax, myLatBeg, myLatEnd
 449    integer :: ni, lonPerPE, lonPerPEmax, myLonBeg, myLonEnd
 450    integer :: nLevEns_M, nLevEns_T
 451    integer :: nLevInc_M, nLevInc_T
 452    integer :: topLevIndex_M, topLevIndex_T
 453    integer, external :: newdate
 454    character(len=4), pointer :: varNames(:)
 455    real(8) :: logP1, logP2
 456    real(8), pointer :: currentProfile(:), meanProfile(:)
 457    real(8), allocatable :: lineVector(:,:), meanPressureProfile(:), multFactor(:)
 458    integer, allocatable :: levIndexFromVarLevIndex(:)
 459    character(len=4), allocatable :: varNameFromVarLevIndex(:)
 460    character(len=2) :: varLevel
 461
 462    if(mmpi_myid == 0) write(*,*) 'bmat1D_setupBEns: Starting'
 463    if(mmpi_myid == 0) write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
 464
 465    if ( nEns <= 0 ) then
 466      if ( mmpi_myid == 0 ) write(*,*) 'bmat1D_setupBEns: no Ensemble members, skipping rest of setup'
 467      cvdim_out = 0
 468      return
 469    end if
 470    
 471    !- 1.1 Number of time step bins
 472    numStep = tim_nstepobsinc
 473    if (numStep /= 1 .and. numStep /= 3.and. numStep /= 5 .and. numStep /= 7) then
 474      call utl_abort('bmat1D_setupBEns: Invalid value for numStep (choose 1 or 3 or 5 or 7)!')
 475    end if
 476    allocate(dateStampList(numStep))
 477    call tim_getstamplist(dateStampList,numStep,tim_getDatestamp())
 478    
 479    hco_ens => hco_in
 480
 481    !- 1.2 Horizontal grid
 482    ni = hco_ens%ni
 483    nj = hco_ens%nj
 484    if (hco_ens%global) then
 485      if (mmpi_myid == 0) write(*,*)
 486      if (mmpi_myid == 0) write(*,*) 'bmat1D_setupBEns: GLOBAL mode activated'
 487    else
 488      if (mmpi_myid == 0) write(*,*)
 489      if (mmpi_myid == 0) write(*,*) 'bmat1D_setupBEns: LAM mode activated'
 490    end if
 491
 492    !- 1.3 Vertical levels
 493    if ( mmpi_myid == 0 ) then
 494      call fln_ensfileName(ensFileName, ensPathName, memberIndex_opt=1)
 495      write(*,*) 'before vco_SetupFromFile'
 496      call vco_SetupFromFile(vco_file, ensFileName)
 497      call gsv_allocate(stateVector, numStep, hco_in, vco_in,  &
 498           hInterpolateDegree_opt='LINEAR', &
 499           dataKind_opt=4, &
 500           dateStamp_opt=tim_getDateStamp(), beSilent_opt=.false.)
 501      call gio_readFromFile(stateVector, ensFileName, '', '')
 502      call gsv_varNamesList(varNames, stateVector)
 503      write(*,*) 'bmat1D_setupBEns: variable names : ', varNames
 504    end if
 505    call vco_mpiBcast(vco_file)
 506
 507    !- Do we need to read all the vertical levels from the ensemble?
 508    useAnlLevelsOnly = vco_subsetOrNot(vco_in, vco_file)
 509    if ( useAnlLevelsOnly ) then
 510      write(*,*)
 511      write(*,*) 'bmat1D_setupBEns: only the analysis levels will be read in the ensemble '
 512      vco_ens  => vco_in ! the ensemble target grid is the analysis grid
 513      call vco_deallocate(vco_file)
 514      vco_file => vco_in ! only the analysis levels will be read in the ensemble
 515      EnsTopMatchesAnlTop = .true.
 516    else
 517      write(*,*)
 518      write(*,*) 'bmat1D_setupBEns: all the vertical levels will be read in the ensemble '
 519      if ( vco_in%nLev_M > 0 .and. vco_in%vgridPresent ) then
 520        pSurfRef = 101000.D0
 521        call czp_fetch1DLevels(vco_in, pSurfRef, profM_opt=pressureProfileInc_M)
 522        call czp_fetch1DLevels(vco_in, pSurfRef, profM_opt=pressureProfileFile_M)
 523      
 524        EnsTopMatchesAnlTop = abs( log(pressureProfileFile_M(1)) - log(pressureProfileInc_M(1)) ) < 0.1d0
 525        write(*,*) 'bmat1D_setupBEns: EnsTopMatchesAnlTop, presEns, presInc = ', &
 526             EnsTopMatchesAnlTop, pressureProfileFile_M(1), pressureProfileInc_M(1)
 527        deallocate(pressureProfileFile_M)
 528        deallocate(pressureProfileInc_M)
 529      else
 530        ! not sure what this mean when no MM levels
 531        write(*,*) 'bmat1D_setupBEns: nLev_M       = ', vco_in%nLev_M
 532        write(*,*) 'bmat1D_setupBEns: vgridPresent = ', vco_in%vgridPresent
 533        EnsTopMatchesAnlTop = .true.
 534      end if
 535
 536      if ( EnsTopMatchesAnlTop ) then
 537        if ( mmpi_myid == 0 ) write(*,*) 'bmat1D_setupBEns: top level of ensemble member and analysis grid match'
 538        vco_ens => vco_in  ! IMPORTANT: top levels DO match, therefore safe
 539                           ! to force members to be on analysis vertical levels
 540      else
 541        if ( mmpi_myid == 0 ) write(*,*) 'bmat1D_setupBEns: top level of ensemble member and analysis grid are different, therefore'
 542        if ( mmpi_myid == 0 ) write(*,*) '                      assume member is already be on correct levels - NO CHECKING IS DONE'
 543        vco_ens => vco_file ! IMPORTANT: top levels do not match, therefore must
 544                            ! assume file is already on correct vertical levels
 545      end if
 546    end if
 547
 548    if (vco_in%Vcode /= vco_ens%Vcode) then
 549      write(*,*) 'bmat1D_setupBEns: vco_in%Vcode = ', vco_in%Vcode, ', vco_ens%Vcode = ', vco_ens%Vcode
 550      call utl_abort('bmat1D_setupBEns: vertical levels of ensemble not compatible with analysis grid')
 551    end if
 552    nLevEns_M = vco_ens%nLev_M
 553    nLevEns_T = vco_ens%nLev_T
 554    nLevInc_M = vco_in%nLev_M
 555    nLevInc_T = vco_in%nLev_T
 556    topLevIndex_M = nLevInc_M-nLevEns_M+1
 557    topLevIndex_T = nLevInc_T-nLevEns_T+1
 558
 559    if (vco_in%Vcode == 5002) then
 560      if ( (nLevEns_T /= (nLevEns_M+1)) .and. (nLevEns_T /= 1 .or. nLevEns_M /= 1) ) then
 561        write(*,*) 'bmat1D_setubBEns: nLevEns_T, nLevEns_M = ',nLevEns_T,nLevEns_M
 562        call utl_abort('bmat1D_setubBEns: Vcode=5002, nLevEns_T must equal nLevEns_M+1!')
 563      end if
 564    else if (vco_in%Vcode == 5005) then
 565      if ( nLevEns_T /= nLevEns_M .and. &
 566           nLevEns_T /= 0 .and. &
 567           nLevEns_M /= 0 ) then
 568        write(*,*) 'bmat1D_setubBEns: nLevEns_T, nLevEns_M = ',nLevEns_T,nLevEns_M
 569        call utl_abort('bmat1D_setubBEns: Vcode=5005, nLevEns_T must equal nLevEns_M!')
 570      end if
 571    else if (vco_in%Vcode == 0) then
 572      if ( nLevEns_T /= 0 .and. nLevEns_M /= 0 ) then
 573        write(*,*) 'bmat1D_setubBEns: nLevEns_T, nLevEns_M = ',nLevEns_T, nLevEns_M
 574        call utl_abort('bmat1D_setubBEns: surface-only case (Vcode=0), nLevEns_T and nLevEns_M must equal 0!')
 575      end if
 576    else
 577      write(*,*) 'bmat1D_setubBEns: vco_in%Vcode = ',vco_in%Vcode
 578      call utl_abort('bmat1D_setubBEns: unknown vertical coordinate type!')
 579    end if
 580
 581    if (nLevEns_M > nLevInc_M) then
 582      call utl_abort('bmat1D_setubBEns: ensemble has more levels than increment - not allowed!')
 583    end if
 584
 585    if (nLevEns_M < nLevInc_M) then
 586      if (mmpi_myid == 0) write(*,*) 'bmat1D_setubBEns: ensemble has less levels than increment'
 587      if (mmpi_myid == 0) write(*,*) '                      some levels near top will have zero increment'
 588    end if
 589
 590    !- 1.4 Bmatrix Weight
 591    if (vco_in%Vcode == 5002 .or. vco_in%Vcode == 5005) then
 592      allocate(scaleFactor_M(nLevEns_M))
 593      allocate(scaleFactor_T(nLevEns_T))
 594      do levIndex = 1, nLevEns_T
 595        if (scaleFactorEns(levIndex) > 0.0d0) then 
 596          scaleFactorEns(levIndex) = sqrt(scaleFactorEns(levIndex))
 597        else
 598          scaleFactorEns(levIndex) = 0.0d0
 599        end if
 600      end do
 601      scaleFactor_T(1:nLevEns_T) = scaleFactorEns(1:nLevEns_T)
 602      if (vco_in%Vcode == 5002) then
 603        scaleFactor_M(1:nLevEns_M) = scaleFactorEns(2:(nLevEns_M+1))
 604      else
 605        scaleFactor_M(1:nLevEns_M) = scaleFactorEns(1:nLevEns_M)
 606      end if
 607
 608      do levIndex = 1, nLevEns_T
 609        if (scaleFactorEnsHumidity(levIndex) > 0.0d0) then 
 610          scaleFactorEnsHumidity(levIndex) = sqrt(scaleFactorEnsHumidity(levIndex))
 611        else
 612          scaleFactorEnsHumidity(levIndex) = 0.0d0
 613        end if
 614      end do
 615      
 616      scaleFactor_SF = scaleFactor_T(nLevEns_T)
 617
 618    else ! vco_in%Vcode == 0
 619      if (scaleFactorEns(1) > 0.0d0) then 
 620        scaleFactor_SF = sqrt(scaleFactorEns(1))
 621      else
 622        call utl_abort('bmat1D_setubBEns: with vCode == 0, the scale factor should never be equal to 0')
 623      end if
 624    end if
 625
 626    !- 1.5 Domain Partionning
 627    call mmpi_setup_latbands(nj, latPerPE, latPerPEmax, myLatBeg, myLatEnd)
 628    call mmpi_setup_lonbands(ni, lonPerPE, lonPerPEmax, myLonBeg, myLonEnd)
 629
 630    !- 1.6 Localization
 631    if ( vLocalize <= 0.0d0 .and. (nLevInc_M > 1 .or. nLevInc_T > 1) ) then
 632      call utl_abort('bmat1D_setubBEns: Invalid VERTICAL localization length scale')
 633    end if     
 634      
 635    !- 1.7 Setup the localization
 636    if ( vco_in%Vcode == 5002 .or. vco_in%Vcode == 5005 ) then
 637      pSurfRef = 101000.D0
 638      call czp_fetch1DLevels(vco_in, pSurfRef, profM_opt=pressureProfileInc_M)
 639      deallocate(pressureProfileInc_M)
 640    end if
 641  
 642    call ens_allocate(ensPerts,  &
 643         nEns, numStep, &
 644         hco_ens,  &
 645         vco_ens, dateStampList, &
 646         hco_core_opt = hco_in, &  ! to generalize later hco_in => hco_core
 647         varNames_opt = bmat1D_includeAnlVar(1:bmat1D_numIncludeAnlVar), &
 648         hInterpolateDegree_opt = hInterpolationDegree)
 649    write(*,*) 'Read ensemble members'
 650    call ens_readEnsemble(ensPerts, ensPathName, biPeriodic=.false.,       &
 651                          vco_file_opt = vco_ens,                          &
 652                          varNames_opt = bmat1D_includeAnlVar(1:bmat1D_numIncludeAnlVar))
 653    
 654    allocate( ensColumns(nEns))
 655    call gsv_allocate(stateVector, numstep, hco_ens, vco_ens, &
 656                     dateStamp_opt=tim_getDateStamp(),  &
 657                     mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
 658                     dataKind_opt=4, allocHeightSfc_opt=.true.)
 659    
 660    call gsv_allocate(stateVectorMean, numstep, hco_ens, vco_ens, &
 661                     dateStamp_opt=tim_getDateStamp(),  &
 662                     mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
 663                     dataKind_opt=4, allocHeightSfc_opt=.true.)
 664    do memberIndex = 1, nEns
 665      write(*,*) 'Copy member ', memberIndex
 666      call ens_copyMember(ensPerts, stateVector, memberIndex)
 667      write(*,*) 'interpolate member ', memberIndex
 668      call col_setVco(ensColumns(memberIndex), vco_ens)
 669      call col_allocate(ensColumns(memberIndex), obs_numheader(obsSpaceData),  &
 670                        mpiLocal_opt=.true., setToZero_opt=.true.)
 671      call s2c_nl( stateVector, obsSpaceData, ensColumns(memberIndex), hco_in, &
 672                   timeInterpType='NEAREST' )
 673      call gsv_add(statevector, statevectorMean, scaleFactor_opt=(1.d0/nEns))
 674    end do
 675    call col_setVco(meanColumn, vco_ens)
 676    call col_allocate(meanColumn, obs_numheader(obsSpaceData),  &
 677            mpiLocal_opt=.true., setToZero_opt=.true.)
 678    call s2c_nl( stateVectorMean, obsSpaceData, meanColumn, hco_in, &
 679            timeInterpType='NEAREST' )
 680
 681    call gsv_deallocate(stateVector)
 682    call gsv_deallocate(stateVectorMean)
 683
 684    nkgdim = 0
 685    do varIndex = 1, bmat1D_numIncludeAnlVar
 686      currentProfile => col_getColumn(meanColumn, var1D_validHeaderIndex(1), varName_opt=bmat1D_includeAnlVar(varIndex))
 687      nkgdim = nkgdim + size(currentProfile)
 688    end do
 689    write(*,*) 'bmat1D_setupBEns: nkgdim', nkgdim
 690    cvDim_out = nkgdim * var1D_validHeaderCount
 691
 692    currentProfile => col_getColumn(meanColumn, var1D_validHeaderIndex(1) )
 693    allocate (levIndexFromVarLevIndex(nkgdim))
 694    allocate (varNameFromVarLevIndex(nkgdim))
 695    allocate (multFactor(nkgdim))
 696    varLevIndex = 0
 697    do varIndex = 1, bmat1D_numIncludeAnlVar
 698      levIndex1 = 0
 699      do levIndex = 1, size(currentProfile)
 700        if ( trim( col_getVarNameFromK(meanColumn,levIndex) ) == trim( bmat1D_includeAnlVar(varIndex) ) ) then
 701          varLevIndex = varLevIndex + 1
 702          levIndex1 = levIndex1 + 1
 703          levIndexFromVarLevIndex(varLevIndex) = levIndex
 704          varNameFromVarLevIndex(varLevIndex) = trim( bmat1D_includeAnlVar(varIndex) )
 705          varLevel = vnl_varLevelFromVarname(varNameFromVarLevIndex(varLevIndex))
 706          if ( varLevel == 'MM' ) then      ! Momentum
 707            multFactor(varLevIndex) = scaleFactor_M(levIndex1)
 708          else if ( varLevel == 'TH' ) then ! Thermo
 709            multFactor(varLevIndex) = scaleFactor_T(levIndex1)
 710          else                              ! SF
 711            multFactor(varLevIndex) = scaleFactor_SF
 712          end if
 713          if (varNameFromVarLevIndex(varLevIndex) == 'HU') then
 714            multFactor(varLevIndex) = multFactor(varLevIndex) * scaleFactorEnsHumidity(levIndex1)
 715          end if
 716          if (mmpi_myid == 0) write(*,*) 'bmat1D_setupBEns:  bmat1D_includeAnlVar ', bmat1D_includeAnlVar(varIndex), varLevIndex, levIndex
 717        end if
 718      end do
 719    end do
 720    
 721    call ens_deallocate( ensPerts )
 722    allocate(bSqrtEns(var1D_validHeaderCount,nkgdim,nkgdim))
 723    bSqrtEns(:,:,:) = 0.d0
 724    allocate(lineVector(1,nkgdim))
 725
 726    !$OMP PARALLEL DO PRIVATE (columnIndex,headerIndex,memberIndex,meanProfile,currentProfile,lineVector)
 727    do columnIndex = 1, var1D_validHeaderCount
 728      headerIndex = var1D_validHeaderIndex(columnIndex)
 729      meanProfile => col_getColumn(meanColumn, headerIndex)
 730      do memberIndex = 1, nEns
 731        currentProfile => col_getColumn(ensColumns(memberIndex), headerIndex)
 732        lineVector(1,:) = currentProfile(levIndexFromVarLevIndex(:)) - meanProfile(levIndexFromVarLevIndex(:))
 733        lineVector(1,:) = lineVector(1,:) * multFactor(:)
 734        bSqrtEns(headerIndex,:,:) = bSqrtEns(headerIndex,:,:) + &
 735            matmul(transpose(lineVector),lineVector)
 736      end do
 737    end do
 738    !$OMP END PARALLEL DO
 739
 740    deallocate(lineVector)
 741    deallocate(multFactor)
 742    allocate(meanPressureProfile(nkgdim))
 743    call lfn_Setup('FifthOrder')
 744
 745    !$OMP PARALLEL DO PRIVATE (columnIndex,headerIndex,meanPressureProfile,levIndex1,varLevIndex1,varLevIndex2,varLevel,levIndexColumn,logP1,logP2,zr)
 746    do columnIndex = 1, var1D_validHeaderCount
 747      headerIndex = var1D_validHeaderIndex(columnIndex)
 748      bSqrtEns(headerIndex,:,:) = bSqrtEns(headerIndex,:,:) / (nEns - 1)
 749      if (vLocalize > 0.0d0) then
 750        do varLevIndex1 = 1, nkgdim
 751          levIndex1 = levIndexFromVarLevIndex(varLevIndex1)
 752          varLevel = vnl_varLevelFromVarname(varNameFromVarLevIndex(varLevIndex1))
 753          if (varLevel=='SF') then
 754            meanPressureProfile(varLevIndex1) = col_getElem(meanColumn, 1, headerIndex, 'P0')
 755          else
 756            levIndexColumn = col_getLevIndexFromVarLevIndex(meanColumn, levIndex1)
 757            meanPressureProfile(varLevIndex1) = col_getPressure(meanColumn, levIndexColumn, headerIndex, varLevel)
 758          end if
 759        end do
 760        do varLevIndex1 = 1, nkgdim
 761          logp1 = log(meanPressureProfile(varLevIndex1))
 762          do varLevIndex2 = 1, nkgdim
 763            !-  do Schurr product with 5'th order function
 764            logP2 = log(meanPressureProfile(varLevIndex2))
 765            zr = abs(logP2  -  logP1)
 766            bSqrtEns(headerIndex, varLevIndex2, varLevIndex1) = &
 767                 bSqrtEns(headerIndex, varLevIndex2, varLevIndex1) * lfn_response(zr, vLocalize)
 768          end do
 769        end do
 770      end if
 771      call utl_matsqrt(bSqrtEns(headerIndex, :, :), nkgdim, 1.d0, printInformation_opt=.false. )
 772    end do
 773    !$OMP END PARALLEL DO
 774
 775    deallocate(levIndexFromVarLevIndex) 
 776    deallocate(varNameFromVarLevIndex)
 777    deallocate(meanPressureProfile)
 778    call col_deallocate(meanColumn)
 779    do memberIndex = 1, nEns
 780      call col_deallocate(ensColumns(memberIndex))
 781    end do
 782    cvDim_mpilocal = cvDim_out
 783    initialized = .true.
 784    
 785    if(mmpi_myid == 0) write(*,*) 'bmat1D_setupBEns: Exiting'
 786    if(mmpi_myid == 0) write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
 787
 788  end subroutine bmat1D_setupBEns

 792  !--------------------------------------------------------------------------
 793  subroutine bmat1D_bSqrtHi(controlVector_in, column, obsSpaceData)
 794    !
 795    !:Purpose: HI component of B square root in 1DVar mode
 796    !
 797    implicit none
 798
 799    ! arguments:
 800    real(8),                 intent(in)    :: controlVector_in(cvDim_mpilocal)
 801    type(struct_columnData), intent(inout) :: column
 802    type(struct_obs),        intent(in)    :: obsSpaceData
 803
 804    ! locals:
 805    integer :: headerIndex, latitudeBandIndex(1), varIndex, columnIndex
 806    real(8), pointer :: currentColumn(:)
 807    real(8), allocatable ::  oneDProfile(:)
 808    real(8) :: latitude
 809    integer :: surfaceType, offset
 810
 811    if (mmpi_myid == 0) write(*,*) 'bmat1D_bsqrtHi: starting'
 812    if (mmpi_myid == 0) write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
 813
 814    if (.not. initialized) then
 815      if (mmpi_myid == 0) write(*,*) 'bmat1D_bsqrtHi: 1Dvar B matrix not initialized'
 816      return
 817    end if
 818    allocate(oneDProfile(nkgdim))
 819
 820    !$OMP PARALLEL DO PRIVATE (columnIndex,headerIndex,oneDProfile,offset,varIndex,currentColumn,latitude,surfaceType,latitudeBandIndex)
 821    do columnIndex = 1, var1D_validHeaderCount 
 822      headerIndex = var1D_validHeaderIndex(columnIndex)
 823      latitude = obs_headElem_r(obsSpaceData, OBS_LAT, headerIndex) !radian 
 824      surfaceType = tvs_ChangedStypValue(obsSpaceData, headerIndex)
 825      if (surfaceType == 1) then !Sea
 826        latitudeBandIndex = minloc( abs( latitude - latSea(:)) )
 827        oneDProfile(:) = matmul(bSqrtSea(latitudeBandIndex(1), :, :), controlVector_in(1+(columnIndex-1)*nkgdim:columnIndex*nkgdim))
 828      else ! Land or Sea Ice
 829        latitudeBandIndex = minloc( abs( latitude - latLand(:)) )
 830        oneDProfile(:) = matmul(bSqrtLand(latitudeBandIndex(1), :, :), controlVector_in(1+(columnIndex-1)*nkgdim:columnIndex*nkgdim))
 831      end if
 832      offset = 0
 833      do varIndex = 1, bmat1D_numIncludeAnlVar
 834        currentColumn => col_getColumn(column, headerIndex, varName_opt=bmat1D_includeAnlVar(varIndex))
 835        currentColumn(:) = currentColumn(:) + oneDProfile(offset+1:offset+size(currentColumn))
 836        offset = offset + size(currentColumn)
 837      end do
 838      if (offset /= nkgdim) then
 839        write(*,*) 'bmat1D_bsqrtHi: offset, nkgdim', offset, nkgdim
 840        call utl_abort('bmat1D_bSqrtHi: inconsistency between Bmatrix and statevector size')
 841      end if
 842    end do
 843    !$OMP END PARALLEL DO
 844
 845    deallocate(oneDProfile)
 846    if (mmpi_myid == 0) write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
 847    if (mmpi_myid == 0) write(*,*) 'bmat1D_bSqrtHi: done'
 848
 849  end subroutine bmat1D_bSqrtHi

 854  subroutine bmat1D_bSqrtHiAd(controlVector_in, column, obsSpaceData)

 869    real(8) :: latitude

 871

 917  !--------------------------------------------------------------------------
 918  ! bmat1D_bSqrtEns
 919  !--------------------------------------------------------------------------
 920  subroutine bmat1D_bSqrtEns(controlVector_in, column)
 921    !
 922    !:Purpose: Ensemble component of B square root in 1DVar mode
 923    !
 924    implicit none
 925
 926    ! arguments:
 927    real(8),                 intent(in)    :: controlVector_in(cvDim_mpilocal)
 928    type(struct_columnData), intent(inout) :: column
