 214  subroutine oer_setObsErrors(obsSpaceData, obserrorMode_in, useTovsUtil_opt)
 215    !
 216    !:Purpose: read and set observation errors (from former sucovo subroutine).
 217    !
 218    type(struct_obs)             :: obsSpaceData
 219    character(len=*), intent(in) :: obserrorMode_in
 220    logical, optional            :: useTovsUtil_opt
 221
 222    namelist /namoer/ new_oer_sw, obsfile_oer_sw, visAndGustAdded
 223    namelist /namoer/ mwAllskyTtInflateByOmp, mwAllskyTtInflateByClwDiff
 224    namelist /namoer/ mwAllskyHuInflateByOmp, mwAllskyHuInflateBySiDiff
 225    namelist /namoer/ amsuaClearCldPredThresh
 226    namelist /namoer/ amsuaInflateErrAllskyCoeff
 227    namelist /namoer/ readOldSymmetricObsErrFile
 228    integer :: fnom, fclos, nulnam, ierr
 229
 230    !
 231    !- 1.  Setup Mode
 232    !
 233    obserrorMode = obserrorMode_in
 234
 235    ! Additional key to allow the use of 'util' column in stats_tovs file
 236    if (present(useTovsUtil_opt)) then
 237      useTovsUtil = useTovsUtil_opt
 238    else
 239      useTovsUtil = .false.
 240    end if
 241
 242    ! read namelist namoer
 243    new_oer_sw = .false.
 244    obsfile_oer_sw  = .false.
 245    visAndGustAdded = .false.
 246    mwAllskyTtInflateByOmp = .false.
 247    mwAllskyTtInflateByClwDiff = .false.
 248    mwAllskyHuInflateByOmp = .false.
 249    mwAllskyHuInflateBySiDiff = .false.
 250    amsuaClearCldPredThresh(:) = 0.03D0
 251    amsuaClearCldPredThresh(1) = 0.05D0
 252    amsuaClearCldPredThresh(4) = 0.02D0
 253    amsuaInflateErrAllskyCoeff = 13.0D0
 254    readOldSymmetricObsErrFile = .true.
 255
 256    if (utl_isNamelistPresent('namoer','./flnml')) then
 257      nulnam = 0
 258      ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
 259      read (nulnam, nml = NAMOER, iostat = ierr)
 260      if (ierr /= 0) call utl_abort('oer_setObsErrors: Error reading namelist')
 261      if (mmpi_myid == 0) write(*,nml=namoer)
 262      ierr = fclos(nulnam)
 263    else
 264      write(*,*)
 265      write(*,*) 'oer_setObsErrors: namoer is missing in the namelist. The default value will be taken.'
 266    end if
 267
 268    !
 269    !- 2.  Read in the observation std dev errors
 270    !
 271
 272    !- 2.1 Radiance data
 273    if (obs_famexist(obsSpaceData,'TO')) then
 274      call oer_readObsErrorsTOVS
 275    else 
 276      write(*,*) "oer_setObsErrors: No brightness temperature observations found."
 277    end if
 278    
 279    !- 2.2 Conventional data
 280    if (obs_famExist(obsSpaceData, 'UA') .or. obs_famExist(obsSpaceData, 'AI') .or. obs_famExist(obsSpaceData, 'SW') .or. &
 281        obs_famExist(obsSpaceData, 'SF') .or. obs_famExist(obsSpaceData, 'GP') .or. obs_famExist(obsSpaceData, 'SC') .or. &
 282        obs_famExist(obsSpaceData, 'PR')) then
 283
 284      call oer_readObsErrorsCONV()
 285
 286    else
 287
 288      write(*,*) "oer_setObsErrors: No conventional weather observations found."
 289
 290    end if
 291
 292    !- 2.3 Constituent data
 293    if (obs_famexist(obsSpaceData,'CH')) then
 294      call chm_read_obs_err_stddev
 295    else
 296      write(*,*) "oer_setObsErrors: No CH observations found."
 297    end if
 298
 299    !- 2.4 Sea ice concentration
 300    if (obs_famexist(obsSpaceData,'GL')) then
 301      call oer_readObsErrorsIce
 302    else
 303      write(*,*) "oer_setObsErrors: No GL observations found."
 304    end if
 305
 306    !- 2.5 SST
 307    if (obs_famexist(obsSpaceData,'TM')) then    
 308      call oer_readObsErrorsSST
 309    else
 310      write(*,*) "oer_setObsErrors: No TM observations found."
 311    end if
 312
 313    !- 2.6 Hydrology
 314    if (obs_famexist(obsSpaceData,'HY')) then
 315      call oer_readObsErrorsHydro
 316    else
 317      write(*,*) "oer_setObsErrors: No HY observations found."
 318    end if
 319
 320    !
 321    !- 3.  Set obs error information in obsSpaceData object
 322    !
 323    call oer_fillObsErrors(obsSpaceData)
 324
 325    !
 326    !- 4.  Deallocate temporary storage
 327    !
 328    if (obs_famExist(obsSpaceData,'CH')) call chm_dealloc_obs_err_stddev
 329
 330  end subroutine oer_setObsErrors

 334  !--------------------------------------------------------------------------
 335  subroutine oer_readObsErrorsTOVS
 336    !
 337    !:Purpose: Read the observation error statistics and
 338    !           utilization flag for TOVS processing. This information
 339    !           resides on an ASCII file and is read using a free format.
 340    !
 341    implicit none
 342
 343    integer, parameter :: bgckColumnIndex = 1
 344    integer, parameter :: analysisColumnIndex = 2
 345
 346    integer,external  :: FNOM, FCLOS
 347    integer :: IER, ILUTOV, ILUTOV2, JI, obsErrorColumnIndex, JL, JM 
 348    integer :: INUMSAT, INUMSAT2, ISAT, IPLF
 349    integer :: IPLATFORM(tvs_maxNumberOfSensors), ISATID(tvs_maxNumberOfSensors)
 350    integer :: IINSTRUMENT(tvs_maxNumberOfSensors), NUMCHN(tvs_maxNumberOfSensors)
 351    integer :: NUMCHNIN(tvs_maxNumberOfSensors)
 352    integer :: IPLATFORM2, ISATID2, IINSTRUMENT2, NUMCHNIN2
 353    integer :: IUTILST(tvs_maxChannelNumber,tvs_maxNumberOfSensors)
 354    integer :: ICHN(tvs_maxChannelNumber,tvs_maxNumberOfSensors)
 355    integer :: ICHNIN(tvs_maxChannelNumber,tvs_maxNumberOfSensors)
 356    integer :: ICHNIN2(tvs_maxChannelNumber)
 357
 358    real(8) :: TOVERRIN(tvs_maxChannelNumber,2,tvs_maxNumberOfSensors)
 359    real(8) :: cldPredThreshInput(tvs_maxChannelNumber,tvs_maxNumberOfSensors,2)
 360    real(8) :: errThreshAllskyInput(tvs_maxChannelNumber,tvs_maxNumberOfSensors,2)
 361    real(8) :: tovsObsInflation(tvs_maxChannelNumber,tvs_maxNumberOfSensors)
 362    real(8) :: clearCldPredThreshInput(tvs_maxChannelNumber,tvs_maxNumberOfSensors)
 363    real(8) :: inflateErrAllskyCoeffInput(tvs_maxNumberOfSensors)
 364    integer :: useStateDepSigmaObsInput(tvs_maxChannelNumber,tvs_maxNumberOfSensors)
 365    integer :: amsuaChannelOffset, amsuaChannelNum  
 366
 367    character (len=132) :: CLDUM,CPLATF,CINSTR
 368
 369    write(*,*) 'oer_readObsErrorsTOVS: reading observation error statistics required for TOVS processing'
 370
 371    !
 372    !    1. Initialize
 373    !       ----------
 374    !
 375    TOVERRST(:,:) = 0.0D0
 376    TOVERRIN(:,:,:) = 0.0D0
 377    cldPredThresh(:,:,:) = 0.0d0
 378    cldPredThreshInput(:,:,:) = 0.0d0
 379    errThreshAllsky(:,:,:) = 0.0d0
 380    errThreshAllskyInput(:,:,:) = 0.0d0
 381    tovsObsInflation(:,:) = 0.0d0
 382    IUTILST(:,:) = 0
 383    useStateDepSigmaObs(:,:) = .false.
 384    useStateDepSigmaObsInput(:,:) = 0
 385    clearCldPredThresh(:,:) = 0.0d0
 386    clearCldPredThreshInput(:,:) = 0.0d0
 387    inflateErrAllskyCoeff(:) = 0.0d0
 388    inflateErrAllskyCoeffInput(:) = 0.0d0
 389
 390    IPLATFORM(:) = 0
 391    NUMCHN(:) = 0
 392    NUMCHNIN(:) = 0
 393    ICHN(:,:) = 0
 394    ICHNIN(:,:) = 0
 395    ICHNIN2(:) = 0
 396
 397    if (tvs_nobtov == 0) return
 398
 399    !
 400    !     2. Open the file
 401    !        -------------
 402    !
 403    ilutov = 0
 404    IER =  FNOM(ILUTOV,'stats_tovs','SEQ+FMT',0)
 405    if (IER < 0) call utl_abort ('oer_readObsErrorsTOVS: Problem opening file stats_tovs')
 406
 407    !
 408    !     3. Read number of satellites
 409    !        -------------------------
 410    !
 411    read(ILUTOV,*)
 412    read(ILUTOV,*) INUMSAT
 413    read(ILUTOV,*)
 414
 415    if (inumsat > tvs_maxNumberOfSensors) then
 416      write(*,'(A)') ' Number of sensors in stats_tovs file is greater than maximum allowed (tvs_maxNumberOfSensors)'
 417      call utl_abort('oer_readObsErrorsTOVS')
 418    end if
 419
 420    !
 421    !     4. Read the satellite identification, the number of channels,
 422    !        the observation errors and the utilization flags
 423    !        ----------------------------------------------------------
 424    !
 425    write(*,*) 'oer_readObsErrorsTOVS: Reading stats_tovs file.'
 426
 427    DO JL = 1, INUMSAT
 428
 429      read(ILUTOV,*)
 430      read(ILUTOV,'(A)') CLDUM
 431      write(*,'(A)') CLDUM
 432      CINSTR=CLDUM
 433      call split(CINSTR," ",CPLATF)
 434      write(*,*) "CINSTR: ",CINSTR
 435      write(*,*) "CPLATF: ",CPLATF
 436      read(ILUTOV,*)
 437      read(ILUTOV,*) ISATID(JL), NUMCHNIN(JL)
 438
 439      do JI = 1, 3
 440        read(ILUTOV,*)
 441      end do
 442
 443      if (CPLATF == 'FY-3C') THEN 
 444         CPLATF = 'FY3-3'
 445         CINSTR = 'MWHS2'
 446      end if
 447
 448      IPLATFORM(JL) =  tvs_getPlatformId(CPLATF)
 449
 450      if (IPLATFORM(JL) == -1) call utl_abort ('oer_readObsErrorsTOVS: Unknown platform!')
 451
 452      IINSTRUMENT(JL) = tvs_getInstrumentId(CINSTR)
 453      
 454      if (IINSTRUMENT(JL) == -1) call utl_abort ('oer_readObsErrorsTOVS: Unknown instrument!')
 455
 456      do JI = 1, NUMCHNIN(JL)
 457        read(ILUTOV,*) ICHNIN(JI,JL), TOVERRIN(ICHNIN(JI,JL),1,JL), TOVERRIN(ICHNIN(JI,JL),2,JL), IUTILST(ICHNIN(JI,JL),JL), tovsObsInflation(ICHNIN(JI,JL),JL)
 458      end do
 459      read(ILUTOV,*)
 460
 461    end do
 462
 463    ! read in the parameters to define the user-defined symmetric TOVS errors
 464    if (tvs_mwAllskyAssim) then
 465      ilutov2 = 10
 466      IER =  FNOM(ILUTOV2,'stats_tovs_symmetricObsErr','SEQ+FMT',0)
 467      if (IER < 0) call utl_abort ('oer_readObsErrorsTOVS: Problem opening symmetricObsErr file.')
 468
 469      read(ILUTOV2,*)
 470      read(ILUTOV2,*) INUMSAT2
 471      read(ILUTOV2,*)
 472
 473      write(*,*) 'oer_readObsErrorsTOVS: Reading symmetricObsErr file.'
 474
 475      DO JL = 1, INUMSAT2
 476
 477        read(ILUTOV2,*)
 478        read(ILUTOV2,'(A)') CLDUM
 479        write(*,'(A)') CLDUM
 480        CINSTR = CLDUM
 481        call split(CINSTR," ",CPLATF)
 482        write(*,*) "CINSTR: ",CINSTR
 483        write(*,*) "CPLATF: ",CPLATF
 484        read(ILUTOV2,*)
 485
 486        ! If reading the old style stats_tovs_symmetricObsErr, the the all-sky parameters are available only for AMSUA.
 487        if (readOldSymmetricObsErrFile) then
 488          read(ILUTOV2,*) ISATID2, NUMCHNIN2
 489          if (CINSTR == "AMSUA") inflateErrAllskyCoeffInput(JL) = amsuaInflateErrAllskyCoeff
 490        else
 491          read(ILUTOV2,*) ISATID2, NUMCHNIN2, inflateErrAllskyCoeffInput(JL)
 492        end if
 493
 494        if (ISATID2 /= ISATID(JL) .or. NUMCHNIN2 /= NUMCHNIN(JL)) &
 495          call utl_abort ('oer_readObsErrorsTOVS: problem with ISATID2, NUMCHNIN2 in symmetricObsErr')
 496
 497        do JI = 1, 3
 498          read(ILUTOV2,*)
 499        end do
 500
 501        IPLATFORM2 = tvs_getPlatformId(CPLATF)
 502        IINSTRUMENT2 = tvs_getInstrumentId(CINSTR)
 503        if (IPLATFORM2 /= IPLATFORM(JL) .or. IINSTRUMENT2 /= IINSTRUMENT(JL)) & 
 504          call utl_abort ('oer_readObsErrorsTOVS: problem with IPLATFORM2, IINSTRUMENT2 in symmetricObsErr')
 505
 506        do JI = 1, NUMCHNIN2
 507          ! If reading the old style stats_tovs_symmetricObsErr, then the all-sky parameters are available only for AMSUA.
 508          if (readOldSymmetricObsErrFile) then

 704    subroutine split(str, delims, before)
 705      !
 706      !:Comment:
 707      ! Code extracted from Benthien's module: http://www.gbenthien.net/strings/index.html
 708      ! Routine finds the first instance of a character from 'delims' in the
 709      ! the string 'str'. The characters before the found delimiter are
 710      ! output in 'before'. The characters after the found delimiter are
 711      ! output in 'str'. 
 712      !
 713
 714      character(len=*) :: str
 715      character(len=*) :: delims
 716      character(len=*) :: before
 717
 718      character :: ch,cha
 719      integer lenstr,i,k,ipos,iposa
 720      str=adjustl(str)
 721      call compact(str)
 722      lenstr=len_trim(str)
 723
 724      if(lenstr == 0) return ! string str is empty
 725      k=0
 726      before=' '
 727      do i=1,lenstr
 728        ch=str(i:i)
 729        
 730        ipos=index(delims,ch)
 731
 732        if(ipos == 0) then ! character is not a delimiter
 733          k=k+1
 734          before(k:k)=ch
 735          cycle
 736        end if
 737        if(ch /= ' ') then ! character is a delimiter that is not a space
 738          str=str(i+1:)
 739          exit
 740        end if
 741
 742        cha=str(i+1:i+1)  ! character is a space delimiter
 743        iposa=index(delims,cha)
 744        if(iposa > 0) then   ! next character is a delimiter 
 745          str=str(i+2:)
 746          exit
 747        else
 748          str=str(i+1:)
 749          exit
 750        end if
 751      end do
 752      if(i >= lenstr) str=''

1734    !
1735    implicit none

1746    integer :: channelNumber, channelIndex

1751    real(8) :: siObs
1752    real(8) :: siFG
1753    real(8) :: deltaE1
1754    real(8) :: deltaE2
1755    real(8) :: ompValue
1756    real(8) :: sigmaObsBeforeInflation
1757    real(8) :: sigmaObsAfterInflation

1760

1781          (.not. mwAllskyTtInflateByOmp .and. .not. mwAllskyTtInflateByClwDiff)) then
1782        return
1783      end if
1784
1785      clwObs = obs_headElem_r(obsSpaceData, OBS_CLWO, headerIndex)
1786      clwFG  = obs_headElem_r(obsSpaceData, OBS_CLWB, headerIndex)
1787

1817

1820          (.not. mwAllskyHuInflateByOmp .and. .not. mwAllskyHuInflateBySiDiff)) then

1822      end if

1825      siFG  = obs_headElem_r(obsSpaceData, OBS_SIB, headerIndex)
1826

1829

1833                          ', sensorIndex=', sensorIndex, &

1839  
1840      ! error inflation for cloud placement

1843          ((siObs - clearCldPredThresh(channelNumber_withOffset,sensorIndex)) * &
1844           (siFG  - clearCldPredThresh(channelNumber_withOffset,sensorIndex)) < 0) .and. &

1868          sigmaObsBeforeInflation, ', sigmaObsInflated=', sigmaObsAfterInflation
1869    end if

1871    call obs_bodySet_r(obsSpaceData, OBS_OER, bodyIndex, sigmaObsAfterInflation)

2638    logical                 :: beSilent
2639    logical                 :: ldata
2640    logical                 :: analysisMode

2642    integer bodyIndex, headerIndex, ityp, iass, IZTDJ, NBRPDATE, ICOUNT, ICOUNT2

2651    REAL*8  ZBTSFC, ZBPSFC, ZBZSFC, ZDZ, ZSTDOMP

2654    !     ZZDERMIN = MIN ZTD OER VALUE (M)
2655    !     ZZDERMAX = MAX ZTD OER VALUE (M)
2656    !     ZTDERFAC = MULTIPLICATION FACTOR FOR FORMAL ZTD MEASUREMENT ERROR

2659    !

2664    DATA ZOPEFAC  /1.0D0/
2665    !

2668    !
2669    REAL*8 ZGAMMA
2670    DATA ZGAMMA /0.0065D0/
2671
2672    !     ----------------------------------------------------------------------------------
2673    !     LINEAR REGRESSION EQUATION CONSTANTS AND COEFFS FOR ZTD ERROR AND STD(O-P):
2674
2675    !     ZRCONST, ZRCOEFF:
2676    !       - From linear regression of Desroziers error estimates binned by observed ZWD.
2677    !       - Gives ZTDerror (mm) as function of ZWD (m):
2678    !            ZTDerror(mm) = ZRCONST + ZRCOEFF*ZWD(m)
2679    !     ZRCONST2, ZRCOEFF2:
2680    !       - From linear regression of Std(O-P) binned by observed ZWD.
2681    !       - Gives Std(O-P) (mm) as function of ZWD (m):
2682    !            Std(O-P)(mm) = ZRCONST2 + ZRCOEFF2*ZWD(m)
2683    !     ----------------------------------------------------------------------------------
2684    REAL*8 ZRCONST, ZRCOEFF, ZRCONST2, ZRCOEFF2
2685    DATA  ZRCONST  /5.12D0/
2686    DATA  ZRCOEFF  /26.4D0/
2687    DATA  ZRCONST2 /6.67D0/
2688    DATA  ZRCOEFF2 /42.6D0/
2689    !
2690    !
2691    if (.not.beSilent) write(*,*) 'ENTER SETERRGPSGB'
2692    !
2693    DEBUG = .FALSE.
2694
2695    LLCZTDE = .FALSE.
2696    LLRZTDE = .FALSE.
2697    LLFZTDE = .FALSE.
2698    IF (gps_gb_YZTDERR  <  0.0D0) then
2699      LLFZTDE = .TRUE.
2700    else if (gps_gb_YZTDERR  >  0.0D0) then
2701      LLCZTDE = .TRUE.
2702    ELSE
2703      LLRZTDE = .TRUE.
2704    end if

2707

2720      LLFER     = .FALSE.
2721      LLZWD     = .FALSE.
2722      ASSIM     = .FALSE.
2723      ERRSET    = .FALSE.
2724      ZZTD      = -1.0D0
2725      ZZWD      = -1.0D0
2726      ZPSFC     = -1.0D0
2727      LESTP     = .FALSE.
2728      ZSTDOMP   = 15.0D0*0.001D0
2729
2730       !   Get Psfc (Pa), Tsfc (K) and model surface height (m) from background profile
2731
2732      ZBPSFC = col_getElem(columnTrlOnTrlLev, 1, headerIndex, 'P0')
2733      ZBTSFC = col_getElem(columnTrlOnTrlLev, nlev_T, headerIndex, 'TT')
2734      ZBZSFC = col_getHeight(columnTrlOnTrlLev, nlev_T, headerIndex, 'TH')
2735       !
2736       !    Loop over all body indices of current report; Set the ZTD error if
2737       !    constant value specified (LLCZTDE=true). Get GPS height and Psfc obs (if any).
2738       !    Get ZTD obs, ZTD formal error and ZWD observation.
2739       !
2740      call obs_set_current_body_list(obsSpaceData, headerIndex)
2741      BODY: do
2742        bodyIndex = obs_getBodyIndex(obsSpaceData)
2743        if (bodyIndex < 0) exit BODY
2744        ityp   = obs_bodyElem_i(obsSpaceData, OBS_VNM, bodyIndex)
2745        iass   = obs_bodyElem_i(obsSpaceData, OBS_ASS, bodyIndex)
2746        zval   = obs_bodyElem_r(obsSpaceData, OBS_VAR, bodyIndex)
2747          !         Store Psfc
2748        IF (ityp == BUFR_NEPS) then
2749          IF (zval  >  0.0D0) ZPSFC = zval
2750        end if
2751          !         Set ZTDOER to constant value (if LLCZTDE); get value of ZTD, 
2752          !         ZTD formal error (OBS_OER) and antenna height (OBS_PPP).
2753        IF (ityp == BUFR_NEZD) then

2757          end if
2758          zlev   = obs_bodyElem_r(obsSpaceData, OBS_PPP, bodyIndex)

2779
2780       !      Replace formal ZTD error with real error for all ZTD to be assimilated.
2781       !      Set Std(O-P) as function of ZWD for ZTD observation and store in OBS_HPHT.

2797            end if

2799            ZHD = 2.2766D-05 * ZPSFC
2800                !               Compute the wet delay (m) from ZTD and ZHD. Avoid negative ZWD.
2801            if (ZHD  >  ZZTD) then
2802              ZWD = 0.0D0
2803            else
2804              ZWD = ZZTD - ZHD
2805            end if
2806          end if
2807             !              Std(O-P) for background check. Limit to 30 mm in case ZTD obs is bad (too high).             
2808          ZSTDOMP = (ZRCONST2 + ZRCOEFF2*ZWD)*0.001D0
2809          ZSTDOMP = MIN(ZZDERMAX, ZSTDOMP)

2813          IF (.NOT. ERRSET) then

2826            end if

2828            IF (LLFER) then

2833              end if

2835            end if
2836          end if

2838          call obs_bodySet_r(obsSpaceData,OBS_OER,IZTDJ, ZTDOER*gps_gb_YZDERRWGT)
2839          IF (.NOT.analysisMode) call obs_bodySet_r(obsSpaceData,OBS_HPHT,IZTDJ, ZSTDOMP)
2840          IF (DEBUG .and. (ICOUNT2  <=  50) .and. LESTP) then
2841            write(*,*) 'TAG    SITE    ZTD    ERROR    ELEV    PSFC    ZWD     STDOMP'
2842            write(*,*) 'ERRDEBUG ', cstnid, &
2843                 ZZTD*1000.D0, ZTDOER*1000.D0, zlev, ZPSFC/100.D0, ZWD*1000.D0, ZSTDOMP*1000.D0
2844          end if
2845        ELSE
2846          call utl_abort('SETERRGPSGB: ERROR:NEGATIVE ZTD VALUE!')
2847        end if
2848      end if
2849
2850       !
2851    end do  HEADER
2852
2853    !      IF (DEBUG) call utl_abort('******DEBUG STOP*******')
2854
2855    IF (.not.ldata) gps_gb_numZTD = 0
2856
2857    IF (ldata .and. .not.beSilent) write(*,*) ' gps_gb_numZTD = ', ICOUNT
2858
2859    if (.not.beSilent) write(*,*) 'EXIT SETERRGPSGB'
2860
2861  END SUBROUTINE OER_SETERRGPSGB
