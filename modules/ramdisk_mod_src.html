 17module ramDisk_mod
 18  ! MODULE ramDisk_mod (prefix='ram' category='8. Low-level utilities and constants')
 19  !
 20  !:Purpose: To control the file manipulations/enquiries on the RAM disk
 21  !
 22  use utilities_mod
 23  use clib_interfaces_mod
 24  implicit none
 25  save
 26  private
 27
 28  ! public procedures
 29  public :: ram_setup, ram_fullWorkingPath, ram_remove, ram_getRamDiskDir
 30
 31  character(len=256) :: ram_disk_dir
 32
 33  logical :: ram_disk_dir_exists
 34  logical :: initialized = .false.
 35
 36contains
 37
 38  !--------------------------------------------------------------------------
 39  ! ram_setup
 40  !--------------------------------------------------------------------------
 41  subroutine ram_setup()
 42    implicit none
 43
 44    integer :: length_ram_disk_dir, status
 45
 46    !
 47    !- Determine the ramdisk directory, if available
 48    !
 49    status = 0
 50    call get_environment_variable('MIDAS_RAMDISKDIR',ram_disk_dir,length_ram_disk_dir,status,.true.)
 51
 52    if (status.gt.1) then
 53      write(*,*) 'ram_setup: Problem when getting the environment variable MIDAS_RAMDISKDIR'
 54    end if
 55    if (status.eq.1) then
 56      write(*,*) 'ram_setup: The environment variable MIDAS_RAMDISKDIR has not been detected!'
 57      write(*,*) '           Assume all files in current working directory'
 58      ram_disk_dir_exists = .false.  
 59      ram_disk_dir = 'DOES_NOT_EXIST'
 60    else
 61      write(*,*)
 62      write(*,*) 'ram_setup: The environment variable MIDAS_RAMDISKDIR has correctly been detected'
 63      write(*,*) 'ram_setup: Files will first be opened from directory: ', trim(ram_disk_dir)
 64      ram_disk_dir_exists = .true.
 65    end if
 66
 67    initialized = .true.
 68
 69  end subroutine ram_setup
 70
 71  !--------------------------------------------------------------------------
 72  ! ram_fullWorkingPath
 73  !--------------------------------------------------------------------------
 74  function ram_fullWorkingPath(fileName, noAbort_opt, copyToRamDisk_opt) result(fullWorkingPath)
 75    !
 76    !:Purpose: Given a filename, return the full path by either adding the 
 77    !          current working directory or the ram disk directory. By default,
 78    !          will copy the file to the ram disk directory, if it exists.
 79    !
 80    implicit none
 81    character(len=512) :: fullWorkingPath
 82    logical, optional  :: noAbort_opt
 83    logical, optional  :: copyToRamDisk_opt
 84    character(len=*)   :: fileName
 85
 86    logical            :: fileExists, noAbort, copyToRamDisk
 87    character(len=256) :: fileName2, subDirectory
 88    integer            :: status
 89
 90    if ( .not. initialized ) then
 91      call ram_setup()
 92    end if
 93
 94    if ( present(noAbort_opt) ) then
 95      noAbort = noAbort_opt
 96    else
 97      noAbort = .false.
 98    end if
 99
100    if ( present(copyToRamDisk_opt) ) then
101      copyToRamDisk = copyToRamDisk_opt
102    else
103      copyToRamDisk = .true.
104    end if
105
106    ! this should make it safe for calls where input and output are the same variable
107    fileName2 = trim(fileName)
108
109    ! first look for file in the ram disk directory
110    if ( ram_disk_dir_exists ) then
111      fullWorkingPath = trim(ram_disk_dir) // '/' // trim(fileName2)
112    else
113      fullWorkingPath = trim(fileName2)
114    end if
115    inquire(file=trim(fullWorkingPath),exist=fileExists)
116
117    ! treat case when no ramdisk exists
118    if ( .not. ram_disk_dir_exists ) then
119      if ( fileExists ) then
120        return
121      else
122        if ( noAbort ) then
123          fullWorkingPath = ' '
124          return
125        else
126          write(*,*) 'ram_fullWorkingPath: file name = ', trim(fileName2)
127          call utl_abort('ram_fullWorkingPath: this file cannot be found on disk.')
128        end if
129      end if
130    end if
131
132    ! treat the case when ramdisk DOES exists
133    if ( fileExists ) then
134      write(*,*) 'ram_fullWorkingPath: this file found on ram disk: ', trim(fileName2)
135    else
136      ! not found on ram disk, so now look in working directory
137      inquire(file='./' // trim(fileName2),exist=fileExists)
138
139      if ( .not. fileExists ) then
140        if ( noAbort ) then
141          fullWorkingPath = ' '
142        else
143          write(*,*) 'ram_fullWorkingPath: file name          = ', trim(fileName2)
144          write(*,*) 'ram_fullWorkingPath: ram disk directory = ', trim(ram_disk_dir)
145          call utl_abort('ram_fullWorkingPath: this file cannot be found.')
146        end if
147      else
148
149        if ( copyToRamDisk ) then
150          ! copy the file from disk to the ramdisk
151          if ( index(trim(filename2),'/') /= 0 ) then
152            status = clib_dirname(trim(filename2),subDirectory)
153            status = clib_mkdir_r(trim(ram_disk_dir) // '/' // trim(subDirectory))
154            if ( status /= clib_ok ) then
155              call utl_abort('ram_fullWorkingPath: problem with mkdir')
156            end if
157            status = clib_isdir(trim(ram_disk_dir) // '/' // trim(subDirectory))
158            if ( status /= clib_ok ) then
159              call utl_abort('ram_fullWorkingPath: problem with checking existence of directory')
160            end if
161          end if
162
163          ! copy the file from disk to the ramdisk
164          status = utl_copyFile(trim(fileName2), trim(ram_disk_dir) // '/' // trim(fileName2))
165
166          fullWorkingPath = trim(ram_disk_dir) // '/' // trim(fileName2)
167          write(*,*) 'ram_fullWorkingPath: file copied to ramdisk: ', trim(fullWorkingPath)
168        else
169          fullWorkingPath = trim(fileName2)
170          write(*,*) 'ram_fullWorkingPath: file left on disk, as requested: ', trim(fullWorkingPath)
171        end if
172
173      end if
174
175    end if
176
177  end function ram_fullWorkingPath
178
179  !--------------------------------------------------------------------------
180  ! ram_remove
181  !--------------------------------------------------------------------------
182  function ram_remove(fullWorkingPath) result(returnCode)
183    !
184    !:Purpose:  Given the full path+filename, remove the file only if 
185    !           it is located on the ram disk (to free up memory)
186    !
187    implicit none
188    character(len=*) :: fullWorkingPath
189    integer          :: returnCode
190    logical          :: fileExists
191
192    if ( .not. initialized ) then
193      call ram_setup()
194    end if
195
196    inquire(file=trim(fullWorkingPath),exist=fileExists)
197    if ( .not. fileExists) then
198      write(*,*) 'ram_Remove: file does not exist: ',trim(fullWorkingPath)
199      returnCode = 0
200      return
201    end if
202
203    if ( .not. ram_disk_dir_exists ) then
204      write(*,*) 'ram_remove: no ram disk in use.'
205      returnCode = 0
206      return
207    end if
208
209    if ( index(trim(fullWorkingPath), trim(ram_disk_dir)) == 1 ) then
210      write(*,*) 'ram_remove: removing file that is on the ram disk: ', trim(fullWorkingPath)
211      returnCode = clib_remove(fullWorkingPath)
212    else
213      write(*,*) 'ram_remove: this file not on ram disk: ', trim(fullWorkingPath)
214      returnCode = 0
215    end if
216
217  end function ram_remove
218

220  function ram_getRamDiskDir() result(fullWorkingPath)
221
222    implicit none
223    character(len=512) :: fullWorkingPath
224
225    if ( ram_disk_dir_exists ) then
226      fullWorkingPath = trim(ram_disk_dir) // '/'
227    else
228      fullWorkingPath = ' '
229    end if
230    
231  end function ram_getRamDiskDir
232
233end module ramDisk_mod
