  99       ip1, ip2, ip3, typvar, nomvar) result(vfstlir)
 100    implicit none

 104    integer :: iun, ni, nj, nk, datev, ip1, ip2, ip3
 105    character(len=*) :: etiket

 107    character(len=*) :: typvar

 130             end do
 131          end do

 255  subroutine utl_matsqrt(matrix, rank, exponentSign, printInformation_opt )
 256    ! 
 257    !:Purpose: Calculate square root of an error covariance matrix
 258    !
 259    implicit none
 260
 261    integer, intent(in)           :: rank
 262    real(8), intent(inout)        :: matrix(rank,rank)
 263    real(8), intent(in)           :: exponentSign
 264    logical, intent(in), optional :: printInformation_opt ! switch to print be more verbose
 265
 266    real(8), allocatable :: eigenValues(:)
 267    real(8), allocatable :: work(:)
 268    real(8), allocatable :: eigenVectors(:,:)
 269    integer :: sizework, info, index, index1, index2 
 270    logical :: printInformation
 271
 272    if (present(printInformation_opt)) then
 273       printInformation = printInformation_opt
 274    else
 275       printInformation = .false.
 276    end if
 277
 278    if (printInformation) then
 279      write(*,*)
 280      write(*,*) 'utl_matsqrt: Starting...'
 281    end if
 282
 283    sizework = 64 * rank
 284    allocate(work(sizework))
 285
 286    allocate(eigenValues (rank))
 287    allocate(eigenVectors(rank,rank))
 288
 289    !- Calculate EigenVectors (V) and EigenValues (D) of B matrix
 290    eigenVectors(:,:) = matrix(:,:)
 291
 292    call dsyev('V','U',rank,   & ! IN
 293               eigenVectors,   & ! INOUT
 294               rank,           & ! IN
 295               eigenValues,    & ! OUT
 296               work, sizework, & ! IN
 297               info )            ! OUT
 298
 299    if ( info /= 0 ) then
 300      write(*,*)
 301      write(*,*) 'dsyev: ',info
 302      call utl_abort('utl_matsqrt: DSYEV failed!')
 303    end if
 304
 305    if (printInformation) then
 306      write(*,*)
 307      write(*,'(1x,"Original EIGEN VALUES: ")')
 308      write(*,'(1x,10f7.3)') (eigenValues(index),index=1,rank)
 309      if (exponentSign < 0.d0) then
 310        write(*,*)
 311        write(*,'(A,1x,e14.6)') "Condition number:", &
 312             maxval(eigenValues(:))/minval(eigenValues(:))
 313      end if
 314    end if
 315
 316    !- Calculate Matrix^0.5 = V D^0.5 V^t
 317    where(eigenValues(:) < 0.d0)
 318      eigenValues = 0.d0
 319    end where
 320
 321    eigenValues(:) = eigenValues(:)**(0.5d0*exponentSign)
 322
 323    do index1 = 1, rank
 324      do index2 = 1, rank
 325        matrix(index1,index2) = sum ( eigenVectors (index1,1:rank)   &
 326                                    * eigenVectors (index2,1:rank)   &
 327                                    * eigenValues(1:rank) )
 328      end do
 329    end do
 330
 331    deallocate(eigenVectors)
 332    deallocate(eigenValues)
 333    deallocate(work)
 334    
 335    if (printInformation) then
 336      write(*,*)
 337      write(*,*) 'utl_matsqrt: Ending...'
 338    end if
 339
 340  end subroutine utl_matsqrt

 345  subroutine utl_matInverse(matrix, rank, inverseSqrt_opt, printInformation_opt)
 346    !
 347    !:Purpose: Calculate the inverse of a covariance matrix 
 348    !          and, optionally, also the inverse square-root.
 349    !
 350    implicit none
 351
 352    ! Arguments
 353    integer, intent(in)              :: rank                 ! order of the matrix
 354    real(8), intent(inout)           :: matrix(:,:)          ! on entry, the original matrix; on exit, the inverse
 355    real(8), intent(inout), optional :: inverseSqrt_opt(:,:) ! if present, the inverse sqrt matrix on exit 
 356    logical, intent(in), optional    :: printInformation_opt ! switch to print be more verbose
 357
 358    ! Local variables
 359    integer :: index1, index2, info, sizework
 360    real(8) :: sizework_r8
 361    real(8), allocatable :: work(:), eigenVectors(:,:), eigenValues(:)
 362    logical :: printInformation
 363
 364    if (present(printInformation_opt)) then
 365      printInformation = printInformation_opt
 366    else
 367      printInformation = .false.
 368    end if
 369
 370    if (printInformation) then
 371      write(*,*)' utl_matInvers: Inverse matrix of a symmetric matrix'
 372    end if
 373
 374    !     1. Computation of eigenvalues and eigenvectors
 375
 376    allocate(eigenVectors(rank,rank))
 377    allocate(eigenValues(rank))
 378
 379    do index2=1,rank
 380      do index1=1,rank
 381        eigenVectors(index1,index2)=matrix(index1,index2)
 382      end do
 383    end do
 384
 385    ! query the size of the 'work' vector by calling 'DSYEV' with 'sizework=-1'
 386    sizework = -1
 387    info = -1
 388    call dsyev('V','U',rank, eigenVectors, rank, eigenValues, sizework_r8, sizework, info)
 389
 390    ! compute the eigenvalues
 391    sizework=int(sizework_r8)
 392    allocate(work(sizework))
 393    call dsyev('V','U',rank, eigenVectors,rank, eigenValues,work, sizework, info)
 394    deallocate(work)
 395
 396    if (printInformation) then
 397      write(*,'(1x,"Original eigen values: ")')
 398      write(*,'(1x,10f7.3)') (eigenValues(index1),index1=1,rank)
 399
 400      if(minval(eigenValues) > 1.0d-10) then
 401        write(*,'(A,1x,e14.6)') "Condition number:", &
 402             maxval(eigenValues)/minval(eigenValues)
 403      end if
 404    end if
 405
 406    !     2.  Take inverse of eigenvalues
 407
 408    do index1=1,rank
 409      if(eigenValues(index1) > 1.0d-10) then
 410        eigenValues(index1)= 1.0d0/eigenValues(index1)
 411      else
 412        write(*,*) 'utl_matInverse: WARNING eigenvalue is too small = ', index1, eigenValues(index1)
 413        eigenValues(index1) = 0.0d0
 414      end if
 415    end do
 416
 417    if (printInformation) then
 418      write(*,'(1x,"Inverse of original eigen values: ")')
 419      write(*,'(1x,10f7.3)') (eigenValues(index1),index1=1,rank)
 420    end if
 421
 422    !     3.  Compute the inverse matrix
 423
 424    do index2 = 1, rank
 425      do index1 = 1, rank
 426        matrix(index1,index2) = sum ( eigenVectors (index1,1:rank)   &
 427                                    * eigenVectors (index2,1:rank)   &
 428                                    * eigenValues(1:rank) )
 429      end do
 430    end do
 431
 432    !     4.  If requested, computed the inverse square-root also
 433
 434    if (present(inverseSqrt_opt)) then
 435      do index1=1,rank
 436        if(eigenValues(index1) > 1.0d-10) then
 437          eigenValues(index1)= sqrt(eigenValues(index1))
 438        else
 439          eigenValues(index1) = 0.0d0
 440        end if
 441      end do
 442      do index2 = 1, rank
 443        do index1 = 1, rank
 444          inverseSqrt_opt(index1,index2) = sum ( eigenVectors (index1,1:rank)   &
 445                                               * eigenVectors (index2,1:rank)   &
 446                                               * eigenValues(1:rank) )
 447        end do
 448      end do
 449    end if
 450
 451    !     5. Deallocate local arrays
 452    deallocate(eigenVectors,eigenValues)
 453
 454    if (printInformation) then
 455      write(*,*) 'utl_matInverse: done'
 456      write(*,*) ' '
 457    end if
 458
 459  end subroutine utl_matInverse

 464  subroutine utl_eigenDecomp(matrix, eigenValues, eigenVectors, tolerance, numReturned, printInformation_opt)
 465    !
 466    !:Purpose: Calculate eigenValues/Vectors and return only those with eigenValues
 467    !          whose magnitude is greater than the specified tolerance.
 468    !
 469    implicit none
 470
 471    ! Arguments
 472    real(8), intent(inout)        :: matrix(:,:)          ! on entry, the original matrix; on exit, the inverse
 473    real(8), intent(out)          :: eigenValues(:)       ! computed eigenValues
 474    real(8), intent(out)          :: eigenVectors(:,:)    ! computed eigenVectors
 475    real(8), intent(in)           :: tolerance            ! threshold for eigenValue magnitude to be returned
 476    integer, intent(out)          :: numReturned          ! number of eigenValues/Vectors returned
 477    logical, intent(in), optional :: printInformation_opt ! switch to print be more verbose
 478
 479    ! Local variables
 480    integer :: rank, index1, index2, info, sizework
 481    real(8) :: sizework_r8
 482    real(8), allocatable :: work(:), eigenVectorsOrig(:,:), eigenValuesOrig(:)
 483    logical :: printInformation
 484
 485    if (present(printInformation_opt)) then
 486      printInformation = printInformation_opt
 487    else
 488      printInformation = .false.
 489    end if
 490
 491    if (printInformation) then
 492      write(*,*)' utl_eigenDecomp: Eigen decomposition of a symmetric matrix'
 493    end if
 494
 495    !     1. Computation of eigenvalues and eigenvectors
 496
 497    rank = size(matrix,1)
 498    allocate(eigenVectorsOrig(rank,rank))
 499    allocate(eigenValuesOrig(rank))
 500
 501    do index2 = 1, rank
 502      do index1 = 1, rank
 503        eigenVectorsOrig(index1,index2)=matrix(index1,index2)
 504      end do
 505    end do
 506
 507    ! Query the size of the 'work' vector by calling 'DSYEV' with 'sizework=-1'
 508    sizework = -1
 509    info = -1
 510    call dsyev('V', 'U', rank, eigenVectorsOrig, rank, eigenValuesOrig,  &
 511               sizework_r8, sizework, info)
 512
 513    ! Compute the eigenvalues/vectors
 514    sizework = int(sizework_r8)
 515    allocate(work(sizework))
 516    call dsyev('V', 'U', rank, eigenVectorsOrig, rank, eigenValuesOrig,  &
 517               work, sizework, info)
 518    deallocate(work)
 519
 520    if (printInformation) then
 521      write(*,'(1x,"Original eigen values: ")')
 522      write(*,'(1x,10f7.3)') (eigenValuesOrig(index1),index1=1,rank)
 523
 524      if(minval(eigenValuesOrig) > tolerance) then
 525        write(*,'(A,1x,e14.6)') "Condition number:", &
 526             maxval(eigenValuesOrig)/minval(eigenValuesOrig)
 527      end if
 528    end if
 529
 530    !     2.  Determine which eigen values/vectors to return
 531
 532    numReturned = 0
 533    do index1 = rank, 1, -1
 534      if (eigenValuesOrig(index1) > tolerance) then
 535        numReturned = numReturned + 1
 536      else
 537        exit
 538      end if
 539    end do
 540
 541    if (printInformation) then
 542      write(*,*) 'Number of eigen values returned =', numReturned, ' out of', rank
 543    end if
 544
 545    !     3.  Copy eigenValues/Vectors into output arrays with reversed order
 546
 547    do index1 = 1, numReturned
 548      ! And set negative values to zero
 549      eigenValues(index1) = max(0.0D0,eigenValuesOrig(rank-index1+1))
 550    end do
 551    do index1 = numReturned+1, rank
 552      eigenValues(index1) = 0.0D0
 553    end do
 554
 555    do index2 = 1, numReturned
 556      do index1 = 1, rank
 557        eigenVectors(index1,index2) = eigenVectorsOrig(index1,rank-index2+1)
 558      end do
 559    end do
 560    do index2 = numReturned+1, rank
 561      do index1 = 1, rank
 562        eigenVectors(index1,index2) = 0.0D0
 563      end do
 564    end do
 565
 566    !     4. Deallocate local arrays
 567    deallocate(eigenVectorsOrig,eigenValuesOrig)
 568
 569    if (printInformation) then
 570      write(*,*) 'utl_eigenDecomp: done'
 571      write(*,*) ' '
 572    end if
 573
 574  end subroutine utl_eigenDecomp

 579  subroutine utl_pseudo_inverse(inputMatrix, pseudoInverse, threshold_opt)
 580    !
 581    !:Purpose: to calculate the More-Penrose pseudo inverse of the matrix inputMatrix
 582    !
 583    implicit none
 584    !Arguments:
 585    real(8), intent(in)           :: inputMatrix(:,:)   ! Input Matrix
 586    real(8), intent(out)          :: pseudoInverse(:,:) ! its Moore Penrose Pseudo-Inverse
 587    real(8), optional, intent(in) :: threshold_opt 
 588    !Locals:
 589    real(8), allocatable :: copyMatrix(:,:), leftSingularVector(:,:), rightSingularVectorT(:,:)
 590    real(8), allocatable :: singularValues(:)
 591    integer :: info, lwork, lineIndex
 592    integer :: lineDim, columnDim, minDim
 593    real(8) :: thresh
 594    real(8), allocatable :: work(:)
 595    character(len=80) :: errorMessage
 596
 597    lineDim = size(inputMatrix, dim=1)
 598    columnDim = size(inputMatrix, dim=2)
 599    minDim = min(lineDim, columnDim)
 600
 601    allocate( copyMatrix(lineDim,columnDim), leftSingularVector(lineDim,lineDim) )
 602    allocate( rightSingularVectorT(columnDim,columnDim), singularValues(minDim) )
 603
 604    copyMatrix(:,:) = inputMatrix(:,:) ! Work with a copy because copyMatrix will be overwriten
 605    lwork = max(10000, max(1, 3 * min(lineDim,columnDim) + max(lineDim,columnDim), 5 * minDim ))
 606    allocate(work(lwork))
 607    call dgesvd("A", "A", lineDim, columnDim, copyMatrix, lineDim, singularValues, &
 608         leftSingularVector, lineDim, rightSingularVectorT, columnDim, work, lwork, info ) 
 609   
 610    if (info /= 0) then
 611      write(errorMessage,*) "utl_pseudo_inverse: Problem in DGESVD ! ",info
 612      call utl_abort(errorMessage)
 613    end if
 614
 615    deallocate(work)
 616
 617    if (present(threshold_opt)) then
 618      thresh = threshold_opt
 619    else
 620      !according to wikipedia... as in matlab or numpy
 621      thresh = epsilon(thresh) * max(lineDim, columnDim) * maxval(singularValues)
 622    end if
 623    print *,"utl_pseudo_inverse: threshold= ",thresh
 624
 625    pseudoInverse(:,:)=0.d0
 626    do lineIndex = 1, minDim
 627      If (singularValues(lineIndex) > thresh) then

1234    !
1235    !           Currently assumes lat/long (or Gaussian) type grids.
1236    !           See hco_SetupFromFile for example toward future generalizations.
1237    !           Generalization would require having xlat and xlong being 2D.
1238    !
1239    !:Arguments:
1240    !                 :fname: input filename
1241    !                 :varName:  search nomvar
1242    !                 :iip1: search ip1
1243    !                 :iip2: search ip2
1244    !                 :iip3: search ip3
1245    !                 :etiketi: search etiket
1246    !                 :ni: ni values
1247    !                 :nj: nj values
1248    !                 :nkeys: number of records satisfying search criteria
1249    !                 :array: data arrray
1250    !                 :xlat_opt: 1D latitude array (optional)
1251    !                 :xlong_opt: 1D longitude array (optional)
1252    !                 :lvls_opt: 1D vertical coordinate array (optional)
1253    !                 :kind_opt: vertical coordinate type according to convip (optional)
1254    !
1255    implicit none
1256
1257    integer, intent(in) :: iip1,iip2,iip3
1258    character(len=*), intent(in) :: varName,fname,etiketi
1259    integer, intent(out) :: ni, nj, nkeys
1260    integer, intent(out), optional :: kind_opt
1261    real(8), intent(out), allocatable :: array(:,:,:)
1262    real(8), intent(out), allocatable, optional :: lvls_opt(:), xlat_opt(:),xlong_opt(:)

1265
1266    real(4) :: lvl_r4
1267
1268    logical :: Exists
1269    character(len=1) :: string
1270

1273
1274    integer, parameter :: maxkeys=1000

1276    integer :: dateo, deet, npas, nbits, datyp
1277    integer :: ip1, ip2, ip3, swa, lng, dltf, ubc
1278    integer :: extra1, extra2, extra3
1279    integer :: ig1, ig2, ig3, ig4  
1280    character*1 clgrtyp
1281    character*2 cltypvar
1282    character*4 nomvar
1283    character*12 cletiket
1284    real(4), allocatable :: buffer(:,:)
1285

1294    else

1296      ier=fstouv(iun,'RND+OLD')
1297    end if
1298
1299    ! Find reports in file for specified varName and iip*.
1300
1301    ier = fstinl(iun,ni,nj,nk,-1,etiketi,iip1,iip2,iip3,'',varName,keys,nkeys,maxkeys) 
1302
1303    if(ier.lt.0.or.nkeys.eq.0) then
1304      write(*,*) 'Search field missing ',varName, ' from file ',fname
1305      call utl_abort('utl_read_fst_field: did not find field.')
1306    else if (nk.gt.1) then
1307      write(*,*) 'Unexpected size nk ',nk,' for ',varName,' of file ',fname 
1308      call utl_abort('utl_read_fst_field')      
1309    end if
1310
1311    if (present(xlat_opt).and.present(xlong_opt)) then
1312
1313       !  Get lat and long if available.
1314
1315       if (allocated(xlat_opt)) deallocate(xlat_opt,xlong_opt)
1316       allocate(xlat_opt(nj),xlong_opt(ni),buffer(ni*nj,1))
1317       xlat_opt(:)=-999.
1318       xlong_opt(:)=-999.
1319
1320       ier = fstprm(keys(1),dateo, deet, npas, ni, nj, nk, nbits,    &         
1321                    datyp, ip1, ip2, ip3, cltypvar, nomvar, cletiket, &
1322                    clgrtyp, ig1, ig2, ig3,                           &
1323                    ig4, swa, lng, dltf, ubc, extra1, extra2, extra3)  
1324

1327          if (ier.ge.0) xlong_opt(:)=buffer(1:ni,1) 
1328       end if

1330          ier=fstlir(buffer,iun,ini,nj,nk,-1,'',ig1,ig2,ig3,'','^^')
1331          if (ier.ge.0) xlat_opt(:)=buffer(1:nj,1)     
1332       end if
1333       deallocate(buffer)
1334
1335       if ( trim(clgrtyp) == 'Z' ) then

1338

1341                  datyp, ip1, ip2, ip3, cltypvar, nomvar, cletiket, & ! OUT
1342                  clgrtyp, ig1, ig2, ig3,                           & ! OUT
1343                  ig4, swa, lng, dltf, ubc, extra1, extra2, extra3 ) ! OUT
1344
1345          call cigaxg (clgrtyp,                                     & ! IN
1346                     xlat1_4, xlon1_4, xlat2_4, xlon2_4,            & ! OUT
1347                     ig1, ig2, ig3, ig4 ) ! IN

1351          
1352       else if (trim(clgrtyp) /= 'G') then
1353
1354         call utl_abort('utl_readFstField: Cannot currently handle grid type ' // trim(clgrtyp) )
1355
1356       end if
1357       
1358    end if 
1359    
1360    ! Get vertical coordinate
1361    
1362    if (present(lvls_opt)) then
1363       if (allocated(lvls_opt)) deallocate(lvls_opt)
1364       allocate(lvls_opt(nkeys))
1365

1368                    datyp, ip1, ip2, ip3, cltypvar, nomvar, cletiket, &
1369                    clgrtyp, ig1, ig2, ig3,                           &
1370                    ig4, swa, lng, dltf, ubc, extra1, extra2, extra3)  
1371          call convip(ip1,lvl_r4,kindi,-1,string,.false.)
1372          lvls_opt(i)=lvl_r4
1373       end do
1374    end if
1375     
1376    if (present(kind_opt)) then
1377        if (present(lvls_opt)) then         
1378            kind_opt=kindi
1379        else
1380            kind_opt=-1
1381        end if
1382    end if
1383
1384    ! Get field
1385    
1386    allocate(array(ni,nj,nkeys),buffer(ni,nj))
1387
1388    do i=1,nkeys
1389       ier=fstluk(buffer,keys(i),ni,nj,nk)
1390       array(:,:,i)=buffer(:,:)
1391    end do
1392    
1393    deallocate(buffer)
1394    
1395    ier=fstfrm(iun)  
1396    ier=fclos(iun)  
1397
1398  end subroutine utl_readFstField

1832  subroutine utl_heapsort2d(array)
1833    !
1834    !:Purpose: Sort a real 2D array in ascending order according
1835    !          to the first column
1836    ! 
1837    implicit none
1838    real(4), intent(inout) :: array(:,:)
1839
1840    real(4) :: values(2) ! temporary value
1841    integer :: i,j,nsize
1842    integer :: ileft,iright
1843
1844    nsize = size(array,1)
1845    ileft=nsize/2+1
1846    iright=nsize
1847
1848    if (nsize == 1) return                  
1849
1850    do 
1851      if(ileft > 1)then
1852        ileft=ileft-1
1853        values(:) = array(ileft,:)
1854      else
1855        values(:) = array(iright,:)
1856        array(iright,:) = array(1,:)
1857        iright = iright-1
1858        if (iright == 1) then
1859          array(1,:) = values(:)
1860          return
1861        end if
1862      end if
1863      i = ileft
1864      j = 2*ileft
1865      do while (j <= iright) 
1866        if (j < iright) then
1867          if (array(j,1) < array(j+1,1)) j=j+1
1868        endif
1869        if (values(1) < array(j,1)) then
1870          array(i,:) = array(j,:)
1871          i = j
1872          j = j+j
1873        else
1874          j = iright+1
1875        end if
1876      end do
1877      array(i,:) = values(:)
1878    end do
1879
1880  end subroutine utl_heapsort2d
