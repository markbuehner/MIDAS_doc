323  integer function codtyp_get_codtyp(name)
324    !
325    !:Purpose: Given a family name, return the codtyp
326    !
327    !          NEW information from namelist NAMCODTYP
328    !
329    implicit none
330    character (len=*),intent(in) :: name
331
332    integer :: i, ilen
333    character (len=codtyp_name_length) :: ctempo
334
335    if (.not.initialized) call codtyp_initialize()
336
337    ! find the codtype based on the name
338    ctempo(:) = ' '
339    codtyp_get_codtyp = -1
340    ilen = len_trim(name)
341    call up2low(name(1:ilen),ctempo(1:ilen))
342    do i=1, ncodtyp
343       if ( trim(ctempo) == trim(cnames(i)) ) then
344          codtyp_get_codtyp = icod(i)
345          exit
346       end if
347    end do
348    
349  end function codtyp_get_codtyp

351  character(len=codtyp_name_length) function codtyp_get_name(codtyp)
352    !
353    !:Purpose: Given a codtyp, return the family name
354    !
355    !          NEW information from namelist NAMCODTYP
356    !
357    implicit none
358    integer,intent(in) :: codtyp
359    
360    integer :: i
361    
362    if (.not.initialized) call codtyp_initialize()
363
364    do i=1, ncodtyp
365       if ( icod(i) == codtyp ) then
366          codtyp_get_name = trim(cnames(i))
367          exit
368       end if
369    end do
370    
371  end function codtyp_get_name
