  91    if ( nmlAlreadyRead ) return

 148  subroutine qlim_saturationLimit_gsv(statevector)
 149    !
 150    !:Purpose: To impose saturation limit on humidity variable of a statevector
 151    !
 152    implicit none
 153
 154    ! Arguments:
 155    type(struct_gsv) :: statevector
 156
 157    ! Locals:
 158    type(struct_vco), pointer :: vco_ptr
 159    real(4), pointer :: hu_ptr_r4(:,:,:,:), tt_ptr_r4(:,:,:,:), psfc_ptr_r4(:,:,:,:), psfcLS_ptr_r4(:,:,:,:)
 160    real(8), pointer :: hu_ptr_r8(:,:,:,:), tt_ptr_r8(:,:,:,:), psfc_ptr_r8(:,:,:,:), psfcLS_ptr_r8(:,:,:,:)
 161    real(8), pointer :: pressure(:,:,:)
 162    real(8)          :: hu, husat, hu_modified, tt
 163    integer          :: lon1, lon2, lat1, lat2, lev1, lev2
 164    real(8), allocatable :: psfc(:,:), psfcLS(:,:)
 165    integer          :: lonIndex, latIndex, levIndex, stepIndex
 166
 167    if (mmpi_myid == 0) write(*,*) 'qlim_saturationLimit_gsv: STARTING'
 168
 169    if( .not. gsv_varExist(statevector,'HU') ) then
 170      if( mmpi_myid == 0 ) write(*,*) 'qlim_saturationLimit_gsv: statevector does not ' // &
 171           'contain humidity ... doing nothing'
 172      return
 173    end if
 174
 175    vco_ptr => gsv_getVco(statevector)
 176    if (stateVector%dataKind == 8) then
 177      call gsv_getField(statevector,hu_ptr_r8,'HU')
 178      call gsv_getField(statevector,tt_ptr_r8,'TT')
 179    else
 180      call gsv_getField(statevector,hu_ptr_r4,'HU')
 181      call gsv_getField(statevector,tt_ptr_r4,'TT')
 182    end if
 183
 184    lon1 = statevector%myLonBeg
 185    lon2 = statevector%myLonEnd
 186    lat1 = statevector%myLatBeg
 187    lat2 = statevector%myLatEnd
 188    lev1 = 1
 189    lev2 = gsv_getNumLev(statevector,'TH')
 190
 191    allocate(psfc(lon2-lon1+1,lat2-lat1+1))
 192    if (vco_ptr%vcode == 5100) allocate(psfcLS(lon2-lon1+1,lat2-lat1+1))
 193    do stepIndex = 1, statevector%numStep
 194      if (stateVector%dataKind == 8) then
 195        call gsv_getField(statevector,psfc_ptr_r8,'P0')
 196        psfc(:,:) = psfc_ptr_r8(:,:,1,stepIndex)
 197        if (vco_ptr%vcode == 5100) then
 198          call gsv_getField(statevector,psfcLS_ptr_r8,'P0LS')
 199          psfcLS(:,:) = psfcLS_ptr_r8(:,:,1,stepIndex)
 200        end if
 201      else
 202        call gsv_getField(statevector,psfc_ptr_r4,'P0')
 203        psfc(:,:) = psfc_ptr_r4(:,:,1,stepIndex)
 204        if (vco_ptr%vcode == 5100) then
 205          call gsv_getField(statevector,psfcLS_ptr_r4,'P0LS')
 206          psfcLS(:,:) = psfcLS_ptr_r4(:,:,1,stepIndex)
 207        end if
 208      end if
 209      if (vco_ptr%vcode == 5100) then
 210        call czp_fetch3DLevels(vco_ptr, psfc, sfcFldLS_opt=psfcLS, fldT_opt=pressure)
 211      else
 212        call czp_fetch3DLevels(vco_ptr, psfc, fldT_opt=pressure)
 213      end if
 214      if (stateVector%dataKind == 8) then
 215        !$OMP PARALLEL DO PRIVATE (levIndex, latIndex, lonIndex, hu, tt, husat, hu_modified)
 216        do levIndex = lev1, lev2
 217          do latIndex = lat1, lat2
 218            do lonIndex = lon1, lon2
 219              hu = hu_ptr_r8(lonIndex,latIndex,levIndex,stepIndex)
 220              tt = tt_ptr_r8(lonIndex,latIndex,levIndex,stepIndex)
 221
 222              ! get the saturated vapor pressure from HU
 223              husat = phf_foqst8(tt, pressure(lonIndex-lon1+1,latIndex-lat1+1,levIndex) )
 224
 225              ! limit the humidity to the saturated humidity
 226              hu_modified = min(husat, hu)
 227              hu_ptr_r8(lonIndex,latIndex,levIndex,stepIndex) = hu_modified
 228
 229            end do ! lonIndex
 230          end do ! latIndex
 231        end do ! levIndex
 232        !$OMP END PARALLEL DO
 233      else
 234        !$OMP PARALLEL DO PRIVATE (levIndex, latIndex, lonIndex, hu, tt, husat, hu_modified)
 235        do levIndex = lev1, lev2
 236          do latIndex = lat1, lat2
 237            do lonIndex = lon1, lon2
 238              hu = hu_ptr_r4(lonIndex,latIndex,levIndex,stepIndex)
 239              tt = tt_ptr_r4(lonIndex,latIndex,levIndex,stepIndex)
 240
 241              ! get the saturated vapor pressure from HU
 242              husat = phf_foqst8(tt, pressure(lonIndex-lon1+1,latIndex-lat1+1,levIndex) )
 243
 244              ! limit the humidity to the saturated humidity
 245              hu_modified = min(husat, hu)
 246              hu_ptr_r4(lonIndex,latIndex,levIndex,stepIndex) = hu_modified
 247
 248            end do ! lonIndex
 249          end do ! latIndex
 250        end do ! levIndex
 251        !$OMP END PARALLEL DO
 252      end if
 253
 254      deallocate(pressure)
 255
 256    end do ! stepIndex
 257
 258    deallocate(psfc)
 259    if (allocated(psfcLS)) deallocate(psfcLS)
 260
 261  end subroutine qlim_saturationLimit_gsv

 264  ! qlim_saturationLimit_ens
 265  !--------------------------------------------------------------------------
 266  subroutine qlim_saturationLimit_ens(ensemble)
 267    !
 268    !:Purpose: To impose saturation limit on humidity variable of an ensemble
 269    !
 270    implicit none
 271
 272    ! Arguments:
 273    type(struct_ens) :: ensemble
 274
 275    ! Locals:
 276    type(struct_vco), pointer :: vco_ptr
 277    real(4), pointer :: hu_ptr_r4(:,:,:,:), tt_ptr_r4(:,:,:,:), psfc_ptr_r4(:,:,:,:), psfcLS_ptr_r4(:,:,:,:)
 278    real(8), pointer :: pressure(:,:,:)
 279    real(8)          :: hu, husat, hu_modified, tt
 280    integer          :: lon1, lon2, lat1, lat2, numLev
 281    real(8), allocatable :: psfc(:,:),psfcLS(:,:)
 282    integer          :: lonIndex, latIndex, levIndex, stepIndex, memberIndex, varLevIndex
 283
 284    if (mmpi_myid == 0) write(*,*) 'qlim_saturationLimit_ens: STARTING'
 285
 286    if (ens_getDataKind(ensemble) == 8) then
 287      call utl_abort('qlim_saturationLimit_ens: Not compatible with dataKind = 8')
 288    end if
 289
 290    if( .not. ens_varExist(ensemble,'HU') ) then
 291      if( mmpi_myid == 0 ) write(*,*) 'qlim_saturationLimit_ens: ensemble does not ' // &
 292           'contain humidity ... doing nothing'
 293      return
 294    end if
 295
 296    vco_ptr => ens_getVco(ensemble)
 297    numLev = ens_getNumLev(ensemble,'TH')
 298    call ens_getLatLonBounds(ensemble, lon1, lon2, lat1, lat2)
 299    allocate(psfc(ens_getNumMembers(ensemble),ens_getNumStep(ensemble)))
 300    if (vco_ptr%vcode == 5100) allocate(psfcLS(ens_getNumMembers(ensemble),ens_getNumStep(ensemble)))
 301
 302    do latIndex = lat1, lat2
 303      do lonIndex = lon1, lon2
 304
 305        ! compute pressure for all members and steps
 306        varLevIndex = ens_getKFromLevVarName(ensemble, 1, 'P0')
 307        psfc_ptr_r4 => ens_getOneLev_r4(ensemble,varLevIndex)
 308        psfc(:,:) = psfc_ptr_r4(:,:,lonIndex,latIndex)
 309        if (vco_ptr%vcode == 5100) then
 310          varLevIndex = ens_getKFromLevVarName(ensemble, 1, 'P0LS')
 311          psfcLS_ptr_r4 => ens_getOneLev_r4(ensemble,varLevIndex)
 312          psfcLS(:,:) = psfcLS_ptr_r4(:,:,lonIndex,latIndex)
 313          call czp_fetch3DLevels(vco_ptr, psfc, sfcFldLS_opt=psfcLS, fldT_opt=pressure)
 314        else
 315          call czp_fetch3DLevels(vco_ptr, psfc, fldT_opt=pressure)
 316        end if
 317
 318        do levIndex = 1, numLev
 319          varLevIndex = ens_getKFromLevVarName(ensemble, levIndex, 'HU')
 320          hu_ptr_r4 => ens_getOneLev_r4(ensemble,varLevIndex)
 321          varLevIndex = ens_getKFromLevVarName(ensemble, levIndex, 'TT')
 322          tt_ptr_r4 => ens_getOneLev_r4(ensemble,varLevIndex)
 323
 324          !$OMP PARALLEL DO PRIVATE (stepIndex, memberIndex, hu, tt, husat, hu_modified)
 325          do stepIndex = 1, ens_getNumStep(ensemble)
 326            do memberIndex = 1, ens_getNumMembers(ensemble)
 327              hu = hu_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex)
 328              tt = tt_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex)
 329
 330              ! get the saturated vapor pressure from HU
 331              husat = phf_foqst8(tt, pressure(memberIndex,stepIndex,levIndex) )
 332
 333              ! limit the humidity to the saturated humidity
 334              hu_modified = min(husat, hu)
 335              hu_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex) = hu_modified
 336
 337            end do ! memberIndex
 338          end do ! stepIndex
 339          !$OMP END PARALLEL DO
 340
 341        end do ! levIndex
 342        deallocate(pressure)
 343
 344      end do ! lonIndex
 345    end do ! latIndex
 346
 347    deallocate(psfc)
 348    if (allocated(psfcLS)) deallocate(psfcLS)
 349
 350  end subroutine qlim_saturationLimit_ens

 355  subroutine qlim_rttovLimit_gsv(statevector, varName_opt, applyLimitToCloud_opt)
 356    !
 357    !:Purpose: To impose RTTOV limits on humidity/cloud
 358    !
 359    implicit none
 360
 361    ! Arguments:
 362    type(struct_gsv),        intent(inout) :: statevector
 363    character(len=*), optional, intent(in) :: varName_opt
 364    logical,          optional, intent(in) :: applyLimitToCloud_opt
 365
 366    ! Locals:
 367    type(struct_vco), pointer :: vco_ptr
 368    real(8), allocatable :: press_rttov(:), qmin_rttov(:), qmax_rttov(:)
 369    real(8), allocatable :: psfc(:,:),psfcLS(:,:)
 370    real(8), allocatable :: qmin3D_rttov(:,:,:), qmax3D_rttov(:,:,:)
 371    real(8), pointer     :: hu_ptr_r8(:,:,:,:), psfc_ptr_r8(:,:,:,:), psfcLS_ptr_r8(:,:,:,:)
 372    real(8), pointer     :: cld_ptr_r8(:,:,:,:)
 373    real(4), pointer     :: hu_ptr_r4(:,:,:,:), psfc_ptr_r4(:,:,:,:), psfcLS_ptr_r4(:,:,:,:)
 374    real(4), pointer     :: cld_ptr_r4(:,:,:,:)
 375    real(8), pointer     :: pressure(:,:,:)
 376    real(8)              :: hu, hu_modified
 377    real(8)              :: cld, cld_modified
 378    real(8)              :: minValueCld, maxValueCld
 379    integer              :: lon1, lon2, lat1, lat2, lev1, lev2, varNameIndex
 380    integer              :: lonIndex, latIndex, levIndex, stepIndex
 381    integer              :: ni, nj, numLev, numLev_rttov
 382    integer              :: fnom, fclos, ierr, nulfile
 383    character(len=256)   :: fileName
 384    character(len=4)     :: varName
 385    logical, save        :: firstTime=.true.
 386    logical              :: applyLimitToAllVarname
 387    logical              :: applyLimitToCloud
 388
 389    if (mmpi_myid == 0) write(*,*) 'qlim_rttovLimit_gsv: STARTING'
 390
 391    if (present(varName_opt)) then
 392      applyLimitToAllVarname = .false.
 393      varName = varName_opt
 394    else
 395      applyLimitToAllVarname = .true.
 396      varName = 'XXXX'  
 397    end if
 398
 399    ! for cloud, limits are applied to ALL cloud variables.
 400    if (present(applyLimitToCloud_opt)) then
 401      applyLimitToCloud = applyLimitToCloud_opt
 402      if (.not. applyLimitToCloud) call utl_abort('qlim_rttovLimit_gsv: remove applyLimitToCloud_opt argument')
 403    else
 404      if (vnl_isCloudVar(varName)) then
 405        if (mmpi_myid == 0) write(*,*) 'qlim_rttovLimit_gsv: limits are applied to ALL cloud variables'
 406        applyLimitToCloud = .true.
 407      else
 408        applyLimitToCloud = .false.
 409      end if
 410    end if
 411
 412    if (applyLimitToCloud) applyLimitToAllVarname = .false.
 413
 414    if ((applyLimitToAllVarname .or. trim(varName) == 'HU') .and. &
 415         gsv_varExist(statevector,'HU')) then
 416
 417      if (mmpi_myid == 0) write(*,*) 'qlim_rttovLimit_gsv: applying limits to HU.'
 418
 419      ! Read in RTTOV humidity limits
 420      fileName = "rttov_h2o_limits.dat"
 421      nulfile = 0
 422      ierr = fnom(nulfile, fileName, "FMT+OLD+R/O", 0)
 423      if( ierr /= 0 ) then
 424        if ( mmpi_myid == 0 ) write(*,*) 'fileName = ', fileName
 425        call utl_abort('qlim_rttovLimit_gsv: error opening the humidity limits file')
 426      end if
 427
 428      read(nulfile,*) numLev_rttov
 429      if ( mmpi_myid == 0 .and. firstTime ) write(*,*) 'qlim_rttovLimit_gsv: rttov number of levels = ', numLev_rttov
 430      allocate(press_rttov(numLev_rttov))
 431      allocate(qmin_rttov(numLev_rttov))
 432      allocate(qmax_rttov(numLev_rttov))
 433      do levIndex = 1, numLev_rttov
 434        read(nulfile,*) press_rttov(levIndex), qmax_rttov(levIndex), qmin_rttov(levIndex)
 435      end do
 436      ierr = fclos(nulfile)
 437      press_rttov(:) = press_rttov(:) * mpc_pa_per_mbar_r8
 438      qmin_rttov(:) = qmin_rttov(:) / mixratio_to_ppmv
 439      qmax_rttov(:) = qmax_rttov(:) / mixratio_to_ppmv
 440
 441      if (firstTime) then
 442        write(*,*) ' '
 443        do levIndex = 1, numLev_rttov
 444          if ( mmpi_myid == 0 ) write(*,fmt='(" qlim_rttovLimit_gsv:   LEVEL = ",I4,", PRES = ",F9.0,", HUMIN = ",E10.2,", HUMAX = ",E10.2)') &
 445              levIndex, press_rttov(levIndex), qmin_rttov(levIndex), qmax_rttov(levIndex)
 446        end do
 447        firstTime = .false.
 448      end if
 449
 450      vco_ptr => gsv_getVco(statevector)
 451      if (statevector%dataKind == 8) then
 452        call gsv_getField(statevector,hu_ptr_r8,'HU')
 453      else
 454        call gsv_getField(statevector,hu_ptr_r4,'HU')
 455      end if
 456
 457      lon1 = statevector%myLonBeg
 458      lon2 = statevector%myLonEnd
 459      lat1 = statevector%myLatBeg
 460      lat2 = statevector%myLatEnd
 461      lev1 = 1
 462      lev2 = gsv_getNumLev(statevector,'TH')
 463
 464      ni = lon2 - lon1 + 1
 465      nj = lat2 - lat1 + 1
 466      numLev = lev2 - lev1 + 1
 467      allocate( qmin3D_rttov(ni,nj,numLev) )
 468      allocate( qmax3D_rttov(ni,nj,numLev) )
 469      allocate( psfc(lon2-lon1+1,lat2-lat1+1) )
 470      if (vco_ptr%vcode == 5100) allocate( psfcLS(lon2-lon1+1,lat2-lat1+1) )
 471
 472      do stepIndex = 1, statevector%numStep
 473        if (statevector%dataKind == 8) then
 474          call gsv_getField(statevector,psfc_ptr_r8,'P0')
 475          psfc(:,:) = psfc_ptr_r8(:,:,1,stepIndex)
 476          if (vco_ptr%vcode == 5100) then
 477            call gsv_getField(statevector,psfcLS_ptr_r8,'P0')
 478            psfcLS(:,:) = psfcLS_ptr_r8(:,:,1,stepIndex)
 479          end if
 480        else
 481          call gsv_getField(statevector,psfc_ptr_r4,'P0')
 482          psfc(:,:) = real(psfc_ptr_r4(:,:,1,stepIndex),8)
 483          if (vco_ptr%vcode == 5100) then
 484            call gsv_getField(statevector,psfcLS_ptr_r4,'P0')
 485            psfcLS(:,:) = psfcLS_ptr_r4(:,:,1,stepIndex)
 486          end if
 487        end if
 488        if (vco_ptr%vcode == 5100) then
 489          call czp_fetch3DLevels(vco_ptr, psfc, sfcFldLS_opt=psfcLS, fldT_opt=pressure)
 490        else
 491          call czp_fetch3DLevels(vco_ptr, psfc, fldT_opt=pressure)
 492        end if
 493
 494        ! Interpolate RTTOV limits onto model levels
 495        call qlim_lintv_minmax(press_rttov, qmin_rttov, qmax_rttov, numLev_rttov, &
 496            ni, nj, numLev, pressure, qmin3D_rttov, qmax3D_rttov)
 497
 498        !$OMP PARALLEL DO PRIVATE (levIndex, latIndex, lonIndex, hu, hu_modified)
 499        do levIndex = lev1, lev2
 500          do latIndex = lat1, lat2
 501            do lonIndex = lon1, lon2
 502              if (statevector%dataKind == 8) then
 503                hu = hu_ptr_r8(lonIndex,latIndex,levIndex,stepIndex)
 504              else
 505                hu = real(hu_ptr_r4(lonIndex,latIndex,levIndex,stepIndex),8)
 506              end if
 507
 508              ! limit the humidity according to the rttov limits
 509              hu_modified = max(hu, qmin3D_rttov(lonIndex - lon1 + 1, latIndex - lat1 + 1, levIndex) )
 510              hu_modified = min(hu_modified, qmax3D_rttov(lonIndex - lon1 + 1, latIndex - lat1 + 1, levIndex) )
 511              if (statevector%dataKind == 8) then
 512                hu_ptr_r8(lonIndex,latIndex,levIndex,stepIndex) = hu_modified
 513              else
 514                hu_ptr_r4(lonIndex,latIndex,levIndex,stepIndex) = real(hu_modified,4)
 515              end if
 516
 517            end do ! lonIndex
 518          end do ! latIndex
 519        end do ! levIndex
 520        !$OMP END PARALLEL DO
 521
 522        deallocate(pressure)
 523      end do ! stepIndex

 525      deallocate( psfc )

 527      deallocate( qmin3D_rttov )
 528      deallocate( qmax3D_rttov )
 529
 530      deallocate(qmax_rttov)
 531      deallocate(qmin_rttov)
 532      deallocate(press_rttov)
 533
 534    end if
 535
 536    ! apply limits to ALL available cloud variables
 537    if ((applyLimitToAllVarname .or. applyLimitToCloud) .and. cloudExistInStateVector(statevector)) then
 538
 539      do varNameIndex = 1, vnl_numvarmaxCloud
 540        if (.not. gsv_varExist(statevector, vnl_varNameListCloud(varNameIndex))) cycle
 541
 542        if (mmpi_myid == 0) write(*,*) 'qlim_rttovLimit_gsv: applying limits to ', &
 543                                        vnl_varNameListCloud(varNameIndex)
 544
 545        if (statevector%dataKind == 8) then
 546          call gsv_getField(statevector, cld_ptr_r8, vnl_varNameListCloud(varNameIndex))
 547        else
 548          call gsv_getField(statevector, cld_ptr_r4, vnl_varNameListCloud(varNameIndex))
 549        end if
 550
 551        minValueCld = qlim_getMinValueCloud(vnl_varNameListCloud(varNameIndex))
 552        maxValueCld = qlim_getMaxValueCloud(vnl_varNameListCloud(varNameIndex))
 553
 554        lon1 = statevector%myLonBeg
 555        lon2 = statevector%myLonEnd
 556        lat1 = statevector%myLatBeg
 557        lat2 = statevector%myLatEnd
 558        lev1 = 1
 559        lev2 = gsv_getNumLev(statevector,'TH')
 560
 561        do stepIndex = 1, statevector%numStep
 562
 563          !$OMP PARALLEL DO PRIVATE (levIndex, latIndex, lonIndex, cld, cld_modified)
 564          do levIndex = lev1, lev2
 565            do latIndex = lat1, lat2
 566              do lonIndex = lon1, lon2
 567                if (statevector%dataKind == 8) then
 568                  cld = cld_ptr_r8(lonIndex,latIndex,levIndex,stepIndex)
 569                else
 570                  cld = real(cld_ptr_r4(lonIndex,latIndex,levIndex,stepIndex),8)
 571                end if

 574                cld_modified = min(cld_modified,maxValueCld)
 575                if (statevector%dataKind == 8) then
 576                  cld_ptr_r8(lonIndex,latIndex,levIndex,stepIndex) = cld_modified
 577                else
 578                  cld_ptr_r4(lonIndex,latIndex,levIndex,stepIndex) = real(cld_modified,4)
 579                end if
 580
 581              end do ! lonIndex
 582            end do ! latIndex
 583          end do ! levIndex
 584          !$OMP END PARALLEL DO
 585
 586        end do ! stepIndex
 587      end do ! varNameIndex

 589    end if

 593  !--------------------------------------------------------------------------
 594  ! qlim_rttovLimit_ens
 595  !--------------------------------------------------------------------------
 596  subroutine qlim_rttovLimit_ens(ensemble, varName_opt, applyLimitToCloud_opt)
 597    !
 598    !:Purpose: To impose RTTOV limits on humidity/cloud
 599    !
 600    implicit none
 601
 602    ! Arguments:
 603    type(struct_ens),        intent(inout) :: ensemble
 604    character(len=*), optional, intent(in) :: varName_opt
 605    logical,          optional, intent(in) :: applyLimitToCloud_opt
 606
 607    ! Locals:
 608    type(struct_vco), pointer :: vco_ptr
 609    real(8), allocatable :: press_rttov(:), qmin_rttov(:), qmax_rttov(:)
 610    real(8), allocatable :: psfc(:,:),psfcLS(:,:)
 611    real(8), allocatable :: qmin3D_rttov(:,:,:), qmax3D_rttov(:,:,:)
 612    real(4), pointer     :: hu_ptr_r4(:,:,:,:), psfc_ptr_r4(:,:,:,:), psfcLS_ptr_r4(:,:,:,:)
 613    real(4), pointer     :: cld_ptr_r4(:,:,:,:)
 614    real(8), pointer     :: pressure(:,:,:)
 615    real(8)              :: hu, hu_modified
 616    real(8)              :: cld, cld_modified
 617    real(8)              :: minValueCld, maxValueCld
 618    integer              :: lon1, lon2, lat1, lat2, varNameIndex
 619    integer              :: lonIndex, latIndex, levIndex, stepIndex, varLevIndex, memberIndex
 620    integer              :: numMember, numStep, numLev, numLev_rttov
 621    integer              :: fnom, fclos, ierr, nulfile
 622    character(len=256)   :: fileName
 623    character(len=4)     :: varName
 624    logical, save        :: firstTime=.true.
 625    logical              :: applyLimitToAllVarname
 626    logical              :: applyLimitToCloud
 627
 628    if (mmpi_myid == 0) write(*,*) 'qlim_rttovLimit_ens: STARTING'
 629
 630    if (present(varName_opt)) then
 631      applyLimitToAllVarname = .false.
 632      varName = varName_opt
 633    else
 634      applyLimitToAllVarname = .true.
 635      varName = 'XXXX'    
 636    end if
 637
 638    ! for cloud, limits are applied to ALL cloud variables.
 639    if (present(applyLimitToCloud_opt)) then
 640      applyLimitToCloud = applyLimitToCloud_opt
 641      if (.not. applyLimitToCloud) call utl_abort('qlim_rttovLimit_ens: remove applyLimitToCloud_opt argument')
 642    else
 643      if (vnl_isCloudVar(varName)) then
 644        if (mmpi_myid == 0) write(*,*) 'qlim_rttovLimit_ens: limits are applied to ALL cloud variables'
 645        applyLimitToCloud = .true.
 646      else
 647        applyLimitToCloud = .false.
 648      end if
 649    end if
 650
 651    if (applyLimitToCloud) applyLimitToAllVarname = .false.
 652
 653    if ((applyLimitToAllVarname .or. trim(varName) == 'HU') .and. &
 654         ens_varExist(ensemble,'HU')) then
 655
 656      if ( mmpi_myid == 0 ) write(*,*) 'qlim_rttovLimit_ens:  applying limits to HU.'
 657
 658      if (ens_getDataKind(ensemble) == 8) then
 659        call utl_abort('qlim_rttovLimit_ens: Not compatible with dataKind = 8')
 660      end if
 661
 662      ! Read in RTTOV humidity limits
 663      fileName = "rttov_h2o_limits.dat"
 664      nulfile = 0
 665      ierr = fnom(nulfile, fileName, "FMT+OLD+R/O", 0)
 666      if( ierr /= 0 ) then
 667        if ( mmpi_myid == 0 ) write(*,*) 'fileName = ', fileName
 668        call utl_abort('qlim_rttovLimit_ens: error opening the humidity limits file')
 669      end if
 670
 671      read(nulfile,*) numLev_rttov
 672      if ( mmpi_myid == 0 .and. firstTime ) write(*,*) 'qlim_rttovLimit_ens: rttov number of levels = ', numLev_rttov
 673      allocate(press_rttov(numLev_rttov))
 674      allocate(qmin_rttov(numLev_rttov))
 675      allocate(qmax_rttov(numLev_rttov))
 676      do levIndex = 1, numLev_rttov
 677        read(nulfile,*) press_rttov(levIndex), qmax_rttov(levIndex), qmin_rttov(levIndex)
 678      end do
 679      ierr = fclos(nulfile)
 680      press_rttov(:) = press_rttov(:) * mpc_pa_per_mbar_r8
 681      qmin_rttov(:) = qmin_rttov(:) / mixratio_to_ppmv
 682      qmax_rttov(:) = qmax_rttov(:) / mixratio_to_ppmv
 683
 684      if (firstTime) then
 685        write(*,*) ' '
 686        do levIndex = 1, numLev_rttov
 687          if ( mmpi_myid == 0 ) write(*,fmt='(" qlim_rttovLimit_ens:   LEVEL = ",I4,", PRES = ",F9.0,", HUMIN = ",E10.2,", HUMAX = ",E10.2)') &
 688              levIndex, press_rttov(levIndex), qmin_rttov(levIndex), qmax_rttov(levIndex)
 689        end do
 690        firstTime = .false.
 691      end if
 692
 693      vco_ptr => ens_getVco(ensemble)
 694      numLev = ens_getNumLev(ensemble,'TH')
 695      numMember = ens_getNumMembers(ensemble)
 696      numStep = ens_getNumStep(ensemble)
 697      call ens_getLatLonBounds(ensemble, lon1, lon2, lat1, lat2)
 698
 699      allocate( psfc(numMember,numStep) )
 700      if (vco_ptr%vcode == 5100) allocate( psfcLS(numMember,numStep) )
 701      allocate( qmin3D_rttov(numMember,numStep,numLev) )
 702      allocate( qmax3D_rttov(numMember,numStep,numLev) )
 703
 704      do latIndex = lat1, lat2
 705        do lonIndex = lon1, lon2
 706
 707          varLevIndex = ens_getKFromLevVarName(ensemble, 1, 'P0')
 708          psfc_ptr_r4 => ens_getOneLev_r4(ensemble,varLevIndex)
 709          psfc(:,:) = real(psfc_ptr_r4(:,:,lonIndex,latIndex),8)
 710          if (vco_ptr%vcode == 5100) then
 711            varLevIndex = ens_getKFromLevVarName(ensemble, 1, 'P0LS')
 712            psfcLS_ptr_r4 => ens_getOneLev_r4(ensemble,varLevIndex)
 713            psfcLS(:,:) = real(psfcLS_ptr_r4(:,:,lonIndex,latIndex),8)
 714            call czp_fetch3DLevels(vco_ptr, psfc, sfcFldLS_opt=psfcLS, fldT_opt=pressure)
 715          else
 716            call czp_fetch3DLevels(vco_ptr, psfc, fldT_opt=pressure)
 717          end if
 718
 719          ! Interpolate RTTOV limits onto model levels
 720          call qlim_lintv_minmax(press_rttov, qmin_rttov, qmax_rttov, numLev_rttov, &
 721                                numMember, numStep, numLev, pressure,  &
 722                                qmin3D_rttov, qmax3D_rttov)
 723
 724          do levIndex = 1, numLev
 725
 726            varLevIndex = ens_getKFromLevVarName(ensemble, levIndex, 'HU')
 727            hu_ptr_r4 => ens_getOneLev_r4(ensemble,varLevIndex)
 728
 729            !$OMP PARALLEL DO PRIVATE (stepIndex, memberIndex, hu, hu_modified)
 730            do stepIndex = 1, numStep
 731              do memberIndex = 1, numMember
 732
 733                hu = real(hu_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex),8)
 734
 735                ! limit the humidity according to the rttov limits
 736                hu_modified = max(hu, qmin3D_rttov(memberIndex, stepIndex, levIndex) )
 737                hu_modified = min(hu_modified, qmax3D_rttov(memberIndex, stepIndex, levIndex) )
 738                hu_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex) = real(hu_modified,4)
 739
 740              end do ! memberIndex
 741            end do ! stepIndex
 742            !$OMP END PARALLEL DO
 743          end do ! levIndex
 744
 745          deallocate(pressure)
 746
 747        end do ! lonIndex
 748      end do ! latIndex
 749
 750      deallocate( psfc )
 751      if (allocated(psfcLS)) deallocate( psfcLS )
 752      deallocate( qmin3D_rttov )
 753      deallocate( qmax3D_rttov )
 754
 755      deallocate(qmax_rttov)
 756      deallocate(qmin_rttov)
 757      deallocate(press_rttov)
 758    end if
 759
 760    ! apply limits to ALL available cloud variables
 761    if ((applyLimitToAllVarname .or. applyLimitToCloud) .and. cloudExistInEnsemble(ensemble)) then
 762
 763      vco_ptr => ens_getVco(ensemble)
 764      numLev = ens_getNumLev(ensemble,'TH')
 765      numMember = ens_getNumMembers(ensemble)
 766      numStep = ens_getNumStep(ensemble)
 767      call ens_getLatLonBounds(ensemble, lon1, lon2, lat1, lat2)
 768
 769      do varNameIndex = 1, vnl_numvarmaxCloud
 770        if (.not. ens_varExist(ensemble, vnl_varNameListCloud(varNameIndex))) cycle
 771
 772        if (mmpi_myid == 0) write(*,*) 'qlim_rttovLimit_ens:  applying limits to ', &
 773                                        vnl_varNameListCloud(varNameIndex)
 774
 775        minValueCld = qlim_getMinValueCloud(vnl_varNameListCloud(varNameIndex))
 776        maxValueCld = qlim_getMaxValueCloud(vnl_varNameListCloud(varNameIndex))
 777
 778        do latIndex = lat1, lat2
 779          do lonIndex = lon1, lon2
 780
 781            do levIndex = 1, numLev
 782                varLevIndex = ens_getKFromLevVarName(ensemble, levIndex, vnl_varNameListCloud(varNameIndex))
 783                cld_ptr_r4 => ens_getOneLev_r4(ensemble,varLevIndex)
 784
 785                !$OMP PARALLEL DO PRIVATE (stepIndex, memberIndex, cld, cld_modified)
 786                do stepIndex = 1, numStep
 787                  do memberIndex = 1, numMember
 788
 789                      cld = real(cld_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex),8)
 790
 791                      cld_modified = max(cld,minValueCld)
 792                      cld_modified = min(cld_modified,maxValueCld)
 793                      cld_ptr_r4(memberIndex,stepIndex,lonIndex,latIndex) = real(cld_modified,4)
 794
 795                  end do ! memberIndex
 796                end do ! stepIndex
 797                !$OMP END PARALLEL DO
 798
 799            end do ! levIndex
 800
 801          end do ! lonIndex
 802        end do ! latIndex
 803      end do ! varNameIndex
 804
 805    end if
 806
 807  end subroutine qlim_rttovLimit_ens

 809  !--------------------------------------------------------------------------
 810  ! qlim_lintv_minmax
 811  !--------------------------------------------------------------------------
 812  subroutine qlim_lintv_minmax(press_src, qmin_src, qmax_src, numLev_src, &
 813                               ni_dest, nj_dest, numLev_dest, press_dest, &
 814                               qmin_dest, qmax_dest)
 815    !
 816    !:Purpose: To perform the vertical interpolation in log of pressure and
 817    !          and constant value extrapolation of one-dimensional vectors.
 818    implicit none
 819
 820    ! Arguments:
 821    real(8), intent(in) :: press_src(numLev_src) ! Vertical levels, pressure (source)
 822    real(8), intent(in) :: qmin_src(numLev_src)  ! Vectors to be interpolated (source)
 823    real(8), intent(in) :: qmax_src(numLev_src)  ! Vectors to be interpolated (source)
 824    integer, intent(in) :: numLev_src ! Number of input levels (source)
 825    integer, intent(in) :: ni_dest ! Number of profiles
 826    integer, intent(in) :: nj_dest ! Number of profiles
 827    integer, intent(in) :: numLev_dest ! Number of output levels (destination)
 828    real(8)             :: press_dest(:,:,:) ! Vertical levels, pressure (destination)
 829    real(8)             :: qmin_dest(:,:,:)  ! Interpolated profiles (destination)
 830    real(8)             :: qmax_dest(:,:,:)  ! Interpolated profiles (destination)
 831
 832    ! Locals:
 833    integer :: ji, jk, jo, ii, jj, ik, iorder
 834    real(8) :: zpo(numLev_dest)
 835    integer :: il(numLev_dest)
 836    real(8) :: zpi(0:numLev_src+1)
 837    real(8) :: zqmin_src(0:numLev_src+1), zqmax_src(0:numLev_src+1)
 838    real(8) :: zw1, zw2, zp, xi, zrt, zp1, zp2
 839
 840    zpi(0)=200000.d0
 841    zpi(numLev_src+1)=200000.d0
 842
 843    ! Determine if input pressure levels are in ascending or
 844    ! descending order.
 845    if ( press_src(1) < press_src(numLev_src) ) then
 846      iorder = 1
 847    else
 848      iorder = -1
 849    end if
 850
 851    ! Source levels
 852    do jk = 1, numLev_src
 853      zpi(jk) = press_src(jk)
 854      zqmin_src(jk) = qmin_src(jk)
 855      zqmax_src(jk) = qmax_src(jk)
 856    enddo
 857    zqmin_src(0) = qmin_src(1)
 858    zqmin_src(numLev_src+1) = qmin_src(numLev_src)
 859    zqmax_src(0) = qmax_src(1)
 860    zqmax_src(numLev_src+1) = qmax_src(numLev_src)
 861
 862    do jj = 1, nj_dest
 863      do ii = 1, ni_dest
 864        zpo(:) = press_dest(ii,jj,:)
 865
 866        ! Interpolate in log of pressure or extrapolate with constant value
 867        ! for each destination pressure level
 868
 869        ! Find the adjacent level below
 870        il(:) = 0
 871        do ji = 1, numLev_src
 872          do jo = 1, numLev_dest
 873            zrt = zpo(jo)
 874            zp = zpi(ji)
 875            xi = sign(1.0d0,iorder*(zrt-zp))
 876            il(jo) = il(jo) + max(0.0d0,xi)
 877          end do
 878        end do
 879
 880        ! Interpolation/extrapolation
 881        do jo = 1, numLev_dest
 882          ik = il(jo)
 883          zp = zpo(jo)
 884          zp1 = zpi(ik)
 885          zp2 = zpi(ik+1)
 886          zw1 = log(zp/zp2)/log(zp1/zp2)
 887          zw2 = 1.d0 - zw1
 888          qmin_dest(ii,jj,jo) = zw1*zqmin_src(ik) +  zw2*zqmin_src(ik+1)
 889          qmax_dest(ii,jj,jo) = zw1*zqmax_src(ik) +  zw2*zqmax_src(ik+1)
 890        end do
 891
 892      end do ! ii
 893    end do ! jj
 894
 895  end subroutine qlim_lintv_minmax
