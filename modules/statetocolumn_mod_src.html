 123  subroutine pressureProfileMonotonicityCheck(obsSpaceData, column)
 124    !
 125    !:Purpose: Check for non monotonic pressure profiles that can be computed in slantpathmode
 126    !
 127    implicit none
 128
 129    ! arguments
 130    type(struct_obs), intent(inout)       :: obsSpaceData
 131    type(struct_columnData), intent(inout):: column
 132
 133    ! locals
 134    integer, parameter :: numWriteMax = 10
 135    integer :: headerIndex, bodyIndex, iterationCount, singularIndex, levelIndex
 136    integer :: pressureVarIndex
 137    integer :: nlv
 138    integer :: numWrites
 139    real(8), pointer :: pressureProfile(:)
 140    logical :: monotonicProfile
 141    integer, parameter :: nPressureVar =2
 142    character(len=4), parameter :: pressureVarList(nPressureVar)=['P_T ', 'P_M ']
 143
 144    write(*,*) ' '
 145    write(*,*) 'pressureProfileMonotonicityCheck: START'
 146    write(*,*) ' '
 147    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 148
 149    numWrites = 0
 150
 151    call obs_set_current_header_list(obsSpaceData, 'TO')
 152    HEADER: do
 153      headerIndex = obs_getHeaderIndex(obsSpaceData)
 154      if (headerIndex < 0) exit HEADER
 155      do pressureVarIndex = 1, nPressureVar
 156        pressureProfile => col_getColumn(column, headerIndex, pressureVarList(pressureVarIndex))
 157        nlv = size(pressureProfile)
 158        monotonicProfile = .true.
 159        iterationCount = 0
 160        iterationLoop:do
 161          singularIndex = -1
 162          levelSearch:do levelIndex = 1, nlv - 1
 163            if ( pressureProfile(levelIndex) > pressureProfile(levelIndex+1)) then
 164              singularIndex = levelIndex
 165              exit levelSearch
 166            end if
 167          end do levelSearch
 168          if ( singularIndex == -1 ) exit iterationLoop !regular profile or correction OK
 169          iterationCount = iterationCount + 1
 170          if (iterationCount == 1) then
 171            monotonicProfile = .false.
 172            if (numWrites < numWriteMax) then
 173              numWrites = numWrites + 1
 174              write(*,*) 'pressureProfileMonotonicityCheck: found non monotonic pressure profile:', &
 175                   pressureVarList(pressureVarIndex), pressureProfile
 176            end if
 177          end if
 178          if (singularIndex == 1) then !should never happen
 179            write(*,*) 'pressureProfileMonotonicityCheck: ', pressureProfile(1:2)
 180            call utl_abort('pressureProfileMonotonicityCheck: profile in the wrong order ?' &
 181                 // pressureVarList(pressureVarIndex))
 182          end if
 183          pressureProfile(singularIndex) = 0.5d0 * ( pressureProfile(singularIndex - 1) + pressureProfile(singularIndex + 1) )
 184          write(*,*) 'pressureProfileMonotonicityCheck: profile iteration', &
 185               pressureVarList(pressureVarIndex), iterationCount
 186        end do iterationLoop
 187
 188        ! if requested reject the corrected profile
 189        if (.not. monotonicProfile .and. rejectObsNonMonotonicPressure) then
 190          call obs_headSet_i(obsSpaceData,OBS_ST1,headerIndex, &
 191               ibset( obs_headElem_i(obsSpaceData,OBS_ST1,headerIndex), 05))
 192          call obs_set_current_body_list(obsSpaceData, headerIndex)
 193          BODY: do
 194            bodyIndex = obs_getBodyIndex(obsSpaceData)
 195            if (bodyIndex < 0) exit BODY
 196            if (rejectObsNonMonotonicPressure) then
 197              call obs_bodySet_i(obsSpaceData, OBS_ASS, bodyIndex, obs_notAssimilated)
 198              call obs_bodySet_i(obsSpaceData, OBS_FLG, bodyIndex, &
 199                   ibset(obs_bodyElem_i(obsSpaceData, OBS_FLG, bodyIndex),9))
 200            end if
 201          end do BODY
 202        end if
 203      end do ! loop on pressure variables
 204     
 205    end do HEADER
 206
 207    write(*,*) 'pressureProfileMonotonicityCheck: END'
 208
 209  end subroutine pressureProfileMonotonicityCheck

 214  subroutine latlonChecksAnlGrid(obsSpaceData, hco_core, moveObsAtPole)
 215    !
 216    !:Purpose: Check the lat/lon of observations and modify if necessary
 217    !
 218    implicit none
 219
 220    ! arguments
 221    type(struct_obs)          :: obsSpaceData
 222    type(struct_hco), pointer :: hco_core
 223    logical                   :: moveObsAtPole
 224
 225    ! locals
 226    integer :: headerIndex, ierr
 227    integer :: idata, idatend, jdata, subGridIndex
 228    real(4) :: lat_r4, lon_r4, lat_deg_r4, lon_deg_r4
 229    real(4) :: xpos_r4, ypos_r4, xpos2_r4, ypos2_r4
 230    real(4) :: xposLowerBoundAnl_r4, xposUpperBoundAnl_r4
 231    real(8) :: lat_r8, lon_r8
 232    integer, save :: numWrites = 0
 233
 234    ! external functions
 235    integer :: gdllfxy
 236
 237    write(*,*) ' '
 238    write(*,*) 'latlonChecksAnlGrid: STARTING'
 239    write(*,*) ' '
 240    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 241
 242    !
 243    !-    Get the Analysis Grid structure
 244    !
 245    if ( hco_core % global ) then
 246       xposLowerBoundAnl_r4 = - huge(1.0) ! no limit since grid is global (periodic)
 247       xposUpperBoundAnl_r4 = + huge(1.0) ! no limit since grid is global (periodic)
 248    else
 249       xposLowerBoundAnl_r4 = 1.0
 250       xposUpperBoundAnl_r4 = real(hco_core % ni)
 251    end if
 252
 253    header_loop: do headerIndex=1, obs_numheader(obsSpaceData)
 254
 255      !- Get LatLon of observation location
 256      lat_r8 = obs_headElem_r(obsSpaceData,OBS_LAT,headerIndex)
 257      lon_r8 = obs_headElem_r(obsSpaceData,OBS_LON,headerIndex)
 258      lat_r4 = real(lat_r8,4)
 259      lon_r4 = real(lon_r8,4)
 260      if (lon_r4.lt.0.0         ) lon_r4 = lon_r4 + 2.0*MPC_PI_R4
 261      if (lon_r4.ge.2.*MPC_PI_R4) lon_r4 = lon_r4 - 2.0*MPC_PI_R4
 262
 263      lat_deg_r4 = lat_r4 * MPC_DEGREES_PER_RADIAN_R4 ! Radian To Degree
 264      lon_deg_r4 = lon_r4 * MPC_DEGREES_PER_RADIAN_R4
 265
 266      !
 267      !- Find the position in the analysis grid
 268      !
 269      ierr = gpos_getPositionXY( hco_core % EZscintID,  &
 270                                xpos_r4, ypos_r4, xpos2_r4, ypos2_r4, &
 271                                lat_deg_r4, lon_deg_r4, subGridIndex )
 272
 273      !- Test if the obs is outside the analysis grid
 274      if ( xpos_r4 < xposLowerBoundAnl_r4  .or. &
 275           xpos_r4 > xposUpperBoundAnl_r4  .or. &
 276           ypos_r4 < 1.0                   .or. &
 277           ypos_r4 > real(hco_core % nj) ) then
 278
 279        if ( hco_core % global ) then
 280
 281          if ( moveObsAtPole ) then
 282            ! Modify latitude if we have an observation at or near the poles
 283            write(*,*) ''
 284            write(*,*) 'latlonChecksAnlGrid: Moving OBS inside the GLOBAL ANALYSIS grid, ', headerIndex
 285            write(*,*) '  true position : ', lat_deg_r4, lon_deg_r4, ypos_r4, xpos_r4
 286
 287            !- Move the observation to the nearest grid point
 288            if ( ypos_r4 < 1.0 )                ypos_r4 = 1.0
 289            if ( ypos_r4 > real(hco_core % nj) ) ypos_r4 = real(hco_core % nj)
 290
 291            ierr = gdllfxy( hco_core % EZscintID, &    ! IN
 292                            lat_deg_r4, lon_deg_r4, & ! OUT
 293                            xpos_r4, ypos_r4, 1)      ! IN
 294
 295            write(*,*) '  new  position : ', lat_deg_r4, lon_deg_r4, ypos_r4, xpos_r4
 296
 297            lat_r8 = real(lat_deg_r4,8) * MPC_RADIANS_PER_DEGREE_R8
 298            lon_r8 = real(lon_deg_r4,8) * MPC_RADIANS_PER_DEGREE_R8
 299            call obs_headSet_r(obsSpaceData,OBS_LAT,headerIndex, lat_r8) ! IN
 300            call obs_headSet_r(obsSpaceData,OBS_LON,headerIndex, lon_r8) ! IN
 301          else
 302            write(*,*)
 303            write(*,*) 'latlonChecksAnlGrid: OBS outside the GLOBAL ANALYSIS grid, but NOT moved, ', headerIndex
 304          end if
 305
 306        else
 307          ! The observation is outside the domain
 308          ! In LAM Analysis mode we must discard this observation
 309          numWrites = numWrites + 1
 310          if (numWrites < maxNumWrites) then
 311            write(*,*) 'latlonChecksAnlGrid: Rejecting OBS outside the LAM ANALYSIS grid domain, ', headerIndex
 312            write(*,*) '  position : ', lat_deg_r4, lon_deg_r4, ypos_r4, xpos_r4
 313          else if (numWrites == maxNumWrites) then
 314            write(*,*) 'latlonChecksAnlGrid: More rejects, but reached maximum number of writes to the listing.'
 315          end if
 316
 317          idata   = obs_headElem_i(obsSpaceData,OBS_RLN,headerIndex)
 318          idatend = obs_headElem_i(obsSpaceData,OBS_NLV,headerIndex) + idata -1
 319          do jdata = idata, idatend
 320            call obs_bodySet_i(obsSpaceData,OBS_ASS,JDATA, obs_notAssimilated)
 321          end do
 322          call obs_headSet_i(obsSpaceData,OBS_ST1,headerIndex,  &
 323                             ibset( obs_headElem_i(obsSpaceData,OBS_ST1,headerIndex), 05))
 324        end if
 325
 326      end if
 327
 328    end do header_loop
 329
 330    write(*,*) 'latlonChecksAnlGrid: END'
 331
 332  end subroutine latlonChecksAnlGrid

 393    logical :: doSetup3dHeights, lastCall

 399    namelist /nams2c/ useFootprintForTovs, rejectObsNonMonotonicPressure

 417      slantPath_TO_tlad = .false.
 418      slantPath_RO_nl   = .false.
 419      slantPath_RA_nl   = .false.
 420      calcHeightPressIncrOnColumn = .false.
 421      useFootprintForTovs = .false.
 422      rejectObsNonMonotonicPressure =.true.
 423
 424      if ( .not. utl_isNamelistPresent('NAMS2C','./flnml') ) then
 425        if ( mmpi_myid == 0 ) then
 426          write(*,*) 's2c_setupInterpInfo: nams2c is missing in the namelist.'
 427          write(*,*) '                     The default values will be taken.'
 428        end if
 429
 430      else
 431        ! reading namelist variables
 432        nulnam = 0
 433        ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
 434        read(nulnam, nml = nams2c, iostat = ierr)
 435        if ( ierr /= 0 ) call utl_abort('s2c_setupInterpInfo: Error reading namelist')
 436        ierr = fclos(nulnam)
 437      end if
 438      if ( mmpi_myid == 0 ) write(*, nml = nams2c)
 439    end if
 440
 441    doSlantPath = .false.
 442    SlantTO     = .false.
 443    SlantRO     = .false.
 444    if ( slantPath_TO_nl   .and. inputStateVectorType == 'nl' ) then
 445      doSlantPath = .true.
 446      SlantTO     = .true.
 447    endif
 448    if ( slantPath_TO_tlad .and. inputStateVectorType /= 'nl' ) then
 449      doSlantPath = .true.
 450      SlantTO     = .true.
 451    endif
 452    if ( slantPath_RO_nl   .and. inputStateVectorType == 'nl' ) then
 453      doSlantPath = .true.
 454      SlantRO     = .true.
 455    endif
 456    if ( slantPath_RA_nl   .and. inputStateVectorType == 'nl' ) then
 457      doSlantPath = .true.
 458      SlantRA     = .true.
 459    endif

 461               doSlantPath, SlantTO, SlantRO, SlantRA
 462
 463    numStep = stateVector%numStep
 464    numHeader = headerIndexEnd - headerIndexBeg + 1
 465
 466    call oti_setup(interpInfo%oti, obsSpaceData, numStep,  &

 468                   interpType_opt=timeInterpType, flagObsOutside_opt=.true.)
 469
 470    if ((stateVector%heightSfcPresent) .and. ( mmpi_myid == 0)) then
 471      mykBeg = 0 
 472    else
 473      mykBeg = stateVector%mykBeg
 474    end if   
 475
 476    call rpn_comm_allgather(mykBeg, 1,'mpi_integer',       &
 477                            allkBeg,1,'mpi_integer','grid',ierr)
 478
 479    ! Allow for periodicity in Longitude for global Gaussian grid
 480    if ( stateVector%hco%grtyp == 'G' .or. &
 481         (stateVector%hco%grtyp == 'Z' .and. stateVector%hco%global) ) then
 482      niP1 = stateVector%ni + 1
 483    else
 484      niP1 = stateVector%ni
 485    end if
 486
 487    ! First count the number of headers for each stepIndex
 488    allocate(allNumHeaderUsed(numStep,mmpi_nprocs))
 489    allNumHeaderUsed(:,:) = 0
 490    do stepIndex = 1, numStep
 491      numHeaderUsed = 0
 492
 493      header_loop1: do headerIndex = headerIndexBeg, headerIndexEnd
 494
 495        ! if obs inside window, but zero weight for current stepIndex then skip it

 497          cycle header_loop1
 498        end if
 499
 500        numHeaderUsed = numHeaderUsed + 1
 501
 502      end do header_loop1
 503      ! gather the number of obs over all processors for each timestep
 504      call rpn_comm_allgather(numHeaderUsed,                 1, 'MPI_INTEGER', &
 505                              allNumHeaderUsed(stepIndex,:), 1, 'MPI_INTEGER', &
 506                              'GRID',ierr)
 507
 508    end do
 509
 510    numHeaderUsedMax = maxval(allNumHeaderUsed(:,:))

 515    allocate(footprintRadiusVec_r4(numHeaderUsedMax))
 516    headerIndexVec(:,:) = 0
 517
 518    ! copy the horizontal grid object
 519    interpInfo%hco => stateVector%hco

 549    allocate(allLonOneLev(numHeaderUsedMax,mmpi_nprocs))

 556      do stepIndex = 1, numStep
 557        do procIndex = 1, mmpi_nprocs

 564    end if

 570                       .not. gsv_isAllocated(stateVector_1Step) .and. &

 572                       gsv_varExist(stateVector,'Z_M')

 576

 587        call gsv_getField(stateVector,height3D_r4_ptr1)

 591        call gsv_allocate( stateVector_Tiles_allVar_1Step, 1, &

 604        call gsv_getField(stateVector_Tiles_allVar_1Step,height3D_r4_ptr1,'Z_T')
 605        call gsv_getField(stateVector_Tiles_1Step,height3D_r4_ptr2,'Z_T')

 608        call gsv_getField(stateVector_Tiles_allVar_1Step,height3D_r4_ptr1,'Z_M')

 610        height3D_r4_ptr2(:,:,:) = height3D_r4_ptr1(:,:,:)
 611
 612        call gsv_deallocate(stateVector_Tiles_allVar_1Step)
 613
 614      else
 615        stateVector_Tiles_ptr => gvt_getStateVectorTrial('height')
 616
 617        call gsv_allocate( stateVector_Tiles_1Step, 1, &
 618                           stateVector%hco, stateVector%vco, &
 619                           mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
 620                           dataKind_opt=4, varNames_opt=(/'Z_M','Z_T'/) )
 621
 622        call gsv_getField(stateVector_Tiles_ptr,height3D_r8_ptr1,'Z_T')
 623        call gsv_getField(stateVector_Tiles_1Step,height3D_r4_ptr2,'Z_T')
 624        height3D_r4_ptr2(:,:,:) = height3D_r8_ptr1(:,:,:)
 625
 626        call gsv_getField(stateVector_Tiles_ptr,height3D_r8_ptr1,'Z_M')
 627        call gsv_getField(stateVector_Tiles_1Step,height3D_r4_ptr2,'Z_M')
 628        height3D_r4_ptr2(:,:,:) = height3D_r8_ptr1(:,:,:)
 629
 630      end if ! inputStateVectorType 
 631
 632      ! Communicate 3D height fields onto all mpi tasks
 633      call gsv_allocate( stateVector_1Step, 1, &
 634                         stateVector%hco, stateVector%vco, &
 635                         mpi_local_opt=.false., &
 636                         dataKind_opt=4, varNames_opt=(/'Z_M','Z_T'/) )

 638      call gsv_transposeTilesToMpiGlobal(stateVector_1Step, stateVector_Tiles_1Step)

 640      call gsv_getField(stateVector_1Step,height3D_T_r4,'Z_T')
 641      call gsv_getField(stateVector_1Step,height3D_M_r4,'Z_M')
 642    
 643      write(*,*) 's2c_setupInterpInfo, height3D_T_r4='

 646      write(*,*) height3D_M_r4(1,1,:)
 647
 648      write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 649    end if ! doSlantPath 
 650
 651    ! get observation lat-lon and footprint radius onto all mpi tasks
 652    step_loop2: do stepIndex = 1, numStep
 653      numHeaderUsed = 0
 654
 655      footprintRadiusVec_r4(:) = bilinearFootprint
 656
 657      header_loop2: do headerIndex = headerIndexBeg, headerIndexEnd
 658
 659        ! if obs inside window, but zero weight for current stepIndex then skip it

 664        numHeaderUsed = numHeaderUsed + 1

 667        footprintRadiusVec_r4(numHeaderUsed) = s2c_getFootprintRadius(obsSpaceData, &
 668                                                                stateVector, headerIndex)
 669
 670      end do header_loop2
 671
 672      call rpn_comm_allgather(footprintRadiusVec_r4,                numHeaderUsedMax, 'MPI_REAL4', &
 673                              allFootprintRadius_r4(:,stepIndex,:), numHeaderUsedMax, 'MPI_REAL4', &
 674                              'GRID', ierr)
 675
 676      allocate(latColumn(numHeaderUsedMax,allkBeg(1):stateVector%nk))
 677      allocate(lonColumn(numHeaderUsedMax,allkBeg(1):stateVector%nk))
 678      latColumn(:,:) = 0.0d0
 679      lonColumn(:,:) = 0.0d0
 680
 681      if ( doSlantPath .and. &
 682           gsv_varExist(stateVector,'Z_T') .and. &
 683           gsv_varExist(stateVector,'Z_M') ) then
 684
 685        allocate(latLev_T(nlev_T))
 686        allocate(lonLev_T(nlev_T))
 687        allocate(latLev_M(nlev_M))
 688        allocate(lonLev_M(nlev_M))

 690        lonLev_T(:) = 0.0d0
 691        latLev_M(:) = 0.0d0
 692        lonLev_M(:) = 0.0d0
 693
 694        firstHeaderSlantPathTO = .true.
 695        firstHeaderSlantPathRO = .true.
 696        firstHeaderSlantPathRA = .true.
 697        header_loop3: do headerUsedIndex = 1, numHeaderUsed
 698          headerIndex = headerIndexVec(headerUsedIndex,stepIndex)

 704          if (lon_r4 >= 2.0*MPC_PI_R4) lon_r4 = lon_r4 - 2.0*MPC_PI_R4
 705
 706          codeType = obs_headElem_i(obsSpaceData, OBS_ITY, headerIndex)
 707
 708          if ( tvs_isIdBurpTovs(codeType) .and. SlantTO ) then
 709            if ( firstHeaderSlantPathTO ) then
 710              write(*,'(a,i3,a,i8)') 's2c_setupInterpInfo: start slant-path for TOVS. stepIndex = ', &
 711                   stepIndex,' and numHeaderUsed = ',numHeaderUsed
 712              firstHeaderSlantPathTO = .false.
 713            end if
 714
 715            ! calculate lat/lon along the line of sight
 716            call utl_tmg_start(32,'------s2c_Slant')
 717            call slp_calcLatLonTovs( obsSpaceData, stateVector%hco, headerIndex, & ! IN
 718                                     height3D_T_r4, height3D_M_r4,               & ! IN
 719                                     latLev_T, lonLev_T,                         & ! OUT
 720                                     latLev_M, lonLev_M,                         & ! OUT
 721                                     latLev_S, lonLev_S             )              ! OUT
 722            call utl_tmg_stop(32)
 723
 724          else if (codeType == codtyp_get_codtyp('ro') .and. SlantRO ) then
 725            if ( firstHeaderSlantPathRO ) then
 726              write(*,'(a,i3,a,i8)') 's2c_setupInterpInfo: start slant-path for RO. stepIndex = ', &
 727                   stepIndex,' and numHeaderUsed = ',numHeaderUsed
 728              firstHeaderSlantPathRO = .false.
 729            end if
 730
 731            ! Calculate lat/lon along the GPSRO obs
 732            call utl_tmg_start(32,'------s2c_Slant')
 733            call slp_calcLatLonRO( obsSpaceData, stateVector%hco, headerIndex, & ! IN
 734                                   height3D_T_r4, height3D_M_r4,               & ! IN
 735                                   latLev_T, lonLev_T,                         & ! OUT
 736                                   latLev_M, lonLev_M,                         & ! OUT
 737                                   latLev_S, lonLev_S                          ) ! OUT
 738            call utl_tmg_stop(32)
 739          else if (codeType == codtyp_get_codtyp('radar') .and. SlantRA ) then
 740            if ( firstHeaderSlantPathRA ) then
 741              write(*,'(a,i3,a,i8)') 's2c_setupInterpInfo: start slant-path for RADAR. stepIndex=', &
 742                   stepIndex,' and numHeaderUsed=',numHeaderUsed
 743              firstHeaderSlantPathRA = .false.
 744            end if
 745            
 746             ! calculate lat/lon along the radar beam obs
 747             call slp_calcLatLonRadar( obsSpaceData, stateVector%hco, headerIndex, & ! IN
 748                                     height3D_T_r4, height3D_M_r4,                 & ! IN
 749                                     latLev_T, lonLev_T,                           & ! OUT
 750                                     latLev_M, lonLev_M,                           & ! OUT
 751                                     latLev_S, lonLev_S                           ) ! OUT
 752          else
 753
 754            latLev_T(:) = real(lat_r4,8)
 755            lonLev_T(:) = real(lon_r4,8)
 756            latLev_M(:) = real(lat_r4,8)
 757            lonLev_M(:) = real(lon_r4,8)
 758            latLev_S = real(lat_r4,8)
 759            lonLev_S = real(lon_r4,8)
 760          end if

 762          ! check if the slanted lat/lon is inside the domain

 764                              headerIndex, rejectOutsideObs, & ! IN
 765                              latLev_T, lonLev_T,            & ! IN/OUT
 766                              latLev_M, lonLev_M,            & ! IN/OUT
 767                              latLev_S, lonLev_S )             ! IN/OUT
 768
 769          ! put the lat/lon from TH/MM levels to kIndex
 770          do kIndex = allkBeg(1), stateVector%nk
 771            if ( kIndex == 0 ) then
 772              varLevel = 'SF'
 773            else
 774              levIndex = gsv_getLevFromK(stateVector,kIndex)
 775              varLevel = vnl_varLevelFromVarname(gsv_getVarNameFromK(stateVector,kIndex))
 776            end if
 777
 778            if ( varLevel == 'TH' ) then

 783              lonColumn(headerUsedIndex,kIndex) = lonLev_M(levIndex)
 784            else if ( varLevel == 'SF' ) then
 785              latColumn(headerUsedIndex,kIndex) = latLev_S
 786              lonColumn(headerUsedIndex,kIndex) = lonLev_S
 787            else

 789            end if
 790
 791          end do
 792
 793        end do header_loop3

 798        numkToSend = min(mmpi_nprocs,stateVector%nk)

 800        allocate(lat_recv_r8(numHeaderUsedMax,mmpi_nprocs))
 801        lat_recv_r8(:,:) = 0.0d0
 802        allocate(lat_send_r8(numHeaderUsedMax,mmpi_nprocs))

 819        recvdispls(1) = 0

 822        end do

 837              lat_send_r8(1:numHeaderUsed,procIndex) = latColumn(1:numHeaderUsed,kIndex)
 838              lon_send_r8(1:numHeaderUsed,procIndex) = lonColumn(1:numHeaderUsed,kIndex)
 839              sendsizes(procIndex) = numHeaderUsed
 840            else
 841              sendsizes(procIndex) = 0
 842            end if
 843          end do
 844
 845          ! all tasks recv only from those with data
 846          kIndex = kIndexCount + mykBeg - 1

 848            do procIndex = 1, mmpi_nprocs
 849              recvsizes(procIndex) = allNumHeaderUsed(stepIndex,procIndex)
 850            end do
 851          else
 852            recvsizes(:) = 0
 853          end if

 857                             mmpi_comm_grid, ierr)
 858          call mpi_alltoallv(lon_send_r8, sendsizes, senddispls, mmpi_datyp_real8,  &
 859                             lon_recv_r8, recvsizes, recvdispls, mmpi_datyp_real8,  &
 860                             mmpi_comm_grid, ierr)
 861
 862          do procIndex = 1, mmpi_nprocs
 863            ! all tasks copy the received step data into correct slot
 864            kIndex = kIndexCount + mykBeg - 1
 865            if ( kIndex <= stateVector%mykEnd ) then
 866              interpInfo%stepProcData(procIndex,stepIndex)%allLat(:,kIndex) = &
 867                   lat_recv_r8(1:allNumHeaderUsed(stepIndex,procIndex),procIndex)
 868              interpInfo%stepProcData(procIndex,stepIndex)%allLon(:,kIndex) = &
 869                   lon_recv_r8(1:allNumHeaderUsed(stepIndex,procIndex),procIndex)
 870            end if
 871          end do
 872
 873        end do ! kIndexCount
 874
 875        deallocate(lon_send_r8)
 876        deallocate(lon_recv_r8)
 877        deallocate(lat_send_r8)
 878        deallocate(lat_recv_r8)
 879
 880        deallocate(latLev_T)
 881        deallocate(lonLev_T)
 882        deallocate(latLev_M)
 883        deallocate(lonLev_M)
 884
 885      else ! not doSlantPath
 886
 887        allocate(latLev_T(1))
 888        allocate(lonLev_T(1))
 889        allocate(latLev_M(1))
 890        allocate(lonLev_M(1))
 891        latLev_T(:) = 0.0d0
 892        lonLev_T(:) = 0.0d0
 893        latLev_M(:) = 0.0d0
 894        lonLev_M(:) = 0.0d0
 895
 896        do headerUsedIndex = 1, numHeaderUsed
 897          headerIndex = headerIndexVec(headerUsedIndex,stepIndex)
 898
 899          !- Get LatLon of observation location
 900          lat_r4 = real(obs_headElem_r(obsSpaceData, OBS_LAT, headerIndex), 4)
 901          lon_r4 = real(obs_headElem_r(obsSpaceData, OBS_LON, headerIndex), 4)

 903          if (lon_r4 >= 2.0*MPC_PI_R4) lon_r4 = lon_r4 - 2.0*MPC_PI_R4
 904
 905          latLev_T(:) = real(lat_r4,8)
 906          lonLev_T(:) = real(lon_r4,8)
 907          latLev_M(:) = real(lat_r4,8)
 908          lonLev_M(:) = real(lon_r4,8)
 909
 910          ! check if the lat/lon is inside the domain
 911          call latlonChecks ( obsSpaceData, stateVector%hco, & ! IN
 912                              headerIndex, rejectOutsideObs, & ! IN
 913                              latLev_T, lonLev_T,            & ! IN/OUT
 914                              latLev_M, lonLev_M )             ! IN/OUT 
 915
 916          latColumn(headerUsedIndex,allkBeg(1)) = latLev_T(1)
 917          lonColumn(headerUsedIndex,allkBeg(1)) = lonLev_T(1)
 918        end do
 919
 920        ! gather geographical lat, lon positions of observations from all processors
 921        call rpn_comm_allgather(latColumn(:,allkBeg(1)), numHeaderUsedMax, 'MPI_REAL8', &
 922                                allLatOneLev(:,:), numHeaderUsedMax, 'MPI_REAL8', 'GRID', ierr)
 923        call rpn_comm_allgather(lonColumn(:,allkBeg(1)), numHeaderUsedMax, 'MPI_REAL8', &
 924                                allLonOneLev(:,:), numHeaderUsedMax, 'MPI_REAL8', 'GRID', ierr)
 925
 926        k_loop: do kIndex = mykBeg, statevector%mykEnd
 927          do procIndex = 1, mmpi_nprocs
 928            interpInfo%stepProcData(procIndex,stepIndex)%allLat(:,kIndex) = allLatOneLev(1:allNumHeaderUsed(stepIndex,procIndex),procIndex)
 929            interpInfo%stepProcData(procIndex,stepIndex)%allLon(:,kIndex) = allLonOneLev(1:allNumHeaderUsed(stepIndex,procIndex),procIndex)
 930          end do
 931        end do k_loop
 932
 933        deallocate(latLev_T)
 934        deallocate(lonLev_T)
 935        deallocate(latLev_M)
 936        deallocate(lonLev_M)
 937
 938      end if ! doSlantPath 
 939
 940      deallocate(lonColumn)
 941      deallocate(latColumn)
 942
 943    end do step_loop2
 944
 945    if ( gsv_isAllocated(stateVector_1Step) .and. lastCall ) then
 946      write(*,*) 's2c_setupInterpInfo: deallocate height3D fields'
 947      call gsv_deallocate(stateVector_1Step)
 948    end if
 949    deallocate(footprintRadiusVec_r4)
 950
 951    write(*,*) 's2c_setupInterpInfo: latlonChecks and lat/lon MPI comm finished.'
 952
 953    allocate(allHeaderIndex(numHeaderUsedMax,numStep,mmpi_nprocs))
 954    ! gather the headerIndexVec arrays onto all processors
 955    call rpn_comm_allgather(headerIndexVec, numHeaderUsedMax*numStep, 'MPI_INTEGER', &
 956                            allHeaderIndex, numHeaderUsedMax*numStep, 'MPI_INTEGER', &
 957                            'GRID',ierr)
 958
 959    do procIndex = 1, mmpi_nprocs
 960      do stepIndex = 1, numStep
 961        do headerIndex = 1, allNumHeaderUsed(stepIndex,procIndex)
 962          interpInfo%stepProcData(procIndex, stepIndex)%allHeaderIndex(headerIndex) = allHeaderIndex(headerIndex,stepIndex,procIndex)
 963        end do
 964      end do
 965    end do
 966
 967    ! create kdtree to use in footprint operator, if any footprint radius > 0.
 968    interpInfo%inputStateVectorType = inputStateVectorType
 969    if ( any(allFootprintRadius_r4(:,:,:) > 0.0) ) then
 970      if ( (inputStateVectorType == 'nl' .and. .not. associated(tree_nl))   .or. &
 971           (inputStateVectorType == 'tl' .and. .not. associated(tree_tlad)) .or. &
 972           (inputStateVectorType == 'ad' .and. .not. associated(tree_tlad)) ) then

1474        call s2c_deallocInterpInfo(inputStateVectorType='tlad')

1476    end if

1483                         statevector_out%hco, statevector_out%vco, &
1484                         mpi_local_opt=.true., &
1485                         dataKind_opt=gsv_getDataKind(statevector_out), &
1486                         allocHeight_opt=.false., allocPressure_opt=.false. )

1488      call czp_calcZandP_ad(columnAnlInc, columnTrlOnAnlIncLev)
1489    else
1490      stateVector => stateVector_out
1491    end if

1493    nullify(varNames)
1494    call gsv_varNamesList(varNames, statevector)
1495    call gsv_allocate( statevector_VarsLevs, statevector%numstep, &
1496                       statevector%hco, statevector%vco,          &
1497                       mpi_local_opt=.true., mpi_distribution_opt='VarsLevs', &
1498                       dataKind_opt=gsv_getDataKind(statevector), &
1499                       varNames_opt=varNames )
1500    deallocate(varNames)

1502

1504    numHeader = obs_numheader(obsSpaceData)

1506                            'MPI_INTEGER', 'MPI_MAX', 'GRID', ierr)
1507
1508    if ( .not. interpInfo_tlad%initialized ) then
1509      rejectOutsideObs = .false.

1514                                inputStateVectorType='ad' )

1538

1540

1552
1553        do headerIndex = 1, numHeader
1554          cols_send(headerIndex,procIndex) = allCols_ptr(levIndex,headerIndex)
1555        end do
1556      end do proc_loop

1558
1559      call rpn_comm_barrier('GRID',ierr)
1560
1561      ! mpi communication: alltoall for one level/variable
1562      nsize = numHeaderMax
1563      if(mmpi_nprocs > 1) then
1564        call rpn_comm_alltoall(cols_send, nsize, 'MPI_REAL8',  &
1565                               cols_recv, nsize, 'MPI_REAL8', 'GRID', ierr)

1568      end if
1569
1570      if ( kIndex <= stateVector_VarsLevs%mykEnd ) then
1571        varName = gsv_getVarNameFromK(statevector,kIndex)
1572
1573        if ( varName == 'UU' .or. varName == 'VV' ) then
1574          call gsv_getFieldUV(stateVector_VarsLevs, ptr3d_UV, kIndex)
1575        end if
1576

1578        do procIndex = 1, mmpi_nprocs
1579          do stepIndex = 1, numStep
1580            !$OMP PARALLEL DO PRIVATE (headerIndex, headerUsedIndex, weight)
1581            do headerUsedIndex = 1, interpInfo_tlad%allNumHeaderUsed(stepIndex, procIndex)
1582

1584              weight = oti_getTimeInterpWeightMpiGlobal(interpInfo_tlad%oti,  &
1585                                                        headerIndex,stepIndex,procIndex)

1588                   weight * cols_recv(headerIndex,procIndex)
1589
1590            end do
1591            !$OMP END PARALLEL DO
1592          end do
1593        end do
1594
1595        call utl_tmg_start(40,'------s2c_AD_Hinterp')
1596        !$OMP PARALLEL DO PRIVATE (stepIndex, procIndex, yourNumHeader)
1597        step_loop: do stepIndex = 1, numStep
1598          if ( maxval(interpInfo_tlad%allNumHeaderUsed(stepIndex,:)) == 0 ) cycle step_loop
1599
1600          ! interpolate to the columns destined for all procs for all steps and one lev/var
1601          do procIndex = 1, mmpi_nprocs
1602            yourNumHeader = interpInfo_tlad%allNumHeaderUsed(stepIndex,procIndex)
1603            if ( yourNumHeader > 0 ) then
1604              if ( varName == 'UU' ) then
1605                call myezuvint_ad( cols_hint(1:yourNumHeader,stepIndex,procIndex), 'UU',  &
1606                                   ptr4d(:,:,kIndex,stepIndex), ptr3d_UV(:,:,stepIndex),  &
1607                                   interpInfo_tlad, kIndex, stepIndex, procIndex )
1608              else if ( varName == 'VV' ) then

1610                                   ptr3d_UV(:,:,stepIndex), ptr4d(:,:,kIndex,stepIndex),  &
1611                                   interpInfo_tlad, kIndex, stepIndex, procIndex )
1612              else
1613                call myezsint_ad( cols_hint(1:yourNumHeader,stepIndex,procIndex), &
1614                                  ptr4d(:,:,kIndex,stepIndex), interpInfo_tlad, kIndex, &
1615                                  stepIndex, procIndex )
1616              end if
1617            end if
1618          end do
1619
1620        end do step_loop
1621        !$OMP END PARALLEL DO
1622        call utl_tmg_stop(40)
1623
1624      end if ! if kIndex <= mykEnd
1625
1626    end do k_loop
1627
1628    deallocate(cols_hint)
1629    deallocate(cols_send)
1630    deallocate(cols_recv)
1631
1632    call rpn_comm_barrier('GRID',ierr)
1633
1634    call gsv_transposeTilesToVarsLevsAd( statevector_VarsLevs, statevector )
1635
1636    if (calcHeightPressIncrOnColumn) then
1637      call gsv_zero(statevector_out)
1638      call gsv_copy(stateVector, stateVector_out, allowVarMismatch_opt=.true.)
1639    else
1640      ! Adjoint of calculate del Z_T/Z_M and delP_T/delP_M on the grid
1641      call gvt_transform( statevector, 'ZandP_ad' )
1642    end if
1643
1644    call gsv_deallocate( statevector_VarsLevs )
1645
1646    if (slantPath_TO_tlad) call pressureProfileMonotonicityCheck(obsSpaceData, columnTrlOnAnlIncLev)
1647
1648    call utl_tmg_stop(39)
1649
1650    call utl_tmg_stop(30)
1651
1652  end subroutine s2c_ad

1677    logical, optional          :: beSilent_opt

1681    integer :: kIndex, kIndex2, kCount, stepIndex, numStep, mykEndExtended

1687    character(len=4)     :: varName

1691    real(8), allocatable :: cols_send(:,:)

1698
1699    call utl_tmg_start(30,'--StateToColumn')
1700    call utl_tmg_start(34,'----s2c_NL')
1701
1702    if ( present(beSilent_opt) ) then
1703      beSilent = beSilent_opt
1704    else
1705      beSilent = .false.
1706    end if
1707
1708    if ( .not. beSilent ) then
1709      write(*,*) 's2c_nl: STARTING'
1710      write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
1711    end if
1712
1713    if ( .not. gsv_isAllocated(stateVector) ) then 
1714      call utl_abort('s2c_nl: stateVector must be allocated')
1715    end if
1716
1717    if (present(dealloc_opt)) then
1718      dealloc = dealloc_opt
1719    else
1720      dealloc = .true.
1721    end if

1725      numObsBatches = numObsBatches_opt
1726    else
1727      numObsBatches = 1
1728    end if
1729    if (.not. dealloc) then
1730      numObsBatches = 1 ! multiple batches only possible if dealloc=.true.
1731    end if

1735        write(*,*) 's2c_nl: WARNING! Current hco grid parameters differ from allocated interpInfo!'

1746      moveObsAtPole = moveObsAtPole_opt

1761                         stateVector%hco, stateVector%vco, mpi_local_opt=.true., &

1767      call gsv_zero(statevector_VarsLevs)

1771                                        beSilent_opt=beSilent )

1775    if ( .not. interpInfo_nl%initialized ) then
1776      call utl_tmg_stop(34)
1777      call utl_tmg_start(31,'----s2c_Setups')

1794    OBSBATCH: do obsBatchIndex = 1, numObsBatches
1795      headerIndexBeg = 1 + (obsBatchIndex - 1) * (obs_numheader(obsSpaceData) / numObsBatches)

1803                              'MPI_INTEGER', 'MPI_MAX', 'GRID', ierr)

1805                                        headerIndexBeg, headerIndexEnd, numHeader

1810        call utl_tmg_stop(34)
1811        call utl_tmg_start(31,'----s2c_Setups')

1814        call s2c_setupInterpInfo( interpInfo_nl, obsSpaceData, stateVector_VarsLevs,  &
1815                                  headerIndexBeg, headerIndexEnd, &
1816                                  timeInterpType, rejectOutsideObs, &
1817                                  inputStateVectorType='nl', &
1818                                  lastCall_opt=(obsBatchIndex==numObsBatches))
1819        if ( mmpi_myid == 0 .and. verbose ) then
1820          do stepIndex = 1, numStep
1821            write(*,*) 's2c_nl: stepIndex, allNumHeaderUsed = ',  &
1822                       stepIndex, interpInfo_nl%allNumHeaderUsed(stepIndex,:)
1823          end do

1825        call utl_tmg_stop(31)
1826        call utl_tmg_start(34,'----s2c_NL')

1831      cols_hint(:,:,:) = 0.0d0
1832
1833      ! arrays for sending/receiving time interpolated column for 1 level/variable
1834      allocate(cols_send(numHeaderMax,mmpi_nprocs))
1835      cols_send(:,:) = 0.0d0
1836
1837      allocate(cols_recv(numHeaderMax,mmpi_nprocs))

1862            !$OMP PARALLEL DO PRIVATE (procIndex, yourNumHeader)

1864              yourNumHeader = interpInfo_nl%allNumHeaderUsed(stepIndex,procIndex)
1865              if ( yourNumHeader > 0 ) then
1866                if ( varName == 'UU' ) then
1867                  call myezuvint_nl( cols_hint(1:yourNumHeader,stepIndex,procIndex), 'UU',  &
1868                                     ptr4d_r4(:,:,kIndex,stepIndex), ptr3d_UV_r4(:,:,stepIndex), &
1869                                     interpInfo_nl, kindex, stepIndex, procIndex )

1871                  call myezuvint_nl( cols_hint(1:yourNumHeader,stepIndex,procIndex), 'VV',  &
1872                                     ptr3d_UV_r4(:,:,stepIndex), ptr4d_r4(:,:,kIndex,stepIndex), &

1876                                    ptr4d_r4(:,:,kIndex,stepIndex),  &
1877                                    interpInfo_nl, kindex, stepIndex, procIndex )

1879              end if

1884          call utl_tmg_stop(35)

1913        call utl_tmg_start(36,'------s2c_NL_allToAll')

1918                                 cols_recv, nsize, 'MPI_REAL8', 'GRID', ierr)

1925        !$OMP PARALLEL DO PRIVATE (procIndex, kIndex2, headerIndex, headerIndex2)
1926        proc_loop: do procIndex = 1, mmpi_nprocs
1927          kIndex2 = stateVector_VarsLevs%allkBeg(procIndex) + kCount - 1
1928          if ( kIndex2 > stateVector_VarsLevs%allkEnd(procIndex) ) cycle proc_loop
1929          do headerIndex = 1, numHeader
1930            headerIndex2 = headerIndex + headerIndexBeg - 1
1931            allCols_ptr(kIndex2,headerIndex2) = cols_recv(headerIndex,procIndex)
1932          end do
1933        end do proc_loop
1934        !$OMP END PARALLEL DO
1935
1936      end do k_loop
1937
1938      ! impose a lower limit on HU
1939      if(col_varExist(column,'HU')) then
1940        do headerIndex = headerIndexBeg, headerIndexEnd
1941          column_ptr => col_getColumn(column,headerIndex,'HU')
1942          column_ptr(:) = max(column_ptr(:),col_rhumin)
1943        end do
1944      end if
1945
1946      ! impose a lower/upper limits on ALL cloud variables
1947      do varNameIndex = 1, vnl_numvarmaxCloud
1948        if(col_varExist(column, vnl_varNameListCloud(varNameIndex))) then
1949          do headerIndex = headerIndexBeg, headerIndexEnd
1950            column_ptr => col_getColumn(column,headerIndex, vnl_varNameListCloud(varNameIndex))
1951            column_ptr(:) = max(column_ptr(:), qlim_getMinValueCloud(vnl_varNameListCloud(varNameIndex)))
1952            column_ptr(:) = min(column_ptr(:), qlim_getMaxValueCloud(vnl_varNameListCloud(varNameIndex)))
1953          end do
1954        end if
1955      end do
1956
1957      ! Interpolate surface height separately, only exists on mpi task 0
1958      HeightSfcPresent: if ( stateVector_VarsLevs%HeightSfcPresent ) then
1959
1960        if ( mmpi_myid == 0 ) then
1961          varName = 'GZ'
1962          kIndexHeightSfc = 0    
1963          step_loop_height: do stepIndex = 1, numStep
1964
1965            if ( maxval(interpInfo_nl%allNumHeaderUsed(stepIndex,:)) == 0 ) cycle step_loop_height

2547    call gsv_getField(statevector,field_ptr)
2548    zgd(1:statevector%ni,1:statevector%nj,1:statevector%nk)= &
2549         field_ptr(1:statevector%ni,1:statevector%nj,1:statevector%nk)
2550
2551    !
2552    !- 1.  Expand field by repeating meridian 1 into into meridian ni+1
2553    !
2554    if (extraLongitude == 1) then
2555      do jk = 1, statevector%nk
2556        do jgl = 1, statevector%nj
2557          zgd(statevector%ni+1,jgl,jk) = zgd( 1,jgl,jk)
2558        end do
2559      end do
2560    end if
2561
2562    !
2563    !- 2.  Loop over all the headers
2564    !
2565    do headerIndex = 1, col_getNumCol(column)
2566
2567      !- 2.1 Find the obs positin within the analysis grid
2568      lat    = obs_headElem_r(obsSpaceData,OBS_LAT,headerIndex)
2569      lon    = obs_headElem_r(obsSpaceData,OBS_LON,headerIndex)
2570      lat_r4 = real(lat,4)
2571      lon_r4 = real(lon,4)
2572      if (lon_r4.lt.0.0         ) lon_r4 = lon_r4 + 2.0*MPC_PI_R4
2573      if (lon_r4.ge.2.*MPC_PI_R4) lon_r4 = lon_r4 - 2.0*MPC_PI_R4
2574      lat_deg_r4 = lat_r4 * MPC_DEGREES_PER_RADIAN_R4 ! Radian To Degree
2575      lon_deg_r4 = lon_r4 * MPC_DEGREES_PER_RADIAN_R4
2576      ierr = gpos_getPositionXY( stateVector % hco % EZscintID,   &
2577                                xpos_r4, ypos_r4, xpos2_r4, ypos2_r4, &
2578                                lat_deg_r4, lon_deg_r4, subGridIndex )
2579      xpos = real(xpos_r4,8)
2580      ypos = real(ypos_r4,8)
2581
2582      !- Make sure we are within bounds
2583      if ( ypos < 1.d0                        .or. &
2584           ypos > real(statevector%nj    , 8) .or. &
2585           xpos < 1.d0                        .or. &
2586           xpos > real(statevector%ni + extraLongitude, 8) ) then
2587        write(*,*) 's2c_bgcheck_bilin: Obs outside local domain for headerIndex = ', &
2588                   headerIndex
2589        write(*,*) '  obs    lat, lon position            = ', &
2590                   Lat*MPC_DEGREES_PER_RADIAN_R8, Lon*MPC_DEGREES_PER_RADIAN_R8
2591        write(*,*) '  obs    x, y     position            = ', &
2592                   xpos, ypos
2593        write(*,*) '  domain x_end, y_end bounds          = ', &
2594                   statevector%ni + extraLongitude, statevector%nj
2595        call utl_abort('s2c_bgcheck_bilin')
2596      end if
2597
2598      !- 2.2 Find the lower-left grid point next to the observation
2599      if ( xpos == real(statevector%ni + extraLongitude,8) ) then
2600        lonIndex = floor(xpos) - 1
2601      else
2602        lonIndex = floor(xpos)
2603      end if
2604
2605      if ( ypos == real(statevector%nj,8) ) then
2606        ILA = floor(ypos) - 1
2607      else
2608        ILA = floor(ypos)
2609      end if
2610
2611      !- 2.3 Compute the 4 weights of the bilinear interpolation
2612      dldx = xpos - real(lonIndex,8)
2613      dldy = ypos - real(ILA,8)
2614
2615      dlw1 = (1.d0-dldx) * (1.d0-dldy)
2616      dlw2 =       dldx  * (1.d0-dldy)
2617      dlw3 = (1.d0-dldx) *       dldy
2618      dlw4 =       dldx  *       dldy
2619
2620      !- 2.4 Interpolate the model state to the obs point
2621           
2622      do varIndex = 1, vnl_numvarmax
2623        if (.not. col_varExist(column,trim(vnl_varNameList(varIndex)))) cycle
2624        varName=trim(vnl_varNameList(varIndex))
2625        varColumn => col_getColumn(column,headerIndex,varName)
2626        
2627        if(gsv_varExist(statevector,varName)) then
2628          do jk = 1, gsv_getNumLevFromVarName(statevector,varName)      
2629              jk2=jk+gsv_getOffsetFromVarName(statevector,varName)
2630              varColumn(jk) =   dlw1*zgd(lonIndex  ,ila,jk2)  &
2631                                + dlw2*zgd(lonIndex+1,ila,jk2)  &
2632                                + dlw3*zgd(lonIndex  ,ila+1,jk2)  &
2633                                + dlw4*zgd(lonIndex+1,ila+1,jk2)
2634          end do
2635        end if
2636        
2637        nullify(varColumn)
2638      end do
2639      
2640    end do
2641
2642    deallocate(zgd)
2643
2644    call utl_tmg_stop(30)
2645
2646  end subroutine s2c_bgcheck_bilin

2878  ! s2c_setupBilinearInterp
2879  !--------------------------------------------------------------------------
2880  subroutine s2c_setupBilinearInterp(interpInfo, stateVector, headerIndex, kIndex, &
2881                                     stepIndex, procIndex, numGridpt)
2882    !
2883    !:Purpose: To determine the grid points and their associated weights
2884    !          for the bilinear horizontal interpolation. If mask is present
2885    !          we currently can only handle a single 2D mask (like for sea
2886    !          ice or SST analysis). Will abort if multiple ocean levels present.
2887    !
2888    implicit none
2889
2890    ! Arguments:
2891    type(struct_interpInfo), intent(inout) :: interpInfo
2892    type(struct_gsv)       , intent(in)    :: stateVector
2893    integer                , intent(in)    :: headerIndex, kIndex, stepIndex
2894    integer                , intent(in)    :: procIndex
2895    integer                , intent(out)   :: numGridpt(interpInfo%hco%numSubGrid)
2896
2897    ! Locals:
2898    integer :: depotIndex
2899    integer :: ierr, niP1
2900    integer :: latIndex, lonIndex, latIndex2, lonIndex2, lonIndexP1
2901    integer :: subGridIndex, subGridForInterp, numSubGridsForInterp
2902    integer :: ipoint, gridptCount
2903    integer :: latIndexVec(4), lonIndexVec(4)
2904    logical :: mask(2,2)
2905    real(8) :: WeightVec(4)
2906    real(8) :: dldx, dldy
2907    real(8) :: weightsSum
2908    real(4) :: lon_deg_r4, lat_deg_r4
2909    real(4) :: xpos_r4, ypos_r4, xpos2_r4, ypos2_r4
2910    integer, parameter :: leftIndex = 1, rightIndex = 2, bottomIndex = 1, topIndex = 2
2911
2912    numGridpt(:) = 0
2913
2914    lat_deg_r4 = real(interpInfo%stepProcData(procIndex, stepIndex)%allLat(headerIndex, kIndex) *  &
2915                 MPC_DEGREES_PER_RADIAN_R8)
2916    lon_deg_r4 = real(interpInfo%stepProcData(procIndex, stepIndex)%allLon(headerIndex, kIndex) *  &
2917                 MPC_DEGREES_PER_RADIAN_R8)
2918    ierr = gpos_getPositionXY( stateVector%hco%EZscintID,   &
2919                              xpos_r4, ypos_r4, xpos2_r4, ypos2_r4, &
2920                              lat_deg_r4, lon_deg_r4, subGridIndex )
2921
2922    ! Allow for periodicity in Longitude for global Gaussian grid
2923    if ( stateVector%hco%grtyp == 'G' .or. &
2924         (stateVector%hco%grtyp == 'Z' .and. stateVector%hco%global) ) then
2925      niP1 = statevector%ni + 1
2926    else
2927      niP1 = statevector%ni
2928    end if
2929
2930    ! Find the lower-left grid point next to the observation
2931    if ( xpos_r4 >= real(niP1) ) then
2932      xpos_r4 = real(niP1)
2933      lonIndex = niP1 - 1
2934    else if ( xpos_r4 < 1.0 ) then
2935      xpos_r4 = 1.0
2936      lonIndex = 1
2937    else
2938      lonIndex = floor(xpos_r4)
2939    end if
2940    if ( xpos2_r4 >= real(niP1) ) then
2941      xpos2_r4 = real(niP1)
2942      lonIndex2 = niP1 - 1
2943    else if ( xpos2_r4 < 1.0 ) then
2944      xpos2_r4 = 1.0
2945      lonIndex2 = 1
2946    else
2947      lonIndex2 = floor(xpos2_r4)
2948    end if
2949
2950    if ( ypos_r4 >= real(statevector%nj) ) then
2951      ypos_r4 = real(statevector%nj)
2952      latIndex = statevector%nj - 1
2953    else if ( ypos_r4 < 1.0 ) then
2954      ypos_r4 = 1.0
2955      latIndex = 1
2956    else
2957      latIndex = floor(ypos_r4)
2958    end if
2959    if ( ypos2_r4 >= real(statevector%nj) ) then
2960      ypos2_r4 = real(statevector%nj)
2961      latIndex2 = statevector%nj - 1
2962    else if ( ypos2_r4 < 1.0 ) then
2963      ypos2_r4 = 1.0
2964      latIndex2 = 1
2965    else
2966      latIndex2 = floor(ypos2_r4)
2967    end if
2968
2969    if ( stateVector%hco%grtyp == 'U' ) then
2970      if ( ypos_r4 == real(stateVector%nj/2) ) then
2971        latIndex = floor(ypos_r4) - 1
2972      end if
2973      if ( ypos2_r4 == real(stateVector%nj/2) ) then
2974        latIndex2 = floor(ypos2_r4) - 1
2975      end if
2976    end if
2977
2978    ! Handle periodicity in longitude
2979    lonIndexP1 = lonIndex + 1
2980    if ( lonIndexP1 == statevector%ni + 1 ) lonIndexP1 = 1
2981
2982    ! Check if location is in between Yin and Yang (should not happen)
2983    if ( stateVector%hco%grtyp == 'U' ) then
2984      if ( ypos_r4 > real(stateVector%nj/2) .and.  &
2985           ypos_r4 < real((stateVector%nj/2)+1) ) then
2986        write(*,*) 's2c_setupBilinearInterp: WARNING, obs position in between Yin and Yang!!!'
2987        write(*,*) '   xpos, ypos = ', xpos_r4, ypos_r4
2988      end if
2989      if ( ypos2_r4 > real(stateVector%nj/2) .and.  &
2990           ypos2_r4 < real((stateVector%nj/2)+1) ) then
2991        write(*,*) 's2c_setupBilinearInterp: WARNING, obs position in between Yin and Yang!!!'
2992        write(*,*) '   xpos2, ypos2 = ', xpos2_r4, ypos2_r4
2993      end if
2994    end if
2995
2996    if ( subGridIndex == 3 ) then
2997      ! both subGrids involved in interpolation, so first treat subGrid 1
2998      numSubGridsForInterp = 2
2999      subGridIndex = 1
3000    else
3001      ! only 1 subGrid involved in interpolation
3002      numSubGridsForInterp = 1
3003    end if
3004
3005    if ( stateVector%oceanMask%maskPresent ) then
3006      ! abort if 3D mask is present, since we may not handle this situation correctly
3007      if ( stateVector%oceanMask%nLev > 1 ) then
3008        call utl_abort('s2c_setupBilinearInterp: 3D mask present - this case not properly handled')
3009      end if
3010      ! extract the ocean mask
3011      mask(leftIndex ,bottomIndex) = stateVector%oceanMask%mask(lonIndex  ,latIndex    ,1)
3012      mask(rightIndex,bottomIndex) = stateVector%oceanMask%mask(lonIndexP1,latIndex    ,1)
3013      mask(leftIndex ,topIndex   ) = stateVector%oceanMask%mask(lonIndex  ,latIndex + 1,1)
3014      mask(rightIndex,topIndex   ) = stateVector%oceanMask%mask(lonIndexP1,latIndex + 1,1)
3015    else
3016      mask(:,:) = .true.
3017    end if
3018
3019    do subGridForInterp = 1, numSubGridsForInterp
3020
3021      WeightVec(:) = 0
3022      gridptCount = 0
3023
3024      ! Compute the 4 weights of the bilinear interpolation
3025      if ( subGridForInterp == 1 ) then
3026        ! when only 1 subGrid involved, subGridIndex can be 1 or 2
3027        dldx = real(xpos_r4,8) - real(lonIndex,8)
3028        dldy = real(ypos_r4,8) - real(latIndex,8)
3029      else
3030        ! when 2 subGrids, subGridIndex is set to 1 for 1st iteration, 2 for second
3031        subGridIndex = 2
3032        lonIndex = lonIndex2
3033        latIndex = latIndex2
3034        lonIndexP1 = lonIndex2 + 1
3035        dldx = real(xpos2_r4,8) - real(lonIndex,8)
3036        dldy = real(ypos2_r4,8) - real(latIndex,8)
3037      end if
3038
3039      if ( mask(leftIndex ,bottomIndex) ) then
3040        gridptCount = gridptCount + 1
3041        latIndexVec(gridptCount) = latIndex
3042        lonIndexVec(gridptCount) = lonIndex
3043        WeightVec(gridptCount) = (1.d0-dldx) * (1.d0-dldy)
3044      end if
3045
3046      if ( mask(rightIndex,bottomIndex) ) then
3047        gridptCount = gridptCount + 1
3048        latIndexVec(gridptCount) = latIndex
3049        lonIndexVec(gridptCount) = lonIndexP1
3050        WeightVec(gridptCount) =       dldx  * (1.d0-dldy)
3051      end if
3052
3053      if ( mask(leftIndex ,topIndex   ) ) then
3054        gridptCount = gridptCount + 1
3055        latIndexVec(gridptCount) = latIndex + 1
3056        lonIndexVec(gridptCount) = lonIndex
3057        WeightVec(gridptCount) = (1.d0-dldx) *       dldy
3058      end if
3059
3060      if ( mask(rightIndex,topIndex   ) ) then
3061        gridptCount = gridptCount + 1
3062        latIndexVec(gridptCount) = latIndex + 1
3063        lonIndexVec(gridptCount) = lonIndexP1
3064        WeightVec(gridptCount) =       dldx  *       dldy
3065      end if
3066
3067      weightsSum = sum(WeightVec(1:gridptCount))
3068      if ( weightsSum > 0.d0 ) then
3069        WeightVec(1:gridptCount) = WeightVec(1:gridptCount) / weightsSum
3070      end if
3071
3072      ! divide weight by number of subGrids
3073      WeightVec(1:gridptCount) = WeightVec(1:gridptCount) / real(numSubGridsForInterp,8)
3074
