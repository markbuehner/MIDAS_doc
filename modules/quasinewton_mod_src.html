  32      subroutine dcube(t,f,fp,ta,fa,fpa,tlower,tupper)
  33!
  34!:Purpose: [to be completed]
  35!
  36
  37!     
  38!     --- arguments
  39!     
  40      real(8) sign,den,anum,t,f,fp,ta,fa,fpa,tlower,tupper
  41!     
  42!     --- variables locales
  43!     
  44      real(8) z1,b,discri
  45!     
  46!     Using f and fp at t and ta, computes new t by cubic formula
  47!     safeguarded inside [tlower,tupper].
  48!     
  49      z1=fp+fpa-3.d0*(fa-f)/(ta-t)
  50      b=z1+fp
  51!     
  52!     first compute the discriminant (without overflow)
  53!     
  54      if (dabs(z1).le.1.d0) then
  55         discri=z1*z1-fp*fpa
  56      else
  57         discri=fp/z1
  58         discri=discri*fpa
  59         discri=z1-discri
  60         if (z1.ge.0.d0 .and. discri.ge.0.d0) then
  61            discri=dsqrt(z1)*dsqrt(discri)
  62            go to 120
  63         endif
  64         if (z1.le.0.d0 .and. discri.le.0.d0) then
  65            discri=dsqrt(-z1)*dsqrt(-discri)
  66            go to 120
  67         endif
  68         discri=-1.d0
  69      endif
  70      if (discri.lt.0.d0) then
  71         if (fp.lt.0.d0) t=tupper
  72         if (fp.ge.0.d0) t=tlower
  73         go to 900
  74      endif
  75!     
  76!     discriminant nonnegative, compute solution (without overflow)
  77!     
  78      discri=dsqrt(discri)
  79 120  if (t-ta.lt.0.d0) discri=-discri
  80      sign=(t-ta)/dabs(t-ta)
  81      if (b*sign.gt.0.d+0) then
  82         t=t+fp*(ta-t)/(b+discri)
  83      else
  84         den=z1+b+fpa
  85         anum=b-discri
  86         if (dabs((t-ta)*anum).lt.(tupper-tlower)*dabs(den)) then
  87            t=t+anum*(ta-t)/den
  88         else
  89            t=tupper
  90         endif
  91      endif
  92 900  t=dmax1(t,tlower)
  93      t=dmin1(t,tupper)
  94!     return
  95      end subroutine

  97      subroutine ddd (prosca,dtonb,dtcab,n,sscale,nm,depl,aux,jmin,jmax, &
  98                      precos,diag,ybar,sbar,izs,rzs,dzs)
  99      !
 100      !:Purpose: Calcule le produit H.g ou
 101      !
 102      !          * H est une matrice construite par la formule de bfgs inverse
 103      !            a nm memoires a partir de la matrice diagonale diag
 104      !            dans un espace hilbertien dont le produit scalaire
 105      !            est donne par prosca
 106      !            (cf. J. Nocedal, Math. of Comp. 35/151 (1980) 773-782)
 107      !          * g est un vecteur de dimension n (en general le gradient)
 108      !
 109      !          * la matrice diag apparait donc comme un preconditionneur diagonal
 110      !
 111      !          * depl = g (en entree), = H g (en sortie)
 112      !
 113      !          * la matrice H est memorisee par les vecteurs des tableaux
 114      !            ybar, sbar et les pointeurs jmin, jmax
 115      !
 116      !          * alpha(nm) est une zone de travail
 117      !
 118      !          * izs(1),rzs(1),dzs(1) sont des zones de travail pour prosca
 119      !
 120
 121      ! arguments
 122      logical sscale
 123      integer n,nm,jmin,jmax,izs(1)
 124      real rzs(1)
 125      real(8) depl(n),precos,diag(n),alpha(nm),ybar(n,1), &
 126          sbar(n,1),aux(n),dzs(1)
 127      external prosca,dtonb,dtcab
 128!
 129!         variables locales
 130!
 131      integer jfin,i,j,jp
 132      real(8) r,ps
 133!
 134      jfin=jmax
 135      if (jfin.lt.jmin) jfin=jmax+nm
 136!
 137!         phase de descente
 138!
 139      do 100 j=jfin,jmin,-1
 140          jp=j
 141          if (jp.gt.nm) jp=jp-nm
 142          call prosca (n,depl,sbar(1,jp),ps,izs,rzs,dzs)
 143          r=ps
 144          alpha(jp)=r
 145!$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(I) 
 146          do 20 i=1,n
 147              depl(i)=depl(i)-r*ybar(i,jp)
 14820        continue
 149!$OMP END PARALLEL DO
 150100   continue
 151!
 152!         preconditionnement
 153!
 154      if (sscale) then
 155!$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(I) 
 156          do 150 i=1,n
 157              depl(i)=depl(i)*precos
 158  150     continue
 159!$OMP END PARALLEL DO
 160      else
 161          call dtonb (n,depl,aux,izs,rzs,dzs)
 162!$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(I) 
 163          do 151 i=1,n
 164              aux(i)=aux(i)*diag(i)
 165  151     continue
 166!$OMP END PARALLEL DO
 167          call dtcab (n,aux,depl,izs,rzs,dzs)
 168      endif
 169!
 170!         remontee
 171!
 172      do 200 j=jmin,jfin
 173          jp=j
 174          if (jp.gt.nm) jp=jp-nm
 175          call prosca (n,depl,ybar(1,jp),ps,izs,rzs,dzs)
 176          r=alpha(jp)-ps
 177!$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(I) 
 178          do 120 i=1,n
 179              depl(i)=depl(i)+r*sbar(i,jp)
 180120       continue
 181!$OMP END PARALLEL DO
 182200   continue
 183
 184      end subroutine ddd

 186      subroutine ddds (prosca,dtonb,dtcab,n,sscale,nm,depl,aux,jmin, &
 187                       jmax,precos,diag,ybar,sbar,izs,rzs,dzs)
 188      !
 189      !
 190      !:Purpose: This subroutine has the same role as ddd (computation of the
 191      !          product H.g). It supposes however that the (y,s) pairs are not
 192      !          stored in core memory, but on a devise chosen by the user.
 193      !
 194
 195!
 196!         arguments
 197!
 198      logical sscale
 199      integer n,nm,jmin,jmax,izs(1)
 200      real rzs(1)
 201      real(8) depl(n),precos,diag(n),alpha(nm),ybar(n),sbar(n), &
 202          aux(n),dzs(1)
 203      external prosca,dtonb,dtcab
 204!
 205!         variables locales
 206!
 207      integer jfin,i,j,jp
 208      real(8) r,ps
 209!
 210      jfin=jmax
 211      if (jfin.lt.jmin) jfin=jmax+nm
 212!
 213!         phase de descente
 214!
 215      do 100 j=jfin,jmin,-1
 216          jp=j
 217          if (jp.gt.nm) jp=jp-nm
 218          call prosca (n,depl,sbar,ps,izs,rzs,dzs)
 219          r=ps
 220          alpha(jp)=r
 221          do 20 i=1,n
 222              depl(i)=depl(i)-r*ybar(i)
 22320        continue
 224100   continue
 225!
 226!         preconditionnement
 227!
 228      if (sscale) then
 229          do 150 i=1,n
 230              depl(i)=depl(i)*precos
 231  150     continue
 232      else
 233          call dtonb (n,depl,aux,izs,rzs,dzs)
 234          do 151 i=1,n
 235              aux(i)=aux(i)*diag(i)
 236  151     continue
 237          call dtcab (n,aux,depl,izs,rzs,dzs)
 238      endif
 239!
 240!         remontee
 241!
 242      do 200 j=jmin,jfin
 243          jp=j
 244          if (jp.gt.nm) jp=jp-nm
 245          call prosca (n,depl,ybar(1),ps,izs,rzs,dzs)
 246          r=alpha(jp)-ps
 247          do 120 i=1,n
 248              depl(i)=depl(i)+r*sbar(i)
 249120       continue
 250200   continue
 251
 252      end subroutine ddds

 254      subroutine mupdts (sscale,inmemo,n,m,nrz)
 255
 256!
 257!         arguments
 258!
 259      logical sscale,inmemo
 260      integer n,m,nrz
 261!----
 262!
 263!     On entry:
 264!       sscale: .true. if scalar initial scaling,
 265!               .false. if diagonal initial scaling
 266!       n:      number of variables
 267!
 268!     This routine has to return:
 269!       m:      the number of updates to form the approximate Hessien H,
 270!       inmemo: .true., if the vectors y and s used to form H are stored
 271!                  in core memory,
 272!               .false. otherwise (storage of y and s on disk, for
 273!                  instance).
 274!
 275!----
 276!
 277      if (n.eq.0) then
 278         m=0
 279      else
 280         if (sscale) then
 281            m=(nrz-3*n)/(2*n)
 282         else
 283            m=(nrz-4*n)/(2*n)
 284         endif
 285      endif
 286      inmemo=.true.
 287      ! return
 288      end subroutine mupdts

 290      subroutine qna_n1qn3 (simul,prosca,dtonb,dtcab,n,x,f,g,dxmin,df1, &
 291           epsg,impres,io,mode,niter,nsim,iz,dz,ndz, &
 292           izs,rzs,dzs)
 293      !     
 294      !:Purpose: N1QN3, Version 2.0c, June 1995
 295      !          Jean Charles Gilbert, Claude Lemarechal, INRIA.
 296      !     
 297      !          Double precision version of M1QN3.
 298      !     
 299      !          N1qn3 has two running modes: the SID (Scalar Initial Scaling) mode
 300      !          and the DIS (Diagonal Initial Scaling) mode. Both do not require
 301      !          the same amount of storage, the same subroutines, ...
 302      !          In the description below, items that differ in the DIS mode with
 303      !          respect to the SIS mode are given in brakets.
 304      !     
 305      !          Use the following subroutines:
 306      !
 307      !          * N1QN3A
 308      !          * DDD, DDDS
 309      !          * NLIS0 + DCUBE (Dec 88)
 310      !          * MUPDT
 311      !     
 312      !          The following routines are proposed to the user in case the
 313      !          Euclidean scalar product is used:
 314      !          DUCLID, DTONBE, DTCABE.
 315      !     
 316      !          La sous-routine N1QN3 est une interface entre le programme
 317      !          appelant et la sous-routine N1QN3A, le minimiseur proprement dit.
 318      !     
 319      !          Le module PROSCA est sense realiser le produit scalaire de deux
 320      !          vecteurs de Rn; le module DTONB est sense realiser le changement
 321      !          de coordonnees correspondant au changement de bases: base
 322      !          euclidienne -> base orthonormale (pour le produit scalaire
 323      !          PROSCA); le module CTBAB fait la transformation inverse: base
 324      !          orthonormale -> base euclidienne.
 325      !     
 326      !          Iz is an integer working zone for N1QN3A, its dimension is 5.
 327      !          It is formed of 5 scalars that are set by the optimizer:
 328      !
 329      !          - the dimension of the problem,
 330      !          - a identifier of the scaling mode,
 331      !          - the number of updates,
 332      !          - two pointers.
 333      !     
 334      !          Dz est la zone de travail pour N1QN3A, de dimension ndz.
 335      !          Elle est subdivisee en
 336      !
 337      !          - 3 [ou 4] vecteurs de dimension n: d,gg,[diag,]aux
 338      !          - m vecteurs de dimension n: ybar
 339      !          - m vecteurs de dimension n: sbar
 340      !     
 341      !          m est alors le plus grand entier tel que
 342      !
 343      !          - m*(2*n+1)+3*n .le. ndz [m*(2*n+1)+4*n .le. ndz)]
 344      !          - soit m := (ndz-3*n) / (2*n+1) [m := (ndz-4*n) / (2*n+1)].
 345      !          Il faut avoir m >= 1, donc ndz >= 5n+1 [ndz >= 6n+1].
 346      !     
 347      !          A chaque iteration la metrique est formee a partir d'un multiple
 348      !          de l'identite [d'une matrice diagonale] D qui est mise a jour m
 349      !          fois par la formule de BFGS en utilisant les m couples {y,s} les
 350      !          plus recents.
 351      !     
 352
 353!     
 354!     arguments
 355!     
 356      integer n,impres,io,mode,niter,nsim,iz(5),ndz,izs(1)
 357      real rzs(1)
 358      real(8) x(n),f,g(n),dxmin,df1,epsg,dz(ndz),dzs(1)
 359      external simul,prosca,dtonb,dtcab
 360!     
 361!     variables locales
 362!     
 363      logical inmemo,sscale
 364      integer ntravu,id,igg,idiag,iaux,iybar,isbar,m,mmemo,ntotal,ierr
 365      integer m_max
 366      real(8) d1,d2,ps
 367!     
 368!---- impressions initiales et controle des arguments
 369!     
 370      write(io,*) '--------------------------------------------------'
 371      write(io,*) 'N1QN3: calling modified MPI version of modulopt!!!'
 372      write(io,*) '--------------------------------------------------'
 373
 374      call rpn_comm_allreduce(n,ntotal,1,"mpi_integer", &
 375                              "mpi_max","GRID",ierr)
 376
 377      if (impres.ge.1) &
 378           write (io,900) n,dxmin,df1,epsg,niter,nsim,impres
 379 900  format (/" N1QN3 (Version 2.0c, June 1995): entry point"/ &
 380           5x,"dimension of the problem (n):",i9/  &
 381           5x,"absolute precision on x (dxmin):",d9.2/ &
 382           5x,"expected decrease for f (df1):",d9.2/ &
 383           5x,"relative precision on g (epsg):",d9.2/ &
 384           5x,"maximal number of iterations (niter):",i6/ &
 385           5x,"maximal number of simulations (nsim):",i6/ &
 386           5x,"printing level (impres):",i4)
 387      if (ntotal.le.0.or.niter.le.0.or.nsim.le.0.or.dxmin.le.0.d+0.or. &
 388           epsg.le.0.d+0.or.epsg.gt.1.d+0.or.mode.lt.0.or.  &
 389           mode.gt.3) then
 390         mode=2
 391         write (io,901)
 392 901     format (/" >>> n1qn3: inconsistent call")
 393         return
 394      endif
 395!     
 396!---- what method
 397!     
 398      if (mod(mode,2).eq.0) then
 399         if (impres.ge.1) write (io,920)
 400 920     format (/" n1qn3: Diagonal Initial Scaling mode")
 401         sscale=.false.
 402      else
 403         if (impres.ge.1) write (io,921)
 404 921     format (/" n1qn3: Scalar Initial Scaling mode")
 405         sscale=.true.
 406      endif
 407!     
 408      if ( n.gt.0 .and. &
 409           ((ndz.lt.5*n+1).or.((.not.sscale).and.(ndz.lt.6*n+1))) ) then
 410         mode=2
 411         write (io,902)
 412 902     format (/" >>> n1qn3: not enough memory allocated")
 413         return
 414      endif
 415!     
 416!---- Compute m
 417!     
 418      call mupdts (sscale,inmemo,n,m,ndz)
 419      call rpn_comm_allreduce(m,m_max,1,"mpi_integer", &
 420                              "mpi_max","GRID",ierr)
 421      if (m.ne.m_max) then
 422         write(io,*) 'replacing value of m, ',m,' with ',m_max
 423         m=m_max
 424      endif
 425!     
 426!     --- Check the value of m (if (y,s) pairs in core, m will be >= 1)
 427!     
 428      if (m.lt.1) then
 429         mode=2
 430         write (io,9020)
 431 9020    format (/" >>> n1qn3: m is set too small in mupdts")
 432         return
 433      endif
 434!     
 435!     --- mmemo = number of (y,s) pairs in core memory
 436!     
 437      mmemo=1
 438      if (inmemo) mmemo=m
 439!     
 440      ntravu=2*(2+mmemo)*n
 441      if (sscale) ntravu=ntravu-n
 442      write (io,903) ndz,ntravu,m
 443 903  format (/5x,"allocated memory (ndz) :",i9/ &
 444           5x,"used memory :           ",i9/ &
 445           5x,"number of updates :     ",i9)
 446      if (ndz.lt.ntravu) then
 447         mode=2
 448         write (io,902)
 449         return
 450      endif
 451!     
 452      if (impres.ge.1) then
 453         if (inmemo) then
 454            write (io,907)
 455         else
 456            write (io,908)
 457         endif
 458      endif
 459 907  format (5x,"(y,s) pairs are stored in core memory")
 460 908  format (5x,"(y,s) pairs are stored by the user")
 461!     
 462!---- cold start or warm restart ?
 463!     check iz: iz(1)=n, iz(2)=(0 if DIS, 1 if SIS),
 464!     iz(3)=m, iz(4)=jmin, iz(5)=jmax
 465!     
 466      if (mode/2.eq.0) then
 467         if (impres.ge.1) write (io,922)
 468      else
 469         iaux=0
 470         if (sscale) iaux=1
 471!         if (iz(1).ne.n.or.iz(2).ne.iaux.or.iz(3).ne.m.or.iz(4).lt.1 &
 472!              .or.iz(5).lt.1.or.iz(4).gt.iz(3).or.iz(5).gt.iz(3)) then
 473         if (iz(2).ne.iaux.or.iz(3).ne.m.or.iz(4).lt.1 &
 474              .or.iz(5).lt.1.or.iz(4).gt.iz(3).or.iz(5).gt.iz(3)) then
 475            mode=2
 476            write(*,*) 'iz=',iz(:)
 477            write(*,*) 'n,iaux,m=',n,iaux,m
 478            if (impres.ge.1) write (io,923)
 479            return
 480         endif
 481         if (impres.ge.1) write (io,924)
 482      endif
 483 922  format (/" n1qn3: cold start"/,1x)
 484 923  format (/" >>> n1qn3: inconsistent iz for a warm restart")
 485 924  format (/" n1qn3: warm restart"/,1x)
 486      iz(1)=n
 487      iz(2)=0
 488      if (sscale) iz(2)=1
 489      iz(3)=m
 490!     
 491!---- split the working zone dz
 492!     
 493      idiag=1
 494      iybar=idiag+n
 495      if (sscale) iybar=1
 496      isbar=iybar+n*mmemo
 497      id=isbar+n*mmemo
 498      igg=id+n
 499      iaux=igg+n
 500!     
 501!---- call the optimization code
 502!     
 503      call n1qn3a (simul,prosca,dtonb,dtcab,n,x,f,g,dxmin,df1,epsg, &
 504           impres,io,mode,niter,nsim,inmemo,iz(3),iz(4),iz(5), &
 505           dz(id),dz(igg),dz(idiag),dz(iaux), &
 506           dz(iybar),dz(isbar),izs,rzs,dzs)
 507!     
 508!---- impressions finales
 509!     
 510      if (impres.ge.1) write (io,905) mode,niter,nsim,epsg
 511 905  format (/,1x,79("-")/  &
 512           /" n1qn3: output mode is ",i2 &
 513           /5x,"number of iterations: ",i4 &
 514           /5x,"number of simulations: ",i6 &
 515           /5x,"realized relative precision on g: ",d9.2)
 516      call prosca (n,x,x,ps,izs,rzs,dzs)
 517      d1=dsqrt(ps)
 518      call prosca (n,g,g,ps,izs,rzs,dzs)
 519      d2=dsqrt(ps)
 520      if (impres.ge.1) write (io,906) d1,f,d2
 521 906  format (5x,"norm of x = ",d15.8 &
 522           /5x,"f         = ",d15.8 &
 523           /5x,"norm of g = ",d15.8) 
 524      !return
 525      end subroutine qna_n1qn3

 527      subroutine n1qn3a (simul,prosca,dtonb,dtcab,n,x,f,g,dxmin,df1, &
 528                         epsg,impres,io,mode,niter,nsim,inmemo,m,jmin, &
 529                         jmax,d,gg,diag,aux,ybar,sbar,izs,rzs,dzs)
 530     !
 531     !:Purpose: Code d'optimisation proprement dit.
 532     !
 533
 534!
 535!         arguments
 536!
 537      logical inmemo
 538      integer n,impres,io,mode,niter,nsim,m,jmin,jmax,izs(1)
 539      real rzs(1)
 540      real(8) x(n),f,g(n),dxmin,df1,epsg,d(n),gg(n),diag(n), &
 541          aux(n),ybar(n,1),sbar(n,1),dzs(1)
 542      external simul,prosca,dtonb,dtcab
 543!
 544!         variables locales
 545!
 546      logical sscale,cold,warm,ntotal
 547      integer i,itmax,moderl,isim,jcour,indic,ierr,impresmax
 548      real(8) d1,t,tmin,tmin_mpiglobal,tmax,gnorm,eps1,ff, &
 549           preco,precos,ys,den, &
 550           dk,dk1,ps,ps2,hp0
 551!
 552!         parametres
 553!
 554      real(8) rm1,rm2
 555      parameter (rm1=0.0001d+0,rm2=0.9d+0)
 556      real(8) pi
 557      parameter (pi=3.1415927d+0)
 558      real(8) rmin
 559!
 560!---- initialisation
 561!
 562      rmin=1.d-20
 563!
 564      sscale=.true.
 565      if (mod(mode,2).eq.0) sscale=.false.
 566!
 567      warm=.false.
 568      if (mode/2.eq.1) warm=.true.
 569      cold=.not.warm
 570!
 571      itmax=niter
 572      niter=0
 573      isim=1
 574      eps1=1.d+0
 575!
 576      call rpn_comm_allreduce(impres,impresmax,1,"mpi_integer", &
 577                              "mpi_max","GRID",ierr)
 578      call rpn_comm_allreduce(n,ntotal,1,"mpi_integer", &
 579                              "mpi_max","GRID",ierr)
 580!
 581      call prosca (n,g,g,ps,izs,rzs,dzs)
 582      gnorm=dsqrt(ps)
 583      if (impres.ge.1) write (io,900) f,gnorm
 584  900 format (5x,"f         = ",d15.8 &
 585             /5x,"norm of g = ",d15.8)
 586      if (gnorm.lt.rmin) then
 587          mode=2
 588          if (impres.ge.1) write (io,901)
 589          goto 1000
 590      endif
 591  901 format (/" >>> n1qn3a: initial gradient is too small")
 592!
 593!     --- initialisation pour ddd
 594!
 595      if (cold) then
 596          jmin=1
 597          jmax=0
 598      endif
 599      jcour=1
 600      if (inmemo) jcour=jmax
 601!
 602!     --- mise a l'echelle de la premiere direction de descente
 603!
 604      if (cold) then
 605!
 606!         --- use Fletcher's scaling and initialize diag to 1.
 607!
 608          precos=2.d+0*df1/gnorm**2
 609          do 10 i=1,n
 610              d(i)=-g(i)*precos
 611              diag(i)=1.d+0
 612   10     continue
 613          if (impres.ge.5) write(io,902) precos
 614  902     format (/" n1qn3a: descent direction -g: precon = ",d10.3)
 615      else
 616!
 617!         --- use the matrix stored in [diag and] the (y,s) pairs
 618!
 619          if (sscale) then
 620              call prosca (n,ybar(1,jcour),ybar(1,jcour),ps,izs,rzs,dzs)
 621              precos=1.d+0/ps
 622          endif
 623          do 11 i=1,n
 624              d(i)=-g(i)
 625  11      continue
 626          if (inmemo) then
 627              call ddd (prosca,dtonb,dtcab,n,sscale,m,d,aux,jmin,jmax, &
 628                       precos,diag,ybar,sbar,izs,rzs,dzs)
 629          else
 630              call ddds (prosca,dtonb,dtcab,n,sscale,m,d,aux,jmin,jmax, &
 631                         precos,diag,ybar,sbar,izs,rzs,dzs)
 632          endif
 633      endif
 634!
 635      if (impres.eq.3) then
 636          write(io,903)
 637          write(io,904)
 638      endif
 639      if (impres.eq.4) write(io,903)
 640  903 format (/,1x,79("-"))
 641  904 format (1x)
 642!
 643!     --- initialisation pour nlis0
 644!
 645      tmax=1.d+20
 646      call prosca (n,d,g,hp0,izs,rzs,dzs)
 647      if (hp0.ge.0.d+0) then
 648          mode=7
 649          if (impres.ge.1) write (io,905) niter,hp0
 650          goto 1000
 651      endif
 652  905 format (/" >>> n1qn3 (iteration ",i2,"): " &
 653              /5x," the search direction d is not a ", &
 654               "descent direction: (g,d) = ",d12.5)
 655!
 656!     --- compute the angle (-g,d)
 657!
 658      if (warm.and.impresmax.ge.5) then
 659          call prosca (n,g,g,ps,izs,rzs,dzs)
 660          ps=dsqrt(ps)
 661          call prosca (n,d,d,ps2,izs,rzs,dzs)
 662          ps2=dsqrt(ps2)
 663          ps=hp0/ps/ps2
 664          ps=dmin1(-ps,1.d+0)
 665          ps=dacos(ps)
 666          d1=ps*180.d+0/pi
 667          if(impres.ge.5) write (io,906) sngl(d1)
 668      endif
 669  906 format (/" n1qn3: descent direction d: ", &
 670              "angle(-g,d) = ",f5.1," degrees")
 671!
 672!---- Debut de l'iteration. on cherche x(k+1) de la forme x(k) + t*d,
 673!     avec t > 0. On connait d.
 674!
 675!         Debut de la boucle: etiquette 100,
 676!         Sortie de la boucle: goto 1000.
 677!
 678100   niter=niter+1
 679      if (impres.lt.0) then
 680          if (mod(niter,-impres).eq.0) then
 681              indic=1
 682              call simul (indic,n,x,f,g,izs,rzs,dzs)
 683          endif
 684      endif
 685      if (impres.ge.5) write(io,903)
 686      if (impres.ge.4) write(io,904)
 687      if (impres.ge.3) write (io,910) niter,isim,f,hp0
 688  910 format (" n1qn3: iter ",i3,", simul ",i3, &
 689              ", f=",d15.8,", h'(0)=",d12.5)
 690!$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(I) 
 691      do 101 i=1,n
 692          gg(i)=g(i)
 693101   continue
 694!$OMP END PARALLEL DO
 695      ff=f
 696!
 697!     --- recherche lineaire et nouveau point x(k+1)
 698!
 699      if (impres.ge.5) write (io,911)
 700  911 format (/" n1qn3: line search")
 701!
 702!         --- calcul de tmin
 703!
 704      tmin=0.d+0
 705      do 200 i=1,n
 706          tmin=max(tmin,dabs(d(i)))
 707200   continue
 708      call rpn_comm_allreduce(tmin,tmin_mpiglobal,1, &
 709                              "mpi_double_precision", &
 710                              "mpi_max","GRID",ierr)
 711      tmin = tmin_mpiglobal
 712
 713      tmin=dxmin/tmin
 714      t=1.d+0
 715      d1=hp0
 716!
 717      call nlis0 (n,simul,prosca,x,f,d1,t,tmin,tmax,d,g,rm2,rm1, &
 718                 impres,io,moderl,isim,nsim,aux,izs,rzs,dzs)
 719!
 720!         --- nlis0 renvoie les nouvelles valeurs de x, f et g
 721!
 722      if (moderl.ne.0) then
 723          if (moderl.lt.0) then
 724!
 725!             --- calcul impossible
 726!                 t, g: ou les calculs sont impossibles
 727!                 x, f: ceux du t_gauche (donc f <= ff)
 728!
 729              mode=moderl
 730          elseif (moderl.eq.1) then
 731!
 732!             --- descente bloquee sur tmax
 733!                 [sortie rare (!!) d'apres le code de nlis0]
 734!
 735              mode=3
 736              if (impres.ge.1) write(io,912) niter
 737  912         format (/" >>> n1qn3 (iteration ",i3, &
 738                      "): line search blocked on tmax: ", &
 739                      "decrease the scaling")
 740          elseif (moderl.eq.4) then
 741!
 742!             --- nsim atteint
 743!                 x, f: ceux du t_gauche (donc f <= ff)
 744!
 745              mode=5
 746          elseif (moderl.eq.5) then
 747!
 748!             --- arret demande par l'utilisateur (indic = 0)
 749!                 x, f: ceux en sortie du simulateur
 750!
 751              mode=0
 752          elseif (moderl.eq.6) then
 753!
 754!             --- arret sur dxmin ou appel incoherent
 755!                 x, f: ceux du t_gauche (donc f <= ff)
 756!
 757              mode=6
 758          endif
 759          goto 1000
 760      endif
 761!
 762! NOTE: stopping tests are now done after having updated the matrix, so
 763! that update information can be stored in case of a later warm restart
 764!
 765!     --- mise a jour de la matrice
 766!
 767      if (m.gt.0) then
 768!
 769!         --- mise a jour des pointeurs
 770!
 771          jmax=jmax+1
 772          if (jmax.gt.m) jmax=jmax-m
 773          if ((cold.and.niter.gt.m).or.(warm.and.jmin.eq.jmax)) then
 774              jmin=jmin+1
 775              if (jmin.gt.m) jmin=jmin-m
 776          endif
 777          if (inmemo) jcour=jmax
 778!
 779!         --- y, s et (y,s)
 780!
 781!$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(I) 
 782          do 400 i=1,n
 783              sbar(i,jcour)=t*d(i)
 784              ybar(i,jcour)=g(i)-gg(i)
 785400       continue
 786!$OMP END PARALLEL DO
 787          if (impresmax.ge.5) then
 788              call prosca (n,sbar(1,jcour),sbar(1,jcour),ps,izs,rzs,dzs)
 789              dk1=dsqrt(ps)
 790              if (impres.ge.5.and.niter.gt.1) write (io,930) dk1/dk
 791  930         format (/" n1qn3: convergence rate, s(k)/s(k-1) = ", &
 792                      d12.5)
 793              dk=dk1
 794          endif
 795          call prosca (n,ybar(1,jcour),sbar(1,jcour),ys,izs,rzs,dzs)
 796          if (ys.le.0.d+0) then
 797              mode=7
 798              if (impres.ge.1) write (io,931) niter,ys
 799  931         format (/" >>> n1qn3 (iteration ",i2, &
 800                      "): the scalar product (y,s) = ",d12.5 &
 801                      /27x,"is not positive")
 802              goto 1000
 803          endif
 804!
 805!         --- ybar et sbar
 806!
 807          d1=dsqrt(1.d+0/ys)
 808!$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(I) 
 809          do 410 i=1,n
 810              sbar(i,jcour)=d1*sbar(i,jcour)
 811              ybar(i,jcour)=d1*ybar(i,jcour)
 812  410     continue
 813!$OMP END PARALLEL DO
 814!
 815!         --- compute the scalar or diagonal preconditioner
 816!
 817          if (impres.ge.5) write(io,932)
 818  932     format (/" n1qn3: matrix update:")
 819!
 820!             --- Here is the Oren-Spedicato factor, for scalar scaling
 821!
 822          if (sscale) then
 823              call prosca (n,ybar(1,jcour),ybar(1,jcour),ps,izs,rzs,dzs)
 824              precos=1.d+0/ps
 825!
 826              if (impres.ge.5) write (io,933) precos
 827  933         format (5x,"Oren-Spedicato factor = ",d10.3)
 828!
 829!             --- Scale the diagonal to Rayleigh's ellipsoid.
 830!                 Initially (niter.eq.1) and for a cold start, this is
 831!                 equivalent to an Oren-Spedicato scaling of the
 832!                 identity matrix.
 833!
 834          else
 835              call dtonb (n,ybar(1,jcour),aux,izs,rzs,dzs)
 836              ps=0.d0
 837              do 420 i=1,n
 838                  ps=ps+diag(i)*aux(i)*aux(i)
 839  420         continue
 840              call mmpi_allreduce_sumreal8scalar(ps,"GRID")
 841              d1=1.d0/ps
 842              if (impres.ge.5) then
 843                  write (io,934) d1
 844  934             format(5x,"fitting the ellipsoid: factor = ",d10.3)
 845              endif
 846!$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(I) 
 847              do 421 i=1,n
 848                  diag(i)=diag(i)*d1
 849  421         continue
 850!$OMP END PARALLEL DO
 851!
 852!             --- update the diagonal
 853!                 (gg is used as an auxiliary vector)
 854!
 855              call dtonb (n,sbar(1,jcour),gg,izs,rzs,dzs)
 856              ps=0.d0
 857              do 430 i=1,n
 858                  ps=ps+gg(i)*gg(i)/diag(i)
 859  430         continue
 860              call mmpi_allreduce_sumreal8scalar(ps,"GRID")
 861              den=ps
 862!$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(I) 
 863              do 431 i=1,n
 864                  diag(i)=1.d0/ &
 865                         (1.d0/diag(i)+aux(i)**2-(gg(i)/diag(i))**2/den)
 866                  if (diag(i).le.0.d0) then
 867                      if (impres.ge.5) write (io,935) i,diag(i),rmin
 868                      diag(i)=rmin
 869                  endif
 870  431         continue
 871!$OMP END PARALLEL DO
 872  935         format (/" >>> n1qn3-WARNING: diagonal element ",i8, &
 873                       " is negative (",d10.3,"), reset to ",d10.3)
 874!
 875              if (impresmax.ge.5) then
 876                  ps=0.d0
 877                  do 440 i=1,n
 878                      ps=ps+diag(i)
 879  440             continue
 880                  call mmpi_allreduce_sumreal8scalar(ps,"GRID")
 881                  ps=ps/ntotal
 882                  preco=ps
 883!
 884                  ps2=0.d0
 885                  do 441 i=1,n
 886                      ps2=ps2+(diag(i)-ps)**2
 887  441             continue
 888                  call mmpi_allreduce_sumreal8scalar(ps2,"GRID")
 889                  ps2=dsqrt(ps2/ntotal)
 890                  if (impres.ge.5) write (io,936) preco,ps2
 891  936             format (5x,"updated diagonal: average value = ",d10.3, &
 892                         ", sqrt(variance) = ",d10.3)
 893              endif
 894          endif
 895      endif
 896!
 897!     --- tests d'arret
 898!
 899      call prosca(n,g,g,ps,izs,rzs,dzs)
 900      eps1=ps
 901      eps1=dsqrt(eps1)/gnorm
 902!
 903      if (impres.ge.5) write (io,940) eps1
 904  940 format (/" n1qn3: stopping criterion on g: ",d12.5)
 905      if (eps1.lt.epsg) then
 906          mode=1
 907          goto 1000
 908      endif
 909      if (niter.eq.itmax) then
 910          mode=4
 911          if (impres.ge.1) write (io,941) niter
 912  941     format (/" >>> n1qn3 (iteration ",i3, &
 913                  "): maximal number of iterations")
 914          goto 1000
 915      endif
 916      if (isim.gt.nsim) then
 917          mode=5
 918          if (impres.ge.1) write (io,942) niter,isim
 919  942     format (/" >>> n1qn3 (iteration ",i3,"): ",i6, &
 920                  " simulations (maximal number reached)")
 921          goto 1000
 922      endif
 923!
 924!     --- calcul de la nouvelle direction de descente d = - H.g
 925!
 926      if (m.eq.0) then
 927          preco=2.d0*(ff-f)/(eps1*gnorm)**2
 928!$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(I) 
 929          do 500 i=1,n
 930              d(i)=-g(i)*preco
 931  500     continue
 932!$OMP END PARALLEL DO
 933      else
 934!$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(I) 
 935          do 510 i=1,n
 936              d(i)=-g(i)
 937  510     continue
 938!$OMP END PARALLEL DO
 939          if (inmemo) then
 940              call ddd (prosca,dtonb,dtcab,n,sscale,m,d,aux,jmin,jmax, &
 941                        precos,diag,ybar,sbar,izs,rzs,dzs)
 942          else
 943              call ddds (prosca,dtonb,dtcab,n,sscale,m,d,aux,jmin,jmax, &
 944                         precos,diag,ybar,sbar,izs,rzs,dzs)
 945          endif
 946      endif
 947!
 948!         --- test: la direction d est-elle de descente ?
 949!             hp0 sera utilise par nlis0
 950!
 951      call prosca (n,d,g,hp0,izs,rzs,dzs)
 952      if (hp0.ge.0.d+0) then
 953          mode=7
 954          if (impres.ge.1) write (io,905) niter,hp0
 955          goto 1000
 956      endif
 957      if (impresmax.ge.5) then
 958          call prosca (n,g,g,ps,izs,rzs,dzs)
 959          ps=dsqrt(ps)
 960          call prosca (n,d,d,ps2,izs,rzs,dzs)
 961          ps2=dsqrt(ps2)
 962          ps=hp0/ps/ps2
 963          ps=dmin1(-ps,1.d+0)
 964          ps=dacos(ps)
 965          d1=ps
 966          d1=d1*180.d0/pi
 967          if (impres.ge.5) write (io,906) sngl(d1)
 968      endif
 969!
 970!---- on poursuit les iterations
 971!
 972      goto 100
 973!
 974!---- retour
 975!
 976 1000 continue
 977      nsim=isim
 978      epsg=eps1
 979
 980      end subroutine n1qn3a

 982      subroutine nlis0 (n,simul,prosca,xn,fn,fpn,t,tmin,tmax,d,g, &
 983                        amd,amf,imp,io,logic,nap,napmax,x,izs,rzs,dzs)
 984      !
 985      !:Purpose: en sortie logic =
 986      !
 987      !          * 0          descente serieuse
 988      !          * 1          descente bloquee
 989      !          * 4          nap > napmax
 990      !          * 5          retour a l'utilisateur
 991      !          * 6          fonction et gradient pas d'accord
 992      !          * < 0        contrainte implicite active
 993      !
 994
 995!!
 996!!--- arguments
 997!!
 998      external simul,prosca
 999      integer n,imp,io,logic,nap,napmax,izs(*)
1000      real rzs(*)
1001      real(8) xn(n),fn,fpn,t,tmin,tmax,d(n),g(n),amd,amf,x(n),dzs(*)
1002!
1003! --- variables locales
1004!
1005      logical lfound,lfound2
1006      integer i,indic,indica,indicd,ierr,ntotal
1007      real(8) tesf,tesd,tg,fg,fpg,td,ta,fa,fpa,d2,f,fp,ffn,fd, &
1008       fpd,z,test,barmin,barmul,barmax,barr,gauche,droite,taa,ps
1009!
1010 1000 format (/4x,9h nlis0   ,4x,4hfpn=,d10.3,4h d2=,d9.2, &
1011       7h  tmin=,d9.2,6h tmax=,d9.2)
1012 1001 format (/4x,6h mlis0,3x,"stop on tmin",8x,  &
1013         "step",11x,"functions",5x,"derivatives")
1014 1002 format (4x,6h nlis0,37x,d10.3,2d11.3)
1015 1003 format (4x,6h nlis0,d14.3,2d11.3)
1016 1004 format (4x,6h nlis0,37x,d10.3,7h indic=,i3)
1017 1005 format (4x,6h nlis0,14x,2d18.8,d11.3)
1018 1006 format (4x,6h nlis0,14x,d18.8,12h      indic=,i3)
1019 1007 format (/4x,6h mlis0,10x,"tmin forced to tmax")
1020 1008 format (/4x,6h mlis0,10x,"inconsistent call")
1021      call rpn_comm_allreduce(n,ntotal,1,"mpi_integer", &
1022                              "mpi_max","GRID",ierr)
1023      if (ntotal.gt.0 .and. fpn.lt.0.d0 .and. t.gt.0.d0 &
1024       .and. tmax.gt.0.d0 .and. amf.gt.0.d0 &
1025       .and. amd.gt.amf .and. amd.lt.1.d0) go to 5
1026      logic=6
1027      go to 999
1028    5 tesf=amf*fpn
1029      tesd=amd*fpn
1030      barmin=0.01d0
1031      barmul=5.d0
1032      barmax=0.3d0
1033      barr=barmin
1034      td=0.d0
1035      tg=0.d0
1036      fg=fn
1037      fpg=fpn
1038      ta=0.d0
1039      fa=fn
1040      fpa=fpn
1041      call prosca (n,d,d,ps,izs,rzs,dzs)
1042      d2=ps
1043!
1044!               elimination d'un t initial ridiculement petit
1045!
1046      if (t.gt.tmin) go to 20
1047      t=tmin
1048      if (t.le.tmax) go to 20
1049      if (imp.gt.0) write (io,1007)
1050      tmin=tmax
1051   20 if (fn+t*fpn.lt.fn+0.9d0*t*fpn) go to 30
1052      t=2.d0*t
1053      go to 20
1054   30 indica=1
1055      logic=0
1056      if (t.gt.tmax) then
1057          t=tmax
1058          logic=1
1059      endif
1060      if (imp.ge.4) write (io,1000) fpn,d2,tmin,tmax
1061!
1062!     --- nouveau x
1063!
1064      do 50 i=1,n
1065          x(i)=xn(i)+t*d(i)
1066   50 continue
1067!
1068! --- boucle
1069!
1070  100 nap=nap+1
1071      if(nap.gt.napmax) then
1072          logic=4
1073          fn=fg
1074          do 120 i=1,n
1075              xn(i)=xn(i)+tg*d(i)
1076  120     continue
1077          go to 999
1078      endif
1079      indic=4
1080!
1081!     --- appel simulateur
1082!
1083      call simul(indic,n,x,f,g,izs,rzs,dzs)
1084      if(indic.eq.0) then
1085!
1086!         --- arret demande par l'utilisateur
1087!
1088          logic=5
1089          fn=f
1090          do 170 i=1,n
1091              xn(i)=x(i)
1092  170     continue
1093          go to 999
1094      endif
1095      if(indic.lt.0) then
1096!
1097!         --- les calculs n'ont pas pu etre effectues par le simulateur
1098!
1099          td=t
1100          indicd=indic
1101          logic=0
1102          if (imp.ge.4) write (io,1004) t,indic
1103          t=tg+0.1d0*(td-tg)
1104          go to 905
1105      endif
1106!
1107!     --- les tests elementaires sont faits, on y va
1108!
1109      call prosca (n,d,g,ps,izs,rzs,dzs)
1110      fp=ps
1111!
1112!     --- premier test de Wolfe
1113!
1114      ffn=f-fn
1115      if(ffn.gt.t*tesf) then
1116          td=t
1117          fd=f
1118          fpd=fp
1119          indicd=indic
1120          logic=0
1121          if(imp.ge.4) write (io,1002) t,ffn,fp
1122          go to 500
1123      endif
1124!
1125!     --- test 1 ok, donc deuxieme test de Wolfe
1126!
1127      if(imp.ge.4) write (io,1003) t,ffn,fp
1128      if(fp.gt.tesd) then
1129          logic=0
1130          go to 320
1131      endif
1132      if (logic.eq.0) go to 350
1133!
1134!     --- test 2 ok, donc pas serieux, on sort
1135!
1136  320 fn=f
1137      do 330 i=1,n
1138          xn(i)=x(i)
1139  330 continue
1140      go to 999
1141!
1142!
1143!
1144  350 tg=t
1145      fg=f
1146      fpg=fp
1147      if(td.ne.0.d0) go to 500
1148!
1149!              extrapolation
1150!
1151      taa=t
1152      gauche=(1.d0+barmin)*t
1153      droite=10.d0*t
1154      call dcube (t,f,fp,ta,fa,fpa,gauche,droite)
1155      ta=taa
1156      if(t.lt.tmax) go to 900
1157      logic=1
1158      t=tmax
1159      go to 900
1160!
1161!              interpolation
1162!
1163  500 if(indica.le.0) then
1164          ta=t
1165          t=0.9d0*tg+0.1d0*td
1166          go to 900
1167      endif
1168      test=barr*(td-tg)
1169      gauche=tg+test
1170      droite=td-test
1171      taa=t
