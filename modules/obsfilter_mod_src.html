 101  subroutine filt_setup(filterMode_in)
 102    implicit none
 103
 104    character(len=*), intent(in) :: filterMode_in
 105
 106    integer :: nulnam, ierr, elem, jflag, ibit, obsFamilyIndex, elemIndex
 107    integer :: fnom, fclos
 108    integer :: flagIndex, elementIndex
 109
 110    character(len=35) :: CREASON(-8:13)
 111    data creason/'JACOBIAN IMPORTANT ABOVE MODEL TOP', &
 112         'ABS OROGRAPH-PHI                  ', &
 113         'MASQUE TERRE-MER                  ', &
 114         'OROGRAPHIE                        ', &
 115         'REJECTED BY QCVAR                 ', &
 116         'REJECTED BY BACKGROUND CHECK      ', &
 117         'BACKGROUND CHECK  LEVEL 3         ', &
 118         'BACKGROUND CHECK  LEVEL 2         ', &
 119         'BACKGROUND GHECK  LEVEL 1         ', &
 120         'RESERVED                          ', &
 121         'REJECTED BY SELECTION PROCESS     ', &
 122         'GENERATED BY OI                   ', &
 123         'REJECTION BY  OI                  ', &
 124         'ELEMENT ON BLACK LIST             ', &
 125         'RESERVED                          ', &
 126         'CORRECTED ELEMENT                 ', &
 127         'INTERPOLATED ELEMENT              ', &
 128         'DOUBTFUL ELEMENT                  ', &
 129         'POSSIBLY ERRONEOUS ELEMENT        ', &
 130         'ERRONEOUS ELEMENT                 ', &
 131         'ELEMENT EXCEEDS CLIMATE EXTREME   ', &
 132         'ELEMENT MODIFIED OR GEN BY  ADE   '/
 133
 134    ! Namelist variables: (local)
 135    integer :: nelems                    ! MUST NOT BE INCLUDED IN NAMELIST!
 136    integer :: nlist(nelemsMax)          ! list of bufr element IDs to consider for assimilation
 137    integer :: nflags                    ! MUST NOT BE INCLUDED IN NAMELIST!
 138    integer :: nlistflg(nFLagsMax)       ! list of flag 'reference numbers' to use for rejecting obs
 139    integer :: nelems_altDiffMax         ! MUST NOT BE INCLUDED IN NAMELIST!
 140    integer :: list_altDiffMax(numElem)  ! list of bufr element IDs to apply maximum altitude difference
 141    character(len=2) :: list_topoFilt(nTopoFiltFam) ! list of obs family names for applying max altitude
 142    real(8) :: value_altDiffMax(numElem) ! value of maximum difference between model sfc and obs altitude
 143    real(8) :: rlimlvhu                  ! pressure level (in hPa) above which humidity (ES) obs are rejected
 144
 145    namelist /namfilt/nelems, nlist, nflags, nlistflg, rlimlvhu, discardlandsfcwind, &
 146         nelems_altDiffMax, list_altDiffMax, value_altDiffMax, surfaceBufferZone_Pres, &
 147         surfaceBufferZone_Height, list_topoFilt, useEnkfTopoFilt
 148
 149    filterMode = filterMode_in
 150
 151    ! set default values for namelist variables
 152    nlist(:) =  MPC_missingValue_INT
 153    nelems = MPC_missingValue_INT
 154    nlistflg(:) = MPC_missingValue_INT
 155    nflags = MPC_missingValue_INT
 156    list_altDiffMax (:) = MPC_missingValue_INT
 157    value_altDiffMax(:) = MPC_missingValue_R8
 158    nelems_altDiffMax = MPC_missingValue_INT
 159
 160    list_topoFilt(:) = '**'
 161
 162    rlimlvhu = 300.d0
 163    discardlandsfcwind = .true.
 164
 165    surfaceBufferZone_Pres   = 5000.0d0 ! default value in Pascals
 166    surfaceBufferZone_Height =  400.0d0 ! default value in Metres
 167
 168    useEnkfTopoFilt = .false.
 169
 170    nulnam=0
 171    ierr=fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
 172    read(nulnam,nml=namfilt,iostat=ierr)
 173    if(ierr.ne.0) call utl_abort('filt_setup: Error reading namelist! Hint: did you replace ltopofilt by list_topoFilt?')
 174    if(mmpi_myid == 0) write(*,nml=namfilt)
 175    ierr=fclos(nulnam)
 176
 177    filt_rlimlvhu    = rlimlvhu
 178    
 179    if (nelems /= MPC_missingValue_INT) then
 180      call utl_abort('filt_setup: check NAMFILT namelist section; NELEMS should be removed')
 181    end if
 182    filt_nelems = 0
 183    do elementIndex = 1, nelemsMax
 184      if (nlist(elementIndex) /= MPC_missingValue_INT) then
 185        filt_nlist(elementIndex) = nlist(elementIndex)
 186        filt_nelems = filt_nelems + 1
 187      end if
 188    end do
 189    
 190    if (nflags /= MPC_missingValue_INT) then
 191      call utl_abort('filt_setup: check NAMFILT namelist section; NFLAGS should be removed')
 192    end if
 193    filt_nflags = 0
 194    do flagIndex = 1, nflagsMax
 195      if (nlistflg(flagIndex) /= MPC_missingValue_INT) then
 196        filt_nlistflg(flagIndex) = nlistflg(flagIndex)
 197        filt_nflags = filt_nflags + 1
 198      end if
 199    end do
 200
 201    if(mmpi_myid == 0) then
 202      write(*,'(1X,"***********************************")')
 203      write(*,'(1X," ELEMENTS SELECTED FOR ASSIMILATION:",/)')
 204      write(*,'(1X,"***********************************")')
 205      do elem=1,filt_nelems
 206        write(*,'(15X,I5)') filt_nlist(elem)
 207      end do
 208      write(*,'(1X,"***********************************")')
 209      write(*,*) ' REJECT ELEMENTS WITH REJECT FLAG '
 210      write(*,*)'           BIT :  '
 211      do jflag=1,filt_nflags
 212        ibit= filt_nlistflg(jflag)
 213        write(*,*) ibit,' ',creason(ibit)
 214      end do
 215      write(*,'(1X,"***********************************")')
 216    end if
 217
 218    !
 219    !- Set values for altDiffMax
 220    !
 221    if (nelems_altDiffMax /= MPC_missingValue_INT) then
 222      call utl_abort('filt_setup: check namelist section NAMFILT: nelems_altDiffMax should be removed')
 223    end if
 224    do elem = 1, numElem
 225      if (list_altDiffMax(elem) /= MPC_missingValue_INT .and. value_altDiffMax(elem) /= MPC_missingValue_R8) then
 226        elemIndex = findElemIndex(list_altDiffMax(elem))
 227        if ( elemIndex >= 1 .and. elemIndex <= numElem ) then
 228          altDiffMax(elemIndex) = value_altDiffMax(elem)
 229          write(*,*) ' filt_setup: altDiffMax value for ', elemList(elemIndex), ' is set to ', altDiffMax(elemIndex)
 230        else
 231          call utl_abort('filt_setup: Error in value setting for altDiffMax')
 232        end if
 233      end if
 234    end do
 235
 236    !
 237    !- Set the topographic rejection list 
 238    !
 239    if (all(list_topoFilt(:) == '**')) then
 240      ! default list
 241      filtTopoList(1) = 'SF'
 242      filtTopoList(2) = 'UA'
 243      filtTopoList(3) = 'AI'
 244      filtTopoList(4) = 'SW'
 245      filtTopoList(5) = 'PR'
 246      filtTopoList(6) = 'AL'
 247      filtTopoList(7) = 'TO'
 248      filtTopoList(8) = 'CH'
 249    else
 250      do obsFamilyIndex = 1, nTopoFiltFam
 251        if (list_topoFilt(obsFamilyIndex) /= '**') then
 252          filtTopoList(obsFamilyIndex) = list_topoFilt(obsFamilyIndex)
 253        end if
 254      end do
 255    end if
 256
 257    initialized = .true.
 258
 259  end subroutine filt_setup

 263  !--------------------------------------------------------------------------
 264  subroutine filt_suprep(obsSpaceData)
 265    !
 266    !:Purpose: Select the data in the obsSpaceData which are to be assimilated
 267    !
 268    implicit none
 269    type(struct_obs) :: obsSpaceData
 270    integer :: bodyIndex, headerIndex
 271    integer :: ipres, ivco, ierr, loopIndex
 272    integer :: idburp, ivnm, iflg, ibad, iknt, iknt_mpiglobal, ilansea
 273    logical :: llok, llrej, llbogus
 274
 275    call utl_tmg_start(22,'----ObsFiltSuprep')
 276
 277    if(mmpi_myid == 0) write(*,*) 'starting subroutine filt_suprep'
 278
 279    iknt = 0
 280
 281    BODY: do bodyIndex = 1, obs_numbody( obsSpaceData )
 282      headerIndex = obs_bodyElem_i( obsSpaceData, OBS_HIND, bodyIndex   )
 283      ivnm        = obs_bodyElem_i( obsSpaceData, OBS_VNM , bodyIndex   )
 284      idburp      = obs_headElem_i( obsSpaceData, OBS_ITY , headerIndex )
 285      !
 286      ! Unwanted data types via types specified in NLIST
 287      !
 288      llok = .false.
 289      do loopIndex = 1, filt_nelems
 290        llok = ( ivnm == filt_nlist( loopIndex ) ) .or. llok
 291      end do
 292      if (.not.llok) then
 293        call obs_bodySet_i(obsSpaceData,OBS_ASS,bodyIndex,obs_notAssimilated)
 294        cycle BODY
 295      end if
 296      !
 297      ! Allow gz for bogus data only in analysis case 
 298      !
 299      llbogus = ( idburp == 150 .or. idburp == 151 .or. idburp == 152 .or. idburp == 153 )
 300      if  ( (filterMode == 'analysis' .or. filterMode == 'FSO') .and. llok .and. ivnm == BUFR_NEGZ .and. .not.llbogus ) then
 301        llok=.false.
 302      end if
 303      !
 304      ! Ground-based GPS (GP) data (codtyp 189)
 305      ! LLOK = .TRUE. DY DEFAULT IF ELEMENT IS IN NLIST
 306      ! If gps_gb_LASSMET = .FALSE. don't want to assimilate Ps (BUFR_NEPS),
 307      ! Ts (BUFR_NETS), or (T-Td)s (BUFR_NESS)
 308      !
 309      if ( idburp == 189 ) then
 310        if (.not.gps_gb_lassmet .and. ( ivnm == BUFR_NEPS .or.  &
 311                                        ivnm == BUFR_NETS .or.  &
 312                                        ivnm == BUFR_NESS )) then
 313          llok = .false.
 314        end if
 315      end if
 316      !
 317      ! Exclude T-Td above level RLIMLVHU (mbs)
 318      !
 319      ivco  = obs_bodyElem_i( obsSpaceData, OBS_VCO, bodyIndex )
 320      ipres = nint( obs_bodyElem_r( obsSpaceData, OBS_PPP, bodyIndex))
 321      if ( ( ivco == 2) .and. ( ivnm == BUFR_NEES ) .and.  &
 322           ( ipres < nint( filt_rlimlvhu *100.0d0 )) ) then
 323        llok=.false.
 324      end if
 325      !
 326      ! Bad data with quality control flags via bit list specified in NLISTFLG
 327      !
 328      iflg = obs_bodyElem_i( obsSpaceData, OBS_FLG, bodyIndex )
 329      llrej = .false.
 330      do loopIndex = 1, filt_nflags
 331        ibad = 13 - filt_nlistflg( loopIndex )
 332        llrej=( btest(iflg,ibad) ) .or. llrej
 333      end do
 334      !
 335      ! Filter TOVS data: check for invalid land/sea/sea-ice flag
 336      !
 337      if (ivnm == BUFR_NBT1 .or. ivnm == BUFR_NBT2 .or. ivnm == BUFR_NBT3) then
 338        ! Here we have to exclude the ssmis and mwhs2 data since the burp file does not contain the ele 8021 land_sea
 339        if ( ( tvs_isIdBurpTovs(idburp) ) .and. &
 340             ( idburp /= codtyp_get_codtyp('ssmis') ) .and. &
 341             ( idburp /= codtyp_get_codtyp('mwhs2') ) ) then
 342          ilansea  = tvs_ChangedStypValue( obsSpaceData,  headerIndex )
 343          if (ilansea < 0 .or. ilansea > 2  ) llok = .false.
 344        end if
 345      end if
 346      !
 347      ! SAR winds: assimilates wind speed only for SAR winds
 348      !
 349      if ( ivnm == BUFR_NEFS ) then 
 350        if ( idburp .ne. 204 ) then 
 351          llok = .false.
 352        end if
 353      end if
 354      if ( llok .and. .not. llrej ) then
 355        call obs_bodySet_i( obsSpaceData, OBS_ASS, bodyIndex, obs_assimilated )
 356        iknt = iknt + 1
 357      else
 358        call obs_bodySet_i( obsSpaceData, OBS_ASS, bodyIndex, obs_notAssimilated )
 359      end if
 360
 361    end do body
 362
 363    call rpn_comm_allreduce( iknt, iknt_mpiglobal, 1, "MPI_INTEGER", "MPI_SUM", "GRID", ierr )
 364    if(mmpi_myid == 0) write(*,*) '  Number of data to be assimilated: ', iknt_mpiglobal
 365
 366    if(mmpi_myid == 0) write(*,*) 'end of filt_suprep'
 367
 368    ! abort if there is no data to be assimilated
 369    if (iknt_mpiglobal == 0 ) then
 370       call utl_abort('SUPREP. NO DATA TO BE ASSIMILATED')
 371    end if
 372
 373    call utl_tmg_stop(22)
 374
 375  end subroutine filt_suprep

 418  !--------------------------------------------------------------------------
 419  ! filt_topoSurface
 420  !--------------------------------------------------------------------------
 421  subroutine filt_topoSurface(columnTrlOnTrlLev, obsSpaceData, beSilent)
 422    !
 423    !:Purpose: Refuse elements which are too far away from the surface.
 424    !           Replace the pressure of elements which are slightly below
 425    !           the model surface by the pressure of the trial field.
 426    !
 427    implicit none
 428
 429    type(struct_columnData) :: columnTrlOnTrlLev
 430    type(struct_obs)        :: obsSpaceData
 431    logical                 :: beSilent
 432
 433    real(8) :: altitudeDiff
 434    integer :: headerIndex, bodyIndex, familyIndex, elemIndex
 435    integer :: ivnm,countAssim
 436    integer :: countAcc(numElem),countRej(numElem)
 437    integer, parameter :: numFamily = 3
 438    character(len=2) :: list_family(numFamily)
 439    character(len=4) :: varLevel
 440
 441    !
 442    ! reset gps_gb_dzmax for gb gps ztd to value in namelist file
 443    !
 444    altDiffMax(findElemIndex(BUFR_NEZD)) = gps_gb_dzmax
 445
 446    if (  .not.beSilent ) then
 447      write(*,*) ' '
 448      write(*,*) ' filt_topoSurface: '
 449      write(*,*) ' '
 450      write(*,*) '*****************************************************'
 451      write(*,222) 'ELEMENTS                  ',(elemList(elemIndex),elemIndex=1,numElem)
 452      write(*,223) 'REJECTION BOUNDARY(METRE) ',(altDiffMax(elemIndex),elemIndex=1,numElem)
 453      write(*,*) '*****************************************************'
 454      write(*,*) ' '
 455    end if
 456
 457    ! Loop over the families of interest
 458    list_family(1) = 'SF'
 459    list_family(2) = 'UA'
 460    list_family(3) = 'GP'
 461    FAMILY: do familyIndex = 1, numFamily
 462
 463       ! set counters to zero
 464       countRej(:)=0
 465       countAcc(:)=0
 466
 467       ! loop over all header indices of each family
 468       call obs_set_current_header_list(obsSpaceData, &
 469            list_family(familyIndex))
 470       HEADER: do
 471          headerIndex = obs_getHeaderIndex(obsSpaceData)
 472          if (headerIndex < 0) exit HEADER
 473
 474          ! loop over all body indices (still in the same family)
 475          call obs_set_current_body_list(obsSpaceData, headerIndex)
 476          BODY: do 
 477             bodyIndex = obs_getBodyIndex(obsSpaceData)
 478             if (bodyIndex < 0) exit BODY
 479
 480             ! skip this obs if it is not on height levels
 481             if (obs_bodyElem_i(obsSpaceData,OBS_VCO,bodyIndex).ne.1) cycle BODY
 482
 483             ! skip this obs if already flagged to not be assimilated
 484             if (obs_bodyElem_i(obsSpaceData,OBS_ASS,bodyIndex) == obs_notAssimilated) cycle BODY
 485
 486             ivnm   = obs_bodyElem_i(obsSpaceData,OBS_VNM,bodyIndex)
 487             varLevel = vnl_varLevelFromVarnum(ivnm)
 488             altitudeDiff = abs( obs_bodyElem_r(obsSpaceData,OBS_PPP,bodyIndex) -  &
 489                  col_getHeight(columnTrlOnTrlLev,col_getNumLev(columnTrlOnTrlLev,varLevel),headerIndex,varLevel) )
 490             !
 491             ! apply filter to selected elements
 492             !
 493             elemIndex = findElemIndex(ivnm)
 494             if (elemIndex == -1) cycle BODY
 495
 496             if (altitudeDiff <= altDiffMax(elemIndex)) then
 497                ! obs passes the acceptance criteria
 498                countAcc(elemIndex) = countAcc(elemIndex)+1
 499             else
 500                ! obs does not pass the acceptance criteria, reject it
 501                call obs_bodySet_i( obsSpaceData,OBS_FLG,bodyIndex,  &
 502                     ibset(obs_bodyElem_i(obsSpaceData,OBS_FLG,bodyIndex),18) )
 503                call obs_bodySet_i(obsSpaceData,OBS_ASS,bodyIndex,obs_notAssimilated)
 504                countRej(elemIndex) = countRej(elemIndex)+1
 505             end if
 506          end do BODY
 507       end do HEADER
 508
 509       if ( .not.beSilent ) then
 510         write(*,*) ' '
 511         write(*,*) '*****************************************************'
 512         write(*,*) 'FAMILY = ',list_family(familyIndex)
 513         write(*,222) 'ELEMENTS            ', (elemList(elemIndex),elemIndex=1,numElem)
 514         write(*,222) 'ACCEPTED  ',(countAcc(elemIndex),elemIndex=1,numElem)
 515         write(*,222) 'REJECTED  ',(countRej(elemIndex),elemIndex=1,numElem)
 516         write(*,*) '*****************************************************'
 517         write(*,*) ' '
 518       end if
 519222    format(2x,a29,16(2x,i5))
 520223    format(2x,a29,16(2x,f5.0))
 521
 522    end do FAMILY
 523
 524    countAssim=0
 525    do bodyIndex=1,obs_numbody(obsSpaceData)
 526       if ( obs_bodyElem_i(obsSpaceData,OBS_ASS,bodyIndex) == obs_assimilated) countAssim=countAssim+1
 527    end do
 528    if ( .not.beSilent ) write(*,'(1X," NUMBER OF DATA TO BE ASSIMILATED AFTER ADJUSTMENTS: ",i10)') countAssim

 533  !--------------------------------------------------------------------------
 534  ! filt_topoRadiosonde
 535  !--------------------------------------------------------------------------
 536  subroutine filt_topoRadiosonde(columnTrlOnTrlLev, obsSpaceData, beSilent)
 537    !
 538    !:Purpose: Refuse elements which are too far away from the surface of the model
 539    !           Refuse elements which are considered in the free atmosphere of
 540    !           the RAOB but fall in the surface boundary layer of the model atmosphere.
 541    !
 542    implicit none
 543   
 544    ! arguments:
 545    type(struct_columnData) :: columnTrlOnTrlLev
 546    type(struct_obs) :: obsSpaceData
 547    logical :: beSilent
 548
 549    ! locals:
 550    integer :: headerIndex, bodyIndex, listIndex, elemIndex
 551    integer :: ivnm, countAssim
 552    integer :: itotacc(numElem), itotrej(numElem), isblrej(numElem)
 553    integer :: igzacc(numElem), igzrej(numElem), ibndrej(numElem)
 554    real(8) :: zval, obsPressure, altitudeDiff
 555    real(8) :: obsSfcAltitude, colSfcAltitude, colPressureBelow, colPressureAbove, zdelp
 556    logical :: llok
 557    real(8) :: geopotential(1), height(1)
 558    integer :: nlev_M
 559    real(8) :: lat
 560
 561    if ( .not.beSilent ) then
 562      write(*,*) ' '
 563      write(*,*) ' filt_topoRadiosonde: '
 564      write(*,*) ' '
 565      write(*,*) '************************************************'
 566      write(*,222) ' ELEMENTS                  ',(elemList(elemIndex),elemIndex=1,numElem)
 567      write(*,223) ' REJECTION BOUNDARY(METRE) ',(altDiffMax(elemIndex),elemIndex=1,numElem)
 568      write(*,223) ' REJECTION SBL (PASCAL)    ',(surfaceBufferZone_Pres,elemIndex=1,numElem)
 569      write(*,*) '************************************************'
 570      write(*,*) ' '
 571    end if
 572
 573    ! set counters to zero
 574    itotrej(:)=0
 575    itotacc(:)=0
 576    isblrej(:)=0
 577    igzacc(:)=0
 578    igzrej(:)=0
 579    ibndrej(:)=0
 580
 581    nlev_M = col_getNumLev(columnTrlOnTrlLev,'MM')
 582
 583    ! loop over all header indices of the 'UA' family
 584    call obs_set_current_header_list(obsSpaceData, 'UA')
 585    HEADER: do
 586      headerIndex = obs_getHeaderIndex(obsSpaceData)
 587      if (headerIndex < 0) exit HEADER
 588
 589      ! HEIGHT GZ
 590
 591      ! loop over all body indices (still in the 'UA' family)
 592      call obs_set_current_body_list(obsSpaceData, headerIndex)
 593      BODY: do 
 594        bodyIndex = obs_getBodyIndex(obsSpaceData)
 595        if (bodyIndex < 0) exit BODY
 596
 597        ! skip this obs if it is not on pressure level
 598        if( obs_bodyElem_i(obsSpaceData,OBS_VCO,bodyIndex).ne.2 ) cycle BODY
 599
 600        ! skip this obs if already flagged to not be assimilated
 601        if( obs_bodyElem_i(obsSpaceData,OBS_ASS,bodyIndex) == obs_notAssimilated ) cycle BODY
 602
 603        ! skip this obs if it is not GZ
 604        ivnm=obs_bodyElem_i(obsSpaceData,OBS_VNM,bodyIndex)
 605        listIndex = findElemIndex(ivnm)
 606        llok = (ivnm == BUFR_NEGZ .and. listIndex.ne.-1)
 607        if (.not. llok ) cycle BODY
 608
 609        ! convert altitude read from column to geopotential
 610        height(1) = col_getHeight(columnTrlOnTrlLev,0,headerIndex,'SF')
 611        lat = obs_headElem_r(obsSpaceData,OBS_LAT,headerIndex)
 612        call phf_height2geopotential(height,lat,geopotential)
 613
 614        zval = obs_bodyElem_r(obsSpaceData,OBS_VAR,bodyIndex)
 615        altitudeDiff = ( zval - geopotential(1) )/ec_rg
 616        ! obs is above surface, so it is ok, lets jump to the next obs
 617        if(altitudeDiff >= 0.0d0) cycle BODY
 618
 619        if(altitudeDiff >= -1.0d0*altDiffMax(listIndex)) then
 620          ! obs is an acceptably small distance below the surface
 621          itotacc(listIndex) = itotacc(listIndex)+1
 622          igzacc(listIndex) = igzacc(listIndex)+1
 623        else
 624          ! too far below surface, reject
 625          call obs_bodySet_i(obsSpaceData,OBS_FLG,bodyIndex,  &
 626               ibset( obs_bodyElem_i(obsSpaceData,OBS_FLG,bodyIndex), 18 ))
 627          call obs_bodySet_i(obsSpaceData,OBS_ASS,bodyIndex,obs_notAssimilated)
 628          itotrej(listIndex) = itotrej(listIndex)+1
 629          igzrej(listIndex) = igzrej(listIndex)+1
 630        end if
 631      end do BODY
 632      !
 633      !   REJECT ELEMENTS OF U,V,T-TD,T BELOW THE MODEL SURFACE
 634      !   AND THOSE NON SURFACE ELEMENTS PRESENT IN THE SURFACE
 635      !   BOUNDARY LAYER OF THE RAOB OR OF THE MODEL.
 636      !   AT THIS POINT WE WANT TO KEEP OBSERVATIONS IN THE FREE
 637      !   ATMOSPHERE
 638      !
 639      !---Special case if station elevation is above model elevation
 640      !   we want to define colPressureAbove at a level above the station.
 641      !   To approximate that value, we will transform the difference
 642      !   between the 2 elevations into a difference in pressure using
 643      !   the rule of thumb (1Mb =8 metres)
 644      !---Even though TT(element=12001) is not assimmilated
 645      !   it is treated as if it were for the evaluation step.
 646      !   Otherwise we use observations of TT that are too far
 647      !   from the model topography in the verification.
 648
 649      obsSfcAltitude = obs_headElem_r(obsSpaceData,OBS_ALT,headerIndex)
 650      colSfcAltitude = col_getHeight(columnTrlOnTrlLev,0,headerIndex,'SF')
 651      altitudeDiff = obsSfcAltitude - colSfcAltitude
 652
 653      ! Set the body list & start at the beginning of the list
 654      call obs_set_current_body_list(obsSpaceData, headerIndex)
 655      BODY2: do 
 656        bodyIndex = obs_getBodyIndex(obsSpaceData)
 657        if (bodyIndex < 0) exit BODY2
 658
 659        if( obs_bodyElem_i(obsSpaceData,OBS_VCO,bodyIndex).ne.2 ) cycle BODY2
 660
 661        ! skip this obs if already flagged to not be assimilated
 662        if( obs_bodyElem_i(obsSpaceData,OBS_ASS,bodyIndex) == obs_notAssimilated ) cycle BODY2
 663
 664        ivnm = obs_bodyElem_i(obsSpaceData,OBS_VNM,bodyIndex)
 665        listIndex = findElemIndex(ivnm)
 666        llok = (ivnm.ne.BUFR_NEGZ .and. listIndex.ne.-1)
 667        if (.not. llok ) cycle BODY2 ! Proceed with the next bodyIndex
 668
 669        obsPressure = obs_bodyElem_r(obsSpaceData,OBS_PPP,bodyIndex)
 670        colPressureBelow = col_getElem(columnTrlOnTrlLev,1,headerIndex,'P0')
 671        colPressureAbove = colPressureBelow - surfaceBufferZone_Pres
 672
 673        if (useEnkfTopoFilt) then
 674          ! Simpler rules used in the EnKF
 675          if(obsPressure >= colPressureAbove ) then
 676            if(abs(altitudeDiff) >= 50.0D0 .or. obsPressure >= colPressureBelow) then
 677              call obs_bodySet_i(obsSpaceData,OBS_FLG,bodyIndex,  &
 678                   ibset(obs_bodyElem_i(obsSpaceData,OBS_FLG,bodyIndex), 18 ))
 679              call obs_bodySet_i(obsSpaceData,OBS_ASS,bodyIndex,obs_notAssimilated)
 680              itotrej(listIndex) = itotrej(listIndex) + 1
 681              ibndrej(listIndex) = ibndrej(listIndex) + 1
 682            end if
 683          end if
 684        else
 685          ! Original (and confusing) rules used in Var
 686          if (altitudeDiff > 0.0d0) then
 687            zdelp = altitudeDiff * 100.d0 / 8.0d0
 688            colPressureAbove = colPressureBelow - (zdelp + surfaceBufferZone_Pres)
 689          end if
 690
 691          if(abs(altitudeDiff) <= altDiffMax(listIndex)) then
 692            !--Model surface and station altitude are very close
 693            !  Accept observation if obsPressure is within the domain
 694            !  of the trial field
 695            colPressureAbove = col_getPressure(columnTrlOnTrlLev,col_getNumLev(columnTrlOnTrlLev,'MM')-1,headerIndex,'MM')
 696          end if
 697          if(obsPressure > colPressureBelow ) then
 698            call obs_bodySet_i(obsSpaceData,OBS_FLG,bodyIndex,  &
 699                 ibset(obs_bodyElem_i(obsSpaceData,OBS_FLG,bodyIndex), 18 ))
 700            call obs_bodySet_i(obsSpaceData,OBS_ASS,bodyIndex,obs_notAssimilated)
 701            itotrej(listIndex) = itotrej(listIndex) + 1
 702            ibndrej(listIndex) = ibndrej(listIndex) + 1
 703          else if(obsPressure <= colPressureBelow .and. obsPressure > colPressureAbove ) then
 704            call obs_bodySet_i(obsSpaceData,OBS_FLG,bodyIndex,  &
 705                 ibset( obs_bodyElem_i(obsSpaceData,OBS_FLG,bodyIndex), 18 ))
 706            call obs_bodySet_i(obsSpaceData,OBS_ASS,bodyIndex,obs_notAssimilated)
 707            itotrej(listIndex) = itotrej(listIndex) + 1
 708            isblrej(listIndex) = isblrej(listIndex) + 1
 709          end if
 710        end if
 711
 712      end do BODY2
 713    end do HEADER
 714
 715    if ( .not.beSilent ) then
 716      write(*,*) ' '
 717      write(*,*) '***************************************'
 718      write(*,*) 'FAMILY = UA'
 719      write(*,222) ' ELEMENTS          ', (elemList(elemIndex),elemIndex=1,numElem)
 720      write(*,222) ' ACC GZ EXT  ',(igzacc(elemIndex),elemIndex=1,numElem)
 721      write(*,222) ' ACC TOTAL   ',(itotacc(elemIndex),elemIndex=1,numElem)
 722      write(*,*) '***************'
 723      write(*,222) ' REJ GZ EXT  ',(igzrej(elemIndex),elemIndex=1,numElem)
 724      write(*,222) ' REJ OUT BND ',(ibndrej(elemIndex),elemIndex=1,numElem)
 725      write(*,222) ' REJ SBL     ',(isblrej(elemIndex),elemIndex=1,numElem)
 726      write(*,222) ' REJ TOTAL   ',(itotrej(elemIndex),elemIndex=1,numElem)
 727      write(*,*) '***************************************'
 728      write(*,*) ' '
 729    end if
 730222 format(2x,a29,16(2x,i5))
 731223 format(2x,a29,16(2x,f6.0))
 732
 733    countAssim=0
 734    do bodyIndex=1,obs_numbody(obsSpaceData)
 735      if ( obs_bodyElem_i(obsSpaceData,OBS_ASS,bodyIndex) == obs_assimilated) countAssim=countAssim+1
 736    end do
 737    if ( .not.beSilent ) write(*,'(1X," NUMBER OF DATA TO BE ASSIMILATED AFTER ADJUSTMENTS:",i10)') countAssim
 738    if ( .not.beSilent ) write(*,*) ' '
 739
 740  end subroutine filt_topoRadiosonde

1768    !
1769    !:Purpose: Rejects elements which are too far below the model surface
1770    !           or above the model top.
1771    !
1772    !:Comments:
1773    !    Flagging of bit 4 in OBS_FLG done in filt_topoChemistry instead of set_scale_chm
1774    !    since this subroutine is called after chm_setup, allowing use of utl_open_asciifile
1775    !
1776    implicit none
1777    type(struct_columnData) :: columnTrlOnTrlLev
1778    type(struct_obs) :: obsSpaceData
1779    logical :: beSilent
1780
1781    integer :: headerIndex, bodyIndex, listIndex, elemIndex, listIndex_stnid
1782    integer :: ivnm, countAssim, jl, icount
1783    real(8) :: obsAltitude, obsPressure, colTopPressure, colSfcPressure
1784    real(8) :: colAltitudeBelow, colAltitudeAbove
1785    logical :: list_is_empty

1795

1797    countAcc_stnid(:)=0
1798    countRej_stnid(:)=0
1799    countRejflg_stnid(:)=0
1800    Num_stnid_chm=0
1801

1812

1817
1818      ! Set geopotential height and pressure boundaries.

1823      colTopPressure = col_getPressure(columnTrlOnTrlLev,1,headerIndex,'MM')

1844

1850                              iConstituentList,Num_chm,Nmax,listIndex)

1854        if (btest(obs_bodyElem_i(obsSpaceData,OBS_FLG,bodyIndex),4)) &
1855           call obs_bodySet_i(obsSpaceData,OBS_ASS,bodyIndex,obs_notAssimilated)
1856
1857        if (obs_bodyElem_i(obsSpaceData,OBS_ASS,bodyIndex) == obs_notAssimilated) then

1859            ! Already rejected from input marker/flag.
1860
1861            countRej(listIndex)=countRej(listIndex)+1
