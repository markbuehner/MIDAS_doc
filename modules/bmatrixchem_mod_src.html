  17module BmatrixChem_mod 
  18  ! MODULE BmatrixChem_mod (prefix='bchm' category='2. B and R matrices')
  19  !
  20  !:Purpose: Contains routines involving the application of
  21  !           background-error covariance matrix(ces). Matrix based on
  22  !           horizontally homogeneous/isotropic correlations. This module
  23  !           includes the transform from control vector (spectral space) to 
  24  !           analysis increments, related utilites, and the transform's adjoint.
  25  !
  26  !           Based on elements of bmatrixHI_mod.ftn90
  27  !
  28  !:Comments:
  29  !
  30  !   1. Covariances uncoupled from weather variable.
  31  !
  32  !   2. One could potentially make public the functions/routines which are
  33  !      identical to those in bmatrixhi_mod.ftn90 (except possibly in name) so
  34  !      that one copy is present in the code.
  35  !
  36  !
  37  ! Public Subroutines:
  38  !
  39  !    bchm_setupCH:  Must be called first. 
  40  !                   Acquire constituents backgound error standard 
  41  !                   deviations and spectral space correlations which are 
  42  !                   read and prepared by bcsc_setupCH. 
  43  !    bchm_finalize: Deallocate internal module arrays.
  44  !    bchm_BSqrt:    Transformations from control vector to analysis
  45  !                   increments in the minimization process.
  46  !    bchm_BSqrtAd:  Adjoint of bchm_BSqrt.
  47  !    bchm_expand*   MPI manipulations of contol vector(s)
  48  !    bchm_reduce*   MPI manipulations related to contol vector(s)
  49  !
  50  
  51  use midasMpi_mod
  52  use gridStateVector_mod
  53  use gridVariableTransforms_mod
  54  use globalSpectralTransform_mod
  55  use horizontalCoord_mod
  56  use verticalCoord_mod
  57  use varNameList_mod
  58  use utilities_mod
  59  use bCovarSetupChem_mod
  60  
  61  implicit none
  62  save
  63  private
  64
  65  ! public procedures
  66  public :: bchm_setupCH,bchm_finalize,bchm_BSqrt,bchm_BSqrtAd
  67  public :: bchm_expandToMPIglobal,bchm_expandToMPIglobal_r4,bchm_reduceToMPIlocal,bchm_reduceToMPIlocal_r4

 100  !--------------------------------------------------------------------------
 101  ! bchm_setupCH
 102  !--------------------------------------------------------------------------
 103  subroutine bchm_setupCH(hco_in,vco_in,cvDim_out)
 104   
 105    !:Purpose: Acquire constituents backgound error standard deviations (stddev),
 106    !          spectral space correlations (corns) and related elements
 107    !          which are read and prepared by lower level module routine
 108    !          bcsc_setupCH.
 109    !
 110
 111    implicit none
 112
 113    !Arguments
 114    type(struct_hco), intent(in), pointer :: hco_in
 115    type(struct_vco), intent(in) ,pointer :: vco_in
 116    integer, intent(out) :: cvDim_out
 117
 118    !Locals
 119    integer :: jn,jm
 120    integer :: latPerPE, latPerPEmax, lonPerPE, lonPerPEmax, nlev_T_even, ierr
 121    logical :: covarNeeded
 122
 123    ! Read and prepare covariances and related elements
 124    
 125    call bcsc_setupCH(hco_in,vco_in,covarNeeded,'Analysis')
 126    if (.not.covarNeeded) then
 127      ! Assumes CH covariances not needed.
 128      cvDim_out=0
 129      return
 130    end if
 131    
 132    ! Get covariances and related elements required for assimilation
 133        
 134    call bcsc_getCovarCH(bgStats,transformVarKind_opt=transformVarKindCH)
 135        
 136    ! Set vertical dimension
 137    ! Need an even number of levels for spectral transform
 138    
 139    if (mod(bgStats%nlev,2) /= 0) then
 140      nLev_T_even = bgStats%nlev+1
 141    else
 142      nLev_T_even = bgStats%nlev
 143    end if
 144     
 145    ! Spectral transform and MPI setup parameters.
 146
 147    nla_mpiglobal = (bgStats%ntrunc+1)*(bgStats%ntrunc+2)/2    
 148    gstID  = gst_setup(bgStats%ni,bgStats%nj,bgStats%ntrunc,bgStats%nkgdim)
 149    gstID2 = gst_setup(bgStats%ni,bgStats%nj,bgStats%ntrunc,nlev_T_even)
 150    if (mmpi_myid == 0) write(*,*) 'bchm_setupCH: returned value of gstID =',gstID
 151    if (mmpi_myid == 0) write(*,*) 'bchm_setupCH: returned value of gstID2=',gstID2
 152
 153    call mmpi_setup_latbands(bgStats%nj, latPerPE, latPerPEmax, myLatBeg, &
 154      myLatEnd)
 155    call mmpi_setup_lonbands(bgStats%ni, lonPerPE, lonPerPEmax, myLonBeg, &
 156      myLonEnd)
 157
 158    call mmpi_setup_m(bgStats%ntrunc,mymBeg,mymEnd,mymSkip,mymCount)
 159    call mmpi_setup_n(bgStats%ntrunc,mynBeg,mynEnd,mynSkip,mynCount)
 160
 161    call gst_ilaList_mpiglobal(ilaList_mpiglobal,nla_mpilocal,maxMyNla,gstID, &
 162      mymBeg,mymEnd,mymSkip,mynBeg,mynEnd,mynSkip)
 163    call gst_ilaList_mpilocal(ilaList_mpilocal,gstID,mymBeg,mymEnd,mymSkip, &
 164      mynBeg,mynEnd,mynSkip)
 165
 166    ! Compute mpilocal control vector size
 167    do jm = mymBeg, mymEnd, mymSkip
 168      do jn = mynBeg, mynEnd, mynSkip
 169        if (jm <= jn) then
 170          if (jm == 0) then
 171            ! only real component for jm=0
 172            cvDim_mpilocal = cvDim_mpilocal + 1*bgStats%nkgdim
 173          else
 174            ! both real and imaginary components for jm>0
 175            cvDim_mpilocal = cvDim_mpilocal + 2*bgStats%nkgdim
 176          end if
 177        end if
 178      end do
 179    end do
 180    cvDim_out = cvDim_mpilocal
 181
 182    ! Also compute mpiglobal control vector dimension
 183    call rpn_comm_allreduce(cvDim_mpilocal, cvDim_mpiglobal, 1, &
 184                            "mpi_integer", "mpi_sum", "GRID", ierr)
 185
 186    initialized = .true.
 187   
 188  end subroutine bchm_setupCH

 193  subroutine bchm_bSqrt(controlvector_in,statevector, stateVectorRef_opt)
 194    !
 195    !:Purpose: To apply B_CHM^1/2 to a control vector.
 196    !
 197    ! Based on bhi_bsqrt
 198    
 199    implicit none
 200
 201    !Arguments
 202    real(8), intent(inout) :: controlVector_in(cvDim_mpilocal)
 203    type(struct_gsv), intent(inout) :: statevector
 204    type(struct_gsv), intent(in) , optional :: stateVectorRef_opt
 205    
 206    !Locals
 207    real(8) ,allocatable :: gd_out(:,:,:)
 208    real(8)   :: hiControlVector(nla_mpilocal,2,bgStats%nkgdim)
 209    character(len=30) :: transform
 210    integer :: varIndex
 211    
 212    if (.not.initialized) return
 213
 214    if (mmpi_myid == 0) write(*,*) 'bchm_bsqrt: starting'
 215    if (mmpi_myid == 0) write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 216
 217    allocate(gd_out(myLonBeg:myLonEnd,myLatBeg:myLatEnd,bgStats%nkgdim))
 218
 219    call bchm_cain(controlVector_in,hiControlVector)
 220
 221    call bchm_spa2gd(hiControlVector,gd_out)
 222    
 223    call bchm_copyToStatevector(statevector,gd_out)
 224
 225    if ( trim(transformVarKindCH) /= '' ) then  
 226     
 227      transform = trim(transformVarKindCH)//'CH_tlm'
 228      do varIndex= 1, bgStats%numvar3d+bgStats%numvar2d
 229   
 230        if (vnl_varKindFromVarname(vnl_varNameList(varIndex)) /= 'CH') cycle
 231    
 232        if ( present(stateVectorRef_opt) ) then
 233          call gvt_transform( statevector,  &                          ! INOUT
 234                              trim(transform), &                       ! IN
 235                              stateVectorRef_opt=stateVectorRef_opt, & ! IN
 236			      varName_opt=bgStats%varNameList(varIndex) ) ! IN
 237        else
 238          call gvt_transform( statevector,  &                          ! INOUT
 239                              trim(transform), &                       ! IN
 240                              varName_opt=bgStats%varNameList(varIndex) ) ! IN
 241        end if
 242
 243      end do
 244    end if
 245    
 246    deallocate(gd_out)
 247
 248    if (mmpi_myid == 0) write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 249    if (mmpi_myid == 0) write(*,*) 'bchm_bsqrt: done'
 250
 251  end subroutine bchm_bSqrt

 255  !--------------------------------------------------------------------------
 256  subroutine bchm_bSqrtAd(statevector,controlVector_out, stateVectorRef_opt)
 257    !
 258    !:Purpose: To apply adjoint of B_CHM^1/2 to a statevector.
 259    !
 260    !
 261    ! Based on bhi_bSqrtAd.
 262    
 263    implicit none
 264
 265    !Arguments
 266    real(8), intent(inout) :: controlVector_out(cvDim_mpilocal)
 267    type(struct_gsv), intent(inout) :: statevector
 268    type(struct_gsv), intent(in), optional :: stateVectorRef_opt
 269    
 270    !Locals
 271    real(8), allocatable :: gd_in(:,:,:)
 272    real(8)   :: hiControlVector(nla_mpilocal,2,bgStats%nkgdim)
 273    character(len=30) :: transform
 274    integer :: varIndex  
 275
 276    if ( .not.initialized ) then
 277      if (mmpi_myid == 0) write(*,*) 'bMatrixChem not initialized'
 278      return
 279    end if
 280
 281    if (mmpi_myid == 0) write(*,*) 'bchm_bsqrtad: starting'
 282    if (mmpi_myid == 0) write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 283
 284    allocate(gd_in(myLonBeg:myLonEnd,myLatBeg:myLatEnd,bgStats%nkgdim))
 285
 286    if ( trim(transformVarKindCH) /= '' ) then  
 287          
 288      transform = trim(transformVarKindCH)//'CH_ad'            
 289      do varIndex = 1, bgStats%numvar3d+bgStats%numvar2d
 290   
 291        if (vnl_varKindFromVarname(vnl_varNameList(varIndex)) /= 'CH') cycle
 292    
 293        if ( present(stateVectorRef_opt) ) then
 294          call gvt_transform( statevector,  &                          ! INOUT
 295                              trim(transform), &                       ! IN
 296                              stateVectorRef_opt=stateVectorRef_opt, &  ! IN
 297			      varName_opt=bgStats%varNameList(varIndex) ) ! IN
 298        else
 299          call gvt_transform( statevector,  &                          ! INOUT
 300                              trim(transform), &                       ! IN
 301                              varName_opt=bgStats%varNameList(varIndex) ) ! IN
 302        end if
 303
 304      end do
 305    end if
 306
 307    call bchm_copyFromStatevector(statevector,gd_in)
 308
 309    call bchm_spa2gdad(gd_in,hiControlVector)
 310
 311    call bchm_cainad(hiControlVector,controlVector_out)
 312
 313    deallocate(gd_in)
 314
 315    if (mmpi_myid == 0) write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 316    if (mmpi_myid == 0) write(*,*) 'bchm_bsqrtad: done'
 317
 318  end subroutine bchm_bSqrtAd

 320  !--------------------------------------------------------------------------
 321  ! bchm_cain
 322  !--------------------------------------------------------------------------
 323  subroutine bchm_cain(controlVector_in,hiControlVector_out)
 324    !
 325    implicit none
 326
 327    !Arguments
 328    real(8), intent(inout) :: controlVector_in(cvDim_mpilocal)
 329    real(8), intent(inout) :: hiControlVector_out(nla_mpilocal,2,bgStats%nkgdim)
 330
 331    !Locals
 332    integer :: jdim, levelIndex, jm, jn, ila_mpilocal, ila_mpiglobal
 333
 334    jdim = 0
 335    hiControlVector_out(:,:,:) = 0.0d0
 336    do levelIndex = 1, bgStats%nkgdim
 337      do jm = mymBeg, mymEnd, mymSkip
 338        do jn = mynBeg, mynEnd, mynSkip
 339          if (jm <= jn) then
 340            ila_mpiglobal = gst_getNIND(jm,gstID) + jn - jm
 341            ila_mpilocal  = ilaList_mpilocal(ila_mpiglobal)
 342            if (jm == 0) then
 343              ! only real component for jm=0
 344              jdim = jdim + 1
 345              hiControlVector_out(ila_mpilocal,1,levelIndex) = controlVector_in(jdim)
 346            else
 347              ! both real and imaginary components for jm>0
 348              jdim = jdim + 1
 349              hiControlVector_out(ila_mpilocal,1,levelIndex) = controlVector_in(jdim)
 350              jdim = jdim + 1
 351              hiControlVector_out(ila_mpilocal,2,levelIndex) = controlVector_in(jdim)
 352            end if
 353          end if
 354        end do
 355      end do
 356    end do
 357
 358  end subroutine bchm_cain

 361  ! bchm_cainAd
 362  !--------------------------------------------------------------------------
 363  subroutine bchm_cainAd(hiControlVector_in,controlVector_out)
 364    implicit none
 365
 366    !Arguments
 367    real(8), intent(inout) :: controlVector_out(cvDim_mpilocal)
 368    real(8), intent(inout) :: hiControlVector_in(nla_mpilocal,2,bgStats%nkgdim)
 369
 370    !Locals
 371    integer :: jdim, levelIndex, jm, jn, ila_mpilocal, ila_mpiglobal
 372
 373    jdim = 0
 374    do levelIndex = 1, bgStats%nkgdim
 375      do jm = mymBeg, mymEnd, mymSkip
 376        do jn = mynBeg, mynEnd, mynSkip
 377          if (jm <= jn) then
 378            ila_mpiglobal = gst_getNIND(jm,gstID) + jn - jm
 379            ila_mpilocal  = ilaList_mpilocal(ila_mpiglobal)
 380            if (jm == 0) then
 381              ! only real component for jm=0
 382              jdim = jdim + 1
 383              controlVector_out(jdim) = controlVector_out(jdim) + hiControlVector_in(ila_mpilocal,1,levelIndex)
 384            else
 385              ! both real and imaginary components for jm>0
 386              jdim = jdim + 1
 387              controlVector_out(jdim) = controlVector_out(jdim) + hiControlVector_in(ila_mpilocal,1,levelIndex)*2.0d0
 388              jdim = jdim + 1
 389              controlVector_out(jdim) = controlVector_out(jdim) + hiControlVector_in(ila_mpilocal,2,levelIndex)*2.0d0
 390            end if
 391          end if
 392        end do
 393      end do
 394    end do
 395
 396  end subroutine bchm_cainAd

 398  !--------------------------------------------------------------------------
 399  ! bchm_spa2gd
 400  !--------------------------------------------------------------------------
 401  subroutine bchm_spa2gd(hiControlVector_in,gd_out)
 402    implicit none
 403
 404    !Arguments
 405    real(8), intent(inout) :: hiControlVector_in(nla_mpilocal,2,bgStats%nkgdim)
 406    real(8), intent(inout) :: gd_out(myLonBeg:myLonEnd,myLatBeg:myLatEnd,bgStats%nkgdim)
 407
 408    !Locals
 409    real(8) :: sp(nla_mpilocal,2,bgStats%nkgdim)
 410
 411    integer :: jn,jm,ila_mpilocal,ila_mpiglobal,icount
 412    real(8) :: sq2
 413    real(8) , allocatable :: zsp(:,:,:), zsp2(:,:,:)
 414    integer :: levelIndex, lonIndex, latIndex
 415    real(8), target  :: gd(myLonBeg:myLonEnd,myLatBeg:myLatEnd,bgStats%nkgdim)
 416
 417    ! maybe not needed:
 418    sp(:,:,:) = 0.0d0
 419    sq2 = sqrt(2.0d0)
 420
 421    allocate(zsp(bgStats%nkgdim,2,mymCount))
 422    allocate(zsp2(bgStats%nkgdim,2,mymCount))
 423
 424    !$OMP PARALLEL DO PRIVATE(jn,jm,levelIndex,ila_mpiglobal,ila_mpilocal, &
 425      zsp2,zsp,icount)
 426    do jn = mynBeg, mynEnd, mynSkip
 427
 428      icount = 0
 429      do jm = mymBeg, mymEnd, mymSkip
 430        if (jm <= jn) then
 431          icount = icount+1
 432          ila_mpiglobal = gst_getNIND(jm,gstID) + jn - jm
 433          ila_mpilocal = ilaList_mpilocal(ila_mpiglobal)
 434          do levelIndex = 1, bgStats%nkgdim
 435            zsp(levelIndex,1,icount) = hiControlVector_in(ila_mpilocal,1, &
 436	      levelIndex)
 437            zsp(levelIndex,2,icount) = hiControlVector_in(ila_mpilocal,2, &
 438	      levelIndex)
 439          end do
 440        end if
 441      end do
 442      if (icount > 0) then
 443
 444        CALL DGEMM('N','N',bgStats%nkgdim,2*icount,bgStats%nkgdim,1.0d0, &
 445	  bgStats%corns(1,1,jn),bgStats%nkgdim,zsp(1,1,1), &
 446	  bgStats%nkgdim,0.0d0,zsp2(1,1,1),bgStats%nkgdim)
 447
 448        icount = 0
 449        do jm = mymBeg, mymEnd, mymSkip
 450          if (jm <= jn) then
 451            icount = icount+1
 452            ila_mpiglobal = gst_getNIND(jm,gstID) + jn - jm
 453            ila_mpilocal = ilaList_mpilocal(ila_mpiglobal)
 454            do levelIndex = 1, bgStats%nkgdim
 455              sp(ila_mpilocal,1,levelIndex) = zsp2(levelIndex,1,icount)
 456              sp(ila_mpilocal,2,levelIndex) = zsp2(levelIndex,2,icount)
 457            end do
 458          end if
 459        end do
 460
 461      end if
 462
 463      ! make adjustments for jm=0
 464      if (mymBeg == 0) then
 465
 466        ila_mpiglobal = gst_getNind(0,gstID) + jn
 467        ila_mpilocal = ilaList_mpilocal(ila_mpiglobal)
 468
 469        do levelIndex = 1, bgStats%nkgdim
 470          sp(ila_mpilocal,1,levelIndex) = sp(ila_mpilocal,1,levelIndex)*sq2
 471          sp(ila_mpilocal,2,levelIndex) = 0.0d0
 472        end do
 473
 474      end if
 475
 476    end do
 477    !$OMP END PARALLEL DO
 478    deallocate(zsp)
 479    deallocate(zsp2)
 480
 481
 482    !$OMP PARALLEL DO PRIVATE(latIndex,levelIndex,lonIndex)
 483    do levelIndex = 1, bgStats%nkgdim
 484      do latIndex = myLatBeg, myLatEnd
 485        do lonIndex = myLonBeg, myLonEnd
 486          gd(lonIndex,latIndex,levelIndex) = 0.0d0
 487        end do
 488      end do
 489    end do
 490    !$OMP END PARALLEL DO
 491
 492    call gst_setID(gstID)
 493    call gst_speree(sp,gd)
 494    call gst_setID(gstID2)
 495
 496    !$OMP PARALLEL DO PRIVATE(latIndex,levelIndex,lonIndex)
 497    do levelIndex = 1, bgStats%nkgdim
 498      do latIndex = myLatBeg, myLatEnd
 499        do lonIndex = myLonBeg, myLonEnd
 500          gd(lonIndex,latIndex,levelIndex) = gd(lonIndex,latIndex,levelIndex) &
 501	    *bgStats%stddev(lonIndex,latIndex,levelIndex)
 502        end do
 503      end do
 504    end do
 505    !$OMP END PARALLEL DO
 506    
 507    !$OMP PARALLEL DO PRIVATE(latIndex,levelIndex,lonIndex)
 508    do levelIndex = 1, bgStats%nkgdim
 509      do latIndex = myLatBeg, myLatEnd
 510        do lonIndex = myLonBeg, myLonEnd
 511          gd_out(lonIndex,latIndex,levelIndex) = gd(lonIndex,latIndex,levelIndex)
 512        end do
 513      end do
 514    end do
 515    !$OMP END PARALLEL DO
 516
 517  end subroutine bchm_spa2gd

 519  !--------------------------------------------------------------------------
 520  ! bchm_spa2gdad
 521  !--------------------------------------------------------------------------
 522  subroutine bchm_spa2gdad(gd_in,hiControlVector_out)
 523    implicit none
 524
 525    !Arguments
 526    real(8), intent(inout) :: hiControlVector_out(nla_mpilocal,2,bgStats%nkgdim)
 527    real(8), intent(inout) :: gd_in(myLonBeg:myLonEnd,myLatBeg:myLatEnd,bgStats%nkgdim)
 528
 529    !Locals
 530    real(8) :: sp(nla_mpilocal,2,bgStats%nkgdim)
 531
 532    integer :: jn, jm, ila_mpilocal, ila_mpiglobal, icount
 533    real(8) :: sq2
 534    real(8), allocatable :: zsp(:,:,:), zsp2(:,:,:)
 535
 536    integer :: levelIndex, lonIndex, latIndex
 537    real(8), target :: gd(myLonBeg:myLonEnd,myLatBeg:myLatEnd,bgStats%nkgdim)
 538
 539    !$OMP PARALLEL DO PRIVATE(latIndex,levelIndex,lonIndex)
 540    do levelIndex = 1, bgStats%nkgdim
 541      do latIndex = myLatBeg, myLatEnd
 542        do lonIndex = myLonBeg, myLonEnd                                                      
 543          gd(lonIndex,latIndex,levelIndex) = gd_in(lonIndex,latIndex,levelIndex)
 544        end do
 545      end do
 546    end do
 547    !$OMP END PARALLEL DO
 548
 549    !$OMP PARALLEL DO PRIVATE(latIndex,levelIndex,lonIndex)
 550    do levelIndex = 1, bgStats%nkgdim
 551      do latIndex = myLatBeg, myLatEnd
 552        do lonIndex = myLonBeg, myLonEnd
 553          gd(lonIndex,latIndex,levelIndex) = gd(lonIndex,latIndex,levelIndex)* &
 554 	     bgStats%stddev(lonIndex,latIndex,levelIndex)
 555        end do
 556      end do
 557    end do
 558    !$OMP END PARALLEL DO
 559
 560    call gst_setID(gstID)
 561    call gst_speree_ad(sp,gd)
 562
 563    hiControlVector_out(:,:,:) = 0.0d0
 564    sq2 = sqrt(2.0d0)
 565    allocate(zsp(bgStats%nkgdim,2,mymCount))
 566    allocate(zsp2(bgStats%nkgdim,2,mymCount))
 567
