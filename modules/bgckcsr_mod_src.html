 76  subroutine csrbg_bgCheckCSR (obsSpaceData)
 77
 78    !: Purpose: Effectuer le controle que qualite sur les donnees CSR.  
 79    !           Modifier les marqueurs de donnees selon le type de rejet.
 80    implicit none
 81
 82    !argument:
 83    type(struct_obs),     intent(inout)  :: obsSpaceData           ! obspaceData Object
 84
 85    ! Locals
 86
 87    real   , allocatable  :: obsTb(:)            ! brightness temperature (btyp=9248/9264,ele=12163)
 88    real   , allocatable  :: ompTb(:)            ! OMP values
 89    real   , allocatable  :: satZenithAngle(:)   ! satellite zenith angle (btyp=3072,ele=7024)
 90    integer, allocatable  :: obsFlags(:)         ! data flags
 91    integer, allocatable  :: cloudAmount(:)      ! data flags
 92    integer, allocatable  :: obsChannels(:)      ! Tb Channels 
 93    integer, allocatable  :: obsDate(:)          ! date YYYYMMDD
 94    integer, allocatable  :: obsHour(:)          ! Hour HHMM
 95    integer               :: sensorIndex         ! find tvs_sensor index corresponding to current obs
 96    character(len=9)      :: burpFileSatId       ! Platform Name
 97
 98    ! Data derived from elements read in obspace data
 99
100    logical, allocatable  :: maxAngleReached(:)  ! satellite angle exceed max angle at obs
101    logical, allocatable  :: topographicData(:)  ! data flagged as topo data
102    logical, allocatable  :: nonCorrectedData(:) ! data non corrected by bias corr
103    logical, allocatable  :: isTbPresent(:)      ! non missing data
104    logical, allocatable  :: isClearSky(:)       ! clear sky obs
105    logical, allocatable  :: strayLight(:)       !
106    logical, allocatable  :: goesMidi(:)         ! goes noon
107    logical, allocatable  :: isToAssim(:)        ! is channel assimilable
108    logical, allocatable  :: ompOutOfRange(:)        
109    integer               :: categorieRejet(7)        
110    integer               :: headerIndex 
111    integer               :: codtyp
112
113    logical               :: csrDataPresent
114
115
116    
117
118    categorieRejet(:) = 0
119    csrDataPresent = .false.
120    call obs_set_current_header_list(obsSpaceData,'TO')
121    HEADER0: do
122      headerIndex = obs_getHeaderIndex(obsSpaceData)
123      if (headerIndex < 0) exit HEADER0
124      codtyp = obs_headElem_i(obsSpaceData, OBS_ITY, headerIndex)
125      if (tvs_isIdBurpInst(codtyp,'radianceclear'  )) csrDataPresent = .true.
126    end do HEADER0
127
128    if ( .not. csrDataPresent ) then
129      write(*,*) 'WARNING: WILL NOT RUN csrbg_bgCheckCSR since no CSR Data'
130      return
131    end if
132
133    call utl_tmg_start(116,'--BgckCSR')
134    write(*,*) ' CSRBG QC PROGRAM STARTS ....'
135    ! Read Namelist
136    call csrbg_init()
137
138    call obs_set_current_header_list(obsSpaceData,'TO')
139    HEADER: do
140      headerIndex = obs_getHeaderIndex(obsSpaceData)
141      if (headerIndex < 0) exit HEADER
142      codtyp = obs_headElem_i(obsSpaceData, OBS_ITY, headerIndex)
143      if ( .not. tvs_isIdBurpInst(codtyp,'radianceclear') ) then 
144        write(*,*) 'WARNING: Observation with codtyp = ', codtyp, ' is not a CSR data'
145        cycle HEADER
146      end if
147
148      !###############################################################################
149      ! STEP 1) read obs from obsSpacedata to start QC                               !
150      !###############################################################################
151
152      call csrbg_readObsFromObsSpace(obsSpaceData, headerIndex, obsTb, ompTb, satZenithAngle, obsFlags, cloudAmount, &
153                                     obsChannels, sensorIndex, obsDate, obsHour, burpFileSatId, &
154                                     maxAngleReached, topographicData, nonCorrectedData, isTbPresent, &
155                                     isClearSky, strayLight, goesMidi, isToAssim, ompOutOfRange) 
156
157      !###############################################################################
158      ! STEP 2) Controle de qualite des CSR. Data QC flags (obsFlags) are modified here!
159      !###############################################################################
160
161      call csrbg_csrCheckQc(obsFlags, categorierejet, maxAngleReached, topographicData, &
162                            nonCorrectedData, isTbPresent, isClearSky, strayLight, goesMidi, isToAssim, &
163                            ompOutOfRange)
164
165      !###############################################################################
166      ! STEP 3) Update Flags and obs in obsspace data
167      !###############################################################################
168
169      call csrbg_updateObsSpaceAfterQc(obsSpaceData, obsFlags, headerIndex, sensorIndex)
170
171    end do HEADER 
172    write(*,*) "Nombre de donnees rejetees"
173    write(*,*) "Attention, une donnee peut etre rejetee pour plusieurs raisons"
174    write(*,*) "Maxangle, straylight ou goesmid    " , categorieRejet(1)
175    write(*,*) "Topographie                        " , categorieRejet(2)
176    write(*,*) "TB sans correction de biais        " , categorieRejet(3)
177    write(*,*) "Pas TB                             " , categorieRejet(4)
178    write(*,*) "std(O-P)*sigma trop grand          " , categorieRejet(5)
179    write(*,*) "canal non assimile                 " , categorieRejet(6)
180    write(*,*) "Ciel non clair                     " , categorieRejet(7)
181    write(*,*) "*******"
182
183    call utl_tmg_stop(116)
184  end subroutine csrbg_bgCheckCSR

190  subroutine csrbg_readObsFromObsSpace(obsSpaceData, headerIndex, obsTb, ompTb, satZenithAngle, obsFlags, cloudAmount, &
191                                       obsChannels, sensorIndex, obsDate, obsHour, burpFileSatId, &
192                                       maxAngleReached, topographicData, nonCorrectedData, isTbPresent, &
193                                       isClearSky, strayLight, goesMidi, isToAssim, ompOutOfRange)
194
195    !:Purpose:        copy headers and bodies from obsSpaceData object to arrays
196    !                 compute some parameters from the read variables
197
198    implicit None
199
200    
201    integer , intent(in)                 :: headerIndex         ! current header index
202    real   , allocatable, intent(out)    :: obsTb(:)            ! brightness temperature (btyp=9248/9264,ele=12163)
203    real   , allocatable, intent(out)    :: ompTb(:)            ! OMP values
204    real   , allocatable, intent(out)    :: satZenithAngle(:)   ! satellite zenith angle (btyp=3072,ele=7024)
205    integer, allocatable, intent(out)    :: obsFlags(:)         ! data flags
206    integer, allocatable, intent(out)    :: cloudAmount(:)      ! data flags
207    integer, allocatable, intent(out)    :: obsChannels(:)      ! Tb Channels 
208    integer, allocatable, intent(out)    :: obsDate(:)          ! date YYYYMMDD
209    integer, allocatable, intent(out)    :: obsHour(:)          ! Hour HHMM
210    integer,              intent(out)    :: sensorIndex         ! find tvs_sensor index corresponding to current obs
211    character(*),intent(out)             :: burpFileSatId       ! Platform Name
212    type(struct_obs),     intent(inout)  :: obsSpaceData        ! obspaceData Object
213
214    ! Data derived from elements read in obspace data
215
216    logical, allocatable, intent(out)    :: maxAngleReached(:)     ! satellite angle exceed max angle at obs
217    logical, allocatable, intent(out)    :: topographicData(:)  ! data flagged as topo data
218    logical, allocatable, intent(out)    :: nonCorrectedData(:) ! data non corrected by bias corr
219    logical, allocatable, intent(out)    :: isTbPresent(:)      ! non missing data
220    logical, allocatable, intent(out)    :: isClearSky(:)       ! clear sky obs
221    logical, allocatable, intent(out)    :: strayLight(:)       !
222    logical, allocatable, intent(out)    :: goesMidi(:)         ! goes noon
223    logical, allocatable, intent(out)    :: isToAssim(:)        ! is channel assimilable
224    logical, allocatable, intent(out)    :: ompOutOfRange(:)    ! 
225  
226    ! Locals
227    integer                              :: bodyIndex
228    integer                              :: bodyIndexbeg
229    integer                              :: headerCompt
230    integer                              :: currentChannelNumber
231    integer                              :: channelIndex
232    integer                              :: actualnumChannel
233    integer                              :: numObsToProcess
234    integer                              :: iplatform
235    integer                              :: instrum
236    integer                              :: isat, iplatf
237    integer                              :: instr
238    integer                              :: mois
239    integer                              :: jour
240    integer                              :: heure
241    integer                              :: midnight
242    integer                              :: dataIndex
243    integer                              :: indexSat
244    logical                              :: sensorIndexFound
245    logical                              :: indexSatFound
246    real                                 :: errorThreshold
247
248
249
250    ! find tvs_sensor index corresponding to current obs
251
252    iplatf      = obs_headElem_i( obsSpaceData, OBS_SAT, headerIndex )
253    instr       = obs_headElem_i( obsSpaceData, OBS_INS, headerIndex )
254
255    call tvs_mapSat( iplatf, iplatform, isat )
256    call tvs_mapInstrum( instr, instrum )
257
258    sensorIndexFound = .false.
259    do sensorIndex =1, tvs_nsensors
260      if ( iplatform ==  tvs_platforms(sensorIndex)  .and. &
261           isat      ==  tvs_satellites(sensorIndex) .and. &
262           instrum   == tvs_instruments(sensorIndex)       ) then
263        sensorIndexFound = .true.
264        exit
265      end if
266    end do
267    
268    if ( .not. sensorIndexFound ) call utl_abort('csrbg_readObsFromObsSpace: sensor Index not found')
269
270    ! find actual Number of channels
271    actualNumChannel = tvs_coefs(sensorIndex)%coef%fmv_ori_nchn
272    
273
274    headerCompt = 1
275    numObsToProcess = 1
276    ! Allocate Header elements
277    call utl_reAllocate(obsDate, numObsToProcess)
278    call utl_reAllocate(obsHour, numObsToProcess)
279    call utl_reAllocate(satZenithAngle, numObsToProcess)
280
281    ! Allocate Body elements
282    call utl_reAllocate(obsTb, numObsToProcess*actualNumChannel)
283    call utl_reAllocate(ompTb, numObsToProcess*actualNumChannel)
284    call utl_reAllocate(obsFlags, numObsToProcess*actualNumChannel)
285    call utl_reAllocate(cloudAmount, numObsToProcess*actualNumChannel)
286    call utl_reAllocate(obsChannels, numObsToProcess*actualNumChannel)
287
288    !initialization
289    obsTb(:) = csrbg_realMissing
290    ompTb(:) = csrbg_realMissing
291    obsChannels(:) = csrbg_realMissing
292    cloudAmount(:) = csrbg_realMissing
293    !
294    ! Header elements
295    !
296    burpFileSatId                      = obs_elem_c    ( obsSpaceData, 'STID' , headerIndex )
297    satZenithAngle(headerCompt)        = obs_headElem_r( obsSpaceData, OBS_SZA, headerIndex )
298    obsDate(headerCompt)               = obs_headElem_i( obsSpaceData, OBS_DAT, headerIndex )
299    obsHour(headerCompt)               = obs_headElem_i( obsSpaceData, OBS_ETM, headerIndex )
300    !
301    ! Body elements
302    !
303    bodyIndexbeg        = obs_headElem_i( obsSpaceData, OBS_RLN, headerIndex )
304
305    BODY: do bodyIndex =  bodyIndexbeg, bodyIndexbeg + obs_headElem_i( obsSpaceData, OBS_NLV, headerIndex ) - 1
306      currentChannelNumber = nint(obs_bodyElem_r( obsSpaceData,  OBS_PPP, bodyIndex )) - & 
307                             tvs_channelOffset(sensorIndex)
308      obsTb(currentChannelNumber)          = obs_bodyElem_r( obsSpaceData, OBS_VAR, bodyIndex )
309      ompTb(currentChannelNumber)          = obs_bodyElem_r( obsSpaceData, OBS_OMP, bodyIndex )
310      obsFlags(currentChannelNumber)       = obs_bodyElem_i( obsSpaceData, OBS_FLG, bodyIndex )
311      cloudAmount(currentChannelNumber)    = obs_bodyElem_i( obsSpaceData, OBS_CLA, bodyIndex)
312
313    end do BODY
314
315    do channelIndex=1, actualNumChannel
316      obsChannels(channelIndex)    = channelIndex+tvs_channelOffset(sensorIndex)
317    end do
318
319
320    ! Get index for satCloudCoverLimit for current current sat  
321    indexSatFound = .false.
322    do indexSat = 1, maxNumSat
323      if (trim(burpSatName(indexSat)) == trim(burpFileSatId)) then
324        indexSatFound = .true.
325        exit
326      end if 
327    end do
328    if ( .not. indexSatFound ) call utl_abort('csrbg_readObsFromObsSpace: Cloud Cover Limit Not & 
329                                               Found for ' // trim(burpFileSatId))
330    ! compute data for QC
331    call utl_reAllocate(topographicData, numObsToProcess)
332    call utl_reAllocate(maxAngleReached, numObsToProcess)
333    call utl_reAllocate(strayLight, numObsToProcess)
334    call utl_reAllocate(goesMidi, numObsToProcess)
335    call utl_reAllocate(isTbPresent, numObsToProcess*actualNumChannel)
336    call utl_reAllocate(nonCorrectedData, numObsToProcess*actualNumChannel)
337    call utl_reAllocate(isClearSky, numObsToProcess*actualNumChannel)
338    call utl_reAllocate(ompOutOfRange, numObsToProcess*actualNumChannel)
339    call utl_reAllocate(isToAssim, numObsToProcess*actualNumChannel)
340    
341    isTbPresent(:) = .false.
342    isClearSky(:) = .false.
343    nonCorrectedData = .false.
344    
345    do channelIndex=1, numObsToProcess*actualNumChannel
346      if (obsTb(channelIndex) /= csrbg_realMissing) isTbPresent(channelIndex) = .true.
347      if (cloudAmount(channelIndex) /= csrbg_realMissing .and. & 
348          cloudAmount(channelIndex) < satCloudCoverlimit(indexSat,channelIndex)) then
349        isClearSky(channelIndex) = .true.
350      end if 
351      if (.not. btest(obsFlags(channelIndex), 6)) nonCorrectedData(channelIndex) = .true.
352    end do
353
354    topographicData(:) = .false.
355    maxAngleReached(:) = .false.
356    do dataIndex=1, numObsToProcess
357      if (btest(obsFlags(dataIndex), 18)) topographicData(dataIndex) = .true.
358      if (satZenithAngle(dataIndex) > 15250) maxAngleReached(dataIndex) = .true.
359    end do
360
361    !!! check O-P of Tb and set assim flag for each channel of satellite numsat using stat_iutilst
362    !!! If channel not found in stats file, NO ASSIMILATE flag is set and O-P check is skipped
363    ompOutOfRange(:) = .false.
364    isToAssim(:) = .false.
365    do  channelIndex=1, numObsToProcess*actualNumChannel
366      if ( .not. oer_tovutil(obsChannels(channelIndex), sensorIndex) == 0) then
367        isToAssim(channelIndex) = .true.
368      end if
369      errorThreshold = oer_toverrst(obsChannels(channelIndex), sensorIndex)*csrbg_ompThreshold
370      if (abs(ompTb(channelIndex)) > errorThreshold) ompOutOfRange(channelIndex) = .true.
371    end do
372    
373    strayLight(:) = .false.
374    goesMidi(:) = .false.
375    do  dataIndex = 1, numObsToProcess
376      mois  = (obsDate(dataIndex)/100) - (obsDate(dataIndex)/10000)*100
377      jour  = (obsDate(dataIndex)/100) -  (obsDate(dataIndex)/100)*100
378      heure = obsHour(dataIndex)/100 
379      if (burpFileSatId == '^METSAT7') then
380        if ( heure == 21 ) straylight(dataIndex) = 1
381        if ( (heure == 20) .or. (heure == 22) ) then
382          if ( (jour + (mois-1)*30) > 20 .and. (jour + (mois-1)*30) < 135 ) straylight(dataIndex) = .true.
383          if ( (jour + (mois-1)*30) > 210 .and. (jour + (mois-1)*30) < 300 ) straylight(dataIndex) = .true.
384        end if
385      end if
386      if (burpFileSatId == '^GOES11' .or.  burpFileSatId == '^GOES15') midnight = 9
387      if (burpFileSatId == '^GOES13' .or.  burpFileSatId == '^GOES14') midnight = 5
388      if (burpFileSatId == '^GOES11' .or.  burpFileSatId == '^GOES15' .or. &
389          burpFileSatId == '^GOES13' .or.  burpFileSatId == '^GOES14') then
390        if ( (heure == (midnight - 1)) .or. (heure == midnight) ) goesMidi(dataIndex) = .true.
391      end if 
392    end do
393    
394  end subroutine csrbg_readObsFromObsSpace

401  subroutine csrbg_csrCheckQc(obsFlags, categorieRejet, maxAngleReached, topographicData, &
402                              nonCorrectedData, isTbPresent, isClearSky, strayLight, goesMidi, isToAssim, &
403                              ompOutOfRange)
404
405    !:Purpose:        Modify obsFlags
406
407    implicit None
408
409
410    integer,              intent(inout)     :: obsFlags(:)         ! obs Flags to update
411    integer,              intent(inout)     :: categorieRejet(7)   ! the 7 categories of rejections
412    logical,              intent(in)        :: maxAngleReached(:)  ! satellite angle exceed max angle at obs
413    logical,              intent(in)        :: topographicData(:)  ! data flagged as topo data
414    logical,              intent(in)        :: nonCorrectedData(:) ! data non corrected by bias corr
415    logical,              intent(in)        :: isTbPresent(:)      ! non missing data
416    logical,              intent(in)        :: isClearSky(:)       ! clear sky obs
417    logical,              intent(in)        :: strayLight(:)       !
418    logical,              intent(in)        :: goesMidi(:)         ! goes noon
419    logical,              intent(in)        :: isToAssim(:)        ! is channel assimilable
420    logical,              intent(in)        :: ompOutOfRange(:)    ! abs of omp greater than threshold 
421    
422    ! Locals
423    integer                                 :: currentChannelNumber
424    integer                                 :: channelIndex
425    integer                                 :: numObsToProcess
426    integer                                 :: numData
427    integer                                 :: dataIndex
428
429
430    numObsToProcess = 1
431    currentChannelNumber = size(obsFlags)/numObsToProcess
432
433    ! tests de controle de qualite sur les profils et canaux
434    numData = 0
435    do dataIndex = 1, numObsToProcess
436      do channelIndex = 1, currentChannelNumber
437        numData = numData+1
438        !Maxangle & Straylight & Goesmid : Bit 7 et 9 pour tous les canaux
439        if (maxAngleReached(dataIndex) == .true. .or. & 
440            strayLight(dataIndex)      == .true. .or. & 
441            goesMidi(dataIndex)        == .true. ) then 
442          obsFlags(numData) = ibset(obsFlags(numData),7) 
443          obsFlags(numData) = ibset(obsFlags(numData),9) 
444          categorieRejet(1) =  categorieRejet(1) + 1
445        end if
446
447        !Topo : bit 9 et 18 sont dÃ©jÃ  lÃ  provenant du EnVar
448        if (topographicData(dataIndex) == .true.) then
449          obsFlags(numData) = ibset(obsFlags(numData),9) 
450          obsFlags(numData) = ibset(obsFlags(numData),18) 
451          categorieRejet(2) =  categorieRejet(2) + 1
452        end if 
453
454        !Pas corrige : bit 11
455        if (nonCorrectedData(numData) == .true.) then
456          obsFlags(numData) = ibset(obsFlags(numData),11) 
457          categorieRejet(3) =  categorieRejet(3) + 1
458        end if 
459
460        !Tbyela: bit  7 et 9 pour les canaux concernÃ©s
461        if (isTbPresent(numData) == .false.) then 
462          obsFlags(numData) = ibset(obsFlags(numData),7) 
463          obsFlags(numData) = ibset(obsFlags(numData),9) 
464          categorieRejet(4) =  categorieRejet(4) + 1
465        end if
466
467        !omp Out of range: Bit 9 et 16 pour les canaux concernÃ©s
468        if (ompOutOfRange(numData) == .true.) then 
469          obsFlags(numData) = ibset(obsFlags(numData),9)
470          obsFlags(numData) = ibset(obsFlags(numData),16)
471          categorieRejet(5) =  categorieRejet(5) + 1
472        end if
473
474        !Assim: Bit 11 pour les canaux concernÃ©s
475        if (isToAssim(numData) == .false.) then 
476          obsFlags(numData) = ibset(obsFlags(numData),11)
477          categorieRejet(6) =  categorieRejet(6) + 1
478        end if 
479
480        !Clearsky : Bit 7 & Bit 9
481        if (isClearSky(numData) == .false.) then 
482          obsFlags(numData) = ibset(obsFlags(numData),7)
483          obsFlags(numData) = ibset(obsFlags(numData),9)
484          categorieRejet(7) =  categorieRejet(7) + 1
485        end if  
486      end do
487    end do    
488
489  end subroutine csrbg_csrCheckQc

494  subroutine csrbg_updateObsSpaceAfterQc(obsSpaceData, obsFlags, headerIndex, sensorIndex)
495
496    !:Purpose:      Update obspacedata variables (obstTB and obs flags) after QC
497    implicit None
498
499    !Arguments
500    type(struct_obs),     intent(inout)     :: obsSpaceData           ! obspaceData Object
501    integer,              intent(in)        :: obsFlags(:)            ! data flags
502    integer,              intent(in)        :: sensorIndex            ! sensor Index 
503    integer,              intent(in)        :: headerIndex            ! sensor Index 
504    ! Locals
505    integer                                 :: bodyIndex
506    integer                                 :: bodyIndexbeg
507    integer                                 :: currentChannelNumber
508
509    bodyIndexbeg        = obs_headElem_i( obsSpaceData, OBS_RLN, headerIndex )
510
511    BODY: do bodyIndex =  bodyIndexbeg, bodyIndexbeg + obs_headElem_i( obsSpaceData, OBS_NLV, headerIndex ) - 1
512      currentChannelNumber=nint(obs_bodyElem_r( obsSpaceData,  OBS_PPP, bodyIndex ))-tvs_channelOffset(sensorIndex)
513      call obs_bodySet_i(obsSpaceData, OBS_FLG, bodyIndex, obsFlags(currentChannelNumber))
514    end do BODY
515
516  end subroutine csrbg_updateObsSpaceAfterQc
