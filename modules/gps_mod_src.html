 237  pure function gps_gravitysrf(sLat)
 238    !
 239    !:Purpose: Normal gravity on ellipsoidal surface
 240    !
 241    implicit none
 242
 243    real(dp)              :: gps_gravitysrf ! Normal gravity (m/s2)
 244
 245    ! Arguments:
 246    real(dp), intent(in)  :: sLat ! sin(Latitude)
 247
 248    ! Locals:
 249    real(dp)              :: ks2
 250    real(dp)              :: e2s
 251
 252    ks2 = ec_wgs_TNGk * sLat*sLat
 253    e2s = 1._dp - ec_wgs_e2 * sLat*sLat
 254    gps_gravitysrf = ec_wgs_GammaE * (1._dp + ks2) / sqrt(e2s)
 255  end function gps_gravitysrf
 256
 257  pure function gps_gravityalt(sLat, Altitude)
 258    !
 259    !:Purpose: Normal gravity above the ellipsoidal surface
 260    !
 261    implicit none
 262
 263    real(dp)              :: gps_gravityalt ! Normal gravity (m/s2)
 264
 265    ! Arguments:
 266    real(dp), intent(in)  :: sLat     ! sin(Latitude)
 267    real(dp), intent(in)  :: Altitude ! Altitude (m)
 268
 269    real(dp)              :: C1
 270    real(dp)              :: C2
 271
 272    C1 =-2._dp/ec_wgs_a*(1._dp+ec_wgs_f+ec_wgs_m-2*ec_wgs_f*sLat*sLat)
 273    C2 = 3._dp/ec_wgs_a**2
 274    gps_gravityalt = gps_gravitysrf(sLat)*                                   &
 275         (1._dp + C1 * Altitude + C2 * Altitude**2)
 276  end function gps_gravityalt

 278  pure function gps_geopotential(Latitude, Altitude)
 279    !
 280    !:Purpose: Geopotential energy at a given point.
 281    !          Result is based on the WGS84 approximate expression for the
 282    !          gravity acceleration as a function of latitude and altitude,
 283    !          integrated with the trapezoidal rule.
 284    !
 285    implicit none
 286
 287    real(dp)              :: gps_geopotential ! Geopotential (m2/s2)
 288
 289    ! Arguments:
 290    real(dp), intent(in)  :: Latitude ! (rad)
 291    real(dp), intent(in)  :: Altitude ! (m)
 292
 293    ! Locals:
 294    real(dp)              :: dh, sLat
 295    integer               :: n, i
 296    real(dp), allocatable :: hi(:)
 297    real(dp), allocatable :: gi(:)
 298    
 299    dh = 500._dp
 300    n = 1 + int(Altitude/dh)
 301
 302    allocate(hi(0:n))
 303    allocate(gi(0:n))
 304
 305    sLat=sin(Latitude)
 306
 307    do i = 0, n-1
 308       hi(i) = i * dh
 309       gi(i) = gps_gravityalt(sLat, hi(i))
 310    end do
 311    hi(n) = Altitude
 312    gi(n) = gps_gravityalt(sLat, hi(n))
 313
 314    gps_geopotential = 0._dp
 315    do i = 1, n
 316       gps_geopotential = gps_geopotential + 0.5_dp * (gi(i)+gi(i-1)) * (hi(i)-hi(i-1))
 317    end do
 318
 319    deallocate(hi)
 320    deallocate(gi)
 321  end function gps_geopotential

 323  subroutine gpsRadii(Latitude, RadN, RadM)
 324    implicit none
 325
 326    ! Arguments:
 327    real(dp), intent(in)  :: Latitude
 328    real(dp), intent(out) :: RadN, RadM
 329
 330    ! Locals:
 331    real(dp)              :: sLat, e2s
 332
 333    sLat = sin(Latitude)
 334    e2s = 1._dp - ec_wgs_e2 * sLat * sLat
 335    RadN = ec_wgs_a / sqrt(e2s)
 336    RadM = ec_wgs_a * (1._dp - ec_wgs_e2) / (e2s*sqrt(e2s))
 337  end subroutine gpsRadii

 794  subroutine gps_struct1sw(ngpslev,rLat,rLon,rAzm,rMT,Rad,geoid,    &

 796    implicit none
 797

 801    real(dp)        , intent(in)  :: rLon

 811    real(dp)        , intent(in)  :: rUU (ngpssize)

 813    logical         , optional    :: printHeight

 816
 817    ! Locals:
 818    integer(i4)                    :: i
 819
 820    real(dp) , parameter           :: delta = 0.6077686814144_dp
 821
 822    type(gps_diff)                 :: cmp(ngpssize)
 823    real(dp)                       :: h0,dh,Rgh,Eot,Eot2, sLat, cLat
 824    type(gps_diff)                 :: p, t, q, x
 825    type(gps_diff)                 :: tr, z
 826    type(gps_diff)                 :: mold, dd, dw, dx, n0, nd1, nw1, tvm
 827    type(gps_diff)                 :: xi(ngpssize), tv(ngpssize)
 828
 829    prf%ngpslev = ngpslev
 830    prf%rLat    = rLat
 831    prf%rLon    = rLon
 832    prf%rAzm    = rAzm
 833    prf%rMT     = rMT
 834    prf%Rad     = Rad
 835    prf%geoid   = geoid
 836    call gpsRadii(rLat, prf%RadN, prf%RadM)
 837
 838    !
 839    ! Fill pressure placeholders:
 840    !
 841    prf%P0%Var               = 0.01_dp*rP0
 842    prf%P0%DVar              = 0._dp
 843    prf%P0%DVar(2*ngpslev+1) = 0.01_dp
 844    do i=1,ngpslev
 845       prf%pst(i)%Var               = 0.01_dp*rPP(i)
 846       prf%pst(i)%DVar              = 0._dp
 847       prf%pst(i)%DVar(2*ngpslev+1) = 0.01_dp*rDP(i)
 848    end do
 849
 850    !
 851    ! Fill temperature placeholders:
 852    !
 853    do i = 1, ngpslev
 854       prf%tst(i)%Var               = rTT(i)+MPC_K_C_DEGREE_OFFSET_R8
 855       prf%tst(i)%DVar              = 0._dp

 857    end do
 858
 859    !
 860    ! Fill moisture placeholders:
 861    !
 862    do i = 1, ngpslev
 863       prf%qst(i)%Var               = rHU(i)
 864       prf%qst(i)%DVar              = 0._dp
 865       prf%qst(i)%DVar(ngpslev+i)   = 1._dp
 866    end do
 867
 868    ! Compressibility:
 869    do i = 1, ngpslev
 870       cmp(i)= gpscompressibility(prf%pst(i),prf%tst(i),prf%qst(i))
 871    end do
 872
 873    ! Refractivity:
 874    do i = 1, ngpslev
 875       p  = prf%pst(i)
 876       t  = prf%tst(i)
 877       q  = prf%qst(i)
 878       x  = gps_p_wa*q/(1._dp+gps_p_wb*q)
 879
 880       ! Densities (molar, total, dry, water vapor):
 881       mold  = p/t * (100._dp/(p_R*cmp(i)))               ! note that p is in hPa
 882       dd = mold * (1._dp-x) * (gps_p_md/1000._dp)
 883       dw = mold * x         * (gps_p_mw/1000._dp)
 884       ! Aparicio (2011) expression
 885       tr = MPC_K_C_DEGREE_OFFSET_R8/t-1._dp
 886       nd1= ( 222.682_dp+   0.069_dp*tr) * dd
 887       nw1= (6701.605_dp+6385.886_dp*tr) * dw
 888       n0 = (nd1+nw1)
 889       prf%rst(i) = n0*(1._dp+(1.e-6_dp/6._dp)*n0)
 890    end do
 891
 892    !
 893    ! Hydrostatic equation
 894    !
 895    do i = 1, ngpslev
 896       p = prf%pst(i)
 897       t = prf%tst(i)
 898       q = prf%qst(i)
 899       !
 900       ! Log(P)
 901       !
 902       xi(i) = log(p)
 903       !
 904       ! Virtual temperature (K) (corrected of compressibility)
 905       !
 906       tv(i) = (1._dp+delta*q) * t * cmp(i)
 907    end do
 908
 909    sLat=sin(rLat)
 910    cLat=cos(rLat)
 911    dx  = xi(ngpslev)-log(prf%P0)
 912    Rgh = gps_gravitysrf(sLat)
 913    z   = (-p_Rd/Rgh) * tv(ngpslev) * dx
 914    prf%gst(ngpslev) = rMT + z
 915    do i=ngpslev-1,1,-1
 916       dx = xi(i)-xi(i+1)
 917       tvm = 0.5_dp*(tv(i)+tv(i+1))
 918       !
 919       ! Gravity acceleration (includes 2nd-order Eotvos effect)
 920       !
 921       h0  = prf%gst(i+1)%Var
 922       Eot = 2*ec_wgs_OmegaPrime*cLat*rUU(i)
 923       Eot2= (rUU(i)**2+rVV(i)**2)/ec_wgs_a
 924       Rgh = gps_gravityalt(sLat, h0)-Eot-Eot2
 925       dh  = (-p_Rd/Rgh) * tvm%Var * dx%Var
 926       Rgh = gps_gravityalt(sLat, h0+0.5_dp*dh)-Eot-Eot2
 927       !
 928       ! Height increment
 929       !
 930       z   = (-p_Rd/Rgh) * tvm * dx
 931       prf%gst(i) = prf%gst(i+1) + z
 932    end do
 933
 934    if ( present(printHeight) ) then
 935      if ( printHeight ) then
 936        write(*,*) 'gps_struct1sw, height='
 937        write(*,*) prf%gst(1:ngpslev)%Var
 938
 939        printHeight = .false.
 940      end if
 941    end if
 942
 943    prf%bbst=.false.
 944  end subroutine gps_struct1sw

1060  function gpscompressibility(p,t,q)
1061    implicit none
1062
1063    type(gps_diff)              :: gpscompressibility
1064
1065    ! Arguments:
1066    type(gps_diff), intent(in)  :: p,t,q
1067
1068    ! Locals:
1069    real(dp) , parameter   :: a0= 1.58123e-6_dp
1070    real(dp) , parameter   :: a1=-2.9331e-8_dp
1071    real(dp) , parameter   :: a2= 1.1043e-10_dp
1072    real(dp) , parameter   :: b0= 5.707e-6_dp
1073    real(dp) , parameter   :: b1=-2.051e-8_dp
1074    real(dp) , parameter   :: c0= 1.9898e-4_dp
1075    real(dp) , parameter   :: c1=-2.376e-6_dp
1076    real(dp) , parameter   :: d = 1.83e-11_dp
1077    real(dp) , parameter   :: e =-0.765e-8_dp
1078
1079    type(gps_diff)         :: x,tc,pt,tc2,x2
1080
1081    x  = gps_p_wa*q/(1._dp+gps_p_wb*q)
1082    ! Estimate, from CIPM, Picard (2008)
1083    tc = t-MPC_K_C_DEGREE_OFFSET_R8
1084    pt = 1.e2_dp*p/t
1085    tc2= tc*tc
1086    x2 = x*x
1087    gpscompressibility = 1._dp-pt*(a0+a1*tc+a2*tc2+(b0+b1*tc)*x+(c0+c1*tc)*x2)+pt*pt*(d+e*x2)
1088  end function gpscompressibility

2641    !                 nexp in 1/m

2660    rb = ra + th

2736       jlocm = -1000
2737       jlocp = -1000
2738       if (imp1 > imp(1)) then
2739          jlocm = 1
2740          jlocp = 2
2741       end if
2742
2743       do j=1, ngpslev+ngpsxlow-1
2744          if ((imp1 <= imp(j)) .and. (abs(prf%bst(j)%Var) < 1._dp)) then
2745             jlocm = j
2746          end if
2747       end do
2748
2749       do j=jlocm+1, ngpslev+ngpsxlow
2750          if ((imp1 >  imp(j)) .and. (abs(prf%bst(j)%Var) < 1._dp)) then
2751             jlocp = j
2752             exit
2753          end if
2754       end do
2755      
2756       if (jlocm == -1000) jlocm = ngpslev+ngpsxlow-1
2757       if (jlocp == -1000) jlocp = ngpslev+ngpsxlow
2758
2759       !
2760       ! Find properties in that band
2761       !
2762       am = prf%ast(jlocm)
2763       ap = prf%ast(jlocp)
2764       
2765       da = am - ap
2766       dam = (imp1-ap) / da
2767       dap = (am-imp1) / da
2768       
2769       ! Use loglinear interpolation for most data (notably direct rays)
2770       if (prf%bst(jlocm)%Var > 1.e-6_dp .and. prf%bst(jlocp)%Var > 1.e-6_dp) then
2771          bstv(i)=exp(dam*log(prf%bst(jlocm))+dap*log(prf%bst(jlocp)))*(rad/rad0)
2772       else
2773          ! Use linear interpolation for near-zero or negative bending (most reflected rays)
2774          bstv(i)=(dam*prf%bst(jlocm)+dap*prf%bst(jlocp))*(rad/rad0)
2775       end if
2776    end do
2777  end subroutine gpsbndopv

2779  subroutine gps_bndopv1(impv, azmv, nval, prf, bstv)
2780    !:Purpose: Computation of the observation operator for Bending 
2781    !
2782    !:Note: The Operator is based  from Assimilation experiments withCHAMP GPS radio occultation measurements
2783    !         (S. B. HEALY and J.-N. THEPAUT, 2005)
2784    implicit none
2785
2786    ! Arguments:
2787    real(dp)              , intent(in) :: impv(:), azmv(:)
2788    integer(i4)           , intent(in) :: nval
2789    type(gps_profile)                  :: prf
2790    type(gps_diff)        , intent(out):: bstv(:)
2791
2792    ! Locals:
2793    integer                            :: levIndexObs, ngpslev,last_levIndexObs,numLevels,levelHigh,levelLow,levIndexAnl 
2794    type(gps_diff)                     :: h(ngpssize), nu(ngpssize), lnu(ngpssize), n(ngpssize), z(ngpssize)
2795    type(gps_diff)                     :: N0a, N1a, ka, NAa,Aa, Ba, Ba2, Ba3, delta_alpha, delta_alpha_top, z_0, h_0
2796    real(dp)                           :: a2, a, gz, cazm, sazm, last_a
2797
2798    ! model levels 
2799    ngpslev=prf%ngpslev
2800    do levIndexAnl = 1,ngpslev 
2801      h(levIndexAnl)  = prf%geoid+prf%gst(levIndexAnl)
2802      nu(levIndexAnl) = prf%rst(levIndexAnl)
2803      lnu(levIndexAnl)=log(nu(levIndexAnl)) 
2804      n(levIndexAnl)  = 1._dp+nu(levIndexAnl)*1e-6_dp
2805      z(levIndexAnl)  = n(levIndexAnl)*(prf%Rad+prf%geoid+prf%gst(levIndexAnl))
2806    end do
2807    ! number of observed levels in the profile
2808    numLevels  = size(impv)
2809    if (nval < numLevels) numLevels=nval
2810    
2811    do levIndexObs =  numLevels,1,-1
2812      a2 = impv(levIndexObs)*impv(levIndexObs)
2813      a  = impv(levIndexObs)
2814      cazm = cos(azmv(levIndexObs))
2815      sazm = sin(azmv(levIndexObs))
2816      !find model levels that bracket the observation
2817      !   note to self:   like in GEM, level=1 is the highest level
2818      do levIndexAnl = 1, ngpslev-1
2819        levelHigh = levIndexAnl - 1
2820        levelLow  = levIndexAnl
2821        if (z(levIndexAnl)%VaR< a) exit 
2822          levelLow  = 0
2823      end do
2824    
2825      if  (levelLow/=0) then
2826        h_0  = h(levelLow)+(((a-z(levelLow))/(z(levelHigh)-z(levelLow)))*(h(levelHigh)-h(levelLow)))  
2827        N0a  = nu(levelLow)
2828        gz   = (lnu(levelLow+1)%Var-lnu(levelLow)%Var)/(h(levelLow+1)%Var-h(levelLow)%Var)
2829        NAa  = nu(levelLow)*exp(gz*(h_0-h(levelLow)))
2830        delta_alpha = 0.d0
2831        delta_alpha_top = 0.d0
2832        z_0 = a
2833        do while ((levelHigh)>=1) 
2834          N1a = nu(levelHigh)
2835          ka  = log(N0a/N1a)/(z(levelHigh) - z(levelLow))
2836          ! Test of Reflected
2837          do while (ka%Var<0.or.ka%Var>0.1) 
2838            levelHigh = levelHigh - 1
2839            N1a = nu(levelHigh)
2840            ka = log(N0a/N1a)/(z(levelHigh) - z(levelLow))
2841          end do
2842          Aa = 1e-6_dp* sqrt((MPC_PI_R8/2.d0*a)/ka )*NAa* exp(ka*(z_0-a))  
2843          if (z_0%Var==a) then
2844            Ba  = erf(sqrt(ka*(z(levelHigh)-a)))
2845          else  
2846            Ba2 = erf(sqrt(ka*(z(levelHigh)-a)))
2847            Ba3 = erf(sqrt((ka*(z_0-a))))
2848            Ba  = Ba2 - Ba3
2849          end if   
2850          delta_alpha = delta_alpha+2*ka*Ba*Aa 
2851          N0a = N1a
2852          NAa = N1a 
2853          z_0 = z(levelHigh)
2854
2855          levelLow  = levelHigh 
2856          levelHigh = levelLow-1
2857        end do
2858        Ba = erf(1-erf(sqrt((ka*(z_0-a)))))
2859        delta_alpha_top = 2*Aa*ka*Ba 
2860        last_a = a
2861        last_levIndexObs = levIndexObs
2862        bstv(levIndexObs)= delta_alpha +delta_alpha_top
2863      else  ! (levelLow==0) 
2864        ! Use loglinear extrapolation for most data (notably direct rays)
2865        if (a>(1._dp+prf%rst(ngpslev)%Var*1e-6_dp)*prf%Rad) then
2866          bstv(levIndexObs)=bstv(last_levIndexObs)*exp((-1._dp/6500._dp)*(a-last_a))
2867        else
2868        ! Use linear extrapolation (most reflected rays) from Information content in reflected
2869        ! signals during GPS Radio Occultation observations (Josep Aparicio et al.,2017) 
2870          bstv(levIndexObs)=bstv(last_levIndexObs)*exp((-1._dp/6500._dp)*(a-last_a))-2*acos(a/((1._dp+prf%rst(ngpslev)*1e-6_dp)*prf%Rad))
2871        end if
2872
2873      end if
2874    end do
2875  end subroutine gps_bndopv1
