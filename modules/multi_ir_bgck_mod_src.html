 126  subroutine  irbg_init()
 127    !
 128    !:Purpose: This subroutine reads the namelist section NAMBGCKIR
 129    !          for the module.
 130    implicit none
 131
 132    ! Locals:
 133    integer :: nulnam, ierr
 134    logical, save :: first = .true.
 135    integer, external :: fnom, fclos
 136    integer :: instrumentIndex
 137    namelist /NAMBGCKIR/ ninst, inst, iwindow, iwindow_alt, ilist1, ilist2, ilist2_pair
 138    namelist /NAMBGCKIR/ dtw, dtl, pco2min, pco2max, night_ang, crisCloudFractionThreshold
 139
 140    if (first) then
 141
 142      ! set the default values for namelist variables
 143      ninst = MPC_missingValue_INT
 144      inst(:) = '*EMPTY*'
 145      iwindow(:) = 0
 146      iwindow_alt(:) = 0
 147      ilist1(:,:) = 0
 148      ilist2(:,:) = 0
 149      ilist2_pair(:,:) = 0
 150      dtw = 0.0d0
 151      dtl = 0.0d0
 152      pco2min = 0.0d0
 153      pco2max = 0.0d0
 154      night_ang = 0.0d0
 155      crisCloudFractionThreshold = -1.d0
 156
 157      ! read the namelist
 158      nulnam = 0
 159      ierr = fnom(nulnam, './flnml','FTN+SEQ+R/O', 0)
 160      read(nulnam, nml=NAMBGCKIR, iostat=ierr)
 161      if (ierr /= 0) call utl_abort('irbg_init: Error reading namelist')
 162      if (mmpi_myid == 0) write(*, nml=NAMBGCKIR)
 163      if (ninst /= MPC_missingValue_INT) then
 164        call utl_abort('irbg_init: check NAMBGCKIR namelist section: ninst should be removed')
 165      end if
 166      ninst = 0
 167      do instrumentIndex = 1, nmaxinst
 168        if (inst(instrumentIndex) == '*EMPTY*') exit
 169        ninst = ninst + 1
 170      end do
 171      ierr = fclos(nulnam)
 172      first = .false.
 173
 174    end if
 175
 176  end subroutine irbg_init

 182    !
 183    !:Purpose: Memory allocation for the Hyperspectral Infrared

 186    implicit none
 187
 188    ! Locals:
 189    integer :: allocStatus(2)
 190    integer :: tovsIndex, maxChannelNumber
 191    integer :: sensorIndex, channelNumber

 194    !     Memory allocation for background check related variables
 195    allocStatus(:) = 0
 196    allocate( tvs_surfaceParameters(tvs_nobtov), stat=allocStatus(1))
 197    call utl_checkAllocationStatus(allocStatus(1:1), " irbg_setup tvs_surfaceParameters")
 198

 202    do tovsIndex = 1, tvs_nobtov
 203      sensorIndex = tvs_lsensor(tovsIndex)

 224  subroutine irbg_bgCheckIR(columnTrlOnTrlLev, obsSpaceData)
 225    !
 226    !:Purpose: Do background check on all hyperspectral infrared observations
 227    !
 228    implicit none
 229
 230    ! Arguments:
 231    type(struct_columnData) :: columnTrlOnTrlLev
 232    type(struct_obs)        :: obsSpaceData
 233   
 234    ! Locals:
 235    integer,allocatable :: nobir(:)
 236    integer             :: headerIndex, idatyp, sensorIndex, instrumentIndex
 237    logical             :: irDataPresent
 238
 239    call utl_tmg_start(115,'--BgckInfrared')
 240
 241    irDataPresent = .false.
 242    call obs_set_current_header_list(obsSpaceData,'TO')
 243    HEADER0: do
 244      headerIndex = obs_getHeaderIndex(obsSpaceData)
 245      if (headerIndex < 0) exit HEADER0
 246      idatyp = obs_headElem_i(obsSpaceData, OBS_ITY, headerIndex)
 247      if ( tvs_isIdBurpHyperSpectral(idatyp) ) then
 248        irDataPresent = .true.
 249      end if
 250    end do HEADER0
 251
 252    if ( .not. irDataPresent ) then
 253      write(*,*) 'WARNING: WILL NOT RUN irbg_bgCheckIR since no IR'
 254      return
 255    end if
 256
 257    write(*,'(A)') " ****************"
 258    write(*,'(A)') " BEGIN IR BACKGROUND CHECK"
 259    write(*,'(A)') " **************** **************** ****************"
 260
 261    !  Preliminary initializations
 262
 263    tvs_nobtov = 0
 264    call irbg_init()
 265    allocate (nobir(ninst))
 266    nobir = 0
 267  
 268
 269    ! Loop over all header indices of the 'TO' family
 270    ! Set the header list (and start at the beginning of the list)
 271    call obs_set_current_header_list(obsSpaceData,'TO')
 272    HEADER: do
 273      headerIndex = obs_getHeaderIndex(obsSpaceData)
 274      if (headerIndex < 0) exit HEADER
 275      
 276      idatyp = obs_headElem_i(obsSpaceData,OBS_ITY,headerIndex)
 277
 278      if ( .not. tvs_isIdBurpTovs(idatyp) ) then
 279        write(*,*) 'irbg_bgCheckIR: warning unknown radiance codtyp present check NAMTOVSINST', idatyp
 280        cycle HEADER   ! Proceed to the next header_index
 281      end if
 282      tvs_nobtov = tvs_nobtov + 1
 283      do instrumentIndex=1, ninst
 284        if ( tvs_isIdBurpInst(idatyp,inst(instrumentIndex)) ) then
 285          nobir(instrumentIndex) = nobir(instrumentIndex) + 1
 286          exit
 287        end if
 288      end do
 289    end do HEADER
 290    
 291    do instrumentIndex=1, ninst
 292      if (nobir(instrumentIndex) > 0) then
 293        do sensorIndex=1,tvs_nsensors
 294          if (tvs_instruments(sensorIndex) ==  tvs_getInstrumentId(inst(instrumentIndex)) ) then
 295            call irbg_doQualityControl (columnTrlOnTrlLev, obsSpaceData, inst(instrumentIndex), sensorIndex)
 296          end if
 297        end do
 298      end if
 299    end do
 300    deallocate (nobir)
 301
 302    call utl_tmg_stop(115)
 303
 304  end subroutine irbg_bgCheckIR

1268  ! calcreflect
1269  !--------------------------------------------------------------------------
1270  subroutine calcreflect(rad, sunzen, reflect)
1271    !
1272    !:Purpose: Computes Top of Atmosphere Albedo as defined by equation (4)
1273    !           of Rao et al. Int. J. of Remote Sensing, 2003, vol 24, no 9, 1913-1924
1274    !           
1275    implicit none
1276    ! Arguments:
1277    real(8), intent(in) :: rad(nvis)     ! radiances array
1278    real(8), intent(in) :: sunzen        ! Sun zenith angle
1279    real(8), intent(out):: reflect(nvis) ! TOA albedo en %
1280
1281    ! Locals:
1282    real(8),parameter :: solar_filtered_irradiance(nvis) = (/139.873215d0,232.919556d0,14.016470d0/)
1283    ! equivalent widths, integrated solar irradiance,  effective central wavelength
1284    !0.084877,139.873215,0.632815
1285    !0.229421,232.919556,0.841679
1286    !0.056998,14.016470,1.606119
1287    real(8) :: radb ! radiance en W/m2/str
1288    integer :: i
1289    ! *************************************************************
1290
1291    do i = 1, nvis
1292      if (rad(i) >= 0.0d0 ) then
1293        radb = rad(i) / 1000.0d0
1294        reflect(i) = (MPC_PI_R8 * radb) / solar_filtered_irradiance(i)
1295        if (sunzen < 90.0d0 ) reflect(i) = reflect(i) / cos(sunzen * MPC_RADIANS_PER_DEGREE_R8)
1296      else
1297        reflect(i) = -1
1298      end if
1299    end do
1300  
1301  end subroutine calcreflect

1304  ! calcbt
1305  !--------------------------------------------------------------------------
1306  subroutine calcbt(rad, tb, dtbsdrad, freq, offset, slope)
1307    !
1308    !:Purpose: Computes brightness temperature (bt) and the first derivative of
1309    !            bt with respect to radiance, from radiance, frequencies
1310    !           
1311    !
1312    implicit none
1313
1314    ! Arguments:
1315    real(8), intent(in) :: rad(:)              ! Radiance
1316    real(8), intent(in) :: freq(size(rad))     ! Channel wavenumber (cm-1)
1317    real(8), intent(in) :: offset(size(rad))   ! 
1318    real(8), intent(in) :: slope(size(rad))    !
1319    real(8), intent(out):: tb(size(rad))       ! Brightness Temperature
1320    real(8), intent(out):: dtbsdrad(size(rad)) ! Derivative of tb wrt radiance
1321
1322    ! Locals:
1323    integer  :: channelIndex, nchan
1324    real(8)  :: radtotal, tstore, planck1, planck2
1325    real(8), parameter :: c1= 1.19106590d-05   ! First Planck constant
1326    real(8), parameter :: c2= 1.438833d0       ! Second Planck constant 
1327
1328
1329    nchan = size(rad)
1330
1331    do channelIndex = 1, nchan
1332      if (rad(channelIndex) > 1.d-20) then
1333        planck2 = c2 * freq(channelIndex)
1334        planck1 = c1 * ( freq(channelIndex) ** 3 ) 
1335        tstore = planck2 / log( 1.0d0 + planck1 / rad(channelIndex) )
1336        tb(channelIndex) = ( tstore - offset(channelIndex) ) / slope(channelIndex)
1337        
1338        radtotal = rad(channelIndex)
1339        
1340        dtbsdrad(channelIndex) = planck1 * tstore ** 2 / ( planck2 * radtotal * ( radtotal + planck1 ) )
1341        
1342        dtbsdrad(channelIndex) = dtbsdrad(channelIndex) / slope(channelIndex)
1343        
1344      else
1345        tb(channelIndex) = 0.d0
1346        dtbsdrad(channelIndex) = 0.d0
1347      end if
1348      
1349    end do
1350
1351  end subroutine calcbt

1383          end if

1411        if (sumBt2(channelIndex) > 0.d0) then
1412          sumBt2(channelIndex)= sqrt ( sumBt2(channelIndex) )
1413        else
1414          sumBt2(channelIndex) = 0.d0
1415        end if
1416      end if
1417    end do

1459    integer             :: jc_pair
1460    integer             :: iter,niter
1461    real(8), parameter  :: eps = 1.D-12
1462
1463    ptop(:) = -1.d0
1464    ntop(:) = -1
1465    fcloud(:) = -1.d0
1466
1467    !  Profile not assimilated if data from 2 windows channels bad
1468    !  and/or if data from 2 reference co2 channels bad
1469
1470    if ( cldflag == -1 ) return
1471
1472    nlev = size(plev)
1473    nchn = size(rcal)
1474
1475    !  Define closest level jpmax to surface pressure p0
1476
1477    jpmax = nlev
1478    
1479    do J = lev_start, nlev
1480      if ( plev(J) > p0 ) then
1481        jpmax = J
1482        exit
1483      end if
1484    end do
1485    
1486    !     define jmax as last level for co2-slicing calculations
1487  
1488    jmax = jpmax - 1
1489    
1490    !     predetermined clear window channel, all nco2 estimates clear
1491
1492    sumrej = sum(rejflag(ichref,:))
1493
1494    if ( sumrej == 0 ) then
1495      ptop(:) = p0
1496      ntop(:) = jpmax
1497      fcloud(:) = 0.d0
1498      return
1499    end if
1500
1501    allocate(fc(nchn,nlev), drap(nco2,nlev), a_drap(nlev) )
1502
1503    channels: do jch = 1, nco2
1504      
1505      jc = ilist(jch)
1506      jc_pair = ilist_pair(jch)
1507      fc(jc_pair,:) = rcal(jc_pair) - cloudyRadiance(jc_pair,:)
1508      niter = 1
1509      if ( jch > 13) niter = 2 
1510     
1511      iteration: do iter = 1, niter
1512        drap(jch,:)   = 9999.d0
1513        ntop(jch) = -1
1514        !         calcul emi_ratio
1515        if (jch > 13) then       
1516          if ( iter == 1 ) then
1517            emi_ratio = 1.0376d0
1518          else
1519            emi_ratio = 1.09961d0 - 0.09082d0 * fcloud(jch)
1520          end if
1521        else
1522          emi_ratio = 1.0d0
1523        end if
1524
1525        fc(jc,:) = rcal(jc) - cloudyRadiance(jc,:)
1526
1527        !      Gross check failure
1528
1529        if ( rejflag(jc,9) == 1 ) cycle channels
1530        if ( rejflag(jc_pair,9) == 1 ) cycle channels
1531
1532        if ( abs( rcal(jc_pair) - radObs(jc_pair) ) > eps ) then
1533          rapg = (rcal(jc) - radObs(jc)) / (rcal(jc_pair) - radObs(jc_pair))
1534        else
1535          rapg = 0.0d0
1536        end if
1537
1538        do J = lev_start, jpmax
1539          if ( fc(jc,J) > 0.d0 .and. fc(jc_pair,J) > 0.d0 )  &
1540               drap(jch,J) = rapg - (fc(jc,J) / fc(jc_pair,J)) * emi_ratio
1541        end do
1542
1543        a_drap(:) = abs( drap(jch,:) )
1544
1545        levels: do J = lev_start + 1, jmax
1546
1547          ! *         do not allow fc negative (i.e. drap(jch,j) = 9999.)
1548
1549          if ( drap(jch,J) > 9000.d0 .and. &
1550               a_drap(J-1) < eps .and. &
1551               a_drap(J+1) < eps ) cycle channels
1552
1553          val = drap(jch,J) / drap(jch,J - 1)
1554
1555          ! *         find first, hopefully unique, zero crossing
1556
1557          if ( val < 0.d0 ) then
1558
1559            ! *         conditions near zero crossing of isolated minimum need monotonically
1560            ! *         decreasing drap from j-3 to j-1 as well increasing from j to j+1
1561
1562            val1 = drap(jch,J - 2) / drap(jch,J - 1)
1563            val2 = drap(jch,J - 3) / drap(jch,J - 1)
1564            val3 = drap(jch,J) / drap(jch,J + 1)
1565
1566            if ( val1 > 0.d0 .and.  & 
1567                 val2 > 0.d0 .and.  & 
1568                 val3 > 0.d0 .and.  &
1569                 a_drap(J-2) > a_drap(J-1) .and.  &
1570                 a_drap(J-3) > a_drap(J-2) .and.  &
1571                 a_drap(J)   < 9000.d0     .and.  &
1572                 a_drap(J+1) > a_drap(J) )        &
1573                 then
1574              ptop(jch) = plev(J)
1575              ntop(jch) = J
1576            end if
1577            
1578            exit levels
1579                      
1580          end if
1581              
1582        end do levels
1583
1584        j = ntop(jch)
1585
1586        ! *       special cases of no determination
1587
1588        if ( j < 1) then
1589          ptop(jch)   = -1.d0
1590          ntop(jch)   = -1
1591          fcloud(jch) = -1.d0
1592          cycle channels
1593        end if
1594      
1595        if ( J <= lev_start .or. drap(jch,J) > 9000.d0 ) then
1596          !if ( iter == 1) then
1597          ptop(jch) = -1.d0
1598          ntop(jch) = -1
1599          fcloud(jch) = -1.d0
1600          !end if
1601          cycle channels
1602        end if
1603        
1604        if ( abs( cloudyRadiance(jc,J) - rcal(jc) ) > 0.d0 )  &
1605             fcloud(jch) = (radObs(jc) - rcal(jc)) /  &
1606             (cloudyRadiance(jc,J) - rcal(jc))
1607
1608        ! *       find passage to zero if it exists and interpolate to exact pressure
1609
1610        ptop(jch) = plev(J - 1) - drap(jch,J - 1) /    &
1611             ( drap(jch,J) - drap(jch,J-1) ) * ( plev(J) - plev(J-1) )
1612        ! *       find cloud radiance at zero crossing to use to get cloud fraction
1613
1614        fcint = fc(jc,J - 1) + ( fc(jc,J) - fc(jc,J - 1) ) /   &
1615             ( plev(J) - plev(J - 1) ) * ( ptop(jch) - plev(J - 1) )
1616
1617        ! *       find cloud fraction based on exact cloud top
1618
1619        if ( abs(fcint) > 0.d0 )             &
1620             fcloud(jch) = ( rcal(jc) - radObs(jc) ) / fcint
1621
1622        fcloud(jch) = min ( fcloud(jch),  1.5d0 )
1623        fcloud(jch) = max ( fcloud(jch), -0.5d0 )
1624
1625        if (fcloud(jch) < 0.0d0 .or. fcloud(jch) > 1.0d0 )  cycle channels
1626      
1627      end do iteration
1628     
1629    end do channels
1630
1631    deallocate(fc, drap, a_drap )
1632      
1633  end subroutine co2_slicing

1637  !--------------------------------------------------------------------------
1638  subroutine seltop (etop, vtop, ecf, vcf, ngood, he, ht, cf, cfsub, ptop_mb, p0, cldflag, gncldflag)
1639    !
1640    !:Purpose: Select cloud top by averaging co2-slicing results
1641    !           judged correct. all missing values are -1.
1642    !
1643    implicit none
1644
1645    ! Arguments:
1646    real(8), intent(in)  :: he       ! equivalent cloud top heights from a window channel (hPa)
1647    real(8), intent(in)  :: ht(nco2) ! cloud tops from co2-slicing (hPa)
1648    real(8), intent(in)  :: cf(nco2) ! effective cloud fraction for co2-slicing
1649    real(8), intent(in)  :: p0       ! surface pressure in (hPa)
1650    real(8), intent(in)  :: cfsub    ! visible ("subpixel") cloud fraction
1651    integer, intent(in)  :: cldflag  ! Cloudy flag (0 Clear, 1 Cloudy, -1 undefined)
1652    integer, intent(in)  :: gncldflag! Garand Nadon cloudy flag (0 Clear, 1 Cloudy, -1 undefined)
1653    real(8), intent(out) :: etop     ! consensus cloud top (hPa)
1654    real(8), intent(out) :: vtop     ! corresponding variance on etop (hPa)
1655    real(8), intent(out) :: ecf      ! consensus effective cloud fraction
1656    real(8), intent(out) :: vcf      ! corresponding variance on ecf
1657    integer, intent(out) :: ngood    ! number of good estimates
1658    real(8), intent(in)  :: ptop_mb  ! height (mb) from cloud_height subroutine
1659
1660    ! Locals:
1661    integer    :: n,jch
1662    real(8)    :: H(nco2),F(nco2)
1663
1664
1665    etop = -1.d0
1666    vtop = -1.d0
1667    ecf  = -1.d0
1668    vcf  = -1.d0
1669    ngood = 0
1670
1671    !     Profile not assimilated if data from 2 windows channels bad
1672    !     and/or if data from 2 reference co2 channels bad    
1673    if ( cldflag == -1 ) return
1674
1675    n = 0
1676    H(:) = 0.d0
1677    F(:) = 0.d0
1678
1679    do jch = 1, nco2
1680
1681      !     Check for zero cloud fraction
1682
1683      if ( CF(jch) > -0.9d0 .and. CF(jch) < 1.D-6 ) then
1684        n = n + 1
1685        H(n) = p0
1686        F(n) = 0.d0
1687      else
1688
1689
1690        !       Consider only valid values of cloud fraction above some threshold
1691     
1692        !       Important logic: for values above 1.0 of co2-slicing cloud fraction,
1693        !       set it to 1.0 and force the top equal to the effective height he.
1694        !       co2-slicing not allowed to give estimates below he, which happens
1695        !        for cloud fraction cf > 1.0.
1696
1697        if ( ht(jch) > 0.0d0 ) then
1698          n = n + 1
1699          H(n) = ht(jch)
1700          F(n) = min(CF(jch), 1.0d0)
1701          F(n) = max(F(n), 0.d0)
1702          if ( CF(jch) > 1.0d0 ) H(n) = he
1703        end if
1704      end if
1705
1706    end do
1707
1708
1709    ngood = n
1710
1711    !     Compute mean and variance
1712
1713    if ( n >= 1 ) then
1714
1715      call calcul_median_fast(n,H,F,etop,ecf)
1716    
1717      vtop = sqrt ( sum((H(1:n) - etop)**2) / n )
1718      vcf  = sqrt ( sum((F(1:n) - ecf)**2) / n )         
1719
1720      if ( n == 1 ) then
1721        vtop = 50.d0
1722        vcf  = 0.20d0
1723      end if
1724       
1725    else
1726
1727      !    If no solution from co2-slicing, and not predetermined clear, 
1728      !    assume cloudy with top equal to effective height he;
1729      !    however if he is very close to surface pressure p0, assume clear.
1730
1731      etop = he
1732      ecf  = 1.0d0
1733      if (cfsub >= 0.05d0) then
1734        ecf = cfsub
1735        etop = min( min(he,ptop_mb) , p0 - 50.0d0)
1736      end if
1737      vtop = 50.d0
1738      vcf  = 0.30d0
1739      if ( he > (p0 - 10.d0) ) ecf = 0.d0
1740      if ( gncldflag == 0 ) then
1741        ecf = 0.0d0
1742        etop = p0
1743      end if
1744    end if
1745
1746    if ( ecf < 0.05d0 ) then
1747      ecf = 0.0d0
1748      etop = p0
1749    end if
1750  
1751  end subroutine seltop

1794  !--------------------------------------------------------------------------
1795  subroutine min_pres_new( maxheight, minp, pmin, dt1, p0, tau, plev, cldflag, &
1796                           modelTopIndex )
1797    !
1798    !:Purpose: from total transmittance array, find minimum height 
1799    !           level of sensitivity for a number of profiles and channels.
1800    !           this may be used to select for assimilation only the
1801    !           observations without sensitivity to clouds, that is the
1802    !           response function significant only above cloud level.
1803    !           the criterion is that dtau/dplev > 0.01 for a 100 mb layer.
1804    !
1805    !
1806    implicit none
1807
1808    ! Arguments:
1809    real(8), intent(out)  :: maxHeight(:)                    ! Height (hPa) of the maximum of the weighting function
1810    integer, intent(out)  :: minp(size(maxHeight))           ! vertical level corresponding to pmin
1811    real(8), intent(out)  :: pmin(size(maxHeight))           ! minimum height of sensitivity (hPa)
1812    real(8), intent(out)  :: dt1(size(maxHeight))            ! value of 'dtau/dlogp' at model top
1813    real(8), intent(in)   :: p0                              ! surface pressure (hPa)
1814    real(8), intent(in)   :: plev(:)                         ! pressure levels (hPa)
1815    real(8), intent(in)   :: tau(size(maxHeight),size(plev)) ! layer to space transmittances (0.-1.)
1816    integer, intent(in)   :: cldflag                         ! Cloudy flag (0 Clear, 1 Cloudy, -1 undefined)
1817    integer, intent(in)   :: modelTopIndex                   ! rt model level nearest to model top
1818
1819    ! Locals:   
1820    real(8) :: maxwf
1821    integer :: levelIndex, channelIndex, ipos(1), nlev, nchn
1822    real(8),allocatable :: wfunc(:), rap(:)
1823
1824    minp(:) = -1
1825    pmin(:) = -1.d0
1826    dt1(:)  = -1.d0
1827
1828    if ( cldflag == -1 ) return
1829
1830    nlev = size(plev)
1831    nchn = size(maxHeight)
1832
1833    allocate( wfunc(nlev-1), rap(nlev-1) )
1834
1835    do levelIndex = 1, nlev - 1
1836      rap(levelIndex) = log( plev(levelIndex + 1) / plev(levelIndex) )
1837    end do
1838
1839    channels: do channelIndex = 1, nchn
1840
1841      !       Profile not assimilated if data from 2 windows channels bad
1842      !       and/or if data from 2 reference co2 channels bad
1843    
1844      do levelIndex = 1, nlev
1845        if ( tau(channelIndex,levelIndex) < 0.d0) cycle channels
1846      end do
1847
1848      minp(channelIndex) = nlev
1849      pmin(channelIndex) = min(plev(nlev),p0)
1850
1851      !   Compute entire array of dtau/dlog(P)
1852          
1853      do levelIndex = 1, nlev - 1
1854        wfunc(levelIndex) = (tau(channelIndex,levelIndex) - tau(channelIndex,levelIndex + 1)) / rap(levelIndex) 
1855      end do
1856       
1857      dt1(channelIndex) = wfunc(modelTopIndex)
1858
1859      !   If channel sees the surface, don't recalculate minp and pmin
1860
1861      if ( tau(channelIndex,nlev) > 0.01d0 ) cycle channels
1862
1863      ! Recherche du maximum
1864      ipos = maxloc( wfunc(:) )
1865      ! Calcul de la valeur du maximum
1866      maxwf = wfunc(ipos(1))
1867      ! maximum entre les 2 niveaux puisque WF calculee pour une couche finie ( discutable ?)
1868      maxheight(channelIndex)= 0.5d0 * ( plev(ipos(1)) +  plev(ipos(1) + 1)  )
1869
1870      !      If channel doesn't see the surface, see where dtau/dlog(plev) becomes important
1871      !      for recomputation of minp and pmin.
1872
1873      do levelIndex = nlev - 1, ipos(1), -1
1874        if ( ( wfunc(levelIndex)/ maxwf ) > 0.01d0) then
1875          minp(channelIndex) = levelIndex + 1
1876          pmin(channelIndex) = min(plev(levelIndex + 1),p0)
1877          exit
1878        end if
1879      end do
1880     
1881    end do channels
1882
1883    deallocate(wfunc, rap )
1884
1885  end subroutine min_pres_new

1973    real(8), intent(in)    :: ptop_eq         ! Chosen equivalent cloud tops (m)

2012          cldflag=1
2013          return
2014        else
2015          cldflag=0
2016          return
2017        end if
2018      else

2020        if ( ninv == 1 ) then
2021          if ( ptop_eq > 222.d0 ) then
2022            cldflag = 1
2023            return
2024          else
2025            cldflag = 0 
2026            return
2027          end if
2028        else
2029          cldflag = 0
2030          return
2031        end if
2032      end if
2033    end if
2034    
2035    if ( tg - btObs(ichref) > 5.d0 ) then
2036      if ( ntop_eq >= 3 ) then
2037        if ( ptop_eq > 222.d0 ) then
2038          cldflag = 1
2039          return
2040        else
2041          cldflag = 0
2042          return
2043        end if
2044      else

2046        if ( ninv == 1) then

2048            cldflag = 1
2049            return
2050          else
2051            cldflag = 0
2052            return
2053          end if
2054        else
2055          cldflag = 0
2056        end if
2057      end if
2058    else
2059      cldflag = 0
2060    end if

2112    !           radiative transfer equation assuming guess t and q profiles

2114    !           profiles. assumes a real tg (guess) over oceans and a tg 
2115    !           with hypothesis of unity emissivity over land.
2116    !

2118    !         ts = B(ts)*EMI*SFCtau + ATMOS_PART
2119    !         SOLVES FOR ts
2120    !
2121    implicit none
2122
2123    ! Arguments:
2124    integer, intent(in)           :: ichref            ! Reference surface channel (subset values)
2125    integer, intent(in)           :: cldflag           ! Cloudy flag (0 Clear, 1 Cloudy, -1 undefined)
2126    real(8), intent(in)           :: tg                ! Guess skin temperature (K)
2127    real(8), intent(in)           :: emi(:)            ! Surface emissivities from window channel (0.-1.)
2128    real(8), intent(in)           :: rcal(size(emi))   ! Computed clear radiances (mW/m2/sr/cm-1)
2129    real(8), intent(in)           :: radobs(size(emi)) ! Observed radiances (mW/m2/sr/cm-1)
2130    real(8), intent(in)           :: sfctau(size(emi)) ! Surface to space transmittances (0.-1.)
2131    real(8), intent(out)          :: ts                ! Retrieved skin temperature (-1. for missing)
2132    type(rttov_coefs), intent(in) :: myCoefs           ! RTTOV coefficients structure
2133
2134    ! Locals:
2135    real(8)    :: rtg,radtg
2136    real(8)    :: radts,tstore,t_effective
2137  
2138    ts = -1.d0
2139
2140    if ( cldflag /= 0 ) return
2141    if ( ichref == -1 ) return
2142
2143
2144    !   Transform guess skin temperature to plank radiances 
2145
2146    t_effective =  myCoefs % coef % ff_bco(ichref) + myCoefs % coef % ff_bcs(ichref) * TG
2147
2148    radtg =  myCoefs % coef % planck1(ichref) / &
2149         ( exp( myCoefs % coef % planck2(ichref) / t_effective ) - 1.0d0 )

2157    !  Compute true skin planck radiances due to TOA true planck radiances

2168
2169    
2170    !  Transform true skin planck radiances to true skin temperatures
2171
2172    tstore = myCoefs % coef % planck2(ichref) / log( 1.0d0 + myCoefs % coef % planck1(ichref) / radts )
2173
2174    ts = ( tstore - myCoefs % coef % ff_bco(ichref) ) / myCoefs % coef % ff_bcs(ichref)
