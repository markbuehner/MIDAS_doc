17module obsFamilyList_mod
18  ! MODULE varNameList (prefix='ofl' category='7. Low-level data objects')
19  !
20  !:Purpose: Contains a list of all recognizable observation families along with 
21  !           additional information and procedures regarding these families.
22  !
23  
24  implicit none
25  save
26  private
27
28  ! public variables (parameters)
29  public :: ofl_numFamily
30  public :: ofl_familyList
31
32  ! public procedures
33  public :: ofl_isFamilyTypeInList
34
35  integer,          parameter :: ofl_numFamily = 15
36  character(len=2), parameter :: ofl_familyList(ofl_numFamily)= (/ &
37                                 'UA','AI','SF','SC','SW','PR','RO','GP','RA', &
38                                 'TO','CH','TM','AL','GL','HY'/)
39
40  ! Description of obs family types
41  ! -------------------------------
42  ! UA - radiosondes (RAOBS and SURFACE)
43  ! AI - aircraft measurements (AIREPS)
44  ! SF - surface obs (SURFACE)
45  ! SC - scatterometer measurements (SURFACE)
46  ! SW - winds from satellite measurements (SATWINDS) 
47  ! PR - ground-based profiler data 
48  ! RO - radio occultation data (GPS-RO; TT/HU/P0)
49  ! GP - ground-based GPS data (GB-GPS)
50  ! RA - radar precipitation data
51  ! TO - brightness temperatures (TT/HU/P0/constituents)
52  ! CH - retrieved chemical constituent data
53  ! TM - Sea-surface temperature data (SST)
54  ! AL - Aladin lidar horizontal wind data
55  ! GL - Sea-ice concentration data 
56  ! HY - Hydrological data
57  !
58  ! NWP obs families having data as function of pressure level: UA,AI,SW 
59  ! NWP obs families contributing to the surface NWP obs group: UA,SF,SC,GP,RA
60  ! NWP obs families having data as a function of altitude: PR,AL
61                                                   
62  contains
63
64    !--------------------------------------------------------------------------
65    ! ofl_isFamilyTypeInList
66    !--------------------------------------------------------------------------
67    function ofl_isFamilyTypeInList(familyName) result(familyFound)
68      !
69      !:Purpose: To identify if input obs family is part of the available list 
70      !
71
72      implicit none
73
74      !Arguments:
75      character(len=2) :: familyName
76      logical          :: familyFound
77
78      if ( any(ofl_familyList(:) == familyName) ) then
79         familyFound = .true.
80      else
81         familyFound = .false.
82      end if
83      
84    end function ofl_isFamilyTypeInList
85
86end module obsFamilyList_mod
