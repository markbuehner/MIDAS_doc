  28module kdtree2_priority_queue_mod
  29  use utilities_mod
  30  use kdtree2_precision_mod
  31  !
  32  ! maintain a priority queue (PQ) of data, pairs of 'priority/payload', 
  33  ! implemented with a binary heap.  This is the type, and the 'dis' field
  34  ! is the priority.
  35  !
  36  type kdtree2_result
  37      ! a pair of distances, indexes
  38      real(kdkind)    :: dis!=0.0
  39      integer :: idx!=-1   Initializers cause some bugs in compilers.
  40  end type kdtree2_result
  41  !
  42  ! A heap-based priority queue lets one efficiently implement the following
  43  ! operations, each in log(N) time, as opposed to linear time.
  44  !
  45  ! 1)  add a datum (push a datum onto the queue, increasing its length) 
  46  ! 2)  return the priority value of the maximum priority element 
  47  ! 3)  pop-off (and delete) the element with the maximum priority, decreasing
  48  !     the size of the queue. 
  49  ! 4)  replace the datum with the maximum priority with a supplied datum
  50  !     (of either higher or lower priority), maintaining the size of the
  51  !     queue. 
  52  !
  53  !
  54  ! In the k-d tree case, the 'priority' is the square distance of a point in
  55  ! the data set to a reference point.   The goal is to keep the smallest M
  56  ! distances to a reference point.  The tree algorithm searches terminal
  57  ! nodes to decide whether to add points under consideration.
  58  !
  59  ! A priority queue is useful here because it lets one quickly return the
  60  ! largest distance currently existing in the list.  If a new candidate
  61  ! distance is smaller than this, then the new candidate ought to replace
  62  ! the old candidate.  In priority queue terms, this means removing the
  63  ! highest priority element, and inserting the new one.
  64  !
  65  ! Algorithms based on Cormen, Leiserson, Rivest, _Introduction
  66  ! to Algorithms_, 1990, with further optimization by the author.
  67  !
  68  ! Originally informed by a C implementation by Sriranga Veeraraghavan.
  69  !
  70  ! This module is not written in the most clear way, but is implemented such
  71  ! for speed, as it its operations will be called many times during searches
  72  ! of large numbers of neighbors.
  73  !
  74  type pq
  75      !
  76      ! The priority queue consists of elements
  77      ! priority(1:heap_size), with associated payload(:).
  78      !
  79      ! There are heap_size active elements. 
  80      ! Assumes the allocation is always sufficient.  Will NOT increase it
  81      ! to match.
  82      integer :: heap_size = 0
  83      type(kdtree2_result), pointer :: elems(:) 
  84  end type pq

 127  subroutine heapify(a,i_in)
 128    !:Purpose: take a heap rooted at 'i' and force it to be in the
 129    !           heap canonical form.   This is performance critical 
 130    !           and has been tweaked a little to reflect this.
 131    !
 132    type(pq),pointer   :: a
 133    integer, intent(in) :: i_in
 134    !
 135    integer :: i, l, r, largest
 136
 137    real(kdkind)    :: pri_i, pri_l, pri_r, pri_largest
 138
 139
 140    type(kdtree2_result) :: temp
 141
 142    i = i_in
 143
 144bigloop:  do
 145       l = 2*i ! left(i)
 146       r = l+1 ! right(i)
 147       ! 
 148       ! set 'largest' to the index of either i, l, r
 149       ! depending on whose priority is largest.
 150       !
 151       ! note that l or r can be larger than the heap size
 152       ! in which case they do not count.
 153
 154
 155       ! does left child have higher priority? 
 156       if (l .gt. a%heap_size) then
 157          ! we know that i is the largest as both l and r are invalid.
 158          exit 
 159       else
 160          pri_i = a%elems(i)%dis
 161          pri_l = a%elems(l)%dis 
 162          if (pri_l .gt. pri_i) then
 163             largest = l
 164             pri_largest = pri_l
 165          else
 166             largest = i
 167             pri_largest = pri_i
 168          endif
 169
 170          !
 171          ! between i and l we have a winner
 172          ! now choose between that and r.
 173          !
 174          if (r .le. a%heap_size) then
 175             pri_r = a%elems(r)%dis
 176             if (pri_r .gt. pri_largest) then
 177                largest = r
 178             endif
 179          endif
 180       endif
 181
 182       if (largest .ne. i) then
 183          ! swap data in nodes largest and i, then heapify
 184
 185          temp = a%elems(i)
 186          a%elems(i) = a%elems(largest)
 187          a%elems(largest) = temp 
 188          ! 
 189          ! Canonical heapify() algorithm has tail-ecursive call: 
 190          !
 191          !        call heapify(a,largest)   
 192          ! we will simulate with cycle
 193          !
 194          i = largest
 195          cycle bigloop ! continue the loop 
 196       else
 197          return   ! break from the loop
 198       end if
 199    enddo bigloop
 200    return
 201  end subroutine heapify

 230  subroutine pq_extract_max(a,e)
 231    !:Purpose: return the priority and payload of maximum priority
 232    !           element, and remove it from the queue.
 233    !           (equivalent to 'pop()' on a stack)
 234    !
 235    type(pq),pointer :: a
 236    type(kdtree2_result), intent(out) :: e
 237    
 238    if (a%heap_size .ge. 1) then
 239       !
 240       ! return max as first element
 241       !
 242       e = a%elems(1) 
 243       
 244       !
 245       ! move last element to first
 246       !
 247       a%elems(1) = a%elems(a%heap_size) 
 248       a%heap_size = a%heap_size-1
 249       call heapify(a,1)
 250       return
 251    else
 252       call utl_abort('kdtree2_mod-PQ_EXTRACT_MAX: error, attempted to pop non-positive PQ')
 253    end if

 258  real(kdkind) function pq_insert(a,dis,idx) 
 259    !:Purpose: Insert a new element and return the new maximum priority,
 260    !           which may or may not be the same as the old maximum priority.
 261    !
 262    type(pq),pointer  :: a
 263    real(kdkind), intent(in) :: dis
 264    integer, intent(in) :: idx
 265    !    type(kdtree2_result), intent(in) :: e
 266    !
 267    integer :: i, isparent
 268    real(kdkind)    :: parentdis
 269    !
 270
 271    !    if (a%heap_size .ge. a%max_elems) then
 272    !       write (*,*) 'PQ_INSERT: error, attempt made to insert element on full PQ'
 273    !       stop
 274    !    else
 275    a%heap_size = a%heap_size + 1
 276    i = a%heap_size
 277
 278    do while (i .gt. 1)
 279       isparent = int(i/2)
 280       parentdis = a%elems(isparent)%dis
 281       if (dis .gt. parentdis) then
 282          ! move what was in i's parent into i.
 283          a%elems(i)%dis = parentdis
 284          a%elems(i)%idx = a%elems(isparent)%idx
 285          i = isparent
 286       else
 287          exit
 288       endif
 289    end do
 290
 291    ! insert the element at the determined position
 292    a%elems(i)%dis = dis
 293    a%elems(i)%idx = idx
 294
 295    pq_insert = a%elems(1)%dis 
 296    return
 297    !    end if
 298
 299  end function pq_insert

 301  subroutine pq_adjust_heap(a,i)
 302    type(pq),pointer  :: a
 303    integer, intent(in) :: i
 304    !
 305    ! nominally arguments (a,i), but specialize for a=1
 306    !
 307    ! This routine assumes that the trees with roots 2 and 3 are already heaps, i.e.
 308    ! the children of '1' are heaps.  When the procedure is completed, the
 309    ! tree rooted at 1 is a heap.
 310    real(kdkind) :: prichild
 311    integer :: parent, child, N
 312
 313    type(kdtree2_result) :: e
 314
 315    e = a%elems(i) 
 316
 317    parent = i
 318    child = 2*i
 319    N = a%heap_size
 320    
 321    do while (child .le. N)
 322       if (child .lt. N) then
 323          if (a%elems(child)%dis .lt. a%elems(child+1)%dis) then
 324             child = child+1
 325          endif
 326       endif
 327       prichild = a%elems(child)%dis
 328       if (e%dis .ge. prichild) then
 329          exit 
 330       else
 331          ! move child into parent.
 332          a%elems(parent) = a%elems(child) 
 333          parent = child
 334          child = 2*parent
 335       end if
 336    end do
 337    a%elems(parent) = e
 338    return
 339  end subroutine pq_adjust_heap

 342  real(kdkind) function pq_replace_max(a,dis,idx) 
 343    !:Purpose: Replace the extant maximum priority element
 344    !           in the PQ with (dis,idx).  Return
 345    !           the new maximum priority, which may be larger
 346    !           or smaller than the old one.
 347    !
 348    type(pq),pointer         :: a
 349    real(kdkind), intent(in) :: dis
 350    integer, intent(in) :: idx
 351!    type(kdtree2_result), intent(in) :: e
 352    ! not tested as well!
 353
 354    integer :: parent, child, N
 355    real(kdkind)    :: prichild, prichildp1
 356
 357    type(kdtree2_result) :: etmp
 358    
 359    if (.true.) then
 360       N=a%heap_size
 361       if (N .ge. 1) then
 362          parent =1
 363          child=2
 364
 365          loop: do while (child .le. N)
 366             prichild = a%elems(child)%dis
 367
 368             !
 369             ! posibly child+1 has higher priority, and if
 370             ! so, get it, and increment child.
 371             !
 372
 373             if (child .lt. N) then
 374                prichildp1 = a%elems(child+1)%dis
 375                if (prichild .lt. prichildp1) then
 376                   child = child+1
 377                   prichild = prichildp1
 378                endif
 379             endif
 380
 381             if (dis .ge. prichild) then
 382                exit loop  
 383                ! we have a proper place for our new element, 
 384                ! bigger than either children's priority.
 385             else
 386                ! move child into parent.
 387                a%elems(parent) = a%elems(child) 
 388                parent = child
 389                child = 2*parent
 390             end if
 391          end do loop
 392          a%elems(parent)%dis = dis
 393          a%elems(parent)%idx = idx
 394          pq_replace_max = a%elems(1)%dis
 395       else
 396          a%elems(1)%dis = dis
 397          a%elems(1)%idx = idx
 398          pq_replace_max = dis
 399       endif
 400    else
 401       !
 402       ! slower version using elementary pop and push operations.
 403       !
 404       call pq_extract_max(a,etmp) 
 405       etmp%dis = dis
 406       etmp%idx = idx
 407       pq_replace_max = pq_insert(a,dis,idx)
 408    endif
 409    return
 410  end function pq_replace_max

 575  function kdtree2_create(input_data,dim,sort,rearrange) result (mr)
 576    !:Purpose: Create the actual tree structure, given an input array of data.
 577
 578    ! Note, input data is input_data(1:d,1:N), NOT the other way around.
 579    ! THIS IS THE REVERSE OF THE PREVIOUS VERSION OF THIS MODULE.
 580    ! The reason for it is cache friendliness, improving performance.
 581    !
 582    ! Optional arguments:  If 'dim' is specified, then the tree
 583    !                      will only search the first 'dim' components
 584    !                      of input_data, otherwise, dim is inferred
 585    !                      from SIZE(input_data,1).
 586    !
 587    !                      if sort .eqv. .true. then output results
 588    !                      will be sorted by increasing distance.
 589    !                      default=.false., as it is faster to not sort.
 590    !                      
 591    !                      if rearrange .eqv. .true. then an internal
 592    !                      copy of the data, rearranged by terminal node,
 593    !                      will be made for cache friendliness. 
 594    !                      default=.true., as it speeds searches, but
 595    !                      building takes longer, and extra memory is used.
 596    !
 597    ! .. Function Return Cut_value ..
 598    type (kdtree2), pointer :: mr
 599    integer, intent(in), optional      :: dim
 600    logical, intent(in), optional      :: sort
 601    logical, intent(in), optional      :: rearrange
 602    ! ..
 603    ! .. Array Arguments ..
 604    real(kdkind), target :: input_data(:,:)
 605    !
 606    integer :: i, numthread, mythread
 607    integer, external :: omp_get_thread_num, omp_get_num_threads
 608    ! ..
 609    allocate (mr)
 610    mr%the_data => input_data
 611    ! pointer assignment
 612
 613    if (present(dim)) then
 614       mr%dimen = dim
 615    else
 616       mr%dimen = size(input_data,1)
 617    end if
 618    mr%n = size(input_data,2)
 619
 620    if (mr%n < 1) then
 621       !  unlikely to be correct
 622       write (*,*) 'KD_TREE_TRANS: likely user error.'
 623       write (*,*) 'KD_TREE_TRANS: You passed in matrix with D=',mr%dimen
 624       write (*,*) 'KD_TREE_TRANS: and N=',mr%n
 625       write (*,*) 'KD_TREE_TRANS: note, that new format is data(1:D,1:N)'
 626       write (*,*) 'KD_TREE_TRANS: with usually N >> D.   If N =approx= D, then a k-d tree'
 627       write (*,*) 'KD_TREE_TRANS: is not an appropriate data structure.'
 628       call utl_abort('kdtree2_create: supplied array has no data')
 629    end if
 630
 631    !$OMP PARALLEL PRIVATE(mythread)
 632    mythread = omp_get_thread_num()
 633    if ( mythread == 0 ) then
 634      numthread = omp_get_num_threads()
 635    end if
 636    !$OMP END PARALLEL
 637
 638    if ( .not. allocated(sr) ) allocate(sr(0:numthread-1))
 639
 640    call build_tree(mr)
 641
 642    if (present(sort)) then
 643       mr%sort = sort
 644    else
 645       mr%sort = .false.
 646    endif
 647
 648    if (present(rearrange)) then
 649       mr%rearrange = rearrange
 650    else
 651       mr%rearrange = .true.
 652    endif
 653
 654    if (mr%rearrange) then
 655       allocate(mr%rearranged_data(mr%dimen,mr%n))
 656       do i=1,mr%n
 657          mr%rearranged_data(:,i) = mr%the_data(:, &
 658           mr%ind(i))
 659       enddo
 660    else
 661       nullify(mr%rearranged_data)
 662    endif
 663
 664  end function kdtree2_create

 666    subroutine build_tree(tp)
 667      type (kdtree2), pointer :: tp
 668      ! ..
 669      integer :: j
 670      type(tree_node), pointer :: dummy => null()
 671      ! ..
 672      allocate (tp%ind(tp%n))
 673      forall (j=1:tp%n)

 679    recursive function build_tree_for_range(tp,l,u,parent) result (res)
 680      ! .. Function Return Cut_value ..
 681      type (tree_node), pointer :: res
 682      ! ..
 683      ! .. Structure Arguments ..
 684      type (kdtree2), pointer :: tp
 685      type (tree_node),pointer           :: parent
 686      ! ..
 687      ! .. Scalar Arguments ..
 688      integer, intent (In) :: l, u
 689      ! ..
 690      ! .. Local Scalars ..
 691      integer :: i, c, m, dimen
 692      logical :: recompute
 693      real(kdkind)    :: average
 694
 695      ! first compute min and max
 696      dimen = tp%dimen
 697      allocate (res)
 698      allocate(res%box(dimen))
 699
 700      ! First, compute an APPROXIMATE bounding box of all points associated with this node.
 701      if ( u < l ) then
 702         ! no points in this box
 703         nullify(res)
 704         return
 705      end if
 706
 707      if ((u-l)<=bucket_size) then
 708         !
 709         ! always compute true bounding box for terminal nodes.
 710         !
 711         do i=1,dimen
 712            call spread_in_coordinate(tp,i,l,u,res%box(i))
 713         end do
 714         res%cut_dim = 0
 715         res%cut_val = 0.0
 716         res%l = l
 717         res%u = u
 718         res%left =>null()
 719         res%right => null() 
 720      else
 721         ! 
 722         ! modify approximate bounding box.  This will be an
 723         ! overestimate of the true bounding box, as we are only recomputing 
 724         ! the bounding box for the dimension that the parent split on.
 725         !
 726         ! Going to a true bounding box computation would significantly
 727         ! increase the time necessary to build the tree, and usually
 728         ! has only a very small difference.  This box is not used
 729         ! for searching but only for deciding which coordinate to split on.
 730         !
 731         do i=1,dimen
 732            recompute=.true.
 733            if (associated(parent)) then
 734               if (i .ne. parent%cut_dim) then
 735                  recompute=.false.
 736               end if
 737            endif
 738            if (recompute) then
 739               call spread_in_coordinate(tp,i,l,u,res%box(i))
 740            else
 741               res%box(i) = parent%box(i)
 742            endif
 743         end do
 744         
 745
 746         c = maxloc(res%box(1:dimen)%upper-res%box(1:dimen)%lower,1)
 747         !
 748         ! c is the identity of which coordinate has the greatest spread.
 749         !
 750         
 751         if (.false.) then
 752            ! select exact median to have fully balanced tree.
 753            m = (l+u)/2
 754            call select_on_coordinate(tp%the_data,tp%ind,c,m,l,u)
 755         else
 756            !
 757            ! select point halfway between min and max, as per A. Moore,
 758            ! who says this helps in some degenerate cases, or 
 759            ! actual arithmetic average. 
 760            !
 761            if (.true.) then
 762               ! actually compute average
 763               average = sum(tp%the_data(c,tp%ind(l:u))) / real(u-l+1,kdkind)
 764            else
 765               average = (res%box(c)%upper + res%box(c)%lower)/2.0
 766            endif
 767               
 768            res%cut_val = average
 769            m = select_on_coordinate_value(tp%the_data,tp%ind,c,average,l,u)
 770         endif
 771            
 772         ! moves indexes around
 773         res%cut_dim = c
 774         res%l = l
 775         res%u = u
 776!         res%cut_val = tp%the_data(c,tp%ind(m))
 777
 778         res%left => build_tree_for_range(tp,l,m,res)
 779         res%right => build_tree_for_range(tp,m+1,u,res)
 780
 781         if (associated(res%right) .eqv. .false.) then
 782            res%box = res%left%box
 783            res%cut_val_left = res%left%box(c)%upper
 784            res%cut_val = res%cut_val_left
 785         elseif (associated(res%left) .eqv. .false.) then
 786            res%box = res%right%box
 787            res%cut_val_right = res%right%box(c)%lower
 788            res%cut_val = res%cut_val_right
 789         else
 790            res%cut_val_right = res%right%box(c)%lower
 791            res%cut_val_left = res%left%box(c)%upper
 792            res%cut_val = (res%cut_val_left + res%cut_val_right)/2
 793
 794
 795            ! now remake the true bounding box for self.  
 796            ! Since we are taking unions (in effect) of a tree structure,
 797            ! this is much faster than doing an exhaustive
 798            ! search over all points
 799            res%box%upper = max(res%left%box%upper,res%right%box%upper)
 800            res%box%lower = min(res%left%box%lower,res%right%box%lower) 
 801         endif
 802      end if
 803    end function build_tree_for_range

 805    integer function select_on_coordinate_value(v,ind,c,alpha,li,ui) &
 806     result(res)
 807      !:Purpose: Move elts of ind around between l and u, so that all points
 808      !           <= than alpha (in c cooordinate) are first, and then
 809      !           all points > alpha are second. 
 810
 811      !
 812      ! Algorithm (matt kennel). 
 813      !
 814      ! Consider the list as having three parts: on the left,
 815      ! the points known to be <= alpha.  On the right, the points
 816      ! known to be > alpha, and in the middle, the currently unknown
 817      ! points.   The algorithm is to scan the unknown points, starting
 818      ! from the left, and swapping them so that they are added to
 819      ! the left stack or the right stack, as appropriate.
 820      ! 
 821      ! The algorithm finishes when the unknown stack is empty. 
 822      !
 823      ! .. Scalar Arguments ..
 824      integer, intent (In) :: c, li, ui
 825      real(kdkind), intent(in) :: alpha
 826      ! ..
 827      real(kdkind) :: v(1:,1:)
 828      integer :: ind(1:)
 829      integer :: tmp  
 830      ! ..
 831      integer :: lb, rb
 832      !
 833      ! The points known to be <= alpha are in
 834      ! [l,lb-1]
 835      !
 836      ! The points known to be > alpha are in
 837      ! [rb+1,u].  
 838      !
 839      ! Therefore we add new points into lb or
 840      ! rb as appropriate.  When lb=rb
 841      ! we are done.  We return the location of the last point <= alpha.
 842      !
 843      ! 
 844      lb = li; rb = ui
 845
 846      do while (lb < rb)
 847         if ( v(c,ind(lb)) <= alpha ) then
 848            ! it is good where it is.
 849            lb = lb+1
 850         else
 851            ! swap it with rb.
 852            tmp = ind(lb); ind(lb) = ind(rb); ind(rb) = tmp
 853            rb = rb-1
 854         endif
 855      end do
 856      
 857      ! now lb .eq. ub 
 858      if (v(c,ind(lb)) <= alpha) then
 859         res = lb
 860      else
 861         res = lb-1
 862      endif
 863
 864      ! add a check to prevent something going wrong (Buehner)
 865      if (res < li .or. res >= ui) then
 866        !write(*,*) 'patch to handle many equal positions:',li,ui,res,v(c,ind(li:ui))
 867        res = (li + ui)/2 
 868      end if
 869      
 870    end function select_on_coordinate_value

 872    subroutine select_on_coordinate(v,ind,c,k,li,ui)
 873      !:Purpose: Move elts of ind around between l and u, so that the kth
 874      !           element is >= those below, <= those above, in the coordinate c.
 875
 876      ! .. Scalar Arguments ..
 877      integer, intent (In) :: c, k, li, ui
 878      ! ..
 879      integer :: i, l, m, s, t, u
 880      ! ..
 881      real(kdkind) :: v(:,:)
 882      integer :: ind(:)
 883      ! ..
 884      l = li
 885      u = ui
 886      do while (l<u)
 887         t = ind(l)
 888         m = l
 889         do i = l + 1, u
 890            if (v(c,ind(i))<v(c,t)) then
 891               m = m + 1
 892               s = ind(m)
 893               ind(m) = ind(i)
 894               ind(i) = s
 895            end if
 896         end do
 897         s = ind(l)
 898         ind(l) = ind(m)
 899         ind(m) = s
 900         if (m<=k) l = m + 1
 901         if (m>=k) u = m - 1
 902      end do
 903    end subroutine select_on_coordinate

 905   subroutine spread_in_coordinate(tp,c,l,u,interv) 
 906      !:Purpose: the spread in coordinate 'c', between l and u. 
 907      !           Return lower bound in 'smin', and upper in 'smax', 
 908
 909      ! ..
 910      ! .. Structure Arguments ..
 911      type (kdtree2), pointer :: tp
 912      type(interval), intent(out) :: interv
 913      ! ..
 914      ! .. Scalar Arguments ..
 915      integer, intent (In) :: c, l, u
 916      ! ..
 917      ! .. Local Scalars ..
 918      real(kdkind) :: last, lmax, lmin, t, smin,smax
 919      integer :: i, ulocal
 920      ! ..
 921      ! .. Local Arrays ..
 922      real(kdkind), pointer :: v(:,:)
 923      integer, pointer :: ind(:)
 924      ! ..
 925      v => tp%the_data(1:,1:)
 926      ind => tp%ind(1:)
 927      smin = v(c,ind(l))
 928      smax = smin
 929
 930      ulocal = u
 931
 932      do i = l + 2, ulocal, 2
 933         lmin = v(c,ind(i-1))
 934         lmax = v(c,ind(i))
 935         if (lmin>lmax) then
 936            t = lmin
 937            lmin = lmax
 938            lmax = t
 939         end if
 940         if (smin>lmin) smin = lmin
 941         if (smax<lmax) smax = lmax
 942      end do
 943      if (i==ulocal+1) then
 944         last = v(c,ind(ulocal))
 945         if (smin>last) smin = last
 946         if (smax<last) smax = last
 947      end if
 948
 949      interv%lower = smin
 950      interv%upper = smax
 951
 952    end subroutine spread_in_coordinate

1369  recursive subroutine search(node)
1370    !:Purpose: This is the innermost core routine of the kd-tree search.  Along
1371    !           with "process_terminal_node", it is the performance bottleneck. 
1372    !
1373    !           This version uses a logically complete secondary search of
1374    !           "box in bounds", whether the sear
1375    !
1376    type (Tree_node), pointer          :: node
1377    ! ..
1378    type(tree_node),pointer            :: ncloser, nfarther
1379    !
1380    integer                            :: cut_dim, i
1381    ! ..
1382    real(kdkind)                               :: qval, dis
1383    real(kdkind)                               :: ballsize
1384    real(kdkind), pointer           :: qv(:)
1385    type(interval), pointer :: box(:) 
1386    integer :: mythread
1387    integer, external :: omp_get_thread_num
1388
1389    mythread = omp_get_thread_num()
1390
1391    if ((associated(node%left) .and. associated(node%right)) .eqv. .false.) then
1392       ! we are on a terminal node
1393       if (sr(mythread)%nn .eq. 0) then
1394          call process_terminal_node_fixedball(node)
1395       else
1396          call process_terminal_node(node)
1397       endif
1398    else
1399       ! we are not on a terminal node
1400       qv => sr(mythread)%qv(1:)
1401       cut_dim = node%cut_dim
1402       qval = qv(cut_dim)
1403
1404       if (qval < node%cut_val) then
1405          ncloser => node%left
1406          nfarther => node%right
1407          dis = (node%cut_val_right - qval)**2
1408!          extra = node%cut_val - qval
1409       else
1410          ncloser => node%right
1411          nfarther => node%left
1412          dis = (node%cut_val_left - qval)**2
1413!          extra = qval- node%cut_val_left
1414       endif
1415
1416       if (associated(ncloser)) call search(ncloser)
1417
1418       ! we may need to search the second node. 
1419       if (associated(nfarther)) then
1420          ballsize = sr(mythread)%ballsize
1421!          dis=extra**2
1422          if (dis <= ballsize) then
1423             !
1424             ! we do this separately as going on the first cut dimen is often
1425             ! a good idea.
1426             ! note that if extra**2 < sr(mythread)%ballsize, then the next
1427             ! check will also be false. 
1428             !
1429             box => node%box(1:)
1430             do i=1,sr(mythread)%dimen
1431                if (i .ne. cut_dim) then
1432                   dis = dis + dis2_from_bnd(qv(i),box(i)%lower,box(i)%upper)
1433                   if (dis > ballsize) then
1434                      return
1435                   endif
1436                endif
1437             end do
1438             
1439             !
1440             ! if we are still here then we need to search mroe.
1441             !
1442             call search(nfarther)
1443          endif
1444       endif
1445    end if
1446  end subroutine search

1501  subroutine process_terminal_node(node)
1502    !:Purpose: Look for actual near neighbors in 'node', and update
1503    !           the search results on the sr(:) data structure.
1504    !
1505    type (tree_node), pointer          :: node
1506    !
1507    real(kdkind), pointer          :: qv(:)
1508    integer, pointer       :: ind(:)
1509    real(kdkind), pointer          :: data(:,:)
1510    !
1511    integer                :: dimen, i, indexofi, k, centeridx, correltime
1512    real(kdkind)                   :: ballsize, sd, newpri
1513    logical                :: rearrange
1514    type(pq), pointer      :: pqp 
1515    integer :: mythread
1516    integer, external :: omp_get_thread_num
1517
1518    mythread = omp_get_thread_num()
1519
1520    !
1521    ! copy values from sr(:) to local variables
1522    !
1523    !
1524    ! Notice, making local pointers with an EXPLICIT lower bound
1525    ! seems to generate faster code.
1526    ! why?  I don't know.
1527    qv => sr(mythread)%qv(1:) 
1528    pqp => sr(mythread)%pq
1529    dimen = sr(mythread)%dimen
1530    ballsize = sr(mythread)%ballsize 
1531    rearrange = sr(mythread)%rearrange
1532    ind => sr(mythread)%ind(1:)
1533    data => sr(mythread)%Data(1:,1:)     
1534    centeridx = sr(mythread)%centeridx
1535    correltime = sr(mythread)%correltime
1536
1537    !    doing_correl = (centeridx >= 0)  ! Do we have a decorrelation window? 
1538    !    include_point = .true.    ! by default include all points
1539    ! search through terminal bucket.
1540
1541    mainloop: do i = node%l, node%u
1542       if (rearrange) then
1543          sd = 0.0
1544          do k = 1,dimen
1545             sd = sd + (data(k,i) - qv(k))**2
1546             if (sd>ballsize) cycle mainloop
1547          end do
1548          indexofi = ind(i)  ! only read it if we have not broken out
1549       else
1550          indexofi = ind(i)
1551          sd = 0.0
1552          do k = 1,dimen
1553             sd = sd + (data(k,indexofi) - qv(k))**2
1554             if (sd>ballsize) cycle mainloop
1555          end do
1556       endif
1557
1558       if (centeridx > 0) then ! doing correlation interval?
1559          if (abs(indexofi-centeridx) < correltime) cycle mainloop
1560       endif
1561
1562
1563       ! 
1564       ! two choices for any point.  The list so far is either undersized,
1565       ! or it is not.
1566       !
1567       ! If it is undersized, then add the point and its distance
1568       ! unconditionally.  If the point added fills up the working
1569       ! list then set the sr(mythread)%ballsize, maximum distance bound (largest distance on
1570       ! list) to be that distance, instead of the initialized +infinity. 
1571       !
1572       ! If the running list is full size, then compute the
1573       ! distance but break out immediately if it is larger
1574       ! than sr(mythread)%ballsize, "best squared distance" (of the largest element),
1575       ! as it cannot be a good neighbor. 
1576       !
1577       ! Once computed, compare to best_square distance.
1578       ! if it is smaller, then delete the previous largest
1579       ! element and add the new one. 
1580
1581       if (sr(mythread)%nfound .lt. sr(mythread)%nn) then
1582          !
1583          ! add this point unconditionally to fill list.
1584          !
1585          sr(mythread)%nfound = sr(mythread)%nfound +1 
1586          newpri = pq_insert(pqp,sd,indexofi)
1587          if (sr(mythread)%nfound .eq. sr(mythread)%nn) ballsize = newpri
1588          ! we have just filled the working list.
1589          ! put the best square distance to the maximum value
1590          ! on the list, which is extractable from the PQ. 
1591
1592       else
1593          !
1594          ! now, if we get here,
1595          ! we know that the current node has a squared
1596          ! distance smaller than the largest one on the list, and
1597          ! belongs on the list. 
1598          ! Hence we replace that with the current one.
1599          !
1600          ballsize = pq_replace_max(pqp,sd,indexofi)
1601       endif
1602    end do mainloop
1603    !
1604    ! Reset sr(mythread) variables which may have changed during loop
1605    !
1606    sr(mythread)%ballsize = ballsize 
1607
1608  end subroutine process_terminal_node

1822    !    end do
