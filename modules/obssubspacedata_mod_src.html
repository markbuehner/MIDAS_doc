  89  type :: struct_oss_obsdata
  90
  91     !  Structure storing information associated to observations such
  92     !  as BURP file reports (irep) either retrieved from BURP files 
  93     !  themselves or from other sources.  
  94     !     
  95     !  Variable               Description
  96     !  --------               -----------
  97     !  ndim                   number of dimensions of the data arrays
  98     !                         (i.e. ndim=1 for std and ndim=2 for averaging kernels)
  99     !  nrep                   number of reports or observations
 100     !  dim1                   first array dimension for each  report/observation
 101     !  dim2                   second array dimension for each  report/observation (only relevant for ndim=2)
 102     !  data1d                 obs data of dimension (dim1,nrep)
 103     !  data2d                 obs data of dimension (dim1,dim2,nrep)
 104     !  irep                   current report position
 105     !  code                   unique character string for identifying the report/observation
 106     !
 107     !  Follow-up to be made: modify dim1 and dim2 as pointer arrays dependent on irep
 108     !
 109     
 110     real(8), pointer :: data1d(:,:),data2d(:,:,:)
 111     character(len=oss_code_len), pointer :: code(:)
 112     integer :: ndim,nrep,dim1,dim2,irep

 122  subroutine oss_obsdata_alloc(obsdata,nrep,dim1,dim2_opt)
 123    !
 124    !:Purpose: Allocates memory for structure struct_oss_obsdata to hold obs data file information.
 125    !           If dim2 is specified, then the data array associated with each observation/report
 126    !           will be 2D array. will be a 1D array if dim2 is not specified.
 127    ! 
 128    !:Arguments:
 129    !      :obsdata: data structure to allocate
 130    !      :nrep: max number of associated observations/reports for data 
 131    !      :dim1: first dimension length of the array associated to each observation/report
 132    !      :dim2: second dimension length of the array associated to each observation/report (optional)
 133    !
 134
 135    implicit none
 136
 137    ! Arguments:
 138    type(struct_oss_obsdata), intent(inout)        :: obsdata
 139    integer                 , intent(in)           :: nrep
 140    integer                 , intent(in)           :: dim1 
 141    integer                 , intent(in), optional :: dim2_opt
 142    
 143    obsdata%nrep = nrep
 144    obsdata%dim1 = dim1
 145
 146    if (present(dim2_opt)) then
 147       obsdata%dim2 = dim2_opt
 148       obsdata%ndim = 2
 149       allocate(obsdata%data2d(dim1,dim2_opt,nrep))
 150       obsdata%data2d(:,:,:) = 0.0D0
 151       obsdata%data1d => null()
 152    else
 153       obsdata%dim2 = 0
 154       obsdata%ndim = 1
 155       allocate(obsdata%data1d(dim1,nrep))
 156       obsdata%data1d(:,:) = 0.0D0
 157       obsdata%data2d => null()
 158    end if
 159
 160    ! code is a character string assigned to each observation/report to uniquely identify it
 161    allocate(obsdata%code(nrep))
 162    
 163    ! obsdata%irep is a counter used to keep tract of position in the
 164    ! data arrays when extracting values via oss_obsdata_get_array*.
 165    ! The value is initialized to one, pointing to the very first element.
 166    obsdata%irep = 1
 167
 168  end subroutine oss_obsdata_alloc

 190  function oss_obsdata_get_element( obsdata, code, idim1, stat_opt ) result(element)
 191    ! 
 192    !:Purpose: Returns element of array from obsdata 1D data array. The returned element is the
 193    !           one with the specified identifying code.
 194    !
 195    !:Arguments: 
 196    !           :obsdata: struct_oss_obsdata instance
 197    !           :code:    unique identifying code
 198    !           :idim1:   position of element in the dim1 axis
 199    !           :element: retrieved element from obsdata%data1d
 200    !           :stat:    status code
 201    !
 202
 203    type(struct_oss_obsdata), intent(inout)         :: obsdata
 204    character(len=*)        , intent(in)            :: code
 205    integer                 , intent(in)            :: idim1
 206    integer                 , intent(out), optional :: stat_opt
 207
 208    real(8) :: element
 209    
 210    ! find obsdata%irep for current observation
 211    call obsdata_set_index(obsdata,code,stat_opt=stat_opt)
 212    if (present(stat_opt)) then
 213       if (stat_opt.ne.0) then
 214          element = 0.
 215          return
 216       end if
 217    end if
 218
 219    ! get element from data array at current position
 220    element = obsdata%data1d(idim1,obsdata%irep)
 221    
 222    ! increment position in data array
 223    if (obsdata%irep.eq.obsdata%nrep) then
 224       obsdata%irep = 1
 225    else
 226       obsdata%irep=obsdata%irep+1
 227    end if
 228
 229  end function oss_obsdata_get_element

 351  subroutine obsdata_set_index( obsdata, code, stat_opt )
 352    ! 
 353    !:Purpose: Sets the position variable (irep) in struct_oss_obsdata to reference the record
 354    !           that matches the input identifying code.
 355    !
 356    !:Arguments: 
 357    !           :obsdata:      struct_oss_obsdata instance
 358    !           :index:        obs index
 359    !           :code:         code for comparison to those in obsdata
 360    !           :obsdata%irep: current index position for the observation/report
 361    !           :stat_opt:     status of call (optional)
 362    !                    :0: no errors
 363    !                    :1: no reports available
 364    !                    :2: report not found
 365    !
 366    !:Comments: If the optional argument stat_opt is provided and an error occurs, the error code will
 367    !            be returned and the abort will not be called to allow for error handling.
 368    ! 
 369
 370    type(struct_oss_obsdata), intent(inout)         :: obsdata
 371    character(len=*)        , intent(in)            :: code
 372    integer                 , intent(out), optional :: stat_opt
 373
 374    integer :: i
 375
 376    ! Additional declarations for use with obsspace_extra_code_test
 377    
 378    integer :: ref_lat
 379    
 380    if ( obsdata%nrep <= 0 ) then
 381      if ( present( stat_opt ) ) then
 382        stat_opt = 1
 383        return
 384      else
 385        call utl_abort("obsdata_set_index: No reports available. Check for consistency " // &
 386                       "between input BURP file and input NAMBURP_FILTER_*  namelist " // &
 387                       "(and input auxiliary file if part of CH family).")
 388      end if
 389    end if
 390
 391    i=0
 392    
 393    ! Search for matching identifier code
 394    do while (trim(obsdata%code(obsdata%irep)) /= trim(code))
 395       obsdata%irep=obsdata%irep+1
 396       if (obsdata%irep > obsdata%nrep) obsdata%irep=1
 397       if (i > obsdata%nrep) then
 398          if (len_trim(code) >= oss_code_sublen) then
 399       
 400             ! Assumes codes of the form "LAT--LON--YYYYMMDDHHMM*" when len(code)>=oss_code_sublen.
 401	  
 402             ! Upper loop search did not find a match. For valid data near the poles over
 403             ! the global analysis grid, the lat could have been moved to the nearest analysis grid
 404             ! latitude. The following is to account for this, assuming this is the only exception
 405             ! for points near the poles. 
 406             !
 407             ! This is done as a second search step so as not to slow down the normally sufficient 
 408             ! search performed above.
 409
 410             i=0
 411             read(code(1:oss_code_latlen),*) ref_lat
 412             
 413             ! Search for matching identifier code
 414             do while (.not.obsdata_extra_code_test(trim(obsdata%code(obsdata%irep)),code,ref_lat))
 415                obsdata%irep=obsdata%irep+1
 416                if (obsdata%irep > obsdata%nrep) obsdata%irep=1
 417                if (i > obsdata%nrep) exit
 418                i=i+1       
 419             end do
 420             if (i > obsdata%nrep) then
 421                if (present(stat_opt)) then
 422                   stat_opt = 2
 423                   return
 424                else
 425                   call utl_abort("obsdata_set_index: Obs index not found for nrep = " // trim(utl_str(obsdata%nrep)) // " and code = '" // code // "'")
 426                end if
 427             end if
 428          else
 429             if (present(stat_opt)) then
 430                stat_opt = 2
 431                return
 432             else
 433                call utl_abort("obsdata_set_index: Obs index not found for nrep = " // trim(utl_str(obsdata%nrep)) // " and code = '" // code // "'")
 434             end if
 435          end if
 436          exit
 437       end if 
 438       i=i+1       
 439    end do
 440         
 441    if (present(stat_opt)) stat_opt = 0
 442
 443  end subroutine obsdata_set_index

 447  function obsdata_extra_code_test( test_code, ref_code, ref_lat ) result(found)
 448    ! 
 449    !:Purpose: Test matching of code values accounting for rare differences
 450    !           in stored lat (and lon) value(s) when codes are stored as strings in the form  
 451    !           LAT--LON--YYYYMMDDHHMM* (ie. with >= oss_code_sublen characters).
 452    !
 453    !:Caveat: The current version assumes the only source of difference would stem from
 454    !          a shift to the nearest latitude of the analysis grid from points near the pole.
 455    !          (this source of difference identified by M. Sitwell)
 456    !          Also currently assumes that most poleward analysis grid latitudes are within 1 degree
 457    !          away from a pole.
 458    !
 459    !:Arguments:
 460    !           :test_code: code for comparison to ref_code  
 461    !           :ref_lat:   latitude  (x100) part of reference code   
 462    !           :ref_code:  reference code
 463    !           :found:     logical indicating if a match has been found.  
 464    ! 
 465
 466    implicit none
 467
 468    character(len=*), intent(in) :: test_code
 469    character(len=*), intent(in) :: ref_code
 470    integer         , intent(in) :: ref_lat
 471      
 472    integer, parameter :: lat_lim1=-8900    ! Lat*100
 473    integer, parameter :: lat_lim2=8900
 474    integer :: lat
 475    logical :: found
 476          
 477    if (test_code(6:len_trim(test_code)) /= ref_code(oss_code_latlen+1:len_trim(ref_code))) then
 478      found=.false.
 479      return
 480    else 
 481      read(test_code(1:oss_code_latlen),*) lat
 482      if ((lat < lat_lim1 .and. ref_lat < lat_lim1 .and. lat < ref_lat ).or. &
 483          (lat > lat_lim2 .and. ref_lat > lat_lim2 .and. lat > ref_lat ) ) then
 484        found=.true.	     
 485        write(*,*) 'obsdata_extra_code_test: Accounted for lat. mismatch in codes near poles: ', &
 486              lat,ref_lat
 487      else
 488        found=.false.
 489      end if
 490    end if
 491
 492  end function obsdata_extra_code_test

 496  function obsdata_get_header_code_i( ilon, ilat, date, time, stnid ) result(code)
 497    ! 
 498    !:Purpose: Generates a string code to identify an obervation by the header information in
 499    !           a BURP report. The BURP header information is saved as a string in the form  
 500    !           LAT--LON--YYYYMMDDHHMMSTNID----. Intention of this function is to be used for
 501    !           setting the unique identifier 'code' in struct_oss_obsdata. Can be called under
 502    !           the interface oss_obsdata_get_header_code.
 503    !
 504    !:Arguments:
 505    !           :ilon:  longitude integer
 506    !           :ilat:  latitude integer
 507    !           :date:  date in YYYYMMDD
 508    !           :time:  time in HHMM
 509    !           :stnid: station ID
 510    !           :code:  unique code
 511    ! 
 512
 513    implicit none
 514
 515    integer, intent(in) :: ilon,ilat,date,time
 516    character(len=*), intent(in) :: stnid
 517    character(len=oss_code_len) :: code
 518
 519    write(code(1:5),'(I5.5)') ilat
 520
 521    if (ilon > 0) then
 522      write(code(6:10),'(I5.5)') ilon
 523    else
 524      write(code(6:10),'(I5.5)') 36000 + ilon
 525    end if
 526    
 527    write(code(11:18),'(I8.8)') date
 528    write(code(19:22),'(I4.4)') time
 529
 530    write(code(23:oss_code_len),'(A)') stnid(1:min(len_trim(stnid),oss_code_len-oss_code_sublen))
 531
 532  end function obsdata_get_header_code_i

 536  function obsdata_get_header_code_r( lon, lat, date, time, stnid ) result(code)
 537    ! 
 538    !:Purpose: Generates a string code to identify an obervation by the header information in
 539    !           a BURP report. The BURP header information is saved as a string in the form  
 540    !           LAT--LON--YYYYMMDDHHMMSTNID----. Intention of this function is to be used for
 541    !           setting the unique identifier 'code' in struct_oss_obsdata. Can be called under
 542    !           the interface oss_obsdata_get_header_code.
 543    !
 544    !:Arguments:
 545    !           :lon:   longitude real (radians)
 546    !           :lat:   latitude real (radians)
 547    !           :date:  date in YYYYMMDD
 548    !           :time:  time in HHMM
 549    !           :stnid: station ID
 550    !           :code:  unique code
 551    ! 
 552
 553    implicit none
 554
 555    real(8)         , intent(in) :: lon
 556    real(8)         , intent(in) :: lat
 557    integer         , intent(in) :: date
 558    integer         , intent(in) :: time
 559    character(len=*), intent(in) :: stnid
 560    character(len=oss_code_len)  :: code
 561
 562    ! locals
 563    integer :: ilon, ilat
 564    
 565    ilon = nint(100*(lon/MPC_RADIANS_PER_DEGREE_R8))
 566    ilat = nint(100*(90. + lat/MPC_RADIANS_PER_DEGREE_R8))
 567
 568    code = obsdata_get_header_code_i(ilon,ilat,date,time,stnid)

 574  subroutine oss_obsdata_add_data1d(obsdata,val,code,maxsize,dim1_opt)
 575    !
 576    !:Purpose: Add data value(s) to obsdata%data1d with associated identifier code.
 577    !
 578    !:Arguments:
 579    !           obsdata       struct_oss_obsdata instance
 580    !           val           data array to store in obsdata%data1d
 581    !           code          identifying code based on (lat,long,date,hhmm) if not also stnid
 582    !           maxsize       max allowed size for obsdata
 583    !           dim1          value() dimension (optional)
 584    !           obsdata       Updated obsdata 
 585    ! 
 586    !:Comments:
 587    !
 588    !          - Retrieval of values from obsdata%data1d to be done via oss_obsdata_get_element (or oss_obsdata_get_array1d).
 589    !          - If obsdata allocation is required for all processors (such as for use later with obsdata_MPIGather), 
 590    !            allocation and/or initialization of arrays needs to be done at a corresponding appropriate location 
 591    !            outside the obs operations such as in oss_setup to ensure allocation is done for all processors, 
 592    !            including those without associated data. This is to ensure that rpn_comm_allgather will work 
 593    !            in routine obsdata_MPIGather.
 594    !
 595
 596    implicit none 
 597    
 598    type(struct_oss_obsdata), intent(inout)        :: obsdata
 599    real(8)                 , intent(in)           :: val(:)
 600    integer                 , intent(in)           :: maxsize
 601    character(len=*)        , intent(in)           :: code
 602    integer                 , intent(in), optional :: dim1_opt
 603
 604    if (.not.associated(obsdata%data1d)) then
 605      if (present(dim1_opt)) then 
 606         call oss_obsdata_alloc(obsdata,maxsize,dim1=dim1_opt)
 607      else
 608         call oss_obsdata_alloc(obsdata,maxsize,dim1=1)
 609      end if
 610      obsdata%nrep=0
 611    end if
 612
 613    if (obsdata%dim1 > size(val)) &
 614         call utl_abort('obsdata_add_data1d: Insufficient data values provided. ' // trim(utl_str(size(val))) )
 615     
 616    ! nrep counts the number data values/profiles in the data arrays
 617    obsdata%nrep = obsdata%nrep+1 
 618
 619    if (obsdata%nrep > maxsize) &
 620         call utl_abort('obsdata_add_data1d: Reach max size of array ' // trim(utl_str(maxsize)) )
 621  
 622    ! Save unique code
 623    obsdata%code(obsdata%nrep)=trim(code)
 624    
 625    ! Save value(s)
 626    obsdata%data1d(1:obsdata%dim1,obsdata%nrep) = val(1:obsdata%dim1)
 627    
 628  end subroutine oss_obsdata_add_data1d

 632  subroutine oss_obsdata_MPIallgather(obsdata) 
 633    !
 634    !:Purpose: Gathers previously saved obsdata from all processors.
 635    !
 636    !:Arguments:
 637    !           :obsdata: Local struct_oss_obsdata to become global
 638    ! 
 639    !:Comments:
 640    !
 641    !           - Assumes obsdata%dim1 (and obsdata%dim2) the same over all processors.
 642    !
 643
 644    implicit none
 645
 646    type(struct_oss_obsdata), intent(inout) :: obsdata
 647
 648    integer, allocatable :: nrep(:)
 649    character(len=oss_code_len), allocatable :: code_local(:),code_global(:,:)
 650    real(8), allocatable :: data1d_local(:,:),data1d_global(:,:,:),data2d_local(:,:,:),data2d_global(:,:,:,:)
 651    integer :: i,ierr,nproc,nrep_total,nrep_max,irep,array_size
 652
 653    write(*,*) 'Begin oss_obsdata_MPIallgather'
 654      
 655    ! Identify number of processors.
 656    call rpn_comm_size("GRID",nproc,ierr)
 657      
 658    ! Get number of reports on each processor
 659
 660    allocate(nrep(nproc))
 661    nrep(:)=0
 662
 663    call rpn_comm_allgather(obsdata%nrep,1,"MPI_INTEGER",nrep,1,"MPI_INTEGER","GRID",ierr)
 664
 665    nrep_total=sum(nrep)
 666
 667    if (nrep_total == 0) then
 668       deallocate(nrep)
 669       write(*,*) 'Exit oss_obsdata_MPIallgather: no reports'
 670       return
 671    end if
 672        
 673    nrep_max = maxval(nrep)
 674
 675    ! Get values from all processors into global arrays
 676
 677    allocate(code_local(nrep_max))
 678    allocate(code_global(nrep_max,nproc))
 679
 680    code_local(:)=''
 681    if (obsdata%nrep > 0) code_local(1:obsdata%nrep) = obsdata%code(1:obsdata%nrep)
 682
 683    call mmpi_allgather_string(code_local,code_global,nrep_max,oss_code_len,nproc,"GRID",ierr)
 684
 685    if (obsdata%ndim == 1) then
 686
 687       allocate(data1d_local(obsdata%dim1,nrep_max))
 688       allocate(data1d_global(obsdata%dim1,nrep_max,nproc))
 689
 690       data1d_local(:,:)=0.0D0
 691       if (obsdata%nrep > 0) data1d_local(:,1:obsdata%nrep) = obsdata%data1d(:,1:obsdata%nrep)
 692       
 693       array_size = nrep_max*obsdata%dim1
 694
 695       call rpn_comm_allgather(data1d_local,array_size,pre_obsMpiReal,data1d_global,array_size,pre_obsMpiReal,"GRID",ierr)
 696    else 
 697       
 698       allocate(data2d_local(obsdata%dim1,obsdata%dim2,nrep_max))
 699       allocate(data2d_global(obsdata%dim1,obsdata%dim2,nrep_max,nproc))
 700       
 701       data2d_local(:,:,:)=0.0D0
 702       if (obsdata%nrep > 0) data2d_local(:,:,1:obsdata%nrep) = obsdata%data2d(:,:,1:obsdata%nrep)
 703
 704       array_size = nrep_max*obsdata%dim1*obsdata%dim2
 705
 706       call rpn_comm_allgather(data2d_local,array_size,pre_obsMpiReal,data2d_global,array_size,pre_obsMpiReal,"GRID",ierr)
 707
 708    end if
 709  
 710    deallocate(code_local)
 711    if (obsdata%ndim == 1) then 
 712        deallocate(data1d_local)
 713    else
 714        deallocate(data2d_local)
 715    end if 
 716
 717    ! Concatenate values from all processors
 718         
 719    irep = 0
 720    call oss_obsdata_dealloc(obsdata)    
 721
 722    if (obsdata%ndim == 1) then      
 723       
 724       call oss_obsdata_alloc(obsdata,nrep_total,obsdata%dim1)
 725
 726       do i=1,nproc
 727          if (nrep(i) > 0) then
 728             obsdata%code(irep+1:irep+nrep(i)) = code_global(1:nrep(i),i)
 729             obsdata%data1d(1:obsdata%dim1,irep+1:irep+nrep(i)) = data1d_global(1:obsdata%dim1,1:nrep(i),i)
 730             irep = irep+nrep(i)
 731          end if
 732       end do
 733
 734    else
 735
 736       call oss_obsdata_alloc(obsdata,nrep_total,obsdata%dim1,obsdata%dim2)
 737
 738       do i=1,nproc
 739          if (nrep(i) > 0) then
 740             obsdata%code(irep+1:irep+nrep(i)) = code_global(1:nrep(i),i)
 741             obsdata%data2d(1:obsdata%dim1,1:obsdata%dim2,irep+1:irep+nrep(i)) = &
 742                         data2d_global(1:obsdata%dim1,1:obsdata%dim2,1:nrep(i),i)  
 743             irep = irep+nrep(i)
 744          end if
 745       end do
 746 
 747    end if  
 748     
 749    deallocate(nrep,code_global)
 750    if (obsdata%ndim == 1) then 
 751        deallocate(data1d_global)
 752    else
 753        deallocate(data2d_global)
 754    end if 
 755    
 756    write(*,*) 'Exit oss_obsdata_MPIallgather'
 757 
 758  end subroutine oss_obsdata_MPIallgather

 762  subroutine oss_get_comboIdlist( obsSpaceData, stnid_list, varno_list, unilev_list, num_elements, nset )
 763    ! 
 764    !:Purpose: Uses the subroutine oss_comboIdlist to compile a unique list of stnid,  
 765    !           (stnid,varno) or (stnid,varno,multi/uni-level) combinations to be used in searches.
 766    !
 767    !:Arguments:
 768    !           :obsSpaceData: Observation space data
 769    !           :stnid_list:   List of unique stnids
 770    !           :varno_list:   List of unique varno
 771    !           :unilev_list:  List of unique uni/multi-level identifications
 772    !           :num_elements: Number of unique elements in *_list arrrays
 773    !           :nset:         Integer indicating grouping, with values indicating
 774    !
 775    !                    - 1: group by stnid
 776    !                    - 2: group by (stnid,bufr)
 777    !                    - 3: group by (stnid,bufr,multi/uni-level)
 778    !
 779
 780    implicit none
 781
 782    type(struct_obs), intent(inout) :: obsSpaceData
 783
 784    integer, parameter :: nmax=100
 785    integer, intent(out) :: num_elements
 786    integer, intent(out) :: nset
 787    integer, intent(out) :: varno_list(nmax)
 788    character(len=9), intent(out) :: stnid_list(nmax)
 789    logical, intent(out) :: unilev_list(nmax)
 790
 791    integer :: headerIndex,bodyIndex,vco,nlev_obs,varno
 792    logical :: all_combos
 793
 794
 795    call oss_comboIdlist(all_combos_opt=all_combos)
 796    
 797    if (all_combos) then
 798    
 799       ! Loop over obs to find all (stnid,varno) pairs to form a common sequence of search pairs
 800       ! over all processors. The prescribed starting stnid selections can have wild card 
 801       ! characters (via *, see routine oss_comboIdlist).
 802
 803       call obs_set_current_header_list(obsSpaceData,'CH')
 804       HEADER: do
 805          headerIndex = obs_getHeaderIndex(obsSpaceData)
 806          if (headerIndex < 0) exit HEADER
 807         
 808          ! Body info that we only need for first point in the profile
 809          bodyIndex = obs_headElem_i(obsSpaceData,OBS_RLN,headerIndex)     
 810          vco = obs_bodyElem_i(obsSpaceData,OBS_VCO,bodyIndex)
 811
 812          if (vco.ne.1 .and. vco.ne.2 .and. vco.ne.4 .and. vco.ne.5) then
 813             ! Vertical coordinate not handled
 814             write(*,*) 'oss_get_comboIdlist: Currently unaccounted VCO = ',vco
 815             cycle HEADER
 816          end if
 817          
 818          ! Identify varno and nlev_obs (exclude BUFR_SCALE_EXPONENT elements)
 819          nlev_obs = obs_headElem_i(obsSpaceData,OBS_NLV,headerIndex)
 820          call obs_set_current_body_list(obsSpaceData,headerIndex)
 821          do
 822             bodyIndex = obs_getBodyIndex(obsSpaceData)
 823             if (bodyIndex < 0) exit
 824             if (obs_bodyElem_i(obsSpaceData,OBS_VNM,bodyIndex).eq.BUFR_SCALE_EXPONENT) then
 825                nlev_obs = nlev_obs-1
 826             else
 827                varno=obs_bodyElem_i(obsSpaceData,OBS_VNM,bodyIndex)
 828             end if
 829          end do
 830
 831          ! Adds to running list of unique pairs if unique
 832          call oss_comboIdlist(stnid_add_opt=obs_elem_c(obsSpaceData,'STID',headerIndex), varno_add_opt=varno, unilev_add_opt=(nlev_obs.eq.1.and.vco.ge.4))
 833                       
 834       end do HEADER
 835       
 836       ! Get a common sequence of search pairs over all processors. 
 837       
 838       call oss_comboIdlist(gather_mpi_opt=.true.)
 839       
 840    end if
 841    
 842    ! Get list of unique pairs
 843    call oss_comboIdlist(stnid_list_opt=stnid_list, varno_list_opt=varno_list, unilev_list_opt=unilev_list, num_elements_opt=num_elements, nset_opt=nset)
 844
 845  end subroutine oss_get_comboIdlist

 849  subroutine oss_comboIdList( stnid_add_opt, varno_add_opt, unilev_add_opt, stnid_list_opt, &
 850                              varno_list_opt, unilev_list_opt, &
 851                              num_elements_opt, initialize_opt, nset_opt, gather_mpi_opt, all_combos_opt )
 852    ! 
 853    !:Purpose: Provide list of fixed or accumulated stnid, (stnid,varno) or 
 854    !           (stnid,varno,multi/uni-level) combinations to be used in searches.
 855    !
 856    !           Can be used for both single processor and  MPI mode, where 'gather_mpi' must be set
 857    !           to .true. at some point for use with MPI.
 858    !            Called from osd_chem_diagnmostics in file obspacediag_mod.ftn90.
 859    !  
 860    !:Arguments:
 861    !           :stnid_add_opt:    stnid to add to stnid_list if part of unique set
 862    !           :varno_add_opt:    varno to add to varno_list if part of unique set
 863    !           :unilev_add_opt:   unilev logical to add to unilev_list if part of unique set
 864    !           :initialize_opt:   Initialize internal arrays and counters
 865    !           :gather_mpi_opt:   Gather all local MPI process and recompile unique lists  
 866    !           :nset_opt:         Integer indicating grouping of diagnostics. Input variable
 867    !                              if initialize=.true., output variable otherwise.
 868    !                              Values indicate
 869    !
 870    !                              - 1: group by stnid
 871    !                              - 2: group by (stnid,bufr)
 872    !                              - 3: group by (stnid,bufr,multi/uni-level)
 873    !           :all combos_opt:   Indicates if all combinations specified by nset are to
 874    !                              be use, or only those specified in the namelist NAMCHEM
 875    !                              Input variable if initialize=.true., output variable otherwise.
 876    !           :stnid_list_opt:   List of unique stnids
 877    !           :varno_list_opt:   List of unique varno
 878    !           :unilev_list_opt:  List of unique uni/multi-level identifications
 879    !           :num_elements_opt: Number of unique elements in *_list arrrays
 880    !
 881   
 882    implicit none
 883    
 884    integer, parameter :: nmax=100, stnid_len=9
 885
 886    logical                 , intent(in)   , optional :: initialize_opt
 887    logical                 , intent(in)   , optional :: gather_mpi_opt
 888    logical                 , intent(in)   , optional :: unilev_add_opt
 889    integer                 , intent(in)   , optional :: varno_add_opt
 890    character(len=stnid_len), intent(in)   , optional :: stnid_add_opt
 891    integer                 , intent(inout), optional :: nset_opt
 892    logical                 , intent(inout), optional :: all_combos_opt
 893    integer                 , intent(out)  , optional :: varno_list_opt(nmax)
 894    integer                 , intent(out)  , optional :: num_elements_opt
 895    character(len=stnid_len), intent(out)  , optional :: stnid_list_opt(nmax)
 896    logical                 , intent(out)  , optional :: unilev_list_opt(nmax)
 897
 898    ! local arrays of unique values
 899    integer, save :: varno_unique(nmax)
 900    character(len=stnid_len), save :: stnid_unique(nmax)
 901    logical, save :: unilev_unique(nmax)
 902    
 903    ! global arrays of unique values
 904    integer, allocatable :: num_unique_all(:),varno_unique_all(:,:)
 905    character(len=stnid_len),allocatable :: stnid_unique_all(:,:)
 906    logical, allocatable :: unilev_unique_all(:,:)
 907
 908    integer, save :: num_unique ! running count of number of unique elements
 909    integer, save :: iset=2
 910    logical, save :: lall_combos=.true.
 911
 912    logical :: same,init
 913    integer :: i,j,nproc,iproc,ierr
 914
 915    init=.false.
 916    if (present(initialize_opt)) init = initialize_opt
 917
 918    
 919    ! Initialize internal arrays and counters
 920    if (init) then
 921       stnid_unique(:) = ''
 922       varno_unique(:) = 0
 923       unilev_unique(:) = .false.
 924       num_unique = 0
 925       if (present(nset_opt)) iset = nset_opt
 926       if (present(all_combos_opt)) lall_combos = all_combos_opt
 927    end if      
 928
 929
 930    ! Add new elements to internal arrays if not there already
 931    if (present(stnid_add_opt)) then
 932      
 933       if (iset >= 2 .and. (.not. present(varno_add_opt))) call utl_abort('oss_comboIdlist: varno_add must be present to add element for nset>=2.')
 934       if (iset >= 3 .and. (.not. present(unilev_add_opt))) call utl_abort('oss_comboIdlist: unilev_add must be present to add element for nset>=3.')
 935
 936       same = .false.
 937
 938       do i=1,num_unique
 939          same = utl_stnid_equal(stnid_add_opt,stnid_unique(i))
 940          if (iset >= 2) same = same .and. varno_add_opt == varno_unique(i)
 941          if (iset >= 3) same = same .and. unilev_add_opt.eqv.unilev_unique(i)
 942          if (same) exit
 943       end do
 944
 945       if (.not.same) then
 946          num_unique=num_unique+1
 947          if (num_unique > nmax) call utl_abort("oss_comboIDlist: Max allowed dimension exceeded.")
 948          stnid_unique(num_unique) = stnid_add_opt
 949          if (iset >= 2) varno_unique(num_unique) = varno_add_opt
 950          if (iset >= 3) unilev_unique(num_unique) = unilev_add_opt
 951       end if
 952    end if        
 953
 954    ! Gather unique arrays from each local mpi process and compile global unique arrays
 955    if (present(gather_mpi_opt)) then
 956       if (gather_mpi_opt) then
 957
 958          call rpn_comm_size("GRID",nproc,ierr)
 959
 960          allocate(num_unique_all(nproc))
 961          allocate(stnid_unique_all(nmax,nproc))
 962          if (iset >= 2) allocate(varno_unique_all(nmax,nproc))
 963          if (iset >= 3) allocate(unilev_unique_all(nmax,nproc))
 964          
 965          num_unique_all(:) = 0
 966          stnid_unique_all(:,:) = ''
 967          if (iset >= 2) varno_unique_all(:,:) = 0
 968          if (iset >= 3) unilev_unique_all(:,:) = .false.
 969          
 970          if(mmpi_doBarrier) call rpn_comm_barrier("GRID",ierr)
 971
 972          call rpn_comm_allgather(num_unique,1,"MPI_INTEGER",num_unique_all,1,"MPI_INTEGER","GRID",ierr)
 973          call mmpi_allgather_string(stnid_unique,stnid_unique_all,nmax,stnid_len,nproc,"GRID",ierr)
 974          if (iset >= 2) call rpn_comm_allgather(varno_unique,nmax,"MPI_INTEGER",varno_unique_all,nmax,"MPI_INTEGER","GRID",ierr)
 975          if (iset >= 3) call rpn_comm_allgather(unilev_unique,nmax,"MPI_LOGICAL",unilev_unique_all,nmax,"MPI_LOGICAL","GRID",ierr)
 976          
 977          stnid_unique(:) = ''
 978          if (iset >= 2) varno_unique(:) = 0
 979          if (iset >= 3) unilev_unique(:) = .false.
 980          num_unique = 0
 981          
 982          ! Amalgamate unique lists
 983          do iproc=1,nproc
 984             do j=1,num_unique_all(iproc)
 985
 986                same = .false.
 987
 988                do i=1,num_unique
 989                   same = utl_stnid_equal(stnid_unique_all(j,iproc),stnid_unique(i))
 990                   if (iset >= 2) same = same .and. varno_unique_all(j,iproc) == varno_unique(i)
 991                   if (iset >= 3) same = same .and. unilev_unique_all(j,iproc).eqv.unilev_unique(i)
 992                   if (same) exit
 993                end do
 994              
 995                if (.not.same) then
 996                   num_unique=num_unique+1
 997                   if (num_unique > nmax) call utl_abort("oss_comboIDlist: Max allowed dimension exceeded.")
 998                   stnid_unique(num_unique) = stnid_unique_all(j,iproc)
 999                   if (iset >= 2) varno_unique(num_unique) = varno_unique_all(j,iproc)
1000                   if (iset >= 3) unilev_unique(num_unique) = unilev_unique_all(j,iproc)
1001                end if
1002             end do
1003          end do
1004
1005          deallocate(num_unique_all,stnid_unique_all)
1006          if (allocated(varno_unique_all)) deallocate(varno_unique_all)
1007          if (allocated(unilev_unique_all)) deallocate(unilev_unique_all)
1008
1009       end if
1010    end if
1011
1012
1013    ! Return internal arrays (and other info) if requested
1014    if (present(varno_list_opt)) varno_list_opt = varno_unique
1015    if (present(stnid_list_opt)) stnid_list_opt = stnid_unique
1016    if (present(unilev_list_opt)) unilev_list_opt = unilev_unique
1017    if (present(num_elements_opt)) num_elements_opt = num_unique
1018    if (.not. init) then
1019       if (present(nset_opt)) nset_opt = iset
1020       if (present(all_combos_opt)) all_combos_opt = lall_combos
1021    end if
1022    
1023    
1024  end subroutine oss_comboIdList
