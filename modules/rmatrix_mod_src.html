 17module rMatrix_mod
 18  ! MODULE rMatrix_mod (prefix='rmat' category='2. B and R matrices')
 19  !
 20  !:Purpose: Module to handle non-diagonal observation-error covariance
 21  !           matrices for assimilation of radiances
 22  !
 23  use rttov_interfaces_mod
 24  use midasMpi_mod
 25  use rttov_const, only  : errorstatus_success
 26  use utilities_mod
 27  use obsSpaceData_mod
 28  use tovs_nl_mod
 29  implicit none
 30  private
 31  save
 32
 33  ! public variables
 34  public :: rmat_lnondiagr
 35  ! public subroutines
 36  public :: rmat_init,rmat_cleanup,rmat_readCMatrix,rmat_RsqrtInverseOneObs, rmat_RsqrtInverseAllObs
 37
 38 type rmat_matrix
 39    real(8) ,pointer     :: Rmat(:,:)=>null()
 40    integer ,pointer     :: listChans(:)=>null()
 41    integer              :: nchans=0
 42 END type rmat_matrix
 43
 44  type(rmat_matrix),target,allocatable  :: Rcorr_inst(:) ! non diagonal Correlation matrices for each instrument
 45  type(rmat_matrix),target,allocatable  :: R_tovs(:) ! non diagonal R matrices used for the assimilation of all radiances
 46
 47  ! namelist variable
 48  logical :: rmat_lnondiagr ! choose to use non-diagonal R matrix (i.e. non-zero correlations)
 49
 50  contains

 52  subroutine rmat_init(nsensors,nobtovs)
 53   
 54    implicit none
 55
 56    ! Arguments:
 57    integer, intent (in) :: nsensors
 58    integer, intent (in) :: nobtovs
 59
 60    ! locals
 61    integer :: nulnam,ierr
 62    integer ,external:: fnom,fclos
 63    namelist /NAMRMAT/rmat_lnonDiagR
 64
 65    ! Default value for parameter rmat_lnondiagr, don't use interchannel correlation by default
 66    rmat_lnonDiagR = .false.
 67
 68    ! Read the parameters from NAMRMAT
 69    nulnam = 0
 70    ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
 71    read(nulnam,nml=namrmat, iostat=ierr)
 72    if (ierr /= 0) call utl_abort('rmat_init: Error reading namelist')
 73    if (mmpi_myid == 0) write(*,nml=namrmat)
 74    ierr = fclos(nulnam)
 75    if (rmat_lnonDiagR) then
 76      allocate(Rcorr_inst(nsensors))
 77      allocate(R_tovs(nobtovs))
 78    end if
 79
 80  end subroutine rmat_init
 81
 82  subroutine rmat_cleanup()
 83    implicit none
 84    if (rmat_lnondiagr) then
 85      deallocate(Rcorr_inst)
 86      deallocate(R_tovs)
 87    end if
 88  end subroutine rmat_cleanup

 90  subroutine rmat_readCMatrix(instrument, sensor_id, ichan )
 91   
 92    implicit none
 93    
 94    ! Arguments
 95    integer ,intent (in) :: instrument(3)
 96    integer ,intent (in) :: sensor_id
 97    integer ,intent (in) :: ichan(:)
 98
 99    ! locals
100    character (len=64) :: filename
101    integer :: err
102
103    call rttov_coeffname (err, instrument, coeffname=filename, filetype="Cmat")
104    
105    if (err == errorstatus_success) then
106       filename = trim(filename) // ".dat"
107       call rmat_readCMatrixByFileName(filename,Rcorr_inst(sensor_id), ichan )
108    else
109      write(*,*) "Unknown instrument ",instrument(:)
110      call utl_abort("rmat_read_C_matrix")
111    end if
112    
113  end subroutine rmat_readCMatrix

116  subroutine rmat_readCMatrixByFileName(infile,C,chanList_opt)
117    implicit none
118
119    ! Arguments:
120    character (len=*),intent(in) :: infile ! name of input file
121    type(rmat_matrix),intent(inout) :: C    ! correlation matrix structure
122    integer ,intent(in),optional :: chanList_opt(:) ! list of requested channels (if missing will read all file content)
123
124    ! locals
125    integer :: i,j,iu,ierr,count,ich,nchn,nch
126    integer ,external :: fnom,fclos
127    real(8) :: x
128    integer ,allocatable :: index(:)
129
130    nchn = -1
131    if (present(chanList_opt)) then
132      nchn=size(chanList_opt)
133    end if
134
135    iu = 0
136    ierr = fnom(iu,trim(infile),'FTN+SEQ+R/O',0)
137    if (ierr /= 0) then
138      write(*,*) "Cannot open "//trim(infile)
139      call utl_abort("rmat_readCMatrixByFileName")
140    end if
141
142    write(*,*) "rmat_readCMatrixByFileName: Reading "//trim(infile)
143    
144    read(iu,*) nch
145    if (nchn == -1) then
146      nchn = nch
147    else
148      if(nchn > nch) then
149        write(*,*) "Not enough channels in "//trim(infile)
150        write(*,*) nchn,nch
151        call utl_abort("rmat_readCMatrixByFileName")
152      end if
153    end if
154    allocate(index(nch))
155    
156    C%nchans = nchn
157    allocate(C%Rmat(nchn,nchn))
158    allocate(C%listChans(nchn))
159    C%Rmat = 0.d0
160    do i = 1,nchn
161      C%Rmat(i,i) = 1.d0
162    end do
163    count = 0
164    index = -1
165    do i=1,nch
166      read(iu,*) ich
167      if (present(chanList_opt)) then
168        bj:do j=1,nchn
169          if (ich == chanList_opt(j)) then
170            count = count + 1
171            index(i) = j
172            C%listChans(count) = ich
173            exit bj
174          end if
175        end do bj
176      else
177        index(i) = i
178        C%listChans(i) = ich
179        count = count + 1
180      end if
181    end do
182    if (count /= nchn) then
183      write(*,*) "Warning: Missing information in "//trim(infile)
184      do j=1,nchn
185        write(*,*) j, chanList_opt(j) 
186      end do
187      write(*,*) "Not important if there is no observation of this family"
188    end if
189
190    do
191      read(iu,*,iostat=ierr) i,j,x
192      if (ierr /= 0) exit
193      if (index(i) /= -1 .and. index(j) /= -1) then
194        C%Rmat(index(i),index(j)) = x
195        C%Rmat(index(j),index(i)) = x
196      end if
197    end do
198
199    ierr= fclos(iu)
200    deallocate(index)
201
202  end subroutine rmat_readCMatrixByFileName

205  subroutine rmat_RsqrtInverseOneObs(sensor_id,nsubset,obsIn,obsOut,list_sub,list_oer,indexTovs)
206    !
207    !:Purpose: Apply the operator R**-1/2 to obsIn
208    !           result in obsOut for the subset of channels specified
209    !           in list_sub
210    !
211    implicit none
212
213    ! Arguments:
214    integer , intent (in) :: sensor_id
215    integer , intent (in) :: nsubset
216    integer , intent(in)  :: list_sub(nsubset)
217    real(8) , intent(in)  :: list_oer(nsubset)
218    real(8) , intent(in)  :: obsIn(nsubset)
219    real(8) , intent(out) :: obsOut(nsubset)
220    integer , intent(in)  :: indexTovs
221
222    ! locals
223    real (8) :: Rsub(nsubset,nsubset), alpha, beta, product 
224    integer :: index(nsubset)
225    integer :: i,j
226
227    if (R_tovs(indexTovs)%nchans == 0) then
228
229      if (sensor_id <= 0 .or. sensor_id > size(Rcorr_inst)) then
230        write(*,*) "invalid sensor_id",sensor_id,size(Rcorr_inst)
231        call utl_abort('rmat_RsqrtInverseOneObs')
232      end if
233
234      index = -1
235      do i=1,nsubset
236        bj: do j = 1, Rcorr_inst(sensor_id)%nchans
237          if (list_sub(i) == Rcorr_inst(sensor_id)%listChans(j)) then
238            index(i) = j
239            exit bj 
240          end if
241        end do bj
242      end do
243      if (any(index == -1)) then
244        write(*,*) "Missing information for some channel !"
245        write(*,*) list_sub(:)
246        write(*,*) index(:)
247        call utl_abort('rmat_RsqrtInverseOneObs')
248      end if
249      R_tovs(indexTovs)%nchans = nsubset
250      allocate(R_tovs(indexTovs)%listChans(nsubset))
251      R_tovs(indexTovs)%listChans(1:nsubset) = list_sub(1:nsubset)
252      do j=1,nsubset
253        do i=1,nsubset
254          product = list_oer(i) * list_oer(j)
255          Rsub(i,j) = product * Rcorr_inst(sensor_id)%Rmat(index(i),index(j))
256        end do
257      end do
258      ! Calculation of R**-1/2
259      call utl_tmg_start(20,'----RmatMatSqrt')
260      call utl_matSqrt(Rsub,nsubset,-1.d0,.false.)
261      call utl_tmg_stop(20)
262      allocate(R_tovs(indexTovs)%Rmat(nsubset,nsubset))
263      do j=1,nsubset
264        do i=1,nsubset
265          R_tovs(indexTovs)%Rmat(i,j) = Rsub(i,j)
266        end do
267      end do
268    end if
269
270    call utl_tmg_start(21,'----RmatMatMult')
271    alpha = 1.d0
272    beta = 0.d0
273    obsOut = 0.d0
274    ! Optimized symetric matrix vector product from Lapack
275    call dsymv("L", nsubset, alpha, R_tovs(indexTovs)%Rmat, nsubset,obsIn, 1, beta, obsOut, 1)
276    call utl_tmg_stop(21)
277
278  end subroutine rmat_RsqrtInverseOneObs

284  subroutine rmat_RsqrtInverseAllObs( obsSpaceData, elem_dest_i, elem_src_i )
285    !
286    !:Purpose: To apply observation-error variances to ROBDATA8(k_src,*) and to
287    !          store it in the elem_src_s of obsspacedata
288    implicit none
289
290    ! Arguments:
291    type(struct_obs), intent(inout) :: obsspacedata
292    integer, intent(in)  :: elem_dest_i ! destination index
293    integer, intent(in)  :: elem_src_i  ! source index
294
295    ! Locals:
296    integer :: bodyIndex, headerIndex
297    integer :: idata, idatend, idatyp, count, channelNumber, channelIndex
298    real(8) :: obsIn( tvs_maxChannelNumber ), obsOut( tvs_maxChannelNumber )
299    integer :: list_chan( tvs_maxChannelNumber )
300    real(8) :: list_OER( tvs_maxChannelNumber )
301
302    ! NOTE I tried using openMP on this loop, but it increased the cost from 4sec to 80sec!!!
303    do headerIndex =1, obs_numHeader(obsspacedata)
304
305      idata   = obs_headElem_i( obsspacedata, OBS_RLN, headerIndex )
306      idatend = obs_headElem_i( obsspacedata, OBS_NLV, headerIndex ) + idata - 1
307      idatyp  = obs_headElem_i( obsspacedata, OBS_ITY, headerIndex )
308
309      if ( tvs_isIdBurpTovs(idatyp) .and. rmat_lnondiagr ) then
310
311        count = 0
312
313        do bodyIndex = idata, idatend
314
315          if (obs_bodyElem_i( obsspacedata, OBS_ASS, bodyIndex ) == obs_assimilated ) then
316            call tvs_getChannelNumIndexFromPPP( obsSpaceData, headerIndex, bodyIndex, &
317                                                channelNumber, channelIndex )
318            count = count + 1
319            list_chan( count ) = channelNumber
320            list_OER( count ) = obs_bodyElem_r( obsspacedata, OBS_OER, bodyIndex )
321            obsIn( count ) = obs_bodyElem_r( obsspacedata, elem_src_i, bodyIndex )
322          end if
323        
324        end do
325
326        if ( count > 0 .and. tvs_tovsIndex( headerIndex ) > 0 ) then
327
328          call rmat_RsqrtInverseOneObs( tvs_lsensor( tvs_tovsIndex( headerIndex )), count, obsIn(1:count), obsOut(1:count), list_chan(1:count), list_OER(1:count), tvs_tovsIndex(headerIndex) )
329
330          count = 0
331          do bodyIndex = idata, idatend
332            if ( obs_bodyElem_i( obsspacedata, OBS_ASS, bodyIndex ) == obs_assimilated) then
333              count = count + 1
334              call obs_bodySet_r(obsspacedata, elem_dest_i, bodyIndex,obsOut(count))
335            end if
336          end do
337
338        else
339
340          do bodyIndex = idata, idatend
341            call obs_bodySet_r(obsspacedata, elem_dest_i, bodyIndex, 0.d0)
342          end do
343
344        end if 
345
346      else
347       
348        do bodyIndex = idata, idatend
349          if (obs_bodyElem_i( obsspacedata, OBS_ASS, bodyIndex ) == obs_assimilated) then
350            call obs_bodySet_r( obsspacedata, elem_dest_i, bodyIndex, &
351                 obs_bodyElem_r( obsspacedata, elem_src_i, bodyIndex) / obs_bodyElem_r( obsspacedata, OBS_OER, bodyIndex ))
352          end if
353        end do
354
355      end if ! is it a radiance in non diagonal R mode ?
356
357    end do !loop on header
358 
359  end subroutine rmat_RsqrtInverseAllObs
