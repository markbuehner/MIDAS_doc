 17module bgckOcean_mod
 18  ! MODULE bgckOcean_mod (prefix='ocebg' category='1. High-level functionality')
 19  !
 20  !:Purpose: to perform ocean data background Check
 21  !
 22  use midasMpi_mod
 23  use utilities_mod
 24  use obsSpaceData_mod
 25  use columnData_mod
 26  use codtyp_mod
 27  use gridStateVector_mod
 28  use gridStateVectorFileIO_mod
 29  use horizontalCoord_mod
 30  use verticalCoord_mod
 31  use statetocolumn_mod
 32  use bufr_mod
 33  use mathPhysConstants_mod
 34  use timeCoord_mod
 35  use message_mod

 38
 39  save
 40  private
 41
 42  ! Public functions/subroutines
 43  public :: ocebg_bgCheckSST, ocebg_bgCheckSeaIce
 44
 45  ! External functions
 46  integer, external :: fnom, fclos
 47
 48  ! mpi topology
 49  integer           :: myLatBeg, myLatEnd
 50  integer           :: myLonBeg, myLonEnd
 51  integer           :: latPerPE, latPerPEmax, lonPerPE, lonPerPEmax
 52
 53  ! namelist variables with setting default values
 54  character(len=20) :: timeInterpType_nl = 'NEAREST' ! 'NEAREST' or 'LINEAR'
 55  integer           :: numObsBatches     = 20        ! number of batches for calling interp setup

 57  integer           :: ndaysWinds        = 4         ! number of days in the 'winds' file to detect tropical storm (TS)
 58  integer           :: timeStepWinds     = 6         ! in hours, winds are available every timeStepWinds-hours

 61  real(4)           :: maxLatNH = 40.                ! max lat of N. hemisphere lat band where TS is detected
 62  real(4)           :: maxLatExceptionNH = 45.       ! max lat of N. hemisphere lat band allows TS to penetrate further North in some months
 63  integer           :: nmonthsExceptionNH  = 0       ! MUST NOT BE INCLUDED IN NAMELIST!
 64  character(len=3)  :: monthExceptionNH(12) = '   '  ! exceptional months where TS allowed to penetrated further North 
 65  real(4)           :: minLatSH = -35.               ! min lat of Southern hemisphere latutude band where TS is detected
 66  real(4)           :: maxLatSH = -10.               ! max lat of Southern hemisphere latutude band where TS is detected
 67  real(8)           :: smoothLenghtScale = 50000.    ! length scale. in m, to smooth the amplification error field
 68  real(8)           :: globalSelectCriteria(3) = (/5.d0, 25.d0, 30.d0/) ! global selection criteria
 69  logical           :: separateSelectCriteria = .false. ! apply separate selection criteria: sea/inland waters; insitu/satellite
 70  real(8)           :: inlandWaterSelectCriteriaSatData(3) = (/5.d0, 25.d0, 30.d0/) ! inland water, satellite selection criteria
 71  real(8)           :: inlandWaterSelectCriteriaInsitu(3)  = (/5.d0, 25.d0, 30.d0/) ! inland water, insitu selection criteria
 72  real(8)           :: seaWaterSelectCriteriaSatData(3)    = (/5.d0, 25.d0, 30.d0/) ! sea water, satellite selection criteria
 73  real(8)           :: seaWaterSelectCriteriaInsitu(3)     = (/5.d0, 25.d0, 30.d0/) ! sea water, insitu selection criteria
 74  real(4)           :: seaWaterThreshold = 0.1       ! threshold to distinguish inland water from sea water
 75  namelist /namOceanBGcheck/ timeInterpType_nl, numObsBatches, checkWinds, ndaysWinds, timeStepWinds, &
 76                             windForecastLeadtime, minLatNH, maxLatNH, maxLatExceptionNH, nmonthsExceptionNH, &
 77                             monthExceptionNH, minLatSH, maxLatSH, smoothLenghtScale, &
 78                             globalSelectCriteria, separateSelectCriteria, inlandWaterSelectCriteriaSatData, &
 79                             inlandWaterSelectCriteriaInsitu, seaWaterSelectCriteriaSatData, &
 80                             seaWaterSelectCriteriaInsitu, seaWaterThreshold 
 81
 82  character(len=3), parameter :: months(12) = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec']

 86  !----------------------------------------------------------------------------------------
 87  ! ocebg_bgCheckSST
 88  !----------------------------------------------------------------------------------------
 89  subroutine ocebg_bgCheckSST(obsData, dateStamp, columnTrlOnTrlLev, hco)
 90    !
 91    !:Purpose: to compute SST data background Check
 92    !
 93
 94    implicit none
 95
 96    ! Arguments:
 97    type(struct_obs)       , intent(inout)       :: obsData           ! obsSpaceData object
 98    integer                , intent(in)          :: dateStamp         ! date stamp
 99    type(struct_columnData), intent(inout)       :: columnTrlOnTrlLev ! column data on trl levels
100    type(struct_hco)       , intent(in), pointer :: hco               ! horizontal trl grid
101
102    ! Locals:
103    type(struct_gsv)            :: stateVectorFGE        ! state vector containing std B estimation field
104    type(struct_gsv)            :: stateVectorAmplFactor ! state vector for error amplification field
105    real(4), pointer            :: stateVectorAmplFactor_ptr(:,:,:)
106    type(struct_gsv)            :: stateVectorSeaWaterFraction ! statevector for sea water fraction
107    integer                     :: nulnam, ierr, headerIndex, bodyIndex, obsFlag, obsVarno
108    integer                     :: numberObs, numberObsRejected
109    integer                     :: numberObsInsitu, numberObsInsituRejected, codeType  
110    real(8)                     :: OER, OmP, FGE, bgCheck, seaWaterFraction
111    type(struct_columnData)     :: columnFGE, columnSeaWaterFraction
112    real(4), pointer            :: stateVectorFGE_ptr(:,:,:)
113    logical                     :: checkMonth, llok
114    integer                     :: lonIndex, latIndex, monthIndex, exceptMonthIndex
115
116    call msg('ocebg_bgCheckSST', 'performing background check for the SST data...')
117
118    ! get mpi topology
119    call mmpi_setup_lonbands(hco%ni, lonPerPE, lonPerPEmax, myLonBeg, myLonEnd)
120    call mmpi_setup_latbands(hco%nj, latPerPE, latPerPEmax, myLatBeg, myLatEnd)
121    
122    ! Read the namelist
123    if (.not. utl_isNamelistPresent('namOceanBGcheck','./flnml')) then
124      if (mmpi_myid == 0) then
125        call msg('ocebg_bgCheckSST', 'namOceanBGcheck is missing in the namelist.')
126        call msg('ocebg_bgCheckSST', 'the default values will be taken.')
127      end if
128    else
129      ! reading namelist variables
130      nulnam = 0
131      ierr = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
132      read(nulnam, nml = namOceanBGcheck, iostat = ierr)
133      if (ierr /= 0) call utl_abort('ocebg_bgCheckSST: Error reading namelist')
134      ierr = fclos(nulnam)
135      if (nmonthsExceptionNH /=0) then
136        call utl_abort('ocebg_bgCheckSST: check namOceanBGcheck namelist section: nmonthsExceptionNH should be removed')
137      end if
138      do monthIndex = 1, 12
139        if (monthExceptionNH(monthIndex) == '   ') exit
140        nmonthsExceptionNH = nmonthsExceptionNH + 1
141      end do
142    end if
143    call msg('ocebg_bgCheckSST', 'interpolation type: '//timeInterpType_nl)
144    call msg('ocebg_bgCheckSST', 'number obs batches: '//str(numObsBatches))
145    call msg('ocebg_bgCheckSST', 'check winds to detect tropical storms (TS): '//str(checkWinds))
146
147    if (separateSelectCriteria) then
148      call msg('ocebg_bgCheckSST', 'a separate selection criteria will be applied...')
149      write(*,'(a,3f7.2)') 'ocebg_bgCheckSST: inland water selection criteria for INSITU data rejection set to: ', &
150                           inlandWaterSelectCriteriaInsitu(:)
151      write(*,'(a,3f7.2)') 'ocebg_bgCheckSST: inland water selection criteria for SATELLITE data rejection set to: ', &
152                           inlandWaterSelectCriteriaSatData(:)
153      write(*,'(a,3f7.2)') 'ocebg_bgCheckSST: sea water selection criteria for INSITU data rejection set to: ', &
154                           seaWaterSelectCriteriaInsitu(:)
155      write(*,'(a,3f7.2)') 'ocebg_bgCheckSST: sea water selection criteria for SATELLITE data rejection set to: ', &
156                           seaWaterSelectCriteriaSatData(:)
157      call msg('ocebg_bgCheckSST', 'sea water fraction threshold is set to: '//str(seaWaterThreshold))
158      ! read sea water fraction
159      call gsv_allocate(stateVectorSeaWaterFraction, 1, hco, columnTrlOnTrlLev%vco, dataKind_opt = 4, &
160                        datestamp_opt = -1, mpi_local_opt = .true., &
161                        varNames_opt = (/'VF'/), hInterpolateDegree_opt ='LINEAR')
162      call gio_readFromFile(stateVectorSeaWaterFraction, './seaice_analysis', ' ','A', &
163                            unitConversion_opt=.false., containsFullField_opt=.true.)
164      ! Convert sea water fraction stateVector to column object
165      call col_setVco(columnSeaWaterFraction, col_getVco(columnTrlOnTrlLev))
166      call col_allocate(columnSeaWaterFraction, col_getNumCol(columnTrlOnTrlLev), varNames_opt = (/'VF'/))
167      call s2c_nl(stateVectorSeaWaterFraction, obsData, columnSeaWaterFraction, hco, varName_opt = 'VF', &
168                  timeInterpType = timeInterpType_nl, moveObsAtPole_opt = .true., &
169                  numObsBatches_opt = numObsBatches, dealloc_opt = .true.)
170    else
171      write(*,'(a,3f7.2)') 'ocebg_bgCheckSST: global selection criteria for data rejection is set to: ', globalSelectCriteria(:)
172    end if
173
174    ! Read First Guess Error (FGE) and put it into stateVector
175    call gsv_allocate(stateVectorFGE, 1, hco, columnTrlOnTrlLev%vco, dataKind_opt = 4, &
176                      hInterpolateDegree_opt = 'NEAREST', &
177                      datestamp_opt = -1, mpi_local_opt = .true., varNames_opt = (/'TM'/))
178    call gio_readFromFile(stateVectorFGE, './bgstddev', 'STDDEV', 'X', &
179                          unitConversion_opt=.false., containsFullField_opt=.true.)
180    if (checkWinds) then
181      call utl_tmg_start(123, '--checkWindsForSST') 
182      call msg('ocebg_bgCheckSST', 'looking for tropical storms...')
183      call msg('ocebg_bgCheckSST', 'number of days with available winds in the input winds file: '//str(ndaysWinds))
184      call msg('ocebg_bgCheckSST', 'winds are provided every: '//str(timeStepWinds)//' hours')
185      call msg('ocebg_bgCheckSST', 'wind forecast lead time: '//str(windForecastLeadtime)//' hours')
186      do exceptMonthIndex = 1, nmonthsExceptionNH
187        checkMonth = .False.
188        loop_month: do monthIndex = 1, 12
189          if (monthExceptionNH(exceptMonthIndex) == months(monthIndex)) then
190            checkMonth = .True.
191            exit loop_month
192          end if
193        end do loop_month
194        if (.not. checkMonth) then
195          write(*,*) 'ocebg_bgCheckSST: month should be one of these: ',months(:)
196          call utl_abort('ocebg_bgCheckSST: unknown month '//monthExceptionNH(exceptMonthIndex))
197        end if
198      end do
199      
200      ! amplification error field state vector  
201      call gsv_allocate(stateVectorAmplFactor, 1, hco, columnTrlOnTrlLev%vco, dataKind_opt = 4, &
202                        hInterpolateDegree_opt = 'LINEAR', datestamp_opt = dateStamp, &
203                        mpi_local_opt = .true., varNames_opt = (/'TM'/))
204      call gsv_getField(stateVectorAmplFactor, stateVectorAmplFactor_ptr)
205      stateVectorAmplFactor_ptr(myLonBeg:myLonEnd,myLatBeg:myLatEnd,1) = 1.0d0
206
207      call ocebg_getFGEamplification(stateVectorAmplFactor, dateStamp, hco)
208      
209      ! FGE state vector
210      call gsv_getField(stateVectorFGE, stateVectorFGE_ptr)
211
212      ! Apply tropical storm correction to the FGE field:
213      do latIndex = myLatBeg, myLatEnd
214        do lonIndex = myLonBeg, myLonEnd
215          stateVectorFGE_ptr(lonIndex, latIndex, 1) = stateVectorFGE_ptr(lonIndex, latIndex, 1) * &
216                                                      stateVectorAmplFactor_ptr(lonIndex, latIndex, 1)
217        end do
218      end do      
219      call gsv_deallocate(stateVectorAmplFactor)
220      call utl_tmg_stop(123)
221    end if
222
223    ! Convert FGE stateVector to column object
224    call col_setVco(columnFGE, col_getVco(columnTrlOnTrlLev))
225    call col_allocate(columnFGE, col_getNumCol(columnTrlOnTrlLev), varNames_opt = (/'TM'/))
226    call s2c_nl(stateVectorFGE, obsData, columnFGE, hco, varName_opt = 'TM', &
227                timeInterpType = timeInterpType_nl, moveObsAtPole_opt = .true., &
228                numObsBatches_opt = numObsBatches, dealloc_opt = .true.)
229
230    numberObs = 0
231    numberObsRejected = 0
232    numberObsInsitu = 0
233    numberObsInsituRejected = 0
234
235    do headerIndex = 1, obs_numheader(obsData)
236
237      bodyIndex = obs_headElem_i(obsData, obs_rln, headerIndex)
238      obsVarno  = obs_bodyElem_i(obsData, obs_vnm, bodyIndex)
239      llok = (obs_bodyElem_i(obsData, obs_ass, bodyIndex) == obs_assimilated)
240      if (llok) then
241        if (obsVarno == bufr_sst) then
242
243	  FGE = col_getElem(columnFGE, 1, headerIndex, 'TM')
244	  OmP = obs_bodyElem_r(obsData, OBS_OMP, bodyIndex)
245          OER = obs_bodyElem_r(obsData, OBS_OER, bodyIndex)
246	  codeType = obs_headElem_i(obsData, obs_ity, headerIndex)
247
248	  if (FGE /= MPC_missingValue_R8 .and. OmP /= MPC_missingValue_R8) then
249
250	    numberObs = numberObs + 1
251	    if (codeType /= codtyp_get_codtyp('satob')) numberObsInsitu = numberObsInsitu + 1
252	    call obs_bodySet_r(obsData, OBS_HPHT, bodyIndex, FGE)
253	    bgCheck = (OmP)**2 / (FGE**2 + OER**2)
254
255            if (separateSelectCriteria) then
256              seaWaterFraction = col_getElem(columnSeaWaterFraction, 1, headerIndex, 'VF')
257              if (seaWaterFraction <= seaWaterThreshold) then
258                if(codeType == codtyp_get_codtyp('satob')) then
259                  obsFlag = ocebg_setFlag(obsVarno, bgCheck, inlandWaterSelectCriteriaSatData)
260                else
261                  obsFlag = ocebg_setFlag(obsVarno, bgCheck, inlandWaterSelectCriteriaInsitu)
262                end if
263              else
264                if(codeType == codtyp_get_codtyp('satob')) then
265                  obsFlag = ocebg_setFlag(obsVarno, bgCheck, seaWaterSelectCriteriaSatData)
266                else
267                  obsFlag = ocebg_setFlag(obsVarno, bgCheck, seaWaterSelectCriteriaInsitu)
268                end if 
269              end if
270            else
271              obsFlag = ocebg_setFlag(obsVarno, bgCheck, globalSelectCriteria)
272            end if
273	
274            if (obsFlag >= 2) then
275              numberObsRejected = numberObsRejected + 1
276	      if (codeType /= codtyp_get_codtyp('satob')) numberObsInsituRejected = numberObsInsituRejected + 1
277	      write(*,'(i10,a,i5,4f10.4,i5)') numberObsRejected, &
278                                              ', sensor, codtype, lon, lat, obs.value, OmP, flag: '&
279                                              //obs_elem_c(obsData, 'STID' , headerIndex)//' data: ', codeType, &
280                                              obs_headElem_r(obsData, obs_lon, headerIndex) * MPC_DEGREES_PER_RADIAN_R8,&
281                                              obs_headElem_r(obsData, obs_lat, headerIndex) * MPC_DEGREES_PER_RADIAN_R8,&
282                                              obs_bodyElem_r(obsData, obs_var, bodyIndex), OmP, obsFlag 
283            end if
284
285	    ! update background check flags based on bgCheck
286            ! (element flags + global header flags)
287	    if (obsFlag == 1) then
288              call obs_bodySet_i(obsData, obs_flg, bodyIndex  , ibset(obs_bodyElem_i(obsData, obs_flg, bodyIndex)  , 13))
289            else if (obsFlag == 2) then
290              call obs_bodySet_i(obsData, obs_flg, bodyIndex  , ibset(obs_bodyElem_i(obsData, obs_flg, bodyIndex)  , 14))
291              call obs_bodySet_i(obsData, obs_flg, bodyIndex  , ibset(obs_bodyElem_i(obsData, obs_flg, bodyIndex)  , 16))
292              call obs_bodySet_i(obsData, obs_flg, bodyIndex  , ibset(obs_bodyElem_i(obsData, obs_flg, bodyIndex)  , 09))
293              call obs_headSet_i(obsData, obs_st1, headerIndex, ibset(obs_headElem_i(obsData, obs_st1, headerIndex), 06))
294            else if (obsFlag == 3) then
295              call obs_bodySet_i(obsData, obs_flg, bodyIndex  , ibset(obs_bodyElem_i(obsData, obs_flg, bodyIndex)  , 15))
296              call obs_bodySet_i(obsData, obs_flg, bodyIndex  , ibset(obs_bodyElem_i(obsData, obs_flg, bodyIndex)  , 16))
297              call obs_bodySet_i(obsData, obs_flg, bodyIndex  , ibset(obs_bodyElem_i(obsData, obs_flg, bodyIndex)  , 09))
298              call obs_headSet_i(obsData, obs_st1, headerIndex, ibset(obs_headElem_i(obsData, obs_st1, headerIndex), 06))
299            end if
300
301          end if
302	end if
303      end if
304
305    end do
306
307    if (numberObs > 0) then
308      write(*,*)
309      call msg('ocebg_bgCheckSST', 'background check of SST (TM) data is computed')
310      write(*,*) '***************************************************************************************'
311      write(*,'(a, i7,a,i7,a)') 'ocebg_bgCheckSST: total ', numberObsRejected, ' observations out of (ALL) ', numberObs,' rejected'
312      write(*,'(a, i7,a,i7,a)') 'ocebg_bgCheckSST: where ', numberObsInsituRejected, ' insitu observations out of ', &
313                                numberObsInsitu,' insitu obs. rejected'
314      write(*,*) '***************************************************************************************'
315      write(*,*)
316    end if
317
318    call gsv_deallocate(stateVectorFGE)
319    call col_deallocate(columnFGE)
320    if (separateSelectCriteria) then
321      call col_deallocate(columnSeaWaterFraction)
322      call gsv_deallocate(stateVectorSeaWaterFraction)
323    end if      
324
325  end subroutine ocebg_bgCheckSST

327  !----------------------------------------------------------------------------------------
328  ! ocebg_bgCheckSeaIce
329  !----------------------------------------------------------------------------------------
330  subroutine ocebg_bgCheckSeaIce(obsData)
331    !
332    !:Purpose: Compute sea ice data background check.
333    !           The rms of the difference between the observations and
334    !           the background values is calculated over a "swath" of data.
335    !           For satellite observations, this is based on a single passage
336    !           of the satellite over the domain.
337    !           A flag is then set for all observations in the swath
338    !           if the rms value is larger than the threshold
339    !           specified for the observation type.
340    !
341
342    implicit none
343
344    ! Arguments:
345    type(struct_obs), intent(inout) :: obsData           ! obsSpaceData object
346
347    ! Locals:
348    integer              :: nulnam, ierr
349    integer              :: headerIndex, bodyIndex, stationIndex, bodyCount
350    integer              :: obsChid, obsDate, obsTime, obsFlag
351    integer              :: obsDateStamp
352    integer, allocatable :: numberObs(:), bodyIndexList(:,:)
353    integer, allocatable :: minDateStamp(:), maxDateStamp(:), swathID(:)
354    real, allocatable    :: rmsDiff(:)
355    integer              :: swathIndex, nSwath
356    real(8)              :: OmP, deltaHours
357    integer              :: numberObsRejected
358    character(len=12)    :: cstnid
359    integer, external    :: newdate
360    integer              :: imode, prntdate, prnttime
361    integer, parameter   :: numStationMax = 11              ! maximum number of 'idStation' values
362    
363    ! Namelist variables: (local)
364    integer           :: numStation = MPC_missingValue_INT  ! MUST NOT BE INCLUDED IN NAMELIST!
365    character(len=12) :: idStation(numStationMax) = 'null'  ! list of obsSpaceData 'idStation' values to consider
366    real              :: OmpRmsdThresh(numStationMax) = 0.0 ! rejection threshold applied to RMS of O-P for entire swath
367    integer           :: maxSwath = 10                      ! maximum number of swaths
368    integer           :: maxPerSwath = 200000               ! maximum number of data per swath
369    namelist /namIceBGcheck/ numStation, idStation, OmpRmsdThresh, maxSwath, maxPerSwath
370
371    call msg('ocebg_bgCheckSeaIce', 'performing background check for the SeaIce data...')
372 
373    ! Read the namelist
374    if (.not. utl_isNamelistPresent('namIceBGcheck','./flnml')) then
375      if (mmpi_myid == 0) then
376        call msg('ocebg_bgCheckSeaIce', 'namIceBGcheck is missing in the namelist.')
377        call msg('ocebg_bgCheckSeaIce', 'the default values will be taken.')
378      end if
379    else
380      ! reading namelist variables
381      nulnam = 0
382      ierr = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
383      read(nulnam, nml = namIceBGcheck, iostat = ierr)
384      if (ierr /= 0) call utl_abort('ocebg_bgCheckSeaIce: Error reading namelist')
385      ierr = fclos(nulnam)
386      if (numStation /= MPC_missingValue_INT) then
387        call utl_abort('ocebg_bgCheckSeaIce: check namIceBGcheck namelist section: numStation should be removed')
388      end if
389      numStation = 0
390      do stationIndex = 1, numStationMax
391        if (trim(idStation(stationIndex)) == 'null') exit
392        numStation = numStation + 1
393      end do
394    end if
395    write(*, nml = namIceBGcheck)
396
397    allocate(numberObs(maxSwath), bodyIndexList(maxPerSwath,maxSwath))
398    allocate(minDateStamp(maxSwath), maxDateStamp(maxSwath), swathID(maxSwath))
399    allocate(rmsDiff(maxSwath))
400
401    STATION: do stationIndex = 1, numStation
402
403      call msg('ocebg_bgCheckSeaIce', 'doing idStation: '//idStation(stationIndex))
404
405      numberObs(:) = 0
406      rmsDiff(:) = 0.0
407      minDateStamp(:) = tim_getDatestamp()
408      maxDateStamp(:) = tim_getDatestamp()
409      nSwath = 0
410
411      HEADER: do headerIndex = 1, obs_numheader(obsData)
412
413        cstnid = obs_elem_c(obsData, 'STID' , headerIndex )
414        bodyIndex = obs_headElem_i(obsData, obs_rln, headerIndex)
415
416        obsFlag = obs_bodyElem_i(obsData, OBS_FLG, bodyIndex)
417
418        if ( (cstnid /= idStation(stationIndex)) .or. &
419             (obsFlag /= 0 .and. cstnid /= 'CIS_REGIONAL') ) cycle HEADER
420
421        obsChid = obs_headElem_i(obsData, obs_chid, headerIndex)
422        obsDate = obs_headElem_i(obsData, OBS_DAT, headerIndex)
423        obsTime = obs_headElem_i(obsData, OBS_ETM, headerIndex)
424
425        imode = 3 ! printable to stamp
426        ierr = newdate(obsDateStamp, obsDate, obsTime*10000, imode)
427
428        do swathIndex=1, nSwath
429
430          if(obsDateStamp <= maxDateStamp(swathIndex) .and. &
431             obsDateStamp >= minDateStamp(swathIndex) .and. &
432             obsChid == swathID(swathIndex)) exit
433
434          call difdatr(obsDateStamp, minDateStamp(swathIndex), deltaHours)
435            
436          if(abs(deltaHours) <= 0.5d0 .and. obsChid == swathID(swathIndex)) exit
437
438        end do
439
440        nSwath = max(nSwath, swathIndex)
441
442        if(swathIndex <= maxSwath) then
443
444          if(numberObs(swathIndex) == 0) then
445            minDateStamp(swathIndex) = obsDateStamp
446            maxDateStamp(swathIndex) = obsDateStamp
447          else
448            call difdatr(obsDateStamp, minDateStamp(swathIndex), deltaHours)
449            if(deltaHours < 0.0d0) minDateStamp(swathIndex) = obsDateStamp
450            call difdatr(obsDateStamp, maxDateStamp(swathIndex), deltaHours)
451            if(deltaHours > 0.0d0) maxDateStamp(swathIndex) = obsDateStamp
452          end if
453
454          swathID(swathIndex) = obsChid
455
456          OmP = obs_bodyElem_r(obsData, OBS_OMP, bodyIndex)
457
458          if (OmP /= MPC_missingValue_R8) then
459
460            numberObs(swathIndex) = numberObs(swathIndex) + 1
461            rmsDiff(swathIndex) = rmsDiff(swathIndex) + (OmP)**2
462            if (numberObs(swathIndex) <= maxPerSwath) then
463              bodyIndexList(numberObs(swathIndex), swathIndex) = bodyIndex
464            else
465              call utl_abort('ocebg_bgCheckSeaIce: Increase maxPerSwath')
466            end if
467
468          end if
469
470        else
471
472          call utl_abort('ocebg_bgCheckSeaIce: Increase maxSwath')
473
474        end if
475
476      end do HEADER
477
478      SWATH: do swathIndex = 1, maxSwath
479
480        if (numberObs(swathIndex) == 0) cycle SWATH
481
482        rmsDiff(swathIndex) = sqrt(rmsDiff(swathIndex)/numberObs(swathIndex))
483
484        call msg('ocebg_bgCheckSeaIce', 'swathIndex = '//str(swathIndex))
485        call msg('ocebg_bgCheckSeaIce', 'Timespan:')
486        imode = -3 ! stamp to printable
487        ierr = newdate(minDateStamp(swathIndex), prntdate, prnttime, imode)
488        call msg('ocebg_bgCheckSeaIce', 'From: '//str(prntdate)//str(prnttime/100))
489        ierr = newdate(maxDateStamp(swathIndex), prntdate, prnttime, imode)
490        call msg('ocebg_bgCheckSeaIce', 'To  : '//str(prntdate)//str(prnttime/100))
491
492        write(*,'(a,f10.4)') 'ocebg_bgCheckSeaIce: RMSD = ', rmsDiff(swathIndex)
493        write(*,'(a,i10)') 'ocebg_bgCheckSeaIce: nobs = ', numberObs(swathIndex)
494
495        if (rmsDiff(swathIndex) > OmpRmsdThresh(stationIndex)) then
496
497          numberObsRejected = 0
498          do bodyCount = 1, numberObs(swathIndex)
499            numberObsRejected = numberObsRejected + 1
500            ! update background check flag
501            call obs_bodySet_i(obsData, obs_flg, bodyIndexList(bodyCount,swathIndex), ibset(obs_bodyElem_i(obsData, obs_flg, bodyIndexList(bodyCount,swathIndex)), 10))
502          end do
503
504          write(*,'(a,i7,a)')'ocebg_bgCheckSeaIce: ********** reject: ', numberObsRejected, &
505                                    ' '//idStation(stationIndex)//' data'
506
507        end if
508
509      end do SWATH
510
511    end do STATION
512
513    deallocate(numberObs, bodyIndexList)
514    deallocate(minDateStamp, maxDateStamp, swathID)
515    deallocate(rmsDiff)
516
517  end subroutine ocebg_bgCheckSeaIce

551  !--------------------------------------------------------------------------
552  ! ocebg_getFGEamplification
553  !--------------------------------------------------------------------------
554  subroutine ocebg_getFGEamplification(stateVectorAmplFactor, dateStamp, hco)
555    !
556    !:Purpose: Read wind speed fields for the last four days.
557    !           In the operations: 
558    !           The background error used during the background check is then
559    !           amplified in those regions by a factor that varies from 1,
560    !           where the maximum wind speed is 21m/s or less, to 12,
561    !           where the maximum wind speed is 24m/s or more.
562    !           The factor is then filtered to produce a smoothly varying
563    !           field. This amplified background error is used only to
564    !           perform the background check. 
565
566    implicit none
567
568    ! Arguments
569    type(struct_gsv), intent(inout)       :: stateVectorAmplFactor ! state vector to save amplification factor
570    integer         , intent(in)          :: dateStamp             ! date stamp
571    type(struct_hco), intent(in), pointer :: hco                   ! horizontal trl grid
572
573    ! locals
574    type(struct_gsv)          :: stateVector         ! state vector for surface winds
575    type(struct_vco), pointer :: vco_winds           ! vertical grid structure for winds
576    real(4)         , pointer :: uu_ptr4d(:,:,:,:)   ! pointer to get UU wind component
577    real(4)         , pointer :: vv_ptr4d(:,:,:,:)   ! pointer to get VV wind component
578    integer                   :: dataStampList(ndaysWinds * 24 / timeStepWinds) ! datastamp list for wind fields
579    real(4)                   :: windSpeed
580    integer                   :: hour, day, monthNumber
581    integer                   :: yyyy, ndays, timeStepIndex
582    real(8)                   :: deltaT, lat
583    integer                   :: lonIndex, latIndex, monthIndex
584    real(4)         , pointer :: stateVectorAmplFactor_ptr(:,:,:)
585    real(8)                   :: amplFactor
586
587    nullify(vco_winds)
588    
589    ! looking for the earliest valid time:
590    deltaT = - real(ndaysWinds * 24 - windForecastLeadtime)
591    call incdatr(dataStampList(1), dateStamp, deltaT)
592
593    ! scanning the datastamps up from the earliest to the last valid time
594    deltaT = real(timeStepWinds)
595    do timeStepIndex = 2, ndaysWinds * 24 / timeStepWinds
596      call incdatr(dataStampList(timeStepIndex), dataStampList(timeStepIndex - 1), deltaT)
597    end do
598
599    call vco_SetupFromFile(vco_winds,'./winds')
600    call gsv_allocate(stateVector, ndaysWinds * 24 / timeStepWinds, hco, vco_winds, &
601                      dateStampList_opt = dataStampList, dataKind_opt = 4, &
602                      varNames_opt=(/'UU','VV'/), mpi_local_opt=.true., &
603                      hInterpolateDegree_opt='LINEAR')
604      
605    call msg('ocebg_getFGEamplification', 'reading wind speed fields...')
606    do timeStepIndex = 1, ndaysWinds * 24 / timeStepWinds
607      call tim_dateStampToYYYYMMDDHH(dataStampList(timeStepIndex), hour, day, monthNumber, &
608                                     ndays, yyyy, verbose_opt = .False.)
609      call msg('ocebg_getFGEamplification', ' '//str(timeStepIndex)//str(dataStampList(timeStepIndex))//str(yyyy)//str(monthNumber)//str(day)//str(hour))
610      call gio_readFromFile(stateVector, './winds', ' ', ' ', stepIndex_opt = timeStepIndex, &
611                            containsFullField_opt=.true.)
612    end do
613
614    call gsv_getField(stateVector, uu_ptr4d, 'UU')
615    call gsv_getField(stateVector, vv_ptr4d, 'VV')
616    call gsv_getField(stateVectorAmplFactor, stateVectorAmplFactor_ptr)
617
618    write(*,*)
619    call msg('ocebg_getFGEamplification', 'detecting tropical storms (TS)...')
620
621    time_loop: do timeStepIndex = 1, ndaysWinds * 24 / timeStepWinds
622
623      call tim_dateStampToYYYYMMDDHH(dataStampList(timeStepIndex), hour, day, monthNumber, &
624                                     ndays, yyyy, verbose_opt = .False.)
625      do monthIndex = 1, nmonthsExceptionNH
626        if (months(monthNumber) == monthExceptionNH(monthIndex)) then
627          maxLatNH = maxLatExceptionNH
628          call msg('ocebg_getFGEamplification', 'TS is allowed to penetrate up to '//str(maxLatNH) &
629                     //' degrees North; current month: '//months(monthNumber))
630        end if
631      end do
632
633      do latIndex = myLatBeg, myLatEnd
634        do lonIndex = myLonBeg, myLonEnd
635
636          lat = real(hco%lat2d_4(lonIndex, latIndex), 8) * MPC_DEGREES_PER_RADIAN_R8
637          if ((lat > minLatNH .and. lat < maxLatNH) .or. (lat > minLatSH .and. lat < maxLatSH)) then
638            windSpeed = sqrt(uu_ptr4d(lonIndex, latIndex, 1, timeStepIndex) * &
639                             uu_ptr4d(lonIndex, latIndex, 1, timeStepIndex) + &
640                             vv_ptr4d(lonIndex, latIndex, 1, timeStepIndex) * &
641                             vv_ptr4d(lonIndex, latIndex, 1, timeStepIndex))
642            amplFactor = max(1.0d0, min(4.0d0 * max(0.0d0,(windSpeed - 20.75d0)), 12.0d0))
643            ! stateVectorAmplFactor_ptr may already be assigned with a value > 1.0 in a previous state 
644            if(amplFactor > stateVectorAmplFactor_ptr(lonIndex, latIndex, 1)) then
645              stateVectorAmplFactor_ptr(lonIndex, latIndex, 1) = amplFactor
646            end if
647          endif
648
649        end do
650      end do
651
652    end do time_loop
653
654    call gsv_deallocate(stateVector)
655
656    call gio_writeToFile(stateVectorAmplFactor, './amplification', 'ORIG')      
657    call gsv_smoothHorizontal(stateVectorAmplFactor, smoothLenghtScale)
658    call gio_writeToFile(stateVectorAmplFactor, './amplification', 'SMOOTH')        
659
660  end subroutine ocebg_getFGEamplification
