 54  subroutine tim_readNml()
 55    !
 56    !:Purpose: Read the namelist block NAMTIME.
 57    !
 58    !:Namelist parameters:
 59    !         :dstepobs:    time step (hrs) between successive trial fields 
 60    !                       for use in OmP determation. Set to dwindowsize for
 61    !                       single trial field, i.e. use of 3dvar instead of 3dvar-FGAT.
 62    !                       nstepobs = number of trial fields 
 63    !         :dstepobsinc: time step (hrs) between obs groupings in time. Set to
 64    !                       dwindowsize for use of a single obs group.
 65    !                       nstepobsinc = number of obs time intervals 
 66    !         :dwindowsize: Time window size (hrs).
 67    !
 68    !:Comment:
 69    !
 70    !   Provided dates and number of provided trial field files must be
 71    !   consistent with nstepobs, dstepobs and dwindowsize with reference datestamp
 72    !   corresponding to the date of the middle trial field file.
 73    !
 74    implicit none
 75
 76    ! locals
 77    integer :: nulnam, ierr, fnom, fclos
 78    logical, save :: firstCall = .true.
 79
 80    ! namelist variables:
 81    real(8) :: dstepobs      ! time step length for background state (in hours)
 82    real(8) :: dstepobsinc   ! time step length for increment and/or B matrix (in hours)
 83    real(8) :: dwindowsize   ! length of assimilation window (in hours)
 84    character(len=6) :: referencetime  ! location of 'date' within the window: 'middle' or 'start'
 85    logical :: fullyUseExtremeTimeBins ! choose to use full-size bins at both ends of window (usually only half size)
 86
 87    NAMELIST /NAMTIME/dstepobs, dstepobsinc, dwindowsize, referencetime, fullyUseExtremeTimeBins
 88
 89    if (.not.firstCall) then
 90      write(*,*) 'tim_readNml: already initialized, just return'
 91      return
 92    else
 93      firstCall = .false.
 94    end if
 95
 96    ! Set default values for namelist variables
 97    dstepobs       = 6.0d0
 98    dstepobsinc    = 6.0d0      
 99    dwindowsize    = 6.0d0     
100    referenceTime = 'middle'
101    fullyUseExtremeTimeBins = .false.
102
103    ! Read the namelist
104    nulnam = 0
105    ierr = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
106    read(nulnam, nml=namtime, iostat=ierr)
107    if (ierr /= 0) call utl_abort('tim_readNml: Error reading namelist')
108    if (mmpi_myid == 0) write(*,nml=namtime)
109    ierr = fclos(nulnam)
110
111    ! Set the module variables for timestep length, number of timesteps and window length
112    tim_dstepobs      = dstepobs
113    tim_dstepobsinc   = dstepobsinc
114    tim_windowsize    = dwindowsize
115    tim_referencetime = referenceTime
116    tim_fullyUseExtremeTimeBins = fullyUseExtremeTimeBins
117
118    if (tim_fullyUseExtremeTimeBins .and. tim_referencetime=="middle") then
119      write(*,*) "Warning: tim_fullyUseExtremeTimeBins==.true. and tim_referencetime=='middle' is a non-standard combination"
120      write(*,*) "Is it really what you want ?"
121    end if
122
123    if (dstepobs > dwindowsize) then
124      if (mmpi_myid == 0) write(*,*) 'tim_readNml: dstepobs>dwindowsize. Reset to dwindowsize value.'
125      tim_dstepobs = tim_windowsize 
126    end if
127    if (dstepobsinc > dwindowsize) then
128      if (mmpi_myid == 0) write(*,*) 'tim_readNml: dstepobsinc>dwindowsize. Reset to dwindowsize value.'
129      tim_dstepobsinc = tim_windowsize 
130    end if 
131    if (tim_referencetime == "middle") then
132      tim_nstepobs    = 2*nint(((tim_windowsize - tim_dstepobs)/2.d0)/tim_dstepobs) + 1
133      tim_nstepobsinc = 2*nint(((tim_windowsize - tim_dstepobsinc)/2.d0)/tim_dstepobsinc) + 1
134    end if
135
136    if (trim(tim_referencetime) == "start") then
137      tim_nstepobs = max(nint(tim_windowsize/tim_dstepobs), 1)
138      tim_nstepobsinc = max(nint(tim_windowsize/tim_dstepobsinc), 1)
139    end if
140
141  end subroutine tim_readNml

144  subroutine tim_getDateStampFromEnvVar(dateStamp)
145    !
146    !:Purpose: Determine the date from the environment variable MIDAS_DATE.
147    !
148    implicit none
149
150    ! arguments:
151    integer, intent(inout) :: dateStamp
152
153    ! locals:
154    integer    :: lengthValidDateStr, status, datePrint, timePrint, imode, ierr
155    integer(8) :: dateTimePrint
156    character(len=256) :: validDateStr
157    integer    :: newdate
158
159    status = 0
160    call get_environment_variable('MIDAS_DATE',validDateStr,lengthValidDateStr,status,.true.)
161
162    if (status > 1) then
163      call utl_abort('tim_getDateStampFromEnvVar: Problem when getting the environment variable MIDAS_DATE')
164    end if
165    if (status == 1) then
166      write(*,*) 'tim_getDateStampFromEnvVar: WARNING: The environment variable MIDAS_DATE has not been detected!'
167      !call utl_abort('tim_getDateStampFromEnvVar: The environment variable MIDAS_DATE has not been detected!')
168      return
169    end if
170
171    write(*,*)
172    write(*,*) 'tim_getDateStampFromEnvVar: The environment variable MIDAS_DATE has correctly been detected'
173
174    ! convert string to long integer
175    read (validDateStr,*) dateTimePrint
176
177    ! split dateTime long integer into separate date and time values
178    if (lengthValidDateStr == 10) then
179      datePrint = dateTimePrint/100
180      timePrint = (dateTimePrint - datePrint*100) * 1000000
181    else if (lengthValidDateStr == 12) then
182      datePrint = dateTimePrint/10000
183      timePrint = (dateTimePrint - datePrint*10000) * 10000
184    else if (lengthValidDateStr == 14) then
185      datePrint = dateTimePrint/1000000
186      timePrint = (dateTimePrint - datePrint*1000000) * 100
187    else if (lengthValidDateStr == 16) then
188      datePrint = dateTimePrint/100000000
189      timePrint = (dateTimePrint - datePrint*100000000)
190    else
191      write(*,*) 'length of MIDAS_DATE = ', lengthValidDateStr
192      call utl_abort('tim_getDateStampFromEnvVar: Unexpected length of variable MIDAS_DATE')
193    end if
194
195    ! convert to CMC dateStamp
196    imode = 3 ! printable to stamp
197    ierr = newdate(datestamp, datePrint, timePrint, imode)
198
199    write(*,*) 'tim_getDateStampFromEnvVar: envVar, validDate, dateStamp = ', trim(validDateStr), dateTimePrint, dateStamp
200
201  end subroutine tim_getDateStampFromEnvVar

204  subroutine tim_setup(fileNameForDate_opt)
205    !
206    !:Purpose: Setup of obs time window size and related trial field 
207    !           time step for OmP determination. 
208    !
209    implicit none
210
211    ! arguments
212    character(len=*), optional :: fileNameForDate_opt
213
214    ! locals
215    integer :: ierr, newdate, imode, prntdate, prnttime
216    integer :: dateStampEnvVar
217
218    call tim_readNml()
219
220    if (initialized) then
221      write(*,*) 'tim_setup: already initialized, just return'
222      return
223    end if
224
225    ! First try to set dateStamp from MIDAS_DATE
226    dateStampEnvVar = 0
227    call tim_getDateStampFromEnvVar(dateStampEnvVar)
228
229    ! Possibly set the datestamp (except when set later from burp files)
230    if (dateStampEnvVar /= 0) then
231      write(*,*) 'tim_setup: ====================================================='
232      write(*,*) 'tim_setup: DATESTAMP set by value in supplied MIDAS_DATE'
233      write(*,*) 'tim_setup: ====================================================='
234      dateStamp = dateStampEnvVar
235      imode = -3 ! stamp to printable
236      ierr = newdate(datestamp, prntdate, prnttime, imode)
237      write(*,*) 'tim_setup: printdate = ',prntdate
238      write(*,*) 'tim_setup: printtime = ',prnttime
239      write(*,*) 'tim_setup: datestamp = ',datestamp      
240    else if (present(fileNameForDate_opt)) then
241      write(*,*) 'tim_setup: ====================================================='
242      write(*,*) 'tim_setup: DATESTAMP set by value in supplied file'
243      write(*,*) 'tim_setup: ====================================================='
244      datestamp = tim_getDatestampFromFile(fileNameForDate_opt)
245      imode = -3 ! stamp to printable
246      ierr = newdate(datestamp, prntdate, prnttime, imode)
247      write(*,*) 'tim_setup: printdate = ',prntdate
248      write(*,*) 'tim_setup: printtime = ',prnttime
249      write(*,*) 'tim_setup: datestamp = ',datestamp
250    else
251      write(*,*) 'tim_setup: =========================================================='
252      write(*,*) 'tim_setup: DATESTAMP not set in this subroutine, use tim_setDateStamp'
253      write(*,*) 'tim_setup: =========================================================='      
254    end if
255
256    if (mmpi_myid == 0) write(*,*) 'tim_setup: dobs_windowsize=',tim_windowsize
257    if (mmpi_myid == 0) write(*,*) 'tim_setup: dstepobs       =',tim_dstepobs
258    if (mmpi_myid == 0) write(*,*) 'tim_setup: nstepobs       =',tim_nstepobs
259    if (mmpi_myid == 0) write(*,*) 'tim_setup: dstepobsinc    =',tim_dstepobsinc
260    if (mmpi_myid == 0) write(*,*) 'tim_setup: nstepobsinc    =',tim_nstepobsinc
261    if (mmpi_myid == 0) write(*,*) 'tim_setup: tim_referencetime   =',tim_referencetime
262
263    initialized = .true.
264
265  end subroutine tim_setup

277  function tim_getDatestampFromFile(fileName, varNameForDate_opt) result(dateStamp_out)
278    !
279    !:Purpose: to extract the dateStamp from the supplied file.
280    !
281    implicit none
282
283    ! arguments
284    character(len=*)           :: fileName
285    character(len=*), optional :: varNameForDate_opt
286    integer :: dateStamp_out
287
288    ! locals
289    integer :: nulFile, ierr
290    integer, parameter :: maxNumDates = 2000
291    integer :: numDates, ikeys(maxNumDates), varIndex
292    integer :: fnom, fstouv, fstinl, fstprm, fstfrm, fclos, newdate
293    integer :: prntdate, prnttime, imode, windowIndex, windowsPerDay, dateStamp_tmp
294    logical :: fileExists, foundWindow, foundVarNameInFile
295    real(8) :: leadTimeInHours, windowBegHour, windowEndHour, fileHour, middleHour
296    integer :: ideet, inpas, dateStamp_origin, ini, inj, ink, inbits, idatyp
297    integer :: ip1, ip2, ip3, ig1, ig2, ig3, ig4, iswa, ilng, idltf, iubc
298    integer :: iextra1, iextra2, iextra3
299    character(len=2)  :: typvar
300    character(len=4)  :: nomvar
301    character(len=12) :: etiket
302    character(len=1)  :: grtyp
303
304    if (mmpi_myid == 0) then
305
306      ! Check if file for any date within the analysis window (except the last) exists
307      inquire(file=trim(fileName), exist=fileExists)
308      if (.not.fileExists) then
309        call utl_abort('tim_getDateStampFromFile: file not found '//trim(fileName))
310      end if
311
312      ! Determine variable to use for the date (default is P0)
313      varNameForDate = 'P0'
314      if (present(varNameForDate_opt)) then
315      
316	varNameForDate = trim(varNameForDate_opt)
317        write(*,*) 'tim_getDateStampFromFile: defining dateStamp from the variable = ', varNameForDate
318      
319      ! If P0 not present, look for another suitable variable in the file
320      else if (.not. utl_varNamePresentInFile(varNameForDate,fileName_opt=trim(fileName))) then
321      
322        foundVarNameInFile = .false.
323        do varIndex = 1, vnl_numvarmax
324          varNameForDate = vnl_varNameList(varIndex)
325          ! check if variable is in the file
326          if (.not. utl_varNamePresentInFile(varNameForDate,fileName_opt=trim(fileName))) cycle
327          foundVarNameInFile = .true.
328          exit      
329        end do
330
331        if (.not. foundVarNameInFile) then
332          call utl_abort('tim_getDateStampFromFile: NO variables found in the file!!!')
333        end if
334	
335      end if
336
337      ! Extract the datestamp from the file
338      nulFile = 0
339      ierr = fnom(nulFile,trim(fileName),'RND+OLD+R/O',0)
340      ierr = fstouv(nulFile,'RND+OLD')
341      ierr = fstinl(nulFile,ini,inj,ink,-1,' ',-1,-1,-1,' ', &
342                    trim(varNameForDate),ikeys,numdates,maxnumdates)
343      if (ikeys(1) <= 0) then
344        call utl_abort('tim_getDateStampFromFile: Could not find variable ' //  &
345                       trim(varNameForDate) // ' in the supplied file')
346      end if
347      write(*,*) 'tim_getDateStampFromFile: number of dates found = ', numDates
348      ierr = fstprm(ikeys(1), dateStamp_origin, ideet, inpas, ini, inj, &
349                    ink, inbits, idatyp, ip1, ip2, ip3, &
350                    typvar, nomvar, etiket, grtyp, ig1, ig2, ig3, ig4, &
351                    iswa, ilng, idltf, iubc, iextra1, iextra2, iextra3)
352      leadTimeInHours = real(ideet*inpas,8)/3600.0d0
353      call incdatr(dateStamp_out, dateStamp_origin, leadTimeInHours)
354
355      ierr = fstfrm(nulFile)
356      ierr = fclos(nulFile)
357
358    end if
359
360    call rpn_comm_bcast(dateStamp_out, 1, 'MPI_INTEGER', 0, 'GRID', ierr)
361
362    if (tim_referenceTime == 'middle') then
363      ! Modify date to ensure that it corresponds to the middle of the window
364      ! Note: For this, we have to assume that the date in the file
365      !       does NOT correspond to the final time of the window
366      imode = -3 ! stamp to printable, time is HHMMSShh
367      ierr = newdate(datestamp_out, prntdate, prnttime, imode)
368      fileHour = real(prnttime,8)/1000000.0d0
369      windowsPerDay = nint(24.0d0 / tim_windowsize)
370      foundWindow = .false.
371      window_loop: do windowIndex = 0, windowsPerDay
372        windowBegHour = (real(windowIndex,8) * tim_windowsize) - (tim_windowsize/2.0d0)
373        windowEndHour = (real(windowIndex,8) * tim_windowsize) + (tim_windowsize/2.0d0)
374        if (fileHour >= windowBegHour .and. fileHour < windowEndHour) then
375          foundWindow = .true.
376          middleHour = real(windowIndex,8) * tim_windowsize
377          exit window_loop
378        end if
379      end do window_loop
380
381      if (.not. foundWindow) then
382        write(*,*) 'windowsPerDay, fileHour=', windowsPerDay, fileHour
383        call utl_abort('tim_getDateStampFromFile: could not determine assimilation window position')
384      end if
385
386      ! handle special case when window centered on hour 24
387      if (nint(middleHour) == 24) then
388        ! add 24h to dateStamp and recompute prntdate
389        dateStamp_tmp = dateStamp_out
390        call incdatr(dateStamp_out, dateStamp_tmp, 24.0d0)
391        imode = -3 ! stamp to printable, only prntdate will be used
392        ierr = newdate(datestamp_out, prntdate, prnttime, imode)
393
394        ! subtract 24h from middleHour
395        middleHour = 0.0d0
396      end if
397
398      prnttime = nint(middleHour) * 1000000
399      imode = 3 ! printable to stamp
400      ierr = newdate(datestamp_out, prntdate, prnttime, imode)
401
402    end if
403
404  end function tim_getDateStampFromFile

437  subroutine tim_getStampList(datestamplist, numStep, referenceDateStamp)
438    !
439    !:Purpose: Compute a list of STAMPS corresponding to stepobs time
440    !
441    implicit none
442
443    ! arguments
444    integer, intent(in)  :: numStep ! number of step obs
445    integer, intent(in)  :: referenceDateStamp ! Synoptic time
446    integer, intent(out) :: datestamplist(numStep) 
447
448    ! locals
449    integer :: stepIndex
450    real(8) :: dldelt ! delta time in hours between middle time and each step
451    real(8) :: dtstep ! delta time in hours between step obs
452
453    if (.not. initialized) call utl_abort('tim_getStampList: module not initialized')
454
455    if (referenceDateStamp == -1) then
456
457      if (mmpi_myid == 0) write(*,*) 'tim_getStampList: datestamp is not specified, keep as -1'
458      datestamplist(:) = -1
459
460    else
461
462      if (tim_referencetime == "middle") then
463        if (numStep > 1) then
464          dtstep = tim_windowsize/(real(numStep-1,8))
465        else
466          dtstep = tim_windowsize
467        end if
468
469        do stepIndex = 1, numStep
470          dldelt = (stepIndex - ((numStep-1)/2 + 1)) * dtstep
471          call incdatr(datestamplist(stepIndex), referenceDateStamp, dldelt)
472        end do
473      end if
474
475      if (trim(tim_referencetime) == "start") then
476     
477        dtstep = tim_windowsize/(real(numStep,8))
478     
479        do stepIndex = 1, numStep
480          dldelt = (stepIndex - 1) * dtstep
481          call incdatr(datestamplist(stepIndex), referenceDateStamp, dldelt)
482        end do
483
484      end if
485
486    end if
487
488  end subroutine tim_getStampList

491  subroutine tim_getStepObsIndex(dnstepobs, referenceDateStamp, obsYYYMMDD, obsHHMM, numStep)
492    !
493    !:Purpose: Return the stepobs index as a real number (-1.0 if out of range)
494    !
495
496    implicit none
497
498    ! arguments
499    real(8), intent(out) :: dnstepobs          ! number of stepobs from reference time
500    integer, intent(in)  :: referenceDateStamp    ! Synop CMC date-time stamp
501    integer, intent(in)  :: obsYYYMMDD         ! Obs date YYYYMMDD
502    integer, intent(in)  :: obsHHMM            ! Obs time HHMM
503    integer, intent(in)  :: numStep            ! number of stepobs in assimilation window
504
505    ! locals
506    integer :: newdate, istat, imode
507    real(8) :: dddt      ! delta time in hours
508    integer :: istobs    ! obs CMC date-time stamp
509    integer :: itobs     ! obs time HHMMSShh
510    real(8) :: dlhours   ! delta time from synop time
511
512    if (.not. initialized) call utl_abort('tim_getStepObsIndex: module not initialized')
513
514    ! Building observation stamp
515    imode = 3 ! printable to stamp
516    itobs = obsHHMM * 10000
517    istat = newdate(istobs, obsYYYMMDD, itobs, imode)
518
519    ! Difference (in hours) between obs time and reference time
520    call difdatr(istobs, referenceDateStamp, dlhours)
521
522    if (numStep > 1) then
523      ! FGAT: more than 1 trial field in time window
524      if (tim_referencetime == "middle") then
525        dddt = tim_windowsize / (real(numStep-1,8))
526      else
527        dddt = tim_windowsize / (real(numStep,8))
528      end if
529      dnstepobs = dlhours / dddt ! number of step obs from reference (e.g. synoptic)
530      if (tim_referencetime == "middle") then
531        dnstepobs = dnstepobs + real((numStep+1)/2,8)
532      end if
533      if (trim(tim_referencetime) == "start") then
534        dnstepobs = dnstepobs + 1.d0
535      end if
536      if (dnstepobs < 0.5d0.or.dnstepobs > (0.5d0+real(numStep,8))) dnstepobs = -1.0d0
537
538    else
539      ! 3D: only 1 trial field in time window
540      if (tim_referencetime == "middle") then
541        if (dlhours < -tim_windowsize/2.0D0 .or. dlhours > tim_windowsize/2.0D0) then
542          ! outside time window
543          dnstepobs = -1.0d0
544        else
545          ! inside time window
546          dnstepobs = 1.0d0
547        end if
548      else
549        dddt = tim_windowsize
550        if (dlhours < -dddt/2.0d0 .or. dlhours > tim_windowsize + dddt/2.d0) then
551          ! outside time window
552          dnstepobs = -1.0d0
553        else
554          ! inside time window
555          dnstepobs = 1.0d0
556        end if
557      endif
558      
559    end if
560
561  end subroutine tim_getStepObsIndex

566  subroutine tim_dateStampToYYYYMMDDHH(dateStamp, prnttime, dd, mm, ndays, yyyy, verbose_opt)
567    !
568    !:Purpose: to get day (DD), month (MM), number of days in this month 
569    !           and year (YYYY) from dateStamp
570    !  
571    
572    implicit none
573  
574    ! arguments
575    integer, intent(in)           :: dateStamp
576    integer, intent(inout)        :: prnttime, dd, mm, ndays, yyyy
577    logical, intent(in), optional :: verbose_opt
578    
579    ! locals
580    character(len=8)            :: yyyymmdd
581    character(len=3), parameter :: months(12) = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec']
582    integer                     :: ndaysM(12)    
583    integer                     :: imode, ierr, newdate, prntdate
584    logical                     :: verbose = .True.
585
586    ndaysM(:) = [   31,    28,    31,    30,    31,    30,    31,    31,    30,    31,    30,    31]
587    
588    imode = -3 ! stamp to printable
589    ierr = newdate(dateStamp, prntdate, prnttime, imode)
590    write(yyyymmdd,'(i8)') prntdate
591    read (yyyymmdd(7:8), '(i)') dd
592    read (yyyymmdd(5:6), '(i)') mm
593    read (yyyymmdd(1:4), '(i)') yyyy
594
595    ! leap year    
596    if (mm == 2 .and. mod(yyyy,4)==0) ndaysM(mm) = 29
597    ndays = ndaysM(mm)
598    
599    if (present(verbose_opt)) verbose = verbose_opt
600    
601    if(verbose) then
602      write(*,*) 'tim_dateStampToYYYYMMDDHH:  date = ', prntdate
603      write(*,*) 'tim_dateStampToYYYYMMDDHH:  year = ', yyyy
604      write(*,'(a,i5,a,i5,a)') 'tim_dateStampToYYYYMMDDHH: month = ', mm, ' ( '// months(mm)//' where there are ', ndays, ' days)' 
605      write(*,*) 'tim_dateStampToYYYYMMDDHH:   day = ', dd
606      write(*,*) 'tim_dateStampToYYYYMMDDHH:  time = ', prnttime
607    end if     
608  
609  end subroutine tim_dateStampToYYYYMMDDHH

611  !----------------------------------------------------------------------------------------
612  ! tim_getValidDateTimeFromList
613  !----------------------------------------------------------------------------------------
614  subroutine tim_getValidDateTimeFromList(headDateValues, headTimeValues, validDate, validtime)
615    implicit none
616
617    ! arguments:
618    integer, intent(in)  :: headDateValues(:)
619    integer, intent(in)  :: headTimeValues(:)
620    integer, intent(out) :: validDate
621    integer, intent(out) :: validTime
622
623    ! locals:
624    integer                 :: numDates, numWindowsPerDay, windowIndex, timeMin, timeMax, dateMin, dateMax
625    integer                 :: windowBoundaryMin, windowBoundaryMax, validTimeMin, validTimeMax, validDateMin, validDateMax 
626    integer(8)              :: dateTimeMin, dateTimeMax
627    integer(8), allocatable :: dateTimeValues(:), windowBoundaries(:)
628    integer                 :: ier, imode
629    integer                 :: newdate, dateStampIn, dateStampOut
630
631    call tim_readNml()
632
633    numDates = size(headDateValues)
634    checkNumDates: if (numDates > 0) then
635      write(*,*) 'tim_getValidDateTimeFromList: check inputs: time min/max = ', minval(headTimeValues), maxval(headTimeValues)
636      write(*,*) 'tim_getValidDateTimeFromList: check inputs: date min/max = ', minval(headDateValues), maxval(headDateValues)
637      allocate(dateTimeValues(size(headDateValues)))
638      dateTimeValues(:) = headDateValues(:)
639      dateTimeValues(:) = 10000*dateTimeValues(:) + headTimeValues(:)
640
641      dateTimeMin = minval(dateTimeValues(:))
642      dateTimeMax = maxval(dateTimeValues(:))
643      deallocate(dateTimeValues)
644      dateMin = dateTimeMin/10000
645      dateMax = dateTimeMax/10000
646      timeMin = dateTimeMin - 10000*(dateTimeMin/10000)
647      timeMax = dateTimeMax - 10000*(dateTimeMax/10000)
648      ! convert from hhmm to just minutes: hhmm - 100*(hhmm/100) + 60*(hhmm/100)
649      timeMin = timeMin - 100*(timeMin/100) + 60*(timeMin/100)
650      timeMax = timeMax - 100*(timeMax/100) + 60*(timeMax/100)
651      write(*,*) 'tim_getValidDateTimeFromList: min/max DateTime             = ', dateTimeMin, dateTimeMax
652      write(*,*) 'tim_getValidDateTimeFromList: min/max time (in minutes)    = ', timeMin, timeMax
653      if (tim_windowSize < 24.0d0) then
654        numWindowsPerDay = nint(24.0/tim_windowSize)
655
656        ! define boundaries between assimilation windows in minutes relative to 0UTC
657        allocate(windowBoundaries(0:numWindowsPerDay))
658        do windowIndex = 0, numWindowsPerDay
659          ! example for windowSize=6h, boundaries = -3h,+3h,+9h,+15h,+21h
660          windowBoundaries(windowIndex) = nint(windowIndex*60.0*tim_windowSize - 60.0*tim_windowSize/2.0)
661        end do
662        write(*,*) 'tim_getValidDateTimeFromList: boundaries (in minutes) = ', windowBoundaries(:)
663
664        ! find left boundary of window where timeMin/Max are located
665        windowBoundaryMin = -1
666        windowBoundaryMax = -1
667        do windowIndex = 0, numWindowsPerDay
668          if (timeMin >= windowBoundaries(windowIndex)) then
669            windowBoundaryMin = windowIndex
670          end if
671          if (timeMax >= windowBoundaries(windowIndex)) then
672            windowBoundaryMax = windowIndex
673          end if
674        end do
675
676        ! find validTimeMin/Max from left boundary
677        validTimeMin = nint((windowBoundaries(windowBoundaryMin) + 60.0*tim_windowSize/2.0)/60.0)
678        if (validTimeMin >= 24) then
679          validTimeMin = 0
680          imode = 3
681          ier = newdate(dateStampIn, dateMin, validTimeMin, imode)
682          call incdat(dateStampOut, dateStampIn, 24) ! add 1 day to get validDate
683          imode = -3
684          ier = newdate(dateStampOut, validDateMin, validTimeMin, imode)
685          validTimeMin = 0
686        else
687          validDateMin = dateMin
688        end if
689        validTimeMax = nint((windowBoundaries(windowBoundaryMax) + 60.0*tim_windowSize/2.0)/60.0)
690        if (validTimeMax >= 24) then
691          validTimeMax = 0
692          imode = 3
693          ier = newdate(dateStampIn, dateMax, validTimeMax, imode)
694          call incdat(dateStampOut, dateStampIn, 24) ! add 1 day to get validDate
695          imode = -3
696          ier = newdate(dateStampOut, validDateMax, validTimeMax, imode)
697          validTimeMax = 0
698        else
699          validDateMax = dateMax
700        end if
701        write(*,*) 'tim_getValidDateTimeFromList: date from Min/Max = ', validDateMin, validDateMax
702        write(*,*) 'tim_getValidDateTimeFromList: hour from Min/Max = ', validTimeMin, validTimeMax
703        if (validTimeMin /= validTimeMax) call utl_abort('validTimeMin/Max not equal')
704        validTime = validTimeMin
705        validDate = validDateMin
706        deallocate(windowBoundaries)
707      else
708        write(*,*) 'tim_getValidDateTimeFromList: WARNING: window size equal or greater than 1 day, cannot get dateStamp'
709        validTime = 0
710        validDate = 0
711      end if
712    end if checkNumDates
713    
714  end subroutine tim_getValidDateTimeFromList
