 47  ! sqlf_getDateStamp
 48  !--------------------------------------------------------------------------
 49  subroutine sqlf_getDateStamp(dateStamp, sqliteFileName)
 50    !
 51    ! Purpose: get dateStamp from an SQLite file
 52    !
 53  
 54    implicit none
 55    
 56    ! arguments
 57    integer         , intent(out) :: dateStamp
 58    character(len=*), intent(in)  :: sqliteFileName
 59    
 60    ! locals
 61    logical              :: fileExists 
 62    integer              :: ier, imode, validTime, validDate, validDateRecv, validTimeRecv
 63    integer              :: newdate
 64    integer, allocatable :: headDateValues(:), headTimeValues(:)
 65
 66    validDate = MPC_missingValue_INT 
 67    validTime = MPC_missingValue_INT 
 68    
 69    inquire(file = trim(sqliteFileName), exist = fileExists)
 70
 71    if (fileExists) then
 72      call sqlr_getColumnValuesDate(headDateValues, headTimeValues, fileName=trim(sqliteFileName))
 73      call tim_getValidDateTimeFromList(headDateValues, headTimeValues, validDate, validTime)
 74    end if
 75
 76    ! Make sure all mpi tasks have a valid date (important for split sqlite files)
 77    call rpn_comm_allreduce(validDate, validDateRecv, 1, "MPI_INTEGER", "MPI_MAX", "GRID", ier)
 78    call rpn_comm_allreduce(validTime, validTimeRecv, 1, "MPI_INTEGER", "MPI_MAX", "GRID", ier)
 79    
 80    if (validDateRecv == MPC_missingValue_INT .or. validTimeRecv == MPC_missingValue_INT) then
 81      write(*,*) 'sqlf_getDateStamp: WARNING: Error in getting valid date and time!'
 82      dateStamp = 0
 83    else    
 84      ! printable to stamp, validTime must be multiplied with 1e6 to make newdate work
 85      imode = 3
 86      ier = newdate(dateStamp, validDateRecv, validTimeRecv * 1000000, imode)
 87      write(*,*)'sqlf_getDateStamp: SQLite files valid date (YYYYMMDD): ', validDateRecv
 88      write(*,*)'sqlf_getDateStamp: SQLite files valid time       (HH): ', validTimeRecv
 89      write(*,*)'sqlf_getDateStamp: SQLite files dateStamp            : ', datestamp
 90    end if
 91
 92  end subroutine sqlf_getDateStamp

 97  subroutine sqlf_readFile(obsdat, fileName, familyType, fileIndex)
 98    implicit none
 99    ! arguments
100    type (struct_obs), intent(inout) :: obsdat
101    character(len=*)                 :: fileName
102    character(len=*)                 :: familyType
103    integer                          :: fileIndex
104    ! locals
105    integer :: bodyIndex, bodyIndexBegin, bodyIndexEnd, headerIndexBegin, headerIndexEnd, headerIndex
106    integer :: numBody, numHeader
107    character(len=*), parameter :: my_name = 'sqlf_readFile'
108    character(len=*), parameter :: my_warning = '****** '// my_name //' WARNING: '
109    character(len=*), parameter :: my_error   = '******** '// my_name //' ERROR: '
110    real(pre_obsReal)           :: missingValue
111
112    write(*,*)' '
113    write(*,*)'                '//trim(my_name)//': Starting          '
114    write(*,*)' '
115    missingValue = real(MPC_missingValue_R8,pre_obsReal)
116    write(*,*) my_name//': FileName   : ', trim(FileName)
117    write(*,*) my_name//': FamilyType : ', FamilyType
118
119    bodyIndexBegin   = obs_numbody(obsdat) + 1
120    headerIndexBegin = obs_numheader(obsdat) + 1
121    call sqlr_readSqlite(obsdat, trim(familyType), trim(fileName))
122    bodyIndexEnd   = obs_numbody(obsdat)
123    headerIndexEnd = obs_numheader(obsdat)
124    if (trim(familyType) == 'TO') then
125      call sqlr_readSqlite_avhrr(obsdat, trim(fileName), headerIndexBegin, headerIndexEnd)
126    end if
127
128    if (trim(familyType) /= 'TO') then
129      call ovt_transformObsValues      (obsdat, headerIndexBegin, headerIndexEnd)
130      call ovt_adjustHumGZ             (obsdat, headerIndexBegin, headerIndexEnd)
131      call obsu_computeVertCoordSurfObs(obsdat, headerIndexBegin, headerIndexEnd)
132    end if
133
134    do headerIndex = headerIndexBegin, headerIndexEnd
135      call obs_headSet_i(obsdat, OBS_OTP, headerIndex, fileIndex)
136      call obs_headSet_i(obsdat, OBS_IDF, headerIndex, fileIndex)
137      call obs_setFamily(obsdat, trim(familyType), headerIndex)
138    end do
139
140    do bodyIndex = bodyIndexBegin, bodyIndexEnd
141      if (obs_columnActive_RB(obsdat, OBS_OMA))  call obs_bodySet_r(obsdat, OBS_OMA , bodyIndex, missingValue)
142      if (obs_columnActive_RB(obsdat, OBS_OMA0))  call obs_bodySet_r(obsdat, OBS_OMA0, bodyIndex, missingValue)
143      if (obs_columnActive_RB(obsdat, OBS_OMP))  call obs_bodySet_r(obsdat, OBS_OMP , bodyIndex, missingValue)
144      if (obs_columnActive_RB(obsdat, OBS_OMP6))  call obs_bodySet_r(obsdat, OBS_OMP6, bodyIndex, missingValue)
145      if (obs_columnActive_RB(obsdat, OBS_OER))  call obs_bodySet_r(obsdat, OBS_OER , bodyIndex, missingValue)
146      if (obs_columnActive_RB(obsdat, OBS_HPHT)) call obs_bodySet_r(obsdat, OBS_HPHT, bodyIndex, missingValue)
147      if (obs_columnActive_RB(obsdat, OBS_HAHT))  call obs_bodySet_r(obsdat, OBS_HAHT, bodyIndex, missingValue)
148      if (obs_columnActive_RB(obsdat, OBS_WORK)) call obs_bodySet_r(obsdat, OBS_WORK, bodyIndex, missingValue)
149      if (obs_columnActive_RB(obsdat, OBS_SIGI))  call obs_bodySet_r(obsdat, OBS_SIGI, bodyIndex, missingValue)
150      if (obs_columnActive_RB(obsdat, OBS_SIGO))  call obs_bodySet_r(obsdat, OBS_SIGO, bodyIndex, missingValue)
151      if (obs_columnActive_RB(obsdat, OBS_ZHA))  call obs_bodySet_r(obsdat, OBS_ZHA , bodyIndex, missingValue)
152      if (obs_columnActive_RB(obsdat, OBS_BCOR)) call obs_bodySet_r(obsdat, OBS_BCOR ,bodyIndex, missingValue)
153    end do
154
155    ! For GP family, initialize OBS_OER to element 15032 (ZTD formal error) 
156    ! for all ZTD data (element 15031)
157    if (trim(familyType) == 'GP') then
158      write(*,*)' Initializing OBS_OER for GB-GPS ZTD to formal error (ele 15032)'
159      call obsu_setGbgpsError(obsdat, headerIndexBegin, headerIndexEnd)
160    end if
161
162    numHeader = obs_numHeader(obsdat)
163    numBody   = obs_numBody(obsdat)
164    write(*,*) my_name//': obs_numheader', trim(familyType), numHeader
165    write(*,*) my_name//': obs_numbody  ', trim(familyType), numBody
166    write(*,*)' '
167    write(*,*)'      '//trim(my_name)//'     END                   '
168    write(*,*)' '
169
170  end subroutine  sqlf_readFile

175  subroutine sqlf_updateFile(obsSpaceData, fileName, familyType, fileIndex)
176    implicit none
177    ! arguments
178    type (struct_obs), intent(inout) :: obsSpaceData
179    character(len=*)                 :: fileName
180    character(len=*)                 :: familyType
181    integer                          :: fileIndex
182    ! locals
183    character(len=*), parameter :: myName = 'sqlf_updateFile'
184    character(len=*), parameter :: myWarning = '****** '// myName //' WARNING: '
185    character(len=*), parameter :: myError   = '******** '// myName //' ERROR: '
186
187    call utl_tmg_start(13,'----UpdateSqliteFile')
188    write(*,*) myName//' Starting'
189    write(*,*) myName//': FileName   : ',trim(fileName)
190    write(*,*) myName//': FamilyType : ',FamilyType
191
192    call fSQL_open(db, fileName, statusSqlite)
193    if (fSQL_error(statusSqlite) /= FSQL_OK) then
194      write(*,*) 'fSQL_open: ', fSQL_errmsg(statusSqlite)
195      write(*,*) myError, fSQL_errmsg(statusSqlite)
196    end if
197
198    call sqlr_updateSqlite(db, obsSpaceData, familyType, fileName, fileIndex)
199    call sqlr_insertSqlite(db, obsSpaceData, familyType, fileName, fileIndex)
200
201    write(*,*)'  closed database -->', trim(FileName)
202    call fSQL_close(db, statusSqlite)
203    write(*,*)' '
204    write(*,*)'================================================='
205    write(*,*)'                '//trim(myName)//'    END               '
206    write(*,*)'================================================='
207    write(*,*)' '
208    call utl_tmg_stop(13)
209
210  end subroutine sqlf_updateFile
