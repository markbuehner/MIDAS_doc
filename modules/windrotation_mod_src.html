 50  subroutine uvr_setup( uvr, hco_in )
 51    !
 52    !:Purpose:    Setup the information for wind rotation
 53    !
 54    implicit none
 55
 56    ! Arguments:
 57    type(struct_uvr), pointer    :: uvr    ! Wind rotation object
 58    type(struct_hco), intent(in) :: hco_in ! Horizontal grid object
 59
 60    if ( associated(uvr) ) then
 61      if ( uvr%initialized ) then
 62        if ( mmpi_myid == 0 ) write(*,*) 'uvr_setup: already initialized, returning'
 63        return
 64      else
 65        call utl_abort('uvr_setup: the supplied non-null uvr pointer is not initialized!')
 66      end if
 67    end if
 68
 69    !
 70    !-  Compute the rotation matrices (grd_rot_8 and grd_rotinv_8)
 71    !
 72    if ( mmpi_myid == 0 ) then
 73      write(*,*)
 74      write(*,*) 'uvr_setup: Starting...  for grid type = ', hco_in%grtyp
 75    end if
 76
 77    allocate(uvr)
 78
 79    if ( mmpi_myid == 0 ) then
 80      write(*,*) hco_in % xlon1
 81      write(*,*) hco_in % xlat1
 82      write(*,*) hco_in % xlon2
 83      write(*,*) hco_in % xlat2
 84    end if
 85
 86    call sugrdpar(uvr, 1, hco_in % xlon1, hco_in % xlat1, hco_in % xlon2, hco_in % xlat2 )
 87
 88    if ( hco_in%grtyp == 'U' ) then
 89
 90      if ( mmpi_myid == 0 ) then
 91        write(*,*) 'uvr_setup: doing setup for YAN grid'
 92        write(*,*) hco_in % xlon1_yan
 93        write(*,*) hco_in % xlat1_yan
 94        write(*,*) hco_in % xlon2_yan
 95        write(*,*) hco_in % xlat2_yan
 96      end if
 97
 98      call sugrdpar(uvr, 2, hco_in % xlon1_yan, hco_in % xlat1_yan, hco_in % xlon2_yan, hco_in % xlat2_yan )
 99
100    end if
101
102    uvr%initialized = .true.
103
104    if ( mmpi_myid == 0 ) then
105      write(*,*)
106      write(*,*) 'uvr_setup: Done!'
107    end if
108
109  end subroutine uvr_setup

111  subroutine sugrdpar( uvr, subGridIndex, grd_xlon1, grd_xlat1, grd_xlon2, grd_xlat2 )
112    !
113    !:Purpose: Compute the rotation matrix (r_8) that allows transformation
114    !           from the non-rotated to the rotated spherical coordinate system.
115    !
116    implicit none
117
118    ! Arguments:
119    type(struct_uvr), pointer :: uvr          ! Wind rotation object
120    integer                   :: subGridIndex ! Horizontal subGrid index = 2
121    real(8), intent(in)       :: grd_xlon1    ! Horizontal grid xlon1_yan
122    real(8), intent(in)       :: grd_xlat1    ! Horizontal grid xlat1_yan
123    real(8), intent(in)       :: grd_xlon2    ! Horizontal grid xlon2_yan
124    real(8), intent(in)       :: grd_xlat2    ! Horizontal grid xlat2_yan
125
126    ! locals
127    integer :: j1, j2
128    real(8) :: zxlon1_8,zxlat1_8,zxlon2_8,zxlat2_8
129    real(8) :: a_8, b_8, c_8, d_8, xyz1_8(msize), xyz2_8(msize)
130    real(8) :: zunit(msize,msize)
131
132    zxlon1_8 = grd_xlon1
133    zxlat1_8 = grd_xlat1
134    zxlon2_8 = grd_xlon2
135    zxlat2_8 = grd_xlat2
136
137    if ( zxlon1_8 == zxlon2_8 .and. zxlat1_8 == zxlat2_8 ) then
138      !
139      !- 1. Non rotated grid (for test case only)
140      !
141      if ( mmpi_myid == 0 ) then
142        write(*,*)
143        write(*,*) 'uvr_sugrdpar: Warning: This grid is not rotated !!!'
144      end if
145      uvr%grd_rot_8(1,1,subGridIndex) = 1.0d0       
146      uvr%grd_rot_8(1,2,subGridIndex) = 0.0d0
147      uvr%grd_rot_8(1,3,subGridIndex) = 0.0d0
148      uvr%grd_rot_8(2,1,subGridIndex) = 0.0d0
149      uvr%grd_rot_8(2,2,subGridIndex) = 1.0d0
150      uvr%grd_rot_8(2,3,subGridIndex) = 0.0d0
151      uvr%grd_rot_8(3,1,subGridIndex) = 0.0d0
152      uvr%grd_rot_8(3,2,subGridIndex) = 0.0d0
153      uvr%grd_rot_8(3,3,subGridIndex) = 1.0d0
154
155    else
156      !
157      !- 2. Rotated grid
158      !
159      call vllacar( xyz1_8, zxlon1_8, zxlat1_8 )
160      call vllacar( xyz2_8, zxlon2_8, zxlat2_8 )
161
162      !- 2.1 Compute a = cos(alpha) & b = sin(alpha)
163      a_8 = (xyz1_8(1)*xyz2_8(1)) + (xyz1_8(2)*xyz2_8(2)) &
164                                  + (xyz1_8(3)*xyz2_8(3))
165
166      b_8 = sqrt (((xyz1_8(2)*xyz2_8(3)) - (xyz2_8(2)*xyz1_8(3)))**2 &
167               +  ((xyz2_8(1)*xyz1_8(3)) - (xyz1_8(1)*xyz2_8(3)))**2 &
168               +  ((xyz1_8(1)*xyz2_8(2)) - (xyz2_8(1)*xyz1_8(2)))**2)
169
170      !- 2.2 Compute c = norm(-r1) & d = norm(r4)
171      c_8 = sqrt ( xyz1_8(1)**2 + xyz1_8(2)**2 + xyz1_8(3)**2 )
172    
173      d_8 = sqrt ( ( ( (a_8*xyz1_8(1)) - xyz2_8(1) ) / b_8 )**2 + &
174                 ( ( (  a_8*xyz1_8(2)) - xyz2_8(2) ) / b_8 )**2 + &
175                 ( ( (  a_8*xyz1_8(3)) - xyz2_8(3) ) / b_8 )**2  )
176
177      !- 2.3 Compute the forward rotation matrix
178      uvr%grd_rot_8(1,1,subGridIndex) = -xyz1_8(1) / c_8
179      uvr%grd_rot_8(1,2,subGridIndex) = -xyz1_8(2) / c_8
180      uvr%grd_rot_8(1,3,subGridIndex) = -xyz1_8(3) / c_8
181      uvr%grd_rot_8(2,1,subGridIndex) = ( ((a_8*xyz1_8(1)) - xyz2_8(1)) / b_8 ) / d_8
182      uvr%grd_rot_8(2,2,subGridIndex) = ( ((a_8*xyz1_8(2)) - xyz2_8(2)) / b_8 ) / d_8
183      uvr%grd_rot_8(2,3,subGridIndex) = ( ((a_8*xyz1_8(3)) - xyz2_8(3)) / b_8 ) / d_8
184      uvr%grd_rot_8(3,1,subGridIndex) = ( (xyz1_8(2)*xyz2_8(3)) - (xyz2_8(2)*xyz1_8(3))) / b_8
185      uvr%grd_rot_8(3,2,subGridIndex) = ( (xyz2_8(1)*xyz1_8(3)) - (xyz1_8(1)*xyz2_8(3))) / b_8
186      uvr%grd_rot_8(3,3,subGridIndex) = ( (xyz1_8(1)*xyz2_8(2)) - (xyz2_8(1)*xyz1_8(2))) / b_8
187
188    end if
189
190    do j1 = 1, msize
191      do j2 = 1, msize
192        if ( mmpi_myid == 0 ) then
193          write(*,*) 'sugrdpar: grd_rot_8(j1,j2) =',j1,j2,uvr%grd_rot_8(j1,j2,subGridIndex)
194        end if
195      end do
196    end do
197
198    !
199    !- 3. Compute the inverse matrix (transpose is the inverse)
200    !
201    do j1 = 1, msize
202      do j2 = 1, msize
203        uvr%grd_rotinv_8(j1,j2,subGridIndex) = uvr%grd_rot_8(j2,j1,subGridIndex)
204      end do
205    end do
206
207    zunit(:,:) = 0.d0
208    call mxma8x(zunit,uvr%grd_rotinv_8(:,:,subGridIndex),uvr%grd_rot_8(:,:,subGridIndex),msize,msize,msize)
209    do j1 = 1, msize
210      do j2 = 1, msize
211        if ( mmpi_myid == 0 ) then
212          write(*,*) 'sugrdpar: unit = ', j1, j2, zunit(j1,j2)
213        end if
214      end do
215    end do
216
217  end subroutine sugrdpar

219  subroutine vllacar( F_xyz_8, F_lon, F_lat )
220    !
221    !:Purpose:    Compute parameters of rotated grid
222    ! 
223    implicit none
224
225    real(8), intent(out) :: F_xyz_8(msize) ! output
226    real(8), intent(in)  :: F_lon          ! Input in degrees  
227    real(8), intent(in)  :: F_lat          ! Input in degrees
228
229    F_xyz_8(1) = cos(MPC_RADIANS_PER_DEGREE_R8*F_lat) * cos(MPC_RADIANS_PER_DEGREE_R8*F_lon)
230    F_xyz_8(2) = cos(MPC_RADIANS_PER_DEGREE_R8*F_lat) * sin(MPC_RADIANS_PER_DEGREE_R8*F_lon)
231    F_xyz_8(3) = sin(MPC_RADIANS_PER_DEGREE_R8*F_lat)
232
233  end subroutine vllacar

235  subroutine mxma8x( pmat3, pmat1, pmat2, kdimi1, kdimj1, kdimj2 )
236    !
237    !:Purpose:    Compute a product of two matrices.
238    !
239    implicit none
240
241    ! Arguments:
242    real(8) :: pmat3(kdimi1,kdimj2)  ! output
243    real(8) :: pmat1(kdimi1,kdimj1)  ! input matrix one
244    real(8) :: pmat2(kdimj1,kdimj2)  ! input matrix two
245    integer :: kdimi1                ! first  dimension of the first  matrix
246    integer :: kdimj1                ! second dimension of the first  matrix 
247    integer :: kdimj2                ! second dimension of the second matrix
248
249    ! locals
250    integer :: ji1,jj2,jj
251
252    pmat3(:,:) = 0.d0
253
254    do jj2 = 1, kdimj2
255      do jj = 1, kdimj1
256        do ji1 = 1, kdimi1
257          pmat3(ji1,jj2) = pmat3(ji1,jj2) + pmat1(ji1,jj) * pmat2(jj,jj2)
258        end do
259      end do
260    end do
261 
262  end subroutine mxma8x

264  subroutine uvr_rotateWind_nl( uvr, subGridIndex, uwind, vwind, Lat, Lon, LatRot, LonRot, mode )
265    !
266    !:Purpose: Go from tangential wind components from one sphere to another
267    !           (same origin!). Original ezsint version used for computing innovation.
268    !
269    implicit none
270
271    ! Arguments:
272    type(struct_uvr), pointer :: uvr          ! Wind rotation object
273    integer, intent(in)       :: subGridIndex ! Current subgrid index
274    real(8), intent(inout)    :: uwind        ! interpUU
275    real(8), intent(inout)    :: vwind        ! interpVV
276    real(8), intent(in)       :: Lat          ! Latitude in radians
277    real(8), intent(in)       :: Lon          ! Longitude in radians
278    real(8), intent(in)       :: LatRot       ! Rotated latitude in radians
279    real(8), intent(in)       :: LonRot       ! Rotated longitude in radians 
280    character(*), intent(in)  :: mode         ! ToMetWind or ToRotWind
281
282    ! locals
283    integer :: index1, index2
284    real(8) :: coslatr, sinlatr, coslonr, sinlonr, coslat, sinlat, coslon, sinlon, ezCoeff_C, ezCoeff_D
285    real(8) :: xyz(msize), uvcart(msize)
286
287    if ( .not. uvr%initialized ) then
288      write(*,*)
289      call utl_abort('uvr_rotateWind_nl: WindRotation module is not initialize')
290    endif
291
292    coslatr = cos(LatRot)
293    sinlatr = sin(LatRot)
294    coslonr = cos(LonRot)
295    sinlonr = sin(LonRot)
296    coslat  = cos(Lat)
297    sinlat  = sin(Lat)
298    coslon  = cos(Lon)
299    sinlon  = sin(Lon)
300    
301    if ( trim(mode) == 'ToMetWind' ) then 
302
303      xyz(1) = -uwind*sinlonr - vwind*coslonr*sinlatr
304      xyz(2) =  uwind*coslonr - vwind*sinlonr*sinlatr
305      xyz(3) =                  vwind*coslatr
306    
307      uvcart(:) = 0.0d0
308      do index2 = 1, msize
309        do index1 = 1, msize
310          uvcart(index1) = uvcart(index1) +   &
311               uvr%grd_rotinv_8(index1,index2,subGridIndex)*xyz(index2)
312        end do
313      end do
314
315      uwind  = uvcart(2)*coslon - uvcart(1)*sinlon
316      ezCoeff_C = uvcart(1)*coslon + uvcart(2)*sinlon
317      ezCoeff_D = sqrt( ezCoeff_C**2 + uvcart(3)**2 )
318      vwind     = sign( ezCoeff_D, uvcart(3)*coslat - ezCoeff_C*sinlat )
319
320    else if ( trim(mode) == 'ToRotWind' ) then
321      write(*,*) 
322      call utl_abort('uvr_rotateWind_nl: mode ToRotWind is not available yet')
323    else
324      write(*,*) 
325      write(*,*) 'uvr_rotateWind_nl: Unknown transform name: ', trim(mode)
326      write(*,*) '                mode = ToMetWind or ToRotWind'
327      call utl_abort('uvr_rotateWind_nl')
328    end if
329
330  end subroutine uvr_rotateWind_nl

332  subroutine uvr_rotateWind_tl( uvr, subGridIndex, uwind, vwind, Lat_in, Lon_in, LatRot_in, LonRot_in, mode )
333    !
334    !:Purpose: Go from tangential wind components from one sphere to another
335    !           (same origin!). Fast version used by Variational analysis. 
336    !
337    implicit none
338
339    ! Arguments:
340    type(struct_uvr), pointer :: uvr          ! Wind rotation object
341    integer, intent(in)       :: subGridIndex ! Current subgrid index
342    real(8), intent(inout)    :: uwind        ! interpUU
343    real(8), intent(inout)    :: vwind        ! interpVV
344    real(8), intent(in)       :: Lat_in       ! Latitude in radians
345    real(8), intent(in)       :: Lon_in       ! Longitude in radians
346    real(8), intent(in)       :: LatRot_in    ! Rotated latitude in radians
347    real(8), intent(in)       :: LonRot_in    ! Rotated longitude in radians 
348    character(*), intent(in)  :: mode         ! ToMetWind or ToRotWind
349
350    ! locals
351    integer :: index1, index2
352    real(8) :: coslatr, sinlatr, coslonr, sinlonr, coslon, sinlon, rsinlat
353    real(8) :: lat, lon, latRot, lonRot
354    real(8) :: xyz(msize), uvcart(msize)
355
356    if ( .not. uvr%initialized ) then
357      write(*,*)
358      call utl_abort('uvr_rotateWind_tl: WindRotation module is not initialize')
359    endif
360
361    ! Special case for the equator
362    if (abs(lat_in) < lat_minValue) then
363      lat = lat_minValue
364      lon = lon_in
365      call uvr_RotateLatLon( uvr,   & ! INOUT
366                             subGridIndex,     & ! IN
367                             latRot, lonRot,   & ! OUT (radians)
368                             lat, lon,         & ! IN  (radians)
369                             'ToLatLonRot')      ! IN
370    else
371      lat = lat_in
372      lon = lon_in
373      latRot = latRot_in
374      lonRot = lonRot_in
375    end if
376
377    coslatr = cos(LatRot)
378    sinlatr = sin(LatRot)
379    coslonr = cos(LonRot)
380    sinlonr = sin(LonRot)
381    coslon  = cos(Lon)
382    sinlon  = sin(Lon)
383    if ( Lat /= 0.0d0 ) then
384      rsinlat = 1.0d0/sin(Lat)
385    else
386      call utl_abort('uvr_rotateWind_tl: cannot be used for points on the equator')
387    end if
388
389    if ( trim(mode) == 'ToMetWind' ) then 
390
391      xyz(1) = -uwind*sinlonr - vwind*coslonr*sinlatr
392      xyz(2) =  uwind*coslonr - vwind*sinlonr*sinlatr
393      xyz(3) =                  vwind*coslatr
394    
395      uvcart(:) = 0.0d0
396      do index2 = 1, msize
397        do index1 = 1, msize
398          uvcart(index1) = uvcart(index1) +   &
399               uvr%grd_rotinv_8(index1,index2,subGridIndex)*xyz(index2)
400        end do
401      end do
402
403      uwind = -uvcart(1)*sinlon         + uvcart(2)*coslon
404      vwind = -uvcart(1)*rsinlat*coslon - uvcart(2)*rsinlat*sinlon
405
406    else if ( trim(mode) == 'ToRotWind' ) then
407      write(*,*) 
408      call utl_abort('uvr_rotateWind_tl: mode ToRotWind is not available yet')
409    else
410      write(*,*) 
411      write(*,*) 'uvr_rotateWind_tl: Unknown transform name: ', trim(mode)
412      write(*,*) '                mode = ToMetWind or ToRotWind'
413      call utl_abort('uvr_rotateWind_tl')
414    end if
415
416  end subroutine uvr_rotateWind_tl

418  subroutine uvr_rotateWind_ad( uvr, subGridIndex, uwind, vwind, Lat_in, Lon_in, LatRot_in, LonRot_in, mode )
419    !
420    !:Purpose: Adjoint of : Go from tangential wind components from one sphere to another
421    !           (same origin!). Fast version used by Variational analysis. 
422    !
423    implicit none
424
425    ! Arguments:
426    type(struct_uvr), pointer :: uvr          ! Wind rotation object
427    integer, intent(in)       :: subGridIndex ! Current subgrid index
428    real(8), intent(inout)    :: uwind        ! interpUU
429    real(8), intent(inout)    :: vwind        ! interpVV
430    real(8), intent(in)       :: Lat_in       ! Latitude in radians
431    real(8), intent(in)       :: Lon_in       ! Longitude in radians
432    real(8), intent(in)       :: LatRot_in    ! Rotated latitude in radians
433    real(8), intent(in)       :: LonRot_in    ! Rotated longitude in radians 
434    character(*), intent(in)  :: mode         ! ToMetWind or ToRotWind
435
436    ! locals
437    integer :: index1, index2
438    real(8) :: coslatr, sinlatr, coslonr, sinlonr, coslon, sinlon, rsinlat
439    real(8) :: lat, lon, latRot, lonRot
440    real(8) :: xyz(msize), uvcart(msize)
441
442    if ( .not. uvr%initialized ) then
443      write(*,*)
444      call utl_abort('uvr_rotateWind_ad: WindRotation module is not initialize')
445    endif
446
447    ! Special case for the equator
448    if (abs(lat_in) < lat_minValue) then
449      lat = lat_minValue
450      lon = lon_in
451      call uvr_RotateLatLon( uvr,   & ! INOUT
452                             subGridIndex,     & ! IN
453                             latRot, lonRot,   & ! OUT (radians)
454                             lat, lon,         & ! IN  (radians)
455                             'ToLatLonRot')      ! IN
456    else
457      lat = lat_in
458      lon = lon_in
459      latRot = latRot_in
460      lonRot = lonRot_in
461    end if
462
463    coslatr = cos(LatRot)
464    sinlatr = sin(LatRot)
465    coslonr = cos(LonRot)
466    sinlonr = sin(LonRot)
467    coslon  = cos(Lon)
468    sinlon  = sin(Lon)
469    if ( Lat /= 0.0d0 ) then
470      rsinlat = 1.0d0/sin(Lat)
471    else
472      call utl_abort('uvr_rotateWind_ad: cannot be used for points on the equator')
473    end if
474
475    if ( trim(mode) == 'ToMetWind' ) then 
476
477      uvcart(1) = -uwind*sinlon - vwind*rsinlat*coslon
478      uvcart(2) =  uwind*coslon - vwind*rsinlat*sinlon
479      uvcart(3) = 0.0d0
480
481      xyz(:) = 0.0d0
482      do index2 = 1, msize
483        do index1 = 1, msize
484          xyz(index1) = xyz(index1) + uvr%grd_rot_8(index1,index2,subGridIndex)*uvcart(index2)
485        end do
486      end do
487
488      uwind = -xyz(1)*sinlonr         + xyz(2)*coslonr
489      vwind = -xyz(1)*coslonr*sinlatr - xyz(2)*sinlonr*sinlatr + xyz(3)*coslatr
490
491    else if ( trim(mode) == 'ToRotWind' ) then
492      write(*,*) 
493      call utl_abort('uvr_rotateWind_ad: mode ToRotWind is not available yet')
494    else
495      write(*,*) 
496      write(*,*) 'uvr_rotateWind_ad: Unknown transform name: ', trim(mode)
497      write(*,*) '                   mode = ToMetWind or ToRotWind'
498      call utl_abort('uvr_rotateWind_ad')
499    end if
500
501  end subroutine uvr_rotateWind_ad

503  subroutine uvr_rotateLatLon( uvr, subGridIndex, LatOut, LonOut, LatIn, LonIn, mode )
504    !
505    !:Purpose: Go from (lat,lon) of one Cartesian frame to (lat,lon) of another
506    !           Cartesian frame given the rotation matrix.
507    !
508    implicit none
509
510    ! Arguments:
511    type(struct_uvr), pointer :: uvr          ! Wind rotation object
512    integer, intent(in)       :: subGridIndex ! Current subgrid index
513    real(8), intent(in)       :: LatIn        ! Input latitude in radians
514    real(8), intent(in)       :: LonIn        ! Input longitude in radians
515    real(8), intent(out)      :: LatOut       ! Output latitude in radians
516    real(8), intent(out)      :: LonOut       ! Output longitude in radians 
517    character(*), intent(in)  :: mode         ! ToLatLonRot or ToLatLon
518
519    ! locals
520    real(8) :: CartIn(msize),CartOut(msize)
521    real(8) :: rLon,rLat
522
523    if ( .not. uvr%initialized ) then
524      write(*,*)
525      call utl_abort('uvr_rotateLatLon: WindRotation module is not initialize')
526    endif
527
528    rLon = LonIn * MPC_DEGREES_PER_RADIAN_R8 ! To degress
529    rLat = LatIn * MPC_DEGREES_PER_RADIAN_R8 ! To degrees
530
531    call vllacar( CartIn,      & ! OUT
532                  rLon, rLat )   ! IN
533
534    if ( trim(mode) == 'ToLatLonRot' ) then
535      call mxv( CartOut,              & ! OUT
536                uvr%grd_rot_8(:,:,subGridIndex), CartIn,    & ! IN
537                msize, msize)           ! IN
538    else if ( trim(mode) == 'ToLatLon' ) then 
539      call mxv( CartOut,              & ! OUT
540                uvr%grd_rotinv_8(:,:,subGridIndex), CartIn, & ! IN
541                msize, msize)           ! IN
542    else
543      write(*,*) 
544      write(*,*) 'uvr_rotateLatLon: Unknown transform name: ', trim(mode)
545      write(*,*) '                  mode = ToLatLonRot or ToLatLon'
546      call utl_abort('uvr_rotateLatLon')
547    end if
548
549    call carall( LonOut, LatOut, & ! OUT
550                 CartOut )         ! IN
551
552    LonOut = LonOut * MPC_RADIANS_PER_DEGREE_R8 ! To radians
553    LatOut = LatOut * MPC_RADIANS_PER_DEGREE_R8 ! To radians
554
555  end subroutine uvr_rotateLatLon

558  subroutine carall( plon, plat, pcart )
559    !
560    !:Purpose: Returns (lat,lon) (degrees) of an input Cartesian position vector 
561    !           on the unit sphere.
562    !
563    implicit none
564
565    ! Arguments:
566    real(8), intent(out) :: plat         ! output latitude 
567    real(8), intent(out) :: plon         ! output longitude
568    real(8), intent(in)  :: pcart(msize) ! input Cartesian vector
569
570    plat = asin(pcart(3))
571    plat = plat * MPC_DEGREES_PER_RADIAN_R8
572    
573    if ( pcart(1) == 0.d0 ) then
574       if ( pcart(2) == 0.d0 ) then  ! point is located at the pole
575          plon = 0.0d0  ! can be any longitude... set it to zero simply.
576       else if ( pcart(2) > 0.0d0 ) then
577          plon = 90.0d0
578       else if ( pcart(2) < 0.0d0 ) then
579          plon = 270.0d0
580       end if
581    else
582       plon = atan2(pcart(2),pcart(1))
583       if (plon < 0.0d0) plon = plon + 2.d0 * MPC_PI_R8
584       plon = plon * MPC_DEGREES_PER_RADIAN_R8
585    end if
586
587  end subroutine carall

589  subroutine mxv( pvec2, pmat, pvec1, kdimi, kdimj )
590    !
591    !:Purpose: Compute a product : matrix times vector.
592    !
593    implicit none
594
595    ! Arguments:
596    real(8) :: pvec2(kdimi)      ! output vector
597    real(8) :: pmat(kdimi,kdimj) ! input matrix
598    real(8) :: pvec1(kdimj)      ! input vector
599    integer :: kdimi             ! first dimension
600    integer :: kdimj             ! second dimension
601    
602
603    ! locals
604    integer :: ji,jj
605
606    pvec2(:) = 0.0d0
607    do jj = 1,kdimj
608      do ji = 1,kdimi
609        pvec2(ji) = pvec2(ji) + pmat(ji,jj) * pvec1(jj)
610      end do
611    end do
612
613  end subroutine mxv
