 108  subroutine min_setup( nvadim_mpilocal_in, hco_anl_in, oneDVarMode_opt, &
 109                        varqc_opt, nwoqcv_opt )
 110    !
 111    !:Purpose: Reading NAMMIN namelist to setup variables for minimization.
 112    !
 113    implicit none
 114
 115    ! Arguments:
 116    integer, intent(in)                   :: nvadim_mpilocal_in
 117    type(struct_hco), pointer, intent(in) :: hco_anl_in
 118    logical, intent(in), optional         :: oneDVarMode_opt
 119    logical, intent(out), optional        :: varqc_opt
 120    integer, intent(out), optional        :: nwoqcv_opt
 121
 122    ! Locals:
 123    integer :: ierr,nulnam
 124    integer,external :: fnom,fclos
 125
 126    call utl_tmg_start(90,'--Minimization')
 127
 128    if ( nvadim_mpilocal_in /= cvm_nvadim ) then
 129      write(*,*) 'nvadim_mpilocal_in,cvm_nvadim=',nvadim_mpilocal_in,cvm_nvadim
 130      call utl_abort('min_setup: control vector dimension not consistent')
 131    endif
 132
 133    nvadim_mpilocal = nvadim_mpilocal_in
 134
 135    if ( present(oneDVarMode_opt) ) then
 136      oneDVarMode = oneDVarMode_opt
 137    else
 138      oneDVarMode = .false.
 139    end if
 140
 141    hco_anl => hco_anl_in
 142
 143    ! set default values for namelist variables
 144    nvamaj = 6
 145    nitermax = 0
 146    rdf1fac  = 0.25d0
 147    nsimmax  = 500
 148    lgrtest  = .false.
 149    lwrthess = .true.
 150    lxbar    = .false.
 151    lvazx    = .false.
 152    repsg    = 1.0d-5
 153    lvarqc   = .false.
 154    nwoqcv   = 5
 155
 156    ! read in the namelist NAMMIN
 157    nulnam=0
 158    ierr=fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
 159    read(nulnam,nml=nammin,iostat=ierr)
 160    if(ierr.ne.0) call utl_abort('min_setup: Error reading namelist')
 161    write(*,nml=nammin)
 162    ierr=fclos(nulnam)
 163
 164    IF(N1GC == 3)THEN
 165      NMTRA = (4 + 2*NVAMAJ)*nvadim_mpilocal
 166    ELSE
 167      call utl_abort('min_setup: only N1GC=3 currently supported!') 
 168    END IF
 169    WRITE(*,9401)N1GC,NVAMAJ,NMTRA
 170 9401 FORMAT(4X,'N1GC = ',I2,4X,'NVAMAJ = ',I3,/5X,"NMTRA =",1X,I14)
 171
 172    if ( present(varqc_opt) ) varqc_opt = lvarqc
 173    if ( present(nwoqcv_opt) ) nwoqcv_opt = nwoqcv
 174
 175    if(LVARQC .and. mmpi_myid == 0) write(*,*) 'VARIATIONAL QUALITY CONTROL ACTIVATED.'
 176
 177    initialized=.true.
 178
 179    call utl_tmg_stop(90)
 180
 181  end subroutine min_setup

 184  subroutine min_minimize( outerLoopIndex_in, columnTrlOnAnlIncLev, obsSpaceData, controlVectorIncrSum, &
 185                           vazx, numIterMaxInnerLoop, deallocHessian_opt, &
 186                           isMinimizationFinalCall_opt, numIterMaxInnerLoopUsed_opt )
 187    !
 188    !:Purpose: Minimizing cost function to get the increments.
 189    !           The maximum number of inner-loop iterations is set to nitermax if the
 190    !           namelist variable nitermax is provided. Otherwise, it is set to the
 191    !           numIterMaxInnerLoop supplied by the calling subroutine/program.
 192    !           numIterMaxInnerLoopUsed_opt is passing the maximum number of inner-loop
 193    !           iterations to the calling subroutine/program.
 194    !
 195    implicit none
 196
 197    ! Arguments:
 198    integer, intent(in)                    :: outerLoopIndex_in
 199    type(struct_columnData), intent(inout) :: columnTrlOnAnlIncLev
 200    type(struct_obs),        intent(inout) :: obsSpaceData
 201    real(8), intent(inout)        , target :: controlVectorIncrSum(:)
 202    real(8), intent(inout)                 :: vazx(:)
 203    integer, intent(in)                    :: numIterMaxInnerLoop
 204    integer, intent(out),         optional :: numIterMaxInnerLoopUsed_opt
 205    logical, intent(in),          optional :: deallocHessian_opt
 206    logical, intent(in),          optional :: isMinimizationFinalCall_opt
 207
 208    ! Locals:
 209    type(struct_columnData) :: columnAnlInc
 210    integer                 :: get_max_rss
 211
 212    call utl_tmg_start(90,'--Minimization')
 213
 214    write(*,*) '--------------------------------'
 215    write(*,*) '--Starting subroutine minimize--'
 216    write(*,*) '--------------------------------'
 217    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 218
 219    initializeForOuterLoop = .true.
 220    outerLoopIndex = outerLoopIndex_in
 221
 222    if ( present(deallocHessian_opt) ) then
 223      deallocHessian = deallocHessian_opt
 224    else
 225      deallocHessian = .true.
 226    end if
 227    if ( present(isMinimizationFinalCall_opt) ) then
 228      isMinimizationFinalCall = isMinimizationFinalCall_opt
 229    else
 230      isMinimizationFinalCall = .true.
 231    end if
 232
 233    if ( (nitermax > 0 .and. numIterMaxInnerLoop > 0) .or. &
 234         (nitermax == 0 .and. numIterMaxInnerLoop == 0) ) then
 235      call utl_abort('min_minimize: one of nitermax or numIterMaxInnerLoop should be zero and the other positive')
 236    end if
 237
 238    if ( nitermax > 0 ) then
 239      numIterMaxInnerLoopUsed = nitermax
 240    else if ( numIterMaxInnerLoop > 0 ) then
 241      numIterMaxInnerLoopUsed = numIterMaxInnerLoop
 242    else
 243      call utl_abort('min_minimize: one of the variables nIterMax and numIterMaxInnerLoop must be positive')
 244    end if
 245    if ( present(numIterMaxInnerLoopUsed_opt) ) numIterMaxInnerLoopUsed_opt = numIterMaxInnerLoopUsed
 246
 247    controlVectorIncrSum_ptr => controlVectorIncrSum
 248
 249    ! zero array for writing to hessian
 250    allocate(controlVectorIncrSumZero(cvm_nvadim))
 251    controlVectorIncrSumZero(:) = 0.0d0
 252
 253    call col_setVco(columnAnlInc,col_getVco(columnTrlOnAnlIncLev))
 254    call col_allocate(columnAnlInc,col_getNumCol(columnTrlOnAnlIncLev),mpiLocal_opt=.true.)
 255
 256    write(*,*) 'oti_timeBinning: For 4D increment'
 257    call oti_timeBinning(obsSpaceData,tim_nstepobsinc)
 258
 259    call quasiNewtonMinimization( columnAnlInc, columnTrlOnAnlIncLev, obsSpaceData, vazx )
 260
 261    call col_deallocate(columnAnlInc)
 262
 263    deallocate(controlVectorIncrSumZero)
 264
 265    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 266    write(*,*) '--Done subroutine minimize--'
 267
 268    call utl_tmg_stop(90)
 269
 270  end subroutine min_minimize

 273  subroutine quasiNewtonMinimization( columnAnlInc, columnTrlOnAnlIncLev, obsSpaceData, vazx )
 274      !
 275      !:Purpose: 3D/En-VAR minimization
 276      !
 277      implicit none
 278
 279      ! Arguments:
 280      type(struct_columnData), target :: columnAnlInc
 281      type(struct_columnData), target :: columnTrlOnAnlIncLev
 282      type(struct_obs),        target :: obsSpaceData
 283      real(8)                         :: vazx(:)
 284
 285      ! Locals:
 286      integer              :: nulout = 6
 287      integer              :: impres
 288      INTEGER              :: NGRANGE = 10 ! range of powers of 10 used for gradient test
 289
 290      real    :: zzsunused(1)
 291      integer :: intUnused(1)
 292
 293      real(8),allocatable :: vazg(:)
 294
 295      real(8) :: dlds(1)
 296      logical :: llvarqc, lrdvatra, llxbar
 297
 298      integer :: itermax, iterdone, itermaxtodo, isimmax, indic, iitnovqc
 299      integer :: ierr, isimdone, jdata, isimnovqc
 300      integer :: ibrpstamp, isim3d
 301      real(8) :: zjsp, zxmin, zeps1
 302      real(8) :: dlgnorm, dlxnorm
 303
 304      real(8) :: zeps0_000,zdf1_000
 305      integer :: iterdone_000,isimdone_000
 306
 307      if (lvarqc .and. outerLoopIndex==1) call vqc_setup(obsSpaceData)
 308
 309      min_nsim=0 
 310
 311      if(mmpi_myid == 0) then
 312        impres=5
 313      else 
 314        impres=0
 315      endif
 316
 317      ! Check for preconditioning file
 318      inquire(file=preconFileName,exist=preconFileExists)
 319      if(preconFileExists) then
 320        if(mmpi_myid == 0) write(*,*) 'PRECONDITIONING FILE FOUND:',preconFileName
 321      else
 322        if(mmpi_myid == 0) write(*,*) 'NO PRECONDITIONING FILE FOUND:',preconFileName
 323      endif
 324
 325      ! Initialize Hessian only at first outerLoop (mpilocal)
 326      if ( outerLoopIndex == 1 ) then
 327        allocate(vatra(nmtra))
 328        vatra(:) = 0.0d0
 329      end if
 330
 331      allocate(vazg(nvadim_mpilocal),stat=ierr)
 332      if(ierr.ne.0) then
 333        write(*,*) 'minimization: Problem allocating memory! id=2',ierr
 334        call utl_abort('min quasiNewtonMinimization')
 335      endif
 336
 337      ! set module variable pointers for obsspacedata and the two column objects
 338      obsSpaceData_ptr         => obsSpaceData
 339      columnAnlInc_ptr         => columnAnlInc
 340      columnTrlOnAnlIncLev_ptr => columnTrlOnAnlIncLev
 341
 342      ! Set-up the minimization
 343
 344      ! initialize iteration/simulation counters to zero
 345      ITERTOT = 0
 346      isimtot = 0
 347
 348      ! initialize control vector related arrays to zero
 349      vazx(:) = 0.0d0
 350      vazg(:) = 0.0d0
 351
 352
 353      ! If minimization start without qcvar : turn off varqc to compute
 354      ! innovations and test the gradients
 355
 356      if (outerLoopIndex == 1) zeps0 = repsg
 357
 358      if (preconFileExists) then
 359        if ( mmpi_myid == 0 ) write(*,*) 'quasiNewtonMinimization : Preconditioning mode'
 360        lrdvatra = .true.
 361        imode = 2
 362        llvazx = lvazx ! from namelist (default is .false.)
 363        llxbar = lxbar ! from namelist (default is .false.)
 364      else
 365        lrdvatra = .false.
 366        imode = 0
 367      endif
 368
 369      ! read the hessian from preconin file at first outer-loop iteration
 370      if ( lrdvatra .and. outerLoopIndex == 1 ) then
 371        ibrpstamp = tim_getDatestamp() ! ibrpstamp is a I/O argument of hessianIO
 372
 373        call hessianIO (preconFileName,0,                    &
 374          isim3d,ibrpstamp,zeps0_000,zdf1_000,iterdone_000,  &
 375          isimdone_000,iztrl,vatra,controlVectorIncrSumZero, &
 376          vazx,llxbar,llvazx,n1gc,imode)
 377      endif
 378
 379      iterdone = 0
 380      isimdone = 0
 381      itermax = numIterMaxInnerLoopUsed
 382      itermaxtodo = itermax
 383      isimmax = nsimmax
 384
 385      ! do the gradient test for the starting point of minimization
 386      if ( lgrtest .and. outerLoopIndex == 1 ) then
 387        ! save user-requested varqc switch
 388        llvarqc = lvarqc
 389
 390        if ( nwoqcv > 0 ) lvarqc = .false.
 391        call utl_tmg_start(91,'----QuasiNewton')
 392        call grtest2(simvar,nvadim_mpilocal,vazx,ngrange)
 393        call utl_tmg_stop(91)
 394
 395        lvarqc = llvarqc
 396      endif
 397
 398      itertot = iterdone
 399      isimtot = isimdone
 400
 401      llvarqc = lvarqc
 402      if ( nwoqcv > 0 .and. outerLoopIndex == 1 ) lvarqc = .false.
 403      INDIC =2
 404      call utl_tmg_start(91,'----QuasiNewton')
 405      call simvar(indic,nvadim_mpilocal,vazx,zjsp,vazg)
 406      call utl_tmg_stop(91)
 407      lvarqc = llvarqc
 408      
 409      if ( outerLoopIndex == 1 ) zdf1 = rdf1fac * ABS(zjsp)
 410
 411      CALL PRSCAL(nvadim_mpilocal,VAZG,VAZG,DLGNORM)
 412      DLGNORM = DSQRT(DLGNORM)
 413      CALL PRSCAL(nvadim_mpilocal,VAZX,VAZX,DLXNORM)
 414      DLXNORM = DSQRT(DLXNORM)
 415      WRITE(*,*)' |X| = ', DLXNORM
 416      WRITE(*,FMT=9220) ZJSP, DLGNORM
 417 9220 FORMAT(/4X,'J(X) = ',G23.16,4X,'|Grad J(X)| = ',G23.16)
 418
 419      ! Iterations of the minimization algorithm
 420
 421      ZXMIN = epsilon(ZXMIN)
 422      WRITE(*,FMT=9320)ZXMIN,ZDF1,ZEPS0,IMPRES,ITERMAX,NSIMMAX
 423
 424 9320 FORMAT(//,10X,' Minimization QNA_N1QN3 starts ...',/  &
 425             10x,'DXMIN =',G23.16,2X,'DF1 =',G23.16,2X,'EPSG =',G23.16  &
 426             /,10X,'IMPRES =',I3,2X,'NITER = ',I3,2X,'NSIM = ',I3)
 427
 428      ! Begin the minimization
 429      if ( numIterMaxInnerLoopUsed > 0 ) then
 430
 431        ! First do iterations without var-QC only at the beginning of first outer-loop.
 432        if (lvarqc .and. nwoqcv > 0 .and. iterdone < nwoqcv .and. outerLoopIndex == 1 ) then
 433          iitnovqc = min(nwoqcv - iterdone,itermax)
 434          isimnovqc = isimmax
 435          lvarqc = .false.
 436          call utl_tmg_start(91,'----QuasiNewton')
 437
 438          zeps1 = zeps0
 439
 440          call qna_n1qn3(simvar, dscalqn, dcanonb, dcanab, nvadim_mpilocal, vazx,  &
 441              zjsp,vazg, zxmin, zdf1, zeps1, impres, nulout, imode,       &
 442              iitnovqc, isimnovqc ,iztrl, vatra, nmtra, intUnused,   &
 443              zzsunused, dlds)
 444          call utl_tmg_stop(91)
 445          call fool_optimizer(obsSpaceData)
 446
 447          isimnovqc = isimnovqc - 1
 448          itermaxtodo = itermaxtodo - iitnovqc + 1
 449          isimmax = isimmax - isimnovqc + 1
 450
 451          itertot = itertot + iitnovqc
 452          isimtot = isimtot + isimnovqc
 453
 454          zeps0 = zeps0/zeps1
 455          lvarqc = .true.
 456
 457          if ((imode == 4 .or. imode == 1) .and. itertot < itermax) then
 458            imode = 2
 459            INDIC = 2
 460            call utl_tmg_start(91,'----QuasiNewton')
 461            call simvar(indic,nvadim_mpilocal,vazx,zjsp,vazg)
 462            call utl_tmg_stop(91)
 463          else
 464            write(*,*) 'minimization_mod: qna_n1qn3 imode = ', imode
 465            call utl_abort('minimization_mod: qna_n1qn3 mode not equal to 1 or 4')
 466          endif
 467        endif
 468
 469        ! Now do main minimization with var-QC
 470        call utl_tmg_start(91,'----QuasiNewton')
 471
 472        if ( outerLoopIndex > 1 ) imode = 2
 473
 474        zeps1 = zeps0
 475
 476        call qna_n1qn3(simvar, dscalqn, dcanonb, dcanab, nvadim_mpilocal, vazx,  &
 477            zjsp,vazg, zxmin, zdf1, zeps1, impres, nulout, imode,   &
 478            itermaxtodo,isimmax, iztrl, vatra, nmtra, intUnused, zzsunused,   &
 479            dlds)
 480        call utl_tmg_stop(91)
 481        call fool_optimizer(obsSpaceData)
 482
 483        itertot = itertot + itermaxtodo
 484        isimtot = isimtot + isimmax
 485
 486        zeps0 = zeps0/zeps1
 487
 488
 489        WRITE(*,FMT=9500) imode,iterdone,itertot-iterdone,itertot,isimdone,isimtot-isimdone,isimtot
 490 9500   FORMAT(//,20X,20('*'),2X    &
 491        ,/,20X,'              Minimization ended with MODE:',I4  &
 492        ,/,20X,' Number of iterations done in previous job:',I4  &
 493        ,/,20X,'          Number of iterations in this job:',I4  &
 494        ,/,20X,'                Total number of iterations:',I4  &
 495        ,/,20X,'Number of simulations done in previous job:',I4  &
 496        ,/,20X,'         Number of simulations in this job:',I4  &
 497        ,/,20X,'               Total number of simulations:',I4)
 498
 499        min_niter = itertot
 500
 501        ! Test the gradient at the final point
 502        if ( lgrtest .and. isMinimizationFinalCall ) then
 503          WRITE(*,FMT=9400)
 504 9400     FORMAT(//,12X,40('**'),/,12X,'TESTING THE GRADIENT AT THE FINAL POINT',/,40('**'))
 505          call utl_tmg_start(91,'----QuasiNewton')
 506          call grtest2(simvar,nvadim_mpilocal,vazx,ngrange)
 507          call utl_tmg_stop(91)
 508        end if
 509
 510        ! Print some contents of obsSpaceData to the listing
 511        if(mmpi_myid == 0) then
 512          do jdata = 1, min(1,obs_numHeader(obsSpaceData))
 513            call obs_prnthdr(obsSpaceData,jdata)
 514            call obs_prntbdy(obsSpaceData,jdata)
 515          end do
 516        endif
 517
 518      else
 519
 520        ! no analysis performed for ensemble mean, ensure mean increment is zero
 521        vazx(:) = 0.0d0
 522        min_niter = 0
 523
 524      endif ! if numIterMaxInnerLoopUsed > 0
 525
 526      ! deallocate the gradient
 527      deallocate(vazg)
 528      if ( deallocHessian .and. .not. lwrthess ) deallocate(vatra)
 529
 530  end subroutine quasiNewtonMinimization

 533  subroutine min_writeHessian(vazx)
 534    implicit none
 535
 536    real(8) :: vazx(:)
 537
 538    call utl_tmg_start(90,'--Minimization')
 539
 540    if ( lwrthess ) then
 541      ! Write out the Hessian to file
 542      if ( mmpi_myid == 0 ) write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 543
 544      ! zero array for writing to hessian
 545      if ( .not. allocated(controlVectorIncrSumZero) ) then
 546        allocate(controlVectorIncrSumZero(cvm_nvadim))
 547      end if
 548      controlVectorIncrSumZero(:) = 0.0d0
 549
 550      call hessianIO (preconFileNameOut,1,  &
 551        min_nsim,tim_getDatestamp(),zeps0,zdf1,itertot,isimtot,  &
 552        iztrl,vatra,controlVectorIncrSumZero,vazx,.true.,llvazx,n1gc,imode)
 553
 554      deallocate(controlVectorIncrSumZero)
 555
 556      if ( mmpi_myid == 0 ) write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 557    endif
 558
 559    if ( lwrthess ) then
 560      deallocate(vatra)
 561    end if
 562
 563    call utl_tmg_stop(90)
 564
 565  end subroutine min_writeHessian

 568  subroutine simvar(na_indic,na_dim,da_v,da_J,da_gradJ)
 569    !
 570    !:Purpose: To implement the Variational solver as described in
 571    !          Courtier, 1997, Dual Formulation of Four-Dimentional Variational
 572    !          Assimilation, Q.J.R., pp2449-2461.
 573    !
 574    !:Arguments:
 575    !    :na_indic:
 576    !               =1 No action taken
 577    !
 578    !               =2 Same as 4 (compute J and gradJ) but do not interrupt
 579    !                  timer of the minimizer.
 580    !
 581    !               =3 Compute Jo and gradJo only.
 582    !
 583    !               =4 Both J(u) and its gradient are computed.
 584    !
 585    !               .. Note:: 1 and 4 are reserved values for call back from
 586    !                         m1qn3. For direct calls, use a value other than 1
 587    !                         and 4.
 588    implicit none
 589
 590    ! Arguments:
 591    integer :: na_indic
 592    integer :: na_dim! Control-vector dimension, forecast-error covariance space
 593    real(8) :: da_v(na_dim) ! Control variable, forecast-error covariance space
 594    real*8  :: da_J ! Cost function of the Variational algorithm
 595    real(8) :: da_gradJ(na_dim) ! Gradient of the Variational Cost funtion
 596
 597    ! Locals:
 598    real*8, dimension(na_dim) :: dl_v
 599    real*8 :: dl_Jb, dl_Jo
 600    type(struct_gsv), save :: statevector
 601    type(struct_vco), pointer :: vco_anl
 602
 603    call utl_tmg_stop(91)
 604    call utl_tmg_stop(90)
 605
 606    if (na_indic .ne. 1) then ! No action taken if na_indic == 1
 607       min_nsim = min_nsim + 1
 608
 609       if(mmpi_myid == 0) then
 610         write(*,*) 'Entering simvar for simulation ',min_nsim
 611         write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
 612       endif
 613
 614       ! note: controlVectorIncrSum_ptr is sum of previous outer-loops
 615       dl_v(1:nvadim_mpilocal) = da_v(1:nvadim_mpilocal) + controlVectorIncrSum_ptr(1:nvadim_mpilocal)     
 616
 617       ! Computation of background term of cost function:
 618       dl_Jb = dot_product(dl_v(1:nvadim_mpilocal),dl_v(1:nvadim_mpilocal))/2.d0  
 619       call mmpi_allreduce_sumreal8scalar(dl_Jb,"GRID")
 620
 621       if (oneDVarMode) then
 622         call bmat1D_sqrtB(da_v, nvadim_mpilocal, columnAnlInc_ptr, obsSpaceData_ptr)
 623         call cvt_transform(columnAnlInc_ptr, 'ZandP_tl', columnTrlOnAnlIncLev_ptr)
 624       else
 625         if (.not.gsv_isAllocated(statevector)) then
 626           write(*,*) 'min-simvar: allocating increment stateVector'
 627           vco_anl => col_getVco(columnTrlOnAnlIncLev_ptr)
 628           call gsv_allocate(statevector, tim_nstepobsinc, hco_anl, vco_anl, &
 629                dataKind_opt=pre_incrReal, mpi_local_opt=.true.)
 630           call gio_readMaskFromFile(statevector,'./analysisgrid')
 631         end if
 632
 633         call bmat_sqrtB(da_v,nvadim_mpilocal,statevector)
 634
 635         ! put in columnAnlInc H_horiz dx
 636         call s2c_tl(statevector,columnAnlInc_ptr,columnTrlOnAnlIncLev_ptr,obsSpaceData_ptr)
 637       end if
 638
 639       ! Save as OBS_WORK: H_vert H_horiz dx = Hdx
 640       call utl_tmg_start(10,'--Observations')
 641       call utl_tmg_start(18,'----ObsOper_TL')
 642       call oop_Htl(columnAnlInc_ptr,columnTrlOnAnlIncLev_ptr,obsSpaceData_ptr,min_nsim, &
 643                    initializeLinearization_opt=initializeForOuterLoop) 
 644       call utl_tmg_stop(18)
 645       call utl_tmg_stop(10)
 646
 647       ! Calculate OBS_OMA from OBS_WORK : d-Hdx
 648       call res_compute(obsSpaceData_ptr)  
 649
 650       call bcs_calcbias_tl(da_v,OBS_OMA,obsSpaceData_ptr,columnTrlOnAnlIncLev_ptr)
 651
 652       ! Save as OBS_WORK : R**-1/2 (d-Hdx)
 653       call utl_tmg_start(10,'--Observations')
 654       call rmat_RsqrtInverseAllObs(obsSpaceData_ptr,OBS_WORK,OBS_OMA)  
 655       call utl_tmg_stop(10)
 656
 657       ! Store J-obs in OBS_JOBS : 1/2 * R**-1 (d-Hdx)**2
 658       call cfn_calcJo(obsSpaceData_ptr) 
 659
 660       ! Store modified J_obs in OBS_JOBS : -ln((gamma-exp(J))/(gamma+1)) 
 661       IF ( LVARQC ) THEN
 662         call vqc_NlTl(obsSpaceData_ptr)
 663       endif
 664
 665       dl_Jo = 0.d0
 666       call utl_tmg_start(90,'--Minimization')
 667       call utl_tmg_start(92,'----SumCostFunction')
 668       call cfn_sumJo(obsSpaceData_ptr,dl_Jo)
 669       da_J = dl_Jb + dl_Jo
 670       if (na_indic  ==  3) then
 671          da_J = dl_Jo
 672          IF(mmpi_myid == 0) write(*,FMT='(6X,"SIMVAR:  JO = ",G23.16,6X)') dl_Jo
 673       else
 674          da_J = dl_Jb + dl_Jo
 675          IF(mmpi_myid == 0) write(*,FMT='(6X,"SIMVAR:  Jb = ",G23.16,6X,"JO = ",G23.16,6X,"Jt = ",G23.16)') dl_Jb,dl_Jo,da_J
 676       endif
 677       call utl_tmg_stop(92)
 678       call utl_tmg_stop(90)
 679
 680       ! Modify OBS_WORK : R**-1 (d-Hdx)
 681       call utl_tmg_start(10,'--Observations')
 682       call rmat_RsqrtInverseAllObs(obsSpaceData_ptr,OBS_WORK,OBS_WORK)  
 683       call utl_tmg_stop(10)
 684
 685       IF ( LVARQC ) THEN
 686         call vqc_ad(obsSpaceData_ptr)
 687       endif
 688
 689       ! Calculate adjoint of d-Hdx (mult OBS_WORK by -1)
 690       call res_computeAd(obsSpaceData_ptr)  
 691
 692       call col_zero(columnAnlInc_ptr)
 693
 694       ! Put in column : -H_vert**T R**-1 (d-Hdx)
 695       call utl_tmg_start(10,'--Observations')
 696       call utl_tmg_start(19,'----ObsOper_AD')
 697       call oop_Had(columnAnlInc_ptr,columnTrlOnAnlIncLev_ptr,obsSpaceData_ptr, &
 698                    initializeLinearization_opt=initializeForOuterLoop)
 699       call utl_tmg_stop(19)
 700       call utl_tmg_stop(10)
 701
 702       ! Put in statevector -H_horiz**T H_vert**T R**-1 (d-Hdx)
 703       if (oneDVarMode) then
 704         ! no interpolation needed for 1Dvar case
 705       else
 706         call s2c_ad(statevector,columnAnlInc_ptr,columnTrlOnAnlIncLev_ptr,obsSpaceData_ptr)
 707       end if
 708
 709       da_gradJ(:) = 0.d0
 710       call bcs_calcbias_ad(da_gradJ,OBS_WORK,obsSpaceData_ptr)
 711
 712       if (oneDVarMOde) then
 713         call cvt_transform( columnAnlInc_ptr, 'ZandP_ad', columnTrlOnAnlIncLev_ptr)      ! IN
 714         call bmat1D_sqrtBT(da_gradJ, nvadim_mpilocal, columnAnlInc_ptr, obsSpaceData_ptr)
 715       else
 716         call bmat_sqrtBT(da_gradJ,nvadim_mpilocal,statevector)
 717       end if
 718
 719       if (na_indic .ne. 3) then
 720         da_gradJ(1:nvadim_mpilocal) = dl_v(1:nvadim_mpilocal) + da_gradJ(1:nvadim_mpilocal)
 721       endif
 722    endif
 723
 724    call utl_tmg_start(90,'--Minimization')
 725    call utl_tmg_start(91,'----QuasiNewton')
 726
 727    initializeForOuterLoop = .false.
 728
 729    if(mmpi_myid == 0) write(*,*) 'end of simvar'
 730
 731  end subroutine simvar

 840  subroutine hessianIO (cfname,status,                             &
 841                        nsim,kbrpstamp,zeps1,zdf1,itertot,isimtot, &
 842                        iztrl,vatra,vazxbar,vazx,llxbar,llvazx,k1gc,imode)
 843    !
 844    !:Purpose: Read-Write Hessian and increment (possibly for outer loop) on a
 845    !          file
 846    !
 847    !:Arguments:
 848    !   * i   cfname
 849    !   * i   status
 850    !   * i   nsim
 851    !   * io  kbrpstamp
 852    !   * i   zeps1
 853    !   * i   zdf1
 854    !   * i   itertot
 855    !   * i   isimtot
 856    !   * i   iztrl
 857    !   * i   vatra
 858    !   * i   vazxbar
 859    !   * i   vazx
 860    !   * i   llxbar
 861    !   * i   llvazx
 862    !   * i   k1gc
 863    !   * o   imode
 864    !
 865    implicit none
 866
 867    ! Arguments:
 868    character(len=*) :: cfname ! precon file
 869    integer status  ! = 0 if READ, = 1 if WRITE
 870    integer nsim    ! Number of simulations in QNA_N1QN3
 871    integer kbrpstamp ! Date
 872    real*8 :: zeps1    ! Parameter in QNA_N1QN3
 873    real*8 :: zdf1     ! Parameter in QNA_N1QN3
 874    integer itertot ! Parameter in QNA_N1QN3
 875    integer isimtot ! Parameter in QNA_N1QN3
 876    integer, target:: iztrl(5)     ! Localisation parameters for Hessian
 877    real*8, target :: vatra(nmtra) ! Hessian
 878    real*8, target :: vazxbar(nvadim_mpilocal) ! Vazx of previous loop
 879    real*8, target :: vazx(nvadim_mpilocal) ! Current state of the minimization
 880    logical llxbar  ! read in vaxzbar if dates are compatible
 881    logical llvazx  ! Logical to read vazx
 882    integer k1gc    ! Minimizer ID (2: m1qn2, 3: m1qn3)
 883    integer imode   ! If status=0, set imode=0 (no prec) or 2 (prec)
 884
 885    ! Locals:
 886    real*4, allocatable :: vatravec_r4_mpiglobal(:)
 887    real*4, allocatable :: vatra_r4(:)
 888    real*8, allocatable :: vazxbar_mpiglobal(:),vazx_mpiglobal(:)
 889
 890    integer :: ibrpstamp,ireslun, ierr, fnom, fclos
 891    integer :: nvadim_mpiglobal,nmtra_mpiglobal
 892    integer :: ivadim, itrunc
 893    integer :: ivamaj
 894    integer :: jvec, i1gc,ictrlvec,ii
 895    integer, dimension(10), target, save :: iztrl_io
 896
 897    character(len=3) :: cl_version
 898
 899    if (status == 0) then
 900      if (mmpi_myid == 0) write(*,*) 'Read  Hessian in min_hessianIO from file ', cfname
 901      call utl_tmg_start(93,'----ReadHess')
 902    elseif (status == 1) then
 903      if (mmpi_myid == 0) write(*,*) 'Write Hessian in min_hessianIO to file ', cfname
 904      call utl_tmg_start(94,'----WriteHess')
 905    else
 906      call utl_abort('min_hessianIO: status not valid ')
 907    endif
 908
 909    call rpn_comm_allreduce(nvadim_mpilocal,nvadim_mpiglobal,1,"mpi_integer","mpi_sum","GRID",ierr)
 910    call rpn_comm_allreduce(nmtra          ,nmtra_mpiglobal, 1,"mpi_integer","mpi_sum","GRID",ierr)
 911
 912    ireslun=0
 913
 914    if (status == 0) then
 915      !
 916      !- 1.  Read Hessian on processor 0 and distribute the data to the other processors
 917      !
 918
 919      if (mmpi_myid == 0) then
 920         !- Open the Hessian matrix file
 921         ierr = fnom(ireslun,cfname,'FTN+SEQ+UNF+OLD+R/O',0)
 922
 923         !- Checking version number
 924         read (ireslun) cl_version, i1gc
 925         if (trim(cl_version) == 'V5') then
 926            write(*,*) 'min_hessianIO: using single precision V5 Hessian'
 927         else if (trim(cl_version) == 'V4') then
 928            write(*,*) 'min_hessianIO: using DEPRECIATED single precision V5 Hessian'
 929         else
 930            write(*,*)
 931            write(*,*) 'min_hessianIO : Only V5 Hessian are supported, found ', trim(cl_version)
 932            call utl_abort('min_hessianIO')
 933         endif
 934
 935         if (i1gc == 3 .and. i1gc == k1gc) then
 936            write(*,*) trim(cl_version),' M1QN3'
 937         else
 938            write(*,*) 'Version, n1gc =',trim(cl_version),i1gc
 939            call utl_abort('min_hessianIO: Inconsistant input hessian')
 940         endif
 941
 942         rewind (ireslun)
 943
 944         read(ireslun) cl_version,i1gc,nsim,ibrpstamp,zeps1,zdf1,itertot,isimtot,ivadim,itrunc
 945         read(ireslun) ivamaj,iztrl_io
 946         if ((ivamaj.ne.nvamaj).or.(nvadim_mpiglobal.ne.ivadim)) then
 947            write(*,*) nvamaj,ivamaj,nvadim_mpiglobal,ivadim
 948            call utl_abort('min_hessianIO : ERROR, size of V5 Hessian not consistent')
 949         endif
 950
 951      end if
 952
 953      ! ibrpstamp and iztrl_io must be broadcasted
 954      call rpn_comm_bcast(ibrpstamp,  1, "MPI_INTEGER", 0, "GRID", ierr)
 955      call rpn_comm_bcast(iztrl_io , 10, "MPI_INTEGER", 0, "GRID", ierr)
 956
 957      !- Read the Hessian
 958      if(mmpi_myid == 0) then 
 959         write(*,*) 'min_hessianIO : reading Hessian'
 960         allocate(vatravec_r4_mpiglobal(nvadim_mpiglobal))
 961      else
 962         allocate(vatravec_r4_mpiglobal(1))
 963      end if
 964      allocate(vatra_r4(nvadim_mpilocal))
 965
 966      if (k1gc == 3) ictrlvec = 2*nvamaj+1
 967      do jvec = 1, ictrlvec
 968 
 969         if (mmpi_myid == 0) then
 970            read(ireslun) vatravec_r4_mpiglobal
 971         end if
 972
 973         call bmat_reduceToMPILocal_r4( vatra_r4,            & ! OUT
 974                                        vatravec_r4_mpiglobal) ! IN (contains data only on proc 0)
 975         !$OMP PARALLEL DO PRIVATE(ii)
 976         do ii = 1, nvadim_mpilocal
 977           vatra((jvec-1)*nvadim_mpilocal+ii) = real(vatra_r4(ii),8)
 978         enddo
 979         !$OMP END PARALLEL DO
 980
 981      end do
 982
 983      deallocate(vatravec_r4_mpiglobal)       
 984      deallocate(vatra_r4)       
 985
 986      imode = 2
 987      iztrl(:) = iztrl_io(1:5)
 988      if(k1gc == 3) then
 989         iztrl(1) = nvadim_mpilocal
 990         iztrl(2) = 0
 991         iztrl(3) = nvamaj
 992         iztrl(4) = iztrl_io(4)
 993         iztrl(5) = iztrl_io(5)
 994      end if
 995
 996      !- Read VAZXBAR 
 997      if (ibrpstamp == kbrpstamp .and. llxbar) then
 998         if (mmpi_myid == 0) then 
 999            write(*,*) 'min_hessianIO : reading vazxbar'
1000            allocate(vazxbar_mpiglobal(nvadim_mpiglobal))
1001            read(ireslun) vazxbar_mpiglobal
1002         else
1003            allocate(vazxbar_mpiglobal(1))
1004         end if
1005         call bmat_reduceToMPILocal( vazxbar,          & ! OUT
1006                                     vazxbar_mpiglobal ) ! IN (contains data only on proc 0)
1007         deallocate(vazxbar_mpiglobal)
1008      end if
1009
1010      !- Read VAZX
1011      if (ibrpstamp == kbrpstamp .and. llvazx) then
1012         if (mmpi_myid == 0) then 
1013            write(*,*) 'min_hessianIO : reading vazx'
1014            allocate(vazx_mpiglobal(nvadim_mpiglobal))
1015            read(ireslun) vazx_mpiglobal
1016         else
1017            allocate(vazx_mpiglobal(1))
1018         end if
1019         call bmat_reduceToMPILocal( vazx,           & ! OUT
1020                                     vazx_mpiglobal )  ! IN (contains data only on proc 0)
1021         deallocate(vazx_mpiglobal)
1022      end if
1023
1024      if (ibrpstamp /= kbrpstamp) then
1025         kbrpstamp = ibrpstamp
1026      end if
1027
1028      if (mmpi_myid == 0) ierr = fclos(ireslun)
1029
1030    elseif (status == 1) then
1031      !
1032      !- 2.  Write Hessian
1033      !
1034
1035      !- Open the Hessian matrix file on processor 0 and write some metadata 
1036      if (mmpi_myid == 0) ierr = fnom(ireslun,cfname, 'FTN+SEQ+UNF' , 0)
1037      cl_version = 'V5'
1038      itrunc=0
1039      iztrl_io(1: 5) = iztrl(:)
1040      iztrl_io(6:10) = 0 ! dummy values for hessian size legacy purposes
1041      if(mmpi_myid == 0) write(ireslun) cl_version,k1gc,nsim,kbrpstamp,zeps1,zdf1,itertot,isimtot,nvadim_mpiglobal,itrunc
1042      if(mmpi_myid == 0) write(ireslun) nvamaj,iztrl_io
1043
1044      !- Write the Hessian
1045      if (mmpi_myid == 0) then
1046         allocate(vatravec_r4_mpiglobal(nvadim_mpiglobal))
1047      else
1048         allocate(vatravec_r4_mpiglobal(1))
1049      endif
1050      allocate(vatra_r4(nvadim_mpilocal))
1051
1052      if (k1gc == 3) ictrlvec = 2*nvamaj+1
1053      do jvec = 1, ictrlvec
1054        !$OMP PARALLEL DO PRIVATE(ii)
1055        do ii = 1, nvadim_mpilocal
1056          vatra_r4(ii) = vatra((jvec-1)*nvadim_mpilocal+ii)
1057        enddo
1058        !$OMP END PARALLEL DO
1059        call bmat_expandToMPIGlobal_r4( vatra_r4,              & ! IN
1060                                        vatravec_r4_mpiglobal )  ! OUT
1061        if (mmpi_myid == 0) write(ireslun) vatravec_r4_mpiglobal
1062      enddo
1063      deallocate(vatravec_r4_mpiglobal)
1064      deallocate(vatra_r4)
1065
1066      !- Write VAZXBAR
1067      if (mmpi_myid == 0) then
1068         allocate(vazxbar_mpiglobal(nvadim_mpiglobal))
1069      else
1070         allocate(vazxbar_mpiglobal(1)) ! dummy
1071      end if
1072      call bmat_expandToMPIGlobal( vazxbar,          & ! IN
1073                                   vazxbar_mpiglobal ) ! OUT
1074      if (mmpi_myid == 0) write(ireslun) vazxbar_mpiglobal(1:nvadim_mpiglobal)
1075      deallocate(vazxbar_mpiglobal)
1076
1077      !- Write VAZX
1078      if (mmpi_myid == 0) then
1079         allocate(vazx_mpiglobal(nvadim_mpiglobal))
1080      else
1081         allocate(vazx_mpiglobal(1))
1082      end if
1083      call bmat_expandToMPIGlobal( vazx,          & ! IN
1084                                   vazx_mpiglobal ) ! OUT
1085      if (mmpi_myid == 0) write(ireslun) vazx_mpiglobal(1:nvadim_mpiglobal)
1086      deallocate(vazx_mpiglobal)
1087
1088      !- Close the Hessian matrix file
1089      if (mmpi_myid == 0) ierr = fclos(ireslun)
1090    else
1091      call utl_abort('min_hessianIO: status not valid')
1092    endif
1093
1094    if (status == 0) then
1095      call utl_tmg_stop(93)
1096    elseif (status == 1) then
1097      call utl_tmg_stop(94)
1098    endif
1099
1100  end subroutine hessianIO

1103  subroutine grtest2(simul,na_dim,da_x0,na_range)
1104  !
1105  !:Purpose: To compare the variation of the functional against what the
1106  !          gradient gives for small changes in the control variable. This test
1107  !          should be accurate for values as small as DLALPHA =  SQRT(machine
1108  !          precision). (see Courtier, 1987)
1109  !
1110  !:Arguments:
1111  !    :na_range:  the test will be carried over values of ALPHA ranging between
1112  !                10**(-NA_RANGE) < ALPHA < 0.1
1113  !
1114  implicit none
1115
1116  ! Arguments:
1117  external simul ! simulator: return cost function estimate and its gradient
1118  integer, intent(in) :: na_dim ! Size of the control vector
1119  real*8,  intent(in) :: da_x0(na_dim) ! Control vector
1120  integer, intent(in) :: na_range
1121
1122  ! Locals:
1123  integer :: nl_indic, nl_j
1124  real*8  :: dl_wrk(na_dim),dl_gradj0(na_dim), dl_x(na_dim)
1125  real*8  :: dl_J0, dl_J, dl_test, dl_start,dl_end
1126  real*8  :: dl_alpha, dl_gnorm0
1127
1128
1129  ! 1. Initialize dl_gradj0 at da_x0
1130  !    ------------------------------------
1131
1132  nl_indic = 2
1133  call simul(nl_indic,na_dim,da_x0,dl_j0,dl_gradj0)
1134  dl_gnorm0 = dot_product(dl_gradj0,dl_gradj0)
1135  call mmpi_allreduce_sumreal8scalar(dl_gnorm0,"GRID")
1136  dl_start = 1.d0
1137  dl_end   = 10.0d0**(-na_range)
1138  write(*,FMT=9100) dl_start,dl_end, dl_j0, dl_gnorm0
1139  ! 2. Perform the test
1140  !    ----------------
1141
1142  if(dl_gnorm0 == 0.d0)then
1143     write(*,FMT=9101)
1144     return
1145  end if
1146  write(*,FMT=9200)
1147  do  nl_j = 1, na_range
1148     dl_alpha = 10.0d0**(- nl_j)
1149     dl_x(:) = da_x0(:) - dl_alpha*dl_gradJ0(:)
1150     call simul(nl_indic,na_dim,dl_x,dl_j,dl_wrk)
1151     dl_test = (dl_j-dl_j0)/(-dl_alpha * dl_gnorm0)
1152     write(*,FMT=9201)nl_j, dl_alpha, dl_j, dl_test
1153  end do
1154
11559100 format(//,4X,&
1156          'GRTEST- The gradient is being tested for',&
1157          G23.16,' <= ALPHA <= ',G23.16,/,12X,&
1158          'Initial value of J(X):',1x,G23.16,4x,&
1159          'Norm of GRAD J(X)**2: ',G23.16)
11609101 format(/,4X,'-In GRTEST: gradient vanishes exactly',&
1161          '. Gradient test cannot be performed at this point')
11629200 format(/,4X,'J',8X,'ALPHA',11X,'J(X)',12X,'TEST')
1163
11649201 format(2X,'GRTEST: step',2X,I3,4X,G23.16,4X,G23.16,4X,&
1165          G23.16)
1166  return
