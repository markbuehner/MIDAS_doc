 55  subroutine cvm_setupSubVector(label, BmatrixType, dimVector)
 56    implicit none
 57
 58    ! Arguments
 59    character(len=*) :: label
 60    character(len=*) :: BmatrixType
 61    integer :: dimVector
 62
 63    ! Locals
 64    integer :: ierr, dimVector_mpiglobal
 65
 66    if ( numVectors == maxNumVectors ) then
 67      call utl_abort('cvm_setupSubVector: number of allocated subvectors already at maximum allowed')
 68    end if
 69
 70    call rpn_comm_allreduce(dimVector, dimVector_mpiglobal,  &
 71                            1, 'MPI_INTEGER', 'MPI_SUM', 'GRID', ierr)
 72
 73    ! just return if subVector dimension is zero on all MPI tasks
 74    if ( dimVector_mpiglobal == 0 ) return
 75
 76    numVectors = numVectors + 1
 77
 78    if ( any(cvm_vector(:)%label == label) ) then
 79      write(*,*) 'cvm_setupSubVector: label = ', trim(label)
 80      call utl_abort('cvm_setupSubVector: this label is already present')
 81    end if
 82
 83    cvm_vector(numVectors)%label = label
 84    cvm_vector(numVectors)%BmatrixType = BmatrixType
 85    cvm_vector(numVectors)%dimVector = dimVector
 86    cvm_vector(numVectors)%dimVector_mpiglobal = dimVector_mpiglobal
 87
 88    if ( numVectors == 1 ) then
 89      cvm_vector(numVectors)%subVectorBeg = 1
 90      cvm_vector(numVectors)%subVectorEnd = cvm_vector(numVectors)%dimVector
 91
 92      cvm_vector(numVectors)%subVectorBeg_mpiglobal = 1
 93      cvm_vector(numVectors)%subVectorEnd_mpiglobal = cvm_vector(numVectors)%dimVector_mpiglobal
 94    else
 95      cvm_vector(numVectors)%subVectorBeg = 1 + cvm_vector(numVectors-1)%subVectorEnd
 96      cvm_vector(numVectors)%subVectorEnd = cvm_vector(numVectors)%dimVector +   &
 97                                            cvm_vector(numVectors-1)%subVectorEnd
 98
 99      cvm_vector(numVectors)%subVectorBeg_mpiglobal = 1 + cvm_vector(numVectors-1)%subVectorEnd_mpiglobal
100      cvm_vector(numVectors)%subVectorEnd_mpiglobal = cvm_vector(numVectors)%dimVector_mpiglobal +  &
101                                                      cvm_vector(numVectors-1)%subVectorEnd_mpiglobal
102    end if
103
104    cvm_nvadim = sum(cvm_vector(1:numVectors)%dimVector)
105    cvm_nvadim_mpiglobal = sum(cvm_vector(1:numVectors)%dimVector_mpiglobal)
106
107    write(*,*) 'cvm_setupSubVector: '
108    write(*,*) '   added subVector with label                 = ', cvm_vector(numVectors)%label
109    write(*,*) '   added subVector of type                    = ', cvm_vector(numVectors)%Bmatrixtype
110    write(*,*) '   added subVector with dimension             = ', cvm_vector(numVectors)%dimVector
111    write(*,*) '   added subVector with dimension (mpiglobal) = ', cvm_vector(numVectors)%dimVector_mpiglobal
112    write(*,*) '   current total dimension             = ', cvm_nvadim
113    write(*,*) '   current total dimension (mpiglobal) = ', cvm_nvadim_mpiglobal
114
115  end subroutine cvm_setupSubVector

118  function cvm_indexFromLabel(subVectorLabel) result(subVectorIndex)
119    implicit none
120    integer :: subVectorIndex
121
122    ! Arguments
123    character(len=*) :: subVectorLabel
124
125    ! Locals
126    logical :: found
127
128    found = .false.
129    index_loop: do subVectorIndex = 1, numVectors
130      if ( trim(cvm_vector(subVectorIndex)%label) == trim(subVectorLabel) ) then
131        found = .true.
132        exit index_loop
133      end if
134    end do index_loop
135
136    if ( .not.found ) then
137      subVectorIndex = -1
138    end if
139
140  end function cvm_indexFromLabel

143  function cvm_subVectorExists(subVectorLabel) result(exists)
144    implicit none
145    logical :: exists
146
147    ! Arguments
148    character(len=*) :: subVectorLabel
149
150    ! Locals
151    integer :: subVectorIndex
152
153    subVectorIndex = cvm_indexFromLabel(subVectorLabel)
154
155    if ( subVectorIndex < 0 ) then
156      exists = .false.
157      return
158    end if
159
160    if ( cvm_vector(subVectorIndex)%dimVector_mpiglobal > 0 ) then
161      exists = .true.
162    else
163      exists = .false.
164    end if
165
166  end function cvm_subVectorExists

169  function cvm_getSubVector(controlVector,subVectorLabel) result(subVector)
170    implicit none
171    real*8, pointer :: subVector(:)
172
173    ! Arguments
174    character(len=*) :: subVectorLabel
175    real*8, target  :: controlVector(:)
176
177    ! Locals
178    integer         :: subVectorIndex, indexBeg, indexEnd
179
180    subVectorIndex = cvm_indexFromLabel(subVectorLabel)
181
182    if( subVectorIndex < 0 ) then
183      call utl_abort('cvm_getSubVector: invalid subVector label')
184    end if
185
186    indexBeg = cvm_vector(subVectorIndex)%subVectorBeg
187    indexEnd = cvm_vector(subVectorIndex)%subVectorEnd
188    subVector => controlVector(indexBeg:indexEnd)
189
190  end function cvm_getSubVector
