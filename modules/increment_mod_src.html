 17module increment_mod
 18  ! MODULE increment_mod (prefix='inc' category='1. High-level functionality')
 19  !
 20  !:Purpose: To add a 4D increment to a given 4D background/reference state and
 21  !           to output the results
 22  !
 23  use codePrecision_mod
 24  use midasMpi_mod
 25  use timeCoord_mod
 26  use gridStateVector_mod
 27  use gridStateVectorFileIO_mod
 28  use interpolation_mod
 29  use horizontalCoord_mod
 30  use verticalCoord_mod
 31  use humidityLimits_mod
 32  use utilities_mod
 33  use message_mod
 34  use gridVariableTransforms_mod
 35  use BMatrix_mod
 36  use varNamelist_mod
 37  implicit none
 38  save
 39  private
 40
 41  ! public procedures
 42  public :: inc_computeHighResAnalysis, inc_writeIncAndAnalHighRes, inc_getIncrement, inc_writeIncrement
 43  public :: inc_writeAnalysis, inc_analPostProcessing
 44
 45  ! namelist variables
 46  integer           :: writeNumBits          ! number of bits to use when writing analysis and high-res increment
 47  logical           :: writeHiresIncrement   ! choose to write the high-res increment to a file
 48  logical           :: imposeRttovHuLimits   ! choose to impose "rttov" humidity limits to analysis
 49  logical           :: useAnalIncMask        ! for LAM only, choose to apply scale factor from a mask file to the increment  
 50  character(len=12) :: etiket_anlm           ! 'etiket' used when writing the analysis
 51  character(len=12) :: etiket_rehm           ! 'etiket' used when writing the high-res increment
 52  character(len=12) :: etiket_rebm           ! 'etiket' used when writing the low-res increment
 53  character(len=12) :: hInterpolationDegree  ! type of interpolation to use: 'LINEAR' or 'CUBIC'

 55  logical           :: SSTSpread             ! choose to apply spatial spreading of the SST increment onto land
 56  integer           :: SSTSpreadMaxBoxSize   ! control the amount of SST increment spreading

 61  !--------------------------------------------------------------------------
 62  ! readNameList
 63  !--------------------------------------------------------------------------
 64  subroutine readNameList
 65    !
 66    !:Purpose: Reading NAMINC namelist by any subroutines in increment_mod module.
 67    !
 68    implicit none
 69
 70    integer :: nulnam, ierr
 71    integer, external :: fnom, fclos
 72    logical, save :: nmlAlreadyRead = .false.
 73    NAMELIST /NAMINC/ writeHiresIncrement, etiket_rehm, etiket_anlm, &
 74         etiket_rebm, writeNumBits, imposeRttovHuLimits, hInterpolationDegree, &
 75         useAnalIncMask, applyLiebmann, SSTSpread, SSTSpreadMaxBoxSize, SSTSubgrid
 76
 77    if ( .not. nmlAlreadyRead ) then
 78      nmlAlreadyRead = .true.
 79
 80      !- Setting default values
 81      writeHiresIncrement = .true.
 82      imposeRttovHuLimits = .true.
 83      useAnalIncMask      = .false.
 84      etiket_rehm = 'INCREMENT'
 85      etiket_rebm = 'INCREMENT'
 86      etiket_anlm = 'ANALYSIS'
 87      writeNumBits = 16
 88      hInterpolationDegree = 'LINEAR'
 89      applyLiebmann = .false.
 90      SSTSpread  = .false.
 91      SSTSpreadMaxBoxSize = 0
 92      SSTsubgrid = '   '
 93
 94      if ( .not. utl_isNamelistPresent('NAMINC','./flnml') ) then
 95        call msg('readNameList (inc)', &
 96             'NAMINC is missing in the namelist. The default values will be taken.', &
 97             mpiAll_opt=.false.)
 98      else
 99        ! Reading the namelist
100        nulnam = 0
101        ierr = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
102        read(nulnam, nml=naminc, iostat=ierr)
103        if ( ierr /= 0) call utl_abort('readNameList: Error reading namelist')
104        ierr = fclos(nulnam)
105      end if
106      if ( mmpi_myid == 0 ) write(*,nml=naminc)
107    end if
108
109  end subroutine readNameList

111  !--------------------------------------------------------------------------
112  ! inc_computeHighResAnalysis
113  !--------------------------------------------------------------------------
114  subroutine inc_computeHighResAnalysis( statevectorIncLowRes,                            & ! IN
115                                         stateVectorUpdateHighRes, stateVectorPsfcHighRes)  ! OUT
116    !
117    !:Purpose: Computing high-resolution analysis on the trial grid.
118    !
119    implicit none
120
121    ! Arguments:
122    type(struct_gsv), intent(in) :: statevectorIncLowRes
123    type(struct_gsv), intent(inout) :: stateVectorUpdateHighRes
124    type(struct_gsv), intent(out) :: stateVectorPsfcHighRes
125
126    ! Locals:
127    type(struct_gsv) :: statevectorRef, statevectorRefPsfc
128    type(struct_gsv) :: statevectorIncRefLowRes
129    type(struct_gsv) :: statevectorTrlRefVars
130    type(struct_gsv) :: statevectorTrlLowResTime
131    type(struct_gsv) :: statevectorTrlLowResVert
132    type(struct_gsv) :: statevector_mask
133    type(struct_gsv) :: stateVectorHighRes
134
135    type(struct_vco), pointer :: vco_trl, vco_inc
136    type(struct_hco), pointer :: hco_trl, hco_inc
137
138    real(pre_incrReal), pointer :: PsfcIncPrior(:,:,:,:), PsfcIncMasked(:,:,:,:)
139    real(pre_incrReal), pointer :: analIncMask(:,:,:)
140
141    integer              :: stepIndex, numStep_inc, numStep_trl
142    integer, allocatable :: dateStampList(:)
143
144    character(len=4), allocatable :: varNamesRef(:), varNamesPsfc(:)
145    logical  :: allocHeightSfc, refStateNeeded
146
147    call msg('inc_computeHighResAnalysis', 'START', verb_opt=2)
148    call msg_memUsage('inc_computeHighResAnalysis')
149
150    call utl_tmg_start(80,'--Increment')
151    call utl_tmg_start(81,'----ComputeHighResAnalysis')
152
153    ! Set/Read values for the namelist NAMINC
154    call readNameList
155
156    if ( gsv_isAllocated(stateVectorPsfcHighRes) ) then
157      call utl_abort('inc_computeHighResAnalysis: '&
158                      //'stateVectorPsfcHighRes should not be allocated yet')
159    end if
160
161    ! If P0 present it is infered the statevector is a 3D atmospheric state
162    refStateNeeded =  gsv_varExist(varName='P0')
163
164    ! Setup timeCoord module (date read from trial file)
165    numStep_inc = tim_nstepobsinc ! low-res time
166    numStep_trl = tim_nstepobs    ! high-res time
167    allocate(dateStampList(numStep_inc))
168    call tim_getstamplist(dateStampList,numStep_inc,tim_getDatestamp())
169
170    hco_trl => gsv_getHco(stateVectorUpdateHighRes)
171    vco_trl => gsv_getVco(stateVectorUpdateHighRes)
172    hco_inc => gsv_getHco(statevectorIncLowRes)
173    vco_inc => gsv_getVco(statevectorIncLowRes)
174
175    if (vco_trl%Vcode == 0 .or. .not. gsv_varExist(varName='P0')) then
176      allocHeightSfc = .false.
177    else
178      allocHeightSfc = stateVectorUpdateHighRes%heightSfcPresent
179    end if
180
181    ! Read the analysis mask (in LAM mode only) - N.B. different from land/sea mask!!!
182    if (.not. hco_trl%global .and. useAnalIncMask) then
183      call gio_getMaskLAM(statevector_mask, hco_trl, vco_trl, hInterpolationDegree)
184    end if
185
186    ref_building: if ( refStateNeeded ) then
187      ! Build a reference variables
188      !
189      ! - Allocate the target reference statevector for vertical interpolation
190      ! that will be done in inc_interpolateAndAdd.
191      ! The reference needs to have the vertical structure of the input which is the
192      ! increment, but since horizontal interpolation is done first, it needs the
193      ! trial horizontal structure.
194      if (vco_trl%Vcode == 5100) then
195        allocate(varNamesRef(4))
196        varNamesRef = (/'P0', 'P0LS', 'TT', 'HU'/)
197        allocate(varNamesPsfc(2))
198        varNamesPsfc = (/'P0','P0LS'/)
199      else
200        allocate(varNamesRef(3))
201        varNamesRef = (/'P0', 'TT', 'HU'/)
202        allocate(varNamesPsfc(1))
203        varNamesPsfc = (/'P0'/)
204      end if
205      call gsv_allocate( statevectorRef, numStep_inc, hco_trl, vco_inc, &
206                         dataKind_opt=pre_incrReal, &
207                         dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true.,  &
208                         varNames_opt=varNamesRef, allocHeightSfc_opt=allocHeightSfc, &
209                         hInterpolateDegree_opt=hInterpolationDegree )
210
211      ! - Restriction of increment to only reference state variables
212      call gsv_allocate( statevectorIncRefLowRes, numStep_inc, hco_inc, vco_inc,  &
213                         dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true.,  &
214                         dataKind_opt=pre_incrReal, varNames_opt=varNamesRef, &
215                         allocHeightSfc_opt=allocHeightSfc)
216      call gsv_copy(statevectorIncLowRes, statevectorIncRefLowRes, &
217                    allowVarMismatch_opt=.true.)
218
219      ! - Horizontal *only* interpolation of analysis increment (ref state variables)
220      !   using int_interp_gsv since it also does needed mpi transpositions
221      call msg('inc_computeHighResAnalysis', &
222               'horizontal interpolation of the Psfc increment', mpiAll_opt=.false.)
223      call int_interp_gsv(statevectorIncRefLowRes,statevectorRef)
224      call gsv_deallocate(statevectorIncRefLowRes)
225      !- Apply mask to increment in LAM
226      if (.not. hco_trl%global .and. useAnalIncMask) then
227        call gsv_getField(statevectorRef,PsfcIncPrior,'P0')
228        call gsv_getField(statevectorRef,PsfcIncMasked,'P0')
229        call gsv_getField(statevector_mask,analIncMask)
230        do stepIndex = 1, statevectorRef%numStep
231          PsfcIncMasked(:,:,1,stepIndex) = PsfcIncPrior(:,:,1,stepIndex) * analIncMask(:,:,1)
232        end do
233        if (gsv_varExist(statevectorRef,'P0LS')) then
234          call gsv_getField(statevectorRef,PsfcIncPrior,'P0LS')
235          call gsv_getField(statevectorRef,PsfcIncMasked,'P0LS')
236          do stepIndex = 1, statevectorRef%numStep
237            PsfcIncMasked(:,:,1,stepIndex) = PsfcIncPrior(:,:,1,stepIndex) * analIncMask(:,:,1)
238          end do
239        end if
240      end if
241
242      ! - Compute analysis of reference state variables to use for vertical interpolation
243      !   of increment
244      call msg('inc_computeHighResAnalysis', &
245           'Computing reference variables analysis to use for interpolation of increment', &
246           mpiAll_opt=.false.)
247
248      !   - build a restriction to reference state variables of trial
249      call gsv_allocate(statevectorTrlRefVars, numStep_trl, &
250                        hco_trl, vco_trl, dateStamp_opt=tim_getDateStamp(), &
251                        mpi_local_opt=.true., dataKind_opt=pre_incrReal, &
252                        varNames_opt=varNamesRef, allocHeightSfc_opt=allocHeightSfc )
253      call gsv_copy(stateVectorUpdateHighRes, statevectorTrlRefVars,&
254                    allowVarMismatch_opt=.true.)
255      !   - bring trial to low time res
256      call gsv_allocate(statevectorTrlLowResTime, numStep_inc, hco_trl, vco_trl,  &
257                        dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true.,  &
258                        dataKind_opt=pre_incrReal, varNames_opt=varNamesRef, &
259                        allocHeightSfc_opt=allocHeightSfc)
260      call gsv_copy(statevectorTrlRefVars, statevectorTrlLowResTime, &
261                    allowTimeMismatch_opt=.true.)
262      !   - bring trial to low res (increment) vertical structure
263      !     (no vertical interpolation reference needed as it is a full state,
264      !     not a incremental state)
265      call gsv_allocate(statevectorTrlLowResVert, numStep_inc, hco_trl, vco_inc,  &
266                        dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true.,  &
267                        dataKind_opt=pre_incrReal, varNames_opt=varNamesRef, &
268                        allocHeightSfc_opt=allocHeightSfc)
269      call int_vInterp_gsv(statevectorTrlLowResTime, statevectorTrlLowResVert)
270      call gsv_deallocate(statevectorTrlLowResTime)
271      call gsv_deallocate(statevectorTrlRefVars)
272      !   - build the analysis reference state for the increment vertical interpolation.
273      !     At that stage, statevectorRef contains the increment on the trial
274      !     horizontal grid, but analysis vertical structure; consistent for addition.
275      call gsv_add(statevectorTrlLowResVert, statevectorRef)
276      call gsv_deallocate(statevectorTrlLowResVert)
277
278      ! - Time interpolation to get high-res Psfc analysis increment to output
279      call msg('inc_computeHighResAnalysis', &
280               'Time interpolation to get high-res Psfc analysis increment', &
281               mpiAll_opt=.false.)
282
283      !   - Restriction to P0 only + vco set on vco_trl for later consistency
284      call gsv_allocate(statevectorRefPsfc, numStep_inc, hco_trl, vco_trl, &
285                        dataKind_opt=pre_incrReal, &
286                        dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true.,  &
287                        varNames_opt=varNamesPsfc, allocHeightSfc_opt=allocHeightSfc, &
288                        hInterpolateDegree_opt=hInterpolationDegree )
289      call gsv_copy(statevectorRef, statevectorRefPsfc, allowVarMismatch_opt=.true., &
290                    allowVcoMismatch_opt=.true.)
291      !   - high res time interpolation
292      call gsv_allocate(stateVectorPsfcHighRes, numStep_trl, hco_trl, vco_trl, &
293                        dataKind_opt=pre_incrReal, &
294                        dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true.,  &
295                        varNames_opt=varNamesPsfc, allocHeightSfc_opt=allocHeightSfc, &
296                        hInterpolateDegree_opt=hInterpolationDegree)
297      call int_tInterp_gsv(statevectorRefPsfc, stateVectorPsfcHighRes)
298    else ref_building
299      call msg('inc_computeHighResAnalysis','P0 not present, not computing reference', &
300               verb_opt=3)
301    end if ref_building
302
303    ! Compute the analysis
304    call msg('inc_computeHighResAnalysis', 'compute the analysis', mpiAll_opt=.false.)
305
306    ! Copy of trial to be passed to inc_interpolateAndAdd
307    call gsv_allocate( stateVectorHighRes, numStep_trl, hco_trl, vco_trl, &
308                       dataKind_opt=stateVectorUpdateHighRes%dataKind, &
309                       dateStamp_opt=tim_getDateStamp(), &
310                       mpi_local_opt=stateVectorUpdateHighRes%mpi_local, &
311                       allocHeightSfc_opt=stateVectorUpdateHighRes%heightSfcPresent, &
312                       hInterpolateDegree_opt=hInterpolationDegree, &
313                       allocHeight_opt=.false., allocPressure_opt=.false. )
314    call gsv_copy( stateVectorUpdateHighRes,stateVectorHighRes, &
315                   allowVarMismatch_opt=.true.)
316
317    ! Interpolate low-res increments to high-res and add to the initial state
318    if (.not. hco_trl%global .and. useAnalIncMask) then
319      if ( refStateNeeded ) then
320        call inc_interpolateAndAdd(statevectorIncLowRes, stateVectorHighRes, &
321                                   statevectorRef_opt=statevectorRef, &
322                                   statevectorMaskLAM_opt=statevector_mask)
323      else
324        call inc_interpolateAndAdd(statevectorIncLowRes, stateVectorHighRes, &
325                                   statevectorMaskLAM_opt=statevector_mask)
326      end if
327    else
328      if ( refStateNeeded ) then
329        call inc_interpolateAndAdd(statevectorIncLowRes, stateVectorHighRes, &
330                                   statevectorRef_opt=statevectorRef)
331      else
332        call inc_interpolateAndAdd(statevectorIncLowRes, stateVectorHighRes)
333      end if
334    end if
335
336    call gsv_copy( stateVectorHighRes, stateVectorUpdateHighRes, &
337                   allowVarMismatch_opt=.true.)
338    call gsv_deallocate(stateVectorHighRes)
339
340    if ( gsv_isAllocated(statevectorRef) ) call gsv_deallocate(statevectorRef)
341    if ( gsv_isAllocated(statevector_mask) ) call gsv_deallocate(statevector_mask)
342
343    call utl_tmg_stop(81)
344    call utl_tmg_stop(80)
345
346    call msg('inc_computeHighResAnalysis', 'END', verb_opt=2)
347
348  end subroutine inc_computeHighResAnalysis

351  ! inc_analPostProcessing
352  !--------------------------------------------------------------------------
353  subroutine inc_analPostProcessing (stateVectorPsfcHighRes, stateVectorUpdateHighRes, & ! IN
354                                     stateVectorTrial, stateVectorPsfc, stateVectorAnal) ! OUT
355    !
356    !:Purpose: Post processing of the high resolution analysis including degrading
357    !           temporal resolution, variable transform, and humidity clipping.
358    !
359    implicit none
360
361    ! Arguments:
362    type(struct_gsv), intent(inout)  :: stateVectorPsfcHighRes
363    type(struct_gsv), intent(in)     :: stateVectorUpdateHighRes
364    type(struct_gsv), intent(out)    :: stateVectorTrial
365    type(struct_gsv), intent(out)    :: stateVectorPsfc
366    type(struct_gsv), intent(out)    :: stateVectorAnal
367
368    ! Locals:
369    type(struct_vco), pointer :: vco_trl => null()
370    type(struct_hco), pointer :: hco_trl => null()
371    character(len=4), allocatable :: varNamesPsfc(:)
372
373    real(pre_incrReal), pointer :: oceanIce_ptr(:,:,:,:)
374
375    logical  :: allocHeightSfc
376
377    call msg('inc_analPostProcessing', 'START', verb_opt=2)
378    call msg_memUsage('inc_analPostProcessing')
379
380    call utl_tmg_start(80,'--Increment')
381    call utl_tmg_start(82,'----AnalPostProcessing')
382
383    ! Re-read trials to make stateVectorTrial with degraded timesteps available
384    hco_trl => gsv_getHco(stateVectorUpdateHighRes)
385    vco_trl => gsv_getVco(stateVectorUpdateHighRes)
386    allocHeightSfc = ( vco_trl%Vcode /= 0 )
387
388    call gsv_allocate(stateVectorTrial, tim_nstepobsinc, hco_trl, vco_trl,  &
389                      dataKind_opt = pre_incrReal, &
390                      dateStamp_opt = tim_getDateStamp(), mpi_local_opt = .true., &
391                      allocHeightSfc_opt = allocHeightSfc, hInterpolateDegree_opt = 'LINEAR', &
392                      allocHeight_opt = .false., allocPressure_opt = .false.)
393    call gsv_zero(stateVectorTrial)
394    call gio_readTrials(stateVectorTrial)
395    call msg_memUsage('inc_analPostProcessing')
396
397    if (vco_trl%Vcode == 0 .or. .not. gsv_varExist(varName='P0')) then
398      allocHeightSfc = .false.
399    else
400      allocHeightSfc = stateVectorTrial%heightSfcPresent
401    end if
402
403    if (vco_trl%Vcode == 5100) then
404      allocate(varNamesPsfc(2))
405      varNamesPsfc = (/'P0','P0LS'/)
406    else
407      allocate(varNamesPsfc(1))
408      varNamesPsfc = (/'P0'/)
409    end if
410
411    ! Degrade timesteps stateVector high-res Psfc, and high-res analysis
412    if (gsv_varExist(varName='P0')) then
413      call gsv_allocate(stateVectorPsfc, tim_nstepobsinc, hco_trl, vco_trl, &
414                        dataKind_opt = pre_incrReal, &
415                        dateStamp_opt = tim_getDateStamp(), mpi_local_opt = .true.,  &
416                        varNames_opt = varNamesPsfc, allocHeightSfc_opt = allocHeightSfc, &
417                        hInterpolateDegree_opt = hInterpolationDegree)
418      call gsv_copy(stateVectorPsfcHighRes, stateVectorPsfc, &
419                    allowTimeMismatch_opt = .true., allowVarMismatch_opt=.true.)
420      call gsv_deallocate(stateVectorPsfcHighRes)
421    end if
422
423    call gsv_allocate(stateVectorAnal, tim_nstepobsinc, hco_trl, vco_trl,  &
424                      dataKind_opt = pre_incrReal, &
425                      dateStamp_opt = tim_getDateStamp(), mpi_local_opt = .true., &
426                      allocHeightSfc_opt = allocHeightSfc, hInterpolateDegree_opt = 'LINEAR', &
427                      allocHeight_opt = .false., allocPressure_opt = .false.)
428    call gsv_copy(stateVectorUpdateHighRes, stateVectorAnal, &
429                  allowVarMismatch_opt = .true., allowTimeMismatch_opt = .true.)
430
431    if (gsv_varExist(stateVectorAnal, 'GL')) then
432      ! Impose limits [0,1] on sea ice concentration analysis
433      call gsv_getField(stateVectorAnal, oceanIce_ptr, 'GL')
434      oceanIce_ptr(:,:,:,:) = min(oceanIce_ptr(:,:,:,:), 1.0d0)
435      oceanIce_ptr(:,:,:,:) = max(oceanIce_ptr(:,:,:,:), 0.0d0)
436    end if
437
438    if (applyLiebmann) then
439
440      ! Start the variable transformations
441      if (gsv_varExist(stateVectorAnal, 'GL')) then
442        if (gsv_varExist(stateVectorAnal, 'LG')) then
443          ! Compute the continuous sea ice concentration field (LG)
444          call gvt_transform(stateVectorAnal, 'oceanIceContinuous', stateVectorRef_opt = stateVectorTrial, varName_opt = 'LG')
445        end if
446      end if
447
448      ! Start the variable transformations
449      if(gsv_varExist(stateVectorAnal,'TM')) then
450        ! Compute the continuous SST field (TM)
451        call gvt_transform(stateVectorAnal, 'oceanIceContinuous', stateVectorRef_opt = stateVectorTrial, varName_opt = 'TM')
452      end if
453
454    end if
455
456    if (SSTSpread) then
457      if(gsv_varExist(stateVectorAnal,'TM')) then
458        ! Compute SST spreading (TM) on neigbouring land points
459        call gvt_transform(stateVectorAnal, 'SSTSpread', varName_opt = 'TM', maxBoxSize_opt = SSTSpreadMaxBoxSize, subgrid_opt = SSTSubgrid)
460      end if
461    end if
462
463    ! Convert all transformed variables into model variables (e.g. LVIS->VIS, LPR->PR) for analysis
464    call gvt_transform(stateVectorAnal, 'AllTransformedToModel', allowOverWrite_opt = .true.)
465
466    ! Impose limits on humidity analysis
467    call msg('inc_analPostProcessing', 'calling qlim_saturationLimit')
468    call qlim_saturationLimit(stateVectorAnal)
469    if (imposeRttovHuLimits) call qlim_rttovLimit(stateVectorAnal)
470
471    if (gsv_varKindExist('CH')) then
472      ! Apply boundaries to analysis of CH kind variables as needed.
473      call msg('inc_analPostProcessing', &
474           'applying minimum values to analysis for variables of CH kind')
475      call gvt_transform(stateVectorAnal, 'CH_bounds')
476    end if
477
478    call utl_tmg_stop(82)
479    call utl_tmg_stop(80)
480
481    call msg('inc_analPostProcessing', 'END', verb_opt=2)
482
483  end subroutine inc_analPostProcessing

485  !--------------------------------------------------------------------------
486  ! inc_writeIncAndAnalHighRes
487  !--------------------------------------------------------------------------
488  subroutine inc_writeIncAndAnalHighRes( stateVectorTrial, stateVectorPsfc, &
489                                         stateVectorAnal )
490    !
491    !:Purpose: Write the high-resolution analysis increments to the rehm file.
492    !
493    implicit none
494
495    ! Arguments:
496    type(struct_gsv), intent(inout), target :: stateVectorTrial
497    type(struct_gsv), intent(inout) :: stateVectorPsfc
498    type(struct_gsv), intent(inout) :: stateVectorAnal
499
500    ! Locals:
501    type(struct_gsv) :: stateVectorIncHighRes
502    type(struct_gsv) :: stateVector_1step_r4, stateVectorPsfc_1step_r4
503    character(len=4), allocatable :: varNamesPsfc(:)
504
505    type(struct_vco), pointer :: vco_trl => null()
506    type(struct_hco), pointer :: hco_trl => null()
507
508    integer              :: stepIndex, stepIndexBeg, stepIndexEnd, stepIndexToWrite, numStep
509    integer              :: dateStamp, numBatch, batchIndex, procToWrite
510    integer, allocatable :: dateStampList(:)
511
512    character(len=256)  :: incFileName, anlFileName
513    character(len=4)    :: coffset
514    character(len=4), pointer :: varNames(:)
515
516    real(8)             :: deltaHours
517    real(8), pointer    :: HeightSfc_increment(:,:), HeightSfc_trial(:,:)
518
519    logical  :: allocHeightSfc, writeHeightSfc
520
521    call msg('inc_writeIncAndAnalHighRes', 'START', verb_opt=2)
522    call msg_memUsage('inc_writeIncAndAnalHighRes')
523
524    call utl_tmg_start(80,'--Increment')
525    call utl_tmg_start(83,'----WriteIncAndAnalHighRes')
526
527    ! Setup timeCoord module (date read from trial file)
528    numStep = tim_nstepobsinc
529    allocate(dateStampList(numStep))
530    call tim_getstamplist(dateStampList,numStep,tim_getDatestamp())
531
532    hco_trl => gsv_getHco(stateVectorTrial)
533    vco_trl => gsv_getVco(stateVectorTrial)
534    if (vco_trl%Vcode == 0 .or. .not. gsv_varExist(varName='P0')) then
535      allocHeightSfc = .false.
536    else
537      allocHeightSfc = stateVectorTrial%heightSfcPresent
538    end if
539    writeHeightSfc = allocHeightSfc
540
541    if (vco_trl%Vcode == 5100) then
542      allocate(varNamesPsfc(2))
543      varNamesPsfc = (/'P0','P0LS'/)
544    else
545      allocate(varNamesPsfc(1))
546      varNamesPsfc = (/'P0'/)
547    end if
548
549    ! Convert all transformed variables into model variables (e.g. LVIS->VIS, LPR->PR) for original trial
550    call gvt_transform(stateVectorTrial,   'AllTransformedToModel',allowOverWrite_opt=.true.)
551
552    ! Copy the surface height from trial into stateVectorPsfc and stateVectorAnal
553    if ( allocHeightSfc ) then
554      HeightSfc_trial     => gsv_getHeightSfc(stateVectorTrial)
555      HeightSfc_increment => gsv_getHeightSfc(stateVectorPsfc)
556      HeightSfc_increment(:,:) = HeightSfc_trial(:,:)
557
558      nullify(HeightSfc_increment)
559      HeightSfc_increment => gsv_getHeightSfc(stateVectorAnal)
560      HeightSfc_increment(:,:) = HeightSfc_trial(:,:)
561    end if
562
563    ! reAllocate incHighRes with the names of the model variables (e.g. VIS, PR)
564    nullify(varNames)
565    call gsv_varNamesList(varNames, stateVectorAnal)
566    call gsv_allocate(stateVectorIncHighRes, numStep, hco_trl, vco_trl, &
567                      dataKind_opt=pre_incrReal,  &
568                      dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true., &
569                      allocHeightSfc_opt=allocHeightSfc, hInterpolateDegree_opt=hInterpolationDegree, &
570                      varNames_opt=varNames )
571
572    ! Recompute increments and write out the files in parallel with respect to time steps
573    call gsv_copy(stateVectorAnal, stateVectorIncHighRes)
574    call gsv_add(stateVectorTrial, stateVectorIncHighRes, -1.0d0)
575
576    ! figure out number of batches of time steps for writing
577    numBatch = ceiling(real(stateVectorAnal%numStep) / real(mmpi_nprocs))
578    call msg('inc_writeIncAndAnalHighRes', &
579         'writing will be done by number of batches = '//str(numBatch))
580
581    batch_loop: do batchIndex = 1, numBatch
582
583      stepIndexBeg = 1 + (batchIndex - 1) * mmpi_nprocs
584      stepIndexEnd = min(stateVectorAnal%numStep, stepIndexBeg + mmpi_nprocs - 1)
585      call msg('inc_writeIncAndAnalHighRes', 'batchIndex = '//str(batchIndex) &
586           //', stepIndexBeg = '//str(stepIndexBeg) &
587           //', stepIndexEnd = '//str(stepIndexEnd))
588
589      ! figure out which time step I will write, if any (-1 if none)
590      stepIndexToWrite = -1
591      do stepIndex = stepIndexBeg, stepIndexEnd
592        procToWrite = nint( real(stepIndex - stepIndexBeg) * real(mmpi_nprocs) / real(stepIndexEnd - stepIndexBeg + 1) )
593        if ( procToWrite == mmpi_myid ) stepIndexToWrite = stepIndex
594        call msg('inc_writeIncAndAnalHighRes', ' stepIndex = '//str(stepIndex) &
595             //', procToWrite = '//str(procToWrite), mpiAll_opt=.false.)
596      end do
597
598      ! determine date and allocate stateVector for storing just 1 time step, if I do writing
599      if ( stepIndexToWrite /= -1 ) then
600
601        dateStamp = stateVectorAnal%dateStampList(stepIndexToWrite)
602        call difdatr(dateStamp,tim_getDatestamp(),deltaHours)
603        if(nint(deltaHours*60.0d0).lt.0) then
604          write(coffset,'(I4.3)') nint(deltaHours*60.0d0)
605        else
606          write(coffset,'(I3.3)') nint(deltaHours*60.0d0)
607        endif
608
609        call gsv_allocate( stateVector_1step_r4, 1, stateVectorAnal%hco, stateVectorAnal%vco, &
610                           dateStamp_opt=dateStamp, mpi_local_opt=.false., dataKind_opt=4,        &
611                           allocHeightSfc_opt=allocHeightSfc, &
612                           varNames_opt=varNames )
613        call gsv_allocate( stateVectorPsfc_1step_r4, 1, stateVectorAnal%hco, stateVectorAnal%vco, &
614                           dateStamp_opt=dateStamp, mpi_local_opt=.false., dataKind_opt=4,        &
615                           varNames_opt=varNamesPsfc, allocHeightSfc_opt=allocHeightSfc )
616        call gsv_zero(stateVectorPsfc_1step_r4)
617      end if
618
619      ! transpose ANALYSIS data from Tiles to Steps
620      call gsv_transposeTilesToStep(stateVector_1step_r4, stateVectorAnal, stepIndexBeg)
621
622      ! write the ANALYSIS file for one timestep on all tasks with data
623      if ( stepIndexToWrite /= -1 ) then
624        call msg_memUsage('inc_writeIncAndAnalHighRes')
625        anlFileName = './anlm_' // trim(coffset) // 'm'
626        call gio_writeToFile( stateVector_1step_r4, trim(anlFileName), etiket_anlm,  &
627                              typvar_opt='A', writeHeightSfc_opt=writeHeightSfc, &
628                              numBits_opt=writeNumBits, containsFullField_opt=.true. )
629      end if
630
631      if (writeHiresIncrement) then
632
633        ! transpose INCREMENT data from Tiles to Steps
634        call gsv_transposeTilesToStep(stateVector_1step_r4, stateVectorIncHighRes, stepIndexBeg)
635
636        ! write the INCREMENT file for one timestep on all tasks with data
637        if ( stepIndexToWrite /= -1 ) then
638          call msg_memUsage('inc_writeIncAndAnalHighRes')
639          incFileName = './rehm_' // trim(coffset) // 'm'
640          call gio_writeToFile( stateVector_1step_r4, trim(incFileName), etiket_rehm,  &
641                                typvar_opt='R', &
642                                numBits_opt=writeNumBits, containsFullField_opt=.false. )
643        end if
644
645        if (gsv_varExist(varName='P0')) then
646
647          ! transpose ANALYSIS PSFC AND height_SFC ONLY data from Tiles to Steps
648          call gsv_transposeTilesToStep(stateVectorPsfc_1step_r4, stateVectorPsfc, stepIndexBeg)
649
650          ! Also write analysis value of Psfc and surface height to increment file
651          if ( stepIndexToWrite /= -1 ) then
652            call msg_memUsage('inc_writeIncAndAnalHighRes')
653            incFileName = './rehm_' // trim(coffset) // 'm'
654            call gio_writeToFile( stateVectorPsfc_1step_r4, trim(incFileName), etiket_rehm,  &
655                                  typvar_opt='A', writeHeightSfc_opt=writeHeightSfc, &
656                                  numBits_opt=writeNumBits, containsFullField_opt=.true. )
657          end if
658
659        end if
660
661      end if ! writeHiresIncrement
662
663      if ( stepIndexToWrite /= -1 ) then
664        call gsv_deallocate(stateVector_1step_r4)
665        call gsv_deallocate(stateVectorPsfc_1step_r4)
666      end if
667
668    end do batch_loop
669
670    call gsv_deallocate(stateVectorAnal)
671    if (gsv_varExist(varName='P0')) then
672      call gsv_deallocate(stateVectorPsfc)
673    end if
674    call gsv_deallocate(stateVectorIncHighRes)
675    call gsv_deallocate(stateVectorTrial)
676
677    call utl_tmg_stop(83)
678    call utl_tmg_stop(80)
679
680    call msg('inc_writeIncAndAnalHighRes', 'END', verb_opt=2)
681
682  end subroutine inc_writeIncAndAnalHighRes

685  ! inc_getIncrement
686  !--------------------------------------------------------------------------
687  subroutine inc_getIncrement(incr_cv,statevector_incr,nvadim_mpilocal)
688    !
689    !:Purpose: Get true analysis increment from control vector.
690    !
691    implicit none
692
693    ! arguments
694    real(8)          :: incr_cv(:)
695    type(struct_gsv) :: statevector_incr
696    integer          :: nvadim_mpilocal
697
698    call utl_tmg_start(80,'--Increment')
699    call utl_tmg_start(84,'----GetIncrement')
700
701    ! compute increment from control vector (multiply by B^1/2)
702    call bmat_sqrtB( incr_cv, nvadim_mpilocal, statevector_incr )
703
704    ! Compute new diagnotics based on NAMSTATE
705    if ( gsv_varExist(statevector_incr,'QR') .and. gsv_varExist(statevector_incr,'DD') ) then
706       call msg('inc_getIncrement', 'User is asking for Vort-Div analysis increment')
707       call gvt_transform( statevector_incr, & ! INOUT
708                           'UVtoVortDiv' )     ! IN
709       if ( gsv_varExist(statevector_incr,'PP') .and. gsv_varExist(statevector_incr,'CC') ) then
710          call msg('inc_getIncrement', 'User is asking for Psi-Chi analysis increment')
711          call gvt_transform( statevector_incr, & ! INOUT
712                              'VortDivToPsiChi')  ! IN
713       end if
714    end if
715
716    call utl_tmg_stop(84)
717    call utl_tmg_stop(80)
718
719  end subroutine inc_getIncrement

830  !--------------------------------------------------------------------------
831  subroutine inc_interpolateAndAdd(statevector_in,statevector_inout, &
832                                   statevectorRef_opt, statevectorMaskLAM_opt, &
833                                   scaleFactor_opt)
834    !
835    !:Purpose: Interpolate the low-resolution increments to trial grid and add to
836    !           get the high-resolution analysis.
837    !
838    implicit none
839
840    ! Arguments:
841    type(struct_gsv), target,   intent(in)    :: statevector_in
842    type(struct_gsv),           intent(inout) :: statevector_inout
843    type(struct_gsv), optional, intent(in)    :: statevectorRef_opt         ! Reference statevector providing optional fields (P0, TT, HU)
844    type(struct_gsv), optional, intent(in)    :: statevectorMaskLAM_opt
845    real(8),          optional, intent(in)    :: scaleFactor_opt
846
847    ! Locals:
848    type(struct_gsv)          :: statevector_in_hvInterp
849    type(struct_gsv)          :: statevector_in_hvtInterp
850    type(struct_gsv), target  :: statevector_in_withP0LS
851    type(struct_gsv), pointer :: statevector_in_ptr
852
853    character(len=4), pointer :: varNamesToInterpolate(:), varNamesToInterpolate_withP0LS(:)
854
855    call msg('inc_interpolateAndAdd', 'START', verb_opt=2)
856
857    ! Error traps
858    if ( .not. gsv_isAllocated(statevector_in) ) then
859      call utl_abort('inc_interpolateAndAdd: gridStateVector_in not yet allocated! Aborting.')
860    end if
861    if ( .not. gsv_isAllocated(statevector_inout) ) then
862      call utl_abort('inc_interpolateAndAdd: gridStateVector_inout not yet allocated! Aborting.')
863    end if
864    if ( present(statevectorRef_opt) ) then
865      if ( statevector_in%numstep /= statevectorRef_opt%numstep) then
866        call utl_abort('inc_interpolateAndAdd: statevector_in and statevectorRef_opt numstep inconsistent')
867      end if
868    end if
869
870    nullify(varNamesToInterpolate)
871    call vnl_varNamesFromExistList(varNamesToInterpolate, statevector_in%varExistlist(:))
872
873    ! Check if P0LS needs to be added to the increment
874    if (statevector_inout%vco%vcode == 5100 .and. &
875         .not.gsv_varExist(statevector_in, 'P0LS')) then
876      varNamesToInterpolate_withP0LS => vnl_addToVarNames(varNamesToInterpolate,'P0LS')
877      call gsv_allocate(statevector_in_withP0LS, statevector_in%numstep,                       &
878                        statevector_in%hco, statevector_in%vco,                                &
879                        dateStamp_opt=tim_getDateStamp(),                                      &
880                        mpi_local_opt=statevector_in%mpi_local, mpi_distribution_opt='Tiles',  &
881                        dataKind_opt=statevector_in%dataKind,                                  &
882                        allocHeightSfc_opt=statevector_in%heightSfcPresent,                    &
883                        varNames_opt=varNamesToInterpolate_withP0LS,                           &
884                        hInterpolateDegree_opt=statevector_in%hInterpolateDegree)
885      call gsv_zero(statevector_in_withP0LS)
886      call gsv_copy(statevector_in, statevector_in_withP0LS, &
887                    allowVarMismatch_opt=.true.)
888      statevector_in_ptr => statevector_in_withP0LS
889      deallocate(varNamesToInterpolate)
890      varNamesToInterpolate => varNamesToInterpolate_withP0LS
891    else
892      statevector_in_ptr => statevector_in
893    end if
894    ! Do the spatial interpolation of statevector_in onto the grid of statevector_inout
895    call gsv_allocate(statevector_in_hvInterp, statevector_in%numstep,                          &
896                      statevector_inout%hco, statevector_inout%vco,                             &
897                      dateStamp_opt=tim_getDateStamp(),                                         &
898                      mpi_local_opt=statevector_inout%mpi_local, mpi_distribution_opt='Tiles',  &
899                      dataKind_opt=statevector_inout%dataKind,                                  &
900                      allocHeightSfc_opt=statevector_inout%heightSfcPresent,                    &
901                      varNames_opt=varNamesToInterpolate,                                       &
902                      hInterpolateDegree_opt=statevector_inout%hInterpolateDegree,              &
903                      hExtrapolateDegree_opt='VALUE' )
904
905    call int_interp_gsv(statevector_in, statevector_in_hvInterp, &
906                        statevectorRef_opt=statevectorRef_opt)
907
908    ! Do the time interpolation
909    call gsv_allocate(statevector_in_hvtInterp, statevector_inout%numstep,                      &
910                      statevector_inout%hco, statevector_inout%vco,                             &
911                      dateStamp_opt=tim_getDateStamp(),                                         &
912                      mpi_local_opt=statevector_inout%mpi_local, mpi_distribution_opt='Tiles',  &
913                      dataKind_opt=statevector_inout%dataKind,                                  &
914                      allocHeightSfc_opt=statevector_inout%heightSfcPresent,                    &
915                      varNames_opt=varNamesToInterpolate,                                       &
916                      hInterpolateDegree_opt=statevector_inout%hInterpolateDegree,              &
917                      hExtrapolateDegree_opt='VALUE' )
918    call int_tInterp_gsv(statevector_in_hvInterp, statevector_in_hvtInterp)
919    call gsv_deallocate(statevector_in_hvInterp)
920
921    ! Masking
922    if (present(statevectorMaskLAM_opt)) then
923      call gsv_applyMaskLAM(statevector_in_hvtInterp,statevectorMaskLAM_opt)
924    end if
925
926    ! Do the summation
927    call gsv_add(statevector_in_hvtInterp,statevector_inout,scaleFactor_opt=scaleFactor_opt)
928
929    call gsv_deallocate(statevector_in_hvtInterp)
930    deallocate(varNamesToInterpolate)
931
932    call msg('inc_interpolateAndAdd', 'END', verb_opt=2)
933
934  end subroutine inc_interpolateAndAdd
