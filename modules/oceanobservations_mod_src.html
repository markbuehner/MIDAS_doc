 45  integer, external :: fnom, fclos

 48  integer :: myLatBeg, myLatEnd
 49  integer :: myLonBeg, myLonEnd

 54  !----------------------------------------------------------------------------------------
 55  ! oobs_pseudoSST
 56  !----------------------------------------------------------------------------------------
 57  subroutine oobs_pseudoSST(hco, vco, iceFractionThreshold, outputSST, outputFreshWaterST, &
 58                            iceThinning, outputFileName, seaWaterThreshold)
 59    !
 60    !:Purpose: to generate pseudo SST data  
 61    !           
 62   
 63    implicit none
 64    
 65    ! Arguments
 66    type(struct_hco) , intent(inout), pointer :: hco                  ! horizontal grid structure
 67    type(struct_vco) , intent(in)   , pointer :: vco                  ! vertical grid structure
 68    real(4)          , intent(in)             :: iceFractionThreshold ! consider no ice condition below this threshold
 69    real(4)          , intent(in)             :: outputSST            ! output SST value for pseudo observations
 70    real(4)          , intent(in)             :: outputFreshWaterST   ! output fresh water surface temperature for pseudo observations
 71    integer          , intent(in)             :: iceThinning          ! generate pseudo obs in every 'iceThinning' points   
 72    character(len=*) , intent(in)             :: outputFileName    
 73    real(4)          , intent(in)             :: seaWaterThreshold    ! to distinguish inland water from sea water  
 74    
 75    ! Locals:
 76    type(struct_gsv)            :: stateVector_ice, stateVector_seaWater
 77    real(4), pointer            :: seaIce_ptr(:, :, :), seaWater_ptr(:, :, :)
 78    type(struct_ocm)            :: oceanMask
 79    integer                     :: numberIceCoveredPoints, lonIndex, latIndex, dateStamp, inlandWaterPoints
 80    integer                     :: datePrint, timePrint, imode, seaWaterPoints
 81    integer                     :: randomSeed, newDate, ierr 
 82    integer, allocatable        :: iceDomainIndexesAux(:), iceDomainIndexes(:)
 83    real(4), allocatable        :: seaWaterFractionAux(:), iceLonsAux(:), iceLatsAux(:)  
 84    real(4), allocatable        :: seaWaterFraction(:), iceLons(:), iceLats(:)
 85    type(struct_obs)            :: obsData   
 86    
 87    ! get mpi topology
 88    call mmpi_setup_lonbands(hco%ni, lonPerPE, lonPerPEmax, myLonBeg, myLonEnd)
 89    call mmpi_setup_latbands(hco%nj, latPerPE, latPerPEmax, myLatBeg, myLatEnd)
 90
 91    ! get latest sea-ice analysis
 92    call gsv_allocate(stateVector_ice, 1, hco, vco, dataKind_opt = 4, &
 93                      datestamp_opt = -1, mpi_local_opt = .false., varNames_opt = (/'LG'/))
 94    call gio_readFromFile(stateVector_ice, './seaice_analysis', ' ','A', &
 95                          unitConversion_opt=.false., containsFullField_opt=.true.)
 96    call gsv_getField(stateVector_ice, seaIce_ptr)
 97
 98    ! read sea water fraction
 99    call gsv_allocate(stateVector_seaWater, 1, hco, vco, dataKind_opt = 4, &
100                      datestamp_opt = -1, mpi_local_opt = .false., varNames_opt = (/'VF'/))
101    call gio_readFromFile(stateVector_seaWater, './seaice_analysis', ' ','A', &
102                          unitConversion_opt=.false., containsFullField_opt=.true.)
103    call gsv_getField(stateVector_seaWater, seaWater_ptr)
104
105    ! Get land mask from analysisgrid file (1=water, 0=land)
106    call ocm_readMaskFromFile(oceanMask, hco, vco, './analysisgrid')
107
108    allocate(iceDomainIndexesAux((myLonEnd - myLonBeg + 1) * (myLatEnd - myLatBeg + 1)))
109    allocate(iceLonsAux((myLonEnd - myLonBeg + 1) * (myLatEnd - myLatBeg + 1)))
110    allocate(iceLatsAux((myLonEnd - myLonBeg + 1) * (myLatEnd - myLatBeg + 1)))
111    allocate(seaWaterFractionAux((myLonEnd - myLonBeg + 1) * (myLatEnd - myLatBeg + 1)))
112    
113    numberIceCoveredPoints = 0
114    inlandWaterPoints = 0
115    seaWaterPoints = 0
116    
117    do lonIndex = myLonBeg, myLonEnd 
118      do latIndex = myLatBeg, myLatEnd
119        if (oceanMask%mask(lonIndex, latIndex, 1)) then
120          if (seaice_ptr(lonIndex, latIndex, 1) > iceFractionThreshold) then
121	  
122            numberIceCoveredPoints = numberIceCoveredPoints + 1
123	    iceDomainIndexesAux(numberIceCoveredPoints) = numberIceCoveredPoints
124	    seaWaterFractionAux(numberIceCoveredPoints) = seaWater_ptr(lonIndex, latIndex, 1)
125
126	    if(seaWater_ptr(lonIndex, latIndex, 1) <= seaWaterThreshold) then
127	      inlandWaterPoints = inlandWaterPoints + 1
128	    else
129	      seaWaterPoints = seaWaterPoints + 1
130	    end if  	      
131	  
132	    iceLonsAux(numberIceCoveredPoints) = hco%lon2d_4 (lonIndex, latIndex)
133	    iceLatsAux(numberIceCoveredPoints) = hco%lat2d_4 (lonIndex, latIndex)	
134          
135	  end if
136        end if
137      end do
138    end do
139    call ocm_deallocate(oceanMask)
140    call gsv_deallocate(stateVector_ice)
141    call gsv_deallocate(stateVector_seaWater)
142    write(*,*) 'oobs_pseudoSST: ', numberIceCoveredPoints, ' ice-covered points found'
143    write(*,*) 'oobs_pseudoSST: where ', inlandWaterPoints, ' are inland water points'
144    write(*,*) 'oobs_pseudoSST: ', seaWaterPoints, ' sea water points found'
145
146    if (numberIceCoveredPoints > 0) then
147      allocate(iceDomainIndexes(1:numberIceCoveredPoints))
148      iceDomainIndexes(:) = iceDomainIndexesAux(1:numberIceCoveredPoints)
149      allocate(seaWaterFraction(1:numberIceCoveredPoints))      
150      seaWaterFraction(:) = seaWaterFractionAux(1:numberIceCoveredPoints)
151      allocate(iceLons(1:numberIceCoveredPoints))
152      allocate(iceLats(1:numberIceCoveredPoints))
153      iceLons(:) = iceLonsAux(1:numberIceCoveredPoints)
154      iceLats(:) = iceLatsAux(1:numberIceCoveredPoints)
155    end if  
156    deallocate(iceLonsAux)
157    deallocate(iceLatsAux)
158    deallocate(iceDomainIndexesAux)
159    deallocate(seaWaterFractionAux)
160        
161    dateStamp = tim_getDatestampFromFile('./seaice_analysis', varNameForDate_opt = 'LG')
162    write(*,*) 'oobs_pseudoSST: datestamp: ', dateStamp 
163    ! compute random seed from the date for randomly forming sea-ice subdomain
164    imode = -3 ! stamp to printable date and time: YYYYMMDD, HHMMSShh
165    ierr = newdate(dateStamp, datePrint, timePrint, imode)
166    timePrint = timePrint / 1000000
167    datePrint =  datePrint * 100 + timePrint
168    
169    ! Remove the century, keeping 2 digits of the year
170    randomSeed = datePrint - 100000000 * (datePrint / 100000000)
171    write(*,*) 'oobs_pseudoSST: datePrint, timePrint: ', datePrint, timePrint 
172
173    if (numberIceCoveredPoints > 0) then
174
175      call utl_randomOrderInt(iceDomainIndexes, randomSeed)
176      write(*,*) 'oobs_pseudoSST: seed for random shuffle of sea-ice points: ', randomSeed
177    
178      call oobs_computeObsData(obsData, iceDomainIndexes, iceLons, iceLats, &
179                               iceThinning, outputSST, outputFreshWaterST, &
180                               outputFileName, datePrint, timePrint, &
181                               seaWaterFraction, seaWaterThreshold, inlandWaterPoints)
182    else 
183    
184      call obs_initialize(obsData, numHeader_max = 0, numBody_max = 0, mpi_local= .true.)
185      call sqlr_writeEmptyPseudoSSTobsFile(obsData, 'SF', outputFileName)   
186
187    end if
188
189    if (numberIceCoveredPoints > 0) then
190      deallocate(iceLons)
191      deallocate(iceLats)
192      deallocate(iceDomainIndexes)
193      deallocate(seaWaterFraction)
194    end if  
195    
196    write(*,*) 'oobs_pseudoSST: done'
197    
198  end subroutine oobs_pseudoSST
199
200  !--------------------------------------------------------------------------
201  ! oobs_computeObsData
202  !--------------------------------------------------------------------------
203
204  subroutine oobs_computeObsData(obsData, iceDomainIndexes, iceLons, iceLats, iceThinning, &
205                                 outputSST, outputFreshWaterST, outputFileName, &
206                                 datePrint, timePrint, &
207                                 seaWaterFraction, seaWaterThreshold, inlandWaterPoints)
208    !
209    !:Purpose: pseudo SST data are put into obsSpaceData  
210    !          and written into an SQLite file
211    
212    implicit none
213    
214    ! Arguments
215    type(struct_obs) , intent(inout) :: obsData            ! obsSpaceData   
216    integer          , intent(in)    :: iceDomainIndexes(:)! array of the ice-covered point indexes
217    real(4)          , intent(in)    :: iceLons(:)         ! longitudes of sea ice 
218    real(4)          , intent(in)    :: iceLats(:)         ! latitudes of sea ice 
219    integer          , intent(in)    :: iceThinning        ! generate pseudo obs in every 'iceThinning' points   
220    real(4)          , intent(in)    :: outputSST          ! output SST value for pseudo observations
221    real(4)          , intent(in)    :: outputFreshWaterST ! output fresh water surface temperature for pseudo obs
222    character(len=*) , intent(in)    :: outputFileName    
223    integer          , intent(in)    :: datePrint
224    integer          , intent(in)    :: timePrint
225    real(4)          , intent(in)    :: seaWaterFraction(:)! sea water fraction data: 0: fresh water; 1: sea water
226    real(4)          , intent(in)    :: seaWaterThreshold  ! to distinguish inland water from sea water 
227    integer          , intent(in)    :: inlandWaterPoints  ! number of inland water points 
228     
229    ! Locals
230    real(pre_obsReal)           :: obsLon, obsLat
231    real(4)                     :: obsValue
232    integer                     :: iceIndex, iceDomainDimension, pseudoObsDimension
233    integer                     :: codeType, headerIndex
234    integer                     :: coordinatesIndex, counterThinning, checkInlandWatersCount, checkSeaWatersCount
235    character(len=*), parameter :: myName = 'oobs_computeObsData'
236    
237    iceDomainDimension = size(iceDomainIndexes)
238    pseudoObsDimension = floor(real((iceDomainDimension - inlandWaterPoints) / iceThinning)) + inlandWaterPoints
239      
240    write(*,*) myName//': sea-ice domain dimension: ', iceDomainDimension
241    write(*,*) myName//': pseudo obs vector dimension: ', pseudoObsDimension
242    write(*,*) myName//': pseudo SST obs will be generated in every ', iceThinning, &
243    ' points of the sea-ice field for sea water, '
244    write(*,*) myName//': and in every point for inland waters, where sea water fraction <= ', seaWaterThreshold
245    write(*,*) myName//': number of inland waters points: ', inlandWaterPoints  
246    
247    call obs_initialize(obsData, numHeader_max = pseudoObsDimension, numBody_max = pseudoObsDimension, mpi_local= .true.)
248    codeType = codtyp_get_codtyp('pseudosfc')
249    
250    headerIndex = 1
251    counterThinning = iceThinning
252    checkInlandWatersCount = 0
253    checkSeaWatersCount = 0
254    
255    do iceIndex = 1, iceDomainDimension 
256
257      if (headerIndex > pseudoObsDimension ) cycle
258      
259      coordinatesIndex = iceDomainIndexes(iceIndex)
260      obsLon   = iceLons(coordinatesIndex)
261      obsLat   = iceLats(coordinatesIndex)
262 
263      if (seaWaterFraction(coordinatesIndex) <= seaWaterThreshold) then
264        obsValue = (1.0d0 - seaWaterFraction(coordinatesIndex)) * outputFreshWaterST + &
265                   seaWaterFraction(coordinatesIndex)* outputSST
266        checkInlandWatersCount =  checkInlandWatersCount + 1
267      else
268        if (counterThinning == iceThinning) then  
269          obsValue = outputSST
270          checkSeaWatersCount = checkSeaWatersCount + 1	
271          counterThinning = 1
272        else
273          counterThinning = counterThinning + 1
274          cycle
275        end if
276      end if
277
278      call obs_setFamily(obsData, 'SF'   , headerIndex)
279      call obs_headSet_i(obsData, OBS_ONM, headerIndex, headerIndex)
280      call obs_headSet_i(obsData, OBS_ITY, headerIndex, codeType)
281      call obs_headSet_r(obsData, OBS_LAT, headerIndex, obsLat)
282      call obs_headSet_r(obsData, OBS_LON, headerIndex, obsLon)
283      call obs_bodySet_r(obsData, OBS_VAR, headerIndex, obsValue)
284      call obs_bodySet_i(obsData, OBS_VNM, headerIndex, bufr_sst)  
285      call     obs_set_c(obsData, 'STID' , headerIndex, 'ABOG')
286      call obs_headSet_i(obsData, OBS_NLV, headerIndex, 1)
287      call obs_headSet_i(obsData, OBS_RLN, headerIndex, headerIndex)
288      call obs_headSet_i(obsData, OBS_DAT, headerIndex, datePrint / 100)
289      call obs_headSet_i(obsData, OBS_ETM, headerIndex, timePrint)
290      headerIndex = headerIndex + 1
291    end do 
292    
293    call sqlr_writePseudoSSTobs(obsData, 'SF', outputFileName) 
294 
295    ! Deallocate obsSpaceData
296    call obs_finalize(obsData)
297
298  end subroutine oobs_computeObsData
299
300end module oceanObservations_mod
