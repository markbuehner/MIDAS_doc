177  function bufr_IsAtmosConstituent(varNumber) result(var_chm)
178    !
179    !:Purpose: To determine whether 'varNumber' refers to constituent data from
180    !          the CH family with recognized data units.
181    !
182    implicit none
183
184    ! Arguments:
185    integer, intent(in)           :: varNumber ! BUFR element number
186    logical                       :: var_chm
187      
188    if (any(varNumber.eq. (/ BUFR_UNIT_VMR, BUFR_UNIT_VMR2, BUFR_UNIT_MolePerMole, BUFR_UNIT_MolePerMole2, &
189                             BUFR_UNIT_MMR, BUFR_UNIT_MMR2, BUFR_UNIT_NumberDensity, BUFR_UNIT_MolarDensity,  &
190                             BUFR_UNIT_Density, BUFR_UNIT_Density2, &
191                             BUFR_UNIT_AirDensity, BUFR_UNIT_PMDensity, &
192                             BUFR_UNIT_OptDepth, BUFR_UNIT_OptDepth2, BUFR_UNIT_OptDepth3, BUFR_UNIT_MR_NVaerosol, &
193                             BUFR_UNIT_PartPress, BUFR_UNIT_PartPress2, &
194                             BUFR_UNIT_DU, BUFR_UNIT_DU2, BUFR_UNIT_DU3, BUFR_UNIT_DU4, &
195                             BUFR_UNIT_IntegND, BUFR_UNIT_IntegND2, &
196                             BUFR_UNIT_IntegDens, BUFR_UNIT_IntegDens2, BUFR_UNIT_IntegDens3, &
197                             BUFR_UNIT_IntegMolarDens, BUFR_UNIT_ExtinctCoef, BUFR_UNIT_PhotoDissoc /) )) then          
198       var_chm=.true.
199    else         
200       var_chm=.false.
201    end if
202      
203  end function bufr_IsAtmosConstituent

206  logical function bufr_IsIntegral(varNumber)
207    !
208    !:Purpose: To identify whether obs is a vertically integrated constituent
209    !          measurement.
210    !
211    implicit none
212
213    ! Arguments:
214    integer, intent(in) :: varNumber ! BUFR element number
215 
216    if (any(varNumber .eq. (/ BUFR_UNIT_DU, BUFR_UNIT_DU2, BUFR_UNIT_DU3, BUFR_UNIT_DU4, &
217                              BUFR_UNIT_IntegND, BUFR_UNIT_IntegND2, BUFR_UNIT_IntegDens, &
218                              BUFR_UNIT_IntegDens2, BUFR_UNIT_IntegDens3, BUFR_UNIT_IntegMolarDens /) )) then      
219      bufr_IsIntegral=.true.     
220    else
221      bufr_IsIntegral=.false.
222    end if
223  
224  end function bufr_IsIntegral

227  logical function bufr_isWindComponent(varNumber)
228    !
229    !:Purpose: True if the variable is a wind component
230    !
231    implicit none
232    
233    ! Arguments:
234    integer, intent(in) :: varNumber ! BUFR element number
235
236    select case(varNumber)
237    case(BUFR_NEUU, BUFR_NEVV, BUFR_NEUS, BUFR_NEVS)
238      bufr_isWindComponent=.true.
239    case default
240      bufr_isWindComponent=.false.
241    end select
242
243  end function bufr_isWindComponent

246  logical function bufr_isOceanObs(varNumber)
247    !
248    !:Purpose: True if the variable is an ocean observation
249    !
250    implicit none
251    
252    ! Arguments:
253    integer, intent(in) :: varNumber ! BUFR element number
254
255    select case(varNumber)
256    case(BUFR_SST)
257      bufr_isOceanObs = .true.
258    case default
259      bufr_isOceanObs = .false.
260    end select
261
262  end function bufr_isOceanObs
