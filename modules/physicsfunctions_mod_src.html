  50   !--------------------------------------------------------------------------
  51   ! tetens_coefs_switch
  52   !--------------------------------------------------------------------------
  53   subroutine phf_tetens_coefs_switch
  54     !
  55     !:Purpose: Read water saturation strategy from nml. Options are:
  56     !
  57     !             - 'Tetens_1930' was active before 2018.
  58     !             - 'Tetens_2018a' is a partial update that missed some functions (active before IC4)
  59     !             - 'Tetens_2018' completes the update to the intended 2018 specification.
  60     !
  61     integer            :: NULNAM,IERR,FNOM,FCLOS
  62     character(len=256) :: NAMFILE
  63     logical            :: validOption
  64     NAMELIST /NAMPHY/ saturationCurve
  65
  66     !$omp critical
  67     if (.not.phf_initialized) then
  68
  69       saturationCurve = 'Tetens_2018'
  70
  71       if ( .not. utl_isNamelistPresent('NAMPHY','./flnml') ) then
  72         call msg( 'phf_tetens_coefs_switch', &
  73              'NAMPHY is missing in the namelist. Default values will be taken.', mpiAll_opt=.False.)
  74       else
  75         ! Reading the namelist
  76         nulnam = 0
  77         ierr = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
  78         read(nulnam, nml=namphy, iostat=ierr)
  79         if ( ierr /= 0) call utl_abort('tetens_coefs: Error reading namelist')
  80         ierr = fclos(nulnam)
  81       end if
  82
  83       validOption = (trim(saturationCurve) == 'Tetens_1930'  .or.  &
  84                      trim(saturationCurve) == 'Tetens_2018a' .or.  &
  85                      trim(saturationCurve) == 'Tetens_2018')
  86       if (.not.validOption) then
  87         call utl_abort('phf_tetens: WV Saturation not in expected list')
  88       end if
  89
  90       call msg( 'phf_tetens_coefs_switch ', saturationCurve )
  91       phf_initialized = .true.
  92     end if
  93     !$omp end critical
  94
  95   end subroutine phf_tetens_coefs_switch

 100   real*8 function phf_FOEW8(TTT)
 101     !
 102     !:Purpose: Water vapour saturation pressure (Tetens) - EW or EI as a fct of TT
 103     !
 104     implicit none
 105     real*8 TTT
 106
 107     if (.not.phf_initialized) call phf_tetens_coefs_switch
 108
 109     if (trim(saturationCurve) == 'Tetens_2018') then
 110       ! Updated coefficients 2018
 111       if (TTT > MPC_K_C_DEGREE_OFFSET_R8) then
 112         phf_FOEW8 = 610.94D0*EXP(17.625D0 * (TTT-MPC_TRIPLE_POINT_R8) / (TTT-30.11D0))
 113       else
 114         phf_FOEW8 = 610.94D0*EXP(22.587D0 * (TTT-MPC_TRIPLE_POINT_R8) / (TTT+ 0.71D0))
 115       end if
 116     else
 117       ! Classic Tetens 1930 coefficients
 118       if (TTT > MPC_TRIPLE_POINT_R8) then
 119         phf_FOEW8 = 610.78D0*EXP(17.269D0 * (TTT-MPC_TRIPLE_POINT_R8) / (TTT-35.86D0))
 120       else
 121         phf_FOEW8 = 610.78D0*EXP(21.875D0 * (TTT-MPC_TRIPLE_POINT_R8) / (TTT- 7.66D0))
 122       end if
 123     end if
 124   end function phf_foew8

 142         phf_FODLE8 = 6185.90D0/(TTT+ 0.71D0)**2
 143       end if
 144     else
 145       if (TTT > MPC_TRIPLE_POINT_R8) then
 146         phf_FODLE8 = 4097.93D0/(TTT-35.86D0)**2

 738  !--------------------------------------------------------------------------
 739  ! PHF_CONVERT_Z_TO_PRESSURE
 740  !--------------------------------------------------------------------------
 741  function  phf_convert_z_to_pressure(altitude,rgz_mod,press_mod,nlev,nlev_mod,lat,success) result(press)
 742    !          
 743    !:Purpose: Converts an array of (geometric) altitudes to pressures. Uses linear interpolation
 744    !          in log(p).  
 745    implicit none
 746
 747    real(8), intent(in) :: altitude(nlev)      ! altitudes to convert to pressures (m)
 748    real(8), intent(in) :: rgz_mod(nlev_mod)   ! geopotential heights on model levels (m), assumed to be in decending order
 749    real(8), intent(in) :: press_mod(nlev_mod) ! pressure on model levels, assumed to be in ascending order
 750    real(8), intent(in) :: lat                 ! latitude (rad)
 751    integer, intent(in) :: nlev                ! length of altitude array
 752    integer, intent(in) :: nlev_mod            ! number of model levels
 753    logical, intent(inout) :: success(nlev)    ! flag indicating success
 754    real(8) :: press(nlev)                     ! converted pressures
 755
 756    real(8) :: rgz(nlev)
 757    integer :: ilev,ilev_mod
 758
 759    ! Check model pressures
 760    if (any(press_mod.lt.0.0D0)) call utl_abort("phf_compute_z_to_pressure: Invalid model pressure.")
 761    
 762    ! Convert altitudes to geopotential heights
 763    rgz = phf_convert_z_to_gz(altitude,lat,nlev)
 764
 765    do ilev=1,nlev
 766
 767       ! Check if height is above or below model boundaries
 768       if ( rgz(ilev).gt.rgz_mod(1) .or. rgz(ilev).lt.rgz_mod(nlev_mod) ) then
 769          success(ilev)=.false.
 770       end if
 771
 772       if (success(ilev)) then
 773
 774          ! Find model layers directly above and below rgz(ilev).
 775          ! After exit of loop we will have 
 776          ! rgz_mod(ilev_mod) >= rgz(ilev) > rgz_mod(ilev_mod+1)
 777          do ilev_mod=1,nlev_mod-1
 778             if ( rgz(ilev).le.rgz_mod(ilev_mod) .and. &
 779                  rgz(ilev).gt.rgz_mod(ilev_mod+1) ) exit
 780          end do
 781          
 782          ! Linear interpolation in gz,log(p)
 783          press(ilev) = press_mod(ilev_mod+1) * (press_mod(ilev_mod)/press_mod(ilev_mod+1))**( &
 784               (rgz(ilev)-rgz_mod(ilev_mod+1))/(rgz_mod(ilev_mod)-rgz_mod(ilev_mod+1)) )
 785
 786       else
 787          press(ilev) = 0.0
 788       end if
 789
 790    end do
 791
 792  end function phf_convert_z_to_pressure

 794  !--------------------------------------------------------------------------
 795  ! PHF_CONVERT_Z_TO_GZ
 796  !--------------------------------------------------------------------------
 797  function phf_convert_z_to_gz(altitude,lat,nlev) result(rgz)
 798    !          
 799    !:Purpose: Converts altitudes to geopotential heights. Uses the Helmert formula to
 800    !          parameterize the latitude dependence and uses analytical result of the
 801    !          integral of \int g(z)dz for the altitude dependence (see J.A. Dutton 1976,
 802    !          p.65). At an altitude of 50 km, the altitude and geopotential height
 803    !          differ by around 0.2-0.5 km, depending on the latitude.
 804    !
 805    implicit none
 806
 807    real(8), intent(in) :: altitude(nlev) ! altitudes (m)
 808    real(8), intent(in) :: lat            ! latitude (rad)
 809    integer, intent(in) :: nlev           ! number of levels
 810    real(8) :: rgz(nlev)                  ! geopotential heights (m)
 811
 812    rgz = (ec_rg/9.8) * (1.-2.64D-03*cos(2.*lat)+5.9D-6*cos(2.*lat)**2) * ec_ra*altitude/(ec_ra+altitude)
 813
 814  end function phf_convert_z_to_gz

 817  ! PHF_GET_TROPOPAUSE
 818  !--------------------------------------------------------------------------
 819  function phf_get_tropopause(nmodlev,pressmod,tt,height,hu_opt) result(tropo_press)
 820    !          
 821    !:Purpose: Determines pressure level of tropopause. 
 822    !          Final tropopause is taken as max pressure (lowest altitude) from the
 823    !          water vapour and temperature based tropopauses.
 824    !
 825    implicit none
 826
 827    integer, intent(in) :: nmodlev                   ! Number of model levels
 828    real(8), intent(in) :: pressmod(nmodlev)         ! Model pressure array (Pa)
 829    real(8), intent(in) :: tt(nmodlev)               ! Model temperature (Kelvin)
 830    real(8), intent(in) :: height(nmodlev)           ! Model height (m)
 831    real(8), intent(in), optional :: hu_opt(nmodlev) ! Model specific humidity
 832    real(8) :: tropo_press                           ! Tropopause level in Pa (output)
 833  
 834    integer :: itop,i,k,ilaps
 835    real(8) :: hu_ppmv1,hu_ppmv2,hu_ppmv3,xlaps,tropo_press_hu
 836    real(8), parameter :: press_min=6000.         ! Min tropoause pressure 60 hPa.; equivalent to ~ 20km
 837    real(8), parameter :: height_min=6000.0           ! Min tropopause level in meters.
 838    real(8), parameter :: ppmv_threshold=10.0     
 839    real(8), parameter :: tgrad_threshold=0.002   ! degrees/m (2 degrees/km)
 840    real(8), parameter :: consth=0.160754938e+07  ! conversion from mass mixing ratio to ppmv;  1.0e+06 / (18.015/28.96)
 841
 842    tropo_press=-1.0
 843    if (all(height.lt.0.0))  call utl_abort('phf_get_tropopause: Missing height for determining tropopause pressure')
 844
 845    ! Initialize tropopause pressure level using temperature gradient.
 846    ! Thermal tropopause is defined as the lowest level (above height_min) at which (1) the lapse rate decreases
 847    ! to <= 2 C/km and (2) the average lapse rate between this level and all higher levels within 2 km are <= 2 C/km. 
 848    ! Ref: International Meteorological Vocabulary (2nd ed.). Geneva: Secretariat of the World Meteorological
 849    !      Organization. 1992. p. 636. ISBN 92-63-02182-1.
 850    ! The second requirement, based on hu, may give levels that are to high (pressure too low) in the winter hemisphere.
 851
 852    do itop=3,nmodlev
 853       if (pressmod(itop).ge.press_min) exit
 854    end do
 855    itop=itop-1
 856       
 857    do i=nmodlev,itop+1,-1
 858       if (height(i)-height(nmodlev).lt.height_min) cycle
 859       xlaps=-(tt(i)-tt(i-1))/(height(i)-height(i-1))
 860       if (xlaps.le.tgrad_threshold) then
 861          ilaps=1
 862          do k=i-1,itop,-1
 863             if (height(k)-height(i).gt.2000.0) exit
 864             xlaps=xlaps-(tt(k)-tt(k-1))/(height(k)-height(k-1))
 865             ilaps=ilaps+1
 866          end do
 867          if (xlaps/ilaps.le.tgrad_threshold) exit
 868       end if
 869    enddo
 870    tropo_press=pressmod(i)
 871    
 872    ! Improve on tropopause pressure levels using specific humidity if available,
 873
 874    if (present(hu_opt)) then
 875    
 876      !  Use water vapour
 877      
 878       hu_ppmv1=0.0
 879       do i=itop,nmodlev
 880
 881          ! Convert specific humidity to ppmv mixing ratio.
 882          ! First apply r=q/(1-q) to convert to mass mixing ratio.
 883
 884          if (hu_opt(i).le.0.8.and.hu_opt(i).ge.0) then
 885               hu_ppmv2 = consth*hu_opt(i)/(1.0-hu_opt(i))
 886          else if (hu_opt(i).gt.0.8) then
 887               hu_ppmv2 = consth*0.8/(1.0-0.8)
 888          else if (hu_opt(i).lt.0.0) then
 889               hu_ppmv2 = 0.0
 890          end if
 891
 892          ! Check if transition point reached.
 893          ! Added requirement that levels below also satisfy this condition.
 894
 895          if (hu_ppmv2.ge.ppmv_threshold) then
 896             ilaps=1
 897             do k=i+1,nmodlev
 898                if (height(i)-height(k).gt.5000.0) exit
 899                if (hu_opt(k).le.0.8.and.hu_opt(k).ge.0) then
 900                   hu_ppmv3 = consth*hu_opt(k)/(1.0-hu_opt(k))
 901                else if (hu_opt(k).gt.0.8) then
 902                   hu_ppmv3 = consth*0.8/(1.0-0.8)
 903                else
 904                   hu_ppmv3=0.0
 905                end if
 906                if (hu_ppmv3.lt.ppmv_threshold) ilaps=0
 907             end do
 908             if (ilaps.eq.1) exit
 909          end if
 910          hu_ppmv1=hu_ppmv2
 911       end do
 912       
 913       if (hu_ppmv2.ge.ppmv_threshold.and.ilaps.eq.1) then
 914       
 915            ! Interpolate between levels
 916      
 917             if (abs(hu_ppmv2-hu_ppmv1).lt.0.1) hu_ppmv1=hu_ppmv2-0.1
 918!             tropo_press_hu=(log(pressmod(i))*(ppmv_threshold-hu_ppmv1)+ &
 919!                    log(pressmod(i-1))*(hu_ppmv2-ppmv_threshold)) &
 920!                   /(hu_ppmv2-hu_ppmv1)
 921!             tropo_press_hu=exp(tropo_press_hu)
 922             tropo_press_hu=pressmod(i)
 923             
 924             tropo_press=min(tropo_press,tropo_press_hu)
 925       else
 926          write(*,*) 'phf_get_tropopause: Level and specific humidity: ',itop,hu_ppmv2
 927          call utl_abort('phf_get_tropopause: Specific humidity too small.')
 928       end if
 929                             
 930    end if
 931    
 932  end function phf_get_tropopause

 935  ! PHF_GET_PBL
 936  !--------------------------------------------------------------------------
 937  function phf_get_pbl(nmodlev,pressmod,tt,height,hu_opt,uu_opt,vv_opt) result(pbl_press)
 938    ! 
 939    !:Purpose: Determines pressure level of planetary boundary layer using 
 940    !          a first threshold of 0.5 for the bulk Richadson number (after Mahrt, 1981; 
 941    !          requires availability of uu and vv). Threshold reduced to largest value
 942    !          between 0.25 and 0.5 if first not satisfied. 
 943    !
 944    !          If not found with this approach, applies a variant of the Heffter approach 
 945    !          described in Aliabadi et al (2016), with some local variation.
 946    !
 947    !          References:
 948    !
 949    !          * Aliabadi A.A., R.M. Staebler, J. de Grandpre, A. Zadra, and P.A.
 950    !            Vaillancourt, 2016: Comparison of Estimated Atmospheric
 951    !            Boundary Layer Mixing Height in teh Arctic and Southern Great
 952    !            Plains under Statisticallt Stable Conditions: Experimental
 953    !            and Numerical Aspects, Submitted to Atmosphere-Ocean (2015).
 954    !
 955    !          * Mahrt, L. 1981: Modelling depth of the stable boundary-layer,
 956    !            Bound-Lay. Meteorol., 21, 3-19
 957    !
 958    !          * Heffter, J.L.,1980: Transport layer depth calculations, Second 
 959    !            Joint Conference on Applications of Air Pollution Meteorology,
 960    !            New Orleans, LA, 24-27 March 1980. American Meteorological
 961    !            Society, Boston, MA.
 962    !       
 963    !          Comments: Currently assumes (uu,vv) midlayer levels approximately 
 964    !          at tt, height, and hu levels when size(uu).ne.nmodlev.
 965    !
 966    implicit none
 967
 968    integer, intent(in) :: nmodlev           ! Number of model levels for variables other than uu and vv
 969    real(8), intent(in) :: pressmod(nmodlev) ! Model pressure array (Pa)
 970    real(8), intent(in) :: tt(nmodlev)       ! Model temperature (Kelvin)
 971    real(8), intent(in) :: height(nmodlev)   ! Model height (meters)
 972    real(8), optional :: uu_opt(:)           ! Model zonal wind component (m/s)
 973    real(8), optional :: vv_opt(:)           ! Model meridional wind component (m/s)
 974    real(8), optional :: hu_opt(nmodlev)     ! Specific humidity
 975    real(8) :: pbl_press                     ! PBL level in Pa (output)
 976  
 977    integer :: itop,i,id,igradmax,inv,iRiBmax
 978    real(8) :: RiB1,RiB2,RiBmax,zs,thetavs,thetavh(nmodlev),us,vs,uv,hus,huh,gradmax,grad
 979    real(8), parameter :: kappa = 287.04/1004.67  ! R/Cp
 980    real(8), parameter :: RiB_threshold=0.5, reduced=0.5 
 981
 982    ! Imposed min presssure of PBL height of 200 hPa (extreme; PBL height should normally be under 3km)
 983    real(8), parameter :: press_min=20000.  
 984
 985    real(8) :: huw(nmodlev)
 986
 987    pbl_press=-1.0
 988     
 989    ! Set values for lowest prognostic level 
 990
 991    i = nmodlev   
 992    
 993    if (all(height.lt.0.0))  call utl_abort('phf_get_pbl: Missing height for determining PBL pressure')
 994
 995    ! Convert hu to mass mixing ratio
 996
 997    if (present(hu_opt)) then
 998       huw(:)=hu_opt(:)
 999    else
1000       huw(:)=0.0
1001    end if
1002    
1003    if (huw(i).le.0.8.and.huw(i).ge.0) then
1004        hus = huw(i)/(1.0-huw(i))
1005    else if (huw(i).gt.0.8) then
1006        hus = 0.8/(1.0-0.8)
1007    else if (huw(i).lt.0.0) then
1008        hus = 0.0
1009    end if
1010    zs = height(i)*0.001
1011    
1012    ! Potential virtual temperature at lowest prognostic level
1013
1014    thetavs = tt(i)*(1.D5/pressmod(i))**kappa* (1.0 + 0.61*hus )
1015    thetavh(nmodlev)=thetavs
1016
1017    ! Set max vertical level
1018
1019    do itop=2,nmodlev-1
1020       if (pressmod(itop).ge.press_min) exit
1021    end do
1022
1023    RiB1=0.0
1024    RiB2=0.0
1025    RiBmax=0.0
1026    iRiBmax=0
1027    if (present(uu_opt).and.present(vv_opt)) then
1028       id=nmodlev-size(uu_opt)
1029       if (id.gt.1.or.id.lt.0) then
1030          call utl_abort('phf_get_pbl: Unexpected number of UV levels, nmodlev = ' // trim(utl_str(nmodlev)) // ' , size(uu) = ' // trim(utl_str(size(uu_opt))) )    
1031       end if
1032       us = uu_opt(size(uu_opt))
1033       vs = vv_opt(size(vv_opt))
1034       
1035!      us,vs set to 0.0
1036!       us=0.0
1037!       vs=0.0
1038 
1039       ! Calc RiB from near-surface to level attaining RiB_threshold
1040 
1041       do i=nmodlev-1,itop,-1
1042  
1043           if (huw(i).le.0.8.and.huw(i).ge.0) then
1044               huh = huw(i)/(1.0-huw(i))
1045           else if (huw(i).gt.0.8) then
1046               huh = 0.8/(1.0-0.8)
1047           else if (huw(i).lt.0.0) then
1048               huh = 0.0
1049           end if
1050           thetavh(i) = tt(i)*(1.D5/pressmod(i))**kappa* ( 1.0 + 0.61*huh )
1051
1052           if (id.eq.0) then
1053               uv = max( (uu_opt(i)-us)**2 + (vv_opt(i)-vs)**2, 1.D-8 ) 
1054           else
1055              ! Take layer midpoint values
1056              uv = max( ((uu_opt(i)+uu_opt(i-1))/2.0-us)**2 + ((vv_opt(i)+vv_opt(i-1))/2.0-vs)**2, 1.D-8 ) 
1057           end if
1058         
1059           RiB2 = ec_rg * (thetavh(i)-thetavs) * (height(i)*0.001-zs) / (thetavs*uv)
1060           if (RiBmax.lt.RiB2.and.RiB2.ge.reduced*RiB_threshold) then
1061              RiBmax=RiB2
1062              iRiBmax=i
1063           end if
1064           if (RiB2.ge.RiB_threshold) exit
1065           RiB1=RiB2
1066       end do
1067    else
1068    
1069       ! Calc only theta
1070      
1071       do i=nmodlev-1,itop,-1
1072  
1073           if (huw(i).le.0.8.and.huw(i).ge.0) then
1074               huh = huw(i)/(1.0-huw(i))
1075           else if (huw(i).gt.0.8) then
1076               huh = 0.8/(1.0-0.8)
1077           else if (huw(i).lt.0.0) then
1078               huh = 0.0
1079           end if
1080           thetavh(i) = tt(i)*(1.D5/pressmod(i))**kappa* ( 1.0 + 0.61*huh )
1081       end do
1082    end if   
1083    
1084    if (RiB2.ge.RiB_threshold) then    
1085   
1086      !  Interpolate between levels
1087
1088       pbl_press=(log(pressmod(i))*(RiB_threshold-RiB1)+ &
1089               log(pressmod(i+1))*(RiB2-RiB_threshold)) &
1090               /(RiB2-RiB1)
1091       pbl_press=exp(pbl_press)
1092    else if (RiBmax.ge.reduced*RiB_threshold) then
1093       ! Apply to level with largest RiB between reduced*RiB_threshold and RiB_threshold
1094       pbl_press=pressmod(iRiBmax)
1095    else
1096    
1097       ! Estimate PBL level using the Heffter conditions:
1098       ! First find lowest inversion layer where dtheta>2K.
1099       ! If found, assign mid of layer as PBL level. 
1100       ! Otherwise, assign PBL level as that with largest
1101       ! theta gradient.
1102       
1103       i=nmodlev-1
1104       do while (i.gt.itop) 
1105          !if (thetavh(i)-thetavh(i+1).gt.0.0) then
1106          if ((thetavh(i)-thetavh(i+1))/(height(i)-height(i+1)).ge.0.005) then
1107             ! Near bottom of inversion layer found
1108             inv=i+1
1109             i=i-1
1110             !do while (thetavh(i)-thetavh(i+1).gt.0.0.and.i.gt.itop) 
1111             do while ((thetavh(i)-thetavh(i+1))/(height(i)-height(i+1)).ge.0.005.and.i.gt.itop) 
1112                 i=i-1
1113             end do
1114             if ((thetavh(i+1)-thetavh(inv)).gt.2.0)  then
1115                ! Apply  midlayer as PBL
1116                pbl_press=sqrt(pressmod(i+1)*pressmod(inv))
1117                exit
1118             end if 
1119          else
1120             i=i-1
1121          end if
1122       end do
1123       if (pbl_press.le.0.0) then
1124          gradmax=-1.D30
1125          igradmax=nmodlev-1
1126          do i=nmodlev-1,itop,-1
1127             grad=(thetavh(i)-thetavh(i+1))/(height(i)-height(i+1))
1128             if (gradmax.lt.grad) then
1129                gradmax=(thetavh(i)-thetavh(i+1))/(height(i)-height(i+1))
1130                igradmax=i
1131                if (grad.ge.0.005) then ! Check next layer as well
1132                   if ((thetavh(i-1)-thetavh(i))/(height(i-1)-height(i)).ge.0.005) exit
1133                end if
1134            end if
1135          end do          
1136          pbl_press=pressmod(igradmax)
1137          ! write(*,*) 'phf_get_pbl: Warning2 - Max allowed altitude reached for. ',pbl_press,igradmax,gradmax,RiB2,iRiBmax,RiBmax
1138       !else
1139       !   write(*,*) 'phf_get_pbl: Warning1 - Max allowed altitude reached for. ',pbl_press,i,RiB2,iRiBmax,RiBmax     
1140       end if
1141    end if
1142
1143  end function phf_get_pbl

1148  function phf_calcDistance(lat2, lon2, lat1, lon1) result(distanceInM)
1149    !
1150    !:Purpose: Compute the distance between two point on earth: (lat1,lon1) 
1151    !          and (lat2,lon2). Calcul utilisant la Formule d'Haversine.
1152    !
1153    !          Reference: R.W. Sinnott,'Virtues of Haversine',Sky and 
1154    !          Telescope, vol.68, no.2, 1984, p.159)
1155    !
1156    implicit none
1157
1158    ! arguments
1159    real(8) :: lat1, lon1, lat2, lon2
1160    real(8) :: distanceInM
1161
1162    ! locals
1163    real(8) :: dlat, dlon, a, c
1164
1165    dlon = lon2 - lon1
1166    dlat = lat2 - lat1
1167
1168    a = (sin(dlat/2.d0))**2 + cos(lat1)*cos(lat2)*(sin(dlon/2.d0))**2
1169    c = 2.d0 * atan2(sqrt(a),sqrt(1.d0-a))
1170    distanceInM = ec_ra * c
1171
1172  end function phf_calcDistance

1178    !
1179    !:Purpose: Compute the distance between two point on earth: (lat1,lon1)
1180    !          and (lat2,lon2). Using a quick and dirty formula good for

1240  ! PHF_HEIGHT2GEOPOTENTIAL
1241  !--------------------------------------------------------------------------
1242  subroutine phf_height2geopotential(altitude, latitude, geopotential, &
1243                                     printHeight)
1244    !
1245    !:Purpose: Geopotential energy at a given point.
1246    !          Result is based on the WGS84 approximate expression for the
1247    !          gravity acceleration as a function of latitude and altitude,
1248    !          integrated with the trapezoidal rule.
1249    !
1250    implicit none
1251    real(8), intent(in)   :: altitude(:)     ! altitude (m)
1252    real(8), intent(in)   :: latitude        ! latitude (rad)
1253    real(8), intent(inout):: geopotential(:) ! geopotential (m2/s2)
1254
1255    integer           :: nlev, nlev500m
1256    real(8), allocatable :: alt500m(:), gravity500m(:)
1257    real(8)           :: delAlt, aveGravity, sLat, gravity, gravityM1
1258    integer           :: i, ilev 
1259    logical, optional :: printHeight
1260    
1261
1262    nlev = size(altitude)
1263    sLat = sin(latitude)
1264
1265    nlev500m = int(altitude(nlev) / 500.D0)
1266    if ( nlev500m >= 1) then
1267      allocate(alt500m(0:nlev500m))
1268      allocate(gravity500m(0:nlev500m))
1269
1270      ! Calculate gravity and height of levels for 500m-layers
1271      do i = 0, nlev500m
1272        alt500m(i) = i * 500.0D0
1273        gravity500m(i) = phf_gravityalt(sLat, alt500m(i))
1274      enddo
1275
1276      geopotential(nlev) = 0.0D0
1277      ! integrate from surface on the 500m-layers untill below the desired altitude
1278      do i = 1, nlev500m
1279        delAlt = alt500m(i) - alt500m(i-1)
1280        aveGravity = 0.5D0 * (gravity500m(i) + gravity500m(i-1))
1281        geopotential(nlev) = geopotential(nlev) + delAlt * aveGravity
1282      enddo
1283
1284      ! Add the contribution from top of the last 500m-layer to the altitude  
1285      delAlt = altitude(nlev) - alt500m(nlev500m)
1286      aveGravity = 0.5D0 * (phf_gravityalt(sLat, altitude(nlev)) + gravity500m(nlev500m))
1287      geopotential(nlev) = geopotential(nlev) + delAlt * aveGravity 
1288      gravityM1 = phf_gravityalt(sLat, altitude(nlev))
1289
1290      deallocate(alt500m)
1291      deallocate(gravity500m)
1292
1293    else
1294      ! At surface, use local gravity to get height
1295      gravity = phf_gravityalt(sLat,altitude(nlev))
1296      geopotential(nlev) = altitude(nlev) * gravity
1297      gravityM1 = gravity
1298
1299    endif
1300
1301    ! At upper-levels, integrate on model levels to get height 
1302    do ilev = nlev-1, 1, -1
1303      gravity = phf_gravityalt(sLat,altitude(ilev))
1304      aveGravity = 0.5D0 * (gravity + gravityM1)
1305      delAlt = altitude(ilev) - altitude(ilev+1)
1306      geopotential(ilev) = geopotential(ilev+1) + delAlt * aveGravity
1307      gravityM1 = gravity
1308    enddo
1309
1310    if ( present(printHeight) ) then
1311      if ( printHeight ) then
1312        write(*,*) 'phf_height2geopotential, Z_T:'
1313        write(*,*) geopotential(:)
1314
1315        printHeight = .false.
1316      endif
1317    endif
1318
1319  end subroutine phf_height2geopotential
