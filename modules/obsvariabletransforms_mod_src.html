  66  subroutine ovt_initStructure
  67    !
  68    !:Purpose: To set the transforms handled by this module
  69    ! 
  70    implicit none
  71
  72    ! Locals:
  73    integer :: transformIndex
  74
  75    ! Upper and surface winds
  76    transformIndex = 1
  77    transform(transformIndex)%name = 'windSpeedDirectionToUV'
  78    transform(transformIndex)%nBufrCode = 2
  79    allocate(transform(transformIndex)%sourceBufrCode          (transform(transformIndex)%nBufrCode))
  80    allocate(transform(transformIndex)%sourceBufrCodeExtra     (transform(transformIndex)%nBufrCode))
  81    allocate(transform(transformIndex)%destinationBufrCode     (transform(transformIndex)%nBufrCode))
  82    allocate(transform(transformIndex)%destinationBufrCodeExtra(transform(transformIndex)%nBufrCode))
  83    transform(transformIndex)%sourceBufrCode          (:) = (/bufr_nedd, bufr_neds/) ! direction
  84    transform(transformIndex)%sourceBufrCodeExtra     (:) = (/bufr_neff, bufr_nefs/) ! speed
  85    transform(transformIndex)%destinationBufrCode     (:) = (/bufr_neuu, bufr_neus/) ! u-wind
  86    transform(transformIndex)%destinationBufrCodeExtra(:) = (/bufr_nevv, bufr_nevs/) ! v-wind
  87    transform(transformIndex)%wind = .true.
  88
  89    ! log of visibility
  90    transformIndex = 2
  91    transform(transformIndex)%name = 'visToLogVis'
  92    transform(transformIndex)%nBufrCode = 1
  93    allocate(transform(transformIndex)%sourceBufrCode     (transform(transformIndex)%nBufrCode))
  94    allocate(transform(transformIndex)%destinationBufrCode(transform(transformIndex)%nBufrCode))
  95    transform(transformIndex)%sourceBufrCode     (:) = (/bufr_vis   /) ! visibility
  96    transform(transformIndex)%destinationBufrCode(:) = (/bufr_logVis/) ! log(visibility)
  97    transform(transformIndex)%wind = .false.
  98
  99    ! log of precipitation rate
 100    transformIndex = 3
 101    transform(transformIndex)%name = 'precipToLogPrecip'
 102    transform(transformIndex)%nBufrCode = 1
 103    allocate(transform(transformIndex)%sourceBufrCode     (transform(transformIndex)%nBufrCode))
 104    allocate(transform(transformIndex)%destinationBufrCode(transform(transformIndex)%nBufrCode))
 105    transform(transformIndex)%sourceBufrCode     (:) = (/bufr_radarPrecip   /) ! precipitation
 106    transform(transformIndex)%destinationBufrCode(:) = (/bufr_logRadarPrecip/) ! log(precipitation)
 107    transform(transformIndex)%wind = .false.
 108
 109    ! Skipped variables
 110    nSkippedBufrCodes   = 2
 111    skippedBufrCodes(1) = bufr_neff  ! because we still want to be able ...
 112    skippedBufrCodes(2) = bufr_nefs  ! ... to assimilate wind speed obs alone
 113
 114  end subroutine ovt_initStructure

 116  !--------------------------------------------------------------------------
 117  ! ovt_setup
 118  !--------------------------------------------------------------------------
 119  subroutine ovt_setup(bufrCodeRead)
 120    !
 121    !:Purpose: To determine which transform must be actived
 122    !
 123    implicit none
 124
 125    ! Argument:
 126    integer, intent(in) :: bufrCodeRead(:)          ! The list of bufr code read
 127
 128    ! Locals:
 129    integer, allocatable :: bufrCodeAssimilated(:)  ! The list of bufr code assimilated
 130
 131    integer :: nBufrCodeRead, nBufrCodeAssimilated
 132    integer :: readBufrCodeIndex, transformIndex, assimBufrCodeIndex
 133
 134    logical :: variableTransformNeeded, foundTransformation
 135    logical, save :: firstTime = .true.
 136
 137    if (firstTime) then
 138      call ovt_initStructure
 139      firstTime = .false.
 140    end if
 141
 142    nBufrCodeAssimilated = filt_nBufrCodeAssimilated()
 143    nBufrCodeRead = size(bufrCodeRead)
 144    variableTransformNeeded = .false.
 145
 146    allocate(bufrCodeAssimilated(nBufrCodeAssimilated))
 147    call filt_getBufrCodeAssimilated(bufrCodeAssimilated)
 148
 149    do readBufrCodeIndex = 1, nBufrCodeRead
 150    
 151      ! Check if a transform is neeeded
 152      if (filt_bufrCodeAssimilated(bufrCodeRead(readBufrCodeIndex)) .or. &
 153          bufrCodeRead(readBufrCodeIndex) == bufr_neff              .or. &
 154          bufrCodeRead(readBufrCodeIndex) == bufr_nefs ) then
 155        cycle ! No transformation needed. Move on.
 156              ! Note that this is where we decide that wind speed will be ignored, 
 157              ! because we do all the appropriate wind manipulations when we encounter direction.
 158              ! This strategy allow to assimilate lonely speed observations, like SAR winds.
 159      end if
 160
 161      ! Find and activate the appropriate transform
 162      foundTransformation = .false.
 163
 164      transformLoop : do transformIndex = 1, nTransformSupported
 165        assimBufrCodeLoop : do assimBufrCodeIndex = 1, nBufrCodeAssimilated
 166          if (any(transform(transformIndex)%sourceBufrCode(:) == bufrCodeRead(readBufrCodeIndex))        .and. &
 167              any(transform(transformIndex)%destinationBufrCode(:) == bufrCodeAssimilated(assimBufrCodeIndex)) )  then
 168            if (.not. transform(transformIndex)%active) then
 169              write(*,*) 'ovt_setup: transform activated : ', trim(transform(transformIndex)%name)
 170              transform(transformIndex)%active = .true.
 171            end if
 172            foundTransformation = .true.
 173            exit transformLoop
 174          end if
 175        end do assimBufrCodeLoop
 176      end do transformLoop
 177
 178      if (.not. foundTransformation) then
 179        if ( .not. any( skippedBufrCodes(:) == bufrCodeRead(readBufrCodeIndex) ) ) then
 180          write(*,*)
 181          write(*,*) 'ovt_setup: !WARNING! No transform found for the read bufr code = ', bufrCodeRead(readBufrCodeIndex)
 182          write(*,*) '           We are assuming that this observation is read but not assimilated.'
 183          write(*,*) '           Please consider removing this bufr code from the read observation list'
 184          write(*,*) '           unless the bufr element is read for another purpose.'
 185          nSkippedBufrCodes = nSkippedBufrCodes + 1
 186          skippedBufrCodes(nSkippedBufrCodes) = bufrCodeRead(readBufrCodeIndex)
 187        end if
 188      end if
 189
 190    end do
 191
 192    initialized = .true.
 193
 194    deallocate(bufrCodeAssimilated)
 195
 196  end subroutine ovt_setup

 201  function ovt_bufrCodeSkipped(sourceBufrCode) result(skip)
 202    !
 203    !:Purpose: To NEVER activate a variable transform for this bufr code,
 204    !           even when this bufr_code is read but not found in the assimilated list.
 205    !           So far, this function is only used to "skipped" wind speed reports
 206    !           because we do all the appropriate wind manipulations when we encounter direction.
 207    !
 208    implicit none
 209
 210    ! Arguments:
 211    integer, intent(in) :: sourceBufrCode ! The input bufr code
 212    logical :: skip                       ! The decision
 213    
 214    ! Locals:
 215    integer :: bufrCodeIndex
 216
 217    if (.not. initialized) then
 218      call utl_abort(' ovt_bufrCodeSkipped: this module has not been setup')
 219    end if
 220
 221    skip = .false.
 222    do bufrCodeIndex = 1, nSkippedBufrCodes
 223      if (skippedBufrCodes(bufrCodeIndex) == sourceBufrCode) then
 224        skip = .true.
 225        exit
 226      end if
 227    end do
 228
 229  end function ovt_bufrCodeSkipped

 234  function ovt_getDestinationBufrCode(sourceBufrCode,extra_opt) result(destinationBufrCode)
 235    !
 236    !:Purpose: To get the bufr code of the transformed/destination variable based on the 
 237    !           bufr code of the source variable
 238    !
 239    implicit none
 240
 241    ! Arguments:
 242    integer, intent(in) :: sourceBufrCode    ! The input source bufr code
 243    integer :: destinationBufrCode           ! The returned destination/transform bufr code
 244    logical, optional :: extra_opt           ! Should we look in the "extra" bufr code list or not
 245    
 246    ! Locals:
 247    logical :: extra
 248    integer :: transformIndex, bufrCodeIndex
 249
 250    if (.not. initialized) then
 251      call utl_abort(' ovt_getDestinationBufrCode: this module has not been setup')
 252    end if
 253
 254    if (present(extra_opt)) then
 255      extra = extra_opt
 256    else
 257      extra = .false. ! default
 258    end if
 259
 260    destinationBufrCode = -1
 261
 262    transformLoop : do transformIndex = 1, nTransformSupported
 263      bufrCodeLoop : do bufrCodeIndex = 1, transform(transformIndex)%nBufrCode
 264        if (transform(transformIndex)%sourceBufrCode(bufrCodeIndex) == sourceBufrCode) then
 265          if (extra) then
 266            destinationBufrCode = transform(transformIndex)%destinationBufrCodeExtra(bufrCodeIndex)
 267          else
 268            destinationBufrCode = transform(transformIndex)%destinationBufrCode(bufrCodeIndex)
 269          end if
 270          exit transformLoop
 271        end if
 272      end do bufrCodeLoop
 273    end do transformLoop
 274
 275    if (destinationBufrCode == -1) then
 276      write(*,*)
 277      write(*,*) 'ovt_getDestinationBufrCode: source bufrCode = ', sourceBufrCode
 278      call utl_abort('ovt_getDestinationBufrCode: found no associated bufrCode for the above source variable bufr code')
 279    end if
 280
 281  end function ovt_getDestinationBufrCode

 370    !
 371    !:Purpose: To determine if a bufr code is a transfomed variabled
 372    !

 404  subroutine ovt_transformObsValues(obsSpaceData, headerIndexStart, headerIndexEnd)
 405    !
 406    !:Purpose: To perform observation variable transforms
 407    !
 408    implicit none
 409
 410    ! Arguments:
 411    type (struct_obs), intent(inout) :: obsSpaceData      ! The observation database
 412    integer          , intent(in)    :: headerIndexStart  ! The initial header index to analyse
 413    integer          , intent(in)    :: headerIndexEnd    ! The final header index to analyse
 414
 415    ! Locals:
 416    integer :: transformIndex
 417
 418    if (obs_numHeader(obsSpaceData) == 0) return
 419
 420    if (.not. initialized) then
 421      call utl_abort('ovt_transformObsValues: this module has not been setup')
 422    end if
 423
 424    do transformIndex = 1, nTransformSupported
 425
 426      if (transform(transformIndex)%active) then
 427        select case(trim(transform(transformIndex)%name))
 428        case ('windSpeedDirectionToUV')
 429          call ovt_windSpeedDirectionToUV(obsSpaceData, headerIndexStart, headerIndexEnd)
 430        case ('visToLogVis')
 431          call ovt_visToLogVis           (obsSpaceData, headerIndexStart, headerIndexEnd)
 432        case ('precipToLogPrecip')
 433          call ovt_precipToLogPrecip     (obsSpaceData, headerIndexStart, headerIndexEnd)
 434        case default
 435          call utl_abort('ovt_transformObsValues: Unsupported function ' // trim(transform(transformIndex)%name))
 436        end select
 437      end if
 438
 439    end do
 440
 441  end subroutine ovt_transformObsValues

 484  !--------------------------------------------------------------------------
 485  ! ovt_windSpeedDirectionToUV
 486  !--------------------------------------------------------------------------
 487  subroutine ovt_windSpeedDirectionToUV(obsSpaceData, headerIndexStart, headerIndexEnd)
 488    !
 489    !:Purpose: To transform wind observation in terms of speed and direction to u and v
 490    !
 491    implicit none
 492
 493    ! Arguments:
 494    type (struct_obs), intent(inout) :: obsSpaceData      ! The observation database
 495    integer          , intent(in)    :: headerIndexStart  ! The initial header index to analyse
 496    integer          , intent(in)    :: headerIndexEnd    ! The final header index to analyse
 497
 498    ! Locals:
 499    integer        :: bufrCode, bufrCode2, bufrCode3
 500    integer        :: headerIndex, bodyIndex, bodyIndexStart, bodyIndexEnd, bodyIndex2, bufrCodeAssociated
 501    integer        :: directionFlag, speedFlag, combinedFlag, uWindFlag, vWindFlag
 502    integer        :: speedBufrCode, uWindBufrCode, vWindBufrCode, uWindbodyIndex, vWindBodyIndex
 503
 504    logical        :: direction_missing, speed_missing
 505    logical        :: uWind_present, vWind_present
 506
 507    real(pre_obsReal) :: uWind, vWind, direction, speed
 508    real(pre_obsReal) :: level_direction, level, level2, level3
 509
 510    speedFlag = 0
 511
 512    ! Loop through headers
 513    header: do headerIndex = headerIndexStart, headerIndexEnd
 514      
 515      bodyIndexStart = obs_headElem_i(obsSpaceData, OBS_RLN, headerIndex )
 516      bodyIndexEnd   = obs_headElem_i(obsSpaceData, OBS_NLV, headerIndex ) + bodyIndexStart - 1
 517      
 518      ! Find the wind direction report
 519      body: do bodyIndex = bodyIndexStart, bodyIndexEnd 
 520
 521        direction = obs_missingValue_R
 522        speed     = obs_missingValue_R
 523        bufrCode  = obs_bodyElem_i(obsSpaceData, OBS_VNM, bodyIndex)
 524        direction_missing = .true.
 525      
 526        if ( bufrCode /= bufr_nedd .and. bufrCode /= bufr_neds ) cycle body
 527
 528        if( bufrCode == bufr_neds) then
 529          ! Surface obs
 530          speedBufrCode = bufr_nefs
 531          uWindBufrCode = bufr_neus
 532          vWindBufrCode = bufr_nevs
 533        else
 534          ! Upper air obs
 535          speedBufrCode = bufr_neff
 536          uWindBufrCode = bufr_neuu
 537          vWindBufrCode = bufr_nevv
 538        end if
 539      
 540        direction       = obs_bodyElem_r(obsSpaceData, OBS_VAR, bodyIndex )
 541        directionFlag   = obs_bodyElem_i(obsSpaceData, OBS_FLG, bodyIndex )
 542        level_direction = obs_bodyElem_r(obsSpaceData, OBS_PPP, bodyIndex )
 543        uWind_present = .false.
 544        vWind_present = .false.
 545
 546        ! Check if u and v are present in obsSpaceData
 547        do bodyIndex2 = bodyIndexStart, bodyIndexEnd
 548
 549          level3 = obs_bodyElem_r(obsSpaceData, OBS_PPP, bodyIndex2 )
 550
 551          if (level3 == level_direction) then
 552            bufrCode3 = obs_bodyElem_i(obsSpaceData, OBS_VNM, bodyIndex2 )
 553            
 554            if ( bufrCode3 == uWindBufrCode ) then
 555              uWind_present = .true.
 556              uWindbodyIndex = bodyIndex2
 557            else if ( bufrCode3 == vWindBufrCode) then
 558              vWind_present = .true.
 559              vWindBodyIndex = bodyIndex2
 560            end if
 561
 562          end if
 563
 564        end do
 565
 566        if (.not. uWind_present .or. .not. vWind_present) then
 567          call utl_abort('ovt_windSpeedDirectionToUV: uWind and/or vWind bodyIndex not found!')
 568        end if
 569
 570        ! Find the speed report and compute uWind and vWind
 571        calcuv: do bodyIndex2 = bodyIndexStart, bodyIndexEnd
 572          speed_missing     = .true.
 573          direction_missing = .true.
 574          level = obs_bodyElem_r(obsSpaceData, OBS_PPP, bodyIndex2 )
 575
 576          if ( level /= level_direction) cycle calcuv
 577
 578          bufrCode2 = obs_bodyElem_i(obsSpaceData, OBS_VNM, bodyIndex2 )
 579
 580          if ( bufrCode2 == speedBufrCode ) then
 581
 582            speed     = obs_bodyElem_r(obsSpaceData, OBS_VAR, bodyIndex2 )
 583            speedFlag = obs_bodyElem_i(obsSpaceData, OBS_FLG, bodyIndex2 )
 584
 585            if ( direction == 0.d0 .and. speed > 0. .or. direction > 360. .or. direction < 0. ) then
 586              direction_missing = .true.
 587              speed_missing     = .true.
 588            else if ( direction == obs_missingValue_R .or. speed == obs_missingValue_R ) then
 589              direction_missing = .true.
 590              speed_missing     = .true.
 591            else
 592              direction_missing = .false.
 593              speed_missing     = .false.
 594            end if
 595
 596            if ( direction_missing .or. speed_missing ) then
 597
 598              call obs_bodySet_i(obsSpaceData, OBS_VNM, uWindbodyIndex, -1 )
 599              call obs_bodySet_i(obsSpaceData, OBS_VNM, vWindBodyIndex, -1 )
 600
 601            else
 602
 603              if (speed == 0.d0) direction = 0.d0
 604              direction = direction + 180.
 605              if ( direction > 360.) direction = direction - 360.
 606              direction = direction * mpc_radians_per_degree_r8
 607            
 608              ! (speed,direction) -> (u,v)
 609              uWind = speed * sin(direction)
 610              vWind = speed * cos(direction)
 611          
 612              combinedFlag = ior(directionFlag, speedFlag )
 613
 614              call obs_bodySet_r(obsSpaceData, OBS_VAR, uWindbodyIndex, uWind )
 615              call obs_bodySet_i(obsSpaceData, OBS_FLG, uWindbodyIndex, combinedFlag )
 616              call obs_bodySet_r(obsSpaceData, OBS_VAR, vWindBodyIndex, vWind )
 617              call obs_bodySet_i(obsSpaceData, OBS_FLG, vWindBodyIndex, combinedFlag )
 618

 696  !--------------------------------------------------------------------------
 697  subroutine ovt_UVtoWindSpeedDirection_residual(obsSpaceData, residualTypeID)
 698    !
 699    !:Purpose: To transform wind residuals in terms of u and v to speed and direction
 700    !
 701    implicit none
 702
 703    ! Arguments:
 704    type(struct_obs)    :: obsSpaceData    ! The observation database
 705    integer, intent(in) :: residualTypeID  ! The residual type ID (o-p or o-a)
 706
 707    ! Locals:
 708    integer :: uWindBufrCode, vWindBufrCode, speedBufrCode, directionBufrCode
 709    integer :: headerIndex, headerIndexStart, headerIndexEnd, windTypeIndex, bodyIndex, bodyIndex2
 710
 711    real(pre_obsReal) :: uWindLevel, speed, direction, uWind, vWind
 712
 713    windType: do windTypeIndex = 1, 2
 714      if (windTypeIndex == 1) then
 715        uWindBufrCode = bufr_neuu
 716        vWindBufrCode = bufr_nevv
 717        directionBufrCode = bufr_nedd
 718        speedBufrCode = bufr_neff
 719      else
 720        uWindBufrCode = bufr_neus
 721        vWindBufrCode = bufr_nevs
 722        directionBufrCode = bufr_neds
 723        speedBufrCode = bufr_nefs
 724      end if
 725
 726      ! Process all data within the domain of the model
 727      body: do bodyIndex = 1, obs_numBody(obsSpaceData)
 728
 729        if ( obs_bodyElem_i(obsSpaceData, OBS_ASS, bodyIndex) == obs_assimilated .and. &
 730             obs_bodyElem_i(obsSpaceData, OBS_VNM, bodyIndex) == uWindBufrCode ) then
 731
 732          headerIndex      = obs_bodyElem_i(obsSpaceData, OBS_HIND, bodyIndex  )
 733          headerIndexStart = obs_headElem_i(obsSpaceData, OBS_RLN , headerIndex )
 734          headerIndexEnd   = obs_headElem_i(obsSpaceData, OBS_NLV , headerIndex ) + headerIndexStart - 1
 735          uWindLevel            = obs_bodyElem_r(obsSpaceData, OBS_PPP , bodyIndex )
 736          uWind = - obs_bodyElem_r(obsSpaceData, residualTypeID, bodyIndex ) + obs_bodyElem_r(obsSpaceData, OBS_VAR, bodyIndex )
 737         
 738           body2: do bodyIndex2 = headerIndexStart, headerIndexEnd
 739     
 740             if  ( obs_bodyElem_i(obsSpaceData, OBS_VNM, bodyIndex2) == vWindBufrCode .and. &
 741                   obs_bodyElem_r(obsSpaceData, OBS_PPP, bodyIndex2) == uWindLevel ) then
 742               
 743              vWind = -obs_bodyElem_r(obsSpaceData, residualTypeID, bodyIndex2 ) + obs_bodyElem_r(obsSpaceData, OBS_VAR, bodyIndex2 )
 744
 745              ! Calculate angle
 746              speed = sqrt((uWind**2)+(vWind**2))
 747
 748              if (speed == 0.) then
 749                direction = 0.0d0
 750              else
 751                direction = atan2(vWind,uWind)
 752                direction = (270.0d0 - direction  * MPC_DEGREES_PER_RADIAN_R8 )
 753                ! Change to meteorological definition of wind direction.
 754                if (direction > 360.0d0 ) direction = direction - 360.0d0
 755                if (direction <= 0.0d0  ) direction = direction + 360.0d0
 756              end if
 757
 758            end if
 759
 760          end do body2
 761          
 762          ! insert resduals into obsSpaceData
 763          body2_2: do bodyIndex2 = headerIndexStart, headerIndexEnd
 764
 765            if  ( obs_bodyElem_i( obsSpaceData, OBS_VNM, bodyIndex2) == directionBufrCode .and. &
 766                  obs_bodyElem_r( obsSpaceData, OBS_PPP, bodyIndex2) == uWindLevel ) then
 767              call obs_bodySet_r( obsSpaceData, residualTypeID, bodyIndex2, &
 768                                  obs_bodyElem_r(obsSpaceData, OBS_VAR, bodyIndex2 ) - direction )
 769              
 770              if ( obs_bodyElem_r( obsSpaceData,residualTypeID,bodyIndex2) >  180.0d0)  &
 771                call obs_bodySet_r( obsSpaceData, residualTypeID, bodyIndex2, &
 772                                    obs_bodyElem_r(obsSpaceData, residualTypeID, bodyIndex2 ) - real(360.0d0,pre_obsReal) )
 773              if ( obs_bodyElem_r(obsSpaceData,residualTypeID,bodyIndex2) <= -180.0d0)  &
 774                call obs_bodySet_r( obsSpaceData, residualTypeID, bodyIndex2, &
 775                                    obs_bodyElem_r(obsSpaceData, residualTypeID, bodyIndex2 ) + real(360.0d0,pre_obsReal) )
 776              call obs_bodySet_r( obsSpaceData, residualTypeID, bodyIndex2, &
 777                                  - real(1.0d0,pre_obsReal) * obs_bodyElem_r(obsSpaceData,residualTypeID,bodyIndex2 ) )
 778              call obs_bodySet_r( obsSpaceData, OBS_OER, bodyIndex2, real(1.0d0,pre_obsReal) )
 779              call obs_bodySet_i( obsSpaceData, OBS_ASS, bodyIndex2, obs_assimilated )
 780              call obs_bodySet_i( obsSpaceData, OBS_FLG, bodyIndex2, 0 )
 781            end if
 782            if  ( obs_bodyElem_i( obsSpaceData, OBS_VNM, bodyIndex2 ) == speedBufrCode .and. &
 783                  obs_bodyElem_r( obsSpaceData, OBS_PPP, bodyIndex2 ) == uWindLevel ) then
 784              call obs_bodySet_r( obsSpaceData, residualTypeID,  bodyIndex2, &
 785                                  obs_bodyElem_r(obsSpaceData, OBS_VAR, bodyIndex2 ) - speed )
 786              call obs_bodySet_r( obsSpaceData, OBS_OER, bodyIndex2, real(1.0d0,pre_obsReal) )
 787              call obs_bodySet_i( obsSpaceData, OBS_ASS, bodyIndex2, obs_assimilated)
 788              call obs_bodySet_i( obsSpaceData, OBS_FLG, bodyIndex2, 0)
 789            end if
 790
 791          end do body2_2
 792
 793        end if
 794
 795      end do body
 796
 797    end do windType
 798
 799  end subroutine ovt_UVtoWindSpeedDirection_residual
