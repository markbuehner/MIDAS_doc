 103  ! ose_computeStddev
 104  !--------------------------------------------------------------------------
 105  subroutine ose_computeStddev(columnTrlOnAnlIncLev,hco_anl_in,obsSpaceData)
 106    !
 107    !:Purpose: To set OmP-error std dev when possible. Otherwise 
 108    !          compute background-error stddev in observation space to 
 109    !          estimate OmP-error std dev.
 110    !
 111    implicit none
 112
 113    ! Arguments:
 114    type(struct_columnData) :: columnTrlOnAnlIncLev ! Columns of the background interpolated to analysis levels and to obs horizontal locations
 115    type(struct_hco), pointer :: hco_anl_in
 116    type(struct_obs) :: obsSpaceData         ! Observation-related data
 117    
 118    ! Locals:
 119    real(8) :: HBHT_static, HBHT_ensemble, HBHT_hybrid
 120
 121    logical :: staticHBHT = .false.
 122    logical :: staticOMPE = .false.
 123    logical :: staticHBHT_ch   = .false.
 124    logical :: staticOMPE_ch   = .false.
 125    logical :: ensemble = .false.
 126
 127    integer :: index_body
 128    integer :: fnom, fclos, ierr, nulnam
 129
 130    ! namelist variables
 131    character(len=12) :: hybrid_mode ! can be 'WEIGHTED_SUM' or 'MAX_VALUE'
 132    NAMELIST /NAMHBHT/hybrid_mode
 133
 134    ! set the module variable hco_anl
 135    hco_anl => hco_anl_in
 136
 137    ! return if there is no observation.
 138    if ( obs_numheader(obsSpaceData) == 0 ) return
 139
 140    !- 1.  Compute HBHT (sigma_b in observation space) or OmP error std dev
 141
 142    !- 1.1 Compute error std dev from static statistics
 143    !      OmP error std dev (OBS_OMPE) when possible and required. 
 144    !      Otherwise, compute HBHT
 145    !      obsSpaceData - INOUT ( OmP error std dev outputted in OBS_OMPE or/and
 146    !                            sqrt(diag(H*B*H^T)) with B_static_chm outputted in OBS_HPHT )
 147  
 148    call ose_setStaticErrorStddev( columnTrlOnAnlIncLev, obsSpaceData, staticHBHT, staticHBHT_ch, staticOMPE_ch )
 149
 150    !- 1.2 HBHT from the Bens
 151    !      obsSpaceData - INOUT (HBensHT std. dev. outputted in OBS_WORK)
 152    
 153    call ose_compute_hbht_ensemble( columnTrlOnAnlIncLev,      & 
 154                                    obsSpaceData, &
 155                                    ensemble )                               
 156    
 157    if ( .not. staticHBHT .and. .not. staticOMPE .and. .not. ensemble .and. &
 158         .not. staticHBHT_ch .and. .not. staticOMPE_ch ) &
 159         call utl_abort('ose_computeStddev: no OmP std dev or sqrt(HBHT) was initialized')
 160    
 161    !- 2. Select/Blend HBHT.
 162   
 163    if ( staticHBHT .and. .not. ensemble ) then
 164      ! Bnmc only
 165      write(*,*)
 166      write(*,*) 'ose_computeStddev: Using B_static ONLY'
 167      ! HBnmcHT std. dev. already in OBS_HPHT
 168    else if ( .not. staticHBHT .and. ensemble ) then
 169      write(*,*)
 170      write(*,*) 'ose_computeStddev: Using B_ensemble ONLY'
 171      ! Transfer HBensHT std. dev. values in OBS_WORK to OBS_HPHT
 172      do index_body = 1, obs_numBody(obsSpaceData)
 173        HBHT_ensemble = obs_bodyElem_r(obsSpaceData,OBS_WORK,index_body)
 174        call obs_bodySet_r(obsSpaceData,OBS_HPHT,index_body, HBHT_ensemble)
 175      end do
 176    else if ( staticHBHT .and. ensemble ) then
 177      ! Read Namelist first
 178      hybrid_mode = 'WEIGHTED_SUM' ! default value
 179      nulnam = 0
 180      ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
 181      read(nulnam,nml=namhbht,iostat=ierr)
 182      if ( ierr /= 0) call utl_abort('ose_computeStddev: Error reading namelist')
 183      if ( mmpi_myid == 0 ) write(*,nml=namhbht)
 184      ierr = fclos(nulnam)
 185
 186      write(*,*)
 187      write(*,*) 'ose_computeStddev: Using hybrid approach (blend of B_static and B_ensemble) in mode = ', trim(hybrid_mode)
 188
 189      do index_body = 1, obs_numBody(obsSpaceData)
 190        HBHT_static = obs_bodyElem_r(obsSpaceData,OBS_HPHT,index_body)
 191        HBHT_ensemble = obs_bodyElem_r(obsSpaceData,OBS_WORK,index_body)
 192        if ( HBHT_static <= 0.0d0 ) then
 193          call obs_bodySet_r(obsSpaceData,OBS_HPHT,index_body, HBHT_ensemble)
 194        else      
 195          select case ( trim(hybrid_mode) )
 196          case ('WEIGHTED_SUM')
 197            HBHT_hybrid = sqrt(HBHT_static**2 + HBHT_ensemble**2)
 198          case ('MAX_VALUE')
 199            HBHT_hybrid = max(HBHT_static,HBHT_ensemble)
 200          case default
 201            write(*,*)
 202            write(*,*) 'ose_computeStddev: Unknown hybrid_mode ', trim(hybrid_mode)
 203            call utl_abort('ose_compute_HBHT')
 204          end select
 205          call obs_bodySet_r(obsSpaceData,OBS_HPHT,index_body, HBHT_hybrid)
 206        end if
 207      end do
 208
 209    end if
 210
 211  end subroutine ose_computeStddev

 216  subroutine ose_setStaticErrorStddev( columnTrlOnAnlIncLev, obsSpaceData, statusHBHT, statusHBHT_ch, statusOMPE_ch )
 217    !
 218    !:Purpose: To assign or compute the OmP error standard deviations in
 219    !          observation space where requested. If not possible or available,
 220    !          compute background-error standard deviation.
 221    !
 222    !:Note: OmP error std dev assigment currently only available for the CH obs family. 
 223    ! 
 224    !-------------------------------------------------------------------------
 225    implicit none
 226  
 227    ! Arguments:
 228    type(struct_columnData) :: columnTrlOnAnlIncLev
 229    type(struct_obs)        :: obsSpaceData  ! observation-space data, output saved in OBS_HPHT column
 230    logical, intent(inout)  :: statusHBHT, statusHBHT_ch, statusOMPE_ch
 231    
 232    ! Locals:
 233    integer :: famIndex
 234    character(len=4), allocatable :: availableOMPE(:)
 235    
 236    allocate(availableOMPE(ofl_numFamily)) 
 237    availableOMPE(:) = '    '
 238           
 239    ! Assignment of OBS_OMPE in obsSpaceData according to obs family where possible and required
 240    
 241    do famIndex=1,ofl_numFamily  
 242      if ( .not. obs_famExist(obsSpaceData,ofl_familyList(famIndex),localMPI_opt=.true.) ) cycle
 243      
 244      if ( ofl_familyList(famIndex) == 'CH' ) then
 245        write(*,*)
 246        write(*,*) 'ose_setStaticErrorStddev: Setting of OmP error std dev begins for family ',ofl_familyList(famIndex)
 247        availableOMPE(famIndex) = ose_setOmPstddevCH(obsSpaceData)
 248      else
 249        ! Not available 
 250        availableOMPE(famIndex) = 'None'
 251      end if
 252      if ( trim(availableOMPE(famIndex)) == 'Some' .or. &
 253           trim(availableOMPE(famIndex))  == 'All' ) statusOMPE_ch = .true.
 254
 255      if ( trim(availableOMPE(famIndex)) == '' ) then
 256        write(*,*) 'ose_setStaticErrorStddev: No ',ofl_familyList(famIndex),' obs. Setting of error std dev not required.'
 257      else if ( trim(availableOMPE(famIndex)) == 'None' ) then
 258        if ( ofl_familyList(famIndex) == 'CH' ) then
 259          write(*,*) 'ose_setStaticErrorStddev: Setting of ',ofl_familyList(famIndex),' OmP error std dev to be estimated via HBHT calc for all obs'
 260        end if        
 261      else if ( trim(availableOMPE(famIndex)) == 'Some' ) then
 262        write(*,*) 'ose_setStaticErrorStddev: Setting of ',ofl_familyList(famIndex),' OmP error std dev partially completed (some HBHT calc needed to complete)'       
 263      else
 264        write(*,*) 'ose_setStaticErrorStddev: Setting of ',ofl_familyList(famIndex),' OmP error std dev completed (no HBHT calc needed)'
 265      end if
 266      
 267    end do
 268    
 269    ! Computation of background-error standard deviation for obs families or groups of families when required
 270
 271    ! HBHT from the Bnmc matrix for weather 
 272    ! obsSpaceData - INOUT (HBnmcHT std. dev. output in OBS_HPHT)
 273     
 274    if ( any(ofl_familyList /= 'CH' .and. ofl_familyList /= 'TO' .and. &
 275       ( availableOMPE == 'Some' .or. availableOMPE == 'None' ) ) ) &
 276       call ose_compute_hbht_static( columnTrlOnAnlIncLev, obsSpaceData, statusHBHT )
 277
 278    ! HBHT from the B matrix for constituents
 279    
 280    if ( any(ofl_familyList == 'CH' .and. ( availableOMPE == 'Some' .or. availableOMPE == 'None' ) ) ) &
 281       call ose_compute_hbht_static_chem( columnTrlOnAnlIncLev, obsSpaceData, statusHBHT_ch )
 282    
 283    deallocate(availableOMPE)
 284    
 285  end subroutine ose_setStaticErrorStddev

 290  subroutine ose_compute_hbht_static(columnTrlOnAnlIncLev,lobsSpaceData,active)
 291    !
 292    !:Purpose: To compute background-error stddev in observation space using
 293    !          fixed statistics specific in stats file.
 294    !
 295    implicit none
 296
 297      ! Arguments:
 298      type(struct_obs)        :: lobsSpaceData
 299      type(struct_columnData) :: columnTrlOnAnlIncLev
 300      logical                 :: active
 301      
 302      ! Locals:
 303      type(struct_vco), pointer        :: vco_anl
 304      type(struct_columnData) :: column
 305      type(struct_gsv)        :: statevector
 306
 307      INTEGER JLAT, JLON, JLEV, JOBS
 308
 309      CHARACTER*12 CLETIKET
 310      CHARACTER*2 CLTYPVAR
 311      CHARACTER*1 CLGRTYP
 312      CHARACTER*4 CLNOMVAR
 313
 314      INTEGER IULSSF,IDATEO
 315      INTEGER FSTPRM,FNOM,FSTOUV
 316      INTEGER IKEY,IERR,IDATE
 317
 318      REAL*8, allocatable :: ZBUFFER(:,:)
 319      real*8, pointer     :: height_column(:), tt_column(:), field_ptr(:,:,:)
 320
 321      INTEGER INI,INJ,INK, INPAS, INBITS, IDATYP, IDEET
 322      INTEGER IP1,IP2,IP3,IG1,IG2,IG3,IG4,ISWA,ILENGTH,IDLTF
 323      INTEGER IUBC,IEXTR1,IEXTR2,IEXTR3
 324
 325      integer :: nLev_M,nLev_T,shift_level,Vcode_anl
 326
 327      integer :: cvdim
 328
 329      real(8), allocatable  :: scaleFactor(:)
 330
 331      !- Get the appropriate Vertical Coordinate
 332      vco_anl => col_getVco(columnTrlOnAnlIncLev)
 333
 334      ! Note : Here we can use the global B_hi even if we are in LAM mode since, 
 335      !        in BackgroundCheck mode, the only purpose of bhi_setup is to read 
 336      !        the scaleFactor and to check the consistency between the vco from 
 337      !        analysisgrid and cov file
 338      call bhi_setup( hco_anl,vco_anl,           &  ! IN
 339                      cvdim,                     &  ! OUT
 340                      mode_opt='BackgroundCheck' )  ! IN
 341
 342      if ( cvdim > 0 ) then
 343         write(*,*)
 344         write(*,*) 'Computing HBHT from Bnmc - Start'
 345         active = .true.
 346      else
 347         if ( mmpi_myid == 0 ) write(*,*) 'ose_compute_hbht_static: option NOT ACTIVATED'
 348         active = .false.
 349         return
 350      end if
 351
 352      nlev_T = col_getNumLev(columnTrlOnAnlIncLev,'TH')
 353      nlev_M = col_getNumLev(columnTrlOnAnlIncLev,'MM')
 354
 355      allocate(scaleFactor(max(nLev_M,nLev_T)))
 356      call bhi_getScaleFactor(scaleFactor)
 357
 358      Vcode_anl = vco_anl%Vcode
 359      if ( Vcode_anl == 5001 ) then
 360        shift_level = 0
 361      else if ( Vcode_anl == 5002) then
 362        shift_level = 1
 363      else if ( Vcode_anl == 5005 ) then
 364        shift_level = 0
 365      else
 366        write(*,*) 'Vcode_anl = ',Vcode_anl
 367        call utl_abort('ose_compute_hbht_static: unknown vertical coordinate type!')
 368      end if
 369
 370      allocate(ZBUFFER(HCO_ANL%NI,HCO_ANL%NJ))
 371
 372      call gsv_allocate(statevector, 1, hco_anl, vco_anl, &
 373                        allocHeight_opt=.false., allocPressure_opt=.false.)
 374      call gsv_zero(statevector)
 375
 376      call col_setVco(column,col_getVco(columnTrlOnAnlIncLev))
 377      call col_allocate(column,col_getNumCol(columnTrlOnAnlIncLev))
 378
 379      ! Set the value of OBS_LYR required by setfge routines
 380
 381      call oop_vobslyrs(columnTrlOnAnlIncLev,lobsSpaceData,beSilent=.false.)
 382
 383      ! 1. Opening the statistics file
 384
 385      IULSSF=0
 386      IERR=FNOM(iulssf,'./bgcov','RND+OLD+R/O',0)
 387      IF ( IERR .EQ. 0 ) THEN
 388        write(*,*) 'IBGST - File : ./bgcov'
 389        write(*,*) ' opened as unit file ',iulssf
 390        ierr =  fstouv(iulssf,'RND+OLD')
 391      ELSE
 392        CALL utl_abort('HBHT_static:NO BACKGROUND STAT FILE!!')
 393      ENDIF
 394
 395      ! 2.1 Background error standard deviations
 396
 397      CLETIKET = 'BGCK_STDDEV'
 398      write(*,*) 'HBHT_static: CLETIKET = ',CLETIKET
 399      IDATE    = -1
 400      IP2      = -1
 401      IP3      = -1
 402      CLTYPVAR =' '
 403
 404      ! READ IN STANDARD DEVIATION FOR EACH OBSERVATION TYPE
 405
 406      clnomvar = 'UU'
 407      write(*,*) clnomvar 
 408      call gsv_getField(statevector,field_ptr,'UU')       
 409      do jlev = 1, nlev_M
 410         ip1 = vco_anl%ip1_M(jlev)
 411         ikey = utl_fstlir(zbuffer,iulssf,ini,inj,ink,idate,cletiket,ip1,ip2,ip3,cltypvar,clnomvar)
 412         ierr = fstprm(ikey,idateo,ideet,inpas   &
 413              ,ini,inj,ink, inbits, idatyp       &
 414              ,ip1,ip2,ip3,cltypvar,clnomvar,cletiket,clgrtyp   &
 415              ,ig1,ig2,ig3,ig4,iswa,ilength,idltf   &
 416              ,iubc,iextr1,iextr2,iextr3)
 417         if (ini /= hco_anl%ni .or. inj /= hco_anl%nj .or. ink /= 1) then
 418            write(*,*)
 419            write(*,*) 'HBHT_static: Invalid dimensions for...'
 420            write(*,*) 'nomvar         =', trim(clnomvar)
 421            write(*,*) 'etiket         =', trim(cletiket)
 422            write(*,*) 'Found ni,nj,nk =', ini, inj, ink
 423            write(*,*) 'Should be      =', hco_anl%ni, hco_anl%nj, nlev_M
 424            call utl_abort('ose_compute_hbht_static')
 425         end if
 426         do jlat = 1, hco_anl%nj
 427            do jlon = 1, hco_anl%ni
 428               field_ptr(jlon,jlat,jlev) = scaleFactor(jlev+shift_level)*zbuffer(jlon,jlat)*MPC_M_PER_S_PER_KNOT_R8
 429             end do
 430          end do
 431      end do
 432
 433      clnomvar = 'VV'
 434      write(*,*) clnomvar
 435      call gsv_getField(statevector,field_ptr,'VV')
 436      do jlev = 1, nlev_M
 437         ip1 = vco_anl%ip1_M(jlev)
 438         ikey = utl_fstlir(zbuffer,iulssf,ini,inj,ink,idate,cletiket,ip1,ip2,ip3,cltypvar,clnomvar)
 439         if (ini /= hco_anl%ni .or. inj /= hco_anl%nj .or. ink /= 1) then
 440            write(*,*)
 441            write(*,*) 'HBHT_static: Invalid dimensions for...'
 442            write(*,*) 'nomvar         =', trim(clnomvar)
 443            write(*,*) 'etiket         =', trim(cletiket)
 444            write(*,*) 'Found ni,nj,nk =', ini, inj, ink
 445            write(*,*) 'Should be      =', hco_anl%ni, hco_anl%nj, nlev_M
 446            call utl_abort('ose_compute_hbht_static')
 447         end if
 448
 449         do jlat = 1, hco_anl%nj
 450            do jlon = 1, hco_anl%ni
 451               field_ptr(jlon,jlat,jlev) = scalefactor(jlev+shift_level)*zbuffer(jlon,jlat)*MPC_M_PER_S_PER_KNOT_R8
 452            end do
 453         end do
 454      end do
 455
 456      clnomvar = 'ES'
 457      write(*,*) clnomvar
 458      call gsv_getField(statevector,field_ptr,'HU')
 459      do jlev = 1, nlev_T
 460         ip1 = vco_anl%ip1_T(jlev)
 461         ikey = utl_fstlir(zbuffer,iulssf,ini,inj,ink,idate,cletiket,ip1,ip2,ip3,cltypvar,clnomvar)
 462         if (ini /= hco_anl%ni .or. inj /= hco_anl%nj .or. ink /= 1) then
 463            write(*,*)
 464            write(*,*) 'HBHT_static: Invalid dimensions for...'
 465            write(*,*) 'nomvar         =', trim(clnomvar)
 466            write(*,*) 'etiket         =', trim(cletiket)
 467            write(*,*) 'Found ni,nj,nk =', ini, inj, ink
 468            write(*,*) 'Should be      =', hco_anl%ni, hco_anl%nj, nlev_T
 469            call utl_abort('ose_compute_hbht_static')
 470         end if
 471         do jlat = 1, hco_anl%nj
 472            do jlon = 1, hco_anl%ni
 473               field_ptr(jlon,jlat,jlev) = scaleFactor(jlev)*zbuffer(jlon,jlat)
 474            end do
 475         end do
 476      end do
 477
 478      ! height is put into TT slot in gridStateVector
 479      clnomvar = 'GZ'
 480      write(*,*) clnomvar
 481      call gsv_getField(statevector,field_ptr,'TT')
 482      do jlev = 1, nlev_T
 483         ip1 = vco_anl%ip1_T(jlev)
 484         ikey = utl_fstlir(zbuffer,iulssf,ini,inj,ink,idate,cletiket,ip1,ip2,ip3,cltypvar,clnomvar)
 485         if (ini /= hco_anl%ni .or. inj /= hco_anl%nj .or. ink /= 1) then
 486            write(*,*)
 487            write(*,*) 'HBHT_static: Invalid dimensions for...'
 488            write(*,*) 'nomvar         =', trim(clnomvar)
 489            write(*,*) 'etiket         =', trim(cletiket)
 490            write(*,*) 'Found ni,nj,nk =', ini, inj, ink
 491            write(*,*) 'Should be      =', hco_anl%ni, hco_anl%nj, nlev_T
 492            call utl_abort('ose_compute_hbht_static')
 493         end if
 494         do jlat = 1, hco_anl%nj
 495            do jlon = 1, hco_anl%ni
 496               field_ptr(jlon,jlat,jlev) = scaleFactor(jlev)*zbuffer(jlon,jlat)*ec_rg*10.d0
 497            end do
 498         end do
 499      end do
 500
 501      call s2c_bgcheck_bilin(column,statevector,lobsSpaceData)
 502
 503      ! copy height data from TT to height slot in columnData
 504      do jobs= 1, col_getNumCol(column)
 505         height_column => col_getColumn(column,jobs,'Z_T')
 506         tt_column => col_getColumn(column,jobs,'TT')
 507         do jlev = 1,col_getNumLev(column,'TH')
 508            height_column(jlev)=tt_column(jlev)
 509         enddo
 510      enddo
 511
 512      ! SET THE FIRST-GUESS ERRORS FOR CONVENTIONAL DATA ON PRESSURE LEVELS
 513      ! --------------------------------------------------------------------
 514
 515      call setfgefam('AI',column,columnTrlOnAnlIncLev,lobsSpaceData)
 516      call setfgefam('SW',column,columnTrlOnAnlIncLev,lobsSpaceData)

 806  ! ose_compute_hbht_ensemble
 807  !--------------------------------------------------------------------------
 808  subroutine ose_compute_hbht_ensemble(columnTrlOnAnlIncLev,obsSpaceData,active)
 809    !
 810    !:Purpose: To compute background-error stddev in observation space using
 811    !          ensemble-based statistics.
 812    !
 813    implicit none
 814
 815    ! Arguments:
 816    type(struct_columnData) :: columnTrlOnAnlIncLev      ! Columns of the background interpolated to analysis levels and to obs horizontal locations
 817    type(struct_obs)        :: obsSpaceData ! Observation-related data
 818    logical                 :: active
 819
 820    ! Locals:
 821    type(struct_columnData) :: column
 822    type(struct_gsv)        :: statevector
 823
 824    type(struct_vco), pointer :: vco_anl
 825
 826    real(8), allocatable :: HBHT_ens(:)
 827
 828    integer :: memberIndex, index_body
 829    integer, allocatable :: cvdim(:)
 830
 831    !
 832    !- 1.  Initialization
 833    !
 834
 835    !- 1.1 Get vertical analysis grid attributes
 836    vco_anl => col_getVco(columnTrlOnAnlIncLev)
 837
 838    !- 1.2 Initialize/Read the flow-dependent ensemble perturbations
 839    call ben_Setup( hco_anl, hco_anl,    & ! IN
 840                    vco_anl,             & ! IN
 841                    cvdim,               & ! OUT
 842                    'BackgroundCheck' )    ! IN
 843
 844    if ( cvdim(1) > 0 ) then
 845      write(*,*)
 846      write(*,*) 'Computing HBHT from ensemble perturbations - START'
 847      active = .true.
 848    else
 849      if ( mmpi_myid == 0 ) write(*,*) 'ose_compute_hbht_ensemble: option NOT ACTIVATED'
 850      active = .false.
 851      return
 852    end if
 853
 854    !- 1.3 Create a gridstatevector to store the ensemble perturbations
 855    call gsv_allocate(statevector, tim_nstepobsinc, hco_anl, vco_anl, &
 856      mpi_local_opt=.true., allocHeight_opt=.false., allocPressure_opt=.false.)
 857
 858    !- 1.4 Create column vectors to store the ens perturbation interpolated to obs horizontal locations
 859    call col_setVco(column,vco_anl)
 860    call col_allocate(column,col_getNumCol(columnTrlOnAnlIncLev),mpiLocal_opt=.true.)
 861
 862    !- 1.5 Create a working a array to sum H ensPert HT
 863    allocate(HBHT_ens(obs_numBody(obsSpaceData)))
 864    HBHT_ens(:) = 0.d0
 865
 866    !- 1.6
 867    call oti_timeBinning(obsSpaceData,tim_nstepobsinc)
 868
 869    !
 870    !- 2.  Compute HBHT from the ensemble perturbations
 871    !
 872    do memberIndex = 1, ben_getnEns()
 873
 874      !- 2.1 Extract perturbations from the current memberIndex
 875      write(*,*)
 876      write(*,*) 'Reading ensemble perturbation from member = ', memberIndex
 877      call ben_getPerturbation( statevector,    &
 878                                memberIndex,    &
 879                                'ConstantValue' )
 880
 881      !- 2.2 Interpolation to the observation horizontal locations
 882      !       column - OUT (H_horiz EnsPert)
 883      call s2c_tl( statevector,           &
 884                   column,                &
 885                   columnTrlOnAnlIncLev, obsSpaceData )
 886                   
 887      !- 2.3 Interpolation to observation space
 888      !         obsSpaceData - OUT (Save as OBS_WORK: H_vert H_horiz EnsPert = H EnsPert)
 889      call oop_Htl( column, columnTrlOnAnlIncLev, &
 890                    obsSpaceData,    &
 891                    1 )
 892
 893      !- 2.4 alpha * HBH^T = sum(OBS_WORK^2)
 894      do index_body = 1, obs_numBody(obsSpaceData)
 895        HBHT_ens(index_body) = HBHT_ens(index_body) + &
 896                               (obs_bodyElem_r(obsSpaceData,OBS_WORK,index_body))**2
 897      end do
 898
 899    end do
 900
 901    !- 2.5 Insert the standard deviations in OBS_WORK
 902    do index_body = 1, obs_numBody(obsSpaceData)
 903      call obs_bodySet_r(obsSpaceData,OBS_WORK,index_body,sqrt(HBHT_ens(index_body)))
 904    end do
 905
 906    !
 907    !- 3.  Ending/Deallocation
 908    !
 909    deallocate(HBHT_ens)
 910    call col_deallocate(column)
 911    call gsv_deallocate(statevector)
 912
 913    write(*,*)
 914    write(*,*) 'Computing HBHT from ensemble perturbations - END'
 915
 916  end subroutine ose_compute_hbht_ensemble

 921  ! setfgefam
 922  !--------------------------------------------------------------------------
 923  subroutine setfgefam(cdfam,column,columnTrlOnAnlIncLev,lobsSpaceData)
 924    !
 925    !:Purpose: To interpolate vertically the contents of "column" to
 926    !          the pressure levels of the observations. Then to compute
 927    !          THE FIRST GUESS ERROR VARIANCES. A linear interpolation in ln(p)
 928    !          is performed.
 929    !
 930    implicit none
 931
 932    ! Arguments:
 933    character*2 cdfam
 934    type(struct_columnData) :: column
 935    type(struct_columnData) :: columnTrlOnAnlIncLev
 936    type(struct_obs) :: lobsSpaceData
 937
 938    ! Locals:
 939      INTEGER IPB,IPT
 940      INTEGER INDEX_HEADER,ITYP,IK
 941      INTEGER INDEX_BODY
 942      REAL*8 ZWB,ZWT, columnElem
 943      REAL*8 ZLEV,ZPB,ZPT
 944      character(len=4) :: varLevel
 945
 946      ! loop over all header indices of the CDFAM family
 947      call obs_set_current_header_list(lobsSpaceData,CDFAM)
 948      HEADER: do
 949         index_header = obs_getHeaderIndex(lobsSpaceData)
 950         if (index_header < 0) exit HEADER
 951
 952         ! loop over all body indices for this index_header
 953         call obs_set_current_body_list(lobsSpaceData, index_header)
 954         BODY: do 
 955            index_body = obs_getBodyIndex(lobsSpaceData)
 956            if (index_body < 0) exit BODY
 957            !
 958            !*    1. Computation of sigmap
 959            !     .  -----------------------------
 960            !
 961            IF ( obs_bodyElem_i(lobsSpaceData,OBS_ASS,index_body) == obs_assimilated .AND.   &
 962                 obs_bodyElem_i(lobsSpaceData,OBS_VCO,index_body) .EQ. 2      ) then
 963            IF  (obs_bodyElem_i(lobsSpaceData,OBS_XTR,index_body) .NE. 0) THEN
 964               ITYP = obs_bodyElem_i(lobsSpaceData,OBS_VNM,index_body)
 965               varLevel = vnl_varLevelFromVarnum(ityp)
 966               IK   = col_getNumLev(columnTrlOnAnlIncLev,varLevel)
 967               IPB  = IK + col_getOffsetFromVarno(columnTrlOnAnlIncLev,ityp)
 968               if(ITYP .ne. BUFR_NEGZ) then
 969                 call obs_bodySet_r(lobsSpaceData,OBS_HPHT,index_body,col_getElem(column,IPB,INDEX_HEADER))
 970               else
 971                 call obs_bodySet_r(lobsSpaceData,OBS_HPHT,index_body,col_getHeight(column,IK,INDEX_HEADER,'TH'))
 972               endif
 973            ELSE
 974               ITYP = obs_bodyElem_i(lobsSpaceData,OBS_VNM,index_body)
 975               varLevel = vnl_varLevelFromVarnum(ityp)
 976               ZLEV = obs_bodyElem_r(lobsSpaceData,OBS_PPP,index_body)
 977               IK   = obs_bodyElem_i(lobsSpaceData,OBS_LYR,index_body)
 978               IPT  = IK + col_getOffsetFromVarno(columnTrlOnAnlIncLev,ityp)
 979               IPB  = IPT+1
 980               ZPT  = col_getPressure(columnTrlOnAnlIncLev,IK,INDEX_HEADER,varLevel)
 981               ZPB  = col_getPressure(columnTrlOnAnlIncLev,IK+1,INDEX_HEADER,varLevel)
 982               ZWB  = LOG(ZLEV/ZPT)/LOG(ZPB/ZPT)
 983               ZWT  = 1.0D0 - ZWB
 984
 985               ! FIRST GUESS ERROR VARIANCE
 986
 987               if(ITYP .ne. BUFR_NEGZ) then
 988                 call obs_bodySet_r(lobsSpaceData,OBS_HPHT,index_body,   &
 989                      (ZWB*col_getElem(column,IPB,INDEX_HEADER) + ZWT*col_getElem(column,IPT,INDEX_HEADER)))
 990               else
 991                 call obs_bodySet_r(lobsSpaceData,OBS_HPHT,index_body,   &
 992                      (ZWB*col_getHeight(column,IK+1,INDEX_HEADER,'TH') + ZWT*col_getHeight(column,IK,INDEX_HEADER,'TH')))
 993               endif
 994               if(obs_bodyElem_r(lobsSpaceData,OBS_HPHT,index_body).le.0.d0) then
 995                 write(*,*) 'SETFGEFAM: CDFAM = ',CDFAM
 996                 write(*,*) 'SETFGEFAM: IPB,IPT,ZWB,ZWT,ITYP,ZLEV=',IPB,IPT,ZWB,ZWT,ITYP,ZLEV

1020  subroutine setfgefamz(cdfam, column, columnTrlOnAnlIncLev, obsSpaceData)
1021    !
1022    !:Purpose: To interpolate vertically the contents of "column" to the levels
1023    !          of the observations (in meters). Then to compute THE FIRST GUESS
1024    !          ERROR VARIANCES. A linear interpolation in z is performed.
1025    !
1026    implicit none
1027
1028    ! Arguments:
1029    character(len=2),        intent(in)    :: cdfam
1030    type(struct_columnData), intent(in)    :: column
1031    type(struct_columnData), intent(in)    :: columnTrlOnAnlIncLev
1032    type(struct_obs),        intent(inout) :: obsSpaceData
1033
1034    ! Locals:
1035    integer :: IPB, IPT
1036    integer :: headerIndex, ITYP, IK
1037    integer :: bodyIndex
1038    integer :: bodyIndexStart, bodyIndexEnd, bodyIndex2
1039    real*8  :: ZWB, ZWT, sigmap_uuT, sigmap_vvT , sigmap_uuB, sigmap_vvB 
1040    real*8  :: ZLEV, ZPB, ZPT
1041    real(8) :: azimuth, fge_uu, fge_vv, fge_fam
1042    character(len=4) :: varLevel
1043
1044    ! loop over all header indices of the CDFAM family
1045    call obs_set_current_header_list(obsSpaceData, CDFAM)
1046    HEADER: do
1047      headerIndex = obs_getHeaderIndex(obsSpaceData)
1048      if ( headerIndex < 0 ) exit HEADER
1049
1050      if  ( cdfam == 'RA' ) then
1051        ! Azimuth of the radar beam
1052         azimuth = obs_headElem_r(obsSpaceData, OBS_RZAM, headerIndex )
1053      end if
1054
1055      ! loop over all body indices for this headerIndex
1056      call obs_set_current_body_list(obsSpaceData, headerIndex)
1057      BODY: do 
1058        bodyIndex = obs_getBodyIndex(obsSpaceData)
1059        if ( bodyIndex < 0 ) exit BODY
1060
1061        !*    1. Computation of sigmap
1062        !     .  -----------------------------
1063        if ( obs_bodyElem_i(obsSpaceData, OBS_ASS, bodyIndex) == obs_assimilated .and. &
1064             obs_bodyElem_i(obsSpaceData, OBS_VCO, bodyIndex) == 1 )then
1065          ITYP = obs_bodyElem_i(obsSpaceData, OBS_VNM, bodyIndex)
1066          if ( ITYP == bufr_radarPrecip ) cycle BODY
1067          varLevel = vnl_varLevelFromVarnum(ityp)
1068
1069          ! Interpolate the background-covariance statistics
1070          if ( obs_bodyElem_i(obsSpaceData, OBS_XTR, bodyIndex) /= 0 ) then
1071            IK=col_getNumLev(columnTrlOnAnlIncLev, varLevel)-1
1072            IPT  = IK + col_getOffsetFromVarno(columnTrlOnAnlIncLev, ityp)
1073            IPB  = IPT +1
1074            fge_uu = col_getElem(column, IPB, headerIndex, 'UU')
1075            fge_vv = col_getElem(column, IPB, headerIndex, 'VV')
1076
1077          else
1078            ZLEV = obs_bodyElem_r(obsSpaceData, OBS_PPP, bodyIndex)
1079            IK   = obs_bodyElem_i(obsSpaceData, OBS_LYR, bodyIndex)
1080            IPT  = IK + col_getOffsetFromVarno(columnTrlOnAnlIncLev, ityp)
1081            IPB  = IPT+1
1082            ZPT  = col_getHeight(columnTrlOnAnlIncLev, IK, headerIndex, varLevel)
1083            ZPB  = col_getHeight(columnTrlOnAnlIncLev, IK+1, headerIndex, varLevel)
1084            ZWB  = (ZPT-ZLEV)/(ZPT-ZPB)
1085            ZWT  = 1.d0 - ZWB
1086            fge_uu = ZWB*col_getElem(column, IPB, headerIndex, 'UU') &
1087                   + ZWT*col_getElem(column, IPT, headerIndex, 'UU')
1088            fge_vv = ZWB*col_getElem(column, IPB, headerIndex, 'VV') &
1089                   + ZWT*col_getElem(column, IPT, headerIndex, 'VV')
1090          end if
1091
1092          ! First-Guess Error Variance
1093          if ( cdfam == 'AL' )then
1094            ! Scan body indices for the azimuth
1095            bodyIndexStart = obs_headElem_i(obsSpaceData, OBS_RLN, headerIndex)
1096            bodyIndexEnd   = obs_headElem_i(obsSpaceData, OBS_NLV, headerIndex) &
1097                                  + bodyIndexStart - 1
1098            BODY_SUPP: do bodyIndex2 = bodyIndexStart, bodyIndexEnd
1099              if(obs_bodyElem_i(obsSpaceData, OBS_VNM, bodyIndex2) == 5021)then
1100                azimuth=obs_bodyElem_r(obsSpaceData, OBS_VAR, bodyIndex2) * MPC_RADIANS_PER_DEGREE_R8
1101                    exit BODY_SUPP
1102              end if
1103            end do BODY_SUPP
1104
1105            fge_fam = sqrt((fge_vv*cos(azimuth))**2 + (fge_uu*sin(azimuth))**2)
1106
1107          else if( cdfam == 'PR' )then
1108            fge_fam = ZWB*col_getElem(column, IPB, headerIndex) &
1109                    + ZWT*col_getElem(column, IPT, headerIndex)
1110                  
1111          else if( cdfam == 'RA' .and. ITYP == bufr_radvel ) then    
1112
1113            ! Calculation of sigmap^2 = diag(H*P*H^t) to save sigmap in OBS_HPHT  
1114            ! 
1115            ! H includes vertical interpolation  
1116            !   and projection of U and V wind components along the direction of the beam
1117            !
1118            ! P forecast error covariance
1119            !  
1120            ! H = [ ZWT*sin(az) ZWT*cos(az) ZWB*sin(az) ZWB*cos(az) ] 
1121            ! diag(P) = [ sigmap_uuT^2  sigmap_vvT^2  sigmap_uuB^2  sigmap_vvB^2  ]
1122            sigmap_uuT = col_getElem(column, IPT, headerIndex, 'UU')
1123            sigmap_uuB = col_getElem(column, IPB, headerIndex, 'UU')
1124            sigmap_vvT = col_getElem(column, IPT, headerIndex, 'VV')
1125            sigmap_vvB = col_getElem(column, IPB, headerIndex, 'VV')
1126
1127            fge_fam = sqrt((sigmap_uuT*ZWT*sin(azimuth))**2 + (sigmap_vvT*ZWT*cos(azimuth))**2 + &
1128                           (sigmap_uuB*ZWB*sin(azimuth))**2 + (sigmap_vvB*ZWB*cos(azimuth))**2)
1129                 
1130          else
1131
1132            write(*,*)"ERROR:  The family", cdfam, " is not supported by setfgefamz"
1133            call utl_abort('setfgefamz')
1134
1135          end if
1136
1137          ! Store fge_fam in OBS_HPHT
1138          call obs_bodySet_r(obsSpaceData, OBS_HPHT, bodyIndex, fge_fam)
1139        
1140        end if
1141
1142      end do BODY
1143
1144    end do HEADER
1145
1146  end subroutine setfgefamz

2534

2536

2539           write(*,*) 'latbin levbin Npts      AvgOmP    OmPStddev'
2540        end if
2541        do levIndex=1,OmPstdCH%n_lvl(stnidIndex)
2542        do latIndex=1,OmPstdCH%n_lat(stnidIndex)

2544          if (nSeriest(levIndex,latIndex) > 5*minCount ) then
2545            varOmP =  sumSqrOmP2dt(levIndex,latIndex)
2546            if (varOmP  < minVal*minVal ) then
2547               availableOmP = .false.
2548            else     
2549              if (mmpi_myid == 0 ) write(*,110) latIndex,levIndex,nSeriest(levIndex,latIndex),sumOmP2dt(levIndex,latIndex)/nSeriest(levIndex,latIndex),sqrt(varOmP)
2550            end if
2551          else

2553          end if
2554  
2555          ! Store in structure
2556   
2557          ! Indentify position in structure
2558          if (OmPstdCH%n_lvl(stnidIndex) > 1) then
2559            posIndex=ibegin+(levIndex-1)*OmPstdCH%n_lat(stnidIndex)
2560          else
2561            posIndex=ibegin
2562          end if
2563

2574          end if
2575        
2576        end do
2577        end do
2578      end if  
2579110   format(I5,I7,I7,3x,G11.3,G11.3)
2580

2582      deallocate(nSeriest,sumOmP2dt,sumSqrOmP2dt)

2698  real(8) function ose_getOmPstddevCH(zlat,zlev,stnidIndex,latIndex,monthIndex) result(OmP_err_stddev) 
2699    ! 
2700    !:Purpose: To return the OmP error std dev for a CH family measurement
2701    !
2702    implicit none
2703
2704    ! Arguments:
2705    real(8), intent(in) :: zlat  ! latitude (radians)
2706    real(8), intent(in) :: zlev  ! vertical coordinate value
2707    integer, intent(in) :: stnidIndex ! station and obs type index
2708    integer, intent(in) :: latIndex   ! reference lat for interpolation
2709    integer, intent(in) :: monthIndex ! month index
2710
2711    ! Locals:
2712    real(8) :: levDiff
2713    integer :: ibegin,levIndex,loopIndex,posIndex
2714    real(8), parameter :: minDiff = 0.01*abs(MPC_missingValue_R8)
2715
2716    OmP_err_stddev = MPC_missingValue_R8
2717
2718    ! Get OmP error std. dev.
2719
2720    ibegin=OmPstdCH%ibegin(stnidIndex)-1
2721    
2722    if (OmPstdCH%n_lvl(stnidIndex) > 1) then
2723                 
2724      ! Find nearest vertical level (no vertical interpolation in this version)
2725                 
2726      levDiff=1.0d10
2727      do loopIndex=1,OmPstdCH%n_lvl(stnidIndex)
2728         if ( levDiff > abs(zlev-OmPstdCH%levels(ibegin+loopIndex)) ) THEN
2729            levIndex=loopIndex
2730            levDiff=abs(zlev-OmPstdCH%levels(ibegin+loopIndex))
2731         END IF
2732      END DO
2733      posIndex=ibegin+(levIndex-1)*OmPstdCH%n_lat(stnidIndex)
2734    else
2735      posIndex=ibegin
2736    end if
2737      
2738    if ( OmPstdCH%n_month(stnidIndex) == 1 ) then
2739       posIndex = posIndex + latIndex
2740    else
2741       posIndex = posIndex + (monthIndex-1)*OmPstdCH%n_lat(stnidIndex)*OmPstdCH%n_lvl(stnidIndex) + latIndex
2742    end if   
2743    
2744    if ( OmPstdCH%n_lat(stnidIndex) > 1 ) then
2745      ! Apply interpolation in latitude
2746  
2747      if ( latIndex == 1 .or. latIndex > OmPstdCH%n_lat(stnidIndex)) then
2748        if ( abs(OmPstdCH%std(posIndex) - MPC_missingValue_R8) > minDiff ) then
2749          OmP_err_stddev = MPC_missingValue_R8    
2750        else     
2751          OmP_err_stddev = OmPstdCH%std(posIndex)
2752        end if
2753      else        
2754        if ( abs(OmPstdCH%std(posIndex-1) - MPC_missingValue_R8) < minDiff .and. &
2755             abs(OmPstdCH%std(posIndex) - MPC_missingValue_R8) < minDiff ) then
2756          OmP_err_stddev = MPC_missingValue_R8  
2757        else if ( abs(OmPstdCH%std(posIndex-1) - MPC_missingValue_R8) < minDiff ) then
2758          OmP_err_stddev = OmPstdCH%std(posIndex)
2759        else if ( abs(OmPstdCH%std(posIndex) - MPC_missingValue_R8) < minDiff ) then
2760          OmP_err_stddev = OmPstdCH%std(posIndex-1)
2761        else
2762          OmP_err_stddev = (OmPstdCH%std(posIndex-1)*(OmPstdCH%lat(ibegin+latIndex)-zlat)+ &
2763            OmPstdCH%std(posIndex)*(zlat-OmPstdCH%lat(ibegin+latIndex-1)))/                 &
2764            (OmPstdCH%lat(ibegin+latIndex)-OmPstdCH%lat(ibegin+latIndex-1))
2765        end if
2766      end if       
2767    else
2768      if ( abs(OmPstdCH%std(posIndex) - MPC_missingValue_R8) > minDiff ) then
2769        OmP_err_stddev = MPC_missingValue_R8    
2770      else     
2771        OmP_err_stddev = OmPstdCH%std(posIndex)  
2772      end if           
2773    end if
