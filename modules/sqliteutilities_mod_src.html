 45  function sqlu_sqlColumnExists(fileName, tableName, columnName) result(columnExists)
 46    !
 47    !:Purpose: Check if a column exists in the sqlite file/table
 48    !
 49    implicit none
 50
 51    ! arguments:
 52    character(len=*),              intent(in)  :: fileName
 53    character(len=*),              intent(in)  :: tableName
 54    character(len=*),              intent(in)  :: columnName
 55    logical                                    :: columnExists
 56
 57    ! locals:
 58    integer                     :: ierr
 59    character(len=3000)         :: query, sqliteOutput
 60    character(len=lenSqlName)   :: upperColumnName
 61    type(fSQL_STATUS)           :: stat ! sqlite error status
 62    type(fSQL_DATABASE)         :: db   ! sqlite file handle
 63    logical, parameter          :: debug = .false.
 64    character(len=*), parameter :: myName = 'sqlu_sqlColumnExists'
 65    
 66    ! open the SQLite file
 67    call fSQL_open( db, trim(fileName), status=stat )
 68    if ( fSQL_error(stat) /= FSQL_OK ) then
 69      call utl_abort( myName//fSQL_errmsg(stat) )
 70    end if
 71
 72    upperColumnName = trim(columnName)
 73    ierr = clib_toUpper(upperColumnName)
 74
 75    query = "select count(*) from pragma_table_info('" // &
 76            trim(tableName) // "') where upper(name)='" // trim(upperColumnName) // "';"
 77    if (debug) write(*,*) myName//': query = ', trim(query)
 78
 79    sqliteOutput = sqlu_query(db,trim(query))
 80    if (debug) write(*,*) myName//': output = XXX' // trim(sqliteOutput) // 'XXX'
 81    columnExists = (trim(sqliteOutput) /= '0')
 82
 83    ! close the sqlite file
 84    call fSQL_close( db, stat )
 85
 86  end function sqlu_sqlColumnExists

 91  function sqlu_sqlTableExists(fileName, tableName) result(tableExists)
 92    !
 93    !:Purpose: Check if a table exists in the sqlite file
 94    !
 95    implicit none
 96
 97    ! arguments:
 98    character(len=*),              intent(in)  :: fileName
 99    character(len=*),              intent(in)  :: tableName
100    logical                                    :: tableExists
101
102    ! locals:
103    integer                     :: ierr
104    logical                     :: finished
105    character(len=3000)         :: query, sqliteOutput
106    character(len=lenSqlName)   :: upperTableName
107    type(fSQL_STATUS)           :: stat ! sqlite error status
108    type(fSQL_DATABASE)         :: db   ! sqlite file handle
109    type(fSQL_STATEMENT)        :: stmt ! precompiled sqlite statements
110    logical, parameter          :: debug = .false.
111    character(len=*), parameter :: myName = 'sqlu_sqlTableExists'
112
113    ! open the sqlite file
114    call fSQL_open( db, trim(fileName), status=stat )
115    if ( fSQL_error(stat) /= FSQL_OK ) then
116      call utl_abort( myName//': fSQL_open '//fSQL_errmsg(stat) )
117    end if
118
119    upperTableName = trim(tableName)
120    ierr = clib_toUpper(upperTableName)
121
122    query = "select upper(name) as uppername from sqlite_master where " // &
123            "type='table' and uppername='" // trim(upperTableName) // "';"
124    if (debug) write(*,*) myName//': query = ', trim(query)
125
126    call fSQL_prepare( db, trim(query), stmt, stat)
127    finished = .false.
128    call fSQL_get_row( stmt, finished )
129    call fSQL_get_column( stmt, COL_INDEX = 1, CHAR_VAR = sqliteOutput )
130    call fSQL_get_row( stmt, finished )
131    call fSQL_finalize( stmt )
132    if (debug) write(*,*) myName//': output = XXX' // trim(sqliteOutput) // 'XXX'
133    tableExists = (trim(sqliteOutput) == trim(upperTableName))
134
135    ! close the sqlite file
136    call fSQL_close( db, stat ) 
137
138  end function sqlu_sqlTableExists

143  subroutine sqlu_getSqlColumnNames(sqlColumnNames, fileName, tableName, dataType)
144    !
145    !:Purpose: Read the column names in the sqlite file for the specified table.
146    !
147    implicit none
148
149    ! arguments:
150    character(len=*), allocatable, intent(out) :: sqlColumnNames(:)
151    character(len=*),              intent(in)  :: fileName
152    character(len=*),              intent(in)  :: tableName
153    character(len=*),              intent(in)  :: dataType
154
155    ! locals:
156    integer :: numRows, numColumns, rowIndex, ierr
157    character(len=100), allocatable :: charValues(:,:)
158    character(len=200)       :: dataTypeCriteria
159    character(len=3000)      :: query
160    type(fSQL_STATUS)        :: stat ! sqlite error status
161    type(fSQL_DATABASE)      :: db   ! sqlite file handle
162    type(fSQL_STATEMENT)     :: stmt ! precompiled sqlite statements
163    character(len=*), parameter :: myName = 'sqlu_getSqlColumnNames'
164
165    ! open the sqlite file
166    call fSQL_open( db, trim(fileName), status=stat )
167    if ( fSQL_error(stat) /= FSQL_OK ) then
168      call utl_abort( myName//': fSQL_open '//fSQL_errmsg(stat) )
169    end if
170
171    ! read the column names
172    if (trim(dataType) == 'varchar') then
173      dataTypeCriteria = 'substr(type,1,7)="varchar"'
174    else if (trim(dataType) == 'numeric') then
175      dataTypeCriteria = 'type="real" or type="REAL" or type="double" or ' // &
176                         'type="DOUBLE" or type="integer" or type="INTEGER" or ' // &
177                         'type="INT" or type=""'
178    else if (trim(dataType) /= 'all') then
179      call utl_abort( myName//': invalid dataType = ' // trim(dataType) )
180    end if
181    if (trim(dataType) == 'all') then
182      query = 'select name from pragma_table_info("' // trim(tableName) // '");'
183    else
184      query = 'select name from pragma_table_info("' // trim(tableName) // &
185              '") where ' // trim(dataTypeCriteria) // ' ;'
186    end if
187    call fSQL_prepare( db, trim(query) , stmt, stat )
188    call fSQL_get_many( stmt, nrows=numRows, ncols=numColumns, &
189                        mode=FSQL_CHAR, status=stat )
190    if ( fSQL_error(stat) /= FSQL_OK ) then
191      call utl_abort( myName//': problem with fSQL_get_many '//fSQL_errmsg(stat))
192    end if
193    allocate( charValues(numRows, numColumns) )
194    call fSQL_fill_matrix( stmt, charValues )
195
196    ! copy to output array and ensure they are upper case
197    allocate( sqlColumnNames(numRows) )
198    do rowIndex = 1, numRows
199      sqlColumnNames(rowIndex) = charValues(rowIndex,1)
200      ierr = clib_toUpper(sqlColumnNames(rowIndex))
201    end do
202    deallocate( charValues )
203
204    ! clean up and close the sqlite file
205    call fSQL_free_mem( stmt )
206    call fSQL_finalize( stmt )
207    call fSQL_close( db, stat ) 
208
209  end subroutine sqlu_getSqlColumnNames

214  subroutine sqlu_getColumnValuesNum(columnValues, fileName, tableName, &
215                                     sqlColumnNames, extraQuery_opt)
216    !
217    !:Purpose: Read the column values from sqlite file for the specified table
218    !           and column names.
219    !
220    implicit none
221
222    ! arguments:
223    real(8), allocatable, intent(out) :: columnValues(:,:)
224    character(len=*),     intent(in)  :: sqlColumnNames(:)
225    character(len=*),     intent(in)  :: fileName
226    character(len=*),     intent(in)  :: tableName
227    character(len=*),     intent(in), optional :: extraQuery_opt
228
229    ! locals:
230    integer :: numRows, numColumns, columnIndex
231    character(len=3000)       :: query, extraQuery
232    type(fSQL_STATUS)         :: stat ! sqlite error status
233    type(fSQL_DATABASE)       :: db   ! sqlite file handle
234    type(fSQL_STATEMENT)      :: stmt ! precompiled sqlite statements
235
236    if (present(extraQuery_opt)) then
237      extraQuery = trim(extraQuery_opt)
238    else
239      extraQuery = ''
240    end if
241
242    ! open the sqlite file
243    call fSQL_open( db, trim(fileName), status=stat )
244    if ( fSQL_error(stat) /= FSQL_OK ) then
245      write(*,*) 'sqlu_getColumnValuesNum: fSQL_open: ', fSQL_errmsg(stat)
246      call utl_abort( 'sqlu_getColumnValuesNum: fSQL_open' )
247    end if
248
249    ! build the sqlite query
250    query = 'select'
251    numColumns = size(sqlColumnNames)
252    do columnIndex = 1, numColumns
253      query = trim(query) // ' ' // trim(sqlColumnNames(columnIndex))
254      if (columnIndex < numColumns) query = trim(query) // ','
255    end do
256    query = trim(query) // ' from ' // trim(tableName)
257    query = trim(query) // ' ' // trim(extraQuery) // ';'
258    write(*,*) 'sqlu_getColumnValuesNum: query ---> ', trim(query)
259
260    ! read the values from the file
261    call fSQL_prepare( db, trim(query) , stmt, status=stat )
262    call fSQL_get_many( stmt, nrows=numRows, ncols=numColumns, mode=FSQL_REAL8, &
263                        real8_missing=MPC_missingValue_R8, status=stat )
264    if ( fSQL_error(stat) /= FSQL_OK ) then
265      write(*,*) 'sqlu_getColumnValuesNum: fSQL_get_many: ', fSQL_errmsg(stat)
266      call utl_abort('sqlu_getColumnValuesNum: problem with fSQL_get_many')
267    end if
268    write(*,*) 'sqlu_getColumnValuesNum: numRows = ', numRows, ', numColumns = ', numColumns
269    allocate( columnValues(numRows, numColumns) )
270    columnValues(:,:) = 0.0d0
271    call fSQL_fill_matrix( stmt, columnValues )
272
273    ! close the sqlite file
274    call fSQL_free_mem( stmt )
275    call fSQL_finalize( stmt )
276    call fSQL_close( db, stat ) 
277
278  end subroutine sqlu_getColumnValuesNum

343  subroutine sqlu_getColumnValuesDateStr(columnDateValues, columnTimeValues, fileName, &
344                                         tableName, sqlColumnName)
345    !
346    !:Purpose: Read the column values from sqlite file for the specified table
347    !           and column names.
348    !
349    implicit none
350
351    ! arguments:
352    integer, allocatable, intent(out) :: columnDateValues(:)
353    integer, allocatable, intent(out) :: columnTimeValues(:)
354    character(len=*),     intent(in)  :: sqlColumnName
355    character(len=*),     intent(in)  :: fileName
356    character(len=*),     intent(in)  :: tableName
357
358    ! locals:
359    integer              :: numRows, numColumns, rowIndex
360    character(len=20), allocatable :: columnValuesStr(:,:)
361    character(len=3000)  :: query
362    type(fSQL_STATUS)    :: stat ! sqlite error status
363    type(fSQL_DATABASE)  :: db   ! sqlite file handle
364    type(fSQL_STATEMENT) :: stmt ! precompiled sqlite statements

366    ! open the sqlite file

368    if ( fSQL_error(stat) /= FSQL_OK ) then
369      write(*,*) 'sqlu_getColumnValuesDateStr: fSQL_open: ', fSQL_errmsg(stat)
370      call utl_abort( 'sqlu_getColumnValuesDateStr: fSQL_open' )
371    end if
372
373    ! Get the date and time
374
375    ! build the sqlite query
376    query = "select strftime('%Y%m%d'," // trim(sqlColumnName) // &
377            "), strftime('%H%M'," // trim(sqlColumnName) // ") " // &
378            "from " // trim(tableName) // ";"
379    write(*,*) 'sqlu_getColumnValuesDateStr: query ---> ', trim(query)
380
381    ! read the values from the file
382    call fSQL_prepare( db, trim(query), stmt, status=stat )
383    call fSQL_get_many( stmt, nrows=numRows, ncols=numColumns, &
384                        mode=FSQL_CHAR, status=stat )
385    if ( fSQL_error(stat) /= FSQL_OK ) then
386      write(*,*) 'sqlu_getColumnValuesDateStr: fSQL_get_many: ', fSQL_errmsg(stat)
387      call utl_abort('sqlu_getColumnValuesDateStr: problem with fSQL_get_many')
388    end if
389    write(*,*) 'sqlu_getColumnValuesDateStr: numRows = ', numRows, ', numColumns = ', numColumns
390    allocate( columnValuesStr(numRows,2) )

411  subroutine sqlu_getInitialIdObsData(obsDat, obsFamily, idObs, idData, codeTypeList_opt)
412    !
413    !:Purpose: Compute initial value for idObs and idData that will ensure
414    !           unique values over all mpi tasks
415    !
416    implicit none
417
418    ! Arguments:
419    type(struct_obs)  :: obsdat
420    character(len=*)  :: obsFamily    
421    integer           :: idObs, idData
422    integer, optional :: codeTypeList_opt(:)
423
424    ! Locals:
425    integer                :: headerIndex, numHeader, numBody, codeType, ierr
426    integer, allocatable   :: allNumHeader(:), allNumBody(:)
427
428    numHeader = 0
429    numBody = 0
430    call obs_set_current_header_list(obsdat, obsFamily)
431    HEADERCOUNT: do
432      headerIndex = obs_getHeaderIndex(obsdat)
433      if (headerIndex < 0) exit HEADERCOUNT
434      if (present(codeTypeList_opt)) then
435        codeType  = obs_headElem_i(obsdat, OBS_ITY, headerIndex)
436        if (all(codeTypeList_opt(:) /= codeType)) cycle HEADERCOUNT
437      end if
438      numHeader = numHeader + 1
439      numBody = numBody + obs_headElem_i(obsdat, OBS_NLV, headerIndex)
440    end do HEADERCOUNT
441    allocate(allNumHeader(mmpi_nprocs))
442    allocate(allNumBody(mmpi_nprocs))
443    call rpn_comm_allgather(numHeader,1,'mpi_integer',       &
444                            allNumHeader,1,'mpi_integer','GRID',ierr)
445    call rpn_comm_allgather(numBody,1,'mpi_integer',       &
446                            allNumBody,1,'mpi_integer','GRID',ierr)
447    if (mmpi_myid > 0) then
448      idObs = sum(allNumHeader(1:mmpi_myid))
449      idData = sum(allNumBody(1:mmpi_myid))
450    else
451      idObs = 0
452      idData = 0
453    end if
454    deallocate(allNumHeader)
455    deallocate(allNumBody)
456
457  end subroutine sqlu_getInitialIdObsData

462  function sqlu_query(db,query)
463    !
464    !:Purpose: To create a query to read an SQLite file
465    !
466    implicit none
467
468    ! arguments
469    type(fSQL_DATABASE)  :: db    ! type handle for  SQLIte file
470    character(len = *)   :: query
471    ! locals
472    character(len = 256) :: sqlu_query, result
473    logical finished
474    type(fSQL_STATEMENT) :: stmt !  prepared statement for  SQLite
475    type(fSQL_STATUS)    :: stat !type error status
476
477    result=''
478    call fSQL_prepare(db, trim(query), stmt, stat)
479    if (fSQL_error(stat) /= FSQL_OK) call sqlu_handleError(stat,'fSQL_prepare: ')
480    finished=.false.
481    call fSQL_get_row(stmt, finished)
482
483    ! Put result of query into variable
484    call fSQL_get_column(stmt, COL_INDEX = 1, char_var = result)
485    call fSQL_get_row(stmt, finished)
486    if (.not. finished) write(*,*) ' SQL QUERY ---> QUERY RETURNS MORE THAN ONE ROW...  '
487    call fSQL_finalize(stmt)
488    sqlu_query = trim(result)
489
490  end function sqlu_query
