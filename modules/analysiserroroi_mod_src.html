 17module analysisErrorOI_mod
 18  ! MODULE analysisErrorOI (prefix='aer' category='1. High-level functionality')
 19  !
 20  !:Purpose: Calculate the analysis-error standard deviation.
 21  !           The method used is Optimal Interpolation,
 22  !           where it is assumed that only a subset of the
 23  !           total number of observations influence the analysis at a given grid point.
 24  !           By default, everything in the module is private.
 25  !           The data is accessed by external subroutines through public subroutines
 26  !           and functions calls.
 27  !

 80  !--------------------------------------------------------------------------
 81  ! aer_analysisError
 82  !--------------------------------------------------------------------------
 83  subroutine aer_analysisError(obsSpaceData, hco_ptr, vco_ptr, trlmFileName)
 84    !
 85    !:Purpose: Calculate analysis-error variance.
 86    !
 87    implicit none
 88
 89    ! Arguments
 90    type(struct_obs), intent(in) :: obsSpaceData
 91    type(struct_hco), pointer    :: hco_ptr
 92    type(struct_vco), pointer    :: vco_ptr
 93    character(len=*), intent(in) :: trlmFileName
 94
 95    ! Local Variables
 96    integer :: fnom, fclos, nulnam, ierr
 97
 98    type(struct_gsv) :: stateVectorBkGnd
 99    type(struct_gsv) :: stateVectorAnal
100    integer, allocatable :: numObs(:,:)
101    real(8), pointer :: bkGndErrorStdDev_ptr(:,:,:,:), analysisErrorStdDev_ptr(:,:,:,:)
102
103    real(8), allocatable :: obsOperator(:,:), Bmatrix(:,:), PHiA(:), &
104                            innovCovariance(:,:), obsErrorVariance(:), &
105                            PH(:,:), KH(:), IKH(:), innovCovarianceInverse(:,:)
106    integer, parameter :: maxvar = 15000
107
108    integer :: statei(maxvar)    ! Model grid coordinate i of neighbors
109    integer :: statej(maxvar)    ! Model grid coordinate j of neighbors
110
111    integer :: numVariables, varIndex1, varIndex2, currentAnalVarIndex
112
113    logical :: found
114
115    integer :: stepIndex, levIndex, lonIndex, latIndex, gridIndex, headerIndex, &
116               bodyIndex, kIndex, procIndex
117    integer :: influentObsIndex2, influentObsIndex
118
119    real(8) :: scaling
120    integer :: xStateIndex, yStateIndex
121
122    integer :: numInfluentObs, xIndex1, yIndex1, xIndex2, yIndex2
123    integer :: gridptCount, gridpt
124
125    real(8) :: distance
126
127    real(kdkind), allocatable :: positionArray(:,:)
128    real(8),      allocatable :: latInRad(:,:), lonInRad(:,:)
129
130    character(len=*), parameter :: myName = 'aer_analysisError'
131    character(len=*), parameter :: correlationLengthFileName = './bgstddev'
132    character(len=4), pointer   :: anlVar(:)
133    type(struct_gsv)            :: statevector
134    real(4), pointer            :: field3D_r4_ptr(:,:,:)
135
136    character(len=2 ) :: typvar
137
138    type(struct_columnData) :: column
139    type(struct_columnData) :: columng
140
141    real(8) :: leadTimeInHours
142
143    ! namelist variables:
144    real(8) :: maxAnalysisErrorStdDev ! maximum limit imposed on analysis error stddev
145    namelist /namaer/ maxAnalysisErrorStdDev
146
147    if( mmpi_nprocs > 1 ) then
148      write(*,*) 'mmpi_nprocs = ',mmpi_nprocs
149      call utl_abort( myName// &
150                      ': this version of the code should only be used with one mpi task.')
151    end if
152    if( mmpi_myid > 0 ) return
153
154    write(*,*) '**********************************************************'
155    write(*,*) '** '//myName//': Calculate analysis-error variance **'
156    write(*,*) '**********************************************************'
157
158    ! Read the namelist
159    ! default values
160    maxAnalysisErrorStdDev = 1.0d0
161    if (.not. utl_isNamelistPresent('namaer','./flnml')) then
162      if (mmpi_myid == 0) then
163        call msg(myName, 'namaer is missing in the namelist.')
164        call msg(myName, 'the default values will be taken.')
165      end if
166    else
167      ! reading namelist variables
168      nulnam = 0
169      ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
170      read(nulnam, nml = namaer, iostat = ierr)
171      if ( ierr /= 0 ) call utl_abort( myName//': Error reading namelist')
172      ierr = fclos(nulnam)
173    end if
174    write(*, nml = namaer)
175
176    nullify(anlVar)
177    call gsv_varNamesList(anlVar)
178
179    if (size(anlVar) > 1 .or. anlVar(1) /= 'GL') then
180      call msg(myName, 'The code has only been tested with the analysis variable GL.')
181      call utl_abort( myName//': Check namelist and/or revise the code.')
182    end if
183
184    call gsv_allocate( stateVectorBkGnd, 1, hco_ptr, vco_ptr, dateStamp_opt=-1, &
185                       mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
186                       varNames_opt=(/anlVar(1)/), dataKind_opt=8 )
187    call gsv_allocate( stateVectorAnal,  1, hco_ptr, vco_ptr, dateStamp_opt=-1, &
188                       varNames_opt=(/anlVar(1)/), dataKind_opt=8 )
189
190    typvar = 'P@'
191
192    call gio_readFromFile( stateVectorBkGnd, trlmFileName, aer_backgroundEtiket, typvar )
193
194    leadTimeInHours = real(stateVectorBkGnd%deet*stateVectorBkGnd%npasList(1),8)/3600.0d0
195    call incdatr(stateVectorAnal%dateOriginList(1), stateVectorBkGnd%dateOriginList(1), &
196                 leadTimeInHours)
197
198    call gsv_copyMask(stateVectorBkGnd, stateVectorAnal)
199    stateVectorAnal%etiket = aer_analysisEtiket
200
201    call col_setVco(column, vco_ptr)
202    call col_allocate(column,  obs_numHeader(obsSpaceData), varNames_opt=(/anlVar(1)/))
203    call col_setVco(columng, vco_ptr)
204    call col_allocate(columng, obs_numHeader(obsSpaceData), varNames_opt=(/anlVar(1)/))
205    call s2c_tl(statevectorBkGnd, column, columng, obsSpaceData)
206
207    ni = stateVectorBkGnd%hco%ni
208    nj = stateVectorBkGnd%hco%nj
209
210    allocate( lonInRad( ni, nj ), latInRad( ni, nj) )
211    allocate( Lcorr( ni, nj ) )
212
213    write(*,*) myName// &
214               ': Correlation length scale 2D field will be read from the file: ', &
215               correlationLengthFileName
216    call gsv_allocate( statevector, 1, hco_ptr, vco_ptr, dateStamp_opt=-1, &
217                       dataKind_opt=4, hInterpolateDegree_opt='LINEAR', &
218                       varNames_opt=(/anlVar(1)/) )
219    call gsv_zero( statevector )
220    call gio_readFromFile( statevector, correlationLengthFileName, 'CORRLEN', ' ', &
221                           unitConversion_opt = .false. )
222
223    call gsv_getField( statevector, field3D_r4_ptr, anlVar(1) )
224    ! Convert from km to meters
225    Lcorr(:,:) = 1000.0d0*real(field3D_r4_ptr( :, :, 1 ), 8)
226    write(*,*) myName//': min/max correlation length scale 2D field: ', &
227               minval( Lcorr(:,:) ), maxval( Lcorr(:,:) )
228    call gsv_deallocate( statevector )
229
230    ! create kdtree
231    write(*,*) myName//': start creating kdtree for stateVectorBkGnd'
232    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
233
234    allocate(positionArray(3,ni*nj))
235
236    gridIndex = 0
237    do latIndex = 1, nj
238      do lonIndex = 1, ni
239
240        gridIndex = gridIndex + 1
241        latInRad(lonIndex,latIndex) = real( &
242                                      stateVectorBkGnd%hco%lat2d_4(lonIndex,latIndex), 8)
243        lonInRad(lonIndex,latIndex) = real( &
244                                      stateVectorBkGnd%hco%lon2d_4(lonIndex,latIndex), 8)
245
246        positionArray(:,gridIndex) = kdtree2_3dPosition(lonInRad(lonIndex,latIndex), &
247                                                        latInRad(lonIndex,latIndex))
248
249      end do
250    end do
251
252    write(*,*) myName//': latInRad min/max: ', minval(latInRad(:,:) ), &
253                                               maxval( latInRad(:,:) )
254    write(*,*) myName//': lonInRad min/max: ', minval(lonInRad(:,:) ), &
255                                               maxval( lonInRad(:,:) )
256
257    nullify(tree)
258    tree => kdtree2_create(positionArray, sort=.false., rearrange=.true.) 
259
260    deallocate(positionArray)
261
262    write(*,*) myName//': done creating kdtree for stateVectorBkGnd'
263    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
264
265    ! Go through all observations a first time to get
266    ! the number of influential observations
267    ! in order to allocate memory appropriately.
268
269    allocate(influentObs(ni, nj))
270    allocate(     numObs(ni, nj))
271
272    call findObs(obsSpaceData, stateVectorBkGnd, numObs)
273
274    ! Memory allocation
275
276    do latIndex = 1, nj
277      do lonIndex = 1, ni
278        influentObs(lonIndex,latIndex)%numObs = numObs(lonIndex,latIndex)
279        allocate(influentObs(lonIndex,latIndex)%headerIndex(numObs(lonIndex,latIndex)))
280        allocate(influentObs(lonIndex,latIndex)%bodyIndex(numObs(lonIndex,latIndex)))
281      end do
282    end do
283
284    ! Go through all observations a second time to get
285    ! the indexes of the influential observations.
286    ! This is not efficient to go through all observations twice
287    ! but it saves lot of memory space.
288
289    call findObs(obsSpaceData, stateVectorBkGnd, numObs)
290
291    deallocate(numObs)
292
293    write(*,*) myName//': obs found'
294    write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
295
296    ! Calculate analysis-error one analysis variable (grid point) at a time
297
298    call gsv_getField(stateVectorBkGnd, bkGndErrorStdDev_ptr, anlVar(1))
299    call gsv_getField(stateVectorAnal,  analysisErrorStdDev_ptr, anlVar(1))
300
301    ! Initialisation
302    do stepIndex = 1, stateVectorBkGnd%numStep
303      do levIndex = 1, gsv_getNumLev(stateVectorBkGnd,vnl_varLevelFromVarname(anlVar(1)))
304        do latIndex = 1, nj
305          do lonIndex = 1, ni
306            analysisErrorStdDev_ptr(lonIndex,latIndex,levIndex,stepIndex) = &
307               bkGndErrorStdDev_ptr(lonIndex,latIndex,levIndex,stepIndex)
308          end do
309        end do
310      end do
311    end do
312
313    ! Only variables assigned within or by the loop can be private.
314
315    STEP: do stepIndex = 1, stateVectorBkGnd%numStep
316      LEVEL: do levIndex = 1, &
317                           gsv_getNumLev(stateVectorBkGnd,vnl_varLevelFromVarname(anlVar(1)))
318
319        !$omp parallel do default(shared) schedule(dynamic) private(obsOperator, Bmatrix, &
320        !$omp        PHiA, innovCovariance, innovCovarianceInverse, obsErrorVariance, &
321        !$omp        PH, KH, IKH, &
322        !$omp        statei, statej, &
323        !$omp        numVariables, varIndex1, varIndex2, currentAnalVarIndex, found, &
324        !$omp        influentObsIndex2, influentObsIndex, &
325        !$omp        scaling, xStateIndex, yStateIndex, lonIndex, latIndex, headerIndex, &
326        !$omp        bodyIndex, kIndex, procIndex, &
327        !$omp        numInfluentObs, xIndex1, yIndex1, xIndex2, yIndex2, distance, &
328        !$omp        interpWeight, obsLatIndex, obsLonIndex, gridptCount, gridpt)
329        YINDEX: do latIndex = 1, nj
330          XINDEX: do lonIndex = 1, ni
331
332            numInfluentObs = influentObs(lonIndex,latIndex)%numObs
333
334            if ( numInfluentObs == 0 .or. &
335                 .not. stateVectorBkGnd%oceanMask%mask(lonIndex,latIndex,levIndex) ) cycle
336
337            ! form the observation-error covariance (diagonal) matrix
338
339            allocate(obsErrorVariance(numInfluentObs))
340
341            do influentObsIndex = 1, numInfluentObs
342              bodyIndex = influentObs(lonIndex,latIndex)%bodyIndex(influentObsIndex)
343              obsErrorVariance(influentObsIndex) = ( obs_bodyElem_r(obsSpaceData,OBS_OER, &
344                                                                    bodyIndex) )**2
345            end do
346
347            ! find all model variables involved here
348
349            numVariables = 0
350            statei(:) = 0
351            statej(:) = 0
352            do influentObsIndex = 1, numInfluentObs
353
354              headerIndex = influentObs(lonIndex,latIndex)%headerIndex(influentObsIndex)
355              bodyIndex   = influentObs(lonIndex,latIndex)%bodyIndex(influentObsIndex)
356
357              scaling = oop_iceScaling(obsSpaceData, bodyIndex)
358
359              if ( scaling /= 0.0d0 ) then
360
361                do kIndex = stateVectorBkGnd%mykBeg, stateVectorBkGnd%mykEnd
362                  do procIndex = 1, mmpi_nprocs
363
364                    call s2c_getWeightsAndGridPointIndexes(headerIndex, kIndex, &
365                         stepIndex, procIndex, interpWeight, obsLatIndex, obsLonIndex, &
366                         gridptCount)
367
368                    do gridpt = 1, gridptCount
369
370                      if ( interpWeight(gridpt) /= 0.0d0 ) then
371
372                        xStateIndex = obsLonIndex(gridpt)
373                        yStateIndex = obsLatIndex(gridpt)
374
375                        found = .false.
376                        do varIndex2=1,numVariables
377                          if(xStateIndex == statei(varIndex2) .and. &
378                             yStateIndex == statej(varIndex2)) then
379                            found = .true.
380                            exit
381                          end if
382                        end do
383                        if(.not. found) then
384                          numVariables = numVariables + 1
385                          if(numVariables > maxvar) then
386                            call utl_abort( myName//': Structure state'// &
387                                 ' too small in subroutine'// &
388                                 ' analysis_error_mod'// &
389                                 ' Increase maxvar')
390                          end if
391                          statei(numVariables) = xStateIndex
392                          statej(numVariables) = yStateIndex
393                        end if
394
395                      end if
396
397                    end do
398
399                  end do
400                end do
401
402              end if
403
404            end do
405
406            ! make sure that current analysis variable is part of the state
407            ! vector even if it does not participate in obs calculation
408
409            found = .false.
410            do varIndex2=1,numVariables
411              if(lonIndex == statei(varIndex2) .and. &
412                 latIndex == statej(varIndex2)) then
413                currentAnalVarIndex = varIndex2
414                found = .true.
415                exit
416              end if
417            end do
418            if(.not. found) then
419              numVariables = numVariables + 1
420              if(numVariables > maxvar) then
421                call utl_abort(myName//': Structure state too small in'// &
422                     ' subroutine analysis_error_mod'// &
423                     ' increase maxvar')
424              end if
425              currentAnalVarIndex = numVariables
426              statei(numVariables) = lonIndex
427              statej(numVariables) = latIndex
428            end if
429
430            ! form the observation operator matrix (obsOperator)
431
432            allocate(obsOperator(numVariables,numInfluentObs))
433
434            obsOperator(:,:) = 0.0d0
435
436            varIndex2 = 0
437
438            do influentObsIndex = 1, numInfluentObs
439
440              headerIndex = influentObs(lonIndex,latIndex)%headerIndex(influentObsIndex)
441              bodyIndex   = influentObs(lonIndex,latIndex)%bodyIndex(influentObsIndex)
442
443              scaling = oop_iceScaling(obsSpaceData, bodyIndex)
444
445              if ( scaling /= 0.0d0 ) then
446
447                do kIndex = stateVectorBkGnd%mykBeg, stateVectorBkGnd%mykEnd
448                  do procIndex = 1, mmpi_nprocs
449
450                    call s2c_getWeightsAndGridPointIndexes(headerIndex, kIndex, &
451                         stepIndex, procIndex, interpWeight, obsLatIndex, obsLonIndex, &
452                         gridptCount)
453
454                    do gridpt = 1, gridptCount
455
456                      if ( interpWeight(gridpt) /= 0.0d0 ) then
457
458                        xStateIndex = obsLonIndex(gridpt)
459                        yStateIndex= obsLatIndex(gridpt)
460
461                        found = .false.
462                        do while (.not. found)
463                          if(varIndex2 < numVariables) then
464                            varIndex2 = varIndex2 + 1
465                          else
466                            varIndex2 = 1
467                          end if
468                          if(xStateIndex == statei(varIndex2) .and. &
469                             yStateIndex == statej(varIndex2)) then
470                            found = .true.
471                            obsOperator(varIndex2,influentObsIndex) = scaling* &
472                                                                      interpWeight(gridpt)
473                          end if
474                        end do
475                        if(.not. found) then
476                          write(*,*) 'xStateIndex = ',xStateIndex
477                          write(*,*) 'yStateIndex = ',yStateIndex
478                          write(*,*) 'lonIndex = ',lonIndex
479                          write(*,*) 'latIndex = ',latIndex
480                          write(*,*) 'gridptCount = ',gridptCount
481                          write(*,*) 'gridpt = ',gridpt
482                          write(*,*) 'numVariables = ',numVariables
483                          do varIndex2=1,numVariables
484                            write(*,*) 'varIndex2 statei statej = ',statei(varIndex2), &
485                                                                    statej(varIndex2)
486                          end do
487                          call utl_abort( myName//': not found in state vect.' )
488                        end if
489
490                      end if
491
492                    end do
493
494                  end do
495                end do
496
497              end if
498
499            end do
500
501            ! form the background-error covariance matrix
502
503            allocate(Bmatrix(numVariables,numVariables))
504
505            Bmatrix(:,:) = 0.0d0
506
507            do varIndex2 = 1, numVariables
508
509              xIndex2 = statei(varIndex2)
510              yIndex2 = statej(varIndex2)
511
512              do varIndex1 = varIndex2, numVariables
513
514                xIndex1 = statei(varIndex1)
515                yIndex1 = statej(varIndex1)
516
517                if(xIndex2 == xIndex1 .and. yIndex2 == yIndex1) then
518                  Bmatrix(varIndex1,varIndex2) = bkGndErrorStdDev_ptr(xIndex1,yIndex1,levIndex,stepIndex)**2
519                else if(Lcorr(xIndex2,yIndex2) > 0.0d0) then
520                  distance = phf_calcDistance(latInRad(xIndex2,yIndex2), &
521                                              lonInRad(xIndex2,yIndex2), &
522                                              latInRad(xIndex1,yIndex1), &
523                                              lonInRad(xIndex1,yIndex1))
524                  Bmatrix(varIndex1,varIndex2) = bkGndErrorStdDev_ptr(xIndex2,yIndex2,levIndex,stepIndex)* &
525                                     bkGndErrorStdDev_ptr(xIndex1,yIndex1,levIndex,stepIndex) &
526                                     *exp(-0.5*(distance/Lcorr(xIndex2,yIndex2))**2)
527                end if
528
529                ! symmetric matrix !
530                Bmatrix(varIndex2,varIndex1) = Bmatrix(varIndex1,varIndex2)
531
532              end do
533
534            end do
535
536            ! form the observation background error covariance matrix (PHT)
537
538            allocate(PH(numInfluentObs,numVariables))
539
540            ! PH = matmul (Bmatrix, transpose(obsOperator))
541            do varIndex1 = 1, numVariables
542              do influentObsIndex = 1, numInfluentObs
543                PH(influentObsIndex,varIndex1) = dot_product(Bmatrix(:,varIndex1), &
544                                                 obsOperator(:,influentObsIndex))
545              end do
546            end do
547
548            !  covariance matrix of the innovation (HPHT + R)
549
550            allocate(innovCovariance(numInfluentObs,numInfluentObs), &
551                     innovCovarianceInverse(numInfluentObs,numInfluentObs))
552
553            do influentObsIndex = 1, numInfluentObs
554
555              do influentObsIndex2 = 1, numInfluentObs
556                if ( influentObsIndex == influentObsIndex2 ) then
557                  innovCovariance(influentObsIndex2,influentObsIndex) = &
558                       dot_product(obsOperator(:,influentObsIndex), &
559                       PH(influentObsIndex2,:)) + &
560                       obsErrorVariance(influentObsIndex)
561                else
562                  innovCovariance(influentObsIndex2,influentObsIndex) = &
563                       dot_product(obsOperator(:,influentObsIndex), &
564                       PH(influentObsIndex2,:))
565                end if
566              end do
567
568            end do
569
570            ! Inverse of the covariance matrix of the innovation
571
572            innovCovarianceInverse(:,:) = innovCovariance(:,:)
573            call utl_matInverse(innovCovarianceInverse, numInfluentObs, printInformation_opt=.false.)
574
575            ! Kalman gain; this is the row corresponding to the analysis variable
576
577            allocate(PHiA(numInfluentObs))
578
579            do influentObsIndex = 1, numInfluentObs
580              PHiA(influentObsIndex) = dot_product(PH(:,currentAnalVarIndex), &
581                                       innovCovarianceInverse(influentObsIndex,:))
582            end do
583
584            ! compute the error variance of the analysis
585
586            allocate(KH(numVariables))
587
588            ! KH = matmul (PHiA, obsOperator)
589            do varIndex1 = 1, numVariables
590              KH(varIndex1) = dot_product(PHiA, obsOperator(varIndex1,:))
591            end do
592
593            ! IKH = I - KH
594
595            allocate(IKH(numVariables))
596
597            IKH = -KH
598
599            IKH(currentAnalVarIndex) = 1.0d0 - KH(currentAnalVarIndex)
600
601            analysisErrorStdDev_ptr(lonIndex,latIndex,levIndex,stepIndex) = &
602                 sqrt( dot_product (IKH, Bmatrix(:,currentAnalVarIndex)) )
603
604            if(analysisErrorStdDev_ptr(lonIndex,latIndex,levIndex,stepIndex) < 0.0) then
605              write(*,*) myName//'negative analysis-error Std dev. = ', &
606                   analysisErrorStdDev_ptr(lonIndex,latIndex,levIndex,stepIndex), &
607                   ' reset to zero at grid point (',lonIndex,latIndex,')'
608              analysisErrorStdDev_ptr(lonIndex,latIndex,levIndex,stepIndex) = &
609                   max(analysisErrorStdDev_ptr(lonIndex,latIndex,levIndex,stepIndex), 0.0)
610            end if
611
612            if(analysisErrorStdDev_ptr(lonIndex,latIndex,levIndex,stepIndex) > &
613                  bkGndErrorStdDev_ptr(lonIndex,latIndex,levIndex,stepIndex)) then
614              write(*,*) myName//'analysis-error Std dev. = ', &
615                   analysisErrorStdDev_ptr(lonIndex,latIndex,levIndex,stepIndex), &
616                   ' is larger than background-error ', &
617                   'Std dev. is kept at = ', &
618                   bkGndErrorStdDev_ptr(lonIndex,latIndex,levIndex,stepIndex)
619              analysisErrorStdDev_ptr(lonIndex,latIndex,levIndex,stepIndex) = &
620                   min(bkGndErrorStdDev_ptr(lonIndex,latIndex,levIndex,stepIndex), &
621                   maxAnalysisErrorStdDev)
622            end if
623
624            deallocate(Bmatrix, obsErrorVariance, innovCovariance, &
625                       innovCovarianceInverse, obsOperator, PH, PHiA, KH, IKH)
626            deallocate(influentObs(lonIndex,latIndex)%headerIndex)
627            deallocate(influentObs(lonIndex,latIndex)%bodyIndex)
628
629          end do XINDEX
630        end do YINDEX
631        !$omp end parallel do
632
633      end do LEVEL
634    end do STEP
635
636    deallocate(influentObs)
637    deallocate( Lcorr )
638    deallocate( lonInRad, latInRad )
639
640    call gio_writeToFile( stateVectorAnal, './anlm_000m', '', typvar_opt='A@', &
641                          containsFullField_opt=.true. )
642
643    call col_deallocate( columng )
644    call col_deallocate( column )
645    call gsv_deallocate( stateVectorBkGnd )
646    call gsv_deallocate( stateVectorAnal )
647
648  end subroutine aer_analysisError

650  !---------------------------------------------------------
651  ! findObs
652  !---------------------------------------------------------
653  subroutine findObs(obsSpaceData, stateVectorBkGnd, numObs)
654    !
655    !:Purpose: Find all observations used for the local analysis.
656    !
657    implicit none
658
659    ! Arguments
660    type(struct_obs), intent(in)  :: obsSpaceData
661    type(struct_gsv), intent(in)  :: stateVectorBkGnd
662    integer,          intent(out) :: numObs(ni,nj)
663
664    ! Local variables
665    integer :: headerIndex, bodyIndexBeg, bodyIndexEnd, bodyIndex, kIndex, stepIndex, &
666               procIndex
667    integer :: gridptCount, gridpt, numLocalGridptsFoundSearch
668
669    integer :: lonIndex, latIndex, resultsIndex, gridIndex
670    type(kdtree2_result) :: searchResults(maxNumLocalGridptsSearch)
671    real(kdkind)         :: refPosition(3), maxRadiusSquared
672    real(4) :: footprintRadius_r4 ! (metres)
673    real(4) :: influenceRadius_r4 ! (metres)
674
675    real(8) :: obsLonInRad, obsLatInRad, maxLcorr
676
677    call utl_tmg_start(122,'--AnalErrOI_FindObs')
678
679    maxLcorr = maxval( Lcorr(:,:) )
680
681    numObs(:,:) = 0
682
683    HEADER_LOOP: do headerIndex = 1, obs_numHeader(obsSpaceData)
684
685      bodyIndexBeg = obs_headElem_i(obsSpaceData,OBS_RLN,headerIndex)
686      bodyIndexEnd = obs_headElem_i(obsSpaceData,OBS_NLV,headerIndex) + bodyIndexBeg - 1
687
688      BODY_LOOP: do bodyIndex = bodyIndexBeg, bodyIndexEnd
689
690        if ( obs_bodyElem_i(obsSpaceData,OBS_ASS,bodyIndex) /= obs_assimilated ) &
691             cycle BODY_LOOP
692
693        footprintRadius_r4 = s2c_getFootprintRadius(obsSpaceData, stateVectorBkGnd, &
694                                                    headerIndex)
695        influenceRadius_r4 = max(0.0, footprintRadius_r4) + maxLcorr
696
697        if ( maxLcorr == 0.0d0 ) then
698
699          do kIndex = stateVectorBkGnd%mykBeg, stateVectorBkGnd%mykEnd
700            do stepIndex = 1, stateVectorBkGnd%numStep
701              do procIndex = 1, mmpi_nprocs
702
703                call s2c_getWeightsAndGridPointIndexes(headerIndex, &
704                     kIndex, stepIndex, procIndex, interpWeight, obsLatIndex, &
705                     obsLonIndex, gridptCount)
706
707                GRIDPT_LOOP: do gridpt = 1, gridptCount
708
709                  if ( interpWeight(gridpt) == 0.0d0 ) cycle GRIDPT_LOOP
710
711                  lonIndex = obsLonIndex(gridpt)
712                  latIndex = obsLatIndex(gridpt)
713
714                  numObs(lonIndex,latIndex) = numObs(lonIndex,latIndex) + 1
715                  if(associated(influentObs(lonIndex,latIndex)%bodyIndex)) then
716                    if(numObs(lonIndex,latIndex) > influentObs(lonIndex,latIndex)%numObs) then
717                      call utl_abort('findObs: Array too small')
718                    end if
719                    influentObs(lonIndex,latIndex)%headerIndex(numObs(lonIndex,latIndex)) = headerIndex
720                    influentObs(lonIndex,latIndex)%bodyIndex(numObs(lonIndex,latIndex)) = bodyIndex
721                  end if
722
723                end do GRIDPT_LOOP
724
725              end do
726            end do
727          end do
728
729        else
730
731          ! Determine the grid point nearest the observation.
732
733          obsLonInRad = obs_headElem_r(obsSpaceData, OBS_LON, headerIndex)
734          obsLatInRad = obs_headElem_r(obsSpaceData, OBS_LAT, headerIndex)
735
736          ! do the search
737          maxRadiusSquared = real(influenceRadius_r4, 8) ** 2
738          refPosition(:) = kdtree2_3dPosition(obsLonInRad, obsLatInRad)
739          call kdtree2_r_nearest(tp=tree, qv=refPosition, r2=maxRadiusSquared, &
740               nfound=numLocalGridptsFoundSearch, &
741               nalloc=maxNumLocalGridptsSearch, &
742               results=searchResults)
743
744          if (numLocalGridptsFoundSearch > maxNumLocalGridptsSearch ) then
745            call utl_abort('findObs: parameter maxNumLocalGridptsSearch must be increased')
746          end if
747
748          do resultsIndex = 1, numLocalGridptsFoundSearch
749
750            gridIndex = searchResults(resultsIndex)%idx
751            if ( gridIndex < 1 .or. &
752                 gridIndex > stateVectorBkGnd%hco%ni * stateVectorBkGnd%hco%nj ) then
753              write(*,*) 'analysisErrorStdDev: gridIndex=', gridIndex
754              call utl_abort('findObs: gridIndex out of bound.')
755            end if
756
757            latIndex = (gridIndex - 1) / stateVectorBkGnd%hco%ni + 1
758            lonIndex = gridIndex - (latIndex - 1) * stateVectorBkGnd%hco%ni
759            if ( lonIndex < 1 .or. lonIndex > stateVectorBkGnd%hco%ni .or. &
760                 latIndex < 1 .or. latIndex > stateVectorBkGnd%hco%nj ) then
761              write(*,*) 'analysisErrorStdDev: lonIndex=', lonIndex, &
762                                            ', latIndex=', latIndex
763              call utl_abort('findObs: lonIndex/latIndex out of bound.')
764            end if
765
766            numObs(lonIndex,latIndex) = numObs(lonIndex,latIndex) + 1
767            if(associated(influentObs(lonIndex,latIndex)%bodyIndex)) then
768              if(numObs(lonIndex,latIndex) > influentObs(lonIndex,latIndex)%numObs) then
769                call utl_abort('findObs: Array too small in subroutine findObs')
770              end if
771              influentObs(lonIndex,latIndex)%headerIndex(numObs(lonIndex,latIndex)) = &
772                          headerIndex
773              influentObs(lonIndex,latIndex)%bodyIndex(numObs(lonIndex,latIndex)) = &
774                          bodyIndex
775            end if
776
777          end do
778
779        end if
780
781      end do BODY_LOOP
782
783    end do HEADER_LOOP
784
785    call utl_tmg_stop(122)
786
787  end subroutine findObs

792  subroutine aer_daysSinceLastObs(obsSpaceData, hco_ptr, vco_ptr, trlmFileName)
793    !
794    !:Purpose: Update the field "days since last obs" with the newly assimilated obs.
795    !
796    implicit none
797
798    ! Arguments
799    type(struct_obs), intent(in)  :: obsSpaceData
800    type(struct_hco), pointer    :: hco_ptr
801    type(struct_vco), pointer    :: vco_ptr
802    character(len=*), intent(in) :: trlmFileName
803
804    ! Local Variables
805    type(struct_gsv) :: stateVectorBkGnd
806    type(struct_gsv) :: stateVectorAnal
807    real(8), pointer :: bkGndDaysSinceLastObs_ptr(:,:,:,:), analysisDaysSinceLastObs_ptr(:,:,:,:)
808
809    integer :: stepIndex, levIndex, lonIndex, latIndex, headerIndex
810    integer :: bodyIndexBeg, bodyIndexEnd, bodyIndex, kIndex, procIndex
811
812    integer :: gridptCount, gridpt
813
814    character(len=*), parameter :: myName = 'aer_daysSinceLastObs'
815
816    character(len=2 ) :: typvar
817    character(len=12) :: etiket
818
819    type(struct_columnData) :: column
820    type(struct_columnData) :: columng
821
822    real(8) :: leadTimeInHours
823
824    if( mmpi_nprocs > 1 ) then
825      write(*,*) 'mmpi_nprocs = ',mmpi_nprocs
826      call utl_abort( myName// &
827                      ': this version of the code should only be used with one mpi task.')
828    end if
829    if( mmpi_myid > 0 ) return
830
831    write(*,*) '**********************************************************'
832    write(*,*) '** '//myName//': Update the days since last obs **'
833    write(*,*) '**********************************************************'
834
835    call gsv_allocate( stateVectorBkGnd, 1, hco_ptr, vco_ptr, dateStamp_opt=-1, &
836                       mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
837                       varNames_opt=(/'DSLO'/), dataKind_opt=8 )
838    call gsv_allocate( stateVectorAnal,  1, hco_ptr, vco_ptr, dateStamp_opt=-1, &
839                       varNames_opt=(/'DSLO'/), dataKind_opt=8 )
840
841    etiket = '            '
842    typvar = 'P@'
843
844    call gio_readFromFile( stateVectorBkGnd, trlmFileName, etiket, typvar )
845
846    leadTimeInHours = real(stateVectorBkGnd%deet*stateVectorBkGnd%npasList(1),8)/3600.0d0
847    call incdatr(stateVectorAnal%dateOriginList(1), stateVectorBkGnd%dateOriginList(1), &
848                 leadTimeInHours)

860    nj = stateVectorBkGnd%hco%nj

873        end do
874      end do

877    HEADER_LOOP: do headerIndex = 1, obs_numHeader(obsSpaceData)
878
879      bodyIndexBeg = obs_headElem_i(obsSpaceData,OBS_RLN,headerIndex)
880      bodyIndexEnd = obs_headElem_i(obsSpaceData,OBS_NLV,headerIndex) + bodyIndexBeg - 1
881
882      BODY_LOOP: do bodyIndex = bodyIndexBeg, bodyIndexEnd
883
884        if ( obs_bodyElem_i(obsSpaceData,OBS_ASS,bodyIndex) /= obs_assimilated ) then
885          cycle BODY_LOOP
886        end if
887
888        do kIndex = stateVectorBkGnd%mykBeg, stateVectorBkGnd%mykEnd
889          do stepIndex = 1, stateVectorBkGnd%numStep
890            do procIndex = 1, mmpi_nprocs
891
892              call s2c_getWeightsAndGridPointIndexes(headerIndex, &
893                   kIndex, stepIndex, procIndex, interpWeight, obsLatIndex, &
894                   obsLonIndex, gridptCount)
895
896              GRIDPT_LOOP: do gridpt = 1, gridptCount
897
898                if ( interpWeight(gridpt) == 0.0d0 ) cycle GRIDPT_LOOP
899

901                latIndex = obsLatIndex(gridpt)

918                          containsFullField_opt=.true. )

921    call col_deallocate( column )

923    call gsv_deallocate( stateVectorAnal )

925  end subroutine aer_daysSinceLastObs
