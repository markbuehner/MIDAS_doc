 17module spectralFilter_mod
 18  ! MODULE spectralFilter_mod (prefix='spf' category='8. Low-level utilities and constants')
 19  !
 20  !:Purpose: For computing spectral filter functions
 21  !
 22  implicit none
 23  save
 24  private
 25
 26  ! public procedures
 27  public :: spf_FilterResponseFunction
 28
 29CONTAINS
 30
 31!--------------------------------------------------------------------------
 32! spf_FilterResponseFunction
 33!--------------------------------------------------------------------------
 34  function spf_FilterResponseFunction(totalWaveNumber, waveBandIndex, waveBandPeaks, nWaveBand ) result(ResponseFunction) 
 35    implicit none
 36
 37    real(8) :: totalWaveNumber
 38    real(8) :: ResponseFunction 
 39    integer :: waveBandIndex
 40    integer :: nWaveBand
 41    integer :: waveBandPeaks(:)
 42
 43    real(8) :: linearResponse, lowerLimit, center, upperLimit
 44    real(8), parameter :: pi = 2.d0*asin(1.d0)
 45
 46    if (waveBandIndex == nWaveBand ) then
 47       ! This wave band contains the largest scales.
 48       !
 49       ! The response function is 1 total wave number <= waveBandPeaks(nWaveBand)
 50       ! and decreases to 0 at waveBandPeaks(nWaveBand-1)
 51       !
 52       !                    response=1 |---
 53       !                               |   \
 54       !                               |    \
 55       !                    response=0 |------------
 56       !       waveBandPeaks(nWaveBand) <-|  |-> waveBandPeaks(nWaveBand-1)
 57       !
 58       lowerlimit = real(waveBandPeaks(waveBandIndex  ),8)
 59       upperlimit = real(waveBandPeaks(waveBandIndex-1),8)
 60
 61       if ( totalWaveNumber < lowerlimit ) then
 62          ResponseFunction = 1.d0
 63       else if ( totalWaveNumber <= upperlimit ) then
 64          linearResponse = (upperlimit-totalWaveNumber) / (upperlimit-lowerlimit)
 65          ResponseFunction = sin( (pi/2.d0) * linearResponse)**2
 66       else
 67          ResponseFunction = 0.d0
 68       end if
 69
 70    else if ( waveBandIndex /= 1 ) then
 71       ! This wave band contains intermediate scales (i.e., not the largest or the smallest).
 72       !
 73       ! The response function is 1 (only) for the total wave number = waveBandPeaks(waveBandIndex)
 74       ! and decreases to 0 at both waveBandPeaks(waveBandIndex+1) and waveBandPeaks(waveBandIndex-1)
 75       !
 76       !                    response=1 |      -
 77       !                               |     / \
 78       !                               |    /   \
 79       !                    response=0 |------------
 80       !  waveBandPeaks(waveBandIndex+1) <-|     |-> waveBandPeaks(waveBandIndex-1)
 81       !                                      |-> waveBandPeaks(waveBandIndex)
 82       !
 83       center     = real(waveBandPeaks(waveBandIndex  ),8)
 84       upperlimit = real(waveBandPeaks(waveBandIndex-1),8)
 85       lowerlimit = real(waveBandPeaks(waveBandIndex+1),8)
 86
 87       if (      totalWaveNumber >  lowerlimit .and. &
 88                 totalWaveNumber <= center     ) then
 89          linearResponse = (totalWaveNumber-lowerlimit) / (center-lowerlimit)
 90          ResponseFunction = sin( (pi/2.d0) * linearResponse)**2
 91       else if ( totalWaveNumber >  center      .and. &
 92                 totalWaveNumber <  upperlimit  ) then
 93          linearResponse = (upperlimit-totalWaveNumber) / (upperlimit-center)
 94          ResponseFunction = sin( (pi/2.d0) * linearResponse)**2
 95       else
 96          ResponseFunction = 0.d0
 97       end if
 98
 99    else
100       !
101       ! This wave band contains the smallest scales.
102       !
103       ! The response function is 1 total wave number >= waveBandPeaks(nWaveBand)
104       ! and decreases to 0 at waveBandPeaks(nWaveBand-1)
105       !
106       !                    response=1 |    ---
107       !                               |   /
108       !                               |  / 
109       !                    response=0 |------------
110       !waveBandPeaks(waveBandIndex+1) <-|  |-> waveBandPeaks(1)
111       !
112       upperlimit = real(waveBandPeaks(waveBandIndex  ),8)
113       lowerlimit = real(waveBandPeaks(waveBandIndex+1),8)
114
115       if      ( totalWaveNumber > upperlimit ) then
116          ResponseFunction = 1.d0
117       else if ( totalWaveNumber > lowerlimit ) then
118          linearResponse = (totalWaveNumber-lowerlimit) / (upperlimit-lowerlimit)
119          ResponseFunction = sin( (pi/2.d0) * linearResponse)**2
120       else
121          ResponseFunction = 0.d0
122       end if
123
124    end if
125
126  end function spf_FilterResponseFunction
127
128END MODULE SpectralFilter_mod
