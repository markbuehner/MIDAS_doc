 17program midas_var1D
 18  !
 19  !:Purpose: Main program for one dimensional variational minimization
 20  !
 21  !          ---
 22  !
 23  !:Algorithm: This program performs a similar data assimilation procedure as the var program,
 24  !            except without taking into account the horizontal or temporal dimensions.
 25  !            The assimilation is performed separately at each horizontal location and time where
 26  !            observations are present. The B matrix can be either an explicit representation of
 27  !            the covariances produced by the program extractBmatrixFor1Dvar or from an ensemble
 28  !            (controlled by the namelists). The resulting analysis and analysis increment at the
 29  !            observation locations/times are output in standard files on a "Y" grid.
 30  !
 31  !            --
 32  !
 33  !:File I/O: The required input files and produced output files are listed as follows.
 34  !
 35  !============================================== ==============================================================
 36  ! Input and Output Files (NWP application)        Description of file
 37  !============================================== ==============================================================
 38  ! ``flnml``                                      In - Main namelist file with parameters user may modify
 39  ! ``flnml_static``                               In - The "static" namelist that should not be modified
 40  ! ``trlm_$NN`` (e.g. ``trlm_01``)                In - Background state (a.k.a. trial) files for each timestep
 41  ! ``analysisgrid``                               In - File defining grid for computing the analysis increment
 42  ! ``Bmatrix_sea.bin``                            In - 1DVar Bmatrix file over sea (output from extractBmatrixFor1DVar) 
 43  ! ``Bmatrix_land.bin``                           In - 1DVar Bmatrix file over land (output from extractBmatrixFor1DVar)
 44  ! ``obsfiles_$FAM/obs$FAM_$NNNN_$NNNN``          In - Observation file for each "family" and MPI task
 45  ! ``obserr``                                     In - Observation error statistics
 46  ! ``rttov_h2o_limits.dat``                       In - minimum and maximum humidity profile to clip analysis
 47  ! ``pm1q``                                       In/Out - Preconditioning file (Hessian of the cost function)
 48  ! ``rebm_$MMMm`` (e.g. ``rebm_180m``)            Out - Analysis increment on Y grid
 49  ! ``anlm_$MMMm``                                 Out - Analysis on Y grid
 50  ! ``obsfiles_$FAM.updated/obs$FAM_$NNNN_$NNNN``  Out - Updated obs file for each "family" and MPI task 
 51  ! Remainder are files related to radiance obs:
 52  ! ``stats_tovs``                                 In - Observation error file for radiances
 53  ! ``stats_tovs_symmetricObsErr``                 In - user-defined symmetric TOVS errors for all sky
 54  ! ``Cmat_$PLATFORM_$SENSOR.dat``                 In - Inter-channel observation-error correlations
 55  ! ``rtcoef_$PLATFORM_$SENSOR.H5``                In - RTTOV coefficient file HDF-5 format 
 56  ! ``rtcoef_$PLATFORM_$SENSOR.dat``               In - RTTOV coefficient file ASCII format 
 57  ! ``ozoneclim98``                                In - ozone climatology standard file (Fortuin and Kelder)
 58  !============================================== ==============================================================
 59  !
 60  !           --
 61  !
 62  !:Synopsis: Below is a summary of the ``var1D`` program calling sequence:
 63  !
 64  !             - **Initial setups:**
 65  !
 66  !               - Initialize temporal grid
 67  !
 68  !               - Initialize observation file names and set date stamp
 69  !
 70  !               - Setup horizontal and vertical grid objects for "analysis
 71  !                 grid" from ``analysisgrid`` file
 72  !
 73  !               - Setup ``obsSpaceData`` object and read observations from
 74  !                 files: ``inn_setupObs``
 75  !
 76  !               - Setup ``columnData`` module
 77  !
 78  !               - Setup the observation error statistics in ``obsSpaceData``
 79  !                 object: ``oer_setObsErrors``
 80  !
 81  !               - Setup the gridStateVector module (initialize list of analyzed variables)
 82  !
 83  !               - Get horizontal and vertical grid descriptors from trial fields: ``inn_getHcoVcoFromTrlmFile``,
 84  !                 and allocate a gridStateVector objects 
 85  !
 86  !               - Read the trials: ``gio_readTrials``
 87  !
 88  !               - Setup the 1DVar B matrix: ``bmat1D_bsetup``
 89  !
 90  !               - Setup the ``gridVariableTransforms`` and ``minimization`` modules
 91  !
 92  !               - Horizontally interpolate high-resolution stateVectorUpdate to trial columns: ``inn_setupColumnsOnTrlLev``
 93  !
 94  !               - Interpolate trial columns to analysis levels and setup for linearized H: ``inn_setupColumnsOnAnlIncLev``
 95  !
 96  !               - Compute observation innovations and prepare obsSpaceData for minimization: ``inn_computeInnovation``
 97  !
 98  !             - **Minimization:**
 99  !
100  !               - Do the minimization ``min_minimize`` to obtain ``controlVectorIncr``
101  !
102  !               - Get 1DVar increment from ``controlVectorIncr`` to ``columnAnlInc``: ``bmat1D_get1DvarIncrement``
103  !               - Transfer increment to ``stateVectorIncr``: ``var1D_transferColumnToYGrid``
104  !
105  !               - Write increment to file: ``inc_writeIncrement``
106  !
107  !             - **Final step:**
108  !
109  !               - Release resources
110  !
111  !               - Write hessian if requested to: ``min_writeHessian``
112  !
113  !               - Write updated observation files: ``obsf_writeFiles``
114  !
115  !           --
116  !
117  !:Options: `List of namelist blocks <../namelists_in_each_program.html#var1d>`_
118  !          that can affect the ``var1D`` program.
119  !
120  !          - The B matrix used by the ``var1D`` program is controlled by the namelist block
121  !            ``&nambmat1D`` of module bmatrix1DVar_mod
122  !              * scaleFactorHI scaling factors for HI variances
123  !              * scaleFactorHIHumidity scaling factors for HI humidity variances
124  !              * scaleFactorENs scaling factors for Ens variances
125  !              * scaleFactorEnsHumidity scaling factors for Ens humidity variances
126  !              * nEns number of ensemble members for the ensemble part of  B matrix (negative means no ensemble)
127  !              * vLocalize vertical localization length scale for the ensemble part of B matrix
128  !              * includeAnlVar list of variable for the B matrix
129  !              * numIncludeAnlVar number of variables in  includeAnlVar (to be removed later)
130  !
131  !           --
132  !
133  use version_mod
134  use codePrecision_mod
135  use ramDisk_mod
136  use utilities_mod
137  use midasMpi_mod
138  use mathPhysConstants_mod
139  use horizontalCoord_mod
140  use verticalCoord_mod
141  use timeCoord_mod
142  use obsSpaceData_mod
143  use columnData_mod  
144  use gridStateVector_mod
145  use gridStateVectorFileIO_mod
146  use controlVector_mod
147  use obsFiles_mod
148  use minimization_mod
149  use innovation_mod
150  use obsErrors_mod
151  use gridVariableTransforms_mod
152  use increment_mod
153  use biasCorrectionSat_mod
154  use var1D_mod
155  use bMatrix1Dvar_mod
156 
157  implicit none
158
159  integer :: istamp, exdb, exfin
160  integer :: ierr, dateStampFromObs
161  integer :: get_max_rss
162  character(len=48) :: obsMpiStrategy, varMode
163  real(8), allocatable :: controlVectorIncr(:)
164  real(8), allocatable :: controlVectorIncrSum(:)
165  type(struct_obs),        target :: obsSpaceData
166  type(struct_columnData), target :: columnTrlOnAnlIncLev
167  type(struct_columnData), target :: columnTrlOnTrlLev
168  type(struct_columnData), target :: columnAnlInc
169  type(struct_gsv)                :: stateVectorIncr
170  type(struct_gsv)                :: stateVectorTrialHighRes
171  type(struct_gsv)                :: stateVectorAnalysis
172  type(struct_hco),       pointer :: hco_anl => null()
173  type(struct_vco),       pointer :: vco_anl => null()
174  type(struct_hco),       pointer :: hco_core => null()
175  type(struct_hco),       pointer :: hco_trl => null()
176  type(struct_vco),       pointer :: vco_trl => null()
177
178  integer :: outerLoopIndex, numIterMaxInnerLoop
179  logical :: allocHeightSfc
180
181  istamp = exdb('VAR1D', 'DEBUT', 'NON')
182
183  call ver_printNameAndVersion('var1D', '1D Variational Assimilation')
184
185  ! MPI initialization
186  call mmpi_initialize
187
188  call tmg_init(mmpi_myid, 'TMG_INFO')
189
190  call utl_tmg_start(0,'Main')
191
192  write(*,*)
193  write(*,*) 'Real Kind used for computing the increment =', pre_incrReal
194  write(*,*)
195
196  varMode='analysis'
197
198  ! Setup the ram disk
199  call ram_setup
200
201  ! Do initial set up
202
203  obsMpiStrategy = 'LIKESPLITFILES'
204
205  ! Initialize the Temporal grid and set dateStamp from env variable
206  call tim_setup
207
208  ! Initialize observation file names and set datestamp if not already
209  call obsf_setup(dateStampFromObs, varMode)
210  if (tim_getDateStamp() == 0) then
211    if (dateStampFromObs > 0) then
212      call tim_setDatestamp(dateStampFromObs)
213    else
214      call utl_abort('midas-var1D: DateStamp was not set')
215    end if
216  end if
217
218  ! Initialize constants
219  if (mmpi_myid == 0) call mpc_printConstants(6)
220
221  ! Initialize the Analysis grid
222  if (mmpi_myid == 0) write(*,*)
223  if (mmpi_myid == 0) write(*,*) 'midas-var1D: Set hco parameters for analysis grid'
224  call hco_SetupFromFile(hco_anl, './analysisgrid', 'ANALYSIS', 'Analysis' ) ! IN
225
226  if ( hco_anl % global ) then
227    hco_core => hco_anl
228  else
229    !- Initialize the core (Non-Extended) analysis grid
230    if (mmpi_myid == 0) write(*,*) 'midas-var1D: Set hco parameters for core grid'
231    call hco_SetupFromFile( hco_core, './analysisgrid', 'COREGRID', 'AnalysisCore' ) ! IN
232  end if
233
234  ! Initialisation of the analysis grid vertical coordinate from analysisgrid file
235  call vco_SetupFromFile( vco_anl,        & ! OUT
236                          './analysisgrid') ! IN
237
238  call col_setVco(columnTrlOnAnlIncLev, vco_anl)
239  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
240
241  ! Setup and read observations
242  call inn_setupObs(obsSpaceData, hco_anl, 'VAR', obsMpiStrategy, varMode) ! IN
243  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
244
245  ! Basic setup of columnData module
246  call col_setup
247  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
248
249  ! Memory allocation for background column data
250  call col_allocate(columnTrlOnAnlIncLev, obs_numheader(obsSpaceData), mpiLocal_opt=.true.)
251
252  ! Initialize the observation error covariances
253  call oer_setObsErrors(obsSpaceData, varMode) ! IN
254  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
255
256  ! Initialize list of analyzed variables.
257  call gsv_setup
258  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
259
260  ! Reading trials
261  call inn_getHcoVcoFromTrlmFile( hco_trl, vco_trl )
262  allocHeightSfc = ( vco_trl%Vcode /= 0 )
263
264  call gsv_allocate( stateVectorTrialHighRes, tim_nstepobs, hco_trl, vco_trl,  &
265                     dateStamp_opt=tim_getDateStamp(), mpi_local_opt=.true., &
266                     mpi_distribution_opt='Tiles', dataKind_opt=pre_incrReal,  &
267                     allocHeightSfc_opt=allocHeightSfc, hInterpolateDegree_opt='LINEAR', &
268                     beSilent_opt=.false. )
269  call gsv_zero( stateVectorTrialHighRes )
270  call gio_readTrials( stateVectorTrialHighRes )
271  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
272
273  ! Initialize the background-error covariance, also sets up control vector module (cvm)
274  call bmat1D_bsetup(vco_anl, hco_anl, obsSpaceData)
275  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
276
277  ! Initialize the gridded variable transform module
278  call gvt_setup(hco_anl, hco_core, vco_anl)
279
280  ! Set up the minimization module, now that the required parameters are known
281  ! NOTE: some global variables remain in minimization_mod that must be initialized before
282  !       inn_setupBackgroundColumns
283  call min_setup( cvm_nvadim, hco_anl, oneDVarMode_opt=.true. ) ! IN
284  allocate(controlVectorIncr(cvm_nvadim),stat=ierr)
285  if (ierr /= 0) then
286    write(*,*) 'midas-var1D: Problem allocating memory for ''controlVectorIncr''',ierr
287    call utl_abort('aborting in VAR1D')
288  end if
289  call utl_reallocate(controlVectorIncrSum,cvm_nvadim)
290  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
291
292  ! Horizontally interpolate high-resolution stateVectorUpdate to trial columns
293  call inn_setupColumnsOnTrlLev( columnTrlOnTrlLev, obsSpaceData, hco_core,  &
294                                   stateVectorTrialHighRes )
295
296  ! Interpolate trial columns to analysis levels and setup for linearized H
297  call inn_setupColumnsOnAnlIncLev( columnTrlOnTrlLev,columnTrlOnAnlIncLev )
298
299  ! Compute observation innovations and prepare obsSpaceData for minimization
300  call inn_computeInnovation(columnTrlOnTrlLev, obsSpaceData)
301
302  ! Do minimization of cost function
303  outerLoopIndex = 1
304  numIterMaxInnerLoop = 0
305  controlVectorIncr(:) = 0.0d0
306  call min_minimize( outerLoopIndex, columnTrlOnAnlIncLev, obsSpaceData, controlVectorIncrSum, &
307                     controlVectorIncr, numIterMaxInnerLoop )
308  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
309
310  ! Compute satellite bias correction increment and write to file
311  ! Is is still necessary ? (will do nothing, but does it make sense in 1DVar mode ?)
312  call bcs_writebias(controlVectorIncr)
313
314  call col_setVco(columnAnlInc, col_getVco(columnTrlOnAnlIncLev))
315  call col_allocate(columnAnlInc,col_getNumCol(columnTrlOnAnlIncLev),mpiLocal_opt=.true.)
316  call col_zero(columnAnlInc)
317  ! get final increment
318  call bmat1D_get1DvarIncrement(controlVectorIncr,columnAnlInc,columnTrlOnAnlIncLev,obsSpaceData,cvm_nvadim)
319  call var1D_transferColumnToYGrid( stateVectorIncr, obsSpaceData, columnAnlInc, bmat1D_includeAnlVar)
320
321  ! output the analysis increment
322  call inc_writeIncrement( stateVectorIncr)
323  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
324
325  ! compute and write the analysis (as well as the increment on the trial grid)
326  call var1d_transferColumnToYGrid(stateVectorAnalysis, obsSpaceData, columnTrlOnAnlIncLev, bmat1D_includeAnlVar)
327
328  if (mmpi_myId == 0) call gsv_add(statevectorIncr, statevectorAnalysis)
329
330  call inc_writeAnalysis(stateVectorAnalysis)
331
332  ! Deallocate memory related to B matrices
333  call var1D_finalize()
334  if (mmpi_myId == 0) then
335    call gsv_deallocate(stateVectorIncr)
336    call gsv_deallocate(stateVectorAnalysis)
337  end if
338
339  ! write the Hessian
340  call min_writeHessian(controlVectorIncr)
341  deallocate(controlVectorIncrSum)
342  deallocate(controlVectorIncr)
343
344  ! Deallocate memory related to variational bias correction
345  call bcs_finalize()
346  ! Now write out the observation data files
347  if (min_niter > 0) then
348    if ( .not. obsf_filesSplit() ) then 
349      write(*,*) 'We read/write global observation files'
350      call obs_expandToMpiGlobal(obsSpaceData)
351      if (mmpi_myid == 0) call obsf_writeFiles(obsSpaceData)
352    else
353      ! redistribute obs data to how it was just after reading the files
354      call obs_MpiRedistribute(obsSpaceData, OBS_IPF)
355      call obsf_writeFiles(obsSpaceData)
356    end if
357  end if
358
359  ! Deallocate copied obsSpaceData
360  call obs_finalize(obsSpaceData)
361
362  ! Job termination
363  istamp = exfin('VAR1D','FIN','NON')
364
365  call utl_tmg_stop(0)
366
367  call tmg_terminate(mmpi_myid, 'TMG_INFO')
368
369  call rpn_comm_finalize(ierr) 
370
371end program midas_var1D
