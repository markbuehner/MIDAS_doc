 17program midas_thinning
 18  !
 19  !:Purpose: The thinning program reduces the density of observations for the purpose of
 20  !          assimilation.
 21  !
 22  !
 23  !:Algorithm: After setting up the ``obsSpace_data`` object, the thinning program calls
 24  !            the ``filt_suprep`` routine which rejects certain observations based on
 25  !            blacklists and other checks. 
 26  !
 27  !            -- 
 28  !
 29  !            Specific routines are then called for thinning each observation types. 
 30  !            These routines are found in ``thinning_mod`` and are controlled by the 
 31  !            following namelists:
 32  !
 33  !            |
 34  !
 35  !            ======================= ====== 
 36  !             Namelist                    
 37  !            ======================= ====== 
 38  !             ``thn_surface``
 39  !             ``thn_raobs``
 40  !             ``thn_aircraft``
 41  !             ``thn_satwind``
 42  !             ``thn_gpsro``
 43  !             ``thn_gbgps``
 44  !             ``thn_aladin``
 45  !             ``thn_csr``
 46  !             ``thn_scat``
 47  !             ``thn_tovs``
 48  !             ``thn_hyper``
 49  !             ``thn_thinSatSST``
 50  !            ======================= ====== 
 51  !
 52  !            |
 53  !
 54  !            Observations that are rejected by the thinning routines have their 11th bit *flag* set. 
 55  !            In a subsequent step, these observations are removed from observation files. 
 56  !
 57  !            |
 58  !
 59  !
 60  !:File I/O: 
 61  !
 62  !           ============================================== ================================================================
 63  !            Input and Output Files (NWP applicaton)        Description of file
 64  !           ============================================== ================================================================
 65  !            ``flnml``                                      In - Main namelist file with parameters user may modify
 66  !            ``trlm_$NN`` (e.g. ``trlm_01``)                In - Background state -> necessary for observations in pressure
 67  !                                                           coordinates. 
 68  !            Observation Files                              In/Out - Sqlite/Burp observation files for different families. 
 69  !           ============================================== ================================================================
 70  !
 71  !
 72  use version_mod
 73  use ramDisk_mod
 74  use utilities_mod
 75  use midasMpi_mod
 76  use timeCoord_mod
 77  use obsSpaceData_mod
 78  use obsFiles_mod
 79  use tovs_nl_mod
 80  use obsFilter_mod
 81  use thinning_mod
 82  use fSQLite
 83
 84  implicit none
 85
 86  type(struct_obs) :: obsSpaceData
 87  integer :: istamp,exdb,exfin
 88  integer :: ierr, dateStampFromObs
 89  integer :: get_max_rss
 90
 91  istamp = exdb('THINNING', 'DEBUT', 'NON')
 92
 93  call ver_printNameAndVersion('thinning', 'Observation thinning')
 94
 95  ! MPI initilization
 96  call mmpi_initialize
 97
 98  call tmg_init(mmpi_myid, 'TMG_INFO')
 99  call utl_tmg_start(0, 'Main')
100
101  ! 1. Top level setup
102
103  call ram_setup
104
105  ! 2. configuration the job
106
107  call utl_tmg_start(10,'--Observations')
108
109  !- Initialize observation file names
110  call obsf_setup(dateStampFromObs, 'thinning')
111
112  !- Allocate obsSpaceData
113  call obs_class_initialize('ALL')
114  call obs_initialize(obsSpaceData, mpi_local = .true.)
115
116  !- Setup obsFilter_mod
117  call filt_setup('bgck')
118
119  !- Read observations
120  call utl_tmg_start(11,'----ReadObsFiles')
121  call obsf_readFiles(obsSpaceData)
122  call utl_tmg_stop(11)
123
124  write(*,*) 'Memory Used: ', get_max_rss() / 1024,'Mb'
125
126  !- Initialize TOVS processing
127  if (obs_famExist(obsSpaceData, 'TO')) call tvs_setup
128
129  !- Select the elements to "assimilate" and apply rejection flags
130  call filt_suprep(obsSpaceData)
131
132  call utl_tmg_stop(10)
133
134  !- Setup timeCoord module and set dateStamp from env variable
135  call tim_setup()
136  if (tim_getDateStamp() == 0) then
137    if (dateStampFromObs > 0) then
138      ! use dateStamp from obs if not set by env variable
139      call tim_setDateStamp(dateStampFromObs)
140    else
141      call utl_abort('midas-thinning: DateStamp was not set')
142    end if
143  end if
144
145  ! 3. Do the Thinning
146
147  !- Set bit 11 of flag, process one observation type at a time
148  call thn_thinHyper(obsSpaceData)
149  call thn_thinTovs(obsSpaceData)
150  call thn_thinCSR(obsSpaceData)
151  call thn_thinScat(obsSpaceData)
152  call thn_thinSatWinds(obsSpaceData)
153  call thn_thinAircraft(obsSpaceData)
154  call thn_thinSurface(obsSpaceData, 'SF') ! surface data thinning    
155  if (obs_famExist(obsSpaceData, 'TM')) then
156    call thn_thinSurface(obsSpaceData, 'TM') ! insitu SST thinning
157    call thn_thinSatSST(obsSpaceData)        ! satellite SST thinning
158  end if
159  call thn_thinGbGps(obsSpaceData)
160  call thn_thinGpsRo(obsSpaceData)
161  call thn_thinAladin(obsSpaceData)
162  write(*,*) 'Memory Used: ',get_max_rss() / 1024,'Mb'
163
164  if (obs_famExist( obsSpaceData, 'UA' )) then
165    write(*,*) 'midas-thinning: WARNING: radiosonde observations found in obsSpaceData!'
166    write(*,*) '                These observations cannot be thinned using the stand-alone'
167    write(*,*) '                MIDAS thinning program. Instead they should be thinned'
168    write(*,*) '                in combination with doing the background check in the'
169    write(*,*) '                obsSelection program.'
170  end if
171
172  !- Update obs files and remove rejected obs (bit 11) from file (obsFileClean)
173  call obsf_writeFiles(obsSpaceData)
174  call obsf_cleanObsFiles()
175  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
176
177  ! 4. Job termination
178
179  istamp = exfin('THINNING', 'FIN', 'NON')
180
181  !- deallocate obsSpaceData
182  call obs_finalize(obsSpaceData)
183
184  call utl_tmg_stop(0)
185  call tmg_terminate(mmpi_myid, 'TMG_INFO')
186
187  call rpn_comm_finalize(ierr)
188
189end program midas_thinning
