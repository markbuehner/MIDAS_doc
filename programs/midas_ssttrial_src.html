 17program midas_sstTrial
 18  !
 19  !:Purpose: Main program to compute SST background state.
 20  !
 21  !          ---
 22  !
 23  !:Algorithm: The background state of SST is computed from the previous SST analysis field
 24  !            using a relaxation towards climatology as follows:
 25  !            :math:`X_{b}(t) = (X_{a}(t-1) - X_{clim}(t-1)) * \alpha + X_{clim}(t)`,
 26  !            where :math:`X_{b}(t)` is a resulting background state at time :math:`t`,
 27  !            :math:`X_{a}(t-1)` is analysis state at time :math:`t-1`,
 28  !            :math:`\alpha` is a relaxation coefficient, 
 29  !            :math:`X_{clim}(t)` is a climatology state at time :math:`t`,
 30  !            computed as an interpolation in time 
 31  !            between climatological field of the current and the following month
 32  !            for the current day of the month. 
 33  !           
 34  !            --
 35  ! 
 36  !=========================================================== ======================================================
 37  ! Input and Output Files                                     Description of file
 38  !=========================================================== ======================================================
 39  ! ``analysis``                                               In - SST analysis field
 40  ! ``climatology``                                            In - 12 monthly climatological SST fields 
 41  ! ``trial``                                                  Out - SST background (trial) field
 42  !=========================================================== ======================================================
 43  !
 44  !            --
 45  !
 46  !:Synopsis: Below is a summary of the ``SSTtrial`` program calling sequence:
 47  !
 48  !            - **Initial setups:**
 49  !
 50  !             - Setup horizontal and vertical grid objects for "analysis
 51  !               grid" from ``analysis``.
 52  !
 53  !             - Setup ``gridStateVector`` modules.
 54  !
 55  !             - Initialize a time objects ``tim_setup``
 56  !
 57  !             - Initialize datastamps from climatology file
 58  !
 59  !            - **Computation**
 60  !
 61  !             - ``obgd_getClimatology`` to read SST climatological fields from a standard file,
 62  !               to interpolate the field in time fot the current day :math:`t` in current month :math:`m` as follows   
 63  !               :math:`X_{clim}(t) = X_{clim}(m) + (t - 1) /(N - 1) * (X_{clim}(m+1) - X_{clim}(m))`,
 64  !               where :math:`N` is a number of days in the current month         
 65  !
 66  !             - ``obgd_computeSSTrial`` to compute the background field and save it into a standard file
 67  !
 68  !            --
 69  !
 70  !:Options: `List of namelist blocks <../namelists_in_each_program.html#ssttrial>`_
 71  !          that can affect the ``SSTtrial`` program.
 72  !
 73  !            * The use of ``SSTtrial`` program is controlled by the namelist block
 74  !              ``&NAMSSTTRIAL`` read by the ``SSTtrial`` program.
 75  !                * ``etiketAnalysis`` etiket to put into output standard file
 76  !
 77  !                * ``datestampClim`` list of twelve datestamps of climatological fields
 78  !                  in the ``climatology`` file
 79  !
 80  !                * ``alphaClim`` a parameter defining the relaxation towards climatology.
 81  !
 82  !            --
 83  !
 84  use version_mod
 85  use ramDisk_mod
 86  use utilities_mod
 87  use midasMpi_mod
 88  use horizontalCoord_mod
 89  use verticalCoord_mod
 90  use timeCoord_mod
 91  use gridStateVector_mod
 92  use analysisGrid_mod
 93  use oceanBackground_mod
 94  
 95  implicit none
 96
 97  integer, external :: exdb, exfin, fnom, fclos, get_max_rss
 98  integer :: ierr, istamp, nulnam
 99
100  type(struct_hco), pointer :: hco_anl => null()
101  type(struct_vco), pointer :: vco_anl => null()
102  integer                   :: trialDateStamp    ! output: datestamp for the trial field
103  integer                   :: analysisDateStamp !  input: datastamp of the latest analysis
104  integer, parameter        :: nmonthsClim = 12  ! number of climatological fields
105
106  ! namelist variables
107  character(len=10) :: etiketAnalysis    ! etiket in the analysis file for grid setup
108  integer           :: datestampClim(12) ! datestamps of input climatology fields 
109  real(4)           :: alphaClim         ! scaling factor to relax towards climatology
110  
111  istamp = exdb('SSTTRIAL','DEBUT','NON')
112
113  call ver_printNameAndVersion('SSTtrial','SST trial preparation')
114
115  ! MPI initialization
116  call mmpi_initialize
117
118  call tmg_init(mmpi_myid, 'TMG_INFO')
119  call utl_tmg_start(0,'Main')
120  
121  ! 1. Top level setup
122
123  call ram_setup()
124 
125  ! Do initial set up
126  call SSTtrial_setup(trialDateStamp, analysisDateStamp)
127  
128  call obgd_computeSSTrial(hco_anl, vco_anl, trialDateStamp, analysisDateStamp, &
129                           nmonthsClim, datestampClim, alphaClim, etiketAnalysis)
130			 
131  ! 3. Job termination
132
133  istamp = exfin('SSTTRIAL','FIN','NON')
134  call utl_tmg_stop(0)
135  call tmg_terminate(mmpi_myid, 'TMG_INFO')
136  call rpn_comm_finalize(ierr) 
137
138  contains
139  
140  !----------------------------------------------------------------------------------------
141  ! SSTtrial_setup
142  !----------------------------------------------------------------------------------------
143  subroutine SSTtrial_setup(trialDateStamp, analysisDateStamp)
144    !
145    ! :Purpose:  Control of the preprocessing of trial
146    !
147
148    implicit none
149    
150    !Arguments:
151    integer, intent(out) :: trialDateStamp
152    integer, intent(out) :: analysisDateStamp
153    
154    !Locals:	
155    type(struct_hco), pointer   :: hco_core => null()
156    character(len=*), parameter :: gridFile = './analysis'
157    integer                     :: prntdate, prnttime, imode, newdate, indexMonth
158    namelist /namSSTtrial/ etiketAnalysis, datestampClim, alphaClim
159        
160    write(*,*) ''
161    write(*,*) '-------------------------------------------------'
162    write(*,*) '-- Starting subroutine SSTtrial_setup --'
163    write(*,*) '-------------------------------------------------'
164 
165    ! namelist variables default values
166    etiketAnalysis = ''
167    datestampClim(:) = 0
168    alphaClim = 0.983
169    
170    ! Read the namelist
171    nulnam = 0
172    ierr = fnom( nulnam, './flnml', 'FTN+SEQ+R/O', 0 )
173    read( nulnam, nml = namSSTtrial, iostat = ierr )
174    if ( ierr /= 0) call utl_abort( 'SSTtrial_setup: Error reading namelist')
175    if ( mmpi_myid == 0 ) write(*, nml = namSSTtrial )
176    ierr = fclos(nulnam)
177
178    if(mmpi_myid == 0) then
179      write(*,'(1X,"***********************************")')
180      write(*,'(1X," datestamps for climatology file:",/)')
181      write(*,'(1X,"***********************************")')
182      do indexMonth =1, nmonthsClim
183        write(*,*) indexMonth, datestampClim(indexMonth)
184      end do
185      write(*,'(1X,"***********************************")')
186    end if  
187
188    !
189    !- Initialize the Temporal grid and set dateStamp from gridFile
190    !
191    call tim_setup(fileNameForDate_opt = gridFile)
192    analysisDateStamp = tim_getDateStamp()
193    write(*,*) 'SSTtrial_setup: analysis datestamp  = ', analysisDateStamp
194    write(*,*) 'SSTtrial_setup:          windowsize = ', tim_windowsize
195    
196    call incdatr(trialDateStamp, analysisDateStamp, tim_windowsize)
197    write(*,*) 'SSTtrial_setup:    trial datestamp  = ', trialDateStamp
198    
199    imode = -3 ! stamp to printable
200    ierr = newdate(trialDateStamp, prntdate, prnttime, imode)
201    write(*,*) 'SSTtrial_setup: trial date = ', prntdate
202    write(*,*) 'SSTtrial_setup: trial time = ', prnttime
203    !
204    !- Initialize variables of the model states
205    !
206    call gsv_setup
207    write(*,*) 'SSTtrial_setup: memory Used: ',get_max_rss()/1024,'Mb'
208
209    !
210    !- Initialize the Analysis grid
211    !
212    if(mmpi_myid == 0) write(*,*)''
213    if(mmpi_myid == 0) write(*,*) 'SSTtrial_setup: Set hco parameters for analysis grid'
214    call hco_SetupFromFile(hco_anl, gridFile, trim(etiketAnalysis)) ! IN
215
216    if ( hco_anl % global ) then
217      call agd_SetupFromHCO( hco_anl ) ! IN
218    else
219      !- Initialize the core (Non-Extended) analysis grid
220      if(mmpi_myid == 0) write(*,*) 'SSTtrial_setup: Set hco parameters for core grid'
221      call hco_SetupFromFile( hco_core, gridFile, 'COREGRID', 'AnalysisCore' ) ! IN
222      !- Setup the LAM analysis grid metrics
223      call agd_SetupFromHCO( hco_anl, hco_core ) ! IN
224    end if
225
226    !     
227    !- Initialisation of the analysis grid vertical coordinate from analysisgrid file
228    !
229    call vco_SetupFromFile( vco_anl, & ! OUT
230                            gridFile ) ! IN
231
232    if(mmpi_myid == 0) write(*,*) 'SSTtrial_setup: done.'
233
234  end subroutine SSTtrial_setup
235
236end program midas_sstTrial
