 17program midas_ensembleH
 18  !
 19  !:Purpose: Main program for applying the observation operator to an ensemble
 20  !          of states to compute the innovations and store them in eob*_HX binary files. The
 21  !          ``LETKF`` program can read these binary files instead of computing the innovations.
 22  !
 23  !          ---
 24  !
 25  !:Algorithm: The background (a.k.a trial) ensemble members are read and the non-linear 
 26  !            observation operators are applied to each ensemble member:
 27  !            :math:`H(xb_{i})`,
 28  !            the innovations are computed: 
 29  !            :math:`y-H(xb_{i})`,
 30  !            and stored in unformatted binary files. ``ensembleH`` has the ability 
 31  !            to compute innovations for a batch of ensemble members when the member 
 32  !            indices are sequential. 
 33  !
 34  !            --
 35  !
 36  !============================================== ==============================================================
 37  ! Input and Output Files                        Description of file
 38  !============================================== ==============================================================
 39  ! ``flnml``                                      In - Main namelist file with parameters user may modify
 40  ! ``flnml_static``                               In - The "static" namelist that should not be modified
 41  ! ``trlm_$NN`` (e.g. ``trlm_01``)                In - Background state (a.k.a. trial) files for each timestep
 42  ! ``ensemble/$YYYYMMDDHH_006_$NNNN``             In - Background ensemble member files
 43  ! ``obsfiles_$FAM/obs$FAM_$NNNN_$NNNN``          In - Observation file for each "family" and MPI task
 44  ! ``obserr``                                     In - Observation error statistics
 45  ! ``$YYYYMMDDHH_000_$NNNN``                      Out - Analysis ensemble member files
 46  ! ``obsfiles_$FAM.updated/obs$FAM_$NNNN_$NNNN``  Out - Updated obs file for each "family" and MPI task
 47  ! Remainder are files related to radiance obs:
 48  ! ``stats_$SENSOR_assim``                        In - Satellite radiance observation errors of difference sensors
 49  ! ``stats_tovs``                                 In - Satellite radiance observation errors
 50  ! ``stats_tovs_symmetricObsErr``                 In - User-defined symmetric TOVS errors for all sky
 51  ! ``Cmat_$PLATFORM_$SENSOR.dat``                 In - Inter-channel observation-error correlations
 52  ! ``ceres_global.std``                           In - High-res surface type and water fraction for radiance obs
 53  ! ``rtcoef_$PLATFORM_$SENSOR.dat``               In - RTTOV coefficient files
 54  ! ``rttov_h2o_limits.dat``                       In - Min/max humidity limits applied to analysis
 55  ! ``ozoneclim98``                                In - Ozone climatology
 56  !============================================== ==============================================================
 57  !
 58  !           --
 59  !
 60  !:Synopsis: Below is a summary of the ``ensembleH`` program calling sequence:
 61  !
 62  !             - **Initial setups:**
 63  !
 64  !               - Read the NAMENSEMBLEH namelist and check/modify some values.
 65  !
 66  !               - Various modules are setup: ``obsFiles_mod``,
 67  !                 ``gridStateVector_mod``, ``timeCoord_mod`` (and set up dates
 68  !                 and ``dateStampList`` variables for both trials and
 69  !                 increments/analyses).
 70  !
 71  !               - Setup horizontal and vertical grid objects from first
 72  !                 ensemble member file.
 73  !
 74  !               - Setup ``obsSpaceData`` object and read observations from
 75  !                 files: ``inn_setupObs``.
 76  !
 77  !               - Setup the observation error statistics in ``obsSpaceData``
 78  !                 object: ``oer_setObsErrors``.
 79  !
 80  !               - Allocate objects for ``column_mod``.
 81  !
 82  !               - Allocate and some setup of objects for
 83  !                 ``ensembleObservations_mod``.
 84  !
 85  !               - Allocate some objects for ``gridStateVector_mod``.
 86  !
 87  !               - Allocate ensemble object and read background ensemble members:
 88  !                 ``ens_readEnsemble``.
 89  !
 90  !             - **Computation**
 91  !
 92  !               - Option to read ensemble mean from file (``gio_readFromFile``) or 
 93  !                 compute ensemble mean (``ens_computeMean``)
 94  !
 95  !               - Loop over background ensemble members, computing innovation for each, 
 96  !                 with resulting ``H(xb)`` being stored in ``ensObs`` objects both for
 97  !                 original ensemble and, optionally, for the modulated
 98  !                 ensemble members.
 99  !
100  !               - store the local ``ensObs`` object to binary file.
101  !
102  !             --
103  !  
104  !
105  !:Options: `List of namelist blocks <../namelists_in_each_program.html#ensembleH>`_
106  !          that can affect the ``ensembleH`` program.
107  !
108  !          * The use of ``ensembleH`` program is controlled by the namelist block
109  !           ``&NAMENSEMBLEH`` read by the ``ensembleH`` program.
110  !
111  !          * Some of the other relevant namelist blocks used to configure the
112  !            ``ensembleH`` are listed in the following table:
113  ! 
114  !===================== ================== ===============================================================
115  ! Program/Module        Namelist           Description of what is controlled
116  !===================== ================== ===============================================================
117  ! ``midas_ensembleh``   ``NAMENSEMBLEH``   number of ensemble members, additional parameters to control
118  !                                          generating modulated members from original ensembles, 
119  !                                          the member number the current batch starts with, number of 
120  !                                          members in the batch, option to read ensemble mean from file.
121  ! ``timeCoord_mod``     ``NAMTIME``        assimilation time window length, temporal resolution of
122  !                                          the background state.
123  !===================== ================== ===============================================================
124  !
125  use version_mod
126  use midasMpi_mod
127  use fileNames_mod
128  use gridStateVector_mod
129  use gridStateVectorFileIO_mod
130  use columnData_mod
131  use verticalCoord_mod
132  use horizontalCoord_mod
133  use timeCoord_mod
134  use utilities_mod
135  use ramDisk_mod
136  use statetocolumn_mod
137  use obsFiles_mod
138  use obsSpaceData_mod
139  use obsErrors_mod
140  use innovation_mod
141  use ensembleObservations_mod
142  use ensembleStateVector_mod
143  use enkf_mod
144  implicit none
145
146  type(struct_obs), target  :: obsSpaceData
147  type(struct_ens)          :: ensembleTrl4D
148  type(struct_gsv)          :: stateVectorMeanTrl4D
149  type(struct_gsv)          :: stateVector4D
150  type(struct_gsv)          :: stateVector4Dmod
151  type(struct_gsv)          :: stateVectorWithZandP4D
152  type(struct_gsv)          :: stateVectorHeightSfc
153  type(struct_columnData)   :: column
154
155  type(struct_eob), target  :: ensObs
156  type(struct_eob), pointer :: ensObsGain
157
158  type(struct_vco), pointer :: vco_ens => null()
159  type(struct_hco), pointer :: hco_ens => null()
160
161  integer :: get_max_rss, fclos, fnom, fstopc, ierr
162  integer :: memberIndex, nulnam, dateStampFromObs
163  integer :: nEnsGain, eigenVectorIndex, memberIndexInEnsObs, stepIndex
164  integer, allocatable :: dateStampList(:)
165
166  logical :: useModulatedEns, fileExists
167
168  character(len=256)  :: ensFileName
169  character(len=256)  :: ensMeanFileName
170  character(len=9)    :: obsColumnMode
171  character(len=48)   :: obsMpiStrategy
172  character(len=48)   :: midasMode
173
174  ! namelist variables
175  character(len=256) :: ensPathName       ! path of ensemble member files
176  character(len=20)  :: obsTimeInterpType ! type of time interpolation to obs time
177  integer  :: nEns             ! ensemble size
178  integer  :: numRetainedEigen ! number of retained eigen modes used for modulated ensemble
179  integer  :: fileMemberIndex1 ! first member index in ensemble set to be read
180  integer  :: numFullEns       ! number of full ensemble set (needed only for modulated ensemble)
181  real(8)  :: vLocalize        ! vertical localization lengthscale (needed only for modulated ensemble)
182  logical  :: readEnsMeanFromFile ! choose to read ens mean from file (when reading subset of members)
183  NAMELIST /NAMENSEMBLEH/nEns, ensPathName, obsTimeInterpType, numRetainedEigen, &
184                         vLocalize, fileMemberIndex1, readEnsMeanFromFile, numFullEns
185
186  midasMode = 'analysis'
187  obsColumnMode = 'ENKFMIDAS'
188  obsMpiStrategy = 'LIKESPLITFILES'
189
190  call ver_printNameAndVersion('ensembleH','Program for applying H to ensemble')
191
192  !
193  !- 0. MPI, TMG initialization
194  !
195  call mmpi_initialize
196  call tmg_init(mmpi_myid, 'TMG_INFO')
197
198  call utl_tmg_start(0,'Main')
199  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
200
201  ! Avoid printing lots of stuff to listing for std file I/O
202  ierr = fstopc('MSGLVL','ERRORS',0)
203
204  ! Setup the ramdisk directory (if supplied)
205  call ram_setup
206
207  ! Setting default namelist variable values
208  nEns                   = 10
209  ensPathName            = 'ensemble'
210  obsTimeInterpType      = 'LINEAR'
211  numRetainedEigen       = 0
212  vLocalize              = -1.0D0
213  fileMemberIndex1       = 1
214  readEnsMeanFromFile    = .false.
215  numFullEns             = 0
216
217  ! Read the namelist
218  nulnam = 0
219  ierr = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
220  read(nulnam, nml=namensembleh, iostat=ierr)
221  if (ierr /= 0) call utl_abort('midas-ensembleH: Error reading namelist')
222  if (mmpi_myid == 0) write(*,nml=namensembleh)
223  ierr = fclos(nulnam)
224  
225  if (numRetainedEigen < 0) call utl_abort('midas-ensembleH: numRetainedEigen should be ' // &
226                                             'equal or greater than zero')
227
228  useModulatedEns = (numRetainedEigen > 0)
229  if (useModulatedEns .and. vLocalize <= 0) then
230    call utl_abort('midas-ensembleH: vLocalize should be greater than zero for modulated ens')
231  end if
232
233  if (useModulatedEns .and. numFullEns < nEns) then
234    call utl_abort('midas-ensembleH: For modulated ensembles the number of full ensembles is needed ' // &
235                      'with numFullEns >= nEns')
236  end if
237
238  ! Read the observations
239  call obsf_setup(dateStampFromObs, midasMode)
240
241  ! Use the first ensemble member to initialize datestamp and grid
242  call fln_ensFileName(ensFileName, ensPathName, memberIndex_opt=1, &
243                       fileMemberIndex1_opt=fileMemberIndex1)
244
245  ! Setup timeCoord module, get datestamp from ensemble member
246  call tim_setup(fileNameForDate_opt = ensFileName)
247  allocate(dateStampList(tim_nstepobs))
248  call tim_getstamplist(dateStampList,tim_nstepobs,tim_getDatestamp())
249  
250  write(*,*) 'midas-ensembleH: dateStamp = ',tim_getDateStamp()
251
252  !- Initialize variables of the model states
253  call gsv_setup
254
255  !- Initialize the Ensemble grid
256  if (mmpi_myid == 0) write(*,*) ''
257  if (mmpi_myid == 0) write(*,*) 'midas-ensembleH: Set hco and vco parameters for ensemble grid'
258  call hco_SetupFromFile(hco_ens, ensFileName, ' ', 'ENSFILEGRID')
259  call vco_setupFromFile(vco_ens, ensFileName)
260
261  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
262
263  ! read in the observations
264  call inn_setupObs(obsSpaceData, hco_ens, obsColumnMode, obsMpiStrategy, midasMode)
265
266  ! Initialize obs error covariances
267  call oer_setObsErrors(obsSpaceData, midasMode)
268
269  call col_setup
270  call col_setVco(column, vco_ens)
271  call col_allocate(column, obs_numheader(obsSpaceData))
272  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
273
274  ! Initialize the observation error covariances
275  call oer_setObsErrors(obsSpaceData, midasMode) ! IN
276
277  ! Allocate and initialize eob object for storing HX values
278  call eob_allocate(ensObs, nEns, obs_numBody(obsSpaceData), obsSpaceData, &
279                    fileMemberIndex1_opt=fileMemberIndex1)
280  call eob_zero(ensObs)
281  if (useModulatedEns) then
282    nEnsGain = nEns * numRetainedEigen
283    allocate(ensObsGain)
284    call eob_allocate(ensObsGain, nEnsGain, obs_numBody(obsSpaceData), obsSpaceData, &
285                      fileMemberIndex1_opt=fileMemberIndex1)
286    call eob_zero(ensObsGain)
287  else
288    ensObsGain => ensObs
289  end if
290  
291  ! Set lat, lon, obs values in ensObs
292  call eob_setLatLonObs(ensObs)
293  if (useModulatedEns) call eob_setLatLonObs(ensObsGain)
294
295
296  ! Read the sfc height from ensemble member 1
297  call gsv_allocate(stateVectorHeightSfc, 1, hco_ens, vco_ens, dateStamp_opt=tim_getDateStamp(),  &
298                    mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
299                    dataKind_opt=4, allocHeightSfc_opt=.true., varNames_opt=(/'P0','TT'/))
300  call gio_readFromFile(stateVectorHeightSfc, ensFileName, ' ', ' ',  &
301                        containsFullField_opt=.true., readHeightSfc_opt=.true.)
302
303  ! Allocate statevector related to ensemble mean
304  call gsv_allocate(stateVectorMeanTrl4D, tim_nstepobs, hco_ens, vco_ens, &
305                    dateStamp_opt=tim_getDateStamp(),  &
306                    mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
307                    dataKind_opt=4, allocHeightSfc_opt=.true., &
308                    allocHeight_opt=.false., allocPressure_opt=.false.)
309  call gsv_zero(stateVectorMeanTrl4D)
310  call gsv_allocate(stateVector4D, tim_nstepobs, hco_ens, vco_ens, &
311                    dateStamp_opt=tim_getDateStamp(),  &
312                    mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
313                    dataKind_opt=4, allocHeightSfc_opt=.true., &
314                    allocHeight_opt=.false., allocPressure_opt=.false.)
315  call gsv_zero(stateVector4D)
316  if (useModulatedEns) then
317    ! same as stateVector4D
318    call gsv_allocate(stateVector4Dmod, tim_nstepobs, hco_ens, vco_ens, &
319                      dateStamp_opt=tim_getDateStamp(),  &
320                      mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
321                      dataKind_opt=4, allocHeightSfc_opt=.true., &
322                      allocHeight_opt=.false., allocPressure_opt=.false.)
323    call gsv_zero(stateVector4Dmod)
324  end if
325  
326  ! Allocate statevector for storing state with heights and pressures allocated (for s2c_nl)
327  call gsv_allocate(stateVectorWithZandP4D, tim_nstepobs, hco_ens, vco_ens, &
328                    dateStamp_opt=tim_getDateStamp(),  &
329                    mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
330                    dataKind_opt=4, allocHeightSfc_opt=.true.)
331  call gsv_zero(stateVectorWithZandP4D)
332
333  ! Allocate ensembles, read the Trl ensemble
334  call utl_tmg_start(2,'--ReadEnsemble')
335  call ens_allocate(ensembleTrl4D, nEns, tim_nstepobs, hco_ens, vco_ens, dateStampList, &
336                    fileMemberIndex1_opt=fileMemberIndex1)
337  call ens_readEnsemble(ensembleTrl4D, ensPathName, biPeriodic=.false.)
338  call utl_tmg_stop(2)
339
340  ! Compute ensemble mean and copy to meanTrl stateVectors
341  if (readEnsMeanFromFile) then
342    ! read the mean stateVector and copy to ensembleTrl4D object
343    call fln_ensTrlFileName(ensMeanFileName, '.', tim_getDateStamp())
344    ensMeanFileName = trim(ensMeanFileName) // '_trialmean'
345    inquire(file=trim(ensMeanFileName),exist=fileExists)
346    if (.not. fileExists) then
347      call utl_abort('midas-ensembleH: the ensemble mean file does not exist')
348    end if
349    do stepIndex = 1, tim_nstepobs
350      call gio_readFromFile(stateVectorMeanTrl4D, ensMeanFileName, ' ', ' ',  &
351                            containsFullField_opt=.true., readHeightSfc_opt=.true., &
352                            stepIndex_opt=stepIndex)
353    end do
354    call ens_copyToEnsMean(ensembleTrl4D, stateVectorMeanTrl4D)
355  else
356    call ens_computeMean(ensembleTrl4D)
357    call ens_copyEnsMean(ensembleTrl4D, stateVectorMeanTrl4D)
358  end if
359  
360  do memberIndex = 1, nEns
361
362    write(*,*) ''
363    write(*,*) 'midas-ensembleH: apply nonlinear H to ensemble member ', memberIndex
364    write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
365
366    ! copy 1 member to a stateVector
367    call ens_copyMember(ensembleTrl4D, stateVector4D, memberIndex)
368    call gsv_copy(stateVector4D, stateVectorWithZandP4D, allowVarMismatch_opt=.true., &
369                  beSilent_opt=.true.)
370    call gsv_copyHeightSfc(stateVectorHeightSfc, stateVectorWithZandP4D)
371
372    ! Compute and set Yb in ensObs
373    call s2c_nl(stateVectorWithZandP4D, obsSpaceData, column, hco_ens, &
374                timeInterpType=obsTimeInterpType, dealloc_opt=.false., &
375                beSilent_opt=.true.)
376
377    ! Compute Y-H(X) in OBS_OMP
378    call inn_computeInnovation(column, obsSpaceData, beSilent_opt=.true.)
379
380    ! Copy to ensObs: Y-HX for this member
381    call eob_setYb(ensObs, memberIndex)
382
383    ! Compute and set Yb in ensObsGain
384    do eigenVectorIndex = 1, numRetainedEigen
385      if (mmpi_myid == 0) write(*,*) 'midas-ensembleH: apply nonlinear H to modulated member ', &
386                                        eigenVectorIndex, '/', numRetainedEigen
387
388      ! modulate the member with eigenvectors of vertical localization matrix
389      call enkf_getModulatedState(stateVector4D, stateVectorMeanTrl4D, &
390                                  vLocalize, numRetainedEigen, nEns, &
391                                  eigenVectorIndex, stateVector4Dmod, &
392                                  beSilent=.true.)
393
394      call gsv_copy(stateVector4Dmod, stateVectorWithZandP4D, allowVarMismatch_opt=.true., &
395                    beSilent_opt=.true.)
396      call gsv_copyHeightSfc(stateVectorHeightSfc, stateVectorWithZandP4D)
397
398      call s2c_nl(stateVectorWithZandP4D, obsSpaceData, column, hco_ens, &
399                  timeInterpType=obsTimeInterpType, dealloc_opt=.false., &
400                  beSilent_opt=.true.)
401
402      ! Compute Y-H(X) in OBS_OMP
403      call inn_computeInnovation(column, obsSpaceData, filterObsAndInitOer_opt=.false., &
404                                 beSilent_opt=.true.)
405
406      ! Copy to ensObsGain: Y-HX for this member
407      memberIndexInEnsObs = (eigenVectorIndex - 1) * nEns + memberIndex
408      call eob_setYb(ensObsGain, memberIndexInEnsObs)
409    end do ! eigenVectorIndex
410    
411  end do
412  call gsv_deallocate(stateVectorWithZandP4D)
413  if (gsv_isAllocated(stateVector4Dmod)) call gsv_deallocate(stateVector4Dmod)
414  call gsv_deallocate(stateVector4D)
415  call gsv_deallocate(stateVectorMeanTrl4D)
416  call gsv_deallocate(stateVectorHeightSfc)
417
418  ! write local ensObs to file
419  call eob_writeToFiles(ensObs, outputFilenamePrefix='eob_HX', writeObsInfo=.true.)
420  if (useModulatedEns) then
421    call eob_writeToFiles(ensObsGain, outputFilenamePrefix='eobGain_HX', writeObsInfo=.false., &
422                          numGroupsToDivideMembers_opt=numRetainedEigen, &
423                          maxNumMembersPerGroup_opt=numFullEns)
424  end if
425
426  !
427  !- MPI, tmg finalize
428  !  
429  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
430  call utl_tmg_stop(0)
431
432  call tmg_terminate(mmpi_myid, 'TMG_INFO')
433  call rpn_comm_finalize(ierr) 
434
435  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
436
437  !
438  !- 7.  Ending
439  !
440  if (mmpi_myid == 0) write(*,*) ' --------------------------------'
441  if (mmpi_myid == 0) write(*,*) ' MIDAS-ENSEMBLEH ENDS'
442  if (mmpi_myid == 0) write(*,*) ' --------------------------------'
443
444end program midas_ensembleH
