 17program midas_obsSelection
 18  !
 19  !:Purpose: Main program for background check, and thinning of all observation types.
 20  !
 21  !          ---
 22  !
 23  !:Algorithm: For background check of each observation type, several quality control tests 
 24  !            specific to that observation type are performed on each available observation
 25  !            to determine if the observation meets the standard to be assimilated later.
 26  !            Observation flags are modified for rejected observations as well as for the 
 27  !            assimilated observations of inferior quality.The thinning is performed afterwards 
 28  !            to reduce the number of observation data for assimilation. This is to reduce the 
 29  !            1) computational cost, and 2) observation error correlation during 
 30  !            assimilation stage. The background-checked thinned observations are written
 31  !            to new files, ready for assimilation.
 32  !
 33  !            --
 34  !
 35  !            The computed bias correction values are applied to the observation before the 
 36  !            background check step for certain observation types (e.g. radiances). One of quality
 37  !            control tests during background check is the ``rogue check`` which is the allowed
 38  !            distance of the observation from the background state. Innovation vector  
 39  !            ``y-H(xb)`` is required to measure this distance. The innovations are 
 40  !            computed at the beginning of the program before the background check starts.
 41  !
 42  !            --
 43  !
 44  !============================================== ==============================================================
 45  ! Input and Output Files                        Description of file
 46  !============================================== ==============================================================
 47  ! ``flnml``                                      In - Main namelist file with parameters that user may modify
 48  ! ``flnml_static``                               In - The "static" namelist that should not be modified
 49  ! ``trlm_$NN`` (e.g. ``trlm_01``)                In - Background state (a.k.a. trial) files for each timestep
 50  ! ``analysisgrid``                               In - File defining grid for computing the analysis increment
 51  ! ``obsfiles_$FAM/obs$FAM_$NNNN_$NNNN``          In - Observation file for each "family" and MPI task
 52  ! ``obserr``                                     In - Observation error statistics
 53  ! ``obsfiles_$FAM.updated/obs$FAM_$NNNN_$NNNN``  Out - Updated obs file for each "family" and MPI task
 54  ! Remainder are files related to radiance obs:
 55  ! ``stats_$SENSOR_assim``                        In - Satellite radiance observation errors of difference sensors
 56  ! ``stats_tovs``                                 In - Satellite radiance observation errors
 57  ! ``stats_tovs_symmetricObsErr``                 In - User-defined symmetric TOVS errors for all sky
 58  ! ``Cmat_$PLATFORM_$SENSOR.dat``                 In - Inter-channel observation-error correlations
 59  ! ``dynbcor.coeffs.$SENSOR.*.coeffs_$SENSOR``    In - Dynamic bias correction file
 60  ! ``ceres_global.std``                           In - High-res surface type and water fraction for radiance obs
 61  ! ``rtcoef_$PLATFORM_$SENSOR.dat``               In - RTTOV coefficient files
 62  ! ``rttov_h2o_limits.dat``                       In - Min/max humidity limits applied to analysis
 63  ! ``ozoneclim98``                                In - Ozone climatology
 64  !============================================== ==============================================================
 65  !
 66  !           --
 67  !
 68  !:Synopsis: Below is a summary of the ``obsSelection`` program calling sequence:
 69  !
 70  !             - **Initial setups:**
 71  !
 72  !               - Read the NAMOBSSELECTION namelist and check/modify some values.
 73  !
 74  !               - Various modules are setup: ``obsFiles_mod``, ``timeCoord_mod``.
 75  !
 76  !               - Setup horizontal and vertical grid objects for "analysis
 77  !                 grid" from ``analysisgrid`` file.
 78  !
 79  !               - Setup ``obsSpaceData`` object and read observations from
 80  !                 files: ``inn_setupObs``.
 81  !
 82  !               - Compute and update the stored surface type for some satellite
 83  !                 radiance instruments. 
 84  !
 85  !               - Setup ``columnData`` module (read list of analysis variables 
 86  !                 from namelist) and allocate column object for storing trial 
 87  !                 on analysis levels.
 88  !
 89  !               - Setup the observation error statistics in ``obsSpaceData``
 90  !                 object: ``oer_setObsErrors``
 91  !
 92  !               - Setup ``gridStateVector`` module to initialize the gridstatevector 
 93  !                 objects.
 94  !
 95  !               - Applying optional bias corrections to some observation types.
 96  !
 97  !               - Setup horizontal and vertical grid objects for "trial grid" from
 98  !                 first trial file: ``trlm_01``.
 99  !
100  !               - Allocate a stateVector object on the trial grid and then
101  !                 read the trials: ``gio_readTrials``.
102  !
103  !             - **Computation**
104  !
105  !               - Compute ``columnTrlOnTrlLev`` and ``columnTrlOnAnlIncLev`` from
106  !                 background state: ``inn_setupColumnsOnTrlLev``,
107  !                 ``inn_setupColumnsOnAnlIncLev``.
108  !
109  !               - Compute innovation from background state: ``inn_computeInnovation``.
110  !
111  !               - Do background check for conventional observation: 
112  !                 ``bgck_bgCheck_conv``.
113  !
114  !               - Update radiance bias correction in ``obsSpaceData`` and apply 
115  !                 the bias corrections to the observations and innovations for
116  !                 radiances: ``bcs_calcBias``, ``bcs_applyBiasCorrection``.
117  !
118  !               - Perform background check for multiple observation types.
119  !             
120  !               - If thinning was requested: 
121  !
122  !                 - add some cloud parameters and set missing observation flags in 
123  !                   observation files (specific for radiances) and 
124  !
125  !                 - perform thinning for different observation types.
126  !
127  !               - Write the final background-checked bias corrected results (either
128  !                 thinned or not thinned) into the observation file.
129  ! 
130  !               - If thinning was requested, remove observations which were flagged 
131  !                 not to be assimilated from the observation file.
132  !
133  !             --
134  !
135  !:Options: `List of namelist blocks <../namelists_in_each_program.html#obsSelection>`_
136  !          that can affect the ``obsSelection`` program.
137  !
138  !          * The use of ``obsSelection`` program is controlled by the namelist block
139  !           ``&NAMOBSSELECTION`` read by the ``obsSelection`` program.
140  !
141  !          * Some of the other relevant namelist blocks used to configure the
142  !            ``obsSelection`` are listed in the following table:
143  ! 
144  !=========================== ========================= =========================================
145  ! Module                      Namelist                  Description of what is controlled
146  !=========================== ========================= =========================================
147  ! ``midas_obsSelection``      ``NAMOBSSELECTION``       whether thinning is performed or not.
148  ! ``biasCorrectionConv_mod``  ``NAMBIASCONV``           variables to perform bias correction 
149  !                                                       for conventional observations.
150  ! ``biasCorrectionConv_mod``  ``NAMSONDETYPES``         additional variables to perform bias 
151  !                                                       correction for radiosondes conventional 
152  !                                                       observations.
153  ! ``backgroundCheck_mod``     ``NAMBGCKCONV``           variables to perform background check 
154  !                                                       for conventional observations.
155  ! ``SSTbias_mod``             ``NAMSSTBIASESTIMATE``    variables to perform bias estimation 
156  !                                                       and bias correction for satellite SST.
157  ! ``biasCorrectionSat_mod``   ``NAMBIASSAT``            variables to perform bias correction 
158  !                                                       for satellite radiances.
159  ! ``multi_ir_bgck_mod``       ``NAMBGCKIR``             Variables to perform background check 
160  !                                                       for hyperspectral infrared radiances.
161  ! ``bgckmicrowave_mod``       ``NAMBGCK``               Variables to perform background check 
162  !                                                       for microwave radiances.
163  ! ``bgckcsr_mod``             ``NAMCSR``                Variables To perform background check 
164  !                                                       for CSR radiances.
165  ! ``bgckssmis_mod``           ``NAMBGCK``               Variables to perform background check 
166  !                                                       for SSMIS radiances.
167  ! ``bgckOcean_mod``           ``NAMOCEANBGCHECK``       Variables to perform background check 
168  !                                                       for ocean data.
169  ! ``bgckOcean_mod``           ``NAMICEBGCHECK``         Variables to perform background check 
170  !                                                       for SST data.
171  ! ``burpread_mod``            ``NAMADDTOBURP``          element IDs to add to the BURP file
172  ! ``thinning_mod``            ``THIN_HYPER``            variables to perform thinning on 
173  !                                                       hyperspectral infrared radiances.
174  ! ``thinning_mod``            ``THIN_TOVS``             variables to perform thinning on 
175  !                                                       microwave radiances.
176  ! ``thinning_mod``            ``thin_csr``              variables to perform thinning on 
177  !                                                       CSR radiances.
178  ! ``thinning_mod``            ``thin_raobs``            variables to perform thinning on 
179  !                                                       radiosonde observations.
180  ! ``thinning_mod``            ``thin_scat``             variables to perform thinning on 
181  !                                                       scatterometer wind observations.
182  ! ``thinning_mod``            ``thin_aircraft``         variables to perform thinning on 
183  !                                                       aircraft observations.
184  ! ``thinning_mod``            ``thin_surface``          variables to perform thinning on 
185  !                                                       surface observations.
186  ! ``thinning_mod``            ``thin_gbgps``            variables to perform thinning on 
187  !                                                       ground-based GPS observations.
188  ! ``thinning_mod``            ``thin_gpsro``            variables to perform thinning on 
189  !                                                       GPS radio-occultation observations.
190  ! ``thinning_mod``            ``thin_aladin``           variables to perform thinning on 
191  !                                                       aladin wind observations.
192  ! ``thinning_mod``            ``thin_aladin``           variables to perform thinning on 
193  !                                                       aladin wind observations.
194  ! ``timeCoord_mod``           ``NAMTIME``               assimilation time window length, 
195  !                                                       temporal resolution of the background 
196  !                                                       state.
197  !=========================== ========================= =========================================
198  !

200  use codePrecision_mod
201  use ramDisk_mod
202  use midasMpi_mod
203  use utilities_mod
204  use mathPhysConstants_mod
205  use horizontalCoord_mod
206  use verticalCoord_mod
207  use timeCoord_mod

211  use multi_ir_bgck_mod
212  use innovation_mod
213  use obsSpaceData_mod
214  use columnData_mod
215  use obsFiles_mod
216  use obsErrors_mod
217  use biasCorrectionSat_mod
218  use biasCorrectionConv_mod

228  integer :: dateStampFromObs, headerIndex, ierr, nulnam
229  type(struct_columnData),target :: columnTrlOnAnlIncLev
230  type(struct_columnData),target :: columnTrlOnTrlLev
231  type(struct_obs),       target :: obsSpaceData
232  type(struct_gsv)               :: stateVectorTrialHighRes
233  type(struct_hco),      pointer :: hco_anl => null()
234  type(struct_vco),      pointer :: vco_anl => null()
235  type(struct_hco),      pointer :: hco_trl => null()
236  type(struct_vco),      pointer :: vco_trl => null()
237  type(struct_hco),      pointer :: hco_core => null()
238
239  logical :: allocHeightSfc
240  integer :: get_max_rss, fnom, fclos
241
242  ! Namelist variables
243  logical                        :: doThinning  ! Control whether or not thinning is done
244
245  namelist /namObsSelection/ doThinning
246
247  call ver_printNameAndVersion('obsSelection','Obs Quality Control and Thinning')
248
249  !- 1.0 mpi
250  call mmpi_initialize
251
252  !- 1.1 timings
253  call tmg_init(mmpi_myid, 'TMG_INFO')
254  call utl_tmg_start(0,'Main')
255
256  !- 1.2 Read the namelist for obsSelection program (if it exists)
257  ! set default values for namelist variables
258  doThinning = .false.
259  if (utl_isNamelistPresent('namObsSelection', './flnml')) then
260    nulnam = 0
261    ierr = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
262    if (ierr /= 0) call utl_abort('midas-obsSelection: Error opening file flnml')
263    read(nulnam, nml = namObsSelection, iostat = ierr)
264    if (ierr /= 0) call utl_abort('midas-obsSelection: Error reading namelist namObsSelection')
265    if (mmpi_myid == 0) write(*,nml = namObsSelection)
266    ierr = fclos(nulnam)
267  else
268    write(*,*)
269    write(*,*) 'midas-obsSelection: Namelist block namObsSelection is missing in the namelist.'
270    write(*,*) '                    The default value will be taken.'
271    if (mmpi_myid == 0) write(*, nml = namObsSelection)
272  end if
273
274  !
275  !- Initialize the ram disk
276  !
277  call ram_setup
278
279  !     
280  !- Initialize observation file names, but don't use datestamp
281  !
282  call obsf_setup(dateStampFromObs, 'bgck')
283
284  !
285  !- Initialize the Temporal grid and dateStamp from trial file
286  !
287  call tim_setup(fileNameForDate_opt = './trlm_01')
288
289  !
290  !- Initialize constants
291  !
292  if (mmpi_myid == 0) then
293    call mpc_printConstants(6)
294    call pre_printPrecisions
295  end if
296
297  !
298  !- Initialize the Analysis grid
299  !
300  if(mmpi_myid == 0) write(*,*)
301  if(mmpi_myid == 0) write(*,*) 'midas-obsSelection: Set hco parameters for analysis grid'
302  call hco_SetupFromFile(hco_anl, './analysisgrid', 'ANALYSIS', 'Analysis' ) ! IN
303
304  if ( hco_anl % global ) then
305    hco_core => hco_anl
306  else
307    !- Initialize the core (Non-Extended) analysis grid
308    if( mmpi_myid == 0) write(*,*) 'midas-obsSelection: Set hco parameters for core grid'
309    call hco_SetupFromFile( hco_core, './analysisgrid', 'COREGRID', 'AnalysisCore' ) ! IN
310  end if
311
312  !     
313  !- Initialisation of the analysis grid vertical coordinate from analysisgrid file
314  !
315  call vco_SetupFromFile(vco_anl, './analysisgrid')
316
317  call col_setVco(columnTrlOnAnlIncLev, vco_anl)
318  write(*,*) 'Memory Used: ', get_max_rss()/1024,'Mb'
319
320  !
321  !- Setup and read observations
322  !
323  call inn_setupObs(obsSpaceData, hco_anl, 'ALL', 'LIKESPLITFILES', 'bgck')
324  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
325
326  ! if ssmis, compute the surface type ele and update obspacedata
327
328  call ssbg_computeSsmisSurfaceType(obsSpaceData)
329
330  ! Only for MWHS2 data and if modLSQ option is set to .true. in nambgck namelist, set values for land qualifier
331  ! indice and terrain type based on calculations
332  if (obs_famExist(obsSpaceData, 'TO')) then
333    call mwbg_computeMwhs2SurfaceType(obsSpaceData)
334  end if
335
336  !
337  !- Basic setup of columnData module
338  !
339  call col_setup
340  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
341
342  !
343  !- Memory allocation for background column data
344  !
345  call col_allocate(columnTrlOnAnlIncLev, obs_numheader(obsSpaceData), mpiLocal_opt = .true.)
346
347  !
348  !- Initialize the observation error covariances
349  !
350  call oer_setObsErrors(obsSpaceData, 'bgck')
351
352  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
353
354  !
355  ! Initialize list of analyzed variables.
356  !
357  call gsv_setup
358
359  ! Apply optional bias corrections
360  if (obs_famExist(obsSpaceData, 'TM')) then
361    call sstb_applySatelliteSSTBiasCorrection(obsSpaceData, hco_anl, &
362                                              vco_anl, columnTrlOnAnlIncLev)
363  end if  
364
365  if (obs_famExist(obsSpaceData, 'AI')) call bcc_applyAIBcor(obsSpaceData)    
366  if (obs_famExist(obsSpaceData, 'GP')) call bcc_applyGPBcor(obsSpaceData)
367  if (obs_famExist(obsSpaceData, 'UA')) call bcc_applyUABcor(obsSpaceData)
368    
369  ! Reading trials
370  call inn_getHcoVcoFromTrlmFile(hco_trl, vco_trl)
371  allocHeightSfc = (vco_trl%Vcode /= 0)
372
373  call gsv_allocate(stateVectorTrialHighRes, tim_nstepobs, hco_trl, vco_trl,  &
374                    dateStamp_opt = tim_getDateStamp(), mpi_local_opt = .true., &
375                    mpi_distribution_opt = 'Tiles', dataKind_opt = 4, &
376                    allocHeightSfc_opt = allocHeightSfc, &
377                    hInterpolateDegree_opt = 'LINEAR', beSilent_opt=.false.)
378  call gsv_zero(stateVectorTrialHighRes)
379  call gio_readTrials(stateVectorTrialHighRes)
380  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
381
382  ! Horizontally interpolate trials to trial columns
383  call inn_setupColumnsOnTrlLev(columnTrlOnTrlLev, obsSpaceData, hco_core, &
384                                stateVectorTrialHighRes)
385  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
386
387  ! Interpolate trial columns to analysis levels and setup for linearized H
388  call inn_setupColumnsOnAnlIncLev(columnTrlOnTrlLev, columnTrlOnAnlIncLev)
389
390  ! Compute observation innovations and prepare obsSpaceData for minimization
391  call inn_computeInnovation(columnTrlOnTrlLev, obsSpaceData, analysisMode_opt = .false.)
392
393  ! 2.2 Perform the background check
394  !     The routine also calls compute_HBHT and writes to listings & obsSpaceData
395
396  ! Do the conventional data background check
397  call bgck_bgCheck_conv(columnTrlOnAnlIncLev, columnTrlOnTrlLev, hco_anl, obsSpaceData)
398
399  if (obs_famExist(obsSpaceData, 'TO')) then
400
401    ! Satellite radiance bias correction
402    call bcs_calcBias(obsSpaceData, columnTrlOnTrlLev)        ! Fill in OBS_BCOR obsSpaceData column 
403                                                              ! with computed bias correction
404    call bcs_applyBiasCorrection(obsSpaceData, obs_var, 'TO') ! Apply bias correction to OBS
405    call bcs_applyBiasCorrection(obsSpaceData, obs_omp, 'TO') ! Apply bias correction to O-F
406
407    ! Do the TO background check
408    call irbg_bgCheckIR(columnTrlOnTrlLev, obsSpaceData)
409    call mwbg_bgCheckMW(obsSpaceData)
410    call csrbg_bgCheckCSR(obsSpaceData)
411    call ssbg_bgCheckSsmis(obsSpaceData)
412
413  end if
414
415  ! Do the ocean data background check
416  if (obs_famExist(obsSpaceData, 'TM')) call ocebg_bgCheckSST(obsSpaceData, tim_getDateStamp(), &
417                                                              columnTrlOnTrlLev, hco_trl)
418
419  ! Do the sea ice data gross background check
420  if (obs_famExist(obsSpaceData, 'GL')) call ocebg_bgCheckSeaIce(obsSpaceData)
421
422  if (doThinning) then
423
424    ! Copy original obs files into another directory
425    call obsf_copyObsDirectory('./obsOriginal', direction = 'TO')
426
427    ! 2.3 Write obs files after background check, but before thinning
428    call obsf_writeFiles(obsSpaceData, writeDiagFiles_opt = .false.)
429    
430    ! Add cloud parameter data to burp files (AIRS,IASI,CrIS,ATMS,AMSUA,...)
431    if (obs_famExist(obsSpaceData, 'TO')) then
432      call obsf_updateMissingObsFlags(obsSpaceData)
433      call obsf_addCloudParametersAndEmissivity(obsSpaceData)
434    end if
435
436    ! Copy the pre-thinning files into another directory
437    call obsf_copyObsDirectory('./obsBeforeThinning', direction = 'TO')
438
439    ! Copy original obs files back into usual directory
440    call obsf_copyObsDirectory('./obsOriginal', direction = 'FROM')
441
442    ! 2.4 Thinning:  Set bit 11 of flag, one observation type at a time
443    call thn_thinHyper(obsSpaceData)
444    call thn_thinTovs(obsSpaceData)
445    call thn_thinCSR(obsSpaceData)
446    call thn_thinRaobs(obsSpaceData)
447    call thn_thinScat(obsSpaceData)
448    call thn_thinSatWinds(obsSpaceData)
449    call thn_thinAircraft(obsSpaceData)
450    call thn_thinSurface(obsSpaceData, 'SF') ! surface data thinning
451    if (obs_famExist(obsSpaceData, 'TM')) then
452      call thn_thinSurface(obsSpaceData, 'TM') ! SST thinning
453      call thn_thinSatSST(obsSpaceData)        ! satellite SST thinning
454    end if      
455    call thn_thinGbGps(obsSpaceData)
456    call thn_thinGpsRo(obsSpaceData)
457    call thn_thinAladin(obsSpaceData)
458    ! if requested, dump the thinned predictors and coefficients to sqlite
459    call bcs_dumpBiasToSqliteAfterThinning(obsSpaceData)
460
461  end if
462
463  
464  ! 3 Write the final results
465
466  ! 3.1 Into the listings
467  write(*,*)
468  write(*,*) '> midas-obsSelection: printing the FIRST header and body'
469  do headerIndex = 1, min(1,obs_numHeader(obsSpaceData))
470    call obs_prnthdr(obsSpaceData, headerIndex)
471    call obs_prntbdy(obsSpaceData, headerIndex)
472  end do
473
474  ! 3.2 Into the observation files
475  write(*,*)
476  write(*,*) '> midas-obsSelection: writing to file'
477  call obsf_writeFiles(obsSpaceData)
478  
479  ! Add cloud parameter data to burp files (AIRS,IASI,CrIS,...)
480  if (obs_famExist(obsSpaceData, 'TO')) then
481    call obsf_updateMissingObsFlags(obsSpaceData)
482    call obsf_addCloudParametersAndEmissivity(obsSpaceData)
483  end if
484
485  ! cleaning the observation files
486  if (doThinning) call obsf_cleanObsFiles()
487
488  !
489  ! 4.  Ending
490  !
491  write(*,*)
492  write(*,*) '> midas-obsSelection: Ending'
493  call obs_finalize(obsSpaceData) ! deallocate obsSpaceData
494
495  call rpn_comm_finalize(ierr)
496
497  call utl_tmg_stop(0)
498  call tmg_terminate(mmpi_myid, 'TMG_INFO')
499
500end program midas_obsSelection
