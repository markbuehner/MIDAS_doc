 17program midas_prepcma
 18  !
 19  !:Purpose: Read the observation files (usually after output by the background check) 
 20  !          and apply further quality control and thinning for use by the ``LETKF`` program 
 21  !
 22  !          ---
 23  !
 24  !:Algorithm: After reading the observation files that have been processed by the background check,
 25  !            the ``prepcma`` program rejects more observations based on the
 26  !            various flags and conditions. It also performs further thinnings
 27  !            on the data types of aircraft (AI), scatterometer (SC),
 28  !            satellite winds (SW) and some radiance (TO). The rejection and thinning
 29  !            are controlled by the options in the namelist of ``NAMPREPCMA``.
 30  !            
 31  !            ---
 32  !
 33  !:File I/O: The required input files and produced output files are listed as follows. 
 34  !
 35  !          --
 36  !
 37  !============================================== ==============================================================
 38  ! Input and Output Files                         Description of file
 39  !============================================== ==============================================================
 40  ! ``flnml``                                      In - Main namelist file with parameters user may modify
 41  ! ``flnml_static``                               In - The "static" namelist that should not be modified
 42  ! ``obserr``                                     In - Observation error statistics
 43  ! ``obsfiles_$FAM/obs$FAM_$NNNN_$NNNN``          In - Observation file for each "family"
 44  ! ``stats_tovs``                                 In - Satellite radiance observation errors
 45  ! ``rtcoef_$PLATFORM_$SENSOR.dat``               In - RTTOV coefficient files 
 46  ! ``obsfiles_$FAM.updated/obs$FAM_$NNNN_$NNNN``  Out - final observation file for each family
 47  !============================================== ==============================================================
 48  !
 49  !          ---
 50  !
 51  !:Synopsis: Below is a summary of the ``prepcma`` program calling sequence:
 52  !
 53  !           - **Initial setups:**
 54  !            
 55  !             - Read the NAMPREPCMA namelist and check/modify some values.            
 56  !
 57  !             - ``filt_setup``: set up list of elements to be assimilated and flags for rejection
 58  !                              
 59  !             - ``obsf_setup``: get observation file names and datestamp
 60  !
 61  !           - **Computation:**
 62  !
 63  !             - ``obsf_readFiles``: get the observations 
 64  !
 65  !             - ``filt_suprep``: select the elements to assimilate and apply rejection flags 
 66  !
 67  !             - ``oer_setObsErrors``: initialize obs error covariances and set flag 
 68  !
 69  !             - ``oti_setup``: reject any observations outside the data assimilation window
 70  !
 71  !             - ``enkf_rejectHighLatIR``: reject all IR radiance observation in arctic and antarctic
 72  !
 73  !             - ``enkf_modifyAmsubObsError``: modify the obs error stddev for AMSUB in the tropics 
 74  !
 75  !             - ``thinning_fam``: perform thinning for aircraft (AI), scatterometer (SC),
 76  !                                 satellite winds (SW) and some radiance (TO)
 77  !
 78  !           - **Final steps:**
 79  !
 80  !             - ``obsf_writeFiles``: write to burp/sqlite files 
 81  !
 82  !             - ``obsf_printFiles``: print to ascci file and to unformatted files
 83  !
 84  !          ---
 85  !
 86  !:Options: `List of namelist blocks <../namelists_in_each_program.html#prepcma>`_
 87  !          that can affect the ``prepcma`` program.
 88  !
 89  !          * Some of the relevant namelist blocks used to configure the
 90  !            prepcma are listed in the following table:
 91  !
 92  !          --
 93  !   
 94  !=================== ====================== ===========================================
 95  ! Program/Module     Namelist               Description of what is controlled
 96  !=================== ====================== ===========================================
 97  ! ``midas_prepcma``  ``NAMPREPCMA``         parameters for CMA format and others 
 98  !                                           to modify, reject and thinning some  
 99  !                                           observation data 
100  ! ``timeCoord_mod``  ``NAMTIME``            assimilation time window length, temporal
101  !                                           resolution of the background state and the 
102  !                                           analysis
103  ! ``tovs_nl_mod``    ``NAMTOV``             The list of satellite and instrument
104  !=================== ====================== ===========================================
105  !
106  !
107  use version_mod
108  use obsSpaceData_mod
109  use obsFiles_mod
110  use obsFilter_mod
111  use obsTimeInterp_mod
112  use obsErrors_mod
113  use tovs_nl_mod
114  use timeCoord_mod
115  use enkf_mod
116  use utilities_mod
117  use midasMpi_mod
118  use ramDisk_mod
119  use regions_mod
120  use burpRead_mod

124  integer :: fnom, fclos, nulnam, ierr, dateStampFromObs
125  type(struct_obs), target  :: obsSpaceData
126  type(struct_oti), pointer :: oti => null()
127  real(kind=8) :: hx_dummy(1,1)
128  integer :: ncmahdr, ncmahx, ncmabdy, ncmadim, nobsout, nbrpform
129  logical :: qcvar, numHeader, numBody
130  character(len=7) :: resumeType
131
132  ! number of pressure ranges used for the thinning of aircraft (and other) data:
133  integer, parameter :: npres_ai = 5
134  integer, parameter :: npres_sw = 2
135  integer, parameter :: nai_target = 10
136  integer, parameter :: nsc_target = 10
137  integer, parameter :: nsw_target = 6
138  integer, parameter :: nto_target = 6 
139  real(8) :: nai_pmax(npres_ai) = (/ 25000.0, 40000.0, 60000.0, 80000.0, 110000.0/)
140  real(8) :: nsw_pmax(npres_sw) = (/ 60000.0, 110000.0/)
141  ! For a scalar array, no layer selection will be done
142  real(8) :: nsc_pmax(1) = (/ 0.0 /)
143  real(8) :: nto_pmax(1) = (/ 0.0 /)
144
145  ! Namelist variables:
146  character(len=256) :: cmahdr        ! should not be used anymore
147  character(len=256) :: cmabdy        ! should not be used anymore
148  character(len=256) :: cmadim        ! should not be used anymore
149  character(len=256) :: obsout        ! file name for ascii output
150  character(len=256) :: brpform       ! should not be used anymore
151  logical :: suprep                   ! choose to execute 'suprep' obs filtering
152  logical :: rejectOutsideTimeWindow  ! choose to reject obs outside time window
153  logical :: thinning                 ! choose to apply 'extra' thinning of some obs types
154  logical :: applySatUtil             ! choose to reject satellite obs based on 'util' column of stats_tovs
155  logical :: modifyAmsubObsError      ! choose to modify the obs error stddev for AMSUB/MHS in the tropics
156  logical :: rejectHighLatIR          ! choose to reject IR data in high latitudes
157  logical :: obsClean                 ! choose to remove rejected observations from files
158  logical :: writeObsFiles            ! choose to update the (burp or sqlite) observation files
159  logical :: writeAsciiCmaFiles       ! choose to write ascii output
160
161  NAMELIST /NAMPREPCMA/ cmahdr, cmabdy, cmadim, obsout, brpform,  &
162                        suprep, rejectOutsideTimeWindow, thinning, &
163                        applySatUtil, modifyAmsubObsError, rejectHighLatIR, &
164                        obsClean, writeObsFiles, writeAsciiCmaFiles
165
166  call ver_printNameAndVersion('prepcma','Prepare observations for LETKF')
167
168  !- 1.0 mpi
169  call mmpi_initialize
170
171  !- 1.1 timings
172  call tmg_init(mmpi_myid, 'TMG_INFO')
173  call utl_tmg_start(0,'Main')
174
175  if ( mmpi_myid == 0 ) call utl_writeStatus('PREPCMA_BEG')
176
177  !- Specify default values for namelist variables
178  cmahdr        = 'NOT_DEFINED'
179  cmabdy        = 'NOT_DEFINED'
180  cmadim        = 'NOT_DEFINED'
181  obsout        = 'NOT_DEFINED'
182  brpform       = 'brpform'
183  suprep                  = .true.
184  rejectOutsideTimeWindow = .true.
185  thinning                = .true.
186  applySatUtil            = .true.
187  modifyAmsubObsError     = .true.
188  rejectHighLatIR         = .true.
189  obsClean                = .true.
190  writeObsFiles           = .false.
191  writeAsciiCmaFiles       = .false.
192
193  nulnam = 0
194  ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
195  read(nulnam,nml=namprepcma,iostat=ierr)
196  if (ierr /= 0) call utl_abort('midas-prepcma: Error reading namelist')
197  if (mmpi_myid == 0) write(*,nml=namprepcma)
198  ierr = fclos(nulnam)
199
200  !- RAM disk usage
201  call ram_setup
202
203  call utl_tmg_start(10,'--Observations')
204
205  !- Set up list of elements to be assimilated and flags for rejection (from namelist)
206  call filt_setup('prepcma')
207
208  !- Observation file names and get datestamp
209  call obsf_setup(dateStampFromObs, 'prepcma')
210
211  !- Allocate obsSpaceData
212  call obs_class_initialize('ENKFMIDAS')
213  call obs_initialize( obsSpaceData, mpi_local=obsf_filesSplit() )
214
215  !- Read observations
216  call utl_tmg_start(11,'----ReadObsFiles')
217  call obsf_readFiles( obsSpaceData )
218  call utl_tmg_stop(11)
219
220  numHeader = obs_numheader(obsSpaceData)
221  numBody   = obs_numbody(obsSpaceData)
222  write(*,*) 'midas-prepcma: obs_numheader =', numheader
223  write(*,*) 'midas-prepcma: obs_numbody   =', numbody
224
225  !- Determine if qcvar flag is expected to be present
226  resumeType = brpr_getTypeResume() 
227  write(*,*) 'midas_prepcma: RESUME type =', resumeType
228  qcvar = (resumeType == 'POSTALT')
229  if (qcvar) then 
230    write(*,*) 'midas_prepcma: The input file is a postalt file'
231  else
232    write(*,*) 'midas_prepcma: The input file is NOT a postalt file'
233  end if
234
235  !- Initialize TOVS processing
236  if (obs_famExist(obsSpaceData,'TO')) call tvs_setup
237
238  !- Select the elements to assimilate and apply rejection flags
239  if (suprep) call filt_suprep(obsSpaceData)
240
241  !- Allocation for TOVS
242  if (obs_famExist(obsSpaceData,'TO')) call tvs_setupAlloc(obsSpaceData)
243
244  !- Initialize obs error covariances and set flag using 'util' column of stats_tovs
245  call oer_setObsErrors(obsSpaceData, 'analysis', useTovsUtil_opt=applySatUtil) ! IN
246
247  !- Call suprep again to 'black list' channels according to 'util' column of stats_tovs
248  if (applySatUtil) call filt_suprep(obsSpaceData)
249
250  call utl_tmg_stop(10)
251
252  !- Setup timeCoord module and set dateStamp from env variable
253  call tim_setup()
254  if (tim_getDateStamp() == 0) then
255    if (dateStampFromObs > 0) then
256      ! use dateStamp from obs if not already set by env variable
257      call tim_setDateStamp(dateStampFromObs)
258    else
259      call utl_abort('midas-prepcma: DateStamp was not set')
260    end if
261  end if
262
263  !- Reject any observation outside the data assimilation window
264  if (rejectOutsideTimeWindow) then
265    call oti_setup( oti, obsSpaceData, numStep=1, &
266                    headerIndexBeg=1, headerIndexEnd=obs_numheader(obsSpaceData), &
267                    flagObsOutside_opt=.true. )
268  end if
269  
270  !- Reject all IR radiance observation in arctic and antarctic (.i.e |lat|>60. )
271  if (rejectHighLatIR) call enkf_rejectHighLatIR(obsSpaceData)
272
273  !- Modify the obs error stddev for AMSUB in the tropics
274  if (modifyAmsubObsError) call enkf_modifyAmsubObsError(obsSpaceData)
275
276  !- Perform thinning for several observation types
277  if (thinning) then
278    ! perform thinning for aircraft observations
279    call thinning_fam(obsSpaceData, nai_pmax, nai_target, 'AI')
280    ! perform thinning for scatterometer observations
281    call thinning_fam(obsSpaceData, nsc_pmax, nsc_target, 'SC')
282    ! perform thinning for radiance observations
283    call thinning_fam(obsSpaceData, nto_pmax, nto_target, 'TO')
284    ! perform thinning for satwind observations
285    call thinning_fam(obsSpaceData, nsw_pmax, nsw_target, 'SW')
286  end if
287
288  !- Write the results
289  write(*,*)
290  write(*,*) '> midas-prepcma: writing to files'
291
292  !- Write to burp/sqlite files if requested
293  if (writeObsFiles) then
294    call obsf_writeFiles(obsSpaceData)
295    if ( obsClean ) call obsf_cleanObsFiles()
296  end if
297
298  if (writeAsciiCmaFiles) then
299
300    !- Remove all observations from obsSpaceData that will not be assimilated
301    !- But, unlike the EnKF program, do not check value of OBS_ZHA
302    if (obsClean) then
303      call obs_clean(obsSpaceData, hx_dummy, 0, -1, qcvar, checkZha_opt=.false.)
304    end if
305
306    if (mmpi_nprocs > 1) then
307      call obs_expandToMpiGlobal(obsSpaceData)
308    end if
309
310    if (mmpi_myid == 0) then
311      !- Open file for ascii output
312      nobsout = 0
313      ierr = fnom(nobsout, obsout, 'FMT+SEQ+R/W', 0)
314      call obs_print(obsSpaceData,nobsout)
315      close(nobsout)
316
317      !- Write the results in CMA format
318      ncmahdr = 0
319      ierr = fnom(ncmahdr, cmahdr, 'FTN+SEQ+UNF+R/W', 0)
320      ncmabdy = 0
321      ierr = fnom(ncmabdy, cmabdy, 'FTN+SEQ+UNF+R/W', 0)
322      ncmadim = 0
323      ierr = fnom(ncmadim, cmadim, 'FTN+SEQ+R/W', 0)
324      ncmahx  = -1
325      call obs_write(obsSpaceData, hx_dummy, 0, ncmahdr, ncmabdy, ncmahx, ncmadim)
326      close(ncmahdr)
327      close(ncmabdy)
328      close(ncmadim)
329
330      !! This used to contain a .true. or .false. value indicating if observations passed the QCVar
331      !! Since, this is not the case, we can write .false.
332      nbrpform = 0
333      ierr = fnom(nbrpform, brpform, 'FTN+SEQ+R/W', 0)
334      write(nbrpform,*) .false.
335      close(nbrpform)
336
337    end if
338
339  end if
340
341  !
342  !- 3.  Ending
343  !
344  write(*,*)
345  write(*,*) '> midas-prepcma: Ending'
346  call obs_finalize(obsSpaceData) ! deallocate obsSpaceData
347
348  call utl_tmg_stop(0)
349  call tmg_terminate(mmpi_myid, 'TMG_INFO')
350
351  call rpn_comm_finalize(ierr)
352
353  if ( mmpi_myid == 0 ) then
354    call utl_writeStatus('PREPCMA_END')
355  end if

359  subroutine thinning_fam(obsSpaceData, n_pmax, n_target, cfam)
360    !
361    ! :Purpose: thin the observations of the selected family
362    !
363    implicit none
364
365    ! arguments:
366    type (struct_obs) :: obsSpaceData  ! the data in observation space
367    real(8), intent(in) :: n_pmax(:)   ! pressure levels that separate vertical layers for the thinning
368    integer, intent(in) :: n_target    ! maximum desired amount of data per 3-D box
369    character(len=2) :: cfam           ! family type
370
371    ! locals:
372    type(struct_reg) :: lsc
373    integer :: idist, n_count_thin, iai, iseed(4)
374    integer :: nobs_count, nobs_count_mpiGlobal, nobs_count_thin, nobs_count_thin_mpiGlobal
375    integer :: nrep_count, nrep_count_mpiGlobal, nrep_count_thin, nrep_count_thin_mpiGlobal
376    integer :: headerIndex, bodyIndex, bodyIndexBeg, bodyIndexEnd
377    integer :: iblock, codeType, ilat, incr, ipres, nblocksum, npres, nsize, num_stn
378    logical :: count_obs, allRejected
379    real(4) :: lat_r4, lon_r4
380    real(8) :: pressure, rannum
381    real(8), allocatable :: latcenter(:), latmin(:), latmax(:), ranvals(:)
382    integer, allocatable :: nblockoffset(:), nlonblock(:)
383    integer, allocatable :: ai_indices(:,:), nstation(:,:), nstationMpiGlobal(:,:)
384    real(8), allocatable :: keep_ai(:,:)
385
386    ! box size that is used for observation thinning 
387    ! (the numerator is an approximate distance in km)
388    real(8), parameter :: r0_count_km = 200.0/(2**0.5)
389    ! next two parameters are not used in this program
390    real(8), parameter :: r1_dum = 1.0
391    real(8), parameter :: rz_dum = 1.0
392   
393    num_stn = obs_numheader(obsSpaceData)
394    npres = size(n_pmax,1) 
395    write(*,*) 'Start thinning for ', cfam, ' data'
396    ! at this stage we still have many radiance channels 
397    ! that will be rejected at a later stage.
398    if (cfam == 'XX') then
399      ! never getting here
400      write(*,*) 'count individual observations'   
401      count_obs = .true.
402    else
403      write(*,*) 'count the number of reports'
404      count_obs = .false.
405    end if
406         
407    call reg_init_struct(lsc, r0_count_km, r1_dum, rz_dum)
408    if (mmpi_myid == 0) write(*,*) 'number of latitude bands: ', lsc%nlatband
409    nsize = lsc%nlatband
410    allocate(latmin(nsize))
411    allocate(latmax(nsize))
412    allocate(latcenter(nsize))
413    allocate(nlonblock(nsize))
414    allocate(nblockoffset(nsize))
415
416    call reg_getlatitude(lsc%r0_rad, lsc%nlatband, latmin, latcenter, latmax)
417    if (mmpi_myid == 0) write(*,*) 'number of latitude bands: ',lsc%nlatband
418    do ilat = 1, lsc%nlatband
419      if (mmpi_myid == 0) write(*,*) ' band: ', ilat, ' latitude between ', latmin(ilat), latmax(ilat)
420    end do
421    call reg_getblock(lsc%nlatband, lsc%r0_rad, latmin, latmax, nlonblock)
422    nblocksum = 0
423    do ilat = 1, lsc%nlatband
424      nblockoffset(ilat) = nblocksum
425      nblocksum = nblocksum + nlonblock(ilat)
426      if (mmpi_myid == 0) write(*,*) 'latband: ', ilat, ' no of blocks: ', nlonblock(ilat)
427    end do 
428    if (mmpi_myid == 0) write(*,*) 'total number of blocks: ', nblocksum
429
430    nrep_count = 0
431    nobs_count = 0
432
433    allocate(nstation(nblocksum, npres))
434    allocate(keep_ai(nblocksum, npres))
435 
436    nstation=0
437    ! keep_ai = 1 corresponds to no thinning
438    keep_ai(:,:) = 1.0
439
440    allocate(ai_indices(num_stn,3))
441
442    header_loop: do headerIndex = 1, num_stn
443      codeType= obs_headElem_i(obsSpaceData, obs_ity, headerIndex)
444      if ( (cfam=='AI' .and. (codeType==42  .or. codeType==128 .or. &
445                              codeType==157 .or. codeType==177)) .or. &
446           (cfam=='SC' .and. codeType==254) .or. &
447           (cfam=='SW' .and. (codeType==88  .or. codeType==188)) .or. &
448           (cfam=='TO' .and. tvs_isIdBurpTovs(codeType)) ) then
449
450        ! skip this header if all observations already rejected
451        bodyIndexBeg = obs_headElem_i(obsSpaceData, obs_rln, headerIndex)
452        bodyIndexEnd = obs_headElem_i(obsSpaceData, obs_nlv, headerIndex) + bodyIndexBeg - 1
453        allRejected = .true.
454        body_loop: do bodyIndex = bodyIndexBeg, bodyIndexEnd
455          if (obs_bodyElem_i(obsSpaceData, obs_ass, bodyIndex) == obs_assimilated) then
456            allRejected = .false.
457            exit body_loop
458          end if
459        end do body_loop
460        if (allRejected) cycle header_loop
461
462        lat_r4 = obs_headElem_r(obsSpaceData, obs_lat, headerIndex)
463        lon_r4 = obs_headElem_r(obsSpaceData, obs_lon, headerIndex)
464        call reg_locatestn(lsc%r0_rad, lat_r4, lon_r4, &
465                           lsc%nlatband, nlonblock, &
466                           nblockoffset, iblock)
467        ! note that all data from the aircraft are at the same pressure
468        if (npres == 1) then
469          ipres = 1
470        else
471          pressure= obs_bodyElem_r(obsSpaceData, obs_ppp, bodyIndexBeg)
472          ipres = 1
473          do
474            if ((n_pmax(ipres) > pressure) .or. (ipres > npres)) exit
475            ipres = ipres + 1 
476          end do
477        end if  
478        if (ipres <= npres) then
479          if (count_obs) then
480            incr = obs_headElem_i(obsSpaceData, obs_nlv, headerIndex)
481          else
482            incr = 1
483          end if
484          nstation(iblock,ipres) = nstation(iblock,ipres) + incr
485          nobs_count = nobs_count + obs_headElem_i(obsSpaceData, obs_nlv, headerIndex)
486          nrep_count = nrep_count + 1
487          ai_indices(nrep_count, 1) = headerIndex
488          ai_indices(nrep_count, 2) = iblock
489          ai_indices(nrep_count, 3) = ipres 
490        end if
491      end if
492    end do header_loop
493
494    ! do mpi communication of the accumulators
495    nsize = nblocksum * npres
496    allocate(nstationMpiGlobal(nblocksum, npres))
497    call rpn_comm_allreduce(nstation, nstationMpiGlobal, nsize,  &
498                            'mpi_integer','mpi_sum', 'GRID', ierr)
499    call rpn_comm_allreduce(nrep_count, nrep_count_mpiGlobal, 1,  &
500                            'mpi_integer','mpi_sum', 'GRID', ierr)
501    call rpn_comm_allreduce(nobs_count, nobs_count_mpiGlobal, 1,  &
502                            'mpi_integer','mpi_sum', 'GRID', ierr)
503
504    write(*,*) 'total number of ', cfam, ' reports (local and mpiglobal): ',  &
505         nrep_count, nrep_count_mpiGlobal
506    allocate(ranvals(nrep_count))
507    write(*,*) 'total number of ', cfam, ' observations (local and mpiglobal): ',  &
508         nobs_count, nobs_count_mpiGlobal
509
510    n_count_thin = 0
511    do iblock = 1, nblocksum
512      do ipres = 1, npres
513        if (nstationMpiGlobal(iblock,ipres) .ge. 1) then
514          if (mmpi_myid == 0) write(*,*) 'block ipres and count: ',iblock,ipres, &
515                     nstationMpiGlobal(iblock,ipres)
516          if (nstationMpiGlobal(iblock,ipres) > n_target) then
517            keep_ai(iblock,ipres) = dble(n_target) / dble(nstationMpiGlobal(iblock,ipres))
518            n_count_thin = n_count_thin + n_target
519          else
520            n_count_thin = n_count_thin + nstationMpiGlobal(iblock,ipres)
521          end if
522        end if
523      end do
524    end do
525
526    if (count_obs) then
527      write(*,*) 'Estimated remaining number of ', cfam, ' observations (mpiGlobal): ', n_count_thin
528    else 
529      write(*,*) 'Estimated remaining number of ', cfam, ' reports (mpiGlobal): ', n_count_thin
530    end if
531
532    nrep_count_thin = 0
533    nobs_count_thin = 0
534
535    idist = 1
536    iseed(1) = 1
537    iseed(2) = 5
538    iseed(3) = 9
539    iseed(4) = 11
540    call dlarnv(idist,iseed,nrep_count,ranvals)
541    do iai = 1, nrep_count
542      headerIndex = ai_indices(iai,1)
543      iblock = ai_indices(iai,2)
544      ipres  = ai_indices(iai,3)
545      rannum = ranvals(iai)
546      if (rannum <= keep_ai(iblock,ipres)) then
547        nrep_count_thin = nrep_count_thin + 1
548        nobs_count_thin = nobs_count_thin + obs_headElem_i(obsSpaceData, obs_nlv, headerIndex)
549      else
550        ! reject the profile
551        bodyIndexBeg = obs_headElem_i(obsSpaceData, obs_rln, headerIndex)
552        bodyIndexEnd = obs_headElem_i(obsSpaceData, obs_nlv, headerIndex) + bodyIndexBeg - 1
553        do bodyIndex = bodyIndexBeg, bodyIndexEnd
554          call obs_bodySet_i(obsSpaceData, obs_ass, bodyIndex, obs_notAssimilated)
555          ! also set the 'rejected by selection process' flag (bit 11)
556          call obs_bodySet_i( obsSpaceData, obs_flg, bodyIndex,  &
557                              ibset( obs_bodyElem_i( obsSpaceData, obs_flg, bodyIndex ), 11) )
558        end do       
559      end if
560    end do
561
562    ! mpi communication of accumulators
563    call rpn_comm_allreduce(nrep_count_thin, nrep_count_thin_mpiGlobal, 1,  &
564                            'mpi_integer','mpi_sum', 'GRID', ierr)
565    call rpn_comm_allreduce(nobs_count_thin, nobs_count_thin_mpiGlobal, 1,  &
566                            'mpi_integer','mpi_sum', 'GRID', ierr)
567    
568    write(*,*) 'True remaining number of ', cfam, ' reports (local, mpiGlobal): ',  &
569         nrep_count_thin, nrep_count_thin_mpiGlobal
570    write(*,*) 'True remaining number of ', cfam, ' observations (local, mpiGlobal): ',  &
571         nobs_count_thin, nobs_count_thin_mpiGlobal
572
573    deallocate(ranvals)
574    deallocate(latmin)
575    deallocate(latmax)
576    deallocate(latcenter)
577    deallocate(nlonblock)
578    deallocate(nblockoffset)
579    deallocate(nstation)
580    deallocate(nstationMpiGlobal)
581    deallocate(keep_ai)
582    deallocate(ai_indices)
583
584  end subroutine thinning_fam
585
586end program
