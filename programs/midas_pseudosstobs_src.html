 17program midas_pseudoSSTobs
 18  !
 19  !:Purpose: Main program to produce pseudo SST observations 
 20  !          in ice-covered areas. Pseudo SST observations are needed
 21  !          to prevent the propagation of analysis increments to
 22  !          the ice-covered areas, that may result in undesirable sea-ice melting.
 23  !
 24  !          ---
 25  !
 26  !:Algorithm: Pseudo SST observations are assigned to the ice-covered 
 27  !            water points.
 28  !            First, a global sea-ice analysis is read.
 29  !            The sea-ice analysis file contains a mandatory sea-water
 30  !            fraction field.
 31  !            The grid and land-ocean mask are read 
 32  !            from the ``analysisgrid`` file.
 33  !
 34  !            --
 35  !
 36  !            Second, the number of ice-covered water points, including 
 37  !            concerned inland water points, are computed.
 38  !            If the number of ice-covered water points is zero,
 39  !            an empty observation SQLite file is created.
 40  !            If not, the computation of pseudo observations starts.
 41  !
 42  !            --
 43  !              
 44  !            First, the index array of ice-covered water points are 
 45  !            randomly shuffled to prevent the insertion of pseudo 
 46  !            observations at the same locations
 47  !            that would lead to spatial correlation of observations.
 48  !            Second, the pseudo observations of sea surface temperature :math:`T`
 49  !            are inserted at every ice-covered inland water point :math:`k`, 
 50  !            where the value of observations is computed as follows:
 51  !            :math:`T(k)=(1 - w(k)) * T_{fw} + w(k) * T_{s}`,
 52  !            where :math:`w(k)` is the sea-water fraction at the point :math:`k`,
 53  !            :math:`T_{fw}` is the temperature of fresh water below the ice,
 54  !            :math:`T_{s}` is a temperature of the sea water below the ice.
 55  !            The pseudo observations are inserted into every :math:`N`-th point 
 56  !            of sea water ice-covered points, 
 57  !            where the value of observation is defined as
 58  !            :math:`T_{s}`.
 59  !             
 60  !            --
 61  !  
 62  !            The computed observation values along with the corresponding 
 63  !            coordinates are put into ``obsSpaceData``. 
 64  !            Finally, output SQLite files are created.
 65  !       
 66  !            --
 67  !
 68  !=========================================================== ======================================================
 69  ! Input and Output Files                                     Description of file
 70  !=========================================================== ======================================================
 71  ! ``analysisgrid``                                           In - File defining sea-ice global grid
 72  ! ``seaice_analysis``                                        In - File containing ``LG`` and ``VF`` fields
 73  ! ``obsfiles_sst_pseudo.updated/obssst_pseudo_$NNNN_$NNNN``  Out - Pseudo obs file for each MPI task
 74  !=========================================================== ======================================================
 75  !
 76  !           --
 77  !
 78  !:Synopsis: Below is a summary of the ``pseudoSSTobs`` program calling sequence:
 79  !
 80  !           - **Initial setups:**
 81  !
 82  !             - Setup horizontal and vertical grid objects for "analysis
 83  !               grid" from ``analysisgrid``.
 84  !
 85  !             - Setup ``obsSpaceData`` object.
 86  !
 87  !             - Setup ``gridStateVector`` module.
 88  !
 89  !           - **Computation**
 90  !
 91  !             - ``utl_randomOrderInt`` random shuffle the ice-covered point indices
 92  !
 93  !             - ``oobs_computeObsData`` compute pseudo observation locations and values
 94  !                                       and save them in SQLite files.
 95  !
 96  !           --
 97  !
 98  !:Options: `List of namelist blocks <../namelists_in_each_program.html#pseudoSSTobs>`_
 99  !          that can affect the ``pseudoSSTobs`` program.
100  !
101  !          * The use of ``pseudoSSTobs`` program is controlled by the namelist block
102  !            ``&pseudoSSTobs`` read by the ``pseudoSSTobs`` program.
103  !
104  !          * ``iceFractionThreshold`` the sea-ice fraction threshold to define 
105  !                                      the presence of ice at each particular point
106  !
107  !          * ``outputSST`` the value of :math:`T_{s}` in K; 
108  !
109  !          * ``outputFreshWaterST`` the value of :math:`T_{fw}` in K;
110  !
111  !          * ``seaiceThinning`` pseudo observations are inserted into each :math:`N`-th point,
112  !                                this parameter controls the observation thinning
113  !
114  !          * ``outputFileName`` controls the output file names
115  !
116  !          * ``etiket`` etiket to put into the table "resume" of output SQLite file
117  ! 
118  !          *  ``seaWaterThreshold`` a threshold to distinguish sea and fresh water
119  !
120  !          --
121  !
122  !========================== ================ ==============================================================
123  ! Module                    Namelist         Description of what is controlled
124  !========================== ================ ==============================================================
125  ! ``oceanObservations_mod`` ``pseudoSSTobs`` parameters of pseudo SST observations
126  !========================== ================ ==============================================================
127  !
128  use version_mod
129  use ramDisk_mod
130  use utilities_mod
131  use horizontalCoord_mod
132  use verticalCoord_mod
133  use analysisGrid_mod
134  use midasMpi_mod
135  use oceanObservations_mod
136  use gridStateVector_mod
137  use obsSpaceData_mod
138   
139  implicit none
140
141  integer, external :: exdb, exfin, fnom, fclos, get_max_rss
142  integer :: ierr, istamp, nulnam
143
144  type(struct_hco), pointer   :: hco_anl => null()
145  type(struct_vco), pointer   :: vco_anl => null()
146  character(len=48),parameter :: obsMpiStrategy = 'LIKESPLITFILES'
147  character(len=48),parameter :: varMode        = 'analysis'
148
149  ! namelist variables
150  real(4)                     :: iceFractionThreshold    ! consider no ice condition below this threshold
151  real(4)                     :: outputSST               ! output SST value for pseudo observations
152  real(4)                     :: outputFreshWaterST      ! output fresh water surface temperature for pseudo obs.  
153  integer                     :: seaiceThinning          ! generate pseudo obs in every 'seaiceThinning' points 
154  character(len=100)          :: outputFileName          ! name of the file containing the generated observations
155  real(4)                     :: seaWaterThreshold       ! to distinguish inland water from sea water
156
157  namelist /pseudoSSTobs/ iceFractionThreshold, outputSST, outputFreshWaterST, seaiceThinning, &
158                          outputFileName, seaWaterThreshold
159  
160  istamp = exdb('pseudoSSTobs','DEBUT','NON')
161
162  call ver_printNameAndVersion('pseudoSSTobs','Generation of pseudo SST observations')
163
164  ! MPI initialization
165  call mmpi_initialize
166
167  call tmg_init(mmpi_myid, 'TMG_INFO')
168
169  call utl_tmg_start(0,'Main')
170 
171  ! 1. Top level setup
172
173  call ram_setup()
174 
175  ! Do initial set up
176  call pseudoSSTobs_setup()
177
178  call oobs_pseudoSST(hco_anl, vco_anl, iceFractionThreshold, outputSST, outputFreshWaterST, &
179                      seaiceThinning, outputFileName, seaWaterThreshold)
180
181  ! 3. Job termination
182
183  istamp = exfin('pseudoSSTobs','FIN','NON')
184
185  call utl_tmg_stop(0)
186
187  call tmg_terminate(mmpi_myid, 'TMG_INFO')
188
189  call rpn_comm_finalize(ierr) 
190
191  contains
192
193  subroutine pseudoSSTobs_setup()
194    !
195    ! :Purpose:  Control of the preprocessing of pseudo SST obs
196    !
197
198    implicit none
199    
200    !Locals:	
201    type(struct_hco), pointer   :: hco_core => null()
202    character(len=*), parameter :: myName = 'pseudoSSTobs_setup'
203    character(len=*), parameter :: gridFile = './analysisgrid'
204    
205    write(*,*) ''
206    write(*,*) '-------------------------------------------------'
207    write(*,*) '-- Starting subroutine '//myName//' --'
208    write(*,*) '-------------------------------------------------'
209
210    ! Setting default namelist variable values
211    iceFractionThreshold   = 0.05 
212    outputSST              = 271.4
213    outputFreshWaterST     = 271.4
214    outputFileName         = ''
215    seaiceThinning         = 5
216    seaWaterThreshold      = 0.0
217    
218    ! Read the namelist
219    nulnam = 0
220    ierr = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
221    read(nulnam, nml = pseudoSSTobs, iostat = ierr)
222    if (ierr /= 0) call utl_abort(myName//': Error reading namelist')
223    if (mmpi_myid == 0) write(*, nml = pseudoSSTobs)
224    ierr = fclos(nulnam)
225
226    write(*,*)''
227    write(*,*) myName//': output SST globally: ', outputSST
228    write(*,*) myName//': output fresh water surface temperature  globally: ', outputFreshWaterST
229    write(*,*) myName//': sea-ice fraction threshold: ', iceFractionThreshold
230    write(*,*) myName//': sea water fraction threshold: ', seaWaterThreshold
231    write(*,*) myName//': pseudo SST obs will be generated in every ', seaiceThinning, ' points of the sea-ice field'    
232    write(*,*) myName//': output file name: ', outputFileName
233    !
234    !- Initialize the Analysis grid
235    !
236    if(mmpi_myid == 0) write(*,*)''
237    if(mmpi_myid == 0) write(*,*) myName//': Set hco parameters for analysis grid'
238    call hco_SetupFromFile(hco_anl, gridFile, 'ANALYSIS') ! IN
239
240    if (hco_anl % global) then
241      call agd_SetupFromHCO(hco_anl) ! IN
242    else
243      !- Initialize the core (Non-Extended) analysis grid
244      if(mmpi_myid == 0) write(*,*) myName//': Set hco parameters for core grid'
245      call hco_SetupFromFile(hco_core, gridFile, 'COREGRID', 'AnalysisCore') ! IN
246      !- Setup the LAM analysis grid metrics
247      call agd_SetupFromHCO(hco_anl, hco_core) ! IN
248    end if
249
250    !     
251    !- Initialisation of the analysis grid vertical coordinate from analysisgrid file
252    !
253    call vco_SetupFromFile(vco_anl, & ! OUT
254                           gridFile) ! IN
255
256    !- Initialize variables of the model states
257    !
258    call gsv_setup
259
260    call obs_class_initialize('VAR')
261
262    if(mmpi_myid == 0) write(*,*) myName//': done.'
263    
264  end subroutine pseudoSSTobs_setup
265
266end program midas_pseudoSSTobs
