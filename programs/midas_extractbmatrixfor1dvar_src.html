 17program midas_extractBmatrixFor1Dvar
 18  !
 19  !:Purpose: Main program to extract B matrix to binary file Bmatrix.bin suitable for 1Dvar applications
 20  !          The B matrix is defined at a set of locations specified by the variable lonlatExtract and date 
 21  !          extractDate
 22  !
 23  !          ---
 24  !
 25  !:Algorithm:  The B matrix is computed column by column by application of operators bmat_sqrtBT and bmat_sqrtB 
 26  !
 27  !            --
 28  !
 29  !:File I/O: The required input files and produced output files are listed as follows.
 30  !
 31  !============================================== ==============================================================
 32  ! Input and Output Files                          Description of file
 33  !============================================== ==============================================================
 34  ! ``flnml``                                      In - Main namelist file with parameters user may modify
 35  ! ``flnml_static``                               In - The "static" namelist that should not be modified
 36  ! ``bgcov``                                      In - The B NMC matrix file
 37  ! ``analysisgrid``                               In - analysis grid 
 38  ! ``Bmatrix.bin``                                Out - The Bmatrix binary file for 1DVar
 39  !============================================== ==============================================================
 40  !
 41  !           --
 42  !
 43  !:Synopsis: Below is a summary of the ``extractBmatrixFor1Dvar`` program calling sequence:
 44  !    
 45  !             - **Initial setups:**
 46  !
 47  !               - Read parameters from the program namelist section NAMEXTRACT
 48  !
 49  !               - Setup temporal grid
 50  !
 51  !               - Get vertical and horizontal grid information from analysisgrid file
 52  !
 53  !               - Allocate a gridStatevector object
 54  !
 55  !               - Initialize the B matrix
 56  !
 57  !               - Initialize the gridded variable transform module 
 58  !
 59  !             - **Computation:**
 60  !
 61  !               - For each location specified in the namelist: 
 62  !
 63  !                   - for each element of the stateVector
 64  !
 65  !                      - set this elemnt to one and the other to zero
 66  !
 67  !                      - apply bmat_sqrtBT
 68  !
 69  !                      - apply bmat_sqrtB
 70  !
 71  !                      - get the corresponding column of the B matrix
 72  !
 73  !                   - write the resulting B matrix to file
 74  !           --
 75  !:Options: `List of namelist blocks <../namelists_in_each_program.html#extractbmatrixfor1dvar>`_
 76  !          that can affect the ``extractBmatrixFor1Dvar`` program.
 77  !
 78  !          - The use of ``extractBmatrixFor1Dvar`` program is controlled by the namelist block
 79  !            ``&namextract`` read by the ``extractBmatrixFor1Dvar`` program.
 80  !              *  ``extractdate`` date (YYYYMMDDHH format) for the B matrix extracted
 81  !              *  ``lonlatExtract`` (longitudes, latitudes) pairs definining the locations where the B matrix is to be extracted
 82  !              *  ``varNameExtract`` name of the variable to extract or ``all`` to extract everything in namstate
 83  !              *  ``stepBinExtract`` should be one of ``first``, ``middle`` or ``last`` to define when in the assimilation window the B matrix is valid
 84  !
 85  use version_mod
 86  use midasMpi_mod
 87  use MathPhysConstants_mod
 88  use controlVector_mod
 89  use gridVariableTransforms_mod
 90  use varNameList_mod
 91  use gridStateVector_mod
 92  use bmatrix_mod
 93  use horizontalCoord_mod
 94  use verticalCoord_mod
 95  use timeCoord_mod
 96  use utilities_mod
 97  use ramDisk_mod
 98  implicit none
 99
100  type(struct_gsv) :: statevector
101  type(struct_hco), pointer :: hco_anl  => null()
102  type(struct_hco), pointer :: hco_core => null()
103  type(struct_vco), pointer :: vco_anl  => null()
104  real(4) :: factor1, factor2
105  real(8), pointer :: field4d(:,:,:,:)
106  real(8), allocatable :: controlVector(:)
107  integer, parameter :: nmaxLevs = 100
108  real(4) :: latitude, longitude
109  integer, external :: fclos, fnom, fstopc, newdate, get_max_rss
110  integer :: ierr
111  integer :: varIndex, nkgdim, levIndex1, lonIndex, latIndex, levIndex2
112  integer :: kIndex1, kIndex2, columnProcIdLocal, columnProcIdGlobal, nulmat, varCount
113  integer :: idate, itime, nulnam, dateStamp
114  integer :: stepBinExtractIndex
115  integer :: nLonLatPos, lonLatPosIndex
116
117  integer, parameter :: lonColumn = 1
118  integer, parameter :: latColumn = 2
119
120  character(len=10)  :: datestr
121  character(len=4)   :: varName1, varName2
122  character(len=4),allocatable :: varList(:)
123  real(8), allocatable :: Bmatrix(:,:)
124
125  ! namelist variables
126  integer :: extractdate               ! date for the B matrix extracted
127  integer :: lonlatExtract(nmaxLevs,2) ! lon lat pairs definining the locations where the B matrix is to be extracted
128  character(len=128) :: stepBinExtract ! number of step bins to extract (1 typically for B NMC)
129  character(len=4)   :: varNameExtract ! variable to extract (all to extract everything in namstate)
130  namelist /namextract/ extractdate, lonlatExtract, &
131                    varNameExtract, stepBinExtract
132
133  call ver_printNameAndVersion('extractBmatrix', 'Extract 1Dvar B matrix')
134
135  ! MPI, tmg initialization
136  call mmpi_initialize
137  call tmg_init(mmpi_myid, 'TMG_INFO')
138  call utl_tmg_start(0, 'Main')
139  ierr = fstopc('MSGLVL', 'ERRORS', 0)
140
141  ! Set default values for namelist NAMEXTRACT parameters
142  extractdate        =  2011020100
143  varNameExtract     = 'all'
144  stepBinExtract     = 'middle'
145  lonlatExtract(:,:) = -1
146
147  ! Read the parameters from NAMEXTRACT
148  nulnam = 0
149  ierr = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
150  read(nulnam, nml=namextract, iostat=ierr)
151  if (ierr /= 0) call utl_abort('midas-extractBmatrix: Error reading namelist')
152  write(*, nml=namextract)
153  ierr = fclos(nulnam)
154
155  nLonLatPos = 0
156  do lonlatPosIndex = 1, size(lonlatExtract(:,lonColumn))
157    if (lonlatExtract(lonlatPosIndex,lonColumn) >= 1 .and. lonlatExtract(lonlatPosIndex,latColumn) >= 1) nLonLatPos = nLonLatPos + 1  
158  end do
159
160  if ( nLonLatPos == 0 ) then
161    call utl_abort('midas-extractBmatrix: You should specify at least one lonlat position !')
162  end if
163
164  ! Decompose extractdate(yyyymmddhh) into idate(YYYYMMDD) itime(HHMMSShh)
165  ! and calculate date-time stamp
166  idate = extractdate/100
167  itime = (extractdate-idate*100)*1000000
168  ierr = newdate(dateStamp, idate, itime, 3)
169  write(datestr,'(i10.10)') extractdate
170  write(*,*)' idate= ',idate,' time= ',itime
171  write(*,*)' date= ',extractdate,' stamp= ',dateStamp
172
173  ! Top Level Control setup
174  call ram_setup
175  !- Initialize the Temporal grid
176  call tim_setup
177  if (tim_getDateStamp() == 0) then
178    call tim_setDatestamp(dateStamp)
179  else
180    call utl_abort('midas-extractBmatrix: Code changes needed to use dateStamp from env variable')
181  end if
182  ! Initialize variables of the model states
183  call gsv_setup
184  ! Initialize the Analysis horizontal grid
185  call hco_SetupFromFile( hco_anl, './analysisgrid', 'ANALYSIS', 'Analysis' ) ! IN
186  if ( hco_anl % global ) then
187    hco_core => hco_anl
188  else
189    !- Iniatilized the core (Non-Extended) analysis grid
190    call hco_SetupFromFile( hco_core, './analysisgrid', 'COREGRID', 'AnalysisCore' ) ! IN
191  end if
192  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
193
194  ! Initialize the vertical coordinate from the statistics file
195  call vco_SetupFromFile( vco_anl,        & ! OUT
196                          './analysisgrid') ! IN
197  ! Allocate the statevector
198  call gsv_allocate(statevector, tim_nstepobsinc, hco_anl, vco_anl, &
199                    datestamp_opt=tim_getDatestamp(), mpi_local_opt=.true., &
200                    allocHeight_opt=.false., allocPressure_opt=.false.)
201  call gsv_zero(statevector)
202  nkgdim = statevector%nk
203  allocate( Bmatrix(nkgdim, nkgdim) )
204  ! Setup the B matrix
205  call bmat_setup(hco_anl, hco_core, vco_anl)
206  !- Initialize the gridded variable transform module
207  call gvt_setup(hco_anl, hco_core, vco_anl)
208  call gvt_setupRefFromTrialFiles('HU')
209  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
210
211  !
212  !==============================================
213  !- Compute columns of B matrix
214  !==============================================
215  !
216  select case(trim(stepBinExtract))
217  case ('first')
218   stepBinExtractIndex = 1
219  case ('middle')
220    if (mod(tim_nstepobsinc,2) == 0) then
221      write(*,*)
222      write(*,*) 'odd number of nstepobsinc a required for obs place in the middle of the analysis window '
223      write(*,*) 'tim_nstepobsinc = ', tim_nstepobsinc
224      call utl_abort('midas-extractBmatrix')
225    end if
226   stepBinExtractIndex = (tim_nstepobsinc+1)/2
227  case ('last')
228   stepBinExtractIndex = tim_nstepobsinc
229  case default
230    write(*,*)
231    write(*,*) 'Unsupported stepBinExtract : ', trim(stepBinExtract)
232    call utl_abort('midas-extractBmatrix')
233  end select
234  
235  allocate(controlVector(cvm_nvadim))
236
237  write(*,*) '************************************************************'
238  write(*,*) 'midas-extractBmatrix: Compute columns of B matrix for 1Dvar'
239  write(*,*) '************************************************************'
240  write(*,*)
241  write(*,*) ' temporal location           = ', trim(stepBinExtract), stepBinExtractIndex
242  write(*,*) ' number of lon-lat positions = ', nLonLatPos
243  
244  if (mmpi_myId == 0) then
245    varCount = 0
246    do varIndex=1, vnl_numvarmax
247      if ( .not. gsv_varExist(varName=vnl_varNameList(varIndex)) ) cycle
248      if ( trim(varNameExtract)  /= 'all' .and. (trim(varNameExtract) /= trim(vnl_varNameList(varIndex))) ) cycle
249      varCount = varCount + 1
250    end do
251    allocate(varList(varCount))
252    varCount = 0
253    do varIndex=1, vnl_numvarmax
254      if ( .not. gsv_varExist(varName=vnl_varNameList(varIndex)) ) cycle
255      if ( trim(varNameExtract) /= 'all' .and. (trim(varNameExtract) /= trim(vnl_varNameList(varIndex))) ) cycle
256      varCount = varCount + 1
257      varList(varCount) = vnl_varNameList(varIndex)
258    end do
259    nulmat = 0
260    ierr = fnom(nulmat, './Bmatrix.bin', 'FTN+SEQ+UNF', 0)
261    write(nulmat) extractDate, vco_anl % nlev_T, vco_anl % nlev_M, vco_anl % Vcode, &
262         vco_anl % ip1_sfc, vco_anl % ip1_T_2m, vco_anl % ip1_M_10m, varCount, nkgdim, nLonLatPos
263    write(nulmat) vco_anl % ip1_T(:), vco_anl % ip1_M(:), varList(:)
264  end if
265  
266  locationLoop:do lonLatPosIndex = 1, nLonLatPos
267
268    Bmatrix(:,:) = MPC_missingValue_R8
269
270    latIndex = lonlatExtract(lonLatPosIndex,latColumn)
271    lonIndex = lonlatExtract(lonLatPosIndex,lonColumn)
272    
273    latitude = hco_anl%lat2d_4(lonIndex, latIndex)
274    longitude = hco_anl%lon2d_4(lonIndex, latIndex)
275
276    variableLoop1:do kIndex1 = 1, nkgdim
277      varName1 = gsv_getVarNameFromK(statevector, kIndex1)
278      if ( .not. gsv_varExist(varName=varName1) ) cycle
279      if ( trim(varNameExtract) /= 'all' .and. trim(varNameExtract) /= trim(varName1) ) cycle
280      
281      write(*,*)
282      write(*,*) 'midas-extractBmatrix: simulating a pseudo-observation of ', trim(varName1)
283      
284      factor1 = getConversionFactor( varName1 )
285      levIndex1 = gsv_getLevFromK(statevector, kIndex1)
286      call gsv_zero(statevector)
287      call gsv_getField(statevector,field4d, varName1)
288      if ( latIndex >= statevector%myLatBeg .and. latIndex <= statevector%myLatEnd .and. &
289           lonIndex >= statevector%myLonBeg .and. lonIndex <= statevector%myLonEnd ) then
290        if (vnl_varLevelFromVarname(varName1) == 'SF') then
291          field4d(lonIndex, latIndex, 1, stepBinExtractIndex) = 1.0D0
292        else
293          field4d(lonIndex, latIndex, levIndex1, stepBinExtractIndex) = 1.0D0
294        end if
295      end if
296      !ici field4d est initialise a zero sauf au point qui nous interesse ou on a 1
297      !et donc aussi statevector (puisque field4V est un pointeur vers la partie correspondante de statevector)
298      controlVector(:) = 0.0d0
299      call bmat_sqrtBT(controlVector, cvm_nvadim, statevector)
300      call bmat_sqrtB (controlVector, cvm_nvadim, statevector)
301      ! ici statevector contient la colonne correspondante de la matrice B
302      write(*,*) 'midas-extractBmatrix: writing out the column of B. levIndex1,lonIndex,latIndex=', &
303                 levIndex1,lonIndex,latIndex
304      
305      variableLoop2:do kIndex2 = 1, nkgdim
306        varName2 = gsv_getVarNameFromK(statevector, kIndex2)
307        if ( .not. gsv_varExist(varName= varName2) ) cycle
308        if ( trim(varNameExtract) /= 'all' .and. trim(varNameExtract) /= trim(varName2) ) cycle
309        columnProcIdLocal = -1
310        if ( latIndex >= statevector % myLatBeg .and. latIndex <= statevector % myLatEnd .and. &
311             lonIndex >= statevector % myLonBeg .and. lonIndex <= statevector % myLonEnd ) then
312          columnProcIdLocal = mmpi_myId
313          call gsv_getField(statevector, field4d, varName2)
314          factor2 = getConversionFactor( varName2 )
315          levIndex2 = gsv_getLevFromK(statevector, kIndex2)
316          bmatrix(kIndex2, kIndex1) = factor1 * factor2 * field4d(lonIndex, latIndex, levIndex2, stepBinExtractIndex)
317        end if
318        call rpn_comm_allreduce(columnProcIdLocal, columnProcIdGlobal, 1, "mpi_integer", "mpi_max", "GRID", ierr)
319      end do variableLoop2
320
321    end do variableLoop1
322
323    call RPN_COMM_bcast(Bmatrix, nkgdim * nkgdim, 'MPI_REAL8', columnProcIdGlobal, 'GRID', ierr )
324    if (mmpi_myId ==0) then
325      write(nulmat) latitude, longitude,  Bmatrix(:,:)
326    end if
327
328  end do locationLoop
329    
330  ierr = fclos(nulmat)
331
332  deallocate(Bmatrix)
333  deallocate(controlVector)
334
335  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
336
337  call gsv_deallocate(statevector)
338
339  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
340  
341  ! MPI, tmg finalize
342  call utl_tmg_stop(0)
343  call tmg_terminate(mmpi_myid, 'TMG_INFO')
344  call rpn_comm_finalize(ierr) 
345
346  write(*,*) ' --------------------------------'
347  write(*,*) ' midas-extractBmatrix ENDS'
348  write(*,*) ' --------------------------------'
349
350contains
351
352  real(4) function getConversionFactor(varName)
353    character(len=*), intent(in) :: varName
354    if ( trim(varName) == 'UU' .or. trim(varName) == 'VV') then
355      getConversionFactor = mpc_knots_per_m_per_s_r4 ! m/s -> knots
356    else if ( trim(varName) == 'P0' ) then
357      getConversionFactor = 0.01 ! Pa -> hPa
358    else 
359      getConversionFactor = 1.0 ! no conversion
360    end if
361  end function getConversionFactor
362
363end program midas_extractBmatrixFor1Dvar
