 17program midas_var
 18  !
 19  !:Purpose: Main program for performing data assimilation using one of the
 20  !          following algorithms based on the incremental variational
 21  !          approach:
 22  !
 23  !            - 3D-Var
 24  !            - 3D- or 4D-EnVar
 25  !
 26  !          ---
 27  !
 28  !:Algorithm: The incremental variational data assimilation approach uses
 29  !            observations to compute a correction to the background state
 30  !            (i.e. the analysis increment) while taking into account the
 31  !            specified uncertainties for both the observations and the
 32  !            background state (i.e. the R and B covariance matrices,
 33  !            respectively). The analysis increment is computed by finding the
 34  !            minimum value of a cost function. This cost function is a scalar
 35  !            measure of the weighted departure of the corrected state from
 36  !            both the background state and the observations. The minimization
 37  !            is perfomed with a standard minimization algorithm (currently
 38  !            the quasi-Newton algorithm) that employs the gradient of the
 39  !            cost function to enable the minimum to be found (or at least
 40  !            approximated with sufficient accuracy) with relatively few
 41  !            iterations. The analysis increment is not assumed to be on the
 42  !            same horizontal and vertical grid as the background state. In
 43  !            addition, the temporal resolution of the background state and
 44  !            analysis increment are not assumed to be equal.
 45  !
 46  !            --
 47  !
 48  !            In practical terms, the background state is used to compute the
 49  !            "innovation" vector, that is, the difference between the
 50  !            observations and the background state in observation space:
 51  !            ``y-H(xb)``. The function ``H()`` represents the non-linear
 52  !            observation operators that map a gridded state vector into
 53  !            observation space. The innovation vector is used in the cost
 54  !            function calculation. Versions of the observation operators that
 55  !            are linearized with respect to the background state (or updated
 56  !            background state when using the outer loop) are also used in the
 57  !            cost function calculation to transform the analysis increment
 58  !            into observation space. The gradient of the cost function is
 59  !            computed using the adjoint of these linearized operators.
 60  !
 61  !            --
 62  !
 63  !            After the minimization, the analysis increment is spatially
 64  !            interpolated to the same grid as the background state and then
 65  !            added to this state to obtain the analysis. For some variables a
 66  !            physical constraint is imposed on the analysis
 67  !            (e.g. non-negative humidity or sea-ice concentration between 0
 68  !            and 1) and then the analysis increment is recomputed.
 69  !
 70  !            --
 71  !
 72  !:File I/O: The required input files and produced output files can vary
 73  !           according to the application. Below are tables of files for
 74  !           typical NWP 4D-EnVar (e.g. GDPS) and sea ice or SST 3D-Var
 75  !           applications.
 76  !
 77  !           --
 78  !
 79  !============================================== ==============================================================
 80  ! Input and Output Files (NWP applicaton)        Description of file
 81  !============================================== ==============================================================
 82  ! ``flnml``                                      In - Main namelist file with parameters user may modify
 83  ! ``flnml_static``                               In - The "static" namelist that should not be modified
 84  ! ``trlm_$NN`` (e.g. ``trlm_01``)                In - Background state (a.k.a. trial) files for each timestep
 85  ! ``analysisgrid``                               In - File defining grid for computing the analysis increment
 86  ! ``bgcov``                                      In - Static (i.e. NMC) B matrix file for NWP fields
 87  ! ``bgchemcov``                                  In - Static B matrix file for chemistry fields
 88  ! ``ensemble/$YYYYMMDDHH_006_$NNNN``             In - Ensemble member files defining ensemble B matrix
 89  ! ``obsfiles_$FAM/obs$FAM_$NNNN_$NNNN``          In - Observation file for each "family" and MPI task
 90  ! ``obserr``                                     In - Observation error statistics
 91  ! ``obsinfo_chm``                                In - Something needed for chemistry assimilation?
 92  ! ``preconin``                                   In - Preconditioning file (Hessian of the cost function)
 93  ! ``rebm_$MMMm`` (e.g. ``rebm_180m``)            Out - Analysis increment on the (low-res) analysis grid
 94  ! ``rehm_$MMMm``                                 Out - Analysis increment on the (high-res) trial grid
 95  ! ``anlm_$MMMm``                                 Out - Analysis on the trial grid
 96  ! ``obsfiles_$FAM.updated/obs$FAM_$NNNN_$NNNN``  Out - Updated obs file for each "family" and MPI task
 97  ! Remainder are files related to radiance obs:
 98  ! ``stats_$SENSOR_assim``                        In - Satellite radiance observation errors of difference sensors
 99  ! ``stats_tovs``                                 In - Satellite radiance observation errors
100  ! ``stats_tovs_symmetricObsErr``                 In - User-defined symmetric TOVS errors for all sky
101  ! ``Cmat_$PLATFORM_$SENSOR.dat``                 In - Inter-channel observation-error correlations
102  ! ``ceres_global.std``                           In - High-res surface type and water fraction for radiance obs
103  ! ``rtcoef_$PLATFORM_$SENSOR.dat``               In - RTTOV coefficient files
104  ! ``rttov_h2o_limits.dat``                       In - Min/max humidity limits applied to analysis
105  ! ``ozoneclim98``                                In - Ozone climatology
106  !============================================== ==============================================================
107  !
108  !           --
109  !
110  !============================================== ==============================================================
111  ! Input and Output Files (SST or Sea ice)        Description of file
112  !============================================== ==============================================================
113  ! ``flnml``                                      In - Main namelist file with parameters user may modify
114  ! ``trlm_$NN`` (e.g. ``trlm_01``)                In - Background state (a.k.a. trial) files for each timestep
115  ! ``analysisgrid``                               In - File defining grid for computing the analysis increment
116  ! ``sea_ice_obs-err``                            In - Observation error statistics (sea ice only)
117  ! ``bgstddev``                                   In - Background error stddev
118  ! ``diffusmod.std``                              In - Normalization factor for diffusion operator correlations
119  ! ``obsfiles_$FAM/obs$FAM_$NNNN_$NNNN``          In - Observation file for each "family" and MPI task
120  ! ``rebm_$MMMm`` (e.g. ``rebm_180m``)            Out - Analysis increment on the (low-res) analysis grid
121  ! ``rehm_$MMMm``                                 Out - Analysis increment on the (high-res) trial grid
122  ! ``anlm_$MMMm``                                 Out - Analysis on the trial grid
123  ! ``obsfiles_$FAM.updated/obs$FAM_$NNNN_$NNNN``  Out - Updated obs file for each "family" and MPI task
124  !============================================== ==============================================================
125  !
126  !           --
127  !
128  !:Synopsis: Below is a summary of the ``var`` program calling sequence:
129  !
130  !             - **Initial setups:**
131  !
132  !               - Setup horizontal and vertical grid objects for "analysis
133  !                 grid" from ``analysisgrid`` file and for "trial grid" from
134  !                 first trial file: ``trlm_01``.
135  !
136  !               - Setup ``obsSpaceData`` object and read observations from
137  !                 files: ``inn_setupObs``.
138  !
139  !               - Setup ``columnData`` and ``gridStateVector`` modules (read
140  !                 list of analysis variables from namelist) and allocate column
141  !                 object for storing trial on analysis levels.
142  !
143  !               - Setup the observation error statistics in ``obsSpaceData``
144  !                 object: ``oer_setObsErrors``.
145  !
146  !               - Allocate a stateVector object on the trial grid and then
147  !                 read the trials: ``gio_readTrials``.
148  !
149  !               - Setup the B matrices: ``bmat_setup``.
150  !
151  !               - Setup the ``gridVariableTransforms`` and ``minimization``
152  !                 modules.
153  !
154  !             - **Outer loop** (only 1 iteration when no outer loop is used)
155  !               during which the analysis increment is computed to correct the
156  !               current "updated state" which is used to compute the
157  !               innovation and serve as the starting point for the
158  !               minimization of the following outer loop iteration:
159  !
160  !               - Impose RTTOV humidity limits on updated state (initially the
161  !                 trial) on trial grid: ``qlim_rttovLimit``.
162  !
163  !               - Use the updated state on trial grid to setup the reference
164  !                 state used by ``gridVariableTransforms`` module for height
165  !                 calculations.
166  !
167  !               - Compute ``columnTrlOnTrlLev`` and ``columnTrlOnAnlIncLev`` from
168  !                 updated state: ``inn_setupColumnsOnTrlLev``,
169  !                 ``inn_setupColumnsOnAnlIncLev``
170  !
171  !               - Compute innovation from updated state:
172  !                 ``inn_computeInnovation``.
173  !
174  !               - Use the updated state on trial grid to setup the reference
175  !                 state used by ``gridVariableTransforms`` module for ``LQ``
176  !                 to ``HU`` calculations.
177  !
178  !               - Do the minimization for this outer loop iteration:
179  !                 ``min_minimize`` to obtain ``controlVectorIncr``.
180  !
181  !               - Update sum of all computed increment control vectors:
182  !                 ``controlVectorIncrSum(:)`` which is needed in the
183  !                 background cost function for outer loop.
184  !
185  !               - Compute ``stateVectorIncr`` (on analysis grid) from
186  !                 ``controlVectorIncr``: ``inc_getIncrement``.
187  !
188  !               - Interpolate and add ``stateVectorIncr`` to the updated
189  !                 state: ``inc_computeHighResAnalysis``.
190  !
191  !               - If requested, impose saturation and RTTOV humidity limits on
192  !                 the updated state.
193  !
194  !               - Write increment (or sum of increments when outer loop used)
195  !                 to file: ``inc_writeIncrement``.
196  !
197  !               - End of outer loop.
198  !
199  !             - **Final steps**, after the outer loop:
200  !
201  !               - If requested, compute final cost function value using
202  !                 non-linear observation operators.
203  !
204  !               - Write the final analysis and recomputed complete increment on
205  !                 the trial grid: ``inc_writeincAndAnalHighRes``.
206  !
207  !               - Various final steps, including: write the Hessian to binary
208  !                 file (``min_writeHessian``), update the observation files
209  !                 (``obsf_writeFiles``).
210  !
211  !           --
212  !
213  !:Options: `List of namelist blocks <../namelists_in_each_program.html#var>`_
214  !          that can affect the ``var`` program.
215  !
216  !          * The use of an outer loop is controlled by the namelist block
217  !            ``&NAMVAR`` read by the ``var`` program.
218  !
219  !          * The choice of 3D-Var vs. EnVar is controlled by the weights given
220  !            to the climatological (i.e. static) and ensemble-based background
221  !            error covariance (i.e. B matrix) components. The weights for all
222  !            B matrix components are zero be default and can be set to a
223  !            nonzero value through the namelist variable ``SCALEFACTOR`` in
224  !            the namelist block for each corresponding fortran module.
225  !
226  !          * Either algorithm can be used in combination with "First guess at
227  !            appropriate time" (i.e. FGAT) which is controlled by the namelist
228  !            variable ``DSTEPOBS`` (in ``NAMTIME``) that specifies the length
229  !            of time (in hours) between times when the background state is
230  !            used to compute the innovation (i.e. O-B).
231  !
232  !          * Similarly, the choice between 3D-EnVar and 4D-EnVar is controlled
233  !            by the namelist variable ``DSTEPOBSINC`` (also in ``&NAMTIME``)
234  !            which specifies the length of time (in hours) between times when
235  !            the analysis increment is computed. In the context of EnVar, if
236  !            this is less than the assimilation time window, then the
237  !            ensembles used to construct the ensemble-based B matrix component
238  !            will be used at multiple times within the assimilation window to
239  !            obtain 4D covariances (i.e. 4D-EnVar).
240  !
241  !          * Some of the other relevant namelist blocks used to configure the
242  !            variational analysis are listed in the following table:
243  ! 
244  !======================== ============ ==============================================================
245  ! Module                   Namelist     Description of what is controlled
246  !======================== ============ ==============================================================
247  ! ``minimization_mod``     ``NAMMIN``   maximum number of iterations, convergence criteria and many
248  !                                       additional parameters for controlling the minimization
249  ! ``timeCoord_mod``        ``NAMTIME``  assimilation time window length, temporal resolution of
250  !                                       the background state and increment
251  ! ``bMatrixEnsemble_mod``  ``NAMBEN``   weight and other parameters for ensemble-based B matrix
252  !                                       component
253  ! ``bMatrixHI_mod``        ``NAMBHI``   weight and other parameters for the climatological B matrix
254  !                                       component based on homogeneous-isotropic covariances
255  !                                       represented in spectral space
256  ! Other B matrix modules   various      weight and other parameters for each type of B matrix
257  !======================== ============ ==============================================================

530    end if

534    controlVectorIncr(:) = 0.0d0
535    deallocHessian = ( numOuterLoopIterations == 1 )
536    isMinimizationFinalCall = ( outerLoopIndex == numOuterLoopIterations )
537    call min_minimize( outerLoopIndex, columnTrlOnAnlIncLev, obsSpaceData, controlVectorIncrSum, &
538                       controlVectorIncr, numIterMaxInnerLoop(outerLoopIndex), &
539                       deallocHessian_opt=deallocHessian, &
540                       isMinimizationFinalCall_opt=isMinimizationFinalCall, &
541                       numIterMaxInnerLoopUsed_opt=numIterMaxInnerLoopUsed )
542    numInnerLoopIterDone = numInnerLoopIterDone + numIterMaxInnerLoopUsed
543    call msg_memUsage('var')
544
545    ! Accumulate control vector increments of all the previous iterations
546    controlVectorIncrSum(:) = controlVectorIncrSum(:) + controlVectorIncr(:)
547
548    ! Compute satellite bias correction increment and write to file on last outer-loop 
549    ! iteration
550    if ( outerLoopIndex == numOuterLoopIterations ) then
551      call bcs_writebias(controlVectorIncr)
552    end if
553
554    call tvs_deallocateProfilesNlTlAd
555
556    call gsv_allocate(stateVectorIncr, tim_nstepobsinc, hco_anl, vco_anl, &
557         datestamp_opt=tim_getDatestamp(), mpi_local_opt=.true., &
558         dataKind_opt=pre_incrReal, allocHeight_opt=.false., allocPressure_opt=.false.)
559
560    ! get final increment with mask if it exists
561    call inc_getIncrement( controlVectorIncr, stateVectorIncr, cvm_nvadim )
562    call gio_readMaskFromFile( stateVectorIncr, './analysisgrid' )
563    call msg_memUsage('var')
564
565    ! Compute high-resolution analysis on trial grid
566    call inc_computeHighResAnalysis( stateVectorIncr,                                  & ! IN
567                                     stateVectorUpdateHighRes, stateVectorPsfcHighRes )  ! OUT
568    call msg_memUsage('var')
569
570    ! Impose limits on stateVectorUpdateHighRes only when outer loop is used.
571    if ( limitHuInOuterLoop ) then
572      call msg('var','impose limits on stateVectorUpdateHighRes')
573      call qlim_saturationLimit( stateVectorUpdateHighRes )
574      call qlim_rttovLimit( stateVectorUpdateHighRes )
575    end if
576
577    ! prepare to write incremnt when no outer-loop, or sum of increments at last
578    ! outer-loop iteration.
579    if ( numOuterLoopIterations > 1 .and. &
580         outerLoopIndex == numOuterLoopIterations ) then
581
582      call gsv_allocate(stateVectorIncrSum, tim_nstepobsinc, hco_anl, vco_anl, &
583           datestamp_opt=tim_getDatestamp(), mpi_local_opt=.true., &
584           dataKind_opt=pre_incrReal, allocHeight_opt=.false., allocPressure_opt=.false.)
585      call inc_getIncrement( controlVectorIncrSum, stateVectorIncrSum, cvm_nvadim )
586      call gio_readMaskFromFile( stateVectorIncrSum, './analysisgrid' )
587      call msg_memUsage('var')
588
589      call inc_writeIncrement( stateVectorIncrSum, &     ! IN
590                               ip3ForWriteToFile_opt=0 ) ! IN
591      call gsv_deallocate( stateVectorIncrSum )
592    else if ( numOuterLoopIterations == 1 ) then
593      call inc_writeIncrement( stateVectorIncr, &        ! IN
594                               ip3ForWriteToFile_opt=0 ) ! IN
595    end if
596    call msg_memUsage('var')
597
598    call gsv_deallocate( stateVectorIncr )
599
600    call msg('var','end of outer-loop index='//str(outerLoopIndex))
601  end do outer_loop
602
603  ! Set the QC flags to be consistent with VAR-QC if control analysis
604  if ( varqcActive ) call vqc_listrej(obsSpaceData)
605
606  if ( computeFinalNlJo ) then
607    ! Horizontally interpolate high-resolution stateVectorUpdate to trial columns
608    call inn_setupColumnsOnTrlLev( columnTrlOnTrlLev, obsSpaceData, hco_core, &
609                                   stateVectorUpdateHighRes )
610    call msg_memUsage('var')
611
612    ! Determine if to apply varqc to Jo of non-linear operator
613    applyVarqcOnNlJo = ( varqcActive .and. numInnerLoopIterDone > numIterWithoutVarqc )
614    if ( applyVarqcOnNlJo ) call msg('var','applying varqc to non-linear Jo', mpiAll_opt=.false.)
615
616    ! Compute observation innovations and prepare obsSpaceData for minimization
617    filterObsAndInitOer = .false.
618    call inn_computeInnovation( columnTrlOnTrlLev, obsSpaceData, &
619                                destObsColumn_opt=OBS_OMA, &
620                                filterObsAndInitOer_opt=filterObsAndInitOer, &
621                                applyVarqcOnNlJo_opt=applyVarqcOnNlJo , &
622                                callSetErrGpsgb_opt=filterObsAndInitOer )
623    call msg_memUsage('var')
624  end if
625
626  ! Memory deallocations for non diagonal R matrices for radiances
627  call rmat_cleanup()
628
629  ! Conduct obs-space post-processing diagnostic tasks (some diagnostic
630  ! computations controlled by NAMOSD namelist in flnml)
631  call osd_ObsSpaceDiag( obsSpaceData, columnTrlOnAnlIncLev, hco_anl )
632
633  ! Deallocate memory related to B matrices and update stateVector
634  call bmat_finalize()
635
636  ! Deallocate structures needed for interpolation
637  call s2c_deallocInterpInfo( inputStateVectorType='nl' )
638  call s2c_deallocInterpInfo( inputStateVectorType='tlad' )
639
640  ! Post processing of analyis before writing (variable transform+humidity clipping)
641  call inc_analPostProcessing( stateVectorPsfcHighRes, stateVectorUpdateHighRes, &  ! IN 
642                               stateVectorTrial, stateVectorPsfc, stateVectorAnal ) ! OUT
643  call gsv_deallocate( stateVectorUpdateHighRes )
644
645  ! compute and write the analysis (as well as the increment on the trial grid)
646  call inc_writeIncAndAnalHighRes( stateVectorTrial, stateVectorPsfc, &
647                                   stateVectorAnal )
648  call msg_memUsage('var')
649
650  if (mmpi_myid == 0) then
651    clmsg = 'REBM_DONE'
652    call utl_writeStatus(clmsg)
653  end if
654
655  ! write the Hessian
656  call min_writeHessian(controlVectorIncr)
657  deallocate(controlVectorIncr)
658
659  ! Deallocate memory related to variational bias correction
660  call bcs_finalize()
661
662  ! Now write out the observation data files
663  if (min_niter > 0) then
664    if ( .not. obsf_filesSplit() ) then 
665      call msg('var','reading/writing global observation files')
666      call obs_expandToMpiGlobal(obsSpaceData)
667      if (mmpi_myid == 0) call obsf_writeFiles(obsSpaceData)
668    else
669      ! redistribute obs data to how it was just after reading the files
670      call obs_MpiRedistribute(obsSpaceData,OBS_IPF)
671      call obsf_writeFiles(obsSpaceData)
672    end if
673  end if
674
675  ! Deallocate copied obsSpaceData
676  call obs_finalize(obsSpaceData)
677
678  ! Job termination
679  istamp = exfin('VAR','FIN','NON')
680
681  if (mmpi_myid == 0) then
682    clmsg = 'VAR3D_END'
683    call utl_writeStatus(clmsg)
684  end if
685
686  call utl_tmg_stop(0)
687
688  call tmg_terminate(mmpi_myid, 'TMG_INFO')
689
690  call rpn_comm_finalize(ierr) 
691
692end program midas_var
