 17program midas_randomPert
 18  !
 19  !:Purpose: Main program for generating an ensemble of random perturbations,
 20  !          based on the B matrix (can be homogeneous and isotropic, ensemble-based
 21  !          or a weighted sum of covariance matrices).
 22  !
 23  !          ---
 24  !
 25  !:Algorithm: Uses a gaussian pseudorandom generator to produce an ensemble
 26  !            of control vectors.  These control vectors are transformed into
 27  !            pertubations in physical space, through the action of the square-root
 28  !            of the background-error covariance matrix.  The ensemble of perturbations
 29  !            can then be debiased and some smoothing can be applied.
 30  !            The perturbations thus obtained can then be interpolated and added
 31  !            to a provided ensemble mean and/or be mixed with some previous time
 32  !            perturbations.
 33  !
 34  !            ---
 35  !
 36  !:File I/O:
 37  !
 38  !============================================== ==============================================================
 39  ! Input and Output Files                         Description of file
 40  !============================================== ==============================================================
 41  ! ``analysisgrid``                               In - File defining grid on which perturbations are generated
 42  ! ``targetgrid``                                 In - File defining destination ensemble grid
 43  ! ``bgcov``                                      In - Static (i.e. NMC) B matrix file for NWP fields
 44  ! ``seaIceAnalysis``                             In - Sea-ice analysis, to set under-ice perturbations to zero
 45  ! ``pert_$YYYYMMDDHH_000_$NNNN``                 In - Previous date perturbations, when ensemble not provided
 46  ! ``$YYYYMMDDHH_006_$NNNN``                      In/Out - Ensemble member files
 47  ! ``pert_$YYYYMMDDHH_000_$NNNN``                 Out - Output perturbations
 48  !============================================== ==============================================================
 49  !
 50  !
 51  !:Synopsis: Below is a summary of the ``randomPert`` program calling sequence:
 52  !
 53  !             - **Initial setups:**
 54  !
 55  !               - Initialize the horizontal and vertical grid objects; for the
 56  !                 generation of the perturbations from ``analysisgrid`` and
 57  !                 for the ensemble output from ``targetgrid``.
 58  !
 59  !               - Setup the B matrices: ``bmat_setup``.
 60  !
 61  !               - Allocate arrays and a stateVector object on the ``analysisgrid``,
 62  !                 to receive the perturbations and first statistical moments.
 63  !
 64  !               - Initialize the pseudorandom generator : ``rng_setup``.
 65  !
 66  !             - **Generation of the ensemble of perturbations**.
 67  !               Iterate until the number of required perturbations (``NENS``)
 68  !               is generated:
 69  !
 70  !               - Generate a random control vector from a unit variance gaussian
 71  !                 distribution: ``rng_gaussian``.
 72  !
 73  !               - Transform control vector to physical space perturbation using
 74  !                 ``bmat_sqrtB``.
 75  !
 76  !             - **Unbias the ensemble of perturbations** if ``REMOVE_MEAN`` is true.
 77  !
 78  !             - **Smooth variances** if ``smoothVariances`` is true.
 79  !
 80  !             - **Read ensemble mean** if ``readEnsMean`` is true:
 81  !
 82  !               - Allocate a stateVector object on the ``targetgrid``, and then
 83  !                 read the ensemble mean using ``gio_readFromFile``.
 84  !
 85  !               - if ``setPertZeroUnderIce`` is true, allocate a stateVector
 86  !                 object and read sea-ice analysis using ``gio_readFromFile``.
 87  !                 Where the ice fraction is higher than ``iceFractionThreshold``,
 88  !                 the perturbations are set to zero.
 89  !
 90  !             - **Read and debias previous date perturbations** if
 91  !               ``previousDateFraction`` is higher than zero.
 92  !
 93  !             - **Final steps**:
 94  !
 95  !               - Allocate a statevector object on the ``targetgrid``, copy ocean
 96  !                 mask and interpolate perturbations using ``int_interp_gsv``
 97  !
 98  !               - If ``previousDateFraction`` higher than zero, average previous
 99  !                 date perturbations with the ones generated for the present date.
100  !
101  !               - If ``readEnsMean`` is true, add the ensemble mean to the perturbations.
102  !
103  !               - Write the perturbations using ``gio_writeToFile``.  If ``readEnsMean``
104  !                 is true, write the ensemble mean as well.
105  !
106  !               ---
107  !
108  !:Options: `List of namelist blocks <../namelists_in_each_program.html#randompert>`_
109  !          that can affect the ``randomPert`` program.
110  !
111  !         * ``&NAMENKF`` define specific parameters to configure the ``randomPert``
112  !           program and is read by the program.
113  !
114  !         * The background-error covariance matrix contributions are defined through
115  !           the namelists ``&NAMBHI``, for the homogeneous and isotropic part,
116  !           and ``&NAMBEN`` for the ensemble part.  When the matrix is a weighted
117  !           sum, it is important to make sure to define the ``SCALEFACTOR`` values.
118  !           Note that it would be also possible to define contributions from
119  !           ``&NAMBDIFF`` for diffusion correlations and ``&NAMBCHM`` for homogeneous
120  !           and isotropic chemical constituents covariances.
121  !
122

253

258  if (tim_getDateStamp() == 0) then

265  end if

268  ierr = newdate(dateStamp, datePrint, timePrint, imode)
269  write(dateString, '(I10)') datePrint*100 + timePrint/1000000
270  if( mmpi_myid == 0 ) then
271    write(*,*) ' date= ', datePrint, ' time= ', timePrint, ' stamp= ', dateStamp

278  !- 2.3 Initialize the horizontal grids from analysisgrid and targetgrid files
279  if (mmpi_myid == 0) write(*,*)
280  if (mmpi_myid == 0) write(*,*) 'Set hco parameters for analysis grid'
281  call hco_setupFromFile(hco_anl, './analysisgrid', 'ANALYSIS', 'Analysis' ) ! IN

293    if (mmpi_myid == 0) write(*,*) 'Set hco parameters for target grid'

315                             latPerPEt, latPerPEmaxt, myLatBegt, myLatEndt ) ! OUT

317                             lonPerPEt, lonPerPEmaxt, myLonBegt, myLonEndt ) ! OUT

356                    dateStamp_opt=dateStamp, mpi_local_opt=.true., &

358                    hInterpolateDegree_opt='LINEAR')

361
362  !- 3.2 Allocate auxillary variables

364  allocate(controlVector(cvm_nvadim))

369  !
370  !- 4. Compute an ensemble of random perturbations
371  !
372
373  if( mmpi_myid == 0 ) write(*,*) '******************'
374  if( mmpi_myid == 0 ) write(*,*) 'midas-randomPert: COMPUTE the mean of the random ', &
375                                  'perturbations of all the members'
376
377  if( mpiTopoIndependent ) then
378    call rng_setup(abs(randomSeed))
379  else
380    call rng_setup(abs(randomSeed+mmpi_myid))
381  end if
382
383  gdmean(:,:,:) = 0.0D0
384  call gsv_getField(stateVectorPert,field)
385
386  !- 4.1 Generate a (potentially) biased ensemble
387  do memberIndex = 1, NENS
388
389    if( mmpi_myid == 0 ) write(*,*) '...computing member number= ', memberIndex
390
391    !- 4.1.1 Create a random control vector in spectral space
392
393    if( mpiTopoIndependent ) then
394      !- Global vector (for testing different mpi topology, less efficient)
395      do cvIndex = 1, cvm_nvadim_mpiglobal
396        controlVector_mpiglobal(cvIndex) = rng_gaussian()
397      end do
398      call bmat_reduceToMPILocal( controlVector, controlVector_mpiglobal )
399    else
400      !- Local vector (different randomSeed for each processor, more efficient)
401      do cvIndex = 1, cvm_nvadim
402        controlVector(cvIndex) = rng_gaussian()
403      end do
404    end if
405
406    !- 4.1.2 Transform to control variables in physical space
407    call bmat_sqrtB(controlVector, cvm_nvadim, & ! IN
408                    stateVectorPert           )  ! OUT
409
410    !- 4.1.3 Copy perturbations to big array and update ensemble sum
411    !$OMP PARALLEL DO PRIVATE (levIndex, latIndex, lonIndex)    
412    do levIndex = 1, nkgdim
413      do latIndex = myLatBega, myLatEnda
414        do lonIndex = myLonBega, myLonEnda
415          ensemble_r4(lonIndex, latIndex, levIndex, memberIndex) &
416               = real(field(lonIndex, latIndex, levIndex), 4)
417          gdmean(lonIndex, latIndex, levIndex) &
418               = gdmean(lonIndex, latIndex, levIndex) + field(lonIndex, latIndex, levIndex)
419        end do
420      end do
421    end do
422    !$OMP END PARALLEL DO
423
424  end do
425
426  call gsv_deallocate(stateVectorPert)
427  
428  !- 4.2 Remove the ensemble mean
429  if ( REMOVE_MEAN ) then
430
431    !$OMP PARALLEL DO PRIVATE (levIndex, latIndex, lonIndex)    
432    do levIndex = 1, nkgdim
433      do latIndex = myLatBega, myLatEnda
434        do lonIndex = myLonBega, myLonEnda
435          gdmean(lonIndex, latIndex, levIndex) &
436               = gdmean(lonIndex, latIndex, levIndex) / real(NENS, 8)
437        end do
438      end do
439    end do
440    !$OMP END PARALLEL DO
441
442    !$OMP PARALLEL DO PRIVATE (memberIndex, levIndex, latIndex, lonIndex)    
443    do memberIndex = 1, NENS
444      do levIndex = 1, nkgdim
445        do latIndex = myLatBega, myLatEnda
446          do lonIndex = myLonBega, myLonEnda
447            ensemble_r4(lonIndex, latIndex, levIndex, memberIndex) =  &
448                ensemble_r4(lonIndex, latIndex, levIndex, memberIndex) -  &
449                real(gdmean(lonIndex, latIndex, levIndex), 4)
450          end do
451        end do
452      end do
453    end do
454    !$OMP END PARALLEL DO
455
456  end if
457  
458  !- 4.3 Smooth variances to horizontally constant values
459  if ( smoothVariances ) then
460  
461    allocate(pturb_var(myLonBega:myLonEnda, myLatBega:myLatEnda, stateVectorPert%nk))
462    allocate(avg_pturb_var(stateVectorPert%nk), avg_pturb_var_glb(stateVectorPert%nk))
463    pturb_var(:,:,:) = 0.0D0
464    avg_pturb_var(:) = 0.0D0
465    avg_pturb_var_glb(:) = 0.0D0
466  
467    do memberIndex = 1, NENS  
468      !$OMP PARALLEL DO PRIVATE (levIndex, latIndex, lonIndex)  
469      do levIndex = 1, nkgdim
470        do latIndex = myLatBega, myLatEnda
471          do lonIndex = myLonBega, myLonEnda
472               pturb_var(lonIndex, latIndex, levIndex) = pturb_var(lonIndex, latIndex, levIndex) &
473                   + real(ensemble_r4(lonIndex, latIndex, levIndex, memberIndex), 8)**2
474          end do
475        end do
476      end do
477      !$OMP END PARALLEL DO
478    end do
479  
480    !$OMP PARALLEL DO PRIVATE (levIndex, latIndex, lonIndex)  
481    do levIndex = 1, nkgdim
482      do latIndex = myLatBega, myLatEnda
483        do lonIndex = myLonBega, myLonEnda
484          pturb_var(lonIndex, latIndex, levIndex) &
485               = pturb_var(lonIndex, latIndex, levIndex) / real(NENS, 8)
486        end do
487      end do
488    end do
489    !$OMP END PARALLEL DO
490  
491    do latIndex = myLatBega, myLatEnda
492      do lonIndex = myLonBega, myLonEnda
493        do levIndex = 1, nkgdim
494          avg_pturb_var(levIndex) = avg_pturb_var(levIndex) + pturb_var(lonIndex, latIndex, levIndex)
495        end do
496      end do
497    end do
498  
499    n_grid_point=(lonPerPEa)*(latPerPEa)
500    call rpn_comm_allreduce(n_grid_point, n_grid_point_glb, 1,  &
501                            "mpi_double_precision", "mpi_sum", "GRID", ierr)
502    call rpn_comm_allreduce(avg_pturb_var, avg_pturb_var_glb, nkgdim,  &
503                            "mpi_double_precision", "mpi_sum", "GRID", ierr)
504  
505    !$OMP PARALLEL DO PRIVATE (levIndex, latIndex, lonIndex)  
506    do levIndex = 1, nkgdim
507      do latIndex = myLatBega, myLatEnda
508        do lonIndex = myLonBega, myLonEnda
509          if( pturb_var(lonIndex, latIndex, levIndex) > 0.0d0 ) then
510            pturb_var(lonIndex, latIndex, levIndex) = sqrt( pturb_var(lonIndex, latIndex, levIndex))
511          end if
512        end do
513      end do
514    end do
515    !$OMP END PARALLEL DO
516  
517    do levIndex = 1, nkgdim
518      if( avg_pturb_var_glb(levIndex) > 0.0d0 ) then
519        avg_pturb_var_glb(levIndex) = sqrt( avg_pturb_var_glb(levIndex) / real(n_grid_point_glb, 8) )
520      end if
521    end do
522
523    !$OMP PARALLEL DO PRIVATE (memberIndex, levIndex, latIndex, lonIndex)    
524    do memberIndex = 1, NENS
525      do levIndex = 1, nkgdim
526        if( avg_pturb_var_glb(levIndex) > 0.0d0 ) then
527          do latIndex = myLatBega, myLatEnda
528            do lonIndex = myLonBega, myLonEnda
529              if( pturb_var(lonIndex, latIndex, levIndex) > 0.0d0 ) then
530                ensemble_r4(lonIndex, latIndex, levIndex, memberIndex) =   &
531                    ensemble_r4(lonIndex, latIndex, levIndex, memberIndex) *  &
532                    real(avg_pturb_var_glb(levIndex)/pturb_var(lonIndex, latIndex, levIndex), 4)
533              end if
534            end do
535          end do
536        end if
537      end do
538    end do
539    !$OMP END PARALLEL DO
540
541  end if
542
543  !- 4.4 Read ensemble mean state
544  if ( readEnsMean ) then
545    if (mmpi_myid == 0) write(*,*) 'midas-randomPert: reading ensemble mean state'
546
547    call gsv_allocate(stateVectorEnsMean, 1, hco_target, vco_anl, &
548                      dateStamp_opt=dateStamp, mpi_local_opt=.true., &
549                      allocHeight_opt=.false., allocPressure_opt=.false., &
550                      hInterpolateDegree_opt='LINEAR')
551    call gio_readFromFile(stateVectorEnsMean, ensMeanFileName, ' ', ' ',  &
552                          containsFullField_opt=.true.)
553
554  end if
555
556  !- 4.5 Set perturbations to zero under ice
557  if ( setPertZeroUnderIce ) then
558
559    ! read sea-ice analysis
560    call gsv_allocate(stateVectorIce, 1, hco_anl, vco_anl, dataKind_opt = 4, &
561                      datestamp_opt = -1, mpi_local_opt = .true.,    &
562                      varNames_opt = (/'LG'/), hInterpolateDegree_opt ='LINEAR')
563    call gio_readFromFile(stateVectorIce, './seaIceAnalysis', ' ','A', &
564                          unitConversion_opt=.false., containsFullField_opt=.true.)
565    call gsv_getField(stateVectorIce,seaice_ptr)
566
567    ! set perturbations to zero
568    !$OMP PARALLEL DO PRIVATE (memberIndex, levIndex, latIndex, lonIndex)    
569    do memberIndex = 1, NENS
570      do levIndex = 1, nkgdim
571        do latIndex = myLatBega, myLatEnda
572          do lonIndex = myLonBega, myLonEnda
573            if (seaice_ptr(lonIndex, latIndex, 1) >= iceFractionThreshold) then
574              ensemble_r4(lonIndex, latIndex, levIndex, memberIndex) = 0.0
575            end if
576          end do
577        end do
578      end do
579    end do
580    !$OMP END PARALLEL DO
581
582    call gsv_deallocate(stateVectorIce)
583
584  end if
585
586  !- 4.6 Add a fraction of previous date perturbations
587  if (previousDateFraction > 0.0) then
588
589    previousDateFactPrev  = previousDateFraction
590    ! reduce the amount of random perturbation to maintain steady-state variance
591    previousDateFactNoise = sqrt(1.0d0 - previousDateFraction**2)
592    if (mmpi_myid == 0) then
593      write(*,*) 'midas-randomPert: Scale factor applied to previous date       = ', &
594           previousDateFactPrev
595      write(*,*) 'midas-randomPert: Scale factor applied to random perturbation = ', &
596           previousDateFactNoise
597    end if
598
599    ! determine dateStamp of previous date
600    call incdatr(dateStampPrevious, dateStamp, -tim_windowsize)
601    imode = -3 ! stamp to printable date and time: YYYYMMDD, HHMMSShh
602    ierr    = newdate(dateStampPrevious, datePrint, timePrint, imode)
603    datePrevious =  datePrint*100 + timePrint/1000000
604    write(datePreviousString, '(I10)') datePrevious
605    write(*,*) 'midas-randomPert: previous date, stamp = ', datePrevious, dateStampPrevious
606
607    call gsv_allocate(stateVectorPert, 1, hco_target, vco_anl, &
608                      dateStamp_opt=dateStampPrevious, mpi_local_opt=.true., &
609                      allocHeight_opt=.false., allocPressure_opt=.false., &
610                      hInterpolateDegree_opt='LINEAR')
611    call gsv_getField(stateVectorPert,field)
612
613    deallocate(gdmean)
614    allocate(gdmean(myLonBegt:myLonEndt, myLatBegt:myLatEndt, nkgdim))
615    gdmean(:,:,:) = 0.0D0
616    allocate(ensemblePreviousDate_r4(myLonBegt:myLonEndt, myLatBegt:myLatEndt, nkgdim, nEns))
617
618    do memberIndex = 1, NENS
619
620      ! Read previous date perturbations (or perturbed analyses)
621      write(memberString, '(I4.4)') memberIndex
622      if (readEnsMean) then
623        inFileName = './'//trim(datePreviousString)//'_000_'//trim(memberString)
624      else
625        inFileName = './pert_'//trim(datePreviousString)//'_'//trim(memberString)
626      end if
627      if( mmpi_myid == 0 ) write(*,*) 'midas-randomPert: reading previous date file = ', inFileName
628
629      call gio_readFromFile(stateVectorPert, inFileName, ' ', ' ',  &
630                            containsFullField_opt=.true.)
631
632      ! Copy to big array and accumulate sum for computing mean
633      !$OMP PARALLEL DO PRIVATE (levIndex, latIndex, lonIndex)    
634      do levIndex = 1, nkgdim
635        do latIndex = myLatBegt, myLatEndt
636          do lonIndex = myLonBegt, myLonEndt
637            ensemblePreviousDate_r4(lonIndex, latIndex, levIndex, memberIndex) = &
638                 real(field(lonIndex, latIndex, levIndex), 4)
639            gdmean(lonIndex, latIndex, levIndex) = gdmean(lonIndex, latIndex, levIndex) &
640                                                   + field(lonIndex, latIndex, levIndex)
641          end do
642        end do
643      end do
644      !$OMP END PARALLEL DO
645
646    end do
647
648    ! Finish computing mean and remove it
649    !$OMP PARALLEL DO PRIVATE (levIndex, latIndex, lonIndex)    
650    do levIndex = 1, nkgdim
651      do latIndex = myLatBegt, myLatEndt
652        do lonIndex = myLonBegt, myLonEndt
653          gdmean(lonIndex, latIndex, levIndex) = gdmean(lonIndex, latIndex, levIndex) &
654                                                 / real(NENS, 8)
655        end do
656      end do
657    end do
658    !$OMP END PARALLEL DO
659
660    !$OMP PARALLEL DO PRIVATE (memberIndex, levIndex, latIndex, lonIndex)    
661    do memberIndex = 1, NENS
662      do levIndex = 1, nkgdim
663        do latIndex = myLatBegt, myLatEndt
664          do lonIndex = myLonBegt, myLonEndt
665            ensemblePreviousDate_r4(lonIndex, latIndex, levIndex, memberIndex) =  &
666                ensemblePreviousDate_r4(lonIndex, latIndex, levIndex, memberIndex) -  &
667                real(gdmean(lonIndex, latIndex, levIndex), 4)
668          end do
669        end do
670      end do
671    end do
672    !$OMP END PARALLEL DO
673
674    call gsv_deallocate(stateVectorPert)
675
676  end if
677
678  !- 4.7 Write the perturbations
679  do memberIndex = 1, NENS
680    if( mmpi_myid == 0 ) write(*,*)
681    if( mmpi_myid == 0 ) write(*,*) 'midas-randomPert: pre-processing for writing member number= ', memberIndex
682
683    call gsv_allocate(stateVectorPert, 1, hco_anl, vco_anl, &
684                      dateStamp_opt=dateStamp, mpi_local_opt=.true., &
685                      allocHeight_opt=.false., allocPressure_opt=.false., &
686                      hInterpolateDegree_opt='LINEAR')
687    call gsv_getField(stateVectorPert,field)
688    call gsv_allocate(stateVectorPertInterp, 1, hco_target, vco_anl, &
689                      dateStamp_opt=dateStamp, mpi_local_opt=.true., &
690                      allocHeight_opt=.false., allocPressure_opt=.false., &
691                      hInterpolateDegree_opt='LINEAR')
692
693    ! Copy mask if it exists
694    call gsv_copyMask(stateVectorEnsMean, stateVectorPertInterp)
695
696    !$OMP PARALLEL DO PRIVATE (levIndex, latIndex, lonIndex)    
697    do levIndex = 1, nkgdim
698      do latIndex = myLatBega, myLatEnda
699        do lonIndex = myLonBega, myLonEnda
700          field(lonIndex, latIndex, levIndex) = ensemble_r4(lonIndex, latIndex, levIndex, memberIndex)
701        end do
702      end do
703    end do
704    !$OMP END PARALLEL DO
705
706    ! interpolate perturbation to the target grid
707    call int_interp_gsv(stateVectorPert, stateVectorPertInterp)
708
709    call gsv_getField(stateVectorPertInterp,fieldInterp)
710
711    ! Average current and previous perturbations
712    if (previousDateFraction > 0.0) then
713      !$OMP PARALLEL DO PRIVATE (levIndex, latIndex, lonIndex)    
714      do levIndex = 1, nkgdim
715        do latIndex = myLatBegt, myLatEndt
716          do lonIndex = myLonBegt, myLonEndt
717            fieldInterp(lonIndex, latIndex, levIndex) =  &
718                 previousDateFactPrev * &
719                 ensemblePreviousDate_r4(lonIndex, latIndex, levIndex, memberIndex) + &
720                 previousDateFactNoise * &
721                 fieldInterp(lonIndex, latIndex, levIndex)
722          end do
723        end do
724      end do
725      !$OMP END PARALLEL DO
726    end if
727
728    ! add perturbation to supplied ensemble mean
729    if ( readEnsMean ) then
730      if (mmpi_myid == 0) write(*,*) 'midas-randomPert: adding the ensemble mean and perturbation'
731      call gsv_add(stateVectorEnsMean, stateVectorPertInterp)
732    end if
733
734    ! determine file name and write to file
735    write(memberString, '(I4.4)') memberIndex
736    if (readEnsMean) then
737      outFileName = './'//trim(dateString)//'_000_'//trim(memberString)
738    else
739      outFileName = './pert_'//trim(dateString)//'_'//trim(memberString)
740    end if
741
742    if( mmpi_myid == 0 ) write(*,*) 'midas-randomPert: processing file = ', outFileName
743    stateVectorPertInterp%etiket = 'UNDEFINED'
744    call gio_writeToFile(stateVectorPertInterp, outFileName, out_etiket,              & ! IN
745                         numBits_opt=numBits, unitConversion_opt=.true., &  ! IN
746                         containsFullField_opt=readEnsMean, typvar_opt=typvarOut) 
747
748    call gsv_deallocate(stateVectorPertInterp)
749    call gsv_deallocate(stateVectorPert)
750
751  end do
752
753  ! Write out ensemble mean state, if it exists
754  if (readEnsMean) then
755    ! determine file name and write to file
756    memberIndex = 0
757    write(memberString, '(I4.4)') memberIndex
758    outFileName = './'//trim(dateString)//'_000_'//trim(memberString)
759
760    if( mmpi_myid == 0 ) write(*,*) 'midas-randomPert: processing file = ', outFileName
761    stateVectorEnsMean%etiket = 'UNDEFINED'
762    call gio_writeToFile(stateVectorEnsMean, outFileName, out_etiket,              & ! IN
763                         numBits_opt=numBits, unitConversion_opt=.true., &  ! IN
764                         containsFullField_opt=.true., typvar_opt=typvarOut)
765  end if
766
767  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
768
769  !
770  !- 5.  Memory deallocations
771  !
772  deallocate(gdmean)
773  deallocate(ensemble_r4)
774  if (previousDateFraction > 0.0) then
775    deallocate(ensemblePreviousDate_r4)
776  end if
777  deallocate(controlVector)  
778  deallocate(controlVector_mpiglobal)  
779
780  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
781  call utl_tmg_stop(0)
782
783  !
784  !- 6.  MPI, tmg finalize
785  !  
786  call tmg_terminate(mmpi_myid, 'TMG_INFO')
787  call rpn_comm_finalize(ierr) 
788
789  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
790
791  !
792  !- 7.  Ending
793  !
794  if( mmpi_myid == 0 ) write(*,*) '---------------------------------'
795  if( mmpi_myid == 0 ) write(*,*) ' MIDAS-RANDOMPERT ENDS'
796  if( mmpi_myid == 0 ) write(*,*) '---------------------------------'
797
798end program midas_randomPert
