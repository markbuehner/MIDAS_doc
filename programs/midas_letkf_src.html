 17program midas_letkf
 18  !
 19  !:Purpose: Main program for the local ensemble transform Kalman filter
 20  !          (LETKF). Several different variations of the LETKF algorithm have
 21  !          been implemented. Note that the actual calculation of the analyses
 22  !          is in the subroutine ``enkf_LETKFanalyses``. Many aspects of this
 23  !          program are controlled through the namelist block NAMLETKF.
 24  !
 25  !          ---
 26  !
 27  !:Algorithm: The ``letkf`` programs implements several variations of the LETKF
 28  !            ensemble data assimilation algorithm. The following variations
 29  !            can be chosen through the namelist:
 30  !
 31  !              - **LETKF**: standard LETKF
 32  !
 33  !              - **CVLETKF**: LETKF with deterministic approach to cross validation using
 34  !                the "gain form" (currently operational)
 35  !
 36  !              - **CVLETKF-PERTOBS**: LETKF with stochastic approach to cross validation
 37  !
 38  !              - **LETKF-Gain**: standard LETKF, but using the "gain form"
 39  !
 40  !              - **LETKF-Gain-ME**: standard LETKF, but using the "modulated ensemble" and "gain
 41  !                form" to account for vertical localization
 42  !
 43  !              - **CVLETKF-ME**: LETKF with deterministic approach to cross validation using
 44  !                the "modulated ensemble" and "gain form" to account for
 45  !                vertical localization
 46  !
 47  !            More detail on the LETKF algorithms with and without cross
 48  !            validation (deterministic and stochastic) can be found in the
 49  !            paper:
 50  !            `LETKF with cross validation <https://journals.ametsoc.org/view/journals/mwre/148/6/MWR-D-19-0402.1.xml>`_
 51  !
 52  !            --
 53  !
 54  !============================================== ==============================================================
 55  ! Input and Output Files (NWP applicaton)        Description of file
 56  !============================================== ==============================================================
 57  ! ``flnml``                                      In - Main namelist file with parameters user may modify
 58  ! ``flnml_static``                               In - The "static" namelist that should not be modified
 59  ! ``ensemble/$YYYYMMDDHH_006_$NNNN``             In - Background ensemble member files
 60  ! ``obsfiles_$FAM/obs$FAM_$NNNN_$NNNN``          In - Observation file for each "family" and MPI task
 61  ! ``obserr``                                     In - Observation error statistics
 62  ! ``$YYYYMMDDHH_000_$NNNN``                      Out - Analysis ensemble member files
 63  ! ``obsfiles_$FAM.updated/obs$FAM_$NNNN_$NNNN``  Out - Updated obs file for each "family" and MPI task
 64  ! Remainder are files related to radiance obs:
 65  ! ``stats_$SENSOR_assim``                        In - Satellite radiance observation errors of difference sensors
 66  ! ``stats_tovs``                                 In - Satellite radiance observation errors
 67  ! ``stats_tovs_symmetricObsErr``                 In - User-defined symmetric TOVS errors for all sky
 68  ! ``Cmat_$PLATFORM_$SENSOR.dat``                 In - Inter-channel observation-error correlations
 69  ! ``ceres_global.std``                           In - High-res surface type and water fraction for radiance obs
 70  ! ``rtcoef_$PLATFORM_$SENSOR.dat``               In - RTTOV coefficient files
 71  ! ``rttov_h2o_limits.dat``                       In - Min/max humidity limits applied to analysis
 72  ! ``ozoneclim98``                                In - Ozone climatology
 73  !============================================== ==============================================================
 74  !
 75  !           --
 76  !
 77  !:Synopsis: Below is a summary of the ``letkf`` program calling sequence:
 78  !
 79  !             - **Initial setups:**
 80  !
 81  !               - Read the NAMLETKF namelist and check/modify some values.
 82  !
 83  !               - Various modules are setup: ``obsFiles_mod``,
 84  !                 ``gridStateVector_mod``, ``timeCoord_mod`` (and set up dates
 85  !                 and ``dateStampList`` variables for both trials and
 86  !                 increments/analyses).
 87  !
 88  !               - Setup horizontal and vertical grid objects from first
 89  !                 ensemble member file and determine if this is an NWP or
 90  !                 ocean application.
 91  !
 92  !               - Setup ``obsSpaceData`` object and read observations from
 93  !                 files: ``inn_setupObs``.
 94  !
 95  !               - Setup the observation error statistics in ``obsSpaceData``
 96  !                 object: ``oer_setObsErrors``.
 97  !
 98  !               - Allocate and some setup of objects for
 99  !                 ``ensembleObservations_mod``.
100  !
101  !               - Allocate objects for ``column_mod`` and
102  !                 ``gridStateVector_mod``.
103  !
104  !               - Allocate ensemble object and read trial ensemble:
105  !                 ``ens_readEnsemble``.
106  !
107  !               - Optionally, read a deterministic state for recentering the
108  !                 trial ensemble before the analysis (for special applications
109  !                 using LETKF for deterministic analysis).
110  !
111  !               - Compute ensemble mean: ``ens_computeMean``.
112  !
113  !             - **LETKF computations:**
114  !
115  !               - Loop over trial members, computing innovation for each, with
116  !                 resulting :math:`H(xb)` being stored in ``ensObs`` objects both for
117  !                 original ensemble and, optionally, for the modulated
118  !                 ensemble members.
119  !
120  !               - Compute some additional quantities in ``ensObs`` objects based
121  !                 on :math:`H(xb)` values.
122  !
123  !               - Compute :math:`y-H(xb)` for trial ensemble mean.
124  !
125  !               - Additional observation quality control procedures based on
126  !                 quantities computed from trial ensemble.
127  !
128  !               - Gather ``ensObs`` quantities from all MPI tasks onto all MPI
129  !                 tasks.
130  !
131  !               - Allocate and prepare objects for ``ensembleStateVector_mod``
132  !                 used to store the trial and analysis ensembles with
133  !                 temporal resolution of analysis (can be same as temporal
134  !                 resolution for innovation calculation or only a single time
135  !                 step).
136  !
137  !               - Setup information for interpolating weights from coarse grid
138  !                 to the full model grid: ``enkf_setupInterpInfo``.
139  !
140  !               - Perform LETKF analysis to compute the analysis ensemble:
141  !                 ``enkf_LETKFanalyses``
142  !
143  !             - **Final steps:**
144  !
145  !               - Optionally, compute :math:`H(xa)` for each member and put values in
146  !                 ``ensObs`` object for later output to diag SQLite files.
147  !
148  !               - Write out (update) the observation files: ``obsf_writeFiles``.
149  !
150  !               - Compute :math:`y-H(xa)` for ensemble mean analysis, stored in
151  !                 ``obsSpaceData``.
152  !
153  !               - Optionally, do post-processing of the analysis ensemble (same
154  !                 functionality as the ``ensPostProcess`` program), or just
155  !                 write out the raw analysis ensemble (for later processing by
156  !                 ``ensPostProcess``).
157  !
158  !           --
159  !
160  !:Options: `List of namelist blocks <../namelists_in_each_program.html#letkf>`_
161  !          that can affect the ``letkf`` program.
162  !
163  !          * Some of the other relevant namelist blocks used to configure the
164  !            letkf analysis are listed in the following table:
165  ! 
166  !======================== ============== ==============================================================
167  ! Program/Module           Namelist       Description of what is controlled
168  !======================== ============== ==============================================================
169  ! ``midas_letkf``          ``NAMLETKF``   LETKF algorithm, number of ensemble members and
170  !                                         additional parameters for controlling the LETKF analysis
171  ! ``timeCoord_mod``        ``NAMTIME``    assimilation time window length, temporal resolution of
172  !                                         the background state and the analysis
173  !======================== ============== ==============================================================
174  !

500                           containsFullField_opt=.true., readHeightSfc_opt=.true. )
501  end if
502
503  !- 2.8 Allocate statevector related to ensemble mean
504  call gsv_allocate( stateVectorMeanTrl4D, tim_nstepobs, hco_ens, vco_ens, &
505                     dateStamp_opt=tim_getDateStamp(),  &
506                     mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
507                     dataKind_opt=4, allocHeightSfc_opt=.true., &
508                     allocHeight_opt=.false., allocPressure_opt=.false. )
509  call gsv_zero(stateVectorMeanTrl4D)
510  call gsv_allocate( stateVectorMeanAnl, tim_nstepobsinc, hco_ens, vco_ens, &
511                     dateStamp_opt=tim_getDateStamp(),  &
512                     mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
513                     dataKind_opt=4, allocHeightSfc_opt=.true., &
514                     allocHeight_opt=.false., allocPressure_opt=.false. )
515  call gsv_zero(stateVectorMeanAnl)
516  call gsv_allocate( stateVector4D, tim_nstepobs, hco_ens, vco_ens, &
517                     dateStamp_opt=tim_getDateStamp(),  &
518                     mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
519                     dataKind_opt=4, allocHeightSfc_opt=.true., &
520                     allocHeight_opt=.false., allocPressure_opt=.false. )
521  call gsv_zero(stateVector4D)
522  if ( useModulatedEns ) then
523    ! same as stateVector4D
524    call gsv_allocate( stateVector4Dmod, tim_nstepobs, hco_ens, vco_ens, &
525                       dateStamp_opt=tim_getDateStamp(),  &
526                       mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
527                       dataKind_opt=4, allocHeightSfc_opt=.true., &
528                       allocHeight_opt=.false., allocPressure_opt=.false. )
529    call gsv_zero(stateVector4Dmod)
530  end if
531
532  !- 2.9 Allocate statevector related to an analysis ensemble member  
533  call gsv_allocate( stateVectorMemberAnl, tim_nstepobsinc, hco_ens, vco_ens, &
534                     dateStamp_opt=tim_getDateStamp(),  &
535                     mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
536                     dataKind_opt=4, allocHeightSfc_opt=.true. )
537  call gsv_zero(stateVectorMemberAnl)
538  
539  !- 2.10 Allocate statevector for storing state with heights and pressures allocated (for s2c_nl)
540  call gsv_allocate( stateVectorWithZandP4D, tim_nstepobs, hco_ens, vco_ens, &
541                     dateStamp_opt=tim_getDateStamp(),  &
542                     mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
543                     dataKind_opt=4, allocHeightSfc_opt=.true. )
544  call gsv_zero(stateVectorWithZandP4D)
545
546  !- 2.11 Allocate ensembles, read the Trl ensemble
547  call utl_tmg_start(2,'--ReadEnsemble')
548  call ens_allocate(ensembleTrl4D, nEns, tim_nstepobs, hco_ens, vco_ens, dateStampList)
549  call ens_readEnsemble(ensembleTrl4D, ensPathName, biPeriodic=.false., &
550                        ignoreDate_opt=ignoreEnsDate)
551  call utl_tmg_stop(2)
552
553  !- 2.12 If desired, read a deterministic state for recentering the ensemble
554  if (recenterInputEns) then
555    call gsv_allocate( stateVectorRecenter, tim_nstepobs, hco_ens, vco_ens, &
556                       dateStamp_opt=tim_getDateStamp(),  &
557                       mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
558                       dataKind_opt=4, allocHeightSfc_opt=.false., &
559                       allocHeight_opt=.false., allocPressure_opt=.false. )
560    call gsv_zero(stateVectorRecenter)
561    call fln_ensTrlFileName( recenterFileName, './', tim_getDateStamp() )
562    do stepIndex = 1, tim_nstepobs
563      call gio_readFromFile( stateVectorRecenter, recenterFileName, ' ', ' ',  &
564                             stepIndex_opt=stepIndex, containsFullField_opt=.true., &
565                             readHeightSfc_opt=.false. )
566    end do
567    call ens_recenter( ensembleTrl4D, stateVectorRecenter, recenteringCoeffScalar_opt=1.0d0 )
568    call gsv_deallocate( stateVectorRecenter )
569  end if
570  
571  !- 2.13 Compute ensemble mean and copy to meanTrl and meanAnl stateVectors
572  call ens_computeMean(ensembleTrl4D)
573  call ens_copyEnsMean(ensembleTrl4D, stateVectorMeanTrl4D)
574  if (tim_nstepobsinc < tim_nstepobs) then
575    call gsv_copy4Dto3D(stateVectorMeanTrl4D, stateVectorMeanAnl)
576  else
577    call gsv_copy(stateVectorMeanTrl4D, stateVectorMeanAnl)
578  end if
579  
580  !
581  !- 3. Compute HX values with results in ensObs/ensObsGain
582  !
583
584  !- 3.1 Loop over all members and compute HX for each
585  if ( readEnsObsFromFile ) then
586    call eob_readFromFiles(ensObs, nEns, inputFilenamePrefix='eob_HX', readObsInfo=.true.)
587    if ( useModulatedEns ) then
588      call enkf_setupModulationFactor(stateVectorMeanTrl4D%vco, numRetainedEigen, nEns, vLocalize, &
589                                      beSilent=.true.)
590
591      ! refresh assimilation flag before reading the files
592      call eob_setAssFlag(ensObsGain)
593      call eob_readFromFiles(ensObsGain, nEnsGain, inputFilenamePrefix='eobGain_HX', readObsInfo=.false.)
594    end if
595  else
596    do memberIndex = 1, nEns
597  
598      write(*,*) ''
599      write(*,*) 'midas-letkf: apply nonlinear H to ensemble member ', memberIndex
600      write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
601
602      ! copy 1 member to a stateVector
603      call ens_copyMember(ensembleTrl4D, stateVector4D, memberIndex)
604      call gsv_copy(stateVector4D, stateVectorWithZandP4D, allowVarMismatch_opt=.true., &
605                    beSilent_opt=.true.)
606
607      ! copy the surface height field
608      if (nwpFields) then
609        call gsv_copyHeightSfc(stateVectorHeightSfc, stateVectorWithZandP4D)
610      end if
611
612      ! Compute and set Yb in ensObs
613      call s2c_nl( stateVectorWithZandP4D, obsSpaceData, column, hco_ens, &
614                  timeInterpType=obsTimeInterpType, dealloc_opt=.false., &
615                  beSilent_opt=.true. )
616
617      ! Compute Y-H(X) in OBS_OMP
618      call inn_computeInnovation(column, obsSpaceData, beSilent_opt=.true.)
619
620      ! Copy to ensObs: Y-HX for this member
621      call eob_setYb(ensObs, memberIndex)
622
623      ! Compute and set Yb in ensObsGain
624      do eigenVectorIndex = 1, numRetainedEigen
625        if ( mmpi_myid == 0 .and. debug ) then
626          write(*,*) 'midas-letkf: apply nonlinear H to modulated member ', &
627                    eigenVectorIndex, '/', numRetainedEigen
628        end if
629
630        ! modulate the member with eigenvectors of vertical localization matrix
631        call enkf_getModulatedState( stateVector4D, stateVectorMeanTrl4D, &
632                                    vLocalize, numRetainedEigen, nEns, &
633                                    eigenVectorIndex, stateVector4Dmod, &
634                                    beSilent=.true. )
635        if ( debug ) then
636          call gsv_getField(stateVector4Dmod,field_Psfc,'P0')
637          write(*,*) 'midas-letkf: max(Psfc)=', maxval(field_Psfc), &
638                    ', min(Psfc)=', minval(field_Psfc)
639        end if
640
641        call gsv_copy(stateVector4Dmod, stateVectorWithZandP4D, allowVarMismatch_opt=.true., &
642                      beSilent_opt=.true.)
643        if (nwpFields) then
644          call gsv_copyHeightSfc(stateVectorHeightSfc, stateVectorWithZandP4D)
645        end if
646        call s2c_nl( stateVectorWithZandP4D, obsSpaceData, column, hco_ens, &
647                    timeInterpType=obsTimeInterpType, dealloc_opt=.false., &
648                    beSilent_opt=.true. )
649
650        ! Compute Y-H(X) in OBS_OMP
651        call inn_computeInnovation( column, obsSpaceData, filterObsAndInitOer_opt=.false., &
652                                    beSilent_opt=.true. )
653
654        ! Copy to ensObsGain: Y-HX for this member
655        memberIndexInEnsObs = (eigenVectorIndex - 1) * nEns + memberIndex
656        call eob_setYb(ensObsGain, memberIndexInEnsObs)
657      end do ! eigenVectorIndex
658
659    end do ! memberIndex
660  end if
661  if ( gsv_isAllocated(stateVector4Dmod) ) call gsv_deallocate(stateVector4Dmod)
662
663  ! write local ensObs to file
664  if (writeLocalEnsObsToFile) then
665    call eob_writeToFiles(ensObs, outputFilenamePrefix='eob_HX', writeObsInfo=.true.)
666    if (useModulatedEns) then
667      call eob_writeToFiles(ensObsGain, outputFilenamePrefix='eobGain_HX', writeObsInfo=.false., &
668                            numGroupsToDivideMembers_opt=numRetainedEigen, &
669                            maxNumMembersPerGroup_opt=nEns)
670    end if
671  end if
672
673  !- 3.2 Set some additional information in ensObs/ensObsGain and additional quality
674  !      control before finally communicating ensObs/ensObsGain globally
675
676  ! Compute and remove the mean of Yb
677  call eob_calcAndRemoveMeanYb(ensObs)
678  if ( useModulatedEns ) call eob_calcAndRemoveMeanYb(ensObsGain)
679
680  ! Put HPHT in OBS_HPHT, for writing to obs files
681  call eob_setHPHT(ensObs)
682  if ( useModulatedEns ) call eob_setHPHT(ensObsGain)
683
684  ! Compute random observation perturbations
685  if (trim(algorithm) == 'CVLETKF-PERTOBS') then
686    randomSeedObs = 1 + mmpi_myid
687    call eob_calcRandPert(ensObs, randomSeedObs)
688  end if
689
690  ! Apply obs operators to ensemble mean background for several purposes
691  write(*,*) ''
692  write(*,*) 'midas-letkf: apply nonlinear H to ensemble mean background'
693  write(*,*) ''
694  call gsv_copy(stateVectorMeanTrl4D, stateVectorWithZandP4D, allowVarMismatch_opt=.true.)
695  if (nwpFields) then
696    call gsv_copyHeightSfc(stateVectorHeightSfc, stateVectorWithZandP4D)
697  end if
698  call s2c_nl( stateVectorWithZandP4D, obsSpaceData, column, hco_ens, &
699               timeInterpType=obsTimeInterpType, dealloc_opt=.false. )
700  call tvs_allocTransmission(col_getNumLev(column,'TH')) ! this will cause radiative transmission profiles to be stored for use in eob_setVertLocation
701  call inn_computeInnovation(column, obsSpaceData, beSilent_opt=.false.)
702
703  ! Put y-mean(H(X)) in OBS_OMP for writing to obs files (overwrites y-H(mean(X)))
704  call eob_setMeanOMP(ensObs)
705  if ( useModulatedEns ) call eob_setMeanOMP(ensObsGain)
706
707  ! If doing observation simulation experiment, set y for family of interest to mean(H(x))
708  if ( edaObsImpact ) then
709    call eob_setSimulatedObs(ensObs, edaObsFamily)
710  end if
711
712  ! Set vertical location for all obs for vertical localization (based on ensemble mean pressure and height)
713  if (vLocalize > 0.0d0) then
714    if (nwpFields) then
715      call eob_setTypeVertCoord(ensObs,'logPressure')
716      if ( useModulatedEns ) call eob_setTypeVertCoord(ensObsGain,'logPressure')
717    else if (oceanFields) then
718      call eob_setTypeVertCoord(ensObs,'depth')
719      if ( useModulatedEns ) call eob_setTypeVertCoord(ensObsGain,'depth')
720    end if
721    call eob_setVertLocation(ensObs, column)
722    if ( useModulatedEns ) call eob_setVertLocation(ensObsGain, column)
723  end if
724
725  ! Modify the obs error stddev for AMSUB in the tropics
726  if (modifyAmsubObsError) call enkf_modifyAmsubObsError(obsSpaceData)
727
728  ! Apply a background check (reject limit is set in the routine)
729  if (backgroundCheck) call eob_backgroundCheck(ensObs)
730
731  ! For ice/ocean DA: remove obs that are too close to land
732  if (minDistanceToLand > 0.0D0) then
733    call ens_getMask(ensembleTrl4D,oceanMask)
734    call eob_removeObsNearLand(ensObs, oceanMask, minDistanceToLand)
735  end if
736
737  ! Set values of obs_sigi and obs_sigo before hubernorm modifies obs_oer
738  call eob_setSigiSigo(ensObs)
739
740  ! Apply huber norm quality control procedure (modifies obs_oer)
741  if (huberize) call eob_huberNorm(ensObs)
742
743  !- Reject all IR radiance observation in arctic and antarctic (.i.e |lat|>60. )
744  if (rejectHighLatIR) call enkf_rejectHighLatIR(obsSpaceData)
745
746  ! Reject radiance observations too close to the surface
747  if (rejectRadNearSfc) call eob_rejectRadNearSfc(ensObs)
748
749  ! Compute inverse of obs error variance (done here to use dynamic GPS-RO, GB-GPS based on mean O-P)
750  call eob_setObsErrInv(ensObs)
751  if ( useModulatedEns ) call eob_setObsErrInv(ensObsGain)
752
753  call utl_tmg_start(141,'----Barr')
754  call rpn_comm_barrier('GRID',ierr)
755  call utl_tmg_stop(141)
756
757  ! For EDA-based observation simulation experiments, calculate an inverse of obs error
758  ! variance with variance values inflated for the simulated observations
759  if ( edaObsImpact ) then
760    call eob_setSimObsErrInv(ensObs, edaObsFamily)
761    if ( useModulatedEns ) call eob_setSimObsErrInv(ensObsGain, edaObsFamily)
762  end if
763
764  ! Clean and globally communicate obs-related data to all mpi tasks
765  call eob_allGather(ensObs,ensObs_mpiglobal)
766  if ( useModulatedEns ) then
767    allocate(ensObsGain_mpiglobal)
768    call eob_allGather(ensObsGain,ensObsGain_mpiglobal)
769  else
770    ensObsGain_mpiglobal => ensObs_mpiglobal
771  end if
772
773  ! Print number of assimilated obs per family to the listing
774  write(*,*) 'oti_timeBinning: After extra filtering done in midas-letkf'
775  call oti_timeBinning(obsSpaceData,tim_nstepobs)
776
777  !
778  !- 4. Final preparations for computing analyses
779  !
780
781  !- 4.1 Copy trial ensemble to nstepobsinc time steps
782  if (tim_nstepobsinc < tim_nstepobs) then
783    allocate(ensembleTrl)
784    call ens_allocate(ensembleTrl, nEns, tim_nstepobsinc, hco_ens, vco_ens, dateStampListInc)
785    call ens_copy4Dto3D(ensembleTrl4D,ensembleTrl)
786    call ens_deallocate(ensembleTrl4D)
787  else
788    ! number of timesteps is the same, so just point to it
789    ensembleTrl => ensembleTrl4D
790  end if
791
792  !- 4.2 Copy trl ensemble to anl ensemble
793  call ens_allocate(ensembleAnl, nEns, tim_nstepobsinc, hco_ens, vco_ens, dateStampListInc)
794  call ens_copy(ensembleTrl,ensembleAnl)
795
796  !- 4.3 Setup for interpolating weights from coarse to full resolution
797  call enkf_setupInterpInfo(wInterpInfo, stateVectorMeanAnl%hco, weightLatLonStep,  &
798                            stateVectorMeanAnl%myLonBeg, stateVectorMeanAnl%myLonEnd,  &
799                            stateVectorMeanAnl%myLatBeg, stateVectorMeanAnl%myLatEnd)
800
801  !
802  !- 5. Main calculation of ensemble analyses
803  !
804  
805  !- 5.1 Call to perform LETKF
806  call enkf_LETKFanalyses(algorithm, numSubEns, randomShuffleSubEns,  &
807                          ensembleAnl, ensembleTrl, &
808                          ensObs_mpiglobal, ensObsGain_mpiglobal, &
809                          stateVectorMeanAnl, &
810                          wInterpInfo, maxNumLocalObs,  &
811                          hLocalize, hLocalizePressure, vLocalize, &
812                          mpiDistribution, numRetainedEigen, edaObsImpact)
813
814  !- 5.2 Loop over all analysis members and compute H(Xa_member) (if output is desired) 
815  if ( outputEnsObs ) then
816  
817    do memberIndex = 1, nEns
818
819      write(*,*) ''
820      write(*,*) 'midas-letkf: apply nonlinear H to analysis ensemble member ', memberIndex
821      write(*,*) ''
822
823      ! copy 1 member to a stateVector
824      call ens_copyMember(ensembleAnl, stateVectorMemberAnl, memberIndex)
825      
826      if (tim_nstepobsinc < tim_nstepobs) then
827        ! ensembleAnl is only 3D, so need to make 4D for s2c_nl
828        middleStepIndex = (tim_nstepobs + 1) / 2
829        call gsv_copy(stateVectorMemberAnl, stateVectorWithZandP4D, allowVarMismatch_opt=.true., stepIndexOut_opt=middleStepIndex)
830        call gsv_3dto4d(stateVectorWithZandP4D)
831      else
832        call gsv_copy(stateVectorMemberAnl, stateVectorWithZandP4D, allowVarMismatch_opt=.true.)
833      end if
834      
835      ! copy the surface height field
836      if (nwpFields) then
837        call gsv_copyHeightSfc(stateVectorHeightSfc, stateVectorWithZandP4D)
838      end if  
839      
840      call s2c_nl( stateVectorWithZandP4D, obsSpaceData, column, hco_ens, &
841                   timeInterpType=obsTimeInterpType, dealloc_opt=.false. )
842
843      ! Compute Y-H(Xa) in OBS_OMAM (used instead of OBS_OMA so that obsSpaceData isn't unintentionally modified) 
844      call inn_computeInnovation(column, obsSpaceData, destObsColumn_opt=OBS_OMAM, beSilent_opt=.true., &
845                                 callFiltTopo_opt=.false., callSetErrGpsgb_opt=.false., analysisMode_opt=.false.)
846
847      ! Copy to ensObs: Y-H(Xa_member) for this member
848      call eob_setYa(ensObs, memberIndex, OBS_OMAM)
849
850    end do
851  
852  end if
853  
854  !- 6. Output obs files with mean OMP and (unrecentered) OMA
855
856  ! Compute Y-H(Xa_mean) in OBS_OMA
857  write(*,*) ''
858  write(*,*) 'midas-letkf: apply nonlinear H to ensemble mean analysis'
859  write(*,*) ''
860  if (tim_nstepobsinc < tim_nstepobs) then
861    ! meanAnl is only 3D, so need to make 4D for s2c_nl
862    middleStepIndex = (tim_nstepobs + 1) / 2
863    call gsv_copy(stateVectorMeanAnl, stateVectorWithZandP4D, allowVarMismatch_opt=.true., stepIndexOut_opt=middleStepIndex)
864    call gsv_3dto4d(stateVectorWithZandP4D)
865  else
866    call gsv_copy(stateVectorMeanAnl, stateVectorWithZandP4D, allowVarMismatch_opt=.true.)
867  end if
868  if (nwpFields) then
869    call gsv_copyHeightSfc(stateVectorHeightSfc, stateVectorWithZandP4D)
870  end if
871  call s2c_nl( stateVectorWithZandP4D, obsSpaceData, column, hco_ens, &
872               timeInterpType=obsTimeInterpType )
873  call inn_computeInnovation(column, obsSpaceData, destObsColumn_opt=OBS_OMA, beSilent_opt=.false.)
874  
875  ! Write (update) observation files. 
876  if (outputEnsObs) then
877    call obsf_writeFiles( obsSpaceData, ensObs_opt=ensObs)
878  else
879    call obsf_writeFiles( obsSpaceData)
880  end if
881  
882  !- 7. Post processing of the analysis results (if desired) and write everything to files
883  if (ensPostProcessing) then
884    !- Allocate and read the Trl control member (used to compute control member increment for IAU)
885    call gsv_allocate( stateVectorCtrlTrl, tim_nstepobsinc, hco_ens, vco_ens, &
886                       dateStamp_opt=tim_getDateStamp(),  &
887                       mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
888                       dataKind_opt=4, allocHeightSfc_opt=.true., &
889                       allocHeight_opt=.false., allocPressure_opt=.false. )
890    if (recenterInputEns) then
891      !- Use the deterministic trial, if we are recentering the input ensemble
892      ctrlFileName = trim(recenterFileName)
893    else
894      !- Otherwise, use member 0000
895      call fln_ensFileName(ctrlFileName, ensPathName, memberIndex_opt=0, &
896                           copyToRamDisk_opt=.false.)
897    end if
898    do stepIndex = 1, tim_nstepobsinc
899      call gio_readFromFile( stateVectorCtrlTrl, ctrlFileName, ' ', ' ',  &
900                             stepIndex_opt=stepIndex, containsFullField_opt=.true., &
901                             readHeightSfc_opt=.false. )
902    end do
903
904    call epp_postProcess(ensembleTrl, ensembleAnl, stateVectorHeightSfc, stateVectorCtrlTrl, &
905                         writeTrlEnsemble=.false., outputOnlyEnsMean_opt=outputOnlyEnsMean)
906  else
907    !- Just write the raw analysis ensemble to files
908    if (mmpi_myid == 0) then
909      write(*,*) 'midas-letkf: No ensemble post-processing requested, so just write the raw analysis ensemble'
910    end if
911    call utl_tmg_start(3,'--WriteEnsemble')
912    if (.not. outputOnlyEnsMean) then
913      call ens_writeEnsemble(ensembleAnl, '.', '', etiket_anl, 'A',  &
914                             numBits_opt=16, etiketAppendMemberNumber_opt=.true.,  &
915                             containsFullField_opt=.true.)
916    end if
917    call utl_tmg_stop(3)
918
919  end if
920
921  !
922  !- 8. MPI, tmg finalize
923  !  
924  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
925  call utl_tmg_stop(0)
926
927  call tmg_terminate(mmpi_myid, 'TMG_INFO')
928  call rpn_comm_finalize(ierr) 
929
930  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
931
932  if ( mmpi_myid == 0 ) write(*,*) ' --------------------------------'
933  if ( mmpi_myid == 0 ) write(*,*) ' MIDAS-LETKF ENDS'
934  if ( mmpi_myid == 0 ) write(*,*) ' --------------------------------'
935
936end program midas_letkf
