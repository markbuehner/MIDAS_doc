 17program midas_analysisErrorOI
 18  !
 19  !:Purpose: Calculate analysis-error standard deviation given
 20  !          new assimilated observations. It only works for sea ice variables and
 21  !          uses a simple OI approach.
 22  !
 23  !          ---
 24  !
 25  !:Algorithm: The Optimal Interpolation (OI) is a data assimilation approach
 26  !            where both the state and its estimated error are computed using
 27  !            observations while taking into account the specified 
 28  !            uncertainties for both the observations and the background
 29  !            state (i.e. the R and B covariance matrices, respectively).
 30  !            The computations are done independently at each analysis grid
 31  !            point with only local observations, those that have a 
 32  !            significant influence on the analysis at the grid point
 33  !            location. Here the code only implements the calculation to
 34  !            update the diagonal of the B covariance matrix.
 35  !
 36  !            --
 37  !
 38  !============================================== ==============================================================
 39  ! Input and Output Files                         Description of file
 40  !============================================== ==============================================================
 41  ! ``flnml``                                      In - Main namelist file with parameters user may modify
 42  ! ``trlm_01``                                    In - Background-error standard deviation
 43  ! ``analysisgrid``                               In - File defining grid for computing the analysis error
 44  ! ``sea_ice_obs-err``                            In - Observation error statistics
 45  ! ``bgstddev``                                   In - Static background-error statistics
 46  ! ``bgSeaIceConc``                               In - Background sea ice concentration
 47  ! ``obsfiles_$FAM/obs$FAM_0001_0001``            In - Observation file for each "family" (only 1 MPI task)
 48  ! ``anlm_000m``                                  Out - Analysis-error on the analysis grid
 49  ! ``obsfiles_$FAM.updated/obs$FAM_0001_0001``    Out - Updated obs file for each "family" (only 1 MPI task)
 50  !============================================== ==============================================================
 51  !
 52  !           --
 53  !
 54  !:Synopsis: Below is a summary of the ``analysisErrorOI`` program calling sequence:
 55  !
 56  !             - **Initial setups:**
 57  !
 58  !               - Setup horizontal and vertical grid objects for "analysis
 59  !                 grid" from ``trlm_01`` file.
 60  !
 61  !               - Setup ``obsSpaceData`` object and read observations from
 62  !                 files: ``inn_setupObs``.
 63  !
 64  !               - Setup ``columnData`` and ``gridStateVector`` modules (read
 65  !                 list of analysis variables from namelist) and allocate column
 66  !                 object for storing trial on analysis levels.
 67  !
 68  !               - Setup the observation error statistics in ``obsSpaceData``
 69  !                 object: ``oer_setObsErrors``.
 70  !
 71  !               - Filter out observations from satellites
 72  !                 not specified in the name list: ``filt_iceConcentration``.
 73  !
 74  !               - Filter scatterometer backscatter anisotropy observations
 75  !                 where wind speed is too small: ``filt_backScatAnisIce``.
 76  !
 77  !               - Setup observation-error for scatterometer backscatter
 78  !                 anisotropy observations: ``oer_setErrBackScatAnisIce``.
 79  !
 80  !             - **Main calculation:**
 81  !
 82  !               - Compute the analysis-error: ``aer_analysisError``.
 83  !
 84  !               - Update the Days Since Last Obs: ``aer_daysSinceLastObs``.
 85  !
 86  !               - Update the observation files: ``obsf_writeFiles``.
 87  !
 88  !           --
 89  !
 90  !:Options: `List of namelist blocks <../namelists_in_each_program.html#analysiserroroi>`_
 91  !          that can affect the ``analysisErrorOI`` program.
 92  !
 93  !          * The relevant namelist blocks used to configure the
 94  !            analysis-error calculation are listed in the following table:
 95  ! 
 96  !======================== ================== ==============================================================
 97  ! Module                   Namelist           Description of what is controlled
 98  !======================== ================== ==============================================================
 99  ! ``timeCoord_mod``       ``NAMTIME``         assimilation time window length, temporal resolution of
100  !                                             the background state
101  ! ``columndata_mod``      ``NAMSTATE``        name of the analysis variable (RPN nomvar, 4-character long),
102  !                                             only sea ice concentration (GL) is allowed for now
103  ! ``gridstatevector_mod``       "                                   "
104  ! ``obsspacedata_mod``    ``NAMDIMO``         specify the maximum number of header and body elements
105  ! ``obsfilter_mod``       ``NAMFILT``         list of varno to use and bit flags (13-bit#) for filtering
106  ! ``obsfilter_mod``       ``namPlatformIce``  list of observation platforms to assimilate
107  ! ``sqliteread_mod``      ``NAMSQLgl``        list of varno to read from SQLite files for sea ice
108  ! ``sqliteread_mod``      ``namSQLUpdate``    list of elements to update in SQLite files
109  ! ``sqliteread_mod``      ``namSQLInsert``    place holder, could be empty
110  ! ``analysisErrorOI_mod`` ``NAMAER``          set the maximum analysis-error std dev. allowed
111  !======================== ================== ==============================================================
112  !
113  use version_mod
114  use ramDisk_mod
115  use utilities_mod
116  use midasMpi_mod
117  use message_mod
118  use mathPhysConstants_mod
119  use horizontalCoord_mod
120  use verticalCoord_mod
121  use timeCoord_mod
122  use obsSpaceData_mod
123  use columnData_mod
124  use gridStateVector_mod
125  use obsFiles_mod
126  use innovation_mod
127  use obsErrors_mod
128  use obsFilter_mod  
129  use analysisErrorOI_mod
130
131  implicit none
132
133  integer :: istamp, exdb, exfin
134  integer :: ierr, dateStampFromObs
135  integer :: get_max_rss
136  character(len=48) :: obsMpiStrategy, varMode
137  character(len=20) :: trlmFileName
138  character(len=15), parameter :: myName = 'analysisErrorOI'
139
140  type(struct_obs)       , target :: obsSpaceData
141  type(struct_columnData), target :: trlColumnOnAnlLev
142  type(struct_hco)      , pointer :: hco_anl => null()
143  type(struct_vco)      , pointer :: vco_anl => null()
144
145  istamp = exdb('ANALYSISERROROI','DEBUT','NON')
146
147  call ver_printNameAndVersion(myName,'Program to calculate the analysis-error standard deviation for sea ice using OI.')
148
149  ! MPI initialization
150  call mmpi_initialize
151
152  if( mmpi_nprocs > 1 ) then
153    call msg(myName,'mmpi_nprocs = '//str(mmpi_nprocs))
154    call utl_abort(myName//': this version of the code should only be used with one mpi task.')
155  end if
156
157  call tmg_init(mmpi_myid, 'TMG_INFO')
158
159  call utl_tmg_start(0,'Main')
160
161  varMode='analysis'
162
163  ! Setup the ram disk
164  call ram_setup
165
166  obsMpiStrategy = 'LIKESPLITFILES'
167
168  !
169  !- Initialize the Temporal grid and set dateStamp from env variable
170  !
171  call tim_setup()
172
173  if (tim_nstepobs > 1 .or. tim_nstepobsinc > 1) then
174    call utl_abort(myName//': The program assumes only one time step.')
175  end if
176
177  !
178  !- Initialize observation file names and set datestamp if not already
179  !
180  call obsf_setup(dateStampFromObs, varMode)
181  if (tim_getDateStamp() == 0) then
182    if (dateStampFromObs > 0) then
183      call tim_setDateStamp(dateStampFromObs)
184    else
185      call utl_abort(myName//': DateStamp was not set')
186    end if
187  end if
188
189  !
190  !- Initialize constants
191  !
192  if (mmpi_myid == 0) call mpc_printConstants(6)
193
194  !
195  !- Initialize list of analyzed variables.
196  !
197  call gsv_setup
198  call msg_memUsage(myName)
199
200  trlmFileName = './trlm_01'
201
202  !
203  !- Initialize the Analysis grid
204  !
205  call msg(myName,'Set hco parameters for analysis grid', mpiAll_opt=.false.)
206  call hco_SetupFromFile(hco_anl, trlmFileName, aer_backgroundEtiket) ! IN
207
208  !
209  !- Initialisation of the analysis grid vertical coordinate from analysisgrid file
210  !
211  call vco_SetupFromFile( vco_anl,        & ! OUT
212                          './analysisgrid') ! IN
213
214  call col_setVco(trlColumnOnAnlLev,vco_anl)
215  call msg_memUsage(myName)
216
217  !
218  !- Setup and read observations
219  !
220  call inn_setupObs(obsSpaceData, hco_anl, 'VAR', obsMpiStrategy, varMode) ! IN
221  call msg_memUsage(myName)
222
223  !
224  !- Basic setup of columnData module
225  !
226  call col_setup
227  call msg_memUsage(myName)
228
229  !
230  !- Memory allocation for background column data
231  !
232  call col_allocate(trlColumnOnAnlLev,obs_numheader(obsSpaceData),mpiLocal_opt=.true.)
233
234  !
235  !- Initialize the observation error covariances
236  !
237  call oer_setObsErrors(obsSpaceData, varMode) ! IN
238  call msg_memUsage(myName)
239
240  ! Sea ice concentration
241  call filt_iceConcentration(obsSpaceData, beSilent=.false.)
242  call filt_backScatAnisIce(obsSpaceData, beSilent=.false.)
243  call oer_setErrBackScatAnisIce(obsSpaceData, beSilent=.false.)
244
245  ! Compute the analysis-error
246  call aer_analysisError(obsSpaceData, hco_anl, vco_anl, trlmFileName)
247
248  ! Update the Days Since Last Obs
249  call aer_daysSinceLastObs(obsSpaceData, hco_anl, vco_anl, trlmFileName)
250
251  ! Now write out the observation data files
252  if ( .not. obsf_filesSplit() ) then 
253    call msg(myName,'reading/writing global observation files')
254    call obs_expandToMpiGlobal(obsSpaceData)
255    if (mmpi_myid == 0) call obsf_writeFiles(obsSpaceData)
256  else
257    ! redistribute obs data to how it was just after reading the files
258    call obs_MpiRedistribute(obsSpaceData,OBS_IPF)
259    call obsf_writeFiles(obsSpaceData)
260  end if
261
262  ! Deallocate copied obsSpaceData
263  call obs_finalize(obsSpaceData)
264
265  !
266  ! 3. Job termination
267  !
268  istamp = exfin('ANALYSISERROROI','FIN','NON')
269
270  call utl_tmg_stop(0)
271
272  call tmg_terminate(mmpi_myid, 'TMG_INFO')
273
274  call rpn_comm_finalize(ierr) 
275
276end program midas_analysisErrorOI
