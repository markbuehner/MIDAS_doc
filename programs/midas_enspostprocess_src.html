 17program midas_ensPostProcess
 18  !
 19  !:Purpose: Post-processing program for the local ensemble transform Kalman filter (LETKF).
 20  !          Many aspects of this program are controlled throught the namelist block
 21  !          namEnsPostProc defined in epp_postProcess.
 22  !
 23  !          ---
 24  !
 25  !:Algorithm: The ``ensPostProcess`` program performs several post-processing tasks of
 26  !            LETK ensemble analyses and the ensemble trials. The following tasks are
 27  !            performed based on the namelist options.
 28  !
 29  !              - **Mean and std**: Ensemble mean and standard deviation are computed
 30  !                upon reading the ensemble members and after the processes which may
 31  !                alter the mean or the standard deviation.
 32  !
 33  !              - **RTPP**: Relaxation to the prior perturbation will relax the analysis
 34  !                perturbation to the trial perturbation with a given amount (:math:`\alpha`) in the namelist.
 35  !                It can be formulated as:
 36  !                :math:`x^a_i \leftarrow \overline{x^a} + (1-\alpha)(\overline{x^a}-x^a_i)+\alpha(\overline{x^b}-x^b_i)`
 37  !
 38  !                More details for the RTPP can be found in paper:
 39  !                `Impacts of Initial Estimate and Observation Availability on Convective-Scale
 40  !                Data Assimilation with an Ensemble Kalman Filter
 41  !                <https://journals.ametsoc.org/view/journals/mwre/132/5/1520-0493_2004_132_1238_ioieao_2.0.co_2.xml>`_
 42  !
 43  !              - **RTPS**: Relaxation to the prior spread will relax the analysis ensemble spread (i.e. standard
 44  !                deviation) to the trial ensemble spread with a given amount (:math:`\alpha`) in the namelist.
 45  !                With the ensemble spread of analysis (:math:`\sigma^a`)
 46  !                and trial (:math:`\sigma^b`), it can be formulated as:
 47  !                :math:`x^a_i \leftarrow \overline{x^a} + (\overline{x^a}-x^a_i)(1+\alpha(\sigma^b-\sigma^a)/\sigma^a)`
 48  !
 49  !                More details for the RTPS can be found in paper:
 50  !                `Evaluating Methods to Account for System Errors in Ensemble Data Assimilation
 51  !                <https://journals.ametsoc.org/view/journals/mwre/140/9/mwr-d-11-00276.1.xml>`_
 52  !
 53  !              - **Humidity limits**: Impose the saturation limits and RTTOV limits
 54  !                on the humidity for each member.
 55  !
 56  !              - **Recenter**: The ensemble analyses from the LETKF are recentered
 57  !                around the EnVar analysis based on the input coefficent.
 58  !                It is also called hybrid gain algorithm and more details can be found in
 59  !                paper:
 60  !                `Using the hybrid gain algorithm to sample data assimilation uncertainty
 61  !                <https://rmets.onlinelibrary.wiley.com/doi/10.1002/qj.3426>`_
 62  !
 63  !              - **Subsample**: Select 20 ensemble members for the medium range forecasts.
 64  !
 65  !              - **Random perturbation**: Generate homogeneous and isotropic random perturbations
 66  !                and add them to the ensemble analyses.
 67  !
 68  !              - **Analysis increments**: After all the processes above, the analysis incrememnts are
 69  !                computed at the central time for all the members including the control member and
 70  !                the subsampled 20 members.
 71  !
 72  !              - **Mask analysis increments**: For LAM grid, the analysis increments in the blending
 73  !                area are masked out.
 74  !
 75  !            --
 76  !
 77  !============================================= ==============================================================
 78  ! Input and Output Files                        Description of file
 79  !============================================= ==============================================================
 80  ! ``flnml``                                     In - Main namelist file with parameters user may modify
 81  ! ``ensemble_trial/$YYYYMMDDHH_006_$NNNN``      In - Background ensemble member files
 82  ! ``ensemble_anal/$YYYYMMDDHH_000_$NNNN``       In - Analysis ensemble member files
 83  ! ``bgcov``                                     In - Bnmc matrix for the random perturbations
 84  ! ``analysis_grid``                             In - Horizontal grid file on which the random perturbations
 85  !                                               are generated
 86  ! ``$YYYYMMDDHH_recentering_analysis``          In - Analysis from EnVar for recentering the ensemble analyses
 87  ! ``analinc_mask``                              In - Mask file for masking the analysis increments for LAM grid
 88  ! ``$YYYYMMDDHH_000_inc_$NNNN``                 Out - Ensemble analysis increments for trials
 89  ! ``$YYYYMMDDHH_000_$NNNN``                     Out - Ensemble analyses recentered and perturbed
 90  ! ``$YYYYMMDDHH_000_$NNNN_raw``                 Out - Ensemble analyses from LETKF
 91  ! ``$YYYYMMDDHH_000_analmean``                  Out - Mean ensemble analysis without perturbation
 92  ! ``$YYYYMMDDHH_000_analrms``                   Out - Analysis ensemble spread (i.e. standard deviation) without perturbation
 93  ! ``$YYYYMMDDHH_000_analrms_ascii``             Out - Global average of  ``$YYYYMMDDHH_000_analrms``
 94  ! ``$YYYYMMDDHH_000_analpertmean``              Out - Mean ensemble analysis with perturbation and recentering
 95  ! ``$YYYYMMDDHH_000_analpertrms``               Out - Analysis ensemble spread with perturbation and recentering
 96  ! ``$YYYYMMDDHH_000_analpertrms_ascii``         Out - Global average of  ``$YYYYMMDDHH_000_analpertrms``
 97  ! ``$YYYYMMDDHH_006_trialmean``                 Out - Mean ensemble trial for all time levels
 98  ! ``$YYYYMMDDHH_006_trialrms``                  Out - Trial ensemble spread at 6H
 99  ! ``$YYYYMMDDHH_006_trialrms_ascii``            Out - Global average of ``$YYYYMMDDHH_006_trialrms``
100  ! ``subspace/$YYYYMMDDHH_000_inc_$NNNN``        Out - Subsampled ensemble analysis increments for progs
101  ! ``subspace/$YYYYMMDDHH_000_$NNNN``            Out - Subsampled ensemble analyses recentered and perturbed
102  ! ``subspace_unpert/$YYYYMMDDHH_000_$NNNN``     Out - Unperturbed subsampled ensemble analyses recentered
103  !============================================= ==============================================================
104  !
105  !            --
106  !
107  !:Synopsis: Below is a summary of the ``ensPostProcess`` program calling sequence:
108  !
109  !             - **Initial setups:**
110  !
111  !               - Read the NAMLETKF namelist and check/modify some values.
112  !
113  !               - Various modules are setup: ``gridStateVector_mod``, ``timeCoord_mod`` (and set up dates
114  !                 and ``dateStampList`` variables for both trials and increments/analyses).
115  !
116  !               - Setup horizontal and vertical grid objects from first
117  !                 ensemble member file.
118  !
119  !               - Allocate varaibles and read ensemble analyses and trials
120  !
121  !             - **Ensemble postprocess:**
122  !
123  !               - Compute ensemble mean and spread (i.e. standard deviation) for analysis and trial
124  !
125  !               - Perform RTPP and RTPS and recompute analysis spread
126  !
127  !               - Recenter ensemble mean and recompute analysis mean and spread
128  !
129  !               - Apply humidity saturation limit and RTTOV HU limit and recompute analysis mean
130  !
131  !               - Subsample ensemble analysis
132  !
133  !               - Add random perturbation to the analyses and compute perturbed analysis spread
134  !
135  !               - Add random perturbation to the subsampled analyses and recenter subsampled mean to the
136  !                 global mean.
137  !
138  !               - Compute analysis increments for all analyses and subsampled analyses including the ensemble mean
139  !
140  !               - Mask LAM analysis increments are recompute the ensemble analyses by adding the increments
141  !                 to the trials
142  !
143  !               - Writing the outputs.
144  !
145  !:Options: `List of namelist blocks <../namelists_in_each_program.html#ensPostProcess>`_
146  !          that can affect the ``ensPostProcess`` program.
147  !
148  !========================== ========================== ==============================================================
149  ! Program/Module             Namelist                   Description of what is controlled
150  !========================== ========================== ==============================================================
151  ! ``midas_ensPostProcess``   ``NAMENSPOSTPROC``         Number of ensemble members and read, write control of trials
152  !                                                       and analyses and horizontal interpolation degree
153  ! ``enspostprocess_mod``     ``NAMENSPOSTPROCMODULE``   Control the postprocess algorithms
154  ! ``timeCoord_mod``          ``NAMTIME``                Temporal resolution of the background state and the analysis
155  !========================== ========================== ==============================================================
156  !
157
158  use version_mod
159  use midasMpi_mod
160  use fileNames_mod
161  use ensembleStateVector_mod
162  use gridStateVector_mod
163  use gridStateVectorFileIO_mod
164  use verticalCoord_mod
165  use horizontalCoord_mod
166  use timeCoord_mod
167  use utilities_mod
168  use ramDisk_mod
169  use ensPostProcess_mod
170  implicit none
171
172  type(struct_ens)          :: ensembleTrl
173  type(struct_ens)          :: ensembleAnl
174  type(struct_vco), pointer :: vco_ens => null()
175  type(struct_hco), pointer :: hco_ens => null()
176  type(struct_gsv)          :: stateVectorHeightSfc, stateVectorCtrlTrl
177
178  character(len=256) :: ensPathNameAnl = 'ensemble_anal'
179  character(len=256) :: ensPathNameTrl = 'ensemble_trial'
180  character(len=256) :: ensFileName, gridFileName, ctrlFileName
181  integer, allocatable :: dateStampList(:)
182  integer :: ierr, nulnam, stepIndex
183  logical :: targetGridFileExists
184  integer, external :: get_max_rss, fstopc, fnom, fclos
185
186  integer :: nEns             ! ensemble size
187  logical :: readTrlEnsemble  ! activate reading of trial ensemble
188  logical :: readAnlEnsemble  ! activate reading of analysis ensemble
189  logical :: writeTrlEnsemble ! activate writing of the trial ensemble (useful when it's interpolated)
190  character(len=12) :: hInterpolationDegree ! select degree of horizontal interpolation (if needed)
191  NAMELIST /namEnsPostProc/nEns, readTrlEnsemble, readAnlEnsemble, &
192                           writeTrlEnsemble, hInterpolationDegree
193
194  call ver_printNameAndVersion('ensPostProcess','Program for post-processing of LETKF analysis ensemble')
195
196  !
197  !- 0. MPI, TMG and misc. initialization
198  !
199  call mmpi_initialize
200  call tmg_init(mmpi_myid, 'TMG_INFO')
201
202  call utl_tmg_start(0,'Main')
203  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
204
205  ! Avoid printing lots of stuff to listing for std file I/O
206  ierr = fstopc('MSGLVL','ERRORS',0)
207
208  ! Setup the ramdisk directory (if supplied)
209  call ram_setup
210
211  !- Setting default namelist variable values
212  nEns = 256
213  readTrlEnsemble  = .true.
214  readAnlEnsemble  = .true.
215  writeTrlEnsemble = .false.
216  hInterpolationDegree = 'LINEAR' ! or 'CUBIC' or 'NEAREST'
217
218  !- Read the namelist
219  nulnam = 0
220  ierr = fnom(nulnam, './flnml', 'FTN+SEQ+R/O', 0)
221  read(nulnam, nml=namEnsPostProc, iostat=ierr)
222  if ( ierr /= 0) call utl_abort('midas-ensPostProcess: Error reading namelist')
223  if ( mmpi_myid == 0 ) write(*,nml=namEnsPostProc)
224  ierr = fclos(nulnam)
225
226  if (.not.readTrlEnsemble .and. .not.readAnlEnsemble) then
227    call utl_abort('midas-ensPostProcess: must read either Trial or Analysis ensemble')
228  end if
229
230  if (writeTrlEnsemble .and. .not.readTrlEnsemble) then
231    call utl_abort('midas-ensPostProcess: cannot write Trial ensemble if it is not read')
232  end if
233
234  if (writeTrlEnsemble .and. readAnlEnsemble) then
235    call utl_abort('midas-ensPostProcess: cannot write Trial ensemble when Analysis ensemble is read')
236  end if
237
238  !- 1. Initialize date/time-related info
239
240  ! Setup timeCoord module, set dateStamp with value from trial or analysis ensemble member 1
241  if (readTrlEnsemble) then
242    call fln_ensFileName(ensFileName, ensPathNameTrl, memberIndex_opt=1, &
243                         copyToRamDisk_opt=.false.)
244  else
245    call fln_ensFileName(ensFileName, ensPathNameAnl, memberIndex_opt=1, &
246                         copyToRamDisk_opt=.false.)
247  end if
248  call tim_setup(fileNameForDate_opt=ensFileName)
249  allocate(dateStampList(tim_nstepobsinc))
250  call tim_getstamplist(dateStampList,tim_nstepobsinc,tim_getDatestamp())
251
252  write(*,*) 'midas-ensPostProcess: analysis dateStamp = ', tim_getDatestamp()
253
254  !- 2. Initialize variables and grids
255
256  !- Initialize variables of the model states
257  call gsv_setup
258
259  !- Initialize the grid from targetgrid file or from trial or analysis ensemble member 1
260  if (mmpi_myid == 0) write(*,*) ''
261  if (mmpi_myid == 0) write(*,*) 'midas-ensPostProcess: Set hco and vco parameters for ensemble grid'
262  inquire(file='targetgrid', exist=targetGridFileExists)
263  if (targetGridFileExists) then
264    gridFileName = 'targetgrid'
265  else if (readTrlEnsemble) then
266    call fln_ensFileName(gridFileName, ensPathNameTrl, memberIndex_opt=1, &
267                         copyToRamDisk_opt=.false.)
268  else
269    call fln_ensFileName(gridFileName, ensPathNameAnl, memberIndex_opt=1, &
270                         copyToRamDisk_opt=.false.)
271  end if
272  if (mmpi_myid == 0) then
273    write(*,*) 'midas-ensPostProcess: file use to define grid = ', trim(gridFileName)
274  end if
275  call hco_SetupFromFile(hco_ens, gridFileName, ' ', 'ENSFILEGRID')
276  call vco_setupFromFile(vco_ens, gridFileName)
277
278  !- Read the sfc height from trial ensemble member 1 - only if we are doing NWP!
279  if (vco_getNumLev(vco_ens,'TH') > 0 .or. vco_getNumLev(vco_ens,'MM') > 0) then
280    if (readTrlEnsemble) then
281      call fln_ensFileName(ensFileName, ensPathNameTrl, memberIndex_opt=1, &
282                           copyToRamDisk_opt=.false.)
283    else
284      call fln_ensFileName(ensFileName, ensPathNameAnl, memberIndex_opt=1, &
285                           copyToRamDisk_opt=.false.)
286    end if
287    call gsv_allocate(stateVectorHeightSfc, 1, hco_ens, vco_ens, dateStamp_opt=tim_getDateStamp(),  &
288                      mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
289                      hInterpolateDegree_opt=hInterpolationDegree, &
290                      dataKind_opt=4, allocHeightSfc_opt=.true., varNames_opt=(/'P0','TT'/))
291    call gio_readFromFile(stateVectorHeightSfc, ensFileName, ' ', ' ',  &
292                          containsFullField_opt=.true., readHeightSfc_opt=.true.)
293  end if
294
295  !- 3. Allocate and read ensembles
296
297  call utl_tmg_start(2,'--ReadEnsemble')
298
299  !- Allocate ensembles, read the Anl ensemble
300  if (readAnlEnsemble) then
301    call fln_ensFileName(ensFileName, ensPathNameAnl, resetFileInfo_opt=.true.)
302    call ens_allocate(ensembleAnl, nEns, tim_nstepobsinc, hco_ens, vco_ens, &
303                      dateStampList, hInterpolateDegree_opt=hInterpolationDegree)
304    call ens_readEnsemble(ensembleAnl, ensPathNameAnl, biPeriodic=.false.)
305  end if
306
307  !- Allocate ensembles, read the Trl ensemble
308  if (readTrlEnsemble) then
309    call fln_ensFileName(ensFileName, ensPathNameAnl, resetFileInfo_opt=.true.)
310    call ens_allocate(ensembleTrl, nEns, tim_nstepobsinc, hco_ens, vco_ens, &
311                      dateStampList, hInterpolateDegree_opt=hInterpolationDegree)
312    call ens_readEnsemble(ensembleTrl, ensPathNameTrl, biPeriodic=.false.)
313
314  end if
315
316  call utl_tmg_stop(2)
317
318  !- Allocate and read the Trl control member
319  if (readTrlEnsemble .and. readAnlEnsemble) then
320    !- Allocate and read control member Trl
321    call gsv_allocate( stateVectorCtrlTrl, tim_nstepobsinc, hco_ens, vco_ens, &
322                       dateStamp_opt=tim_getDateStamp(),  &
323                       mpi_local_opt=.true., mpi_distribution_opt='Tiles', &
324                       dataKind_opt=4, allocHeightSfc_opt=.true., &
325                       hInterpolateDegree_opt=hInterpolationDegree, &
326                       allocHeight_opt=.false., allocPressure_opt=.false. )
327    call fln_ensFileName(ctrlFileName, ensPathNameTrl, memberIndex_opt=0, &
328                         copyToRamDisk_opt=.false.)
329    do stepIndex = 1, tim_nstepobsinc
330      call gio_readFromFile( stateVectorCtrlTrl, ctrlFileName, ' ', ' ',  &
331                             stepIndex_opt=stepIndex, containsFullField_opt=.true., &
332                             readHeightSfc_opt=.false. )
333    end do
334  end if
335
336  !- 4. Post processing of the analysis results (if desired) and write everything to files
337  call epp_postProcess(ensembleTrl, ensembleAnl, &
338                       stateVectorHeightSfc, stateVectorCtrlTrl, &
339                       writeTrlEnsemble)
340
341  !
342  !- 5. MPI, tmg finalize
343  !  
344  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
345  call utl_tmg_stop(0)
346
347  call tmg_terminate(mmpi_myid, 'TMG_INFO')
348  call rpn_comm_finalize(ierr) 
349
350  write(*,*) 'Memory Used: ', get_max_rss()/1024, 'Mb'
351
352  if ( mmpi_myid == 0 ) write(*,*) ' --------------------------------'
353  if ( mmpi_myid == 0 ) write(*,*) ' MIDAS-ensPostProcess ENDS'
354  if ( mmpi_myid == 0 ) write(*,*) ' --------------------------------'
355
356end program midas_ensPostProcess
