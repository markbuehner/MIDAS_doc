 17program midas_calcstats
 18  !
 19  !:Purpose: Swiss-knife type program originally used only for computing background-error covariances
 20  !          (**B**) based on homogeneous and isotropic (HI) correlations that was extended with time
 21  !          to compute various statistics and diagnostics from an ensemble of background-error estimates
 22  !
 23  !          ---
 24  !
 25  !:Algorithm: The ``calcStats`` program performs various types of statistics and diagnostics based on
 26  !            two modes defined through namelist options.
 27  !
 28  !              - **BHI**: Compute HI **B**. The approach for  limited-area applications is based on a bi-Fourier
 29  !                         spectral representation and the CVT described in <https://doi.org/10.1175/2010WAF2222401.1>.
 30  !                         For global applications, two formulations based on sperical-harmonics spectral representation
 31  !                         are available and controlled controlled by ``NAMCOMPUTEBHI``: 1) the legacy CVT formulation
 32  !                         described in <https://doi.org/10.1175/MWR-D-11-00097.1> and refereces therein,
 33  !                         and 2) an experimental-only formulation that mimics the CVT model used in the limited-area mode.
 34  !              - **TOOLBOX**: The swiss-knife component of this program controlled by ``NAMTOOLBOX`` from the
 35  !                global and LAM calcstats-related modules. Compute various statistics and diagnostics from
 36  !                an ensemble of background-error estimates in model-variable and/or control-variable space,
 37  !                such as:
 38  !                - HI vertical correlations
 39  !                - HI horizontal correlation functions
 40  !                - Variances
 41  !                - Local correlations
 42  !                - Optimal covariance localization radii
 43  !                - Power spectra
 44  !                Note that the above options are not all available in both global and limited-area
 45  !                applications.
 46  !            ---
 47  !
 48  !============================================= ==============================================================
 49  ! Input and Output Files                        Description of file
 50  !============================================= ==============================================================
 51  ! ``flnml``                                     In - Main namelist file with parameters user may modify
 52  ! ``ensemble/$YYYYMMDDHH_$HHH_$NNNN``           In - Background-error estimates
 53  ! ``various``                                   Out - Too many to be listed here
 54  !============================================= ==============================================================
 55  !
 56  !            --
 57  !
 58  !:Synopsis: Below is a summary of the ``calcStats`` program calling sequence:
 59  !
 60  !             - **Initial setups:**
 61  !
 62  !               - Read the NAMCONF and NAMENS namelist
 63  !
 64  !               - Setup horizontal and vertical grid objects using the first ensemble member
 65  !
 66  !               - Various modules are setup: ``gridStateVector_mod``, ``timeCoord_mod`` and ``bmatrix_mod``
 67  !
 68  !             - **Statistics and Diagnostics:**
 69  !
 70  !               - Intialize the module for global or limited-area applications
 71  !
 72  !               - Select the appropriate master routine based on the chosen mode
 73  !
 74  !               - The I/O and number crunching is performed within the global and limited-area modules
 75  !
 76  !
 77  !======================== ============ ==============================================================
 78  ! Module                   Namelist     Description of what is controlled
 79  !======================== ============ ==============================================================
 80  ! ``timeCoord_mod``        ``NAMTIME``  date of validity of the ensemble of background-error estimates
 81  !                                       the background state and increment
 82  ! ``calcstatslglb_mod``    ``mode-dependent``  Too many to be listed here
 83  ! ``calcstatslam_mod``     ``mode-dependent``  Too many to be listed here
 84  !======================== ============ ==============================================================
 85  !
 86 
 87  use version_mod
 88  use midasMpi_mod
 89  use fileNames_mod
 90  use HorizontalCoord_mod
 91  use VerticalCoord_mod
 92  use calcstatsglb_mod
 93  use calcstatslam_mod
 94  use utilities_mod
 95  use ramDisk_mod
 96  use gridStateVector_mod
 97  use timeCoord_mod
 98  implicit none
 99
100  type(struct_vco), pointer :: vco_ens => null()
101  type(struct_hco), pointer :: hco_ens => null()
102
103  character(len=256), parameter :: enspathname = 'ensemble'
104
105  integer           :: fstopc
106  integer           :: nulnam, ierr, fnom, fclos
107  character(len=256) :: ensFileName
108  character(len=4), pointer :: anlVar(:)
109
110  ! namelist variables
111  character(len=60) :: mode   ! can be 'BHI', 'TOOLBOX', 'STDDEV or 'POWERSPEC'
112  integer           :: nens   ! Ensemble size
113  integer           :: ip2    ! Ensemble lead time (hour) selected within the file
114
115  NAMELIST /NAMCONF/mode
116  NAMELIST /NAMENS/nens,ip2
117
118  call ver_printNameAndVersion('calcStats','Compute the homogeneous-isotropic stats')
119
120  !
121  !- 1.  Initilization
122  !
123  ierr = fstopc('MSGLVL','ERRORS',0)
124
125  !- 1.1 MPI and TMG
126  call mmpi_initialize
127  call tmg_init(mmpi_myid, 'TMG_INFO')
128
129  call utl_tmg_start(0,'Main')
130
131  ! Setup the ramdisk directory (if supplied)
132  call ram_setup
133
134  ! Setup time
135  call tim_setup
136
137  !- 1.2 Read NAMENS namelist
138  nens              = 96                ! default value
139  ip2               = -1                ! default value
140
141  nulnam = 0
142  ierr   = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
143  read (nulnam,nml=namens)
144  write(*     ,nml=namens)
145  ierr=fclos(nulnam)
146
147  !- 2.3 Initialize variables of the model states
148  call gsv_setup
149
150  call fln_ensfileName(ensFileName, ensPathName, memberIndex_opt=1)
151
152  !- 1.3 Initialize the horizontal grid
153  nullify(anlVar)
154  call gsv_varNamesList(anlVar)
155  call hco_SetupFromFile(hco_ens, ensFileName, ' ', 'Ensemble', varName_opt=anlVar(1)) ! IN
156
157  !- 1.4 Initialize the vertical grid
158  call vco_SetupFromFile( vco_ens,        & ! OUT
159                          ensFileName, ' ') ! IN
160
161  !- 1.5 Read NAMCONF namelist to find the mode
162  mode  = 'BHI'  ! default value
163
164  nulnam = 0
165  ierr   = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
166  read (nulnam,nml=namconf)
167  write(*     ,nml=namconf)
168  ierr   = fclos(nulnam)
169
170  !
171  !- 2. Select and launch the appropriate mode
172  !
173
174  !- 2.1 Module initialization
175  if (hco_ens % global) then
176     call csg_setup( nens, hco_ens, vco_ens) ! IN
177  else
178     call csl_setup( nens, hco_ens, vco_ens, ip2) ! IN
179  end if
180
181  !- 2.2 Mode selection
182  select case(trim(mode))
183  case ('BHI')
184     if (hco_ens % global) then
185        call csg_computeBhi
186     else
187        call csl_computeBhi
188     end if
189  case ('TOOLBOX')
190     if (hco_ens % global) then
191        call csg_toolbox
192     else
193        call csl_toolbox
194     end if
195  case default
196     write(*,*)
197     write(*,*) 'Unknown value of MODE in global mode: ',mode
198     call utl_abort('midas-calcstats')
199  end select
200
201  !
202  !- 3.  Ending...
203  !
204
205  write(*,*)
206  write(*,*) '---------------------'
207  write(*,*) '> ENDING CALCBMATRIX '
208  write(*,*) '---------------------'
209
210  !
211  !- 4.  MPI, tmg finalize
212  !  
213  call utl_tmg_stop(0)
214
215  call tmg_terminate(mmpi_myid, 'TMG_INFO')
216  call rpn_comm_finalize(ierr) 
217
218end program midas_calcstats
