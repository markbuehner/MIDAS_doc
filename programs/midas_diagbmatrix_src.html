 17program midas_diagBmatrix
 18  !
 19  !:Purpose: Program for computing diagnostics of the background-error covariances (**B**) and
 20  !          localization (**L**) matrices
 21  !
 22  !          ---
 23  !
 24  !:Algorithm: The ``diagBmatrix`` program performs two different types of diagnostics
 25  !            based on the namelist options.
 26  !
 27  !              - **Columns of B and L**: Allow to extract columns of **B** (the background-error covariances matrix)
 28  !                which is a useful way to simulate single-observation (a.k.a pseudo-obs) experiments without using
 29  !                real observations.
 30  !                See e.g. Figs 5 and 6 in <https://doi.org/10.1175/MWR-D-18-0248.1>.
 31  !                If localization is used, the corresponding columns of the localization matrix (**L**) will be outputed.
 32  !              - **Implied variances**: Compute the variances resulting from an ensemble of random perturbations
 33  !                generated from the provided **B** matrix, including hybrid covariances. This allow to evaluate the impact
 34  !                of various choices in the design of **B** on the modification to the variances of the ensemble of
 35  !                background-error estimates used to derivate **B**, i.e. the "measured" variances.
 36  !
 37  !            ---
 38  !
 39  !============================================= ==============================================================
 40  ! Input and Output Files                        Description of file
 41  !============================================= ==============================================================
 42  ! ``flnml``                                     In - Main namelist file with parameters user may modify
 43  ! ``ensemble/$YYYYMMDDHH_$HHH_$NNNN``           In - Background ensemble member files
 44  ! ``bgcov``                                     In - **B** nmc matrix for the random perturbations
 45  ! ``analysis_grid``                             In - Horizontal grid file on which the random perturbations
 46  !                                                    are generated
 47  ! ``trlm_$NN`` (e.g. ``trlm_01``)               In - Background state (a.k.a. trial) files for each timestep
 48  !                                                    Only needed if a LQ-HU transform is used in **B**
 49  ! ``columnB_$VAR_$YYYYMMDDHH.fst``              Out - Column of **B**
 50  ! ``columnBnorm_$VAR_$YYYYMMDDHH.fst``          Out - Column of **B** normalize by the value at the pseudo-obs location
 51  ! ``columnL_$YYYYMMDDHH.fst``                   Out - Column of **L** (only when ensemble-derived **B** is use) 
 52  ! ``stddev_$YYYYMMDDHH.fst``                    Out - Implied variances
 53  !============================================= ==============================================================
 54  !
 55  !            --
 56  !
 57  !:Synopsis: Below is a summary of the ``diagBmatrix`` program calling sequence:
 58  !
 59  !             - **Initial setups:**
 60  !
 61  !               - Read the NAMDIAG namelist and check/modify some values.
 62  !
 63  !               - Setup horizontal and vertical grid objects from the provided template file (./analysisgrid)
 64  !
 65  !               - Various modules are setup: ``gridStateVector_mod``, ``timeCoord_mod`` and ``bmatrix_mod``
 66  !
 67  !             - **Columns of B and L:**
 68  !
 69  !               - Attribute time bin of the pseudo-obs based on the namelist option.
 70  !
 71  !               - For each variables and prescribed pseudo-obs positions:
 72  !                       1) create a Dirac delta distribution (zero everywhere except at obs location).
 73  !                          This is equivalent to impose (O-F / sigma_obs) = 1 at one location.
 74  !                       2) apply **B** ^(1/2)^T **B** ^1/2 and output the results into a file
 75  !                       3) normalize the results by the value at the pseudo-obs position
 76  !                          and output to a file
 77  !
 78  !               - If bMatrixEnsemble_mod is activated, for each prescribed pseudo-obs positions:
 79  !                       1) create a Dirac delta distribution
 80  !                       2) apply **L** ^(1/2)^T **L** ^1/2 and output the results into a file
 81  !
 82  !             - **Implied variances:**
 83  !
 84  !               - For each member of the chosen random ensemble size:
 85  !                       1) create a random control vector
 86  !                       2) apply **B** ^1/2
 87  !                       3) store the resulting 3D state in randomEns
 88  !
 89  !               - Compute and remove the ensemble mean
 90  !
 91  !               - Compute the ensemble standard deviations at each grid point and output to a file
 92  !
 93  !               - Compute the zonal and domain mean standard deviations and output to the same file
 94  !
 95  !======================== ============ ==============================================================
 96  ! Module                   Namelist     Description of what is controlled
 97  !======================== ============ ==============================================================
 98  ! ``timeCoord_mod``        ``NAMTIME``  assimilation time window length, temporal resolution of
 99  !                                       the background state and increment
100  ! ``bMatrixEnsemble_mod``  ``NAMBEN``   weight and other parameters for ensemble-based **B** matrix
101  !                                       component
102  ! ``bMatrixHI_mod``        ``NAMBHI``   weight and other parameters for the climatological **B** matrix
103  !                                       component based on homogeneous-isotropic covariances
104  !                                       represented in spectral space
105  ! Other **B** matrix modules   various      weight and other parameters for each type of **B** matrix
106  !======================== ============ ==============================================================
107  !
108

164  integer :: myLonBeg, myLonEnd

169  character(len=4)   :: varName

171  character(len=2)   :: instanceIndexString
172

183  character(len=4) :: oneobs_varName ! can be 'all' or a specific variable name (default='all')
184  logical :: writeEnsAmplitude       ! choose to write ensemble amplitude fields (for ensemble B)
185  logical :: writeTextStddev         ! choose to write stddev to text file in addition to standard file
186  logical :: writePsiChiStddev       ! choose to also write stddev of Psi/Chi in addition to UU/VV
187
188  namelist /namdiag/numperturbations, nrandseed, oneobs_levs, oneobs_lonlat, &
189                    oneobs_varName, oneobs_timeStep, writeEnsAmplitude, writeTextStddev, writePsiChiStddev
190
191  call ver_printNameAndVersion('diagBmatrix','Diagnositcs of the B matrix')
192
193  ! MPI, tmg initialization
194  call mmpi_initialize
195  call tmg_init(mmpi_myid, 'TMG_INFO')
196  call utl_tmg_start(0,'Main')
197  ierr = fstopc('MSGLVL','ERRORS',0)
198
199  ! Set default values for namelist NAMDIAG parameters
200  numperturbations  = -1
201  nrandseed         =  1
202  oneobs_varName    = 'all'
203  oneobs_timeStep   = 'middle'
204  oneobs_levs(:)    = -1
205  oneobs_lonlat(:,:)= -1
206  writeEnsAmplitude = .false.
207  writeTextStddev   = .false.
208  writePsiChiStddev = .false.
209
210  ! Read the parameters from NAMDIAG
211  nulnam=0
212  ierr=fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
213  read(nulnam,nml=namdiag,iostat=ierr)
214  if(ierr.ne.0) call utl_abort('midas-diagBmatrix: Error reading namelist')
215  write(*,nml=namdiag)
216  ierr=fclos(nulnam)
217
218  nlevs=0
219  do index = 1, size(oneobs_levs)
220    if (oneobs_levs(index) >= 1) nlevs=nlevs+1
221  end do
222  nLonLatPos=0
223  do index = 1, size(oneobs_lonlat(:,lonPosIndex))
224    if (oneobs_lonlat(index,lonPosIndex) >= 1 .and. oneobs_lonlat(index,latPosIndex) >= 1) nLonLatPos=nLonLatPos+1  
225  end do
226
227  ! Top Level Control setup
228  call ram_setup
229
230  !- Initialize the Temporal grid and set dateStamp from env variable
231  call tim_setup()
232  if (tim_getDateStamp() == 0) then
233    call utl_abort('midas-diagBmatrix: date must be set by env variable MIDAS_DATE')
234  end if
235
236  ! Build date-time string from dateStamp
237  dateStamp = tim_getDateStamp()
238  ierr = newdate(dateStamp,datePrint,timePrint,-3)
239  dateTime = datePrint*100 + timePrint/1000000
240  write(datestr,'(i10.10)') dateTime
241  write(*,*)' datePrint= ',datePrint,' timePrint= ',timePrint
242  write(*,*)' date= ',dateTime,' stamp= ',dateStamp
243
244  ! Initialize variables of the model states
245  call gsv_setup
246
247  ! Initialize the Analysis horizontal grid
248  call hco_SetupFromFile( hco_anl,'./analysisgrid', 'ANALYSIS', 'Analysis' ) ! IN
249
250  if ( hco_anl % global ) then
251    hco_core => hco_anl
252  else
253    !- Iniatilized the core (Non-Exteded) analysis grid
254    call hco_SetupFromFile( hco_core, './analysisgrid', 'COREGRID', 'AnalysisCore' ) ! IN
255  end if
256
257  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
258
259  ! Initialize the vertical coordinate from the statistics file
260  call vco_SetupFromFile( vco_anl,        & ! OUT
261                          './analysisgrid') ! IN
262
263  ! Allocate the statevector
264  call gsv_allocate(statevector, tim_nstepobsinc, hco_anl, vco_anl, &
265                    datestamp_opt=tim_getDatestamp(), mpi_local_opt=.true., &
266                    allocHeight_opt=.false., allocPressure_opt=.false.)
267  call gsv_zero(statevector)
268  nkgdim = statevector%nk
269
270  ! Setup the B matrix
271  call bmat_setup(hco_anl,hco_core,vco_anl)
272
273  !- Initialize the gridded variable transform module
274  call gvt_setup(hco_anl,hco_core,vco_anl)
275  if ( gsv_varExist(varName='HU') ) call gvt_setupRefFromTrialFiles('HU')
276  
277  ! Setup of the L matrix done in bmat_setup
278  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
279
280  !
281  !==============================================
282  !- Compute columns of B and L matrices
283  !==============================================
284  !
285  if ( nLevs >= 1 .and. nLonLatPos >= 1 ) then
286
287    !
288    !- Compute columns of the B matrix
289    !
290    select case(trim(oneobs_timeStep))
291    case ('first')
292      oneobs_timeStepIndex = 1
293    case ('middle')
294      if (mod(tim_nstepobsinc,2) == 0) then
295        write(*,*)
296        write(*,*) 'odd number of nstepobsinc a required for obs place in the middle of the analysis window '
297        write(*,*) 'tim_nstepobsinc = ', tim_nstepobsinc
298        call utl_abort('midas-diagBmatrix')
299      end if
300      oneobs_timeStepIndex = (tim_nstepobsinc+1)/2
301    case ('last')
302      oneobs_timeStepIndex = tim_nstepobsinc
303    case default
304      write(*,*)
305      write(*,*) 'Unsupported oneobs_timeStep : ', trim(oneobs_timeStep)
306      call utl_abort('midas-diagBmatrix')
307    end select
308
309    allocate(controlVector(cvm_nvadim))
310
311    write(*,*) '********************************************'
312    write(*,*) 'midas-diagBmatrix: Compute columns of B matrix'
313    write(*,*) '********************************************'
314    write(*,*)
315    write(*,*) ' temporal location          = ',trim(oneobs_timeStep), oneobs_timeStepIndex
316    write(*,*) ' number of levels            = ',nLevs
317    write(*,*) ' number of lon-lat positions = ', nLonLatPos
318
319    do varIndex = 1, vnl_numvarmax
320
321      if ( .not. gsv_varExist(varName=vnl_varNameList(varIndex)) ) cycle
322      if ( trim(oneobs_varName)  /= 'all' .and. (trim(oneobs_varName) /= trim(vnl_varNameList(varIndex))) ) cycle
323
324      filenameInc    = 'columnB_'      // trim(vnl_varNameList(varIndex)) // '_' // datestr // '.fst'
325      filenameIncNorm= 'columnBnorm_'  // trim(vnl_varNameList(varIndex)) // '_' // datestr // '.fst'
326      filenameEnsAmp = 'ensAmplitude_' // trim(vnl_varNameList(varIndex)) // '_'
327
328      write(*,*)
329      write(*,*) 'midas-diagBmatrix: simulating a pseudo-observation of ', trim(vnl_varNameList(varIndex))
330
331      if(vnl_varLevelFromVarname(vnl_varNameList(varIndex)).eq.'SF') then
332        nlevs2 = 1
333      else
334        nlevs2 = nlevs
335      end if
336
337      ip3 = 0
338      do levIndex = 1, nlevs2
339        do lonLatPosIndex = 1, nLonLatPos
340
341          latIndex = oneobs_lonlat(lonLatPosIndex,latPosIndex)
342          lonIndex = oneobs_lonlat(lonLatPosIndex,lonPosIndex)
343
344          call gsv_zero(statevector)                   
345          call gsv_getField(statevector,field4d,vnl_varNameList(varIndex))
346
347          if ( latIndex >= statevector%myLatBeg .and. latIndex <= statevector%myLatEnd .and. &
348               lonIndex >= statevector%myLonBeg .and. lonIndex <= statevector%myLonEnd ) then
349            if (vnl_varLevelFromVarname(vnl_varNameList(varIndex)) == 'SF') then
350              field4d(lonIndex,latIndex,1                    ,oneobs_timeStepIndex) = 1.0D0
351            else
352              field4d(lonIndex,latIndex,oneobs_levs(levIndex),oneobs_timeStepIndex) = 1.0D0
353            end if
354          end if
355
356          controlVector(:)=0.0d0
357          call bmat_sqrtBT(controlVector,cvm_nvadim,statevector)
358          call bmat_sqrtB (controlVector,cvm_nvadim,statevector)
359          
360          write(*,*)'midas-diagBmatrix: writing out the column of B, levIndex,lonIndex,latIndex=',levIndex,lonIndex,latIndex
361
362          ip3 = ip3 + 1
363          
364          do stepIndexInc = 1, tim_nstepobsinc
365            call gio_writeToFile(statevector,filenameInc,'1OBS_'//trim(vnl_varNameList(varIndex)),  &
366                 stepIndex_opt=stepIndexInc, ip3_opt=ip3, unitConversion_opt=.true.)
367          end do
368
369          ! Normalized the result to get correlation-like pattern
370          centralValueLocal = 0.d0
371          if ( latIndex >= statevector%myLatBeg .and. latIndex <= statevector%myLatEnd .and. &
372               lonIndex >= statevector%myLonBeg .and. lonIndex <= statevector%myLonEnd ) then
373            if (vnl_varLevelFromVarname(vnl_varNameList(varIndex)).eq.'SF') then
374              centralValueLocal = field4d(lonIndex,latIndex,1                    ,oneobs_timeStepIndex)
375            else
376              centralValueLocal = field4d(lonIndex,latIndex,oneobs_levs(levIndex),oneobs_timeStepIndex)
377            end if
378          end if
379          call rpn_comm_allreduce(centralValueLocal, centralValue, 1,  &
380                                  "MPI_DOUBLE_PRECISION", "MPI_SUM", "GRID", ierr)
381          
382          write(*,*) 'midas-diagBmatrix: centralValue found = ', centralValue
383          
384          if (centralValue /= 0.d0) then
385            call gsv_scale(statevector,1.d0/centralValue)
386          else
387            call utl_abort('midas-diagBmatrix: central value equals 0!')
388          end if
389          
390          do stepIndexInc = 1, tim_nstepobsinc
391            call gio_writeToFile(statevector,filenameIncNorm,'1OBSNRM_'//trim(vnl_varNameList(varIndex)), &
392                                 stepIndex_opt=stepIndexInc, ip3_opt=ip3,  &
393                                 unitConversion_opt=.false.)
394          end do
395
396          ! Write the ensemble amplitude fields (i.e., the alphas) when Bens is active
397          if (writeEnsAmplitude) call ben_writeAmplitude('./',filenameEnsAmp, ip3) ! IN
398
399        end do
400      end do
401    end do
402
403    deallocate(controlVector)
404
405    !
406    !- Compute columns of the L matrix
407    !
408    numLoc = ben_getNumLoc()
409    numBensInstance = ben_getNumInstance()
410    numStepAmplitude = ben_getNumStepAmplitudeAssimWindow()
411    amp3dStepIndex   = ben_getAmp3dStepIndexAssimWindow()
412
413    if (numStepAmplitude > 1) then
414      allocate(datestampList(numStepAmplitude))
415      call tim_getstamplist(dateStampList,numStepAmplitude,tim_getDatestamp())
416      nEns = ben_getnEns()
417      adv_amplitudeAssimWindow => ben_getAmplitudeAssimWindow()
418    else
419      allocate(datestampList(1))
420      datestampList(1) = dateStamp
421    end if
422
423    do instanceIndex = 1, numBensInstance
424      do locIndex = 1, numLoc ! (this loop will be done only when localization is used in B)
425        loc => ben_getLoc(locIndex,instanceIndex_opt=instanceIndex)
426
427        if (loc%vco%Vcode == 5002 .or. loc%vco%Vcode == 5005) then
428          varNameALFA(:) = varNameALFAatm(:)
429        else ! vco_anl%Vcode == -1
430          varNameALFA(:) = varNameALFAsfc(:)
431        end if
432
433        call mmpi_setup_latbands(loc%hco%nj, latPerPE, latPerPEmax, myLatBeg, myLatEnd)
434        call mmpi_setup_lonbands(loc%hco%ni, lonPerPE, lonPerPEmax, myLonBeg, myLonEnd)
435
436        call ens_allocate(ensAmplitude, loc%nEnsOverDimension, numStepAmplitude, loc%hco, loc%vco, &
437                          datestampList=dateStampList, varNames_opt=varNameALFA, dataKind_opt=8)
438
439        call gsv_allocate(statevectorEnsAmplitude, numStepAmplitude, loc%hco, loc%vco, &
440                          dateStampList_opt=dateStampList, varNames_opt=varNameALFA, dataKind_opt=8, &
441                          mpi_local_opt=.true.)
442
443        allocate(controlVector(loc%cvDim))
444
445        write(*,*) '********************************************'
446        write(*,*) 'midas-diagBmatrix: Compute columns of L matrix'
447        write(*,*) '********************************************'
448      
449        write(*,*) ' number of levels            = ', nlevs
450        write(*,*) ' number of lon-lat positions = ', nLonLatPos
451
452        if (numLoc > 1) then
453          write(locIndexString,'(i1)') locIndex
454          filename = 'columnL_' // trim(loc%locType) // '_' // locIndexString // '_' // datestr // '.fst'
455        else
456          if (numBensInstance > 1) then
457            write(instanceIndexString,'(I2.2)') instanceIndex
458            filename = 'columnL_i' // trim(instanceIndexString) // '_' // trim(loc%locType) // '_' // datestr // '.fst'
459          else
460            filename = 'columnL_' // trim(loc%locType) // '_' // datestr // '.fst'
461          end if
462        end if
463
464        ip3 = 0
465        do levIndex = 1, nlevs2
466          do lonLatPosIndex = 1, nLonLatPos
467            
468            latIndex = oneobs_lonlat(lonLatPosIndex,latPosIndex)
469            lonIndex = oneobs_lonlat(lonLatPosIndex,lonPosIndex)
470            
471            call ens_zero(ensAmplitude)
472            call gsv_zero(statevectorEnsAmplitude)
473            if ( latIndex >= statevector%myLatBeg .and. latIndex <= statevector%myLatEnd .and. &
474                 lonIndex >= statevector%myLonBeg .and. lonIndex <= statevector%myLonEnd ) then
475              ensAmplitude_oneLev => ens_getOneLev_r8(ensAmplitude,oneobs_levs(levIndex))
476              ensAmplitude_oneLev(:,oneobs_timeStepIndex,lonIndex,latIndex) = 1.d0
477            end if
478            controlVector(:)=0.0d0
479
480            if (numStepAmplitude > 1) then
481              call adv_ensemble_ad(ensAmplitude,                 & ! INOUT
482                                   adv_amplitudeAssimWindow, nEns)  ! IN
483            end if
484
485            call loc_LsqrtAd(loc,           & ! IN
486                             ensAmplitude,  & ! IN
487                             controlVector, & ! OUT
488                             amp3dStepIndex)  ! IN
489            call loc_Lsqrt  (loc,           & ! IN
490                             controlVector, & ! IN
491                             ensAmplitude,  & ! OUT
492                             amp3dStepIndex)  ! IN
493
494            if (numStepAmplitude > 1) then
495              call adv_ensemble_tl(ensAmplitude,                 & ! INOUT
496                                   adv_amplitudeAssimWindow, nEns) ! IN
497            end if
498
499            write(*,*)'midas-diagBmatrix: writing out the column of L, levIndex,lonIndex,latIndex=',levIndex,lonIndex,latIndex
500            call flush(6)
501
502            ip3 = ip3 + 1
503            call ens_copyMember(ensAmplitude, statevectorEnsAmplitude, 1)
504            do stepIndexInc = 1, numStepAmplitude
505              call gio_writeToFile(statevectorEnsAmplitude,filename,'ONEOBS',  &
506                                   stepIndex_opt=stepIndexInc, ip3_opt=ip3,unitConversion_opt=.false.)
507            end do
508
509          end do
510        end do
511
512        deallocate(controlVector)
513        call ens_deallocate(ensAmplitude)
514        call gsv_deallocate(statevectorEnsAmplitude)
515
516      end do ! localization index (if active in B)
517    end do ! Bens instance
518
519  end if ! if any oneobs selected
520
521  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
522
523  !
524  !==============================================
525  !- Compute the stddev from random perturbations
526  !==============================================
527  !
528  if ( numperturbations > 1 ) then
529
530    write(*,*) '********************************************'
531    write(*,*) 'midas-diagBmatrix: Compute the stddev from random perturbations'
532    write(*,*) '********************************************'
533
534    allocate(controlVector(cvm_nvadim))
535    allocate(controlVector_global(cvm_nvadim_mpiglobal))
536
537    call gsv_zero(statevector)
538
539    ! Allocate the randomEns, mean and stddev
540    allocate(randomEns(statevector%myLonBeg:statevector%myLonEnd,statevector%myLatBeg:statevector%myLatEnd,nkgdim,numperturbations))
541    allocate(mean(statevector%myLonBeg:statevector%myLonEnd,statevector%myLatBeg:statevector%myLatEnd,nkgdim))
542    allocate(stddev(statevector%myLonBeg:statevector%myLonEnd,statevector%myLatBeg:statevector%myLatEnd,nkgdim))
543
544    iseed = abs(nrandseed)
545    call rng_setup(iseed)
546
547    call gsv_getField(statevector,field3d)
548
549    !
550    !- Compute the ensemble of random perturbations
551    !
552    do ensIndex = 1, numperturbations
553      write(*,*) 'midas-diagBmatrix: computing member number= ',ensIndex
554      call flush(6)
555
556      !- Global vector (same for each processors)
557      do index = 1, cvm_nvadim_mpiglobal
558        controlVector_global(index) = rng_gaussian()
559      end do
560
561      !- Extract only the subvector for this processor
562      call bmat_reduceToMPILocal(controlVector,       & ! OUT
563                                 controlVector_global ) ! IN
564
565      !- Transform to control variables in physical space
566      call bmat_sqrtB(controlVector,cvm_nvadim,statevector)
567
568      if ( writePsiChiStddev ) call gvt_transform(statevector,'UVtoPsiChi')
569
570      !$OMP PARALLEL DO PRIVATE (lonIndex,latIndex,kIndex)    
571      do kIndex = 1, nkgdim
572        do latIndex = statevector%myLatBeg, statevector%myLatEnd
573          do lonIndex = statevector%myLonBeg, statevector%myLonEnd
574            randomEns(lonIndex,latIndex,kIndex,ensIndex) = field3d(lonIndex,latIndex,kIndex)
575          end do
576        end do
577      end do
578      !$OMP END PARALLEL DO
579
580    end do ! Loop on ens member
581
582    !
583    !- Compute the ensemble mean
584    !
585    mean(:,:,:) = 0.0d0
586    do ensIndex = 1, numperturbations
587      !$OMP PARALLEL DO PRIVATE (lonIndex,latIndex,kIndex)
588      do kIndex = 1, nkgdim
589        do latIndex = statevector%myLatBeg, statevector%myLatEnd
590          do lonIndex = statevector%myLonBeg, statevector%myLonEnd
591            mean(lonIndex,latIndex,kIndex) = mean(lonIndex,latIndex,kIndex) + randomEns(lonIndex,latIndex,kIndex,ensIndex)
592          end do
593        end do
594      end do
595      !$OMP END PARALLEL DO
596    end do
597
598    !$OMP PARALLEL DO PRIVATE (lonIndex,latIndex,kIndex)    
599    do kIndex = 1, nkgdim
600      do latIndex = statevector%myLatBeg, statevector%myLatEnd
601        do lonIndex = statevector%myLonBeg, statevector%myLonEnd
602          mean(lonIndex,latIndex,kIndex) = mean(lonIndex,latIndex,kIndex)/real(numperturbations,8)
603        end do
604      end do
605    end do
606    !$OMP END PARALLEL DO
607
608    !
609    !- Remove the ensemble mean from the ensemble
610    !
611    !$OMP PARALLEL DO PRIVATE (lonIndex,ensIndex,latIndex,kIndex)    
612    do ensIndex = 1, numperturbations
613      do kIndex = 1, nkgdim
614        do latIndex = statevector%myLatBeg, statevector%myLatEnd
615          do lonIndex = statevector%myLonBeg, statevector%myLonEnd
616            randomEns(lonIndex,latIndex,kIndex,ensIndex) = randomEns(lonIndex,latIndex,kIndex,ensIndex) - mean(lonIndex,latIndex,kIndex)
617          end do
618        end do
619      end do
620    end do
621    !$OMP END PARALLEL DO
622    deallocate(mean)
623
624    !
625    !- Compute the ensemble stddev
626    !
627    stddev(:,:,:) = 0.0d0
628
629    do ensIndex = 1, numperturbations
630      !$OMP PARALLEL DO PRIVATE (lonIndex,latIndex,kIndex)
631      do lonIndex = statevector%myLonBeg, statevector%myLonEnd
632        do latIndex = statevector%myLatBeg, statevector%myLatEnd
633          do kIndex = 1, nkgdim
634            stddev(lonIndex,latIndex,kIndex) = stddev(lonIndex,latIndex,kIndex) + &
635                 (randomEns(lonIndex,latIndex,kIndex,ensIndex)**2)/real(numperturbations,8)
636          end do
637        end do
638      end do
639      !$OMP END PARALLEL DO
640    end do
641    deallocate(randomEns)
642
643    !$OMP PARALLEL DO PRIVATE (lonIndex,latIndex,kIndex)
644    do kIndex = 1, nkgdim
645      do latIndex = statevector%myLatBeg, statevector%myLatEnd
646        do lonIndex = statevector%myLonBeg, statevector%myLonEnd
647          stddev(lonIndex,latIndex,kIndex) = sqrt(stddev(lonIndex,latIndex,kIndex))
648        end do
649      end do
650    end do
651    !$OMP END PARALLEL DO
652
653    !- Insert results in statevector
654    !$OMP PARALLEL DO PRIVATE (lonIndex,latIndex,kIndex)    
655    do kIndex = 1, nkgdim
656      do latIndex = statevector%myLatBeg, statevector%myLatEnd
657        do lonIndex = statevector%myLonBeg, statevector%myLonEnd
658          field3d(lonIndex,latIndex,kIndex) = stddev(lonIndex,latIndex,kIndex)
659        end do
660      end do
661    end do
662    !$OMP END PARALLEL DO
663    deallocate(stddev)
664
665    !- Write to file
666    call gio_writeToFile(statevector,'stddev_' // datestr // '.fst','GD_STDDEV',  &
667                         unitConversion_opt=.true.)
668
669    !
670    !- Compute the zonal mean std dev
671    !
672    write(*,*) 'midas-diagBmatrix: Compute the zonal mean stddev'
673    call flush(6)
674
675    allocate(stddev_zm(hco_anl%nj,nkgdim))
676    allocate(stddev_zm2(hco_anl%nj,nkgdim))
677    stddev_zm(:,:) = 0.d0
678    stddev_zm2(:,:) = 0.d0
679
680    !$OMP PARALLEL DO PRIVATE (lonIndex,latIndex,kIndex)
681    do kIndex = 1, nkgdim
682      do latIndex = statevector%myLatBeg, statevector%myLatEnd
683        do lonIndex = statevector%myLonBeg, statevector%myLonEnd
684          stddev_zm(latIndex,kIndex) = stddev_zm(latIndex,kIndex) + (field3d(lonIndex,latIndex,kIndex)**2)/real(hco_anl%ni,8)
685        end do
686      end do
687    end do
688    !$OMP END PARALLEL DO
689
690    nsize = statevector%nj*nkgdim
691    call rpn_comm_allreduce(stddev_zm,stddev_zm2,nsize,  &
692         "MPI_DOUBLE_PRECISION","MPI_SUM","GRID",ierr)
693
694    !- Insert results in statevector
695    !$OMP PARALLEL DO PRIVATE (lonIndex,latIndex,kIndex)    
696    do kIndex = 1, nkgdim
697      do latIndex = statevector%myLatBeg, statevector%myLatEnd
698        do lonIndex = statevector%myLonBeg, statevector%myLonEnd
699          if(stddev_zm2(latIndex,kIndex)> 0.0d0) then
700            field3d(lonIndex,latIndex,kIndex) = sqrt(stddev_zm2(latIndex,kIndex))
701          else
702            field3d(lonIndex,latIndex,kIndex) = 0.0d0
703          end if
704        end do
705      end do
706    end do
707    !$OMP END PARALLEL DO
708    deallocate(stddev_zm)
709    deallocate(stddev_zm2)
710
711    call gio_writeToFile(statevector,'stddev_' // datestr // '.fst','ZM_STDDEV',  &
712                         unitConversion_opt=.true.)
713
714    ! Write the zonal mean stddev to a text file, if requested
715    if ( writeTextStddev ) then
716      allocate(zonalMeanStddev(statevector%latPerPE * mmpi_npey))
717      do varIndex = 1, vnl_numvarmax
718        if (.not. gsv_varExist(varName=vnl_varNameList(varIndex)) ) cycle
719
720        write(*,*) 'midas-diagBmatrix: writing zonal mean stddev to text file for variable: ', vnl_varNameList(varIndex)
721        call gsv_getField(statevector,field3d,vnl_varNameList(varIndex))
722
723        varName = vnl_varNameList(varIndex)
724        if ( varName == 'HU' ) varName = 'LQ'
725        filename = 'stddev_ZM_' // trim(varName) // '_' // datestr // '.txt'
726        nultxt = 0
727        if ( mmpi_myid == 0 ) ierr = fnom(nultxt,trim(filename),'FTN',0)
728
729        do levIndex = 1, gsv_getNumLevFromVarName(statevector,vnl_varNameList(varIndex))
730          nsize = statevector%latPerPE
731          call rpn_comm_gather(field3d(1,:,levIndex), nsize, 'mpi_double_precision',  &
732                               zonalMeanStddev,     nsize, 'mpi_double_precision', 0, 'NS', ierr )
733          if ( mmpi_myid == 0 ) then
734            do latIndex = 1, statevector%nj
735              write(nultxt,*) field3d(1,latIndex,levIndex)
736            end do
737          end if
738        end do
739
740        if ( mmpi_myid == 0 ) ierr = fclos(nulnam)
741
742      end do
743      deallocate(zonalMeanStddev)
744
745    end if
746
747    !
748    !- Compute the domain mean std dev
749    !
750    write(*,*) 'midas-diagBmatrix: Compute the domain mean stddev'
751    call flush(6)
752
753    allocate(stddev_dm(hco_anl%ni,nkgdim))
754    allocate(stddev_dm2(hco_anl%ni,nkgdim))
755    stddev_dm(:,:) = 0.d0
756    stddev_dm2(:,:) = 0.d0
757
758    !$OMP PARALLEL DO PRIVATE (latIndex,kIndex)
759    do kIndex = 1, nkgdim
760      do latIndex = statevector%myLatBeg, statevector%myLatEnd
761        do lonIndex = statevector%myLonBeg, statevector%myLonEnd
762          stddev_dm(lonIndex,kIndex) = stddev_dm(lonIndex,kIndex) + (field3d(lonIndex,latIndex,kIndex)**2)/real(hco_anl%nj,8)
763        end do
764      end do
765    end do
766    !$OMP END PARALLEL DO 
767
768    nsize = statevector%ni*nkgdim
769    call rpn_comm_allreduce(stddev_dm,stddev_dm2,nsize,  &
770         "MPI_DOUBLE_PRECISION","MPI_SUM","GRID",ierr)
771
772    !- Insert results in statevector
773    !$OMP PARALLEL DO PRIVATE (lonIndex,latIndex,kIndex)
774    do kIndex = 1, nkgdim
775      do latIndex = statevector%myLatBeg, statevector%myLatEnd
776        do lonIndex = statevector%myLonBeg, statevector%myLonEnd
777          if(stddev_dm2(lonIndex,kIndex)> 0.0d0) then
778            field3d(lonIndex,latIndex,kIndex) = sqrt(stddev_dm2(lonIndex,kIndex))
779          else
780            field3d(lonIndex,latIndex,kIndex) = 0.0d0
781          end if
782        end do
783      end do
784    end do
785    !$OMP END PARALLEL DO
786    deallocate(stddev_dm)
787    deallocate(stddev_dm2)
788    deallocate(controlVector)
789    deallocate(controlVector_global)
790
791    call gio_writeToFile(statevector,'stddev_' // datestr // '.fst','DM_STDDEV',  &
792                         unitConversion_opt=.true.)
793
794  end if ! if numperturbations.gt.1
795
796  call gsv_deallocate(statevector)
797
798  write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
799
800  ! MPI, tmg finalize
801  call utl_tmg_stop(0)
802  call tmg_terminate(mmpi_myid, 'TMG_INFO')
803  call rpn_comm_finalize(ierr) 
804
805  write(*,*) ' --------------------------------'
806  write(*,*) ' midas-diagBmatrix ENDS'
807  write(*,*) ' --------------------------------'
808
809end program midas_diagBmatrix
