 17program midas_ominusf
 18  !
 19  !:Purpose: Main program for Observation minus Forecast (O-F) computation.
 20  !
 21  !          ---
 22  !
 23  !:Algorithm: The non-linear observation operators map a gridded state vector into the 
 24  !            observation space to compute the difference between the observations 
 25  !            and that state in observation space. The gridded state vector can be 
 26  !            background state or the analysis. In case of background state, the 
 27  !            difference is the innovation vector: ``y-H(xb)``. If asked by 
 28  !            the user, the diagonal of the background errors standard deviation in 
 29  !            observation space, :math:`{diag(H B H^{T})}^{1/2}`, are also computed.
 30  !            The bias corrections are applied for satellite radiances before writing 
 31  !            the observation files.
 32  !
 33  !            --
 34  !
 35  !============================================== ==============================================================
 36  ! Input and Output Files                        Description of file
 37  !============================================== ==============================================================
 38  ! ``flnml``                                      In - Main namelist file with parameters user may modify
 39  ! ``flnml_static``                               In - The "static" namelist that should not be modified
 40  ! ``trlm_$NN`` (e.g. ``trlm_01``)                In - Background state (a.k.a. trial) files for each timestep
 41  ! ``analysisgrid``                               In - File defining grid for computing the analysis increment
 42  ! ``obsfiles_$FAM/obs$FAM_$NNNN_$NNNN``          In - Observation file for each "family" and MPI task
 43  ! ``obserr``                                     In - Observation error statistics
 44  ! ``obsfiles_$FAM.updated/obs$FAM_$NNNN_$NNNN``  Out - Updated obs file for each "family" and MPI task
 45  ! Remainder are files related to radiance obs:
 46  ! ``stats_$SENSOR_assim``                        In - Satellite radiance observation errors of different sensors
 47  ! ``stats_tovs``                                 In - Satellite radiance observation errors
 48  ! ``stats_tovs_symmetricObsErr``                 In - User-defined symmetric TOVS errors for all sky
 49  ! ``Cmat_$PLATFORM_$SENSOR.dat``                 In - Inter-channel observation-error correlations
 50  ! ``dynbcor.coeffs.$SENSOR.*.coeffs_$SENSOR``    In - Dynamic bias correction file
 51  ! ``ceres_global.std``                           In - High-res surface type and water fraction for radiance obs
 52  ! ``rtcoef_$PLATFORM_$SENSOR.dat``               In - RTTOV coefficient files
 53  ! ``rttov_h2o_limits.dat``                       In - Min/max humidity limits applied to analysis
 54  ! ``ozoneclim98``                                In - Ozone climatology
 55  !============================================== ==============================================================
 56  !
 57  !           --
 58  !
 59  !:Synopsis: Below is a summary of the ``oMinusF`` program calling sequence:
 60  !
 61  !             - **Initial setups:**
 62  !
 63  !               - Read the NAMOMF namelist and check/modify some values.
 64  !
 65  !               - Various modules are setup: ``obsFiles_mod``, ``timeCoord_mod``.
 66  !
 67  !               - Setup ``gridStateVector`` module.
 68  !
 69  !               - Setup horizontal and vertical grid objects from either
 70  !                 ``analysisgrid`` or  first trial file: ``trlm_01``.
 71  !
 72  !               - Setup ``obsSpaceData`` object and read observations from
 73  !                 files: ``inn_setupObs``.
 74  !
 75  !               - Applying optional bias corrections to some observation types.
 76  !
 77  !               - Setup ``columnData`` module (read list of analysis variables 
 78  !                 from namelist) and allocate column object.
 79  !
 80  !               - Allocate a stateVector object and then read the state 
 81  !                 (either trials or analysis): ``gio_readTrials``.
 82  !
 83  !             - **Computation**
 84  !
 85  !               - Compute interpolated column on trial level ``columnTrlOnTrlLev`` 
 86  !                 from the state: ``inn_setupColumnsOnTrlLev``.
 87  !
 88  !               - Compute innovation from background state: ``inn_computeInnovation``.
 89  !
 90  !               - For computing background errors in observation space, 
 91  !                 ``columnTrlOnTrlLev`` are interpolated from background to analysis 
 92  !                 levels, ``columnTrlOnAnlIncLev``, and the linearize operators are
 93  !                 initialized: ``inn_setupColumnsOnAnlIncLev``.
 94  !
 95  !               - Update radiance bias correction in ``obsSpaceData`` and apply 
 96  !                 the bias corrections to the observations and innovations for
 97  !                 radiances: ``bcs_calcBias``, ``bcs_applyBiasCorrection``.
 98  !
 99  !               - Write the final bias corrected results into the observation file.
100  !
101  !             --
102  !
103  !:Options: `List of namelist blocks <../namelists_in_each_program.html#ominusf>`_
104  !          that can affect the ``oMinusF`` program.
105  !
106  !          * The use of ``oMinusF`` program is controlled by the namelist block
107  !           ``&NAMOMF`` read by the ``oMinusF`` program.
108  !
109  !          * Some of the other relevant namelist blocks used to configure the
110  !            ``oMinusF`` are listed in the following table:
111  ! 
112  !=========================== ========================= =========================================
113  ! Module                      Namelist                  Description of what is controlled
114  !=========================== ========================= =========================================
115  ! ``biasCorrectionConv_mod``  ``NAMBIASCONV``           variables to perform bias correction 
116  !                                                       for conventional observations.
117  ! ``biasCorrectionConv_mod``  ``NAMSONDETYPES``         additional variables to perform bias 
118  !                                                       correction for radiosondes conventional 
119  !                                                       observations.
120  ! ``biasCorrectionSat_mod``   ``NAMBIASSAT``            variables to perform bias correction 
121  !                                                       for satellite radiances.
122  ! ``burpread_mod``            ``NAMADDTOBURP``          element IDs to add to the BURP file
123  !=========================== ========================= =========================================
124  !
125  use version_mod
126  use oMinusF_mod
127  use obsSpaceData_mod
128  use columnData_mod
129  use obsFiles_mod
130  use utilities_mod
131  use midasMpi_mod
132  use biasCorrectionSat_mod
133  use ensembleObservations_mod
134  use fileNames_mod
135  implicit none
136
137  ! Namelist
138  integer :: nEns       ! ensemble size
139  logical :: addHBHT    ! choose to add the value of HBHT to obsSpaceData so it can be output
140  logical :: addSigmaO  ! choose to add the value of sigma_obs to obsSpaceData so it can be output
141  NAMELIST /NAMOMF/addHBHT, addSigmaO, nEns
142  
143  integer :: fnom, fclos, nulnam, ierr, headerIndex
144  
145  type(struct_columnData),target  :: columnTrlOnAnlIncLev
146  type(struct_columnData),target  :: columnTrlOnTrlLev
147  type(struct_obs),       target  :: obsSpaceData
148  type(struct_eob),       target  :: ensObs
149  
150  character(len=20)  :: oMinusFmode
151  character(len=256) :: ensPathName = 'ensemble'
152  character(len=256) :: ensFileName
153  character(len=256) :: trlFileName = './trlm_01'
154
155  logical :: ensFileExists, trlFileExists
156  
157  call ver_printNameAndVersion('oMinusF','Computation of the innovation')
158
159  !
160  !- 1.  Initialization
161  !
162  
163  !- 1.1 mpi
164  call mmpi_initialize
165
166  !- 1.2 timings
167  call tmg_init(mmpi_myid, 'TMG_INFO')
168  call utl_tmg_start(0,'Main')
169
170  if ( mmpi_myid == 0 ) then
171    call utl_writeStatus('VAR3D_BEG')
172  endif
173
174  !- 1.3 Namelist
175  addHBHT   = .false. ! default value
176  addSigmaO = .false.
177  nEns      = 20
178
179  nulnam = 0
180  ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
181  read(nulnam,nml=namomf,iostat=ierr)
182  if (ierr /= 0) call utl_abort('midas-OminusF: Error reading namelist')
183  if (mmpi_myid == 0) write(*,nml=namomf)
184  ierr = fclos(nulnam)
185
186  !- 1.4 Set mode
187  inquire(file=trim(trlFileName),exist=trlFileExists)
188  call fln_ensFileName(ensFileName, ensPathName, memberIndex_opt=1, &
189                       shouldExist_opt=.false.)
190  inquire(file=trim(ensFileName),exist=ensFileExists)
191
192  if      (trlFileExists) then
193    write(*,*)
194    write(*,*) 'Trial/Prog file found'
195    write(*,*) 'Setting mode to DETERMINISTIC'
196    oMinusFmode = 'deterministic'
197  else if (ensFileExists) then
198    write(*,*)
199    write(*,*) 'Ensemble file found'
200    write(*,*) 'Setting mode to ENSEMBLE'
201    oMinusFmode = 'ensemble'
202  else
203    write(*,*)
204    write(*,*) 'trlFileName = ', trim(trlFileName)
205    write(*,*) 'ensFileName = ', trim(ensFileName)
206    call utl_abort('oMinusF : did not find a trial/prog or ensemble file')
207  end if
208
209  !
210  !- 2.  Calculate the Observation - Forecast difference
211  !
212  if (trim(oMinusFmode) == 'deterministic') then
213
214    !- 2.1 Compute O-F and store in obsSpaceDate
215    call omf_oMinusF(columnTrlOnAnlIncLev, columnTrlOnTrlLev, obsSpaceData, &
216                     'OminusF', addHBHT, addSigmaO)
217
218    !- 2.2 Remove biases
219    call bcs_calcBias(obsSpaceData,columnTrlOnTrlLev) ! Fill in OBS_BCOR obsSpaceData column with computed bias correction
220
221    call bcs_applyBiasCorrection(obsSpaceData,OBS_VAR,"TO") ! Apply bias correction to OBS
222    call bcs_applyBiasCorrection(obsSpaceData,OBS_OMP,"TO") ! Apply bias correction to O-F
223
224    !- 2.3 Write the results
225
226    !  2.3.1 Into the listings
227    write(*,*)
228    write(*,*) '> midas-OminusF: printing the FIRST header and body'
229    do headerIndex = 1, min(1,obs_numHeader(obsSpaceData))
230      call obs_prnthdr(obsSpaceData,headerIndex)
231      call obs_prntbdy(obsSpaceData,headerIndex)
232    end do
233    ! 2.3.2 Into the observation files
234    write(*,*)
235    write(*,*) '> midas-OminusF: writing to file'
236    call obsf_writeFiles(obsSpaceData)
237
238  else ! ensemble
239
240    !- 2.1 Compute O-F and store in ensObs
241    call omf_oMinusFens( ensObs, obsSpaceData, nEns, ensPathName, &
242                         'OminusF', addHBHT, addSigmaO)
243
244    !- 2.3 Write the results
245    call obsf_writeFiles(obsSpaceData, ensObs_opt=ensObs)
246    
247  end if  
248    
249  !
250  !- 3.  Ending
251  !
252  write(*,*)
253  write(*,*) '> midas-OminusF: Ending'
254
255  call obs_finalize(obsSpaceData) ! deallocate obsSpaceData
256  
257  call utl_tmg_stop(0)
258  call tmg_terminate(mmpi_myid, 'TMG_INFO')
259
260  call rpn_comm_finalize(ierr)
261
262  if ( mmpi_myid == 0 ) then
263    call utl_writeStatus('VAR3D_END')
264  endif
265
266end program midas_ominusf
